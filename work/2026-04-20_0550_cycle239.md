### Test 3: Concurrent Tile Submission Race
```python
def test_concurrent_tile_submission(self):
    """Test handling of simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    
    # Simulate concurrent submissions
    def submit_tile(agent_id):
        tile = {"agent": agent_id, "data": f"test_{agent_id}"}
        return room.submit_tile(tile)
    
    # Use threading to simulate concurrency
    import threading
    results = []
    threads = []
    
    for i in range(3):
        t = threading.Thread(target=lambda i=i: results.append(submit_tile(f"agent_{i}")))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    self.assertEqual(len(results), 3)
    self.assertEqual(len(room.get_tile_queue()), 3)
    
    # Check for race condition indicators
    for result in results:
        self.assertIn("submission_id", result)
        self.assertNotIn("conflict", result)
```

### Test 4: Malformed Tile Data Recovery
```python
def test_malformed_tile_recovery(self):
    """Test room's handling of tiles with missing required fields."""
    room = DeadbandRoom()
    
    test_cases = [
        {"data": {}},  # Empty tile
        {"agent": "test_agent"},  # Missing data field
        {"data": "valid", "timestamp": "invalid_date"},  # Invalid timestamp
        {"data": {"nested": "value"}, "metadata": None},  # Null metadata
    ]
    
    recovery_count = 0
    for tile in test_cases:
        try:
            result = room.submit_tile(tile)
            if result.get("recovered", False):
                recovery_count += 1
            self.assertIn("status", result)
        except Exception as e:
            # Some malformed tiles should raise specific errors
            self.assertIsInstance(e, (ValueError, TypeError))
    
    # At least some tiles should be recoverable
    self.assertGreater(recovery_count, 0)
```

### Test 5: Resource Constraint During P2
```python
def test_p2_resource_exhaustion(self):
    """Test P2 behavior when optimization hits resource limits."""
    room = DeadbandRoom(max_compute_time=0.001)  # Very short time limit
    
    # Create a channel that would normally require significant computation
    large_channel = [{"id": f"state_{i}", "score": i/1000} for i in range(1000)]
    
    result = room.apply_p2(large_channel)
    
    # Should return partial results or fallback
    self.assertIn("optimization_status", result)
    status = result["optimization_status"]
    
    if status == "completed":
        self.assertIn("optimal_state", result)
    elif status == "timeout":
        self.assertIn("best_so_far", result)
        self.assertIn("time_limit_exceeded", result["metadata"])
    elif status == "interrupted":
        self.assertIn("fallback_state", result)
    
    # Resource metrics should be reported
    self.assertIn("resource_usage", result)
    self.assertLessEqual(result["resource_usage"]["compute_time"], 0.002)
```
