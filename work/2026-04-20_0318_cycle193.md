### Test 2: Conflicting Constraint Resolution
```python
def test_conflicting_constraints_priority():
    """Test constraint resolution when constraints conflict"""
    room = DeadbandRoom()
    agent_state = {
        "position": "intersection",
        "available_actions": ["move_north", "move_south", "wait"],
        "constraints": [
            {"type": "hard", "action": "move_north", "reason": "wall"},
            {"type": "soft", "action": "move_north", "reason": "high_cost"},
            {"type": "hard", "action": "wait", "reason": "timeout"}
        ]
    }
    
    # P0 should prioritize hard constraints over soft
    negative_space = room.map_negative_space(agent_state)
    assert "move_north" in negative_space  # Hard constraint
    assert "wait" in negative_space  # Hard constraint
    assert "move_south" not in negative_space
    
    # Verify constraint type discrimination
    hard_constraints = [c for c in agent_state["constraints"] if c["type"] == "hard"]
    assert len([a for a in negative_space if any(
        c["action"] == a and c["type"] == "hard" 
        for c in agent_state["constraints"]
    )]) == len(hard_constraints)
```

def test_network_timeout_during_safe_channel_discovery():
    """Test graceful handling of network failure during P1 phase"""
    room = DeadbandRoom()
    
    # Mock external service dependency
    room.external_service_url = "http://constraint-service.fleet"
    
    agent_state = {
        "position": "network_dependent",
        "available_actions": ["query_service", "fallback_action"],
        "constraints": [],
        "requires_external": True
    }
    
    # Simulate network timeout on external query
    with patch('requests.get', side_effect=requests.exceptions.Timeout):
        # P0 should complete (no network needed)
        negative_space = room.map_negative_space(agent_state)
        
        # P1 should handle timeout gracefully
        try:
            safe_channels = room.find_safe_channels(agent_state, negative_space)
            # Should fall back to local-only channels
            assert "fallback_action" in safe_channels
            assert "query_service" not in safe_channels
            assert room.current_phase == "P1"
            assert room.last_error is None  # Error handled internally
        except requests.exceptions.Timeout:
            pytest.fail("P1 should handle network timeout without propagation")
    
    # Verify error was logged but didn't crash protocol
    assert room.error_log is not None
    assert any("timeout" in str(e).lower() for e in room.error_log)
```

def test_malformed_phase_skip_detection():
    """Test detection of non-linear phase skipping (P0→P2 bypass)"""
    room = DeadbandRoom()
    
    agent_state = {
        "position": "malicious_agent",
        "available_actions": ["direct_optimize", "follow_protocol"],
        "constraints": []
    }
    
    # Simulate agent trying to skip P1
    room.current_phase = "P0"
    
    # Malicious attempt to call P2 directly
    try:
        # This should be caught by phase validation
        result = room.optimize_within_channels(
            agent_state=agent_state,
            safe_channels=["direct_optimize"]  # But P1 never ran!
        )
        
        # If we get here, the room failed to detect skip
        # Check that optimization was actually blocked
        assert result.get("error") == "phase_violation"
        assert "P1" in result.get("message", "")
        
    except PhaseViolationError as e:
        # Expected path
        assert "P0 to P2" in str(e)
        assert room.current_phase == "P0"  # Not advanced
    
    # Verify protocol counter tracks violations
    assert room.protocol_violations >= 1
    assert "P0→P2" in room.violation_log[-1]["type"]
```
