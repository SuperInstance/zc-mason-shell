### Test 1: Deadband Protocol Violation - P2 Optimization Without P1 Channel
**Scenario:** Agent attempts P2 optimization when no safe channels (P1) have been established.
**Edge Case:** This tests the strict priority enforcement when P1 returns empty list.
**Expected:** Room should reject P2 action and return appropriate error state.

```python
def test_p2_optimization_without_p1_channels(self, deadband_room):
    """Test P2 action rejected when no P1 channels exist."""
    # Setup: Agent completes P0 but P1 finds no safe channels
    deadband_room.process_action({"priority": 0, "actions": []})
    deadband_room.process_action({"priority": 1, "actions": []})  # Empty channels
    
    # Attempt P2 optimization
    result = deadband_room.process_action({"priority": 2, "actions": ["optimize"]})
    
    # Verify P2 was rejected
    assert result["status"] == "error"
    assert "P1 channels not established" in result["message"]
    assert deadband_room.current_priority == 1  # Should remain at P1
```

### Test 3: Concurrent Tile Submission Race Condition
**Scenario:** Multiple agents attempt simultaneous tile submission.
**Edge Case:** Tests thread safety of tile accumulation mechanism.
**Expected:** All tiles should be recorded without loss or corruption.

```python
def test_concurrent_tile_submission(self, deadband_room):
    """Test thread-safe tile accumulation under concurrent access."""
    import threading
    
    tiles_submitted = []
    errors = []
    
    def submit_tile(agent_id):
        try:
            tile = {"agent": agent_id, "content": f"test from {agent_id}"}
            result = deadband_room.submit_tile(tile)
            if result["status"] == "success":
                tiles_submitted.append(agent_id)
        except Exception as e:
            errors.append(str(e))
    
    # Simulate 10 agents submitting simultaneously
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    assert len(errors) == 0, f"Errors occurred: {errors}"
    assert len(tiles_submitted) == 10
    assert deadband_room.tile_count == 10  # Assuming tile_count property exists
```

### Test 5: Resource Exhaustion - Maximum Tile Capacity
**Scenario:** Room approaches and exceeds configured tile capacity limits.
**Edge Case:** Tests graceful handling of resource constraints.
**Expected:** Room should handle capacity limits with clear messaging.

```python
def test_tile_capacity_limits(self, deadband_room):
    """Test behavior when tile storage capacity is reached."""
    # Assuming room has max_tiles configuration
    max_tiles = deadband_room.config.get("max_tiles", 1000)
    
    # Fill to capacity
    for i in range(max_tiles):
        tile = {"agent": f"agent_{i}", "content": f"tile_{i}"}
        result = deadband_room.submit_tile(tile)
        assert result["status"] == "success"
    
    # Attempt one more tile beyond capacity
    overflow_tile = {"agent": "overflow", "content": "should fail"}
    result = deadband_room.submit_tile(overflow_tile)
    
    # Should reject with capacity error
    assert result["status"] == "error"
    assert "capacity" in result["message"].lower()
    
    # Verify count didn't increase
    assert deadband_room.tile_count == max_tiles
```
