def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "priority": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles (simulating concurrent access)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all("accepted" in str(r).lower() for r in results if not isinstance(r, str))
    
    # Verify room state consistency
    assert len(room.get_tiles()) == len(agent_ids)
```

def test_partial_tile_transmission(self):
    """Test room handling of incomplete tile data (simulated network failure)."""
    room = DeadbandRoom()
    
    # Valid tile
    valid_tile = {
        "agent_id": "test_agent",
        "priority": "P0",
        "content": "Complete tile",
        "timestamp": time.time()
    }
    
    # Partial tiles (missing required fields)
    partial_tiles = [
        {"agent_id": "agent1"},  # Missing priority, content, timestamp
        {"priority": "P1", "content": "test"},  # Missing agent_id, timestamp
        {"agent_id": "agent2", "timestamp": time.time()},  # Missing priority, content
        {}  # Empty tile
    ]
    
    # Submit valid tile first (should succeed)
    valid_result = room.submit_tile(valid_tile)
    assert "accepted" in valid_result.lower()
    
    # Submit partial tiles (should fail with specific errors)
    for partial_tile in partial_tiles:
        try:
            result = room.submit_tile(partial_tile)
            assert False, f"Should have rejected partial tile: {partial_tile}"
        except ValueError as e:
            # Verify error message indicates missing fields
            assert "missing" in str(e).lower() or "required" in str(e).lower()
    
    # Verify only valid tile was stored
    assert len(room.get_tiles()) == 1
```

    # Submit 10 consecutive P0 tiles (negative space mapping)

        tile = {
            "agent_id": agent_id,

    # Room should flag this pattern (implementation-specific check)
    # This tests if the room has starvation detection logic

def test_ghost_tile_integration(self):
    """Test mixing live agent tiles with persisted ghost tiles."""

        {

            "timestamp": time.time() - 86400,  # 1 day ago
            "is_ghost": True
        },
        {
            "agent_id": "ghost_agent_2", 
            "priority": "P1",
            "content": "Historical safe channel",
            "timestamp": time.time() - 43200,  # 12 hours ago
            "is_ghost": True
        }

    # Simulate room initialization with ghost tiles
    # This tests if the room constructor accepts pre-loaded tiles
    room_with_ghosts = DeadbandRoom(initial_tiles=ghost_tiles)
    
    # Submit new live tile

        "agent_id": "live_agent",
        "priority": "P2",

    assert len(all_tiles) == 3
    
    # Verify ghost tiles retain their metadata

    assert "P2" in priorities
