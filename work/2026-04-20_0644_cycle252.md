The existing `test_deadband_room.py` contains 8 tests covering:
1. Basic room initialization
2. P0 negative space mapping
3. P1 safe channel identification
4. P2 optimization within channels
5. Protocol violation detection (skipping to P2)
6. Tile submission validation
7. Agent state transitions
8. Room capacity limits

These tests are functional but focus on happy paths and core protocol logic. Edge cases around boundary conditions, error recovery, and concurrent operations are not covered.

## Edge-Case Test Design
Based on the DeadbandRoom implementation, I identified five edge cases that test:
1. **Concurrent agent submissions** - Race conditions when multiple agents submit tiles simultaneously
2. **Malformed tile data** - Invalid JSON, missing required fields, type mismatches
3. **Protocol state corruption** - Agents attempting operations from invalid states
4. **Resource exhaustion** - Memory/disk limits when processing large tile batches
5. **Network partition simulation** - Partial failures during multi-step operations

# plato-torch/tests/test_deadband_room_edge_cases.py
"""
Edge-case tests for DeadbandRoom preset.
Tests boundary conditions, error recovery, and concurrent operations.
"""

import pytest
import asyncio
import json
from unittest.mock import patch, AsyncMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.exceptions import ProtocolViolationError, TileValidationError


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""
    
    @pytest.mark.asyncio
    async def test_concurrent_tile_submissions(self):
        """Test race condition when multiple agents submit tiles simultaneously.
        
        Edge: Concurrent write operations to shared tile buffer.
        Expected: All tiles processed without data corruption.
        """
        room = DeadbandRoom(room_id="concurrent_test", max_agents=10)
        
        # Simulate 5 agents submitting tiles at the same time
        agents = [f"agent_{i}" for i in range(5)]
        tiles = [
            {"phase": "P0", "content": f"Negative space map {i}", "agent_id": agent}
            for i, agent in enumerate(agents)
        ]
        
        # Submit all tiles concurrently
        tasks = [room.submit_tile(tile) for tile in tiles]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Verify all submissions succeeded
        assert all(r is True for r in results), "Concurrent submissions failed"
        
        # Verify all tiles are present and uncorrupted
        stored_tiles = room.get_tiles()
        assert len(stored_tiles) == 5
        for i, tile in enumerate(stored_tiles):
            assert tile["content"] == f"Negative space map {i}"
            assert tile["agent_id"] == f"agent_{i}"
    
    @pytest.mark.asyncio
    @pytest.mark.parametrize("malformed_tile", [
        {"phase": "P0"},  # Missing required fields
        {"phase": "P0", "content": 123},  # Wrong type for content
        "not_a_dict",  # Not even a dict
        {"phase": "P4", "content": "Invalid phase"},  # Invalid phase value
        {"phase": "P0", "content": "x" * 10001},  # Content too large
    ])
    async def test_malformed_tile_validation(self, malformed_tile):
        """Test validation of various malformed tile structures.
        
        Edge: Invalid input data that could crash or corrupt state.
        Expected: Appropriate TileValidationError raised.
        """
        room = DeadbandRoom(room_id="validation_test")
        
        with pytest.raises(TileValidationError) as exc_info:
            await room.submit_tile(malformed_tile)
        
        # Verify error message is helpful
        assert "validation" in str(exc_info.value).lower()
    
    @pytest.mark.asyncio
    async def test_protocol_state_corruption_recovery(self):
        """Test recovery from corrupted protocol state.
        
        Edge: Agent attempts P2 operation without completing P0/P1.
        Expected: ProtocolViolationError raised, room state remains consistent.
        """
        room = DeadbandRoom(room_id="state_recovery_test")
        agent_id = "rogue_agent"
        
        # Directly manipulate internal state to simulate corruption
        room._agent_states[agent_id] = {"current_phase": "P2", "completed_phases": []}
        
        # Attempt to submit P2 tile without P0/P1 completion
        tile = {
            "phase": "P2",
            "content": "Trying to optimize without safe channels",
            "agent_id": agent_id
        }
        
        with pytest.raises(ProtocolViolationError) as exc_info:
            await room.submit_tile(tile)
        
        assert "P0" in str(exc_info.value) or "P1" in str(exc_info.value)
        
        # Verify room can still accept valid submissions
        valid_tile = {
            "phase": "P0",
            "content": "Proper negative space mapping",
            "agent_id": "good_agent"
        }
        result = await room.submit_tile(valid_tile)
        assert result is True
    
    @pytest.mark.asyncio
    async def test_resource_exhaustion_handling(self):
        """Test behavior when system resources are constrained.
        
        Edge: Large tile batches that could exhaust memory/disk.
        Expected: Graceful degradation or rejection, not crash.
        """
        room = DeadbandRoom(room_id="resource_test", max_tiles=1000)
        
        # Submit tiles until limit
        successful = 0
        for i in range(1500):  # Exceed limit by 50%
            tile = {
                "phase": "P0",
                "content": f"Tile {i}",
                "agent_id": f"agent_{i % 10}"
            }
            try:
                result = await room.submit_tile(tile)
                if result:
                    successful += 1
            except Exception:
                # Some failure expected at limit
                pass
        
        # Should have accepted exactly max_tiles
        assert successful == 1000
        assert len(room.get_tiles()) == 1000
        
        # Verify oldest tiles still present (FIFO behavior)
        tiles = room.get_tiles()
        assert "Tile 0" in tiles[0]["content"]
        assert "Tile 999" in tiles[-1]["content"]
    
    @pytest.mark.asyncio
    async def test_partial_failure_recovery(self):
        """Test recovery from partial network/system failures.
        
        Edge: Multi-step operation fails midway.
        Expected: Transaction rollback or consistent state.
        """
        room = DeadbandRoom(room_id="failure_test")
        
        # Mock storage to fail on second write
        call_count = 0
        original_save = room._save_tile
        
        async def failing_save(tile):
            nonlocal call_count
            call_count += 1
            if call_count == 2:
                raise IOError("Simulated storage failure")
            return await original_save(tile)
        
        room._save_tile = failing_save
        
        # Submit batch of tiles
        tiles = [
            {"phase": "P0", "content": f"Tile {i}", "agent_id": "test_agent"}
            for i in range(3)
        ]
        
        results = []
        for tile in tiles:
            try:
                result = await room.submit_tile(tile)
                results.append(result)
            except Exception as e:
                results.append(e)
        
        # First should succeed, second should fail, third might succeed or fail
        # depending on implementation
        assert results[0] is True
        assert isinstance(results[1], Exception)
        
        # Verify room state is consistent
        # Should have either 1 tile (if transaction rolled back) or 3 (if continued)
        stored_count = len(room.get_tiles())
        assert stored_count in [1, 3], f"Unexpected tile count: {stored_count}"
        
        # Clean up mock
        room._save_tile = original_save


if __name__ == "__main__":
    pytest.main([__file__, "-v"])
