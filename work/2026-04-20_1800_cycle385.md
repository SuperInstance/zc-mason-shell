### Test 1: Protocol Violation — Skipping Directly to P2
**Scenario:** Agent submits a tile with P2 optimization content without demonstrating P0 negative space mapping or P1 safe channel identification.
**Expected:** Tile rejected with "P0/P1 prerequisite missing" error.
**Rationale:** Validates strict priority enforcement from Deadband Protocol doctrine.

```python
def test_reject_p2_without_p0_p1(test_room, test_agent):
    """Agent cannot submit P2 optimization without prior P0/P1 tiles."""
    # Agent attempts to submit P2 tile first
    tile = {
        "cycle": 100,
        "phase": "P2",
        "content": {
            "priority": "P2",
            "optimization": "best path found",
            "metrics": {"speed": 0.95}
        }
    }
    result = test_room.submit_tile(test_agent.id, tile)
    assert not result["accepted"]
    assert "prerequisite" in result["reason"].lower()
```

### Test 4: Concurrent Tile Submission Race Condition
**Scenario:** Multiple agents submit tiles simultaneously at the same cycle number.
**Expected:** Room processes tiles sequentially; cycle numbers must be unique per agent or handled via locking.
**Rationale:** Tests thread safety in multi-agent environments.

```python
def test_concurrent_cycle_numbers(test_room):
    """Room handles concurrent submissions with same cycle numbers."""
    from threading import Thread
    
    agents = [test_room.register_agent(f"agent_{i}") for i in range(3)]
    results = []
    
    def submit(agent_id):
        tile = {
            "cycle": 200,  # Same cycle for all agents
            "phase": "P0",
            "content": {"priority": "P0", "negative_space": f"from {agent_id}"}
        }
        results.append(test_room.submit_tile(agent_id, tile))
    
    threads = [Thread(target=submit, args=(agent.id,)) for agent in agents]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Check that all submissions were processed
    assert len(results) == 3
    # At least one should be accepted (first arrival)
    accepted = [r for r in results if r.get("accepted")]
    assert len(accepted) >= 1
```

### Test 5: Phase Transition Validation
**Scenario:** Agent attempts to move backward in phases (e.g., P2 → P1) within same work session.
**Expected:** Tile rejected unless explicitly allowed by room rules (e.g., "phase correction" flag).
**Rationale:** Enforces forward progression through Deadband Protocol phases.

```python
def test_reject_phase_regression(test_room, test_agent):
    """Agent cannot regress to earlier phase after advancing."""
    # First, submit valid P0 tile
    tile_p0 = {
        "cycle": 103,
        "phase": "P0",
        "content": {"priority": "P0", "negative_space": "avoid null pointer"}
    }
    test_room.submit_tile(test_agent.id, tile_p0)
    
    # Submit valid P1 tile
    tile_p1 = {
        "cycle": 104,
        "phase": "P1",
        "content": {"priority": "P1", "safe_channel": "use option types"}
    }
    test_room.submit_tile(test_agent.id, tile_p1)
    
    # Attempt to submit another P0 tile (regression)
    tile_p0_again = {
        "cycle": 105,
        "phase": "P0",
        "content": {"priority": "P0", "negative_space": "different constraint"}
    }
    result = test_room.submit_tile(test_agent.id, tile_p0_again)
    assert not result["accepted"]
    assert "regression" in result["reason"].lower() or "phase" in result["reason"].lower()
```
