```python
# test_deadband_room_edge_cases.py
import pytest
from plato_torch.presets.deadband_room import DeadbandRoom

def test_state_corruption_recovery():
    """P0: Room should recover from corrupted state dictionary"""
    room = DeadbandRoom()
    # Inject corrupted state
    room.state = {"current_phase": "INVALID", "history": None}
    # Process valid input
    result = room.process("test input")
    # Should reset to valid state
    assert room.state["current_phase"] in ["P0", "P1", "P2"]
    assert isinstance(room.state["history"], list)

def test_malformed_json_input():
    """P0: Handle JSON-like but invalid input without crashing"""
    room = DeadbandRoom()
    # Partial JSON, malformed structure
    inputs = [
        '{"phase": "P1", "missing_quote: "value"}',
        '{"phase": "P2" "extra_comma": "value",}',
        'not json at all but has { brackets }',
    ]
    for inp in inputs:
        result = room.process(inp)
        # Should not raise exception
        assert "error" not in result or room.state["current_phase"] == "P0"

def test_protocol_skip_detection():
    """P0: Detect and correct P2 skipping attempts"""
    room = DeadbandRoom()
    # Start in P0
    room.state["current_phase"] = "P0"
    # Attempt to force P2 without P1
    room.state["force_phase"] = "P2"
    result = room.process("test")
    # Should remain in or revert to P0/P1, not accept P2
    assert room.state["current_phase"] != "P2"
    assert "skip" in result.get("warning", "").lower()

def test_memory_accumulation_limit():
    """P1: Test history accumulation bounds and pruning"""
    room = DeadbandRoom()
    # Generate many inputs
    for i in range(1000):
        room.process(f"input_{i}")
    # History should be bounded
    assert len(room.state.get("history", [])) <= 100  # Assuming reasonable limit
    # Oldest entries should be pruned
    if room.state.get("history"):
        assert "input_0" not in str(room.state["history"][0])

def test_concurrent_state_modification():
    """P2: Simulate race condition in state updates"""
    room = DeadbandRoom()
    import threading
    
    results = []
    def worker(input_val):
        result = room.process(input_val)
        results.append((input_val, room.state["current_phase"]))
    
    threads = []
    for i in range(10):
        t = threading.Thread(target=worker, args=(f"concurrent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # State should be consistent (all same phase or valid transition)
    phases = [phase for _, phase in results]
    # Check: all phases should be valid
    assert all(p in ["P0", "P1", "P2"] for p in phases)
    # Check: no more than 2 phase transitions occurred (P0->P1->P2 max)
    phase_changes = len(set(phases))
    assert phase_changes <= 3
```
