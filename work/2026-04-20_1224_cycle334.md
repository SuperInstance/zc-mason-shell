### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test that room handles multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "priority": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    
    # All tiles should be in room state
    assert len(room.get_tiles()) == 5
    
    # No data corruption between concurrent writes
    stored_tiles = room.get_tiles()
    agent_ids_stored = [t["agent_id"] for t in stored_tiles]
    assert set(agent_ids_stored) == set(agent_ids)
```

### Test 2: Malformed Tile Data Recovery

```python
def test_malformed_tile_recovery(self):
    """Test room recovery from various malformed tile submissions."""
    room = DeadbandRoom()
    
    test_cases = [
        # (malformed_data, expected_error_type)
        (None, "TypeError"),
        ("not_json", "JSONDecodeError"),
        ({}, "ValidationError"),  # Missing required fields
        ({"agent_id": "test", "priority": "P5"}, "ValidationError"),  # Invalid priority
        ({"agent_id": "test", "priority": "P1", "content": ""}, "ValidationError"),  # Empty content
        ({"agent_id": "test", "priority": "P1", "content": "x" * 10001}, "ValidationError"),  # Too large
    ]
    
    for malformed_data, expected_error in test_cases:
        try:
            room.submit_tile(malformed_data)
            assert False, f"Should have raised {expected_error} for {malformed_data}"
        except Exception as e:
            # Room should remain in valid state after each error
            assert room.is_healthy()
            
    # Room should accept valid tile after malformed attempts
    valid_tile = {
        "agent_id": "recovery_test",
        "priority": "P1",
        "content": "Valid after errors",
        "timestamp": time.time()
    }
    result = room.submit_tile(valid_tile)
    assert result["status"] == "accepted"
```

### Test 3: State Persistence Simulation
```python
def test_state_persistence(self):
    """Test that room state can be serialized/deserialized correctly."""
    room1 = DeadbandRoom()
    
    # Add some state
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "priority": "P1",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        room1.submit_tile(tile)
    
    # Simulate server restart: serialize and recreate
    serialized_state = room1.serialize_state()
    
    # Create new room instance with saved state
    room2 = DeadbandRoom()
    room2.load_state(serialized_state)
    
    # Verify state matches
    assert room2.get_tile_count() == 3
    assert room2.get_current_priority() == room1.get_current_priority()
    
    # New room should accept new tiles
    new_tile = {
        "agent_id": "new_agent",
        "priority": "P2",
        "content": "Post-restart tile",
        "timestamp": time.time()
    }
    result = room2.submit_tile(new_tile)
    assert result["status"] == "accepted"
    assert room2.get_tile_count() == 4
```

### Test 4: Priority Violation Recovery
```python
def test_priority_violation_recovery(self):
    """Test agent attempting P2 without completing P1 and room recovery."""
    room = DeadbandRoom()
    
    # Agent tries to skip to P2 (violates Deadband Protocol)
    violating_tile = {
        "agent_id": "impatient_agent",
        "priority": "P2",
        "content": "Trying to optimize before finding safe channels",
        "timestamp": time.time()
    }
    
    try:
        room.submit_tile(violating_tile)
        assert False, "Should reject P2 without P1 completion"
    except ProtocolViolationError as e:
        assert "P1 not completed" in str(e)
    
    # Room should guide agent to correct priority
    assert room.get_suggested_priority() == "P0"
    
    # Agent follows protocol correctly
    p0_tile = {
        "agent_id": "impatient_agent",
        "priority": "P0",
        "content": "Mapping negative space",
        "timestamp": time.time()
    }
    room.submit_tile(p0_tile)
    
    p1_tile = {
        "agent_id": "impatient_agent",
        "priority": "P1",
        "content": "Finding safe channels",
        "timestamp": time.time()
    }
    room.submit_tile(p1_tile)
    
    # Now P2 should be accepted
    p2_tile = {
        "agent_id": "impatient_agent",
        "priority": "P2",
        "content": "Optimizing within channels",
        "timestamp": time.time()
    }
    result = room.submit_tile(p2_tile)
    assert result["status"] == "accepted"
```

### Test 5: Resource Exhaustion Handling
```python
def test_resource_limits(self):
    """Test room behavior under resource constraints."""
    room = DeadbandRoom()
    
    # Test with many small tiles
    successful = 0
    for i in range(1000):
        try:
            tile = {
                "agent_id": f"mass_agent_{i}",
                "priority": "P1",
                "content": f"Tile {i}",
                "timestamp": time.time()
            }
            room.submit_tile(tile)
            successful += 1
        except ResourceLimitError:
            # Room should fail gracefully when limits reached
            break
    
    # Room should have processed some tiles
    assert successful > 0
    assert room.get_tile_count() == successful
    
    # Clear room and test with one very large tile
    room.reset()
    
    large_content = "x" * 5000  # 5KB content
    large_tile = {
        "agent_id": "large_agent",
        "priority": "P1",
        "content": large_content,
        "timestamp": time.time()
    }
    
    try:
        room.submit_tile(large_tile)
        # If succeeds, verify content integrity
        tiles = room.get_tiles()
        assert tiles[0]["content"] == large_content
    except ResourceLimitError:
        # Acceptable if room has size limits
        pass
    
    # Room should remain functional after resource errors
    assert room.is_healthy()
    
    # Should accept normal tile after resource issues
    normal_tile = {
        "agent_id": "recovery_agent",
        "priority": "P1",
        "content": "Back to normal",
        "timestamp": time.time()
    }
    result = room.submit_tile(normal_tile)
    assert result["status"] == "accepted"
```
