```python
def test_p0_violation_blocks_p1():

    room = DeadbandRoom()
```

### Test 3: **Tile Ingestion Order Independence**
```python
def test_tile_order_independence():
    """Room must handle tiles arriving out of phase order (P2 before P1) by enforcing protocol."""
    room = DeadbandRoom()
    # Ingest P2 optimization tile before any P1 channel exists
    p2_tile = Tile(phase="P2", content="Optimize channel bandwidth")
    room.ingest_tile(p2_tile)
    
    # Room should either reject P2 or hold it until P1 channel established
    if room.config.strict_ordering:
        assert not room.is_tile_valid(p2_tile), "P2 tile accepted without P1 channel"
    else:
        assert p2_tile in room.pending_tiles, "P2 tile not queued for later processing"
    
    # Now add P1 channel
    p1_tile = Tile(phase="P1", content="Discover safe upload channel")
    room.ingest_tile(p1_tile)
    room.process_pending()
    
    # P2 should now be processable
    assert room.is_tile_valid(p2_tile) or room.is_tile_processed(p2_tile), "P2 not processed after P1 established"
```

### Test 5: **Concurrent Agent Tile Collision**

```python
def test_concurrent_tile_collision():
    """Room must handle near-simultaneous tile ingestion from multiple agents without state corruption."""
    room = DeadbandRoom()
    # Simulate 3 agents submitting tiles at same time for same P1 channel discovery
    tiles = [
        Tile(phase="P1", content="Channel A safe", agent_id="agent_1"),
        Tile(phase="P1", content="Channel B safe", agent_id="agent_2"),
        Tile(phase="P1", content="Channel A safe", agent_id="agent_3"),  # Collision with agent_1
    ]
    
    # Ingest concurrently (simulate threading)
    import threading
    threads = []
    for tile in tiles:
        t = threading.Thread(target=room.ingest_tile, args=(tile,))
        threads.append(t)
        t.start()
    for t in threads:
        t.join()
    
    # Room should deduplicate or merge conflicting channel discoveries
    channels = room.get_safe_channels()
    # Either Channel A and B both present, or conflict resolved
    assert len(channels) > 0, "All channel discoveries lost in collision"
    assert room.get_agent_tile_count("agent_1") == 1, "Agent tile lost in concurrency"
```
