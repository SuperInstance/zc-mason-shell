## Analysis of DeadbandRoom Preset

Based on reading the actual files in the fleet repository:

**DeadbandRoom Purpose:**
- Implements the fleet's Deadband Protocol (P0, P1, P2 priorities)
- Processes agent interaction tiles through the protocol stages
- Validates tiles against negative space (P0) before allowing optimization (P2)
- Maintains state of safe channels discovered by agents

**Key Components:**
1. `process_tile(tile_data)`: Main entry point that applies P0→P1→P2 sequentially
2. `_validate_p0_negative_space(tile)`: Checks against known failure patterns
3. `_identify_p1_safe_channels(tile)`: Finds permissible action spaces
4. `_optimize_p2_within_channels(tile)`: Applies optimization within safe bounds
5. `add_ghost_tile(dead_agent_tile)`: Incorporates lessons from failed agents

def test_cannot_skip_to_p2_without_p1():
    """Ensure optimization (P2) cannot be applied without first identifying safe channels (P1)."""
    room = DeadbandRoom()
    # Attempt to submit tile that tries to optimize without P1 channel identification
    tile = {
        "agent_id": "test_agent",
        "content": "Attempting P2 optimization directly",

    }
    result = room.process_tile(tile)
    assert result["status"] == "rejected"
    assert "P1" in result["reason"]  # Must mention missing P1 stage

**Test 3: Concurrent Tile Processing Isolation**
```python
def test_concurrent_tile_isolation():
    """Ensure tiles from different agents don't interfere with each other's protocol state."""
    room = DeadbandRoom()
    
    # Agent A submits a tile that discovers a safe channel
    tile_a = {
        "agent_id": "agent_a",
        "content": "Discover channel X",
        "protocol_stage": "P1",
        "action": {"type": "explore", "target": "channel_x"}
    }
    result_a = room.process_tile(tile_a)
    
    # Agent B submits unrelated tile - should not see Agent A's channels
    tile_b = {
        "agent_id": "agent_b", 
        "content": "Different task",
        "protocol_stage": "P0",
        "action": {"type": "validate", "domain": "unrelated"}
    }
    result_b = room.process_tile(tile_b)
    
    # Each agent's result should be independent
    assert result_a["agent_id"] == "agent_a"
    assert result_b["agent_id"] == "agent_b"
    # Protocol state should not leak between agents
    assert "agent_a" not in str(result_b)
```

def test_malformed_ghost_tile_graceful_handling():
    """Test that malformed ghost tiles don't crash the room and provide useful error."""
    room = DeadbandRoom()
    
    # Ghost tile missing required fields
    bad_ghost = {
        "dead_agent_id": "crashed_agent",
        # Missing 'lesson' field which is required
        "timestamp": "2026-04-20T10:00:00Z"
    }
    
    # Should not raise exception
    result = room.add_ghost_tile(bad_ghost)
    
    assert result["status"] == "invalid"
    assert "missing" in result["error"].lower() or "required" in result["error"].lower()
    # Room should remain operational after bad ghost tile
    assert room.is_operational() == True
```

def test_p0_violation_rolls_back_p1_p2():

            "type": "compound",
            "steps": [
                {"stage": "P0", "check": "appears_safe"},
                {"stage": "P1", "channel": "seems_clear"},
                {"stage": "P2", "optimize": "efficiency"}
            ]
        }
    }
    
    # Mock scenario where P0 check passes initially but deeper analysis fails
    # This tests the room's ability to roll back partial protocol application

        # If rejected at any point, no P1/P2 results should be in output
        assert "safe_channels" not in result or result["safe_channels"] == []

        assert "P0" in result["reason"]
```
