### Test 1: Empty Negative Space (P0)
**Scenario:** Agent attempts to proceed when negative space is empty (no P0 constraints defined)
**Edge Case:** Deadband Protocol requires P0 mapping first, but what if there are no constraints?
**Test Purpose:** Ensure room handles empty constraint sets without crashing
```python
def test_empty_negative_space():
    """Test DeadbandRoom behavior when P0 negative space is empty."""
    room = DeadbandRoom(room_id="test_empty_p0")
    agent_context = {"agent_id": "test_agent", "phase": 0}
    
    # Simulate agent trying to map empty negative space
    result = room.process_agent_action(agent_context, {"action": "map_p0", "constraints": []})
    
    # Should not crash, should allow progression with warning
    assert "warning" in result
    assert "proceed" in result["status"]
    assert room.get_agent_phase("test_agent") == 1  # Should advance to P1
```

### Test 2: Overlapping Safe Channels (P1)
**Scenario:** Multiple safe channels identified with overlapping boundaries
**Edge Case:** When safe channels intersect, which should the agent choose?
**Test Purpose:** Test channel prioritization logic
```python
def test_overlapping_safe_channels():
    """Test DeadbandRoom handling of overlapping P1 safe channels."""
    room = DeadbandRoom(room_id="test_overlap_p1")
    
    # Setup overlapping channels
    room.negative_space = ["rock", "wall", "pit"]  # P0 constraints
    room.safe_channels = [
        {"id": "channel_a", "bounds": {"x": (0, 10), "y": (0, 10)}},
        {"id": "channel_b", "bounds": {"x": (5, 15), "y": (5, 15)}},
        {"id": "channel_c", "bounds": {"x": (8, 12), "y": (8, 12)}}
    ]
    
    agent_context = {"agent_id": "test_agent", "phase": 1, "position": {"x": 9, "y": 9}}
    result = room.select_safe_channel(agent_context)
    
    # Should select the most constrained (smallest) safe channel
    assert result["selected_channel"] == "channel_c"
    assert "overlap_resolution" in result
```

### Test 4: Concurrent Agent Protocol Violation
**Scenario:** Multiple agents in room, one violates Deadband Protocol
**Edge Case:** Protocol violation by one agent shouldn't corrupt others' state
**Test Purpose:** Test isolation between agent sessions
```python
def test_concurrent_protocol_violation():
    """Test DeadbandRoom isolation when multiple agents are present."""
    room = DeadbandRoom(room_id="test_concurrent")
    
    # Agent A follows protocol correctly
    agent_a = {"agent_id": "agent_a", "phase": 0}
    room.process_agent_action(agent_a, {"action": "map_p0", "constraints": ["rock"]})
    room.process_agent_action(agent_a, {"action": "find_p1", "channels": ["safe_path"]})
    
    # Agent B tries to skip to P2 immediately (violation)
    agent_b = {"agent_id": "agent_b", "phase": 0}
    violation_result = room.process_agent_action(
        agent_b, 
        {"action": "optimize_p2", "path": "shortcut"}
    )
    
    # Agent B should be rejected
    assert violation_result["status"] == "protocol_violation"
    assert "must_complete_p0_p1_first" in violation_result["error"]
    
    # Agent A should be unaffected
    assert room.get_agent_phase("agent_a") == 2  # Successfully at P2
    assert room.get_agent_phase("agent_b") == 0  # Still at P0
```

### Test 5: Tile Submission with Malformed Data
**Scenario:** Agent submits training tile with invalid structure
**Edge Case:** Non-compliant JSON, missing required fields, or type mismatches
**Test Purpose:** Test robust tile validation
```python
def test_malformed_tile_submission():
    """Test DeadbandRoom handling of invalid tile submissions."""
    room = DeadbandRoom(room_id="test_malformed_tiles")
    
    test_cases = [
        # (tile_data, expected_error_field)
        ({"incomplete": "data"}, "missing_required_field"),
        ({"agent_id": 123, "content": "test"}, "type_mismatch"),  # agent_id should be string
        ("not_a_dict", "invalid_structure"),
        ({"agent_id": "test", "content": "", "metadata": {}}, "empty_content"),
        ({"agent_id": "test", "content": "x" * 10001}, "content_too_large"),  # Size limit
    ]
    
    for tile_data, expected_error in test_cases:
        result = room.submit_tile(tile_data)
        assert result["status"] == "rejected"
        assert expected_error in result["error"]
    
    # Valid tile should still work
    valid_result = room.submit_tile({
        "agent_id": "test_agent",
        "content": "Valid test content",
        "metadata": {"phase": 2, "timestamp": "2026-04-20T06:50:00Z"}
    })
    assert valid_result["status"] == "accepted"
```
