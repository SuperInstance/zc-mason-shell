### Test 1: Protocol State Boundary - Empty P0 Map
**Scenario:** Agent attempts to proceed to P1 with an empty negative space map.
**Risk:** Protocol might allow optimization without safety constraints.
```python
def test_p0_empty_map_progression_blocked(self):
    """P1 should be blocked if P0 map is empty (no constraints learned)."""
    agent = DeadbandAgent()
    agent.p0_map = {}  # Empty negative space
    agent.current_phase = 0
    
    # Attempt to advance to P1
    result = agent.advance_phase()
    
    # Should remain in P0 until constraints are mapped
    self.assertEqual(agent.current_phase, 0)
    self.assertIn("P0 map empty", result["error"])
```

### Test 2: Concurrent Tile Collision
**Scenario:** Multiple agents submit tiles for the same training slot simultaneously.
**Risk:** Race condition could corrupt training data or cause validation failures.
```python
def test_concurrent_tile_submission_collision(self):
    """Handle multiple agents submitting to same room slot."""
    room = DeadbandRoom()
    agent1 = DeadbandAgent(agent_id="agent_1")
    agent2 = DeadbandAgent(agent_id="agent_2")
    
    # Both agents prepare tiles for same training cycle
    tile1 = agent1.generate_tile("test_data_1")
    tile2 = agent2.generate_tile("test_data_2")
    
    # Simulate concurrent submission
    import threading
    results = []
    
    def submit_tile(agent, tile):
        result = room.submit_tile(agent.agent_id, tile)
        results.append(result)
    
    thread1 = threading.Thread(target=submit_tile, args=(agent1, tile1))
    thread2 = threading.Thread(target=submit_tile, args=(agent2, tile2))
    
    thread1.start()
    thread2.start()
    thread1.join()
    thread2.join()
    
    # Only one should succeed, one should get collision error
    success_count = sum(1 for r in results if r["status"] == "accepted")
    self.assertEqual(success_count, 1)
    self.assertTrue(any("collision" in r.get("error", "").lower() for r in results))
```

### Test 3: Malformed Tile Signature
**Scenario:** Agent submits tile with tampered cryptographic signature.
**Risk:** Invalid tiles could bypass validation or corrupt training.
```python
def test_malformed_tile_signature_rejection(self):
    """Reject tiles with invalid or tampered signatures."""
    room = DeadbandRoom()
    agent = DeadbandAgent()
    
    # Generate valid tile then corrupt signature
    valid_tile = agent.generate_tile("valid_data")
    corrupted_tile = valid_tile.copy()
    corrupted_tile["signature"] = "tampered_" + corrupted_tile["signature"]
    
    # Submit corrupted tile
    result = room.submit_tile(agent.agent_id, corrupted_tile)
    
    self.assertEqual(result["status"], "rejected")
    self.assertIn("signature", result["error"].lower())
    self.assertIn("invalid", result["error"].lower())
```

### Test 4: Resource Exhaustion - Memory Limit
**Scenario:** Agent submits extremely large tile exceeding memory limits.
**Risk:** Could crash room server or degrade performance for other agents.
```python
def test_oversized_tile_rejection(self):
    """Reject tiles exceeding configured size limits."""
    room = DeadbandRoom(max_tile_size_mb=10)  # 10MB limit
    
    # Generate tile with excessive data
    oversized_data = "x" * (11 * 1024 * 1024)  # 11MB
    agent = DeadbandAgent()
    
    try:
        oversized_tile = agent.generate_tile(oversized_data)
        result = room.submit_tile(agent.agent_id, oversized_tile)
    except MemoryError as e:
        result = {"status": "rejected", "error": str(e)}
    
    self.assertEqual(result["status"], "rejected")
    self.assertIn("size", result["error"].lower())
    self.assertIn("limit", result["error"].lower())
```

### Test 5: Protocol Phase Regression
**Scenario:** Agent attempts to regress from P2 to P0 after optimization has begun.
**Risk:** Could create inconsistent agent state or training loops.
```python
def test_phase_regression_prevention(self):
    """Prevent agents from regressing phases after advancement."""
    agent = DeadbandAgent()
    
    # Advance through all phases
    agent.p0_map = {"constraint1": "avoid"}
    agent.advance_phase()  # P0 → P1
    
    agent.p1_channels = ["safe_path"]
    agent.advance_phase()  # P1 → P2
    
    # Now at P2, attempt to regress to P0
    agent.current_phase = 0  # Force regression attempt
    
    # Protocol should detect and correct
    self.assertEqual(agent.current_phase, 2)
    
    # Attempt explicit regression via method
    result = agent.set_phase(0)
    self.assertFalse(result["success"])
    self.assertIn("regression", result["error"].lower())
```
