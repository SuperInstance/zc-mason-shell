### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race():
    """Test that concurrent tile submissions don't corrupt internal state."""
    room = DeadbandRoom(room_id="race_test", deadband_threshold=0.1)
    
    def submit_tile(agent_id, value):
        tile = {"agent_id": agent_id, "value": value, "timestamp": time.time()}
        return room.submit_tile(tile)
    
    # Simulate 10 agents submitting simultaneously
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = []
        for i in range(10):
            future = executor.submit(submit_tile, f"agent_{i}", float(i)/10.0)
            futures.append(future)
        
        results = [f.result() for f in futures]
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent submissions should not fail"
    
    # Internal tile count should match
    assert len(room.tiles) == 10, \
        "All tiles should be preserved despite concurrency"
```

```python
def test_protocol_violation_sequence():

    p2_tile = {
        "agent_id": "eager_agent",
        "value": 0.8,
        "protocol_phase": "P2",  # Skipping P1
        "reasoning": "Trying to optimize before finding safe channel"

    result = room.submit_tile(p2_tile)

        "agent_id": "proper_agent",

        "protocol_phase": "P1",
        "reasoning": "Finding safe channel first"

    result = room.submit_tile(p1_tile)

        "Proper protocol sequence should be accepted"
```

```python
def test_byzantine_agent_malicious_payload():
    """Test resilience against malicious/Byzantine agent behavior."""
    room = DeadbandRoom(room_id="byzantine_test", deadband_threshold=0.1)
    
    malicious_payloads = [
        # Injection attempts
        {"agent_id": "malicious_1", "value": 0.5, "code": "__import__('os').system('rm -rf /')"},
        # Extremely deep recursion
        {"agent_id": "malicious_2", "value": 0.5, "nested": {"a": {"a": {"a": "..."}}}},  # Would be truncated in practice
        # Type confusion
        {"agent_id": "malicious_3", "value": "not_a_float", "metadata": float},
        # Memory exhaustion attempt
        {"agent_id": "malicious_4", "value": 0.5, "large": "x" * 10**7},
    ]
    
    for i, payload in enumerate(malicious_payloads):
        try:
            result = room.submit_tile(payload)
            # Should either reject or sanitize
            assert result["status"] in ["accepted", "rejected", "sanitized"], \
                f"Malicious payload {i} should not crash system"
            
            if result["status"] == "accepted":
                # If accepted, ensure it was sanitized
                assert "value" in result.get("tile", {}) and \
                       isinstance(result["tile"]["value"], float), \
                       f"Accepted malicious payload {i} should have sanitized value"
        except Exception as e:
            # Room should handle exceptions gracefully
            assert "crash" not in str(e).lower(), \
                f"Malicious payload {i} caused crash: {e}"
```
