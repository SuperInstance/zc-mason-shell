### Test 1: Concurrent Tile Submission

```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(room_id="concurrent_test")
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions using threads
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        return room.submit_tile(tile)
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=5) as executor:
        results = list(executor.map(submit_tile, agent_ids))
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results), "Concurrent submissions failed"
    # Room tile count should match
    assert len(room.tiles) == 5, f"Expected 5 tiles, got {len(room.tiles)}"
    # State should remain consistent
    assert room.state == "IDLE" or room.state == "MAPPING", f"Unexpected state: {room.state}"
```

### Test 2: Malformed Tile Payloads
```python
def test_malformed_tile_payloads(self):
    """Room should reject tiles with invalid structure and maintain stability."""
    room = DeadbandRoom(room_id="malformed_test")
    
    malformed_tiles = [
        None,  # null payload
        "not a dict",  # string instead of dict
        {"agent_id": "test"},  # missing required fields
        {"agent_id": 123, "phase": "P0", "content": "test"},  # wrong type for agent_id
        {"agent_id": "test", "phase": "P99", "content": "test"},  # invalid phase
        {"agent_id": "test", "phase": "P0", "content": "", "extra": "field"}  # extra field (should be okay)
    ]
    
    for i, tile in enumerate(malformed_tiles):
        result = room.submit_tile(tile)
        if i < 5:  # First 5 should be rejected
            assert result["status"] == "rejected", f"Tile {i} should be rejected: {tile}"
        else:  # Last one with extra field should be accepted (lenient)
            assert result["status"] == "accepted", f"Tile {i} with extra field should be accepted"
    
    assert len(room.tiles) == 1, f"Only valid tile should be stored, got {len(room.tiles)}"
```

### Test 3: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Room should handle agents violating P0→P1→P2 sequence and recover gracefully."""
    room = DeadbandRoom(room_id="violation_recovery_test")
    
    # Agent tries to submit P2 tile without P0/P1
    tile_p2 = {
        "agent_id": "impatient_agent",
        "phase": "P2",
        "content": "Trying to optimize without mapping",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p2)
    assert result["status"] == "rejected", "P2 without prior phases should be rejected"
    assert "P0 required first" in result.get("message", ""), "Should indicate protocol violation"
    
    # Room state should still allow valid submissions
    tile_p0 = {
        "agent_id": "patient_agent",
        "phase": "P0",
        "content": "Proper P0 mapping",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p0)
    assert result["status"] == "accepted", "Valid P0 should be accepted after violation"
    assert room.state == "MAPPING", f"Room should transition to MAPPING, got {room.state}"
```

### Test 4: High-Volume Tile Flooding
```python
def test_high_volume_tile_flooding(self):
    """Room should handle rapid tile submission without crashing or losing data."""
    room = DeadbandRoom(room_id="flood_test")
    BATCH_SIZE = 1000
    
    # Submit large batch of valid tiles
    for i in range(BATCH_SIZE):
        tile = {
            "agent_id": f"flood_agent_{i % 10}",  # 10 different agents
            "phase": "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
            "content": f"Tile {i}",
            "timestamp": time.time() + i  # Ensure unique timestamps
        }
        result = room.submit_tile(tile)
        # Every 100th tile, check room is still responsive
        if i % 100 == 0:
            assert room.state in ["IDLE", "MAPPING", "CHANNEL_FINDING", "OPTIMIZING"], \
                f"Room state corrupted at tile {i}: {room.state}"
    
    # Verify all tiles stored
    assert len(room.tiles) == BATCH_SIZE, f"Lost tiles: expected {BATCH_SIZE}, got {len(room.tiles)}"
    
    # Verify tile ordering preserved (by timestamp)
    timestamps = [t["timestamp"] for t in room.tiles]
    assert timestamps == sorted(timestamps), "Tile order not preserved under load"
```

```python
def test_agent_disconnection_mid_protocol(self):
    """Room should clean up when agent disconnects during active protocol phase."""
    room = DeadbandRoom(room_id="disconnect_test")
    
    # Agent starts P0
    tile_p0 = {
        "agent_id": "disconnecting_agent",
        "phase": "P0",
        "content": "Starting mapping",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p0)
    assert result["status"] == "accepted"
    assert room.state == "MAPPING"
    assert room.active_agent == "disconnecting_agent"
    
    # Simulate agent disconnection (no further tiles)
    # Room should timeout and reset after configured period
    room.last_activity = time.time() - room.INACTIVITY_TIMEOUT - 10  # Force timeout
    
    # Check room state before and after cleanup
    room._check_inactivity()
    assert room.state == "IDLE", f"Room should reset to IDLE after timeout, got {room.state}"
    assert room.active_agent is None, "Active agent should be cleared"
    assert len(room.tiles) == 1, "Tiles should persist despite disconnection"
    
    # New agent should be able to take over
    tile_p0_new = {
        "agent_id": "new_agent",
        "phase": "P0",
        "content": "Fresh start",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p0_new)
    assert result["status"] == "accepted", "New agent should be able to start"
```
