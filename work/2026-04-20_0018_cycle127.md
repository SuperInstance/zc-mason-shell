### Test 1: Protocol Violation - Skipping Directly to P2
**Purpose:** Verify the system correctly rejects attempts to bypass P0/P1 and go straight to optimization.

def test_protocol_violation_skip_to_p2():

    # Try to submit P2 tile immediately

### Test 2: Empty Content Tile Submission

def test_empty_content_tile():
    """Submit tiles with empty, None, or whitespace-only content."""
    room = DeadbandRoom()
    
    test_cases = [
        {"priority": "P0", "content": "", "cycle": 1},
        {"priority": "P0", "content": "   ", "cycle": 1},

### Test 3: Maximum Cycle Rollover

def test_max_cycle_rollover():
    """Handle cycle counter approaching system limits."""
    room = DeadbandRoom()
    
    # Simulate near-max cycle
    max_safe_cycle = 2**31 - 1000  # Leave buffer
    
    # Set to near maximum
    room._cycle = max_safe_cycle
    
    # Submit valid tile
    tile = {
        "priority": "P0",
        "content": "Testing cycle near limit",
        "cycle": max_safe_cycle + 1
    }
    
    result = room.submit_tile(tile)
    
    # System should either:
    # 1. Accept and handle rollover gracefully
    # 2. Reject with appropriate message
    # 3. Reset to safe value
    
    if result["accepted"]:
        next_cycle = room.get_state()["cycle"]
        assert next_cycle > max_safe_cycle or next_cycle == 1
```

### Test 4: Concurrent Tile Submission Race Condition
**Purpose:** Simulate multiple agents submitting tiles simultaneously.

def test_concurrent_submission_race():
    """Test handling of rapid sequential submissions."""
    room = DeadbandRoom()
    
    # Complete P0 phase
    p0_tile = {"priority": "P0", "content": "Negative space mapped", "cycle": 1}
    room.submit_tile(p0_tile)
    
    # Simulate rapid P1 submissions from different agents
    p1_tiles = [
        {"priority": "P1", "content": "Channel A found", "cycle": 2},
        {"priority": "P1", "content": "Channel B found", "cycle": 2},
        {"priority": "P1", "content": "Channel C found", "cycle": 2},
    ]
    
    accepted_count = 0
    for tile in p1_tiles:
        result = room.submit_tile(tile)
        if result["accepted"]:
            accepted_count += 1
    
    # System should either:
    # 1. Accept only the first valid P1 tile
    # 2. Accept multiple P1 tiles if designed for collaboration
    # 3. Reject duplicates based on some heuristic
    
    assert accepted_count >= 1  # At least one should be accepted
    if accepted_count == 1:
        # Verify state moved to P2 after single P1 acceptance
        assert room.get_current_priority() == "P2"
```
