```python
def test_concurrent_tile_submission_race_condition():
    """Test that room handles simultaneous tile submissions without corruption."""
    config = {
        "room_id": "race-test",
        "max_tiles": 1000,
        "validation_strictness": "strict"
    }
    room = DeadbandRoom(config)
    
    # Simulate concurrent submissions
    import threading
    
    results = []
    errors = []
    
    def submit_tile(tile_id):
        try:
            tile = {
                "priority": "P1",
                "content": f"Tile {tile_id}",
                "agent_id": f"agent_{tile_id % 10}",
                "timestamp": time.time() + tile_id * 0.001
            }
            result = room.submit_tile(tile)
            results.append((tile_id, result))
        except Exception as e:
            errors.append((tile_id, str(e)))
    
    threads = []
    for i in range(100):
        t = threading.Thread(target=submit_tile, args=(i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify no data corruption
    assert len(errors) == 0, f"Errors during concurrent submission: {errors}"
    assert len(results) == 100, f"Expected 100 submissions, got {len(results)}"
    assert room.tile_count == 100, f"Room tile count mismatch: {room.tile_count}"
    
    # Verify all tiles are present and unique
    tile_ids = [r[0] for r in results]
    assert len(set(tile_ids)) == 100, "Duplicate or missing tile IDs"
```

```python
    # Should handle without OOM
    result = room.submit_tile(tile)

    assert result["compressed_size"] < len(large_content), "Compression should reduce size"
```

```python
            assert result["status"] == "accepted"
            # Verify normalization
            stored_tile = room.get_tile(result["tile_id"])
            assert stored_tile["priority"] == expected_norm
        else:
            with pytest.raises((ValueError, TypeError)):
                room.submit_tile(tile)
```

```python
def test_network_partition_during_ensign_export():
    """Test ensign export resilience when network fails mid-operation."""
    config = {
        "room_id": "network-test",
        "export_timeout": 5,  # 5 second timeout
        "export_retries": 3
    }
    room = DeadbandRoom(config)
    
    # Add some tiles first
    for i in range(5):
        room.submit_tile({
            "priority": "P1",
            "content": f"Tile {i}",
            "agent_id": f"agent_{i}"
        })
    
    # Mock network failure scenarios
    class FlakyAdapterSaver:
        def __init__(self):
            self.call_count = 0
            self.fail_on_call = 2  # Fail on 2nd call
        
        def save_pretrained(self, *args, **kwargs):
            self.call_count += 1
            if self.call_count == self.fail_on_call:
                raise ConnectionError("Network partition simulated")
            return True
    
    flaky_saver = FlakyAdapterSaver()
    
    # Monkey patch the saver
    original_saver = room._get_adapter_saver
    room._get_adapter_saver = lambda: flaky_saver
    
    try:
        # Export should retry and eventually succeed
        result = room.export_ensign("flaky_adapter", "/tmp/test_export")
        
        assert result["status"] == "success"
        assert result["retries"] == 1  # One retry happened
        assert flaky_saver.call_count == 3  # Initial + retry + success
        
    finally:
        # Restore original
        room._get_adapter_saver = original_saver
    
    # Verify room state is clean after network issues
    assert room.state == "idle"
    assert room.export_queue.empty()
```
