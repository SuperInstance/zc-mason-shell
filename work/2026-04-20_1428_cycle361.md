### Test 3: Concurrent Tile Submission from Same Agent

```python
def test_concurrent_tiles_same_agent(room, agent_state):
    """Agent submits two tiles in same cycle - second should be queued."""
    tile1 = {
        "agent_id": agent_state.id,
        "priority": 1,
        "channel": "channel_a",
        "path": "path_1"
    }
    tile2 = {
        "agent_id": agent_state.id,
        "priority": 1,
        "channel": "channel_b",
        "path": "path_2"
    }
    
    # Process first tile
    result1 = room.process_tile(tile1)
    assert result1["status"] == "accepted"
    
    # Process second tile in same cycle
    result2 = room.process_tile(tile2)
    assert result2["status"] == "queued"
    assert result2["reason"] == "agent has pending tile"
    
    # Verify only first tile is in active state
    assert room.active_tiles[agent_state.id]["path"] == "path_1"
```

### Test 5: Agent State Corruption Recovery
```python
def test_corrupted_agent_state_recovery(room):
    """Room should handle corrupted agent state gracefully."""
    # Inject corrupted state
    room.agent_states["corrupted_agent"] = {
        "id": "corrupted_agent",
        "priority": "invalid_string",  # Wrong type
        "channel": None,
        "path": 12345  # Wrong type
    }
    
    # Valid tile from corrupted agent
    tile = {
        "agent_id": "corrupted_agent",
        "priority": 0,
        "channel": "recovery_channel",
        "path": "recovery_path"
    }
    
    result = room.process_tile(tile)
    # Should either: 1) Reject with state error, or 2) Reset and accept
    assert result["status"] in ["rejected", "accepted"]
    if result["status"] == "rejected":
        assert "state corruption" in result["reason"].lower()
    else:
        # State should be repaired
        state = room.agent_states["corrupted_agent"]
        assert isinstance(state["priority"], int)
        assert isinstance(state["path"], str)
```
