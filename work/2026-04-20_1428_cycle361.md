### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom(capacity=10)
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 1,
            "phase": 1,
            "content": f"Test from {agent_id}",
            "protocol_step": "P0"
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    assert len(room.get_pending_tiles()) == 5
    
    # Verify no tile corruption
    pending = room.get_pending_tiles()
    for agent_id in agent_ids:
        agent_tiles = [t for t in pending if t["agent_id"] == agent_id]
        assert len(agent_tiles) == 1
        assert agent_tiles[0]["content"] == f"Test from {agent_id}"
```

```python
def test_malformed_tile_payloads(self):
    """Room should reject tiles with invalid structure."""
    room = DeadbandRoom(capacity=5)
    
    test_cases = [
        # Missing required field
        {"cycle": 1, "phase": 1, "content": "test"},
        # Invalid JSON (passed as dict but simulating parse error)
        {"agent_id": "test", "cycle": "not_a_number", "content": "test"},
        # Empty content
        {"agent_id": "test", "cycle": 1, "phase": 1, "content": ""},
        # Protocol step violation
        {"agent_id": "test", "cycle": 1, "phase": 1, "content": "test", "protocol_step": "P3"},
        # Exceeds max content length
        {"agent_id": "test", "cycle": 1, "phase": 1, "content": "x" * 10001},
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(tile)
        assert result["status"] == "rejected", f"Test case {i} should be rejected"
        assert "error" in result, f"Test case {i} should include error message"
```

### Test 3: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Agent attempting P2 without P1 should be corrected to P0."""
    room = DeadbandRoom(capacity=3)
    agent_id = "protocol_test_agent"
    
    # Attempt P2 directly (violation)
    tile = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": 1,
        "content": "Attempting P2 without P1",
        "protocol_step": "P2"
    }
    
    result = room.submit_tile(tile)
    
    # Room should reject and provide correction
    assert result["status"] == "rejected"
    assert "P0" in result.get("correction", "")
    
    # Submit corrected P0 tile
    tile["protocol_step"] = "P0"
    tile["content"] = "Starting at P0 as corrected"
    result = room.submit_tile(tile)
    assert result["status"] == "accepted"
    
    # Now P1 should be accepted
    tile["protocol_step"] = "P1"
    tile["content"] = "Now at P1"
    result = room.submit_tile(tile)
    assert result["status"] == "accepted"
```

### Test 4: Room Persistence Simulation
```python
def test_room_persistence_simulation(self):
    """Simulate server restart with pending tiles."""
    room1 = DeadbandRoom(capacity=5)
    
    # Submit tiles to first room instance
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": i,
            "phase": 1,
            "content": f"Tile {i}",
            "protocol_step": "P0"
        }
        room1.submit_tile(tile)
    
    # Simulate persistence: get state and recreate room
    room_state = room1.get_state()
    
    # Create new room instance with saved state
    room2 = DeadbandRoom(capacity=5)
    room2.load_state(room_state)
    
    # Verify state restoration
    assert room2.get_pending_tile_count() == 3
    assert room2.get_agent_count() == 3
    
    # New submissions should work
    tile = {
        "agent_id": "new_agent",
        "cycle": 1,
        "phase": 1,
        "content": "Post-restart tile",
        "protocol_step": "P0"
    }
    result = room2.submit_tile(tile)
    assert result["status"] == "accepted"
    assert room2.get_pending_tile_count() == 4
```
