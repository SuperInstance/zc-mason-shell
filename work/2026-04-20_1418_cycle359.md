# test_deadband_room_edge_cases.py
import unittest
import torch
import numpy as np
from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom

class TestDeadbandRoomEdgeCases(unittest.TestCase):
    
    def setUp(self):
        """Set up test fixture with standard config."""
        self.config = {
            'observation_dim': 10,
            'action_dim': 4,
            'max_steps': 100,
            'p0_threshold': 0.1,
            'p1_threshold': 0.5,
            'p2_threshold': 0.9
        }
        self.room = DeadbandRoom(self.config)
    
    def test_protocol_priority_violation(self):
        """Test that skipping from P0 directly to P2 is rejected."""
        # Force room into P0 state
        self.room.current_phase = 'P0'
        self.room.protocol_state = {'safe_channels': []}
        
        # Agent attempts P2-optimized action
        action = torch.tensor([0.95, 0.95, 0.95, 0.95])  # P2-level action
        
        with self.assertRaises(ValueError) as context:
            self.room.step(action)
        
        self.assertIn('P0', str(context.exception))
        self.assertIn('P2', str(context.exception))
    
    def test_boundary_condition_at_p0_p1_threshold(self):
        """Test behavior at exact P0→P1 transition threshold."""
        # Set observation at exact threshold
        obs = torch.full((self.config['observation_dim'],), 
                        self.config['p0_threshold'])
        
        # Mock the observation generation
        with patch.object(self.room, '_generate_observation', return_value=obs):
            observation, reward, done, info = self.room.step(
                torch.zeros(self.config['action_dim'])
            )
            
            # Should transition to P1 at threshold
            self.assertEqual(self.room.current_phase, 'P1')
            self.assertGreaterEqual(len(info.get('safe_channels', [])), 0)
    
    def test_malformed_tensor_input_resilience(self):
        """Test room's handling of invalid tensor shapes and values."""
        # Test 1: Wrong action dimension
        wrong_dim_action = torch.zeros(self.config['action_dim'] + 1)
        with self.assertRaises(ValueError):
            self.room.step(wrong_dim_action)
        
        # Test 2: NaN values in action
        nan_action = torch.full((self.config['action_dim'],), float('nan'))
        with self.assertRaises(ValueError):
            self.room.step(nan_action)
        
        # Test 3: Infinite values
        inf_action = torch.full((self.config['action_dim'],), float('inf'))
        with self.assertRaises(ValueError):
            self.room.step(inf_action)
    
    def test_state_persistence_across_episodes(self):
        """Test that room maintains protocol learning across resets."""
        # Train through multiple episodes
        protocol_states = []
        
        for episode in range(3):
            self.room.reset()
            
            # Take some steps to accumulate protocol state
            for step in range(5):
                action = torch.randn(self.config['action_dim'])
                _, _, done, info = self.room.step(action)
                if done:
                    break
            
            # Capture protocol state
            protocol_states.append({
                'phase': self.room.current_phase,
                'channels': self.room.protocol_state.get('safe_channels', []).copy(),
                'negative_space': self.room.protocol_state.get('negative_space', []).copy()
            })
        
        # Verify some state persistence (channels should accumulate)
        total_channels = sum(len(state['channels']) for state in protocol_states)
        self.assertGreater(total_channels, 0)
        
        # Verify negative space grows (P0 knowledge accumulates)
        total_negative = sum(len(state['negative_space']) for state in protocol_states)
        self.assertGreaterEqual(total_negative, 0)
    
    def test_concurrent_agent_conflict_resolution(self):
        """Test room handling of multiple agents with conflicting phases."""
        # Simulate multiple agents via batch processing
        batch_size = 3
        batch_action = torch.randn(batch_size, self.config['action_dim'])
        
        # Mock batch processing method if it exists
        if hasattr(self.room, 'step_batch'):
            observations, rewards, dones, infos = self.room.step_batch(batch_action)
            
            # Each agent should have independent phase tracking
            phases = [info.get('phase', 'unknown') for info in infos]
            unique_phases = set(phases)
            
            # Should handle multiple phases concurrently
            self.assertGreater(len(unique_phases), 0)
            
            # No agent should be in invalid state
            for phase in phases:
                self.assertIn(phase, ['P0', 'P1', 'P2', 'unknown'])
        else:
            # Skip test if batch processing not implemented
            self.skipTest("Batch processing not implemented in DeadbandRoom")

if __name__ == '__main__':
    unittest.main()
