def test_deadband_room_malformed_action_recovery():
    """Test room recovery from malformed or adversarial action inputs.
    
    Edge case: Actions with missing required fields, invalid types, or extreme values.
    Room should handle gracefully without crashing protocol state.
    """
    room = DeadbandRoom()
    room.current_phase = "P1"

        # State should not be corrupted
        assert isinstance(room.negative_space, list)

### Test 4: Resource Exhaustion Boundary
```python
def test_deadband_room_resource_exhaustion():
    """Test protocol behavior when hitting resource limits.
    
    Edge case: Maximum number of negative space items, safe channels, or optimization paths.
    Tests graceful degradation rather than crashes.
    """
    room = DeadbandRoom()
    room.current_phase = "P0"
    
    # Fill negative space to theoretical limit
    max_items = 1000  # Assuming reasonable limit
    for i in range(max_items + 10):  # Try to exceed limit
        result = room.process_interaction({
            "action": "map_negative", 
            "data": f"rock_{i}"
        })
        
        if i < max_items:
            assert result["success"] == True
        else:
            # Should handle gracefully when limit reached
            assert result["success"] == False or "limit" in result.get("message", "").lower()
    
    # Verify we can still transition phases
    if len(room.negative_space) >= 10:  # Minimum for P0→P1 transition
        room.current_phase = "P1"
        assert room.current_phase == "P1"
```

### Test 5: Protocol Rollback on Invalid Transition
```python
def test_deadband_room_protocol_rollback():
    """Test that invalid phase transitions are rolled back properly.
    
    Edge case: Agent attempts to skip from P0 directly to P2.
    Protocol should detect violation, rollback any partial state changes,
    and maintain consistent phase state.
    """
    room = DeadbandRoom()
    room.current_phase = "P0"
    
    # Capture initial state
    initial_negative_space = room.negative_space.copy()
    initial_channels = room.safe_channels.copy()
    
    # Attempt illegal P0→P2 transition
    result = room.process_interaction({
        "action": "optimize_path",
        "data": {"path": "fast_but_unsafe"}
    })
    
    # Should fail
    assert result["success"] == False
    assert "P0" in result.get("error", "") or "skip" in result.get("error", "").lower()
    
    # Phase should remain P0
    assert room.current_phase == "P0"
    
    # State should be rolled back (no partial changes)
    assert room.negative_space == initial_negative_space
    assert room.safe_channels == initial_channels
    
    # Should still accept valid P0 actions after failed transition
    valid_result = room.process_interaction({
        "action": "map_negative",
        "data": "valid_rock"
    })
    assert valid_result["success"] == True
```
