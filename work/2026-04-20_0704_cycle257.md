"""
Edge-case tests for DeadbandRoom protocol implementation.
Tests scenarios beyond normal operation to ensure robustness.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
import json
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.exceptions import ProtocolViolationError, TileValidationError


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""
    
    @pytest.mark.asyncio
    async def test_protocol_priority_violation(self):
        """Test that agent cannot skip from P0 directly to P2."""
        # Arrange
        config = {
            "room_id": "edge-test-1",
            "protocol": "deadband",
            "max_agents": 1
        }
        room = DeadbandRoom(config)
        agent_id = "test-agent-1"
        
        # Act & Assert
        # Agent completes P0
        await room.execute_protocol_step(agent_id, "P0", {"negative_space": "test"})
        
        # Attempt to execute P2 without P1 - should raise ProtocolViolationError
        with pytest.raises(ProtocolViolationError) as exc_info:
            await room.execute_protocol_step(agent_id, "P2", {"optimization": "test"})
        
        assert "P1" in str(exc_info.value)  # Error should mention missing P1 step
    
    @pytest.mark.asyncio
    async def test_concurrent_tile_submission(self):
        """Test multiple agents submitting tiles simultaneously."""
        # Arrange
        config = {
            "room_id": "edge-test-2",
            "protocol": "deadband",
            "max_agents": 3
        }
        room = DeadbandRoom(config)
        
        agent_ids = ["agent-1", "agent-2", "agent-3"]
        tiles = [
            {"content": "Tile from agent 1", "phase": "P0"},
            {"content": "Tile from agent 2", "phase": "P0"},
            {"content": "Tile from agent 3", "phase": "P0"}
        ]
        
        # Act - submit tiles concurrently
        tasks = []
        for agent_id, tile in zip(agent_ids, tiles):
            task = asyncio.create_task(
                room.submit_tile(agent_id, tile)
            )
            tasks.append(task)
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Assert - all submissions should succeed without race conditions
        successful = 0
        for result in results:
            if not isinstance(result, Exception):
                successful += 1
        
        assert successful == 3
        assert len(room.get_tile_history()) == 3
    
    @pytest.mark.asyncio
    async def test_resource_exhaustion_handling(self):
        """Test room behavior when system resources are constrained."""
        # Arrange
        config = {
            "room_id": "edge-test-3",
            "protocol": "deadband",
            "max_tiles": 1000,  # Simulate memory limit
            "max_memory_mb": 10  # Artificially low memory limit
        }
        room = DeadbandRoom(config)
        
        # Mock os.getsizeof to simulate memory pressure
        with patch('os.getsizeof', return_value=1024 * 1024):  # 1MB per tile
            # Act - attempt to submit tiles until limit
            submitted = 0
            last_result = None
            
            for i in range(15):  # Should hit limit around 10 tiles
                try:
                    tile = {"content": f"Large tile {i}", "phase": "P0"}
                    last_result = await room.submit_tile(f"agent-{i}", tile)
                    submitted += 1
                except TileValidationError as e:
                    if "memory" in str(e).lower() or "resource" in str(e).lower():
                        last_result = e
                        break
            
            # Assert - room should enforce memory limits
            assert submitted <= 10
            assert isinstance(last_result, TileValidationError)
            assert room.get_resource_usage()["memory_mb"] <= 10
    
    @pytest.mark.asyncio
    async def test_network_partition_simulation(self):
        """Test room behavior with intermittent PLATO server connectivity."""
        # Arrange
        config = {
            "room_id": "edge-test-4",
            "protocol": "deadband",
            "plato_server_url": "http://localhost:8080"
        }
        room = DeadbandRoom(config)
        
        # Mock aiohttp client to simulate network failures
        mock_response_cycle = [
            AsyncMock(status=200, json=AsyncMock(return_value={"valid": True})),  # Success
            Exception("Connection refused"),  # Failure
            AsyncMock(status=200, json=AsyncMock(return_value={"valid": True})),  # Success
            Exception("Timeout"),  # Failure
            AsyncMock(status=200, json=AsyncMock(return_value={"valid": True}))   # Success
        ]
        
        mock_session = AsyncMock()
        mock_session.post.side_effect = mock_response_cycle
        
        # Act - attempt tile submissions with mixed connectivity
        results = []
        with patch('aiohttp.ClientSession', return_value=mock_session):
            for i in range(5):
                tile = {"content": f"Tile {i}", "phase": "P0"}
                try:
                    result = await room.submit_tile(f"agent-{i}", tile)
                    results.append(("success", result))
                except Exception as e:
                    results.append(("error", str(e)))
        
        # Assert - room should handle both successes and failures gracefully
        success_count = sum(1 for status, _ in results if status == "success")
        error_count = sum(1 for status, _ in results if status == "error")
        
        assert success_count == 3
        assert error_count == 2
        # Verify successful tiles are in history, failed ones are not
        assert len(room.get_tile_history()) == 3
