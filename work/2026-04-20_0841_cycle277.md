### Test 1: Empty Action Space at P1
**Scenario:** An agent reaches P1 (find safe channels) but the environment returns an empty list of valid actions.
**Expected Behavior:** The room should transition back to P0 (map negative space) since no safe channels exist.
**Rationale:** Tests resilience when no progress is possible at P1.

```python
def test_p1_empty_action_space_returns_to_p0(self):
    """Test that empty action space at P1 causes fallback to P0."""
    room = DeadbandRoom()
    room.current_phase = "P1"
    
    # Mock environment to return empty valid actions
    with patch.object(room.environment, 'get_valid_actions', return_value=[]):
        result = room.process_phase("P1")
    
    assert result["phase"] == "P0"
    assert "no_safe_channels" in result["feedback"]
    assert room.phase_history[-1] == ("P1", "P0", "empty_action_space")
```

### Test 2: Concurrent Protocol Violation Detection
**Scenario:** Multiple agents attempt protocol violations simultaneously.
**Expected Behavior:** The room should handle concurrent violations without deadlock or state corruption.
**Rationale:** Tests thread safety in multi-agent scenarios.

```python
def test_concurrent_protocol_violations(self):
    """Test handling of multiple simultaneous protocol violations."""
    room = DeadbandRoom()
    
    def attempt_violation(agent_id):
        room.current_phase = "P0"
        # Agent tries to skip directly to P2
        return room.submit_action(agent_id, {"phase": "P2", "action": "optimize"})
    
    # Simulate concurrent violations
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = [executor.submit(attempt_violation, f"agent_{i}") for i in range(3)]
        results = [f.result() for f in futures]
    
    # All should be violations
    assert all(r["violation"] for r in results)
    assert room.violation_count == 3
    # State should remain consistent
    assert room.current_phase == "P0"
```

### Test 3: Malformed Tile Submission
**Scenario:** Agent submits a tile with missing required fields or invalid structure.
**Expected Behavior:** The room should reject the tile with specific error feedback, not crash.
**Rationale:** Tests input validation and graceful error handling.

```python
def test_malformed_tile_submission(self):
    """Test rejection of tiles with invalid structure."""
    room = DeadbandRoom()
    
    test_cases = [
        {"tile": {}, "reason": "empty_tile"},
        {"tile": {"content": "test"}, "reason": "missing_phase"},
        {"tile": {"phase": "P1", "content": 123}, "reason": "invalid_content_type"},
        {"tile": {"phase": "P3", "content": "test"}, "reason": "invalid_phase"},
    ]
    
    for i, test_case in enumerate(test_cases):
        result = room.submit_tile(f"agent_{i}", test_case["tile"])
        assert not result["accepted"]
        assert test_case["reason"] in result["feedback"].lower()
        assert room.rejected_tiles[-1]["reason"] == test_case["reason"]
```

### Test 4: Phase Transition with Partial Progress
**Scenario:** Agent makes partial progress in P2 (optimization) but doesn't complete the objective.
**Expected Behavior:** The room should allow remaining in P2 with incremental feedback, not force phase regression.
**Rationale:** Tests support for incremental optimization within safe channels.

```python
def test_p2_partial_progress_incremental(self):
    """Test that P2 allows incremental optimization without phase regression."""
    room = DeadbandRoom()
    room.current_phase = "P2"
    room.current_channel = "safe_channel_1"
    
    # Simulate incremental optimization steps
    optimization_steps = [
        {"action": "optimize", "progress": 0.3},
        {"action": "optimize", "progress": 0.6},
        {"action": "optimize", "progress": 0.9},
    ]
    
    for step in optimization_steps:
        result = room.process_optimization_step(step)
        assert result["phase"] == "P2"
        assert "incremental_progress" in result["feedback"]
        assert result["current_progress"] > 0
    
    # Should not regress to P1 until optimization complete
    assert room.current_phase == "P2"
```

    # Exhaust all attempts with invalid actions
    for i in range(3):
