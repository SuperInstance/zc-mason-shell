```python
def test_concurrent_tile_submission(self):
    """Test that room handles multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3", "agent_4", "agent_5"]
    
    # Create valid P0 tiles from different agents
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 362,
            "phase": 4,
            "protocol_step": "P0",
            "content": f"Negative space map from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles in quick succession (simulated concurrency)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent tile submission failed"
    
    # All tiles should be in room memory
    assert len(room.memory) == len(agent_ids), \
        "Not all concurrent tiles were stored"
    
    # Verify no tile corruption
    for i, agent_id in enumerate(agent_ids):
        assert room.memory[i]["agent_id"] == agent_id, \
            f"Tile corruption detected for {agent_id}"
```

```python
def test_malformed_tile_handling(self):
    """Test room's resilience to malformed tile structures."""
    room = DeadbandRoom()
    
    test_cases = [
        # Missing required fields
        {"cycle": 362, "phase": 4},  # Missing agent_id, protocol_step
        {"agent_id": "mason", "protocol_step": "P0"},  # Missing cycle, phase
        
        # Wrong data types
        {"agent_id": 123, "cycle": "362", "phase": 4, "protocol_step": "P0"},
        {"agent_id": "mason", "cycle": 362, "phase": "four", "protocol_step": "P0"},
        
        # Invalid protocol steps
        {"agent_id": "mason", "cycle": 362, "phase": 4, "protocol_step": "P3"},
        {"agent_id": "mason", "cycle": 362, "phase": 4, "protocol_step": ""},
        
        # Extra unexpected fields (should be handled gracefully)
        {"agent_id": "mason", "cycle": 362, "phase": 4, 
         "protocol_step": "P0", "unknown_field": "should_ignore"},
    ]
    
    for i, malformed_tile in enumerate(test_cases):
        result = room.submit_tile(malformed_tile)
        
        # Malformed tiles should be rejected with clear error
        assert result["status"] == "rejected", \
            f"Malformed tile {i} was incorrectly accepted"
        assert "error" in result, \
            f"Missing error message for malformed tile {i}"
        assert isinstance(result["error"], str), \
            f"Error message should be string for tile {i}"
```

```python
def test_protocol_violation_escalation(self):
    """Test what happens when P0 is violated after P1/P2 established."""
    room = DeadbandRoom()
    
    # First establish P1 safe channel
    p1_tile = {
        "agent_id": "mason",
        "cycle": 362,
        "phase": 4,
        "protocol_step": "P1",
        "content": "Found safe channel: test execution",
        "timestamp": time.time()
    }
    
    p1_result = room.submit_tile(p1_tile)
    assert p1_result["status"] == "accepted", "P1 tile should be accepted"
    
    # Try to submit P2 without P1 (should fail)
    p2_tile = {
        "agent_id": "mason",
        "cycle": 362,
        "phase": 4,
        "protocol_step": "P2",
        "content": "Optimizing without safe channel",
        "timestamp": time.time()
    }
    
    p2_result = room.submit_tile(p2_tile)
    assert p2_result["status"] == "rejected", \
        "P2 without established P1 should be rejected"
    assert "protocol violation" in p2_result["error"].lower(), \
        "Should mention protocol violation"
    
    # Now violate P0 after P1 established
    p0_violation = {
        "agent_id": "mason",
        "cycle": 362,
        "phase": 4,
        "protocol_step": "P0",
        "content": "Attempting to map negative space that contradicts established P1",
        "timestamp": time.time()
    }
    
    p0_result = room.submit_tile(p0_violation)
    
    # This tests the room's conflict resolution logic
    # Either: reject P0 that contradicts P1, or flag as warning
    assert p0_result["status"] in ["accepted", "rejected", "warning"], \
        "Unexpected status for P0 violation"
    
    if p0_result["status"] == "accepted":
        assert "warning" in p0_result, \
            "Accepted contradictory P0 should have warning flag"
```

### Test 4: Memory Persistence Across Cycles
```python
def test_memory_persistence(self):
    """Test that room maintains state across multiple training cycles."""
    room = DeadbandRoom()
    
    # Simulate multiple training cycles
    cycles = [360, 361, 362, 363, 364]
    submitted_tiles = []
    
    for cycle in cycles:
        tile = {
            "agent_id": "mason",
            "cycle": cycle,
            "phase": 4,
            "protocol_step": "P0",
            "content": f"Cycle {cycle} negative space",
            "timestamp": time.time() + cycle  # Simulate time progression
        }
        
        result = room.submit_tile(tile)
        assert result["status"] == "accepted", \
            f"Tile for cycle {cycle} should be accepted"
        submitted_tiles.append(tile)
    
    # Verify all tiles are in memory
    assert len(room.memory) == len(cycles), \
        f"Expected {len(cycles)} tiles in memory, got {len(room.memory)}"
    
    # Verify chronological order (by timestamp)
    timestamps = [tile["timestamp"] for tile in room.memory]
    assert timestamps == sorted(timestamps), \
        "Tiles should be stored in chronological order"
    
    # Verify cycle progression is recorded
    for i, cycle in enumerate(cycles):
        assert room.memory[i]["cycle"] == cycle, \
            f"Memory corruption at position {i}: expected cycle {cycle}"
    
    # Test memory retrieval by cycle
    for cycle in cycles:
        cycle_tiles = [t for t in room.memory if t["cycle"] == cycle]
        assert len(cycle_tiles) == 1, \
            f"Should have exactly one tile for cycle {cycle}"
        assert cycle_tiles[0]["content"] == f"Cycle {cycle} negative space", \
            f"Content mismatch for cycle {cycle}"
```
