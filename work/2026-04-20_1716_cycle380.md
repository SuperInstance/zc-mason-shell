def test_concurrent_tile_submission_race():
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(room_id="race_test")
    
    # Simulate 5 agents submitting tiles at same time
    tiles = [
        Tile(agent_id=f"agent_{i}", content=f"test_{i}", protocol_phase="P0")
        for i in range(5)
    ]
    
    # Submit all tiles concurrently (simulated with threads)
    import threading
    
    results = []
    errors = []
    
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            errors.append(e)
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    assert len(results) == 5
    assert len(errors) == 0
    
    # Verify tiles were stored in correct order (timestamp-based)
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == 5
    
    # Verify no tile was lost or duplicated
    tile_ids = {t.agent_id for t in stored_tiles}
    assert len(tile_ids) == 5
```

def test_constraint_boundary_deadband():
    """Test constraints at exact boundary of deadband ranges."""
    room = DeadbandRoom(room_id="boundary_test")
    
    # Test case 1: Constraint exactly at lower deadband limit
    constraint_lower = Constraint(
        name="temp_lower",
        value=0.0,  # Exactly at deadband lower limit
        deadband_range=(0.0, 10.0)
    )
    
    # Test case 2: Constraint exactly at upper deadband limit  
    constraint_upper = Constraint(
        name="temp_upper",
        value=10.0,  # Exactly at deadband upper limit
        deadband_range=(0.0, 10.0)
    )
    
    # Test case 3: Constraint exactly at mid-point
    constraint_mid = Constraint(
        name="temp_mid",
        value=5.0,  # Exactly at mid-point
        deadband_range=(0.0, 10.0)
    )
    
    # All should be considered "in deadband" (safe channel)
    assert room._is_in_deadband(constraint_lower) is True
    assert room._is_in_deadband(constraint_upper) is True
    assert room._is_in_deadband(constraint_mid) is True
    
    # Test P1 execution with boundary constraints
    tile = Tile(
        agent_id="boundary_agent",
        content="testing boundaries",
        protocol_phase="P1",
        constraints=[constraint_lower, constraint_upper, constraint_mid]
    )
    
    result = room.submit_tile(tile)
    assert result.status == "accepted"
    
    # Verify all constraints were processed as safe
    processed = room.get_processed_constraints()
    assert len(processed) == 3
    assert all(p.status == "in_deadband" for p in processed)

def test_network_partition_mid_protocol():
    """Test room behavior when network partition occurs during P1→P2 transition."""
    room = DeadbandRoom(room_id="partition_test")
    
    # Start with P0 phase
    tile_p0 = Tile(
        agent_id="partition_agent",
        content="P0 negative space mapping",
        protocol_phase="P0"
    )
    room.submit_tile(tile_p0)
    
    # Begin P1 phase
    tile_p1 = Tile(
        agent_id="partition_agent",
        content="P1 safe channel finding",
        protocol_phase="P1",
        constraints=[Constraint(name="channel", value=5.0, deadband_range=(0.0, 10.0))]
    )
    
    # Simulate network partition after P1 submission but before P2
    # by mocking the network layer to fail
    original_send = room._send_to_network
    
    def failing_send(data):
        raise NetworkError("Partition detected: connection lost")
    
    room._send_to_network = failing_send
    
    # Submit P1 tile - should succeed locally
    result_p1 = room.submit_tile(tile_p1)
    assert result_p1.status == "accepted_local"
    
    # Try to transition to P2 - should fail network sync
    tile_p2 = Tile(
        agent_id="partition_agent",
        content="P2 optimization attempt",
        protocol_phase="P2"
    )
    
    try:
        result_p2 = room.submit_tile(tile_p2)
        # Should not reach here
        assert False, "Expected NetworkError"
    except NetworkError:
        # Expected - network partition
        pass
    
    # Restore network
    room._send_to_network = original_send
    
    # Verify room state is consistent despite partition
    tiles = room.get_tiles()
    assert len(tiles) == 2  # P0 and P1 tiles
    assert tiles[0].protocol_phase == "P0"
    assert tiles[1].protocol_phase == "P1"
    
    # Room should be in "degraded" mode, not "failed"
    assert room.status == "degraded"
    assert room.network_available is False
```
