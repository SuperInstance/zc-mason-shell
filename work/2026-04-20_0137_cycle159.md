def test_p2_before_p1_recovery():
    """Test that room recovers when agent attempts P2 optimization before P1 channel finding."""
    room = DeadbandRoom()
    agent_id = "test_agent_1"
    
    # Agent incorrectly submits P2 tile without P1 context
    violation_tile = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": "P2",
        "content": "Optimizing path without safe channel",
        "priority": 2
    }
    
    # Room should reject and provide corrective feedback
    result = room.submit_tile(violation_tile)
    assert not result["accepted"]
    assert "P1" in result["feedback"]  # Should reference missing P1
    
    # Agent then submits proper P1 tile
    correction_tile = {
        "agent_id": agent_id,
        "cycle": 2,
        "phase": "P1",
        "content": "Found safe channel in network layer",
        "priority": 1
    }
    
    result = room.submit_tile(correction_tile)
    assert result["accepted"]
    assert room.get_agent_state(agent_id)["last_valid_phase"] == "P1"

def test_concurrent_tile_submission():
    """Test handling of simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(3)]
    
    # Simulate concurrent submissions
    tiles = []
    for i, agent_id in enumerate(agent_ids):
        tile = {
            "agent_id": agent_id,
            "cycle": 10,
            "phase": "P1",
            "content": f"Channel analysis {i}",
            "priority": 1,
            "timestamp": time.time()  # Nearly identical timestamps
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        results.append(room.submit_tile(tile))
    
    # All should be accepted with unique sequence numbers
    accepted = [r for r in results if r["accepted"]]
    assert len(accepted) == len(agent_ids)
    
    # Check for unique sequence assignment
    seq_numbers = [r["sequence"] for r in accepted]
    assert len(set(seq_numbers)) == len(seq_numbers)
    
    # Room state should reflect all agents
    for agent_id in agent_ids:
        assert room.agent_exists(agent_id)
```

def test_memory_boundary_handling():
    """Test room behavior when approaching memory limits."""
    room = DeadbandRoom()
    agent_id = "stress_agent"
    
    # Submit tiles until near limit (simulated)
    max_tiles = room.config.get("max_tiles_per_agent", 1000)
    test_limit = min(50, max_tiles // 2)  # Test reasonable boundary
    
    accepted_count = 0
    for i in range(test_limit * 2):  # Attempt double the reasonable limit
        tile = {
            "agent_id": agent_id,
            "cycle": i,
            "phase": "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
            "content": "x" * 1000,  # Large content
            "priority": i % 3
        }
        
        result = room.submit_tile(tile)
        if result["accepted"]:
            accepted_count += 1
        else:
            # Should provide clear feedback when limits approached
            assert "limit" in result["feedback"].lower() or "memory" in result["feedback"].lower()
            break
    
    # Should accept reasonable number, reject when limits hit
    assert accepted_count >= test_limit
    assert accepted_count < test_limit * 2

def test_malformed_tile_recovery():
    """Test room handling of structurally invalid tiles."""
    room = DeadbandRoom()
    
    malformed_tiles = [
        None,  # Null tile
        {},  # Empty dict
        {"agent_id": "test"},  # Missing required fields
        {"agent_id": "test", "cycle": "not_a_number", "phase": "P1"},  # Wrong type
        {"agent_id": "test", "cycle": -1, "phase": "P1"},  # Invalid cycle
        {"agent_id": "test", "cycle": 1, "phase": "P99"},  # Invalid phase
        {"agent_id": "test", "cycle": 1, "phase": "P1", "priority": 99},  # Invalid priority
        {"agent_id": "test" * 1000, "cycle": 1, "phase": "P1"},  # Excessive ID length
    ]
    
    for i, tile in enumerate(malformed_tiles):
        try:
            result = room.submit_tile(tile)
            # Should not crash; should return rejection with feedback
            assert not result["accepted"]
            assert "error" in result or "feedback" in result
        except Exception as e:
            # Room should handle gracefully, not crash
            assert False, f"Room crashed on malformed tile {i}: {type(e).__name__}: {e}"
    
    # After malformed tiles, room should still accept valid tile
    valid_tile = {
        "agent_id": "recovery_agent",
        "cycle": 100,
        "phase": "P1",
        "content": "Valid after malformed storm",
        "priority": 1
    }
    
    result = room.submit_tile(valid_tile)
    assert result["accepted"], "Room should recover and accept valid tiles after malformed ones"
```

        "content": "Channel finding based on old state",
        "priority": 1
    }
    
    result2 = room.submit_tile(tile2)
    
    # Room should either:

    # 2. Accept and correct the cycle internally

        assert result3["accepted"]

        # If accepted, room should have corrected cycle
        current_state = room.get_agent_state(agent_id)
        assert current_state["last_cycle"] >= tile2["cycle"]
