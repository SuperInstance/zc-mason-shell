### Test 1: Protocol Priority Violation Recovery
**Scenario**: Agent attempts P2 optimization before completing P0 mapping
**Edge Case**: Protocol violation with partial P0 data
**Expected**: Room should reject tile, log violation, and maintain P0 state
**Test Focus**: State machine resilience

```python
def test_protocol_priority_violation_recovery(deadband_room, mock_agent):
    """Test room handles P2 attempts before P0 completion."""
    # Agent submits P2 tile without P0 mapping
    violation_tile = {
        "agent_id": mock_agent.id,
        "phase": "P2",
        "content": {"optimization": "test"},
        "timestamp": time.time()
    }
    
    # Should reject due to protocol violation
    with pytest.raises(ProtocolViolationError) as exc_info:
        deadband_room.submit_tile(violation_tile)
    
    assert "P0 mapping required before P2" in str(exc_info.value)
    assert deadband_room.get_agent_state(mock_agent.id)["current_phase"] == "P0"
```

### Test 2: Concurrent Agent Phase Collision
**Scenario**: Multiple agents in same room attempt conflicting protocol phases
**Edge Case**: Race condition in phase tracking
**Expected**: Room should serialize phase transitions or detect conflicts
**Test Focus**: Thread safety and conflict resolution

```python
def test_concurrent_agent_phase_collision(deadband_room, mock_agents):
    """Test room handles concurrent phase transitions."""
    agents = mock_agents(3)
    
    # All agents start at P0
    for agent in agents:
        deadband_room.register_agent(agent)
    
    # Simulate concurrent phase advancement
    def advance_phase(agent_id):
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": {"safe_channels": ["test"]},
            "timestamp": time.time()
        }
        return deadband_room.submit_tile(tile)
    
    # Submit concurrently (simulated with threads)
    results = []
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = [executor.submit(advance_phase, agent.id) for agent in agents]
        for future in futures:
            results.append(future.result())
    
    # All should succeed or conflict be handled gracefully
    assert len(results) == 3
    assert all(r["status"] == "accepted" for r in results)
```

### Test 4: Protocol Reset After Catastrophic Failure
**Scenario**: Room experiences internal error during phase transition
**Edge Case**: Partial state corruption
**Expected**: Room should recover to last consistent state or reset safely
**Test Focus**: Failure recovery and state integrity

```python
def test_protocol_reset_after_failure(deadband_room, mock_agent):
    """Test room recovers from internal errors during phase transitions."""
    deadband_room.register_agent(mock_agent)
    
    # Submit valid P0 tile
    p0_tile = {
        "agent_id": mock_agent.id,
        "phase": "P0",
        "content": {"negative_space": ["rock1", "rock2"]},
        "timestamp": time.time()
    }
    deadband_room.submit_tile(p0_tile)
    
    # Simulate internal error during P1 transition
    original_method = deadband_room._advance_phase
    def failing_advance(*args, **kwargs):
        raise RuntimeError("Simulated internal failure")
    
    deadband_room._advance_phase = failing_advance
    
    try:
        p1_tile = {
            "agent_id": mock_agent.id,
            "phase": "P1",
            "content": {"safe_channels": ["channel1"]},
            "timestamp": time.time()
        }
        with pytest.raises(RuntimeError):
            deadband_room.submit_tile(p1_tile)
    finally:
        deadband_room._advance_phase = original_method
    
    # Agent should be in recoverable state
    state = deadband_room.get_agent_state(mock_agent.id)
    assert state["current_phase"] in ["P0", "P1"]  # Either original or safe fallback
    assert state["recovery_required"] is True
```

### Test 5: Cross-Phase Tile Dependency Validation
**Scenario**: P2 tile references P1 data that doesn't exist
**Edge Case**: Cross-phase data dependency violations
**Expected**: Room should validate dependencies between phases
**Test Focus**: Protocol coherence across phases

```python
def test_cross_phase_tile_dependency(deadband_room, mock_agent):
    """Test room validates dependencies between protocol phases."""
    deadband_room.register_agent(mock_agent)
    
    # Complete P0
    p0_tile = {
        "agent_id": mock_agent.id,
        "phase": "P0",
        "content": {"negative_space": ["danger_zone"]},
        "timestamp": time.time()
    }
    deadband_room.submit_tile(p0_tile)
    
    # Skip P1, attempt P2 with reference to non-existent safe channel
    p2_tile = {
        "agent_id": mock_agent.id,
        "phase": "P2",
        "content": {
            "optimization": "test_opt",
            "based_on_channel": "non_existent_channel"  # Never defined in P1
        },
        "timestamp": time.time()
    }
    
    # Should reject due to missing dependency
    with pytest.raises(DependencyError) as exc_info:
        deadband_room.submit_tile(p2_tile)
    
    assert "non_existent_channel" in str(exc_info.value)
    assert "P1" in str(exc_info.value)
```
