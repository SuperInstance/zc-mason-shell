import pytest
import numpy as np
from plato_torch.presets.deadband_room import DeadbandRoom

class TestDeadbandRoomEdgeCases:
    """Edge-case tests for DeadbandRoom preset."""

    def test_p0_violation_with_invalid_action_type(self):
        """Test that room handles non-dict action input gracefully (P0: negative space)."""
        room = DeadbandRoom()
        room.reset()
        # Action should be a dict with keys like 'phase', 'channel', etc.
        invalid_action = "not_a_dict"
        with pytest.raises(ValueError) as exc_info:
            room.step(invalid_action)
        assert "Action must be a dictionary" in str(exc_info.value)

    def test_p1_channel_boundary_overflow(self):
        """Test safe channel (P1) identification when state values exceed normal bounds."""
        room = DeadbandRoom()
        room.reset()
        # Force internal state to extreme values (simulate numeric overflow)
        room._state = np.array([float('inf'), -float('inf'), 1e10], dtype=np.float32)
        safe_channels = room._find_safe_channels()  # Assume private method
        # Should still return a list (empty or clipped), not crash
        assert isinstance(safe_channels, list)
        # No channel should contain infinite values
        for channel in safe_channels:
            assert all(np.isfinite(channel))

    def test_p2_optimization_with_zero_safe_channels(self):
        """Test optimization (P2) when no safe channels exist (empty P1 result)."""
        room = DeadbandRoom()
        room.reset()
        # Mock to return no safe channels
        room._find_safe_channels = lambda: []
        action = room._optimize_within_channels()  # Assume private method
        # Should fallback to a default safe action (e.g., wait) or raise specific signal
        assert action is not None
        assert action.get('phase') == 0  # Default to P0 re-evaluation

    def test_concurrent_step_calls_race_condition(self):
        """Test room stability under rapid, concurrent step calls (simulated)."""
        room = DeadbandRoom()
        room.reset()
        actions = [
            {'phase': 0, 'channel': 'scan'},
            {'phase': 1, 'channel': 'navigate'},
            {'phase': 2, 'channel': 'optimize'}
        ]
        # Simulate rapid sequential calls (not truly threaded, but stress order)
        states = []
        for act in actions:
            state, reward, done, info = room.step(act)
            states.append(state)
            assert not done  # Should not terminate early
        # State should evolve consistently
        for i in range(1, len(states)):
            assert not np.array_equal(states[i], states[i-1])

    def test_recovery_from_corrupted_state_after_reset(self):
        """Test room fully resets even if previous state was corrupted (e.g., NaN)."""
        room = DeadbandRoom()
        room.reset()
        # Corrupt state manually
        room._state = np.array([np.nan, np.nan, np.nan])
        # Reset should clear corruption
        initial_state = room.reset()
        assert initial_state is not None
        assert all(np.isfinite(initial_state))
        # Step should work post-reset
        action = {'phase': 0, 'channel': 'scan'}
        state, reward, done, info = room.step(action)
        assert np.isfinite(reward)
