**Edge:** Network transmission limits, memory allocation failures.

def test_add_tile_exceeds_max_size(self):
    """Test that a tile exceeding MAX_TILE_SIZE raises TileTooLargeError."""
    room = DeadbandRoom(name="edge-room", max_tiles=1000)
    oversized_tile = {"content": "x" * (room.MAX_TILE_SIZE + 1), "metadata": {}}
    with self.assertRaises(TileTooLargeError):
        room.add_tile(oversized_tile)

### 2. Test: `train_ensign()` with tiles containing malformed sentiment scores
**Purpose:** Validate robustness when sentiment scores are out of bounds (-1.0 to 1.0).  
**Edge:** NPC sentiment integration in holodeck-rust expects normalized values.  
**Implementation:**
```python
def test_train_ensign_with_out_of_bounds_sentiment(self):
    """Test training with tiles containing sentiment scores outside [-1.0, 1.0]."""
    room = DeadbandRoom(name="sentiment-edge")
    # Tile with sentiment > 1.0
    room.add_tile({"content": "test", "sentiment": 2.5, "phase": 1})
    # Tile with sentiment < -1.0
    room.add_tile({"content": "test2", "sentiment": -3.0, "phase": 1})
    
    # Should clamp or handle gracefully without crashing
    ensign = room.train_ensign()
    self.assertIsNotNone(ensign)
    # Verify sentiment was normalized during training
    self.assertTrue(all(-1.0 <= s <= 1.0 for s in room._get_sentiment_cache()))
```

### 3. Test: Concurrent `add_tile()` from multiple threads
**Purpose:** Ensure thread safety for fleet-simulator concurrent tile submissions.  
**Edge:** Race conditions on tile count and memory buffer.  
**Implementation:**
```python
def test_concurrent_add_tile_thread_safety(self):
    """Test that add_tile() is thread-safe for concurrent fleet submissions."""
    room = DeadbandRoom(name="concurrent-room", max_tiles=500)
    tiles = [{"content": f"tile_{i}", "phase": i % 3} for i in range(100)]
    
    def worker(tile):
        room.add_tile(tile)
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=10) as executor:
        executor.map(worker, tiles)
    
    # Should have exactly 100 tiles, no duplicates or losses
    self.assertEqual(room.tile_count, 100)
    self.assertEqual(len(room.get_tiles()), 100)
```

### 4. Test: `export_ensign()` with custom LoRA adapter path that is read-only
**Purpose:** Map filesystem permission edge cases (P0).  
**Edge:** Deployment in restricted containers, read-only filesystems.  
**Implementation:**
```python
def test_export_ensign_read_only_path(self):
    """Test export to a read-only directory raises PermissionError."""
    room = DeadbandRoom(name="readonly-test")
    # Add minimum tiles and train
    for i in range(room.MIN_TILES_FOR_TRAINING):
        room.add_tile({"content": f"train_{i}", "phase": 1})
    room.train_ensign()
    
    # Create a read-only temporary directory
    import tempfile, os
    with tempfile.TemporaryDirectory() as tmpdir:
        read_only_dir = os.path.join(tmpdir, "readonly")
        os.makedirs(read_only_dir, mode=0o444)
        
        with self.assertRaises(PermissionError):
            room.export_ensign(adapter_path=read_only_dir)
```

### 5. Test: `add_tile()` with Unicode emoji and zero-width characters
**Purpose:** Ensure content encoding handles fleet communication artifacts.  
**Edge:** Herald agent messages, sentiment-aware NPC text processing.  
**Implementation:**
```python
def test_add_tile_unicode_edge_cases(self):
    """Test tiles with emoji, zero-width joiners, and bidirectional text."""
    room = DeadbandRoom(name="unicode-edge")
    edge_tiles = [
        {"content": "Hello 👨‍👩‍👧‍👦 World", "phase": 1},  # Emoji zwj sequence
        {"content": "A\u200B\u200C\u200Dinvisible", "phase": 2},  # Zero-width chars
        {"content": "السلام عليكم", "phase": 3},  # RTL text
        {"content": "Hello\x00World", "phase": 4},  # Null byte
    ]
    
    for tile in edge_tiles:
        room.add_tile(tile)
    
    # All tiles should be stored and retrievable
    stored = room.get_tiles()
    self.assertEqual(len(stored), 4)
    # Content should be preserved exactly
    self.assertEqual(stored[0]["content"], "Hello 👨‍👩‍👧‍👦 World")
```
