### Test 2: Malformed Tile Data Handling
```python
def test_submit_corrupted_tile_json():
    """Room should reject tiles with malformed JSON structure."""
    room = DeadbandRoom(config=valid_config)
    corrupted_tile = {
        "agent_id": "test_agent",
        "content": b"invalid_bytes",  # Non-serializable bytes
        "timestamp": "2026-04-19T22:15:00Z"
    }
    # Missing required 'phase' field
    incomplete_tile = {"agent_id": "test_agent"}
    
    with pytest.raises(TileValidationError):
        room.submit_tile(corrupted_tile)
    with pytest.raises(TileValidationError):
        room.submit_tile(incomplete_tile)
    assert len(room.tiles) == 0  # No tiles should be accepted
```

### Test 3: Room State Persistence After Simulated Crash
```python
def test_state_recovery_after_crash():
    """Room should recover state correctly after simulated failure."""
    room = DeadbandRoom(config=valid_config)
    # Add some valid tiles
    for i in range(3):
        room.submit_tile(create_valid_tile(phase="P0"))
    
    # Simulate crash: serialize state
    saved_state = room.serialize_state()
    
    # Create new room instance with saved state
    recovered_room = DeadbandRoom(config=valid_config)
    recovered_room.restore_state(saved_state)
    
    assert len(recovered_room.tiles) == 3
    assert recovered_room.current_phase == room.current_phase
    # Verify tile content integrity
    for original, recovered in zip(room.tiles, recovered_room.tiles):
        assert original["content"] == recovered["content"]
```

### Test 5: Unauthorized Agent Tile Submission
```python
def test_tile_submission_with_invalid_credentials():
    """Room should reject tiles from unauthorized agents."""
    room = DeadbandRoom(config=valid_config)
    
    # Valid agent
    valid_agent_tile = create_valid_tile(agent_id="authorized_agent")
    room.submit_tile(valid_agent_tile)
    
    # Unauthorized agent (not in config's allowed_agents)
    unauthorized_tile = create_valid_tile(agent_id="rogue_agent")
    
    with pytest.raises(AuthorizationError):
        room.submit_tile(unauthorized_tile)
    
    # Only valid tile should be accepted
    assert len(room.tiles) == 1
    assert room.tiles[0]["agent_id"] == "authorized_agent"
```
