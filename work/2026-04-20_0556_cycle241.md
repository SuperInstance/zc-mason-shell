### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(room_id="edge-concurrent")
    agent_ids = [f"agent-{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all("accepted" in str(r).lower() or "error" in str(r).lower() for r in results)
    
    # Verify room state consistency
    assert room.get_tile_count() == len([r for r in results if "accepted" in str(r).lower()])
```

### Test 2: Protocol Priority Violation Detection
```python
def test_protocol_priority_violation(self):
    """Room should reject tiles that skip P0 or P1 phases."""
    room = DeadbandRoom(room_id="edge-priority")
    
    # Agent attempts to start at P2 without P0/P1
    invalid_tile = {
        "agent_id": "impatient-agent",
        "phase": "P2",
        "content": "Trying to optimize without mapping",
        "timestamp": time.time()
    }
    
    # Should be rejected
    result = room.submit_tile(invalid_tile)
    assert "rejected" in result.lower() or "error" in result.lower()
    
    # Verify proper error message
    assert "p0" in result.lower() or "priority" in result.lower()
    
    # Now submit proper P0 tile
    valid_tile = {
        "agent_id": "impatient-agent",
        "phase": "P0",
        "content": "Mapping negative space",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(valid_tile)
    assert "accepted" in result.lower()
```

### Test 3: Memory Exhaustion Protection

```python
def test_large_tile_rejection(self):
    """Room should reject tiles exceeding size limits."""
    room = DeadbandRoom(room_id="edge-memory")
    
    # Create excessively large content
    large_content = "x" * (1024 * 1024 * 10)  # 10MB
    
    oversized_tile = {
        "agent_id": "bloated-agent",
        "phase": "P0",
        "content": large_content,
        "timestamp": time.time()
    }
    
    result = room.submit_tile(oversized_tile)
    assert "rejected" in result.lower() or "error" in result.lower()
    assert "size" in result.lower() or "limit" in result.lower()
    
    # Verify room didn't crash
    assert room.is_healthy()
```

### Test 4: Network Partition Simulation
```python
def test_partial_connectivity_recovery(self):
    """Room should handle agents with intermittent connectivity."""
    room = DeadbandRoom(room_id="edge-network")
    
    # Simulate flaky agent
    flaky_agent_id = "flaky-agent"
    
    # First submission succeeds
    tile1 = {
        "agent_id": flaky_agent_id,
        "phase": "P0",
        "content": "First mapping attempt",
        "timestamp": time.time()
    }
    
    result1 = room.submit_tile(tile1)
    assert "accepted" in result1.lower()
    
    # Simulate network drop (agent disappears)
    # Room should maintain state
    
    # Agent reconnects with same ID
    tile2 = {
        "agent_id": flaky_agent_id,
        "phase": "P1",
        "content": "Finding channels after reconnect",
        "timestamp": time.time()
    }
    
    result2 = room.submit_tile(tile2)
    
    # Should either accept (if state preserved) or handle gracefully
    assert "accepted" in result2.lower() or "reconnect" in result2.lower()
    
    # Verify agent's progress wasn't lost
    agent_state = room.get_agent_state(flaky_agent_id)
    assert agent_state is not None
```

### Test 5: Ghost Tile Integration
```python
def test_ghost_tile_integration(self):
    """Room should incorporate ghost tiles without disrupting live agents."""
    room = DeadbandRoom(room_id="edge-ghosts")
    
    # Add ghost tile (from terminated agent)
    ghost_tile = {
        "agent_id": "ghost-agent-001",
        "phase": "P0",
        "content": "Warning: avoid null pointer in constraint solver",
        "timestamp": time.time() - 3600,  # One hour ago
        "ghost": True
    }
    
    # Room should accept ghost tiles
    ghost_result = room.submit_tile(ghost_tile)
    assert "accepted" in ghost_result.lower()
    
    # Live agent submits tile
    live_tile = {
        "agent_id": "live-agent",
        "phase": "P0",
        "content": "Testing with ghost knowledge",
        "timestamp": time.time()
    }
    
    live_result = room.submit_tile(live_tile)
    assert "accepted" in live_result.lower()
    
    # Verify both tiles coexist
    tile_count = room.get_tile_count()
    assert tile_count >= 2
    
    # Ghost tiles should be marked as non-training
    tiles = room.get_recent_tiles(limit=10)
    ghost_tiles = [t for t in tiles if t.get("ghost")]
    assert len(ghost_tiles) > 0
    assert all(t.get("training_weight", 1.0) < 1.0 for t in ghost_tiles)
```
