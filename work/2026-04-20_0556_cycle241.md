### Test 3: Concurrent Tile Submission Race
```python
def test_concurrent_tile_submission():
    """P1: Multiple simultaneous submissions should not corrupt cycle count."""
    room = DeadbandRoom.default()
    
    def submit_tile():
        tile = {"content": f"tile from thread", "priority": 0}
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(submit_tile) for _ in range(5)]
        results = [f.result() for f in futures]
    
    # Only one should be accepted per cycle
    accepted = sum(1 for r in results if r["accepted"])
    assert accepted == 1
    assert room.current_cycle == 1
```
