import pytest

        """P0: Test with floating-point threshold at machine epsilon boundaries."""
        # Very small positive threshold
        room = DeadbandRoom(deadband_threshold=1e-10)
        assert room.deadband_threshold == 1e-10
        
        # Large threshold
        room = DeadbandRoom(deadband_threshold=1e10)
        assert room.deadband_threshold == 1e10
        
        # Threshold exactly at float min positive normal
        import sys
        room = DeadbandRoom(deadband_threshold=sys.float_info.min)
        assert room.deadband_threshold == sys.float_info.min
    
    def test_process_input_none_and_empty(self):
        """P0: Test process_input with None, empty dict, and malformed input."""
        room = DeadbandRoom(deadband_threshold=0.1)
        
        # None input
        with pytest.raises(TypeError):
            room.process_input(None)
        
        # Empty dict
        with pytest.raises(KeyError):
            room.process_input({})
        
        # Missing 'priority' key
        with pytest.raises(KeyError):
            room.process_input({'data': 'test'})
        
        # Invalid priority type
        with pytest.raises(ValueError):
            room.process_input({'priority': 'invalid', 'data': {}})
    
    def test_consecutive_process_input_state_evolution(self):
        """P1: Test state changes across multiple process_input calls."""
        room = DeadbandRoom(deadband_threshold=0.5)
        
        # Mock the internal _apply_deadband method to track calls
        call_log = []
        original_apply = room._apply_deadband
        def tracked_apply(data, priority):
            call_log.append((priority, data.get('value', 0)))
            return original_apply(data, priority)
        room._apply_deadband = tracked_apply
        
        # Sequence of inputs
        inputs = [
            {'priority': 0, 'value': 0.8},
            {'priority': 1, 'value': 0.3},
            {'priority': 2, 'value': 0.9},
            {'priority': 0, 'value': 0.1},  # Should trigger deadband
        ]
        
        for inp in inputs:
            room.process_input(inp)
        
        # Verify call sequence and deadband application
        assert len(call_log) == 4
        assert call_log[0][0] == 0  # First call P0
        assert call_log[3][0] == 0  # Last call P0
        # The room should track that value 0.1 is within deadband of previous P0 value 0.8
        # (threshold=0.5, difference=0.7 > 0.5, so should process)
    
    def test_reset_during_processing(self):
        """P0: Test reset() while room is in mid-processing state."""
        room = DeadbandRoom(deadband_threshold=0.2)
        
        # Set some internal state
        room.process_input({'priority': 0, 'value': 0.7})
        assert hasattr(room, '_last_values')
        
        # Mock train to be in a "processing" state
        with patch.object(room, 'train') as mock_train:
            mock_train.side_effect = lambda: time.sleep(0.1)

            import threading
            processing = False
            def process():
                nonlocal processing
                processing = True
                try:
                    room.process_input({'priority': 1, 'value': 0.4})
                finally:
                    processing = False
            
            thread = threading.Thread(target=process)
            thread.start()
            
            # Small delay to ensure processing started
