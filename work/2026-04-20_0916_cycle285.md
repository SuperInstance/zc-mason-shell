"""
Edge-case tests for DeadbandRoom protocol implementation.
Focuses on boundary conditions, error recovery, and protocol violations.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
import json
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import TrainingTile
from plato_torch.exceptions import ProtocolViolationError, StateCorruptionError


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom protocol implementation."""
    
    @pytest.fixture
    def room(self):
        """Fresh DeadbandRoom instance for each test."""
        return DeadbandRoom(room_id="test-edge-room")
    
    @pytest.fixture
    def valid_tile(self):
        """Valid training tile following protocol structure."""
        return TrainingTile(
            agent_id="test-agent",
            cycle=1,
            content={"phase": "P0", "observation": "Test negative space"},
            metadata={"protocol_version": "1.0"}
        )
    
    # Test 1: Protocol Priority Violation (P2 before P1)
    def test_protocol_priority_violation(self, room, valid_tile):
        """P0: Test that P2 optimization cannot occur before P1 safe channels."""
        
        # Create tile that attempts P2 without completing P1
        violating_tile = TrainingTile(
            agent_id="violating-agent",
            cycle=1,
            content={
                "phase": "P2",  # Direct to P2 - protocol violation
                "optimization": "premature optimization",
                "p0_complete": False,  # Never completed P0
                "p1_complete": False   # Never completed P1
            },
            metadata={"protocol_version": "1.0"}
        )
        
        # Should raise ProtocolViolationError
        with pytest.raises(ProtocolViolationError) as exc_info:
            room.process_tile(violating_tile)
        
        assert "P2 before P1" in str(exc_info.value)
        assert room.state["violation_count"] == 1
        
        # Verify room still accepts valid tiles after violation
        result = room.process_tile(valid_tile)
        assert result["status"] == "processed"
    
    # Test 2: State Boundary Conditions
    @pytest.mark.parametrize("state_size", [0, 10000, 100000])
    def test_state_size_boundaries(self, room, state_size):
        """P1: Test room behavior at different state size boundaries."""
        
        # Fill room state to specified size
        for i in range(state_size):
            tile = TrainingTile(
                agent_id=f"boundary-agent-{i}",
                cycle=i,
                content={"phase": "P0", "observation": f"Boundary test {i}"},
                metadata={"protocol_version": "1.0"}
            )
            
            if i < 1000:  # Process normally for first 1000
                result = room.process_tile(tile)
                assert result["status"] in ["processed", "queued"]
            else:
                # After 1000, room should handle gracefully (queue or reject)
                try:
                    result = room.process_tile(tile)
                    assert result["status"] in ["processed", "queued", "rejected"]
                except Exception as e:
                    # Acceptable if room has resource limits
                    assert "resource" in str(e).lower() or "limit" in str(e).lower()
        
        # Verify room can still process after boundary test
        recovery_tile = TrainingTile(
            agent_id="recovery-agent",
            cycle=state_size + 1,
            content={"phase": "P0", "observation": "Recovery test"},
            metadata={"protocol_version": "1.0"}
        )
        
        result = room.process_tile(recovery_tile)
        assert result is not None
    
    # Test 3: Concurrent Tile Processing Race Conditions
    @pytest.mark.asyncio
    async def test_concurrent_tile_race_condition(self, room):
        """P0: Test concurrent tile processing for race conditions."""
        
        async def process_concurrent_tile(agent_id, delay):
            await asyncio.sleep(delay)
            tile = TrainingTile(
                agent_id=agent_id,
                cycle=1,
                content={"phase": "P0", "observation": f"Concurrent from {agent_id}"},
                metadata={"protocol_version": "1.0"}
            )
            return room.process_tile(tile)
        
        # Launch concurrent tile processing with staggered delays
        tasks = []
        for i in range(10):
            delay = i * 0.01  # Staggered to increase race condition likelihood
            task = process_concurrent_tile(f"concurrent-agent-{i}", delay)
            tasks.append(task)
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Count successful vs failed processing
        successful = 0
        exceptions = 0
        
        for result in results:
            if isinstance(result, Exception):
                exceptions += 1
                # Exceptions should be specific, not generic
                assert any(keyword in str(result).lower() 
                          for keyword in ["concurrent", "lock", "race", "protocol"])
            else:
                successful += 1
                assert result["status"] in ["processed", "queued"]
        
        # Room should handle concurrent access gracefully
        assert successful + exceptions == 10
        assert room.state["processed_count"] <= 10  # Some may be queued
    
    # Test 4: Recovery from Corrupted State
    def test_corrupted_state_recovery(self, room):
        """P1: Test room recovery from intentionally corrupted internal state."""
        
        # First, process some valid tiles to establish state
        for i in range(5):
            tile = TrainingTile(
                agent_id=f"pre-corrupt-{i}",
                cycle=i,
                content={"phase": "P0", "observation": f"Pre-corruption {i}"},
                metadata={"protocol_version": "1.0"}
            )
            room.process_tile(tile)
        
        original_state = room.state.copy()
        
        # Corrupt the internal state (simulating bug or disk corruption)
        room.state["p0_negative_space"] = None  # Should be dict
        room.state["p1_channels"] = "not-a-list"  # Should be list
        room.state["cycle_counter"] = -1  # Invalid negative
        
        # Attempt to process tile with corrupted state
        recovery_tile = TrainingTile(
            agent_id="recovery-agent",
            cycle=100,
            content={"phase": "P0", "observation": "Attempt recovery"},
            metadata={"protocol_version": "1.0"}
        )
        
        try:
            result = room.process_tile(recovery_tile)
            
            # If room recovers, it should either:
            # 1. Process successfully (with state repair)
            # 2. Raise StateCorruptionError
            if result["status"] == "processed":
                # Verify state was repaired
                assert isinstance(room.state["p0_negative_space"], dict)
                assert isinstance(room.state["p1_channels"], list)
                assert room.state["cycle_counter"] >= 0
            else:
                assert result["status"] == "rejected"
                assert "corrupt" in result.get("reason", "").lower()
                
        except StateCorruptionError:
            # Room detected corruption and raised appropriate error
            pass
        except Exception as e:
            # Any other exception should be documented
            pytest.fail(f"Unexpected exception during recovery: {e}")
