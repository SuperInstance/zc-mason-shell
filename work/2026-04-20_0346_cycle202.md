## 5 Additional Edge-Case Tests

### Test 1: Overlapping Constraint Boundaries
**Scenario:** Constraints with overlapping negative spaces that create complex safe channels.
**Purpose:** Test the room's ability to resolve intersecting P0 regions and find valid P1 channels.
```python
def test_overlapping_constraint_boundaries():
    """Test DeadbandRoom with constraints that have overlapping negative spaces."""
    constraints = [
        {"name": "temp", "bounds": [(-10, 50)], "type": "range"},
        {"name": "temp", "bounds": [(30, 100)], "type": "range"},  # Overlaps with first
        {"name": "pressure", "bounds": [(0, 20)], "type": "range"}
    ]
    
    room = DeadbandRoom(constraints=constraints)
    
    # P0: Should identify combined negative space
    negative_space = room.map_negative_space()
    assert "temp" in negative_space
    assert len(negative_space["temp"]) == 2  # Both constraint ranges
    
    # P1: Should find channels avoiding both temp ranges
    channels = room.find_safe_channels()
    assert len(channels) > 0
    # Valid channels should have temp outside both ranges
    for channel in channels:
        assert channel["temp"] < -10 or channel["temp"] > 100
```

### Test 2: Conflicting Constraint Types

```python
def test_conflicting_constraint_types():

        {"name": "threshold", "bounds": [(0, 1)], "type": "range"},

    negative_space = room.map_negative_space()

    channels = room.find_safe_channels()

        assert 0 <= channel["threshold"] <= 1
        assert channel["custom"] % 2 == 0
```

### Test 3: High-Dimensional Constraint Space

```python
def test_high_dimensional_constraint_space():
    """Test DeadbandRoom with many constraints creating sparse solution space."""
    constraints = []
    for i in range(15):
        constraints.append({
            "name": f"dim_{i}",
            "bounds": [(i, i + 5)],
            "type": "range"
        })
    
    room = DeadbandRoom(constraints=constraints, max_iterations=1000)
    
    # P0: Should map all dimensions
    negative_space = room.map_negative_space()
    assert len(negative_space) == 15

    channels = room.find_safe_channels()
```

### Test 4: Dynamic Constraint Modification
**Scenario:** Constraints modified after room initialization.
**Purpose:** Test the room's response to runtime constraint changes.
```python
def test_dynamic_constraint_modification():
    """Test DeadbandRoom when constraints are modified after initialization."""
    constraints = [
        {"name": "speed", "bounds": [(0, 100)], "type": "range"}
    ]
    
    room = DeadbandRoom(constraints=constraints)
    
    # Initial P1
    channels = room.find_safe_channels()
    assert all(0 <= c["speed"] <= 100 for c in channels)
    
    # Modify constraints
    room.constraints = [
        {"name": "speed", "bounds": [(50, 150)], "type": "range"}
    ]
    
    # New P1 should reflect updated constraints
    new_channels = room.find_safe_channels()
    assert all(50 <= c["speed"] <= 150 for c in new_channels)
    
    # P0 should also update
    negative_space = room.map_negative_space()
    assert negative_space["speed"] == [(50, 150)]
```

### Test 5: Protocol Phase Transition Edge Cases
**Scenario:** Attempting P2 optimization from invalid starting points.
**Purpose:** Test proper error handling when protocol phases are misapplied.
```python
def test_protocol_phase_transition_edge_cases():
    """Test edge cases in protocol phase transitions."""
    constraints = [
        {"name": "value", "bounds": [(10, 20)], "type": "range"}
    ]
    
    room = DeadbandRoom(constraints=constraints)
    
    # Case 1: Valid P1 channel, then P2 optimization
    channels = room.find_safe_channels()
    assert channels
    start_channel = channels[0]
    
    optimized = room.optimize_within_channel(start_channel, objective="minimize")
    assert 10 <= optimized["value"] <= 20
    
    # Case 2: Attempt P2 from point in negative space (should fail)
    invalid_point = {"value": 5}  # Outside P1 bounds
    with pytest.raises(ValueError, match="Starting point not in safe channel"):
        room.optimize_within_channel(invalid_point, objective="minimize")
    
    # Case 3: Empty P1 result, then attempt P2 (should fail)
    impossible_constraints = [
        {"name": "impossible", "bounds": [(1, 2)], "type": "range"},
        {"name": "impossible", "bounds": [(3, 4)], "type": "range"}  # No overlap
    ]
    impossible_room = DeadbandRoom(constraints=impossible_constraints)
    empty_channels = impossible_room.find_safe_channels()
    assert empty_channels == []
    
    # Cannot optimize from empty channels
    with pytest.raises(ValueError, match="No safe channels available"):
        impossible_room.optimize_within_channel({}, objective="minimize")
```
