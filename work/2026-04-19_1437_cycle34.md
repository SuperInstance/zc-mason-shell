## 5 Additional Edge-Case Tests

### Test 1: P0 Rejection When Negative Space is Empty
**Scenario:** Agent submits P0 tile with empty `negative_space` list.
**Expected:** Tile rejected with error "P0 requires at least one negative space mapping."
**Rationale:** Ensures P0 phase actually maps prohibitions before proceeding.

```python
def test_p0_rejects_empty_negative_space(self):
    """P0 tile with empty negative_space list should be rejected."""
    tile = {
        "phase": "P0",
        "negative_space": [],  # Empty list
        "channel": None,
        "optimization": None,
        "metadata": {"agent": "test", "timestamp": "2026-04-19T14:30:00Z"}
    }
    result = self.room.submit_tile(tile)
    self.assertFalse(result["accepted"])
    self.assertIn("P0 requires at least one negative space mapping", result["error"])
```

### Test 2: P1 Channel Exhaustion (No Safe Channels)
**Scenario:** After mapping negative space, agent cannot find any safe channels.
**Expected:** Room remains in P1 state with empty channels list; tile accepted but marked "stalled".
**Rationale:** Tests deadband protocol behavior when no progress is possible.

```python
def test_p1_channel_exhaustion(self):
    """P1 tile with empty channels list indicates exhaustion."""
    # First submit valid P0
    self.room.submit_tile({
        "phase": "P0",
        "negative_space": ["rock", "wall", "pit"],
        "channel": None,
        "optimization": None,
        "metadata": {"agent": "test"}
    })
    
    # Submit P1 with no channels found
    tile = {
        "phase": "P1",
        "negative_space": [],
        "channel": [],  # Empty channel list
        "optimization": None,
        "metadata": {"agent": "test", "timestamp": "2026-04-19T14:31:00Z"}
    }
    result = self.room.submit_tile(tile)
    self.assertTrue(result["accepted"])
    self.assertEqual(self.room.current_phase, "P1")
    self.assertEqual(self.room.safe_channels, [])
```

### Test 3: P2 Optimization With Conflicting Constraints
**Scenario:** Agent submits P2 optimization that violates previously mapped negative space.
**Expected:** Tile rejected with error "Optimization violates P0 constraints."
**Rationale:** Ensures P2 doesn't undo P0 safety mappings.

```python
def test_p2_rejects_constraint_violation(self):
    """P2 optimization that violates P0 negative space should be rejected."""
    # Setup P0 and P1
    self.room.submit_tile({
        "phase": "P0",
        "negative_space": ["avoid_water"],
        "channel": None,
        "optimization": None,
        "metadata": {"agent": "test"}
    })
    self.room.submit_tile({
        "phase": "P1",
        "negative_space": [],
        "channel": ["path_a", "path_b"],
        "optimization": None,
        "metadata": {"agent": "test"}
    })
    
    # Submit P2 that violates P0
    tile = {
        "phase": "P2",
        "negative_space": [],
        "channel": "path_a",
        "optimization": {"path": "through_water"},  # Violates "avoid_water"
        "metadata": {"agent": "test", "timestamp": "2026-04-19T14:32:00Z"}
    }
    result = self.room.submit_tile(tile)
    self.assertFalse(result["accepted"])
    self.assertIn("violates P0 constraints", result["error"])
```

### Test 5: Concurrent Tile Submission Race Condition
**Scenario:** Simulate two agents submitting tiles simultaneously.
**Expected:** Room state updates atomically; last valid tile determines state.
**Rationale:** Tests thread safety in tile processing.

```python
def test_concurrent_tile_submission(self):
    """Multiple tiles submitted in rapid succession should process correctly."""
    import threading
    
    results = []
    
    def submit_tile_thread(tile_data):
        result = self.room.submit_tile(tile_data)
        results.append(result)
    
    # Create two competing tiles
    thread1 = threading.Thread(target=submit_tile_thread, args=({
        "phase": "P0",
        "negative_space": ["rock"],
        "channel": None,
        "optimization": None,
        "metadata": {"agent": "agent1", "timestamp": "2026-04-19T14:33:00Z"}
    },))
    
    thread2 = threading.Thread(target=submit_tile_thread, args=({
        "phase": "P0",
        "negative_space": ["wall"],
        "channel": None,
        "optimization": None,
        "metadata": {"agent": "agent2", "timestamp": "2026-04-19T14:33:01Z"}
    },))
    
    thread1.start()
    thread2.start()
    thread1.join()
    thread2.join()
    
    # One should be accepted, one rejected (or both accepted with merge)
    accepted_count = sum(1 for r in results if r.get("accepted", False))
    self.assertGreaterEqual(accepted_count, 1)
    self.assertLessEqual(accepted_count, 2)
```
