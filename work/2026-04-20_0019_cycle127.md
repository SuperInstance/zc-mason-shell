### 1. **Test: Concurrent Tile Submission Race Condition**
```python
def test_concurrent_tile_submission_race_condition():
    """Test that concurrent tile submissions from multiple agents don't corrupt room state."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (in rapid succession)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r["status"] == "accepted" for r in results)
    
    # Verify room state integrity
    assert len(room.tiles) == len(agent_ids)
    assert set(t["agent_id"] for t in room.tiles) == set(agent_ids)
```

### 4. **Test: Room State Persistence Across Resets**
```python
def test_room_state_persistence_across_resets():
    """Test that room maintains certain state (like ghost tiles) across soft resets."""
    room = DeadbandRoom()
    
    # Add some tiles including a ghost tile
    tiles = [
        {"agent_id": "active_agent", "phase": "P0", "content": "Active tile", "timestamp": time.time()},
        {"agent_id": "ghost_agent", "phase": "P0", "content": "Ghost tile", "timestamp": time.time(), "ghost": True},
        {"agent_id": "active_agent", "phase": "P1", "content": "Another active", "timestamp": time.time()},
    ]
    
    for tile in tiles:
        room.submit_tile(tile)
    
    # Verify initial state
    assert len(room.tiles) == 3
    ghost_tiles = [t for t in room.tiles if t.get("ghost")]
    assert len(ghost_tiles) == 1
    
    # Perform soft reset
    room.reset(soft=True)
    
    # Verify ghost tile persists, active tiles cleared
    assert len(room.tiles) == 1
    assert room.tiles[0].get("ghost") == True
    assert room.tiles[0]["agent_id"] == "ghost_agent"
    
    # Verify room is ready for new submissions
    new_tile = {
        "agent_id": "new_agent",
        "phase": "P0",
        "content": "Post-reset tile",
        "timestamp": time.time()
    }
    result = room.submit_tile(new_tile)
    assert result["status"] == "accepted"
    assert len(room.tiles) == 2
```

### 5. **Test: Tile Timestamp Validation and Ordering**
```python
def test_tile_timestamp_validation_and_ordering():
    """Test handling of tiles with invalid timestamps and correct temporal ordering."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    current_time = time.time()
    
    test_cases = [
        # Future timestamp (should be rejected or clamped)
        {"agent_id": agent_id, "phase": "P0", "content": "From future", "timestamp": current_time + 3600},
        # Past timestamp (far in past)
        {"agent_id": agent_id, "phase": "P0", "content": "From past", "timestamp": current_time - 86400},
        # Missing timestamp (should use server time)
        {"agent_id": agent_id, "phase": "P0", "content": "No timestamp"},
        # Invalid timestamp type
        {"agent_id": agent_id, "phase": "P0", "content": "Bad timestamp", "timestamp": "not-a-number"},
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(tile)
        
        if "timestamp" not in tile or not isinstance(tile.get("timestamp"), (int, float)):
            # Should be rejected or have timestamp added
            if result["status"] == "accepted":
                assert "timestamp" in room.tiles[-1]
                assert isinstance(room.tiles[-1]["timestamp"], (int, float))
        elif tile["timestamp"] > current_time + 300:  # 5 minutes in future
            # Future timestamps beyond threshold should be rejected or clamped
            assert result["status"] == "rejected" or room.tiles[-1]["timestamp"] <= current_time
    
    # Verify tiles are stored in chronological order
    timestamps = [t["timestamp"] for t in room.tiles if "timestamp" in t]
    assert timestamps == sorted(timestamps)
```
