### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(deadband_room):
    """Test handling of near-simultaneous tile submissions from different agents."""
    import threading
    
    results = []
    def submit_tile(agent_id):
        tile = Tile(
            content=f"Test from {agent_id}",
            priority=1,
            agent_id=agent_id,
            timestamp=time.time()
        )
        results.append(deadband_room.submit_tile(tile))
    
    threads = [
        threading.Thread(target=submit_tile, args=("agent_1",)),
        threading.Thread(target=submit_tile, args=("agent_2",)),
    ]
    
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # At least one submission should succeed
    assert any(r["accepted"] for r in results)
    # Room state should remain consistent
    assert deadband_room.state in ["active", "paused", "complete"]
```

def test_network_partition_recovery(deadband_room):
    """Test room recovery after simulated network partition."""

### Test 5: Cross-Room Tile Migration
```python
def test_cross_room_tile_migration(deadband_room):
    """Test migrating tiles between different DeadbandRoom instances."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    # Create source and destination rooms
    source_room = DeadbandRoom(room_id="source")
    dest_room = DeadbandRoom(room_id="dest")
    
    # Submit tile to source
    tile = Tile(
        content="Tile to migrate",
        priority=1,
        agent_id="migrating_agent",
        timestamp=time.time()
    )
    
    source_result = source_room.submit_tile(tile)
    assert source_result["accepted"] == True
    
    # Migrate tile from source to dest
    migrated = source_room.migrate_tile(tile.tile_id, dest_room)
    
    if migrated:
        # Verify tile removed from source
        assert tile.tile_id not in [t.tile_id for t in source_room.tiles]
        # Verify tile added to dest
        assert any(t.tile_id == tile.tile_id for t in dest_room.tiles)
        # Verify priority preserved
        dest_tile = next(t for t in dest_room.tiles if t.tile_id == tile.tile_id)
        assert dest_tile.priority == tile.priority
    else:
        # Migration failed - tile should remain in source
        assert any(t.tile_id == tile.tile_id for t in source_room.tiles)
```
