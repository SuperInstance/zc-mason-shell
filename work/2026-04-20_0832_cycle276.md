## 5 Additional Edge-Case Tests

### Test 1: Empty Constraint Sets
**Scenario:** What happens when P0, P1, or P2 constraints are empty lists?
**Edge Case:** Empty P0 (no prohibitions) should allow any action. Empty P1 (no safe channels) should make P2 optimization impossible.
**Test Purpose:** Verify room handles empty constraint definitions without crashing.
```python
def test_empty_constraint_sets():
    """Test DeadbandRoom with empty P0, P1, or P2 constraint lists."""
    room = DeadbandRoom(
        room_id="empty_constraints_test",
        p0_constraints=[],  # No prohibitions
        p1_constraints=[],  # No safe channels
        p2_constraints=[]   # No optimization criteria
    )
    
    # Should initialize without error
    assert room.room_id == "empty_constraints_test"
    
    # Tile with any action should pass P0 (no prohibitions)
    tile = {"action": "any_action", "phase": 0}
    result = room.submit_tile(tile)
    # Expect rejection at P1 phase since no safe channels defined
    assert not result["accepted"]
    assert "no safe channels" in result["reason"].lower()
```

### Test 2: Overlapping Constraint Conflicts
**Scenario:** When a constraint appears in both P0 (prohibited) and P1 (safe).
**Edge Case:** Contradictory constraints where the same action is both forbidden and allowed.
**Test Purpose:** Test constraint resolution priority (P0 should override P1).
```python
def test_p0_overrides_p1_conflict():
    """Test that P0 prohibitions override P1 safe channels when conflicting."""
    room = DeadbandRoom(
        room_id="conflict_test",
        p0_constraints=["use_absolute_claims"],  # Prohibited
        p1_constraints=["use_absolute_claims"],  # But also marked as safe
        p2_constraints=["minimize_word_count"]
    )
    
    tile = {
        "action": "write_output",
        "content": "This is guaranteed to work always.",
        "phase": 1
    }
    
    result = room.submit_tile(tile)
    # Should be rejected at P0 despite being in P1
    assert not result["accepted"]
    assert "p0 violation" in result["reason"].lower()
    assert "absolute" in result["reason"].lower()
```

### Test 4: Malformed Tile Recovery
**Scenario:** Tile submission that crashes mid-processing.
**Edge Case:** Tile with valid JSON but invalid internal structure that causes exceptions during processing.
**Test Purpose:** Verify room handles processing errors gracefully without corrupting state.
```python
def test_malformed_tile_recovery():
    """Test room recovery after processing a malformed tile."""
    room = DeadbandRoom(
        room_id="recovery_test",
        p0_constraints=["prohibit_crash"],
        p1_constraints=["allow_safe"],
        p2_constraints=["optimize_speed"]
    )
    
    # First, a valid tile to establish baseline
    valid_tile = {"action": "safe_action", "phase": 1}
    valid_result = room.submit_tile(valid_tile)
    initial_state = room.get_state()
    
    # Malformed tile that might cause processing error
    # Simulating a tile with circular reference or unserializable object
    class Unserializable:
        def __init__(self):
            self.self_ref = self
    
    malformed_tile = {
        "action": "problematic",
        "phase": 2,
        "problematic_data": Unserializable()  # This should fail
    }
    
    # Room should catch exception and reject tile
    result = room.submit_tile(malformed_tile)
    assert not result["accepted"]
    assert "processing error" in result["reason"].lower()
    
    # Room state should remain intact
    current_state = room.get_state()
    assert current_state["total_tiles"] == initial_state["total_tiles"] + 1
    assert current_state["accepted_tiles"] == initial_state["accepted_tiles"]
```

### Test 5: Protocol Phase Skipping with Partial Compliance
**Scenario:** Tile attempts to skip from P0 to P2 but has partial P1 compliance.
**Edge Case:** Mixed compliance where some P1 constraints are met but not all.
**Test Purpose:** Test nuanced protocol violation detection.
```python
def test_partial_p1_compliance_skipping():
    """Test detection when tile partially complies with P1 but skips to P2."""
    room = DeadbandRoom(
        room_id="partial_compliance_test",
        p0_constraints=["use_absolute_language"],
        p1_constraints=["cite_sources", "be_specific", "avoid_hallucination"],
        p2_constraints=["minimize_token_count"]
    )
    
    # Tile that meets 2/3 P1 constraints but claims phase 2
    tile = {
        "action": "generate_response",
        "content": "Based on fleet-knowledge/FLEET-CONTEXT.md...",
        "phase": 2,  # Skipping P1 declaration
        "cites_sources": True,  # Meets one P1
        "is_specific": True,    # Meets another P1
        "avoids_hallucination": False  # Fails one P1
    }
    
    result = room.submit_tile(tile)
    
    # Should be rejected for protocol violation (skipping phase)
    # Not for partial P1 compliance
    assert not result["accepted"]
    assert "protocol violation" in result["reason"].lower()
    assert "skip" in result["reason"].lower()
    
    # The rejection reason should mention phase skipping, not just P1 failure
    assert "phase" in result["reason"].lower()
```
