### Test 2: Overlapping Safe Channels (P1)
```python
def test_deadband_room_p1_overlapping_channels():
    """Test discovery of overlapping safe channels."""
    room = DeadbandRoom()
    # Provide constraints that create overlapping valid regions
    constraints = [
        {"type": "avoid", "region": "A"},
        {"type": "avoid", "region": "B"}
    ]
    room.apply_p0(constraints)
    
    channels = room.discover_p1_channels()
    # Should identify channels that don't violate any constraints
    assert len(channels) > 0
    # Verify no channel contains avoided regions
    for channel in channels:
        assert "A" not in channel["regions"]
        assert "B" not in channel["regions"]
```
