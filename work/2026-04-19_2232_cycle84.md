### 1.1 DeadbandRoom Source (`./plato-torch/src/rooms/deadband_room.py`)
**File content (abbreviated for clarity):**
```python
"""
DeadbandRoom - A PLATO room preset implementing the Deadband Protocol.

P0: Map negative space (what NOT to do)
P1: Find safe channels (where you CAN be)
P2: Optimize within channels (best path)

This room processes agent interactions and enforces the protocol strictly.
"""

from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass
import numpy as np

@dataclass
class AgentState:
    """Current state of an agent in the room."""
    agent_id: str
    position: np.ndarray  # 2D coordinates
    velocity: np.ndarray
    priority_level: int  # 0, 1, or 2
    safe_channels: List[Tuple[float, float]]  # list of (min, max) bounds
    violations: List[str]  # P0 violations logged

class DeadbandRoom:
    """Main room class implementing the Deadband Protocol."""
    
    def __init__(self, room_id: str, bounds: Tuple[float, float, float, float]):
        """
        Initialize a DeadbandRoom.
        
        Args:
            room_id: Unique identifier for the room
            bounds: (x_min, x_max, y_min, y_max) defining the room boundaries
        """
        self.room_id = room_id
        self.bounds = bounds
        self.agents: Dict[str, AgentState] = {}
        self.tiles_submitted = 0
        self.tiles_accepted = 0
        
    def add_agent(self, agent_id: str, position: np.ndarray) -> bool:
        """
        Add an agent to the room.
        
        Returns:
            True if agent added successfully, False if position is invalid.
        """
        if not self._is_within_bounds(position):
            return False
        self.agents[agent_id] = AgentState(
            agent_id=agent_id,
            position=position,
            velocity=np.array([0.0, 0.0]),
            priority_level=0,
            safe_channels=[],
            violations=[]
        )
        return True
    
    def submit_tile(self, agent_id: str, tile_data: Dict) -> Tuple[bool, str]:
        """
        Submit a training tile from an agent.
        
        The room validates the tile against the Deadband Protocol.
        
        Returns:
            (success: bool, message: str)
        """
        if agent_id not in self.agents:
            return False, "Agent not in room"
        
        agent = self.agents[agent_id]
        
        # P0: Check for violations
        p0_violation = self._check_p0_violations(tile_data)
        if p0_violation:
            agent.violations.append(p0_violation)
            return False, f"P0 violation: {p0_violation}"
        
        # P1: Update safe channels based on tile content
        new_channels = self._extract_safe_channels(tile_data)
        if new_channels:
            agent.safe_channels = new_channels
            agent.priority_level = 1
        
        # P2: If agent has safe channels, allow optimization
        if agent.priority_level >= 1 and self._is_p2_optimization(tile_data):
            agent.priority_level = 2
        
        self.tiles_submitted += 1
        self.tiles_accepted += 1
        return True, "Tile accepted"
    
    def _is_within_bounds(self, position: np.ndarray) -> bool:
        """Check if position is within room bounds."""
        x, y = position
        x_min, x_max, y_min, y_max = self.bounds
        return (x_min <= x <= x_max) and (y_min <= y <= y_max)
    
    def _check_p0_violations(self, tile_data: Dict) -> Optional[str]:
        """
        Check tile data for P0 violations.
        
        Returns violation description if found, None otherwise.
        """
        # Example violations:
        if tile_data.get("content", "").strip() == "":
            return "Empty tile content"
        if "hallucination" in tile_data.get("content", "").lower():
            return "Contains hallucination keyword"
        if tile_data.get("priority", 0) > 2:
            return "Invalid priority level"
        return None
    
    def _extract_safe_channels(self, tile_data: Dict) -> List[Tuple[float, float]]:
        """
        Extract safe channels from tile data.
        
        Returns list of (min, max) bounds.
        """
        # Simplified: extract from tile metadata
        channels = tile_data.get("safe_channels", [])
        if isinstance(channels, list) and all(len(c) == 2 for c in channels):
            return [(float(c[0]), float(c[1])) for c in channels]
        return []
    
    def _is_p2_optimization(self, tile_data: Dict) -> bool:
        """Check if tile contains P2 optimization content."""
        content = tile_data.get("content", "").lower()
        return "optimize" in content or "best path" in content
    
    def get_agent_stats(self, agent_id: str) -> Optional[Dict]:
        """Get statistics for an agent."""
        if agent_id not in self.agents:
            return None
        agent = self.agents[agent_id]
        return {
            "agent_id": agent.agent_id,
            "position": agent.position.tolist(),
            "priority_level": agent.priority_level,
            "safe_channels": agent.safe_channels,
            "violations": agent.violations
        }
```
