```rust
#[tokio::test]
async fn test_deadband_protocol_violation_skip_to_p2() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use plato_torch::core::protocols::DeadbandViolation;
    
    let config = DeadbandRoomConfig {
        max_iterations: 100,
        p0_threshold: 0.1,
        p1_threshold: 0.3,
        p2_threshold: 0.7,
        allow_protocol_skip: false, // Explicitly disallow skipping
    };
    
    let mut room = DeadbandRoom::new(config);
    let init_result = room.initialize().await;
    assert!(init_result.is_ok());
    
    // Simulate agent trying to skip directly to P2 optimization
    let violation_result = room.process_action(
        "skip_to_p2",
        vec!["optimize", "without", "p0_p1"]
    ).await;
    
    // Should detect protocol violation
    assert!(violation_result.is_err());
    
    if let Err(e) = violation_result {
        assert!(e.to_string().contains("DeadbandProtocolViolation"));
        assert!(e.to_string().contains("P2"));
    }
    
    // Room should enter error state
    assert_eq!(room.state(), RoomState::Error);
}
```

### Test 2: Negative Space Overflow - Too Many P0 Constraints
```rust
#[tokio::test]
async fn test_negative_space_overflow() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use plato_torch::core::constraints::ConstraintOverflow;
    
    let config = DeadbandRoomConfig {
        max_iterations: 50,
        p0_threshold: 0.05, // Very low threshold
        p1_threshold: 0.2,
        p2_threshold: 0.5,
        max_constraints: 10, // Limit on P0 constraints
    };
    
    let mut room = DeadbandRoom::new(config);
    let init_result = room.initialize().await;
    assert!(init_result.is_ok());
    
    // Add constraints until overflow
    for i in 0..15 {
        let constraint_result = room.add_constraint(
            format!("p0_constraint_{}", i),
            ConstraintType::NegativeSpace
        ).await;
        
        if i < 10 {
            assert!(constraint_result.is_ok());
        } else {
            // Should overflow after 10 constraints
            assert!(constraint_result.is_err());
            if let Err(e) = constraint_result {
                assert!(e.to_string().contains("ConstraintOverflow"));
            }
            break;
        }
    }
    
    // Room should handle overflow gracefully
    assert_ne!(room.state(), RoomState::Crashed);
}
```

### Test 3: Channel Boundary Conditions - P1 Threshold Edge Cases
```rust
#[tokio::test]
async fn test_p1_channel_boundary_conditions() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use plato_torch::core::channels::ChannelState;
    
    let config = DeadbandRoomConfig {
        max_iterations: 100,
        p0_threshold: 0.1,
        p1_threshold: 0.299999, // Just below 0.3 boundary
        p2_threshold: 0.7,
        floating_point_tolerance: 0.000001,
    };
    
    let mut room = DeadbandRoom::new(config);
    let init_result = room.initialize().await;
    assert!(init_result.is_ok());
    
    // Add metrics that should keep us in P0
    for _ in 0..5 {
        room.add_metric("p0_metric", 0.05).await.unwrap();
    }
    
    // Try to transition to P1
    let transition_result = room.transition_to_p1().await;
    
    // Should fail because we're below P1 threshold
    assert!(transition_result.is_err());
    assert!(room.current_channel() == ChannelState::P0);
    
    // Now add metrics to reach exactly P1 threshold
    room.add_metric("boundary_metric", 0.25).await.unwrap();
    
    // With floating point tolerance, should now transition
    let transition_result = room.transition_to_p1().await;
    assert!(transition_result.is_ok());
    assert!(room.current_channel() == ChannelState::P1);
}
```

### Test 4: Concurrent Agent Collision in Safe Channels
```rust
#[tokio::test]
async fn test_concurrent_agent_collision_p1() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use std::sync::Arc;
    use tokio::task;
    
    let config = DeadbandRoomConfig {
        max_iterations: 100,
        p0_threshold: 0.1,
        p1_threshold: 0.3,
        p2_threshold: 0.7,
        max_concurrent_agents: 3,
    };
    
    let room = Arc::new(tokio::sync::Mutex::new(DeadbandRoom::new(config)));
    let init_result = room.lock().await.initialize().await;
    assert!(init_result.is_ok());
    
    // Simulate multiple agents trying to enter same P1 channel simultaneously
    let mut handles = vec![];
    for agent_id in 0..5 {
        let room_clone = Arc::clone(&room);
        let handle = task::spawn(async move {
            let mut room = room_clone.lock().await;
            
            // Each agent tries to claim a position in P1 channel
            room.enter_channel(
                ChannelState::P1,
                format!("agent_{}", agent_id)
            ).await
        });
        handles.push(handle);
    }
    
    let mut successes = 0;
    let mut collisions = 0;
    
    for handle in handles {
        let result = handle.await.unwrap();
        match result {
            Ok(_) => successes += 1,
            Err(e) if e.to_string().contains("ChannelOccupied") => collisions += 1,
            _ => {}
        }
    }
    
    // Should have exactly 3 successes (max_concurrent_agents)
    // and 2 collisions (5 agents - 3 slots)
    assert_eq!(successes, 3);
    assert_eq!(collisions, 2);
}
```
