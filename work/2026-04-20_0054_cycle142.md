```python
def test_concurrent_tile_submission(self):
    """Multiple agents should be able to submit tiles without state corruption."""
    room = DeadbandRoom(room_id="concurrent_test")
    agent_ids = ["agent_alpha", "agent_beta", "agent_gamma"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Safe channel analysis from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions succeeded
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent tile submission should not corrupt room state"
    
    # Verify tile count matches
    assert len(room.get_tiles()) == len(agent_ids), \
        "Room should contain all submitted tiles"
```

```python
def test_malformed_tile_validation(self):

    room = DeadbandRoom(room_id="validation_test")
    
    test_cases = [
        # Missing required fields
        {"phase": "P1", "content": "test"},
        {"agent_id": "test_agent", "content": "test"},

        # Invalid phase values
        {"agent_id": "test_agent", "phase": "P3", "content": "test"},
        {"agent_id": "test_agent", "phase": "invalid", "content": "test"},
        
        # Malformed content (non-string)
        {"agent_id": "test_agent", "phase": "P1", "content": 123},
        {"agent_id": "test_agent", "phase": "P1", "content": None},
        
        # Empty content
        {"agent_id": "test_agent", "phase": "P1", "content": ""},
        {"agent_id": "test_agent", "phase": "P1", "content": "   "},
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(tile)
        assert result["status"] == "rejected", \
            f"Test case {i} should be rejected: {tile}"
        assert "validation_error" in result, \
            f"Rejection should include error details: {tile}"
```

### Test 3: Room Capacity Limits
```python
def test_room_capacity_limits(self):
    """Room should enforce maximum tile and agent limits."""
    room = DeadbandRoom(room_id="capacity_test", max_tiles=10, max_agents=3)
    
    # Test tile limit
    for i in range(15):  # Exceeds 10 tile limit
        tile = {
            "agent_id": f"agent_{i % 3}",
            "phase": "P1",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        
        if i < 10:
            assert result["status"] == "accepted", \
                f"Tile {i} should be accepted (under limit)"
        else:
            assert result["status"] == "rejected", \
                f"Tile {i} should be rejected (over limit)"
            assert "capacity_limit" in result.get("reason", "").lower(), \
                f"Rejection reason should mention capacity: {result}"
    
    # Test agent limit (implicit via unique agent tracking)
    unique_agents = set()
    for tile in room.get_tiles():
        unique_agents.add(tile["agent_id"])
    
    assert len(unique_agents) <= 3, \
        f"Room should not exceed max agents. Found: {len(unique_agents)}"
```

### Test 4: Network Partition Simulation
```python
def test_network_partition_recovery(self):
    """Room should handle partial state sync after network issues."""
    room = DeadbandRoom(room_id="partition_test")
    
    # Create initial state
    initial_tiles = []
    for i in range(5):
        tile = {
            "agent_id": "primary_agent",
            "phase": ["P0", "P1", "P2"][i % 3],
            "content": f"Initial tile {i}",
            "timestamp": time.time() - 100 + i  # Staggered timestamps
        }
        room.submit_tile(tile)
        initial_tiles.append(tile)
    
    # Simulate network partition - create isolated room instance
    partitioned_room = DeadbandRoom(room_id="partition_test")
    
    # Submit tiles during partition (should create divergent state)
    partition_tile = {
        "agent_id": "isolated_agent",
        "phase": "P1",
        "content": "Submitted during network partition",
        "timestamp": time.time()
    }
    partitioned_room.submit_tile(partition_tile)
    
    # Simulate network recovery - attempt state merge
    merge_result = room.merge_state(partitioned_room.get_state())
    
    # Verify merge preserves all tiles
    final_tiles = room.get_tiles()
    assert len(final_tiles) == 6, \
        f"Merge should preserve all tiles. Found: {len(final_tiles)}"
    
    # Verify conflict resolution (timestamp-based)
    tile_contents = [t["content"] for t in final_tiles]
    assert "Submitted during network partition" in tile_contents, \
        "Partition-submitted tile should be preserved"
```
