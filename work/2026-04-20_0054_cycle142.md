```python
def test_protocol_violation_skip_to_p2():
    """P0: Agent attempts to optimize (P2) without mapping negative space (P0) first."""
    room = DeadbandRoom()
    agent_id = "test_agent_violator"
    
    # Attempt P2 operation without completing P0
    with pytest.raises(ProtocolViolationError) as exc:

            "priority": 2
        })

    assert room.get_agent_state(agent_id)["violations"] == 1
```

### Test 3: Concurrent Agent Isolation
```python
def test_concurrent_agent_isolation():
    """P1: Multiple agents in same room should not interfere with each other's state."""
    room = DeadbandRoom()
    agent_a = "agent_alpha"
    agent_b = "agent_beta"
    
    # Agent A completes P0
    room.submit_tile(agent_a, {
        "phase": "P0",
        "content": "A's negative space",
        "priority": 0
    })
    
    # Agent B should start fresh at P0
    assert room.get_agent_state(agent_b)["current_phase"] == "P0"
    
    # Agent B submits P0
    room.submit_tile(agent_b, {
        "phase": "P0",
        "content": "B's different negative space",
        "priority": 0
    })
    
    # Both should have independent states
    state_a = room.get_agent_state(agent_a)
    state_b = room.get_agent_state(agent_b)
    
    assert state_a["current_phase"] == "P1"
    assert state_b["current_phase"] == "P1"
    assert state_a["tile_count"] == 1
    assert state_b["tile_count"] == 1
    assert state_a["last_content"] != state_b["last_content"]
```

```python
def test_resource_limit_enforcement():
    """P0: Room should enforce tile storage limits."""
    room = DeadbandRoom(max_tiles_per_agent=3)
    agent_id = "test_agent_hoarder"
    
    # Fill up to limit
    for i in range(3):
        room.submit_tile(agent_id, {
            "phase": ["P0", "P1", "P2"][i],
            "content": f"Tile {i}",
            "priority": i
        })
    
    # Attempt to exceed limit
    result = room.submit_tile(agent_id, {
        "phase": "P2",
        "content": "Excess tile",
        "priority": 2
    })
    
    assert result["status"] == "rejected"
    assert "tile limit exceeded" in result["message"]
    
    # Verify count remains at limit
    assert room.get_agent_state(agent_id)["tile_count"] == 3
    
    # Room should still accept tiles from other agents
    other_agent = "test_agent_fresh"
    result = room.submit_tile(other_agent, {
        "phase": "P0",
        "content": "New agent tile",
        "priority": 0
    })
    assert result["status"] == "accepted"
```
