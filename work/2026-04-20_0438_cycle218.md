```python
def test_p1_safe_channels_empty_constraint_list():
    """Test P1 phase with empty constraint list (should return all channels as safe)"""
    room = DeadbandRoom(config=valid_config)
    room.current_phase = "P1"
    
    # Empty constraints means no restrictions
    safe_channels = room.find_safe_channels(constraints=[])
    
    # With no constraints, all possible channels should be considered safe
    assert len(safe_channels) == len(room.ALL_POSSIBLE_CHANNELS)
    assert set(safe_channels) == set(room.ALL_POSSIBLE_CHANNELS)
```

```python
def test_protocol_phase_regression():
```

### Test 4: Concurrent Operations - Simulated Race Condition
```python
def test_concurrent_phase_transitions():
    """Test handling of concurrent phase transition attempts"""
    room = DeadbandRoom(config=valid_config)
    room.current_phase = "P0"
    
    # Simulate concurrent attempts to advance phase
    def attempt_transition(target_phase):
        try:
            room.current_phase = target_phase
            return True
        except ProtocolViolationError:
            return False
    
    # Multiple threads/processes attempting different transitions
    results = []
    for target_phase in ["P1", "P2", "P1"]:  # Note duplicate P1
        results.append(attempt_transition(target_phase))
    
    # Only first valid transition should succeed
    assert results == [True, False, False]  # P0→P1 ok, P0→P2 fails, second P1 fails
    assert room.current_phase == "P1"
```

### Test 5: Recovery Path - After Protocol Violation
```python
def test_recovery_after_skip_detection():
    """Test room recovery after detecting P0→P2 skip"""
    room = DeadbandRoom(config=valid_config)
    room.current_phase = "P0"
    
    # First, cause a protocol violation
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.execute_phase("P2", sample_data)  # Skip P1
    
    assert "Cannot skip from P0 to P2" in str(exc_info.value)
    
    # Verify room enters recovery state
    assert room.in_recovery_mode is True
    assert room.recovery_data["last_violation"] == "P0→P2 skip"
    
    # Should be able to continue with correct phase
    room.execute_phase("P1", sample_data)  # This should work
    assert room.current_phase == "P1"
    assert room.in_recovery_mode is False  # Recovery complete
```
