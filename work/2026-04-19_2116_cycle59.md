### Test 1: Empty Negative Space (No P0 Constraints)
```rust
#[test]
fn test_empty_negative_space() {
    let mut room = DeadbandRoom::new();
    let constraints = Constraints {
        p0: vec![], // No negative space mapped
        p1: vec![SafeChannel { bounds: (0.0, 10.0) }],
        p2: None,
    };
    
    let result = room.apply_protocol(&constraints);
    assert!(result.is_ok());
    // With no P0, P1 channel should be directly available
    assert_eq!(result.unwrap().selected_channel, Some((0.0, 10.0)));
}
```

### Test 2: Overlapping Safe Channels in P1
```rust
#[test]
fn test_overlapping_p1_channels() {
    let mut room = DeadbandRoom::new();
    let constraints = Constraints {
        p0: vec![NegativeRegion { bounds: (5.0, 7.0) }],
        p1: vec![
            SafeChannel { bounds: (0.0, 6.0) },
            SafeChannel { bounds: (4.0, 10.0) },
            SafeChannel { bounds: (6.5, 9.0) },
        ],
        p2: None,
    };
    
    let result = room.apply_protocol(&constraints);
    assert!(result.is_ok());
    let solution = result.unwrap();
    // System should select one valid channel despite overlap
    assert!(solution.selected_channel.is_some());
    // Verify selected channel doesn't intersect P0
    let channel = solution.selected_channel.unwrap();
    assert!(!ranges_overlap(channel, (5.0, 7.0)));
}
```

### Test 4: High-Dimensional Constraint Space
```rust
#[test]
fn test_4d_constraint_space() {
    let mut room = DeadbandRoom::new();
    // 4D example: (x, y, z, t) with constraints in each dimension
    let constraints = Constraints {
        p0: vec![
            NegativeRegion { bounds: (0.0, 2.0) }, // x-axis
            NegativeRegion { bounds: (5.0, 8.0) }, // y-axis
        ],
        p1: vec![
            SafeChannel { bounds: (2.0, 10.0) }, // x-axis
            SafeChannel { bounds: (0.0, 5.0) },  // y-axis
            SafeChannel { bounds: (0.0, 100.0) }, // z-axis (unconstrained)
            SafeChannel { bounds: (0.0, 1.0) },   // t-axis
        ],
        p2: Some(OptimizationTarget::MinimizeDistance),
    };
    
    let result = room.apply_protocol(&constraints);
    assert!(result.is_ok());
    let solution = result.unwrap();
    // Should find intersection of all safe channels
    assert!(solution.selected_channel.is_some());
    // In 4D, channel is hyper-rectangle intersection
    assert_eq!(solution.dimensions, 4);
}
```

### Test 5: Protocol Interruption and State Reset
```rust
#[test]
fn test_protocol_interruption() {
    let mut room = DeadbandRoom::new();
    
    // First application
    let constraints1 = Constraints {
        p0: vec![NegativeRegion { bounds: (0.0, 5.0) }],
        p1: vec![SafeChannel { bounds: (5.0, 10.0) }],
        p2: None,
    };
    
    let result1 = room.apply_protocol(&constraints1);
    assert!(result1.is_ok());
    
    // Interrupt and reset room state
    room.reset();
    
    // Second application with different constraints
    let constraints2 = Constraints {
        p0: vec![NegativeRegion { bounds: (8.0, 10.0) }],
        p1: vec![SafeChannel { bounds: (0.0, 8.0) }],
        p2: None,
    };
    
    let result2 = room.apply_protocol(&constraints2);
    assert!(result2.is_ok());
    
    // Results should be independent
    assert_ne!(
        result1.unwrap().selected_channel,
        result2.unwrap().selected_channel
    );
}
```

## Helper Function for Range Overlap Check
```rust
fn ranges_overlap(a: (f64, f64), b: (f64, f64)) -> bool {
    a.0 < b.1 && b.0 < a.1
}
```
