## Edge-Case Test Proposals
Based on the Deadband Protocol's strict priority rules and the room's role in training, I identify these edge cases:

1. **Concurrent Phase Transitions** – Multiple agents attempting phase transitions simultaneously.
2. **Tile Validation Failures** – Server rejects tiles (malformed, duplicate, invalid signature).
3. **Agent Timeout/Disconnect** – Agent leaves mid-phase, room must handle orphaned state.
4. **Resource Exhaustion** – Memory/disk limits reached during tile accumulation.
5. **Protocol Violation Recovery** – Room detects P2→P0 jump and enforces reset logic.

### Test 1: Concurrent Phase Transitions
```python
def test_concurrent_phase_transitions(self):
    """Multiple agents should not corrupt phase state when transitioning concurrently."""
    room = DeadbandRoom(capacity=5)
    agent_ids = [f"agent_{i}" for i in range(3)]
    
    # All agents start at P0
    for aid in agent_ids:
        room.register_agent(aid)
    
    # Simulate concurrent P0→P1 attempts
    def transition_agent(aid):
        room.submit_tile(aid, {"phase": "P0", "content": f"test from {aid}"})
        room.advance_phase(aid)
    
    threads = [threading.Thread(target=transition_agent, args=(aid,)) for aid in agent_ids]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # All agents should be at P1, room phase state consistent
    for aid in agent_ids:
        self.assertEqual(room.get_agent_phase(aid), "P1")
    # Room should have recorded all transitions
    self.assertEqual(len(room.transition_log), 3)
```

### Test 2: Tile Validation Failure Recovery

def test_tile_validation_failure_recovery(self):
    """Room should handle server rejection gracefully and maintain agent state."""
    room = DeadbandRoom(capacity=3)
    agent_id = "test_agent"
    room.register_agent(agent_id)
    
    # Mock server to reject first tile
    with unittest.mock.patch.object(room.tile_server, 'submit') as mock_submit:
        mock_submit.side_effect = [
            TileRejectionError("Invalid signature"),
            {"status": "accepted", "tile_id": "123"}
        ]
        
        # First submission fails
        with self.assertRaises(TileRejectionError):
            room.submit_tile(agent_id, {"phase": "P0", "content": "bad tile"})
        
        # Agent remains at P0 (no phase advance)
        self.assertEqual(room.get_agent_phase(agent_id), "P0")
        
        # Second submission succeeds
        result = room.submit_tile(agent_id, {"phase": "P0", "content": "good tile"})
        self.assertEqual(result["status"], "accepted")
        
        # Now can advance
        room.advance_phase(agent_id)
        self.assertEqual(room.get_agent_phase(agent_id), "P1")
```

### Test 3: Agent Disconnect Mid-Phase

def test_agent_disconnect_mid_phase(self):
    """Room should clean up resources when agent disconnects during active phase."""
    room = DeadbandRoom(capacity=2)
    agent_id = "disconnecting_agent"
    room.register_agent(agent_id)
    room.submit_tile(agent_id, {"phase": "P0", "content": "initial"})
    room.advance_phase(agent_id)  # Now at P1
    
    # Agent starts P1 work but disconnects
    room.submit_tile(agent_id, {"phase": "P1", "content": "partial work"})
    
    # Simulate disconnect (timeout/cleanup)
    room.handle_agent_disconnect(agent_id)
    
    # Agent should be removed from active agents
    self.assertNotIn(agent_id, room.active_agents)
    # Phase resources should be cleaned up
    self.assertEqual(room.get_phase_resource_count("P1"), 0)
    # Tile should be marked as orphaned
    orphaned_tiles = room.get_orphaned_tiles()
    self.assertEqual(len(orphaned_tiles), 1)
    self.assertEqual(orphaned_tiles[0]["content"], "partial work")
```

### Test 4: Resource Exhaustion Handling
```python
def test_resource_exhaustion_handling(self):
    """Room should enforce limits and degrade gracefully when resources exhausted."""
    room = DeadbandRoom(capacity=2, max_tiles_per_phase=100)
    
    # Fill phase P0 with tiles up to limit
    agent_id = "resource_agent"
    room.register_agent(agent_id)
    
    with unittest.mock.patch('psutil.virtual_memory') as mock_mem:
        # Simulate memory exhaustion after 50 tiles
        mock_mem.return_value.percent = 95  # Critical memory usage
        
        with self.assertLogs(level='WARNING') as log:
            for i in range(55):  # Attempt to exceed safe limit
                try:
                    room.submit_tile(agent_id, {"phase": "P0", "content": f"tile_{i}"})
                except ResourceExhaustedError:
                    break
            
            # Should log warning and stop accepting tiles
            self.assertTrue(any("memory threshold" in msg for msg in log.output))
        
        # Room should have entered degraded mode
        self.assertTrue(room.degraded_mode)
        # Tile count should be limited
        self.assertLessEqual(room.get_tile_count("P0"), 50)
```

### Test 5: Protocol Violation Detection and Reset
```python
def test_protocol_violation_detection_reset(self):
    """Room should detect P2→P0 jumps and enforce reset to P1 with penalty."""
    room = DeadbandRoom(capacity=1)
    agent_id = "violating_agent"
    room.register_agent(agent_id)
    
    # Normal progression to P2
    room.submit_tile(agent_id, {"phase": "P0", "content": "p0 work"})
    room.advance_phase(agent_id)  # P1
    room.submit_tile(agent_id, {"phase": "P1", "content": "p1 work"})
    room.advance_phase(agent_id)  # P2
    room.submit_tile(agent_id, {"phase": "P2", "content": "p2 work"})
    
    # Agent attempts to jump back to P0 (violation)
    with self.assertRaises(ProtocolViolationError) as ctx:
        room.submit_tile(agent_id, {"phase": "P0", "content": "illegal regression"})
    
    self.assertIn("P2→P0", str(ctx.exception))
    
    # Room should auto-reset agent to P1 with penalty flag
    self.assertEqual(room.get_agent_phase(agent_id), "P1")
    self.assertTrue(room.get_agent_penalty(agent_id))
    # Previous P2 work should be marked as suspect
    p2_tiles = room.get_tiles_by_phase("P2")
    self.assertTrue(p2_tiles[0].get("suspect", False))
```
