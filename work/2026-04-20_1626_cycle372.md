### Test 3: Protocol State Machine Edge: P2 Without P1
**Scenario:** Attempt to execute P2 (optimization) step before P1 (channel finding) is complete.
**Rationale:** The Deadband Protocol requires strict P0→P1→P2 progression. The room should enforce this.
```python
def test_protocol_sequence_violation(tmp_path, mock_llm):
    """Test attempting P2 step before P1 is complete."""
    room = DeadbandRoom(
        name="test_sequence",
        data_dir=tmp_path,
        llm_client=mock_llm
    )
    
    # Start with P0 (always allowed)
    room.execute_protocol_step("P0")
    assert room.current_protocol_step == "P0"
    
    # Try to jump to P2 without P1
    with pytest.raises(RuntimeError) as exc_info:
        room.execute_protocol_step("P2")
    assert "P1" in str(exc_info.value)  # Error should mention missing P1
    
    # Complete P1 properly
    room.execute_protocol_step("P1")
    assert room.current_protocol_step == "P1"
    
    # Now P2 should work
    room.execute_protocol_step("P2")
    assert room.current_protocol_step == "P2"
```

```python
def test_corrupted_state_recovery(tmp_path, mock_llm):
    """Test room initialization with corrupted state file."""
    room_name = "test_corrupt"
    state_file = tmp_path / f"{room_name}_state.json"
    
    # Write invalid JSON to state file
    state_file.write_text("{invalid json: true")
    
    # Room should initialize despite corrupted state
    room = DeadbandRoom(
        name=room_name,
        data_dir=tmp_path,
        llm_client=mock_llm
    )
    
    # Should either load with defaults or create fresh state
    assert room.name == room_name
    # State should be either default or recovered
    assert room.current_protocol_step in ["P0", None]  # Starting state
    
    # New state file should be valid
    if state_file.exists():
        import json
        reloaded = json.loads(state_file.read_text())
        assert "current_protocol_step" in reloaded
```
