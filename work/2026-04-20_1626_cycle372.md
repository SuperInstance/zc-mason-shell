1. **P2 optimization with multiple valid channels** – When P1 returns >1 safe channel, does P2 select optimally?
2. **State corruption mid-cycle** – If room state is mutated externally during protocol execution.
3. **Tile validation failure cascade** – When a submitted tile is invalid, does room handle rejection without crashing?
4. **Concurrent agent interactions** – Simulated race condition where two agents submit tiles simultaneously.
5. **Resource exhaustion** – Memory/CPU spikes during large state processing.

```python
def test_handles_state_mutation_during_processing():
    """Edge: External mutation of state dict while room is processing."""
    import threading
    import time
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom(room_id="test_edge_2")
    shared_state = {
        "negative_space": [],
        "safe_channels": [{"id": "safe_1"}],
        "current_phase": "P1"
    }
    
    def mutate_state():
        time.sleep(0.001)  # Small delay
        shared_state["negative_space"].append("injected_rock")
    
    # Start mutation thread
    mutator = threading.Thread(target=mutate_state)
    mutator.start()
    
    # Process while mutation occurs
    result = room.process_agent_state(shared_state.copy(), agent_id="test_agent")
    
    mutator.join()
    
    # Room should either use initial state or handle mutation gracefully
    assert result is not None
    # Should not crash with KeyError or TypeError
```

```python
def test_recovers_after_tile_rejection():
    """Edge: Tile server rejects tile, room should not enter invalid state."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from unittest.mock import Mock
    
    room = DeadbandRoom(room_id="test_edge_3")
    
    # Mock tile server to reject first tile, accept second

        {"status": "rejected", "reason": "invalid_format"},
        {"status": "accepted"}
    ]
    
    state = {
        "negative_space": ["rock_x"],
        "safe_channels": [{"id": "safe_a"}],
        "current_phase": "P0"
    }
    
    # First attempt - should be rejected
    result1 = room.process_agent_state(state, agent_id="test_agent")
    assert result1.get("tile_status") == "rejected"
    
    # Second attempt with corrected state - should succeed
    state["corrected_format"] = True
    result2 = room.process_agent_state(state, agent_id="test_agent")
    assert result2.get("tile_status") == "accepted"
    
    # Room state should still be valid
    assert room.room_state is not None
```

### Test 4: Concurrent Agent Simulation
```python
def test_handles_concurrent_agent_submissions():
    """Edge: Simulate 3 agents submitting tiles simultaneously."""
    import concurrent.futures
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from unittest.mock import Mock
    
    room = DeadbandRoom(room_id="test_edge_4")
    room.tile_server = Mock()
    room.tile_server.submit_tile.return_value = {"status": "accepted"}
    
    def agent_action(agent_id):
        state = {
            "negative_space": [f"rock_{agent_id}"],
            "safe_channels": [{"id": f"channel_{agent_id}"}],
            "current_phase": "P1"
        }
        return room.process_agent_state(state, agent_id=agent_id)
    
    # Submit concurrently
    with concurrent.futures.ThreadPoolExecutor(max_workers=3) as executor:
        futures = [
            executor.submit(agent_action, f"agent_{i}")
            for i in range(3)
        ]
        results = [f.result() for f in concurrent.futures.as_completed(futures)]
    
    # All should complete without deadlock or corruption
    assert len(results) == 3
    assert all(r is not None for r in results)
    # Tile server should have been called 3 times
    assert room.tile_server.submit_tile.call_count == 3
```

```python
def test_processes_large_state_without_resource_exhaustion():
    """Edge: State with 10,000 negative space entries and 1,000 channels."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from unittest.mock import Mock
    import sys
    
    room = DeadbandRoom(room_id="test_edge_5")
    room.tile_server = Mock()
    room.tile_server.submit_tile.return_value = {"status": "accepted"}
    
    # Generate large state
    large_state = {
        "negative_space": [f"rock_{i}" for i in range(10000)],
        "safe_channels": [{"id": f"channel_{j}", "cost": j % 10, "reward": j % 20}
                         for j in range(1000)],
        "current_phase": "P1"
    }
    
    import resource
    mem_before = resource.getrusage(resource.RUSAGE_SELF).ru_maxrss
    
    result = room.process_agent_state(large_state, agent_id="stress_agent")
    
    mem_after = resource.getrusage(resource.RUSAGE_SELF).ru_maxrss
    mem_increase_kb = mem_after - mem_before
    
    # Should process without crashing
    assert result is not None
    # Memory increase should be reasonable (< 50MB for this operation)
    assert mem_increase_kb < 51200  # 50MB in KB
    # Should identify at least one safe channel
    assert len(result.get("identified_channels", [])) > 0
```
