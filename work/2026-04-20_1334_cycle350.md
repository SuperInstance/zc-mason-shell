### Test 1: Agent Leaving Room
```python
def test_agent_leaving_room():
    """Test that an agent can leave the room and is removed from active agents."""
    room = DeadbandRoom(config={"room_id": "test_leave"})
    agent_id = "agent_123"
    
    # Agent joins
    room.join_room(agent_id)
    assert agent_id in room.active_agents
    
    # Agent leaves
    result = room.leave_room(agent_id)
    assert result["success"] is True
    assert agent_id not in room.active_agents
    assert len(room.active_agents) == 0
```

### Test 3: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission():
    """Simulate concurrent tile submissions from multiple agents."""
    room = DeadbandRoom(config={"room_id": "test_concurrent"})
    
    # Add multiple agents
    agents = [f"agent_{i}" for i in range(5)]
    for agent_id in agents:
        room.join_room(agent_id)
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agents:
        tile = {
            "agent_id": agent_id,
            "content": f"Test content from {agent_id}",
            "timestamp": "2026-04-20T14:00:00Z"
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["success"] for r in results)
    assert len(room.pending_tiles) == len(agents)
```

### Test 4: Room Capacity Enforcement
```python
def test_room_capacity_limit():
    """Test that room enforces maximum agent capacity."""
    max_agents = 3
    room = DeadbandRoom(config={
        "room_id": "test_capacity",
        "max_agents": max_agents
    })
    
    # Fill room to capacity
    for i in range(max_agents):
        agent_id = f"agent_{i}"
        result = room.join_room(agent_id)
        assert result["success"] is True
    
    # Try to exceed capacity
    excess_agent = "agent_excess"
    result = room.join_room(excess_agent)
    assert result["success"] is False
    assert "capacity" in result.get("error", "").lower()
    assert excess_agent not in room.active_agents
```

### Test 5: State Persistence After Reset

```python
def test_state_persistence_simulation():
    """Test room state consistency after simulated reset/restart."""
    room = DeadbandRoom(config={"room_id": "test_persistence"})
    
    # Initial state
    agent_id = "persistent_agent"
    room.join_room(agent_id)
    
    initial_tile = {
        "agent_id": agent_id,
        "content": "Initial test content",
        "timestamp": "2026-04-20T14:00:00Z"
    }
    room.submit_tile(initial_tile)
    
    # Simulate room state capture
    captured_state = {
        "active_agents": room.active_agents.copy(),
        "pending_tiles": room.pending_tiles.copy(),
        "processed_tiles": room.processed_tiles.copy()
    }
    
    # Simulate restart - create new room with same config
    new_room = DeadbandRoom(config={"room_id": "test_persistence"})
    
    # Restore state (simulating persistence layer)
    new_room.active_agents = captured_state["active_agents"]
    new_room.pending_tiles = captured_state["pending_tiles"]
    new_room.processed_tiles = captured_state["processed_tiles"]
    
    # Verify state consistency
    assert agent_id in new_room.active_agents
    assert len(new_room.pending_tiles) == 1
    assert new_room.pending_tiles[0]["content"] == "Initial test content"
```
