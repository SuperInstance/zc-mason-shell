"""
plato-torch/tests/test_presets/test_deadband_room.py
Additional edge-case tests for DeadbandRoom.
"""

import pytest
import torch
from plato_torch.presets.deadband_room import DeadbandRoom

class TestDeadbandRoomEdgeCases:
    """Edge-case tests for DeadbandRoom preset."""
    
    def test_deadband_range_zero(self):
        """Edge case: deadband_range = 0 should behave like identity outside zero."""
        room = DeadbandRoom(deadband_range=0.0)
        # Values outside deadband (any non-zero) should pass unchanged
        assert room.apply_deadband(5.0, 0.0) == 5.0
        assert room.apply_deadband(-3.0, 0.0) == -3.0
        # Zero input should remain zero
        assert room.apply_deadband(0.0, 0.0) == 0.0
    
    def test_large_deadband_range_exceeds_input(self):
        """Edge case: deadband_range larger than input magnitude forces output to zero."""
        room = DeadbandRoom(deadband_range=10.0)
        # Input within deadband (|input| <= range)
        assert room.apply_deadband(5.0, 10.0) == 0.0
        assert room.apply_deadband(-5.0, 10.0) == 0.0
        # Input exactly at deadband edge
        assert room.apply_deadband(10.0, 10.0) == 0.0
        assert room.apply_deadband(-10.0, 10.0) == 0.0
    
    def test_tensor_input_broadcasting(self):
        """Edge case: tensor inputs with varying deadband ranges per-element."""
        room = DeadbandRoom(deadband_range=1.0)
        input_tensor = torch.tensor([-2.0, -0.5, 0.0, 0.5, 2.0])
        deadband_ranges = torch.tensor([1.0, 0.2, 1.0, 0.2, 1.0])
        output = room.apply_deadband(input_tensor, deadband_ranges)
        expected = torch.tensor([
            -1.0,   # -2.0 + 1.0 = -1.0 (outside deadband)
            0.0,    # -0.5 within ±0.2 deadband -> 0.0
            0.0,    # 0.0 within ±1.0 deadband -> 0.0
            0.0,    # 0.5 within ±0.2 deadband -> 0.0
            1.0     # 2.0 - 1.0 = 1.0 (outside deadband)
        ])
        assert torch.allclose(output, expected, atol=1e-7)
    
    def test_nan_and_inf_input(self):
        """Edge case: NaN and Inf inputs should propagate or raise? (Current: propagate)."""
        room = DeadbandRoom(deadband_range=1.0)
        # NaN input -> output NaN (deadband logic doesn't apply)
        assert torch.isnan(room.apply_deadband(float('nan'), 1.0))
        # Inf input -> output Inf (outside any finite deadband)
        assert room.apply_deadband(float('inf'), 1.0) == float('inf')
        assert room.apply_deadband(float('-inf'), 1.0) == float('-inf')
    
    def test_deadband_range_as_tensor_with_negative_check(self):
        """Edge case: deadband_range as tensor with any negative element raises ValueError."""
        room = DeadbandRoom(deadband_range=1.0)
        # Single negative element in tensor should raise
        with pytest.raises(ValueError, match="deadband_range must be >= 0"):
            room.apply_deadband(5.0, torch.tensor([1.0, -0.5, 2.0]))
        # All non-negative tensor should work
        valid_ranges = torch.tensor([0.0, 1.0, 2.0])
        output = room.apply_deadband(torch.tensor([0.5, 1.5, 3.0]), valid_ranges)
        expected = torch.tensor([0.0, 0.5, 1.0])  # [within 0.0, 1.5-1.0, 3.0-2.0]
        assert torch.allclose(output, expected, atol=1e-7)
