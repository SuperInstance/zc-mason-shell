### Test 3: Concurrent Agent Protocol Violation
**Scenario:** Multiple agents simultaneously attempt to skip from P0 directly to P2.
**Rationale:** Tests thread safety and protocol enforcement under concurrent load.
```python
def test_concurrent_protocol_violation(deadband_room):
    """Room should correctly handle multiple agents violating protocol simultaneously."""
    import threading
    
    violations_detected = []
    
    def attempt_violation(agent_id):
        try:
            # Skip P1, go directly to P2
            result = deadband_room.process_phase(agent_id, "P2", ["optimize_this"])
            if "violation" in result or "error" in result:
                violations_detected.append(agent_id)
        except Exception as e:
            violations_detected.append(f"{agent_id}_error")
    
    threads = []
    for i in range(10):
        agent_id = f"concurrent_agent_{i}"
        t = threading.Thread(target=attempt_violation, args=(agent_id,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    assert len(violations_detected) == 10  # All violations should be caught
```

```python
def test_tile_submission_malformed_metadata(deadband_room):
    """Room should reject tiles with invalid metadata while maintaining state."""
    agent_id = "test_agent_4"
    
    # Complete all phases first
    deadband_room.process_phase(agent_id, "P0", ["constraint1"])
    deadband_room.process_phase(agent_id, "P1", [{"min": 0, "max": 10}])
    deadband_room.process_phase(agent_id, "P2", ["optimize"])
    
    # Attempt to submit tile with problematic metadata
    malicious_tile = {
        "content": "valid work here",
        "metadata": {
            "agent_id": agent_id,
            "phase": "P2",
            "injection": "<script>alert('xss')</script>",  # Attempted injection
            "null_byte": "valid\x00truncated",  # Null byte injection
            "nested_depth": {"a": {"b": {"c": {"d": {"e": "too deep"}}}}}  # Deep nesting
        }
    }
    
    result = deadband_room.submit_tile(agent_id, malicious_tile)
    
    # Should either sanitize or reject, but not crash
    assert "rejected" in result or "sanitized" in result
    assert deadband_room.get_agent_phase(agent_id) == "P2"  # State preserved
```
