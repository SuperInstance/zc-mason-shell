## New Edge-Case Tests

### Test 1: Agent Submits Tile While in "dead" State
**Scenario:** An agent that has been marked `dead` (e.g., due to protocol violation) attempts to submit a new tile. According to the room logic, dead agents should be ignored.
```python
def test_dead_agent_cannot_submit_tile(room, agent):
    """P0: Dead agents cannot submit tiles."""
    room.register_agent(agent)
    room.agents[agent["id"]]["state"] = "dead"
    
    tile = {"type": "test", "content": "should be rejected"}
    result = room.submit_tile(agent["id"], tile)
    
    assert result["status"] == "rejected"
    assert "dead" in result["reason"].lower()
    assert len(room.tiles) == 0
```

### Test 3: Rapid-Fire Tile Submissions from Multiple Agents
**Scenario:** Simulate high concurrency where multiple agents submit tiles in the same cycle. Tests thread safety and queue handling.
```python
def test_concurrent_tile_submissions(room):
    """P2: Handle multiple agents submitting tiles simultaneously."""
    agents = [{"id": f"agent{i}", "state": "idle", "tiles": []} for i in range(5)]
    for agent in agents:
        room.register_agent(agent)
    
    # All agents submit a tile in the same cycle
    results = []
    for agent in agents:
        tile = {"type": "concurrent", "content": f"from {agent['id']}"}
        results.append(room.submit_tile(agent["id"], tile))
    
    # All should be accepted
    assert all(r["status"] == "accepted" for r in results)
    assert len(room.tiles) == 5
    # Verify tiles are in submission order (FIFO)
    assert [t["content"] for t in room.tiles] == [f"from agent{i}" for i in range(5)]
```

### Test 5: Room Capacity Enforcement with Mixed Valid/Invalid Submissions
**Scenario:** Room has a maximum tile capacity (e.g., 1000 tiles). When near capacity, valid tiles should be accepted until full, then rejected. Tests the room's cleanup/garbage collection logic.
```python
def test_room_capacity_edge_case(room, agent):
    """P1/P2: Room accepts tiles until at capacity, then rejects new ones."""
    room.register_agent(agent)
    room.max_tiles = 10  # Small limit for testing
    
    # Fill room to capacity
    for i in range(room.max_tiles):
        tile = {"type": "fill", "content": str(i)}
        result = room.submit_tile(agent["id"], tile)
        assert result["status"] == "accepted"
    
    assert len(room.tiles) == room.max_tiles
    
    # Next submission should be rejected
    tile = {"type": "overflow", "content": "should be rejected"}
    result = room.submit_tile(agent["id"], tile)
    assert result["status"] == "rejected"
    assert "capacity" in result["reason"].lower()
    
    # After processing a cycle (which may clear old tiles), new submissions might be accepted again
    # This tests the integration between capacity and cycle processing
    room.process_cycle()
    # Assuming process_cycle clears tiles from this cycle
    if len(room.tiles) < room.max_tiles:
        tile = {"type": "post_cycle", "content": "new"}
        result = room.submit_tile(agent["id"], tile)
        assert result["status"] == "accepted"
```
