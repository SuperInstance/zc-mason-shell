    @pytest.mark.asyncio
    async def test_concurrent_tile_submission(self):
        """Test P1: Multiple agents submitting tiles simultaneously.
        
        Verifies thread safety and proper tile ordering.
        """
        room = DeadbandRoom(room_id="edge_concurrent", max_tiles=100)
        
        # Simulate 10 agents submitting at the same time
        async def submit_tile(agent_id):
            tile = Tile(
                agent_id=agent_id,
                content=f"Test from {agent_id}",
                protocol_phase="P1",
                metadata={"timestamp": 123456}
            )
            return await room.submit_tile(tile)
        
        tasks = [submit_tile(f"agent_{i}") for i in range(10)]
        results = await asyncio.gather(*tasks)
        
        # All submissions should succeed
        assert all(r["status"] == "accepted" for r in results)
        
        # Tiles should be in submission order (FIFO)
        tiles = room.get_tiles()
        assert len(tiles) == 10
        assert tiles[0].agent_id == "agent_0"
        assert tiles[9].agent_id == "agent_9"
    
    @pytest.mark.asyncio

        """Test P0: Partial write during network partition.
        
        Simulates a scenario where tile submission succeeds locally
        but fails to persist to storage.

        room = DeadbandRoom(room_id="edge_partition", max_tiles=50)
        
        # Mock the storage layer to fail on every other write
        call_count = 0
        original_save = room._storage.save_tile
        
        async def failing_save(tile):
            nonlocal call_count
            call_count += 1
            if call_count % 2 == 0:
                raise IOError("Network partition: storage unreachable")
            return await original_save(tile)
        
        room._storage.save_tile = failing_save
        
        # Submit 5 tiles - some should fail
        results = []
        for i in range(5):
            tile = Tile(
                agent_id=f"agent_{i}",
                content=f"Tile {i}",
                protocol_phase="P1",
                metadata={"seq": i}
            )
            try:
                result = await room.submit_tile(tile)
                results.append(result)
            except IOError as e:
                results.append({"status": "failed", "error": str(e)})
        
        # Verify mixed results
        successes = [r for r in results if r.get("status") == "accepted"]
        failures = [r for r in results if r.get("status") == "failed"]
        
        assert len(successes) > 0, "Should have some successful writes"
        assert len(failures) > 0, "Should have some failed writes"
        
        # Room state should reflect only successful submissions
        assert len(room.get_tiles()) == len(successes)
