### Test 3: Concurrent Processing Simulation
**Scenario:** Simulate concurrent access to DeadbandRoom instance
**Purpose:** Test thread safety for potential multi-agent use
**Expected:** Room should handle concurrent calls without data corruption

```python
import threading

def test_concurrent_processing():
    """Simulate concurrent access to DeadbandRoom."""
    room = DeadbandRoom(threshold=0.15, deadband=0.07)
    
    results = []
    errors = []
    
    def worker(worker_id, observations):
        try:
            result = room.process_observations(observations)
            results.append((worker_id, result))
        except Exception as e:
            errors.append((worker_id, str(e)))
    
    # Create multiple observation sets
    threads = []
    for i in range(5):
        obs = [float(j) * 0.1 + i * 0.3 for j in range(5)]
        t = threading.Thread(target=worker, args=(i, obs))
        threads.append(t)
        t.start()
    
    # Wait for all threads
    for t in threads:
        t.join()
    
    # No errors should occur
    assert len(errors) == 0, f"Concurrent errors: {errors}"
    
    # All results should have valid structure
    for worker_id, result in results:
        assert isinstance(result, dict)
        assert 'p0_violations' in result
```

### Test 5: Protocol Phase Boundary Conditions
**Scenario:** Observations that sit exactly on threshold/deadband boundaries
**Purpose:** Test edge-case classification accuracy
**Expected:** Room should consistently classify boundary cases

```python
def test_boundary_conditions():
    """Test observations at exact threshold/deadband boundaries."""
    threshold = 0.5
    deadband = 0.2
    room = DeadbandRoom(threshold=threshold, deadband=deadband)
    
    # Boundary test cases
    test_cases = [
        # (observations, expected_primary_phase)
        ([0.49, 0.51], 'P0'),      # Crosses threshold boundary
        ([0.3, 0.31], 'P1'),       # Within deadband
        ([0.29, 0.31], 'P0'),      # Crosses deadband boundary
        ([0.5, 0.5], 'P1'),        # Exactly at threshold (edge case)
        ([0.2, 0.2], 'P1'),        # Exactly at deadband (edge case)
    ]
    
    for observations, expected_phase in test_cases:
        result = room.process_observations(observations)
        
        # Determine primary phase from result
        if result['p0_violations']:
            primary_phase = 'P0'
        elif result['p1_channels']:
            primary_phase = 'P1'
        else:
            primary_phase = 'P2'
        
        # Phase should match expectation
        assert primary_phase == expected_phase, \
            f"Observations {observations}: expected {expected_phase}, got {primary_phase}"
```
