### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (simulated)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results if isinstance(r, dict))
    
    # Verify no tile corruption
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == len(agent_ids)
```

### Test 2: Malformed Tile Data Handling
```python
def test_malformed_tile_validation(self):
    """Test room handling of invalid tile structures."""
    room = DeadbandRoom()
    
    test_cases = [
        # (tile_data, expected_error)
        (None, "Tile data cannot be None"),
        ({}, "Missing required field: agent_id"),
        ({"agent_id": "test"}, "Missing required field: phase"),
        ({"agent_id": "test", "phase": "P5"}, "Invalid phase: P5"),
        ({"agent_id": "test", "phase": "P1", "content": 123}, "Content must be string"),
        ({"agent_id": "test", "phase": "P1", "content": "x" * 10001}, "Content too long"),
    ]
    
    for tile_data, expected_error in test_cases:
        with self.assertRaises(ValueError) as context:
            room.submit_tile(tile_data)
        assert expected_error in str(context.exception)
```

### Test 3: Protocol Regression Prevention
```python
def test_protocol_regression_blocked(self):
    """Test that agents cannot regress to earlier phases."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    
    # Normal progression
    room.set_agent_phase(agent_id, "P0")
    room.set_agent_phase(agent_id, "P1")
    room.set_agent_phase(agent_id, "P2")
    
    # Attempt regression should fail
    with self.assertRaises(ValueError) as context:
        room.set_agent_phase(agent_id, "P1")
    assert "Cannot regress from P2 to P1" in str(context.exception)
    
    # Verify current phase unchanged
    assert room.get_agent_phase(agent_id) == "P2"
```

### Test 4: Resource Limit Enforcement
```python
def test_resource_limits_enforced(self):
    """Test room behavior when resource limits are reached."""
    room = DeadbandRoom(max_tiles=10, max_memory_mb=5)
    
    # Fill tile storage
    for i in range(10):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P1",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # 11th tile should be rejected
    with self.assertRaises(RuntimeError) as context:
        tile = {
            "agent_id": "agent_11",
            "phase": "P1",
            "content": "Excess tile",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    assert "Tile storage limit reached" in str(context.exception)
    
    # Simulate memory pressure
    large_content = "x" * (5 * 1024 * 1024)  # 5MB
    with self.assertRaises(MemoryError):
        room._simulate_memory_pressure(large_content)
```

```python
def test_partial_network_connectivity(self):
    """Test room behavior with simulated network partitions."""
    room = DeadbandRoom()
    
    # Create multiple agents
    agents = [f"agent_{i}" for i in range(3)]
    
    # Simulate network partition: agent_1 cannot communicate
    room._simulate_network_partition(["agent_1"])
    
    # Agent_0 and agent_2 should work normally
    for agent_id in ["agent_0", "agent_2"]:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"From {agent_id}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # Agent_1 should fail
    with self.assertRaises(ConnectionError):
        tile = {
            "agent_id": "agent_1",
            "phase": "P1",
            "content": "From partitioned agent",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # Restore connectivity
    room._restore_network_connectivity()
    
    # Agent_1 should now succeed
    tile = {
        "agent_id": "agent_1",
        "phase": "P1",
        "content": "After restore",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile)
    assert result["status"] == "accepted"
```
