### 1. Test Concurrent Tile Submission
```python
def test_concurrent_tile_submission(mock_room_config, mock_tile_db):
    """P0: Ensure room handles simultaneous tile adds without data corruption."""
    from threading import Thread
    room = DeadbandRoom(mock_room_config, mock_tile_db)
    tiles = []
    
    def add_tile(agent_id):
        tile = {"agent_id": agent_id, "phase": "P1", "content": f"test_{agent_id}"}
        room.add_tile(tile)
        tiles.append(agent_id)
    
    threads = [Thread(target=add_tile, args=(i,)) for i in range(10)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all tiles recorded
    assert len(room.tiles) == 10
    assert sorted([t["agent_id"] for t in room.tiles]) == list(range(10))
```

def test_p2_before_p1_violation(mock_room_config, mock_tile_db):
    """P0: Tile attempting P2 optimization without P1 channel should be rejected."""
    room = DeadbandRoom(mock_room_config, mock_tile_db)
    
    violating_tile = {
        "agent_id": "eager_agent",
        "P0": {},  # No negative space mapped
        "P1": {},  # No safe channel found (empty)
        "P2": {"optimization": "premature"}  # Attempting P2 anyway
    }
    
    # According to Deadband Protocol, this should fail validation
    is_valid, reason = room.validate_tile(violating_tile)
    
    assert is_valid is False
    assert "P1 channel not established" in reason or "protocol violation" in reason.lower()
