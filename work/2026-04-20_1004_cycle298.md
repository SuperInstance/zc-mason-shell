### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents should be able to submit tiles without race conditions."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3", "agent_4"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "priority": "P0",
            "content": f"Negative space map from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent tile submissions should all be accepted"
    
    # All tiles should be in room memory
    assert len(room.memory) == len(agent_ids), \
        f"Expected {len(agent_ids)} tiles, got {len(room.memory)}"
```

### Test 2: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Room should reject P0→P2 jumps but allow recovery to P1."""
    room = DeadbandRoom()
    agent_id = "violating_agent"
    
    # Submit P0 tile (valid)
    p0_tile = {
        "agent_id": agent_id,
        "priority": "P0",
        "content": "Mapping negative space",
        "timestamp": time.time()
    }
    p0_result = room.submit_tile(p0_tile)
    assert p0_result["status"] == "accepted", "P0 should be accepted"
    
    # Try to jump directly to P2 (should be rejected)
    p2_tile = {
        "agent_id": agent_id,
        "priority": "P2",
        "content": "Trying to optimize prematurely",
        "timestamp": time.time()
    }
    p2_result = room.submit_tile(p2_tile)
    assert p2_result["status"] == "rejected", "P0→P2 jump should be rejected"
    assert "P1" in p2_result.get("message", ""), "Should suggest P1"
    
    # Submit P1 (should be accepted for recovery)
    p1_tile = {
        "agent_id": agent_id,
        "priority": "P1",
        "content": "Finding safe channels after violation",
        "timestamp": time.time()
    }
    p1_result = room.submit_tile(p1_tile)
    assert p1_result["status"] == "accepted", "P1 recovery should be accepted"
```

### Test 3: Large Tile Payload Handling

def test_large_tile_payload(self):
    """Room should handle or reject excessively large tiles appropriately."""
    room = DeadbandRoom()
    agent_id = "data_heavy_agent"
    
    # Generate large content (simulating memory-intensive tile)
    large_content = "X" * (1024 * 1024)  # 1MB of data
    
    tile = {
        "agent_id": agent_id,
        "priority": "P0",
        "content": large_content,
        "timestamp": time.time()
    }
    
    result = room.submit_tile(tile)
    
    # Either accept with truncation or reject with clear message
    assert result["status"] in ["accepted", "rejected"], \
        "Should have clear status for large payload"
    
    if result["status"] == "accepted":
        # If accepted, content might be truncated
        stored_tile = room.get_agent_tiles(agent_id)[0]
        assert len(stored_tile["content"]) <= len(large_content), \
            "Stored content should not exceed original"
    else:
        # If rejected, should indicate size issue
        assert "size" in result.get("message", "").lower() or \
               "memory" in result.get("message", "").lower(), \
            "Rejection should mention size/memory constraint"
```

def test_partial_state_update_recovery(self):
    """Room should handle incomplete state updates gracefully."""
    room = DeadbandRoom()
    
    # Simulate partial update (missing required fields)
    partial_tiles = [
        {"agent_id": "agent_1"},  # missing priority and content
        {"priority": "P0", "content": "test"},  # missing agent_id
        {"agent_id": "agent_2", "priority": "P0"},  # missing content
        {"agent_id": "agent_3", "content": "test"},  # missing priority
    ]
    
    for tile in partial_tiles:
        tile["timestamp"] = time.time()  # Add timestamp for consistency
        result = room.submit_tile(tile)
        assert result["status"] == "rejected", \
            f"Partial tile should be rejected: {tile.keys()}"
        assert "missing" in result.get("message", "").lower() or \
               "required" in result.get("message", "").lower(), \
            "Should indicate missing required fields"
    
    # Valid tile should still work after partial failures
    valid_tile = {
        "agent_id": "recovery_agent",
        "priority": "P0",
        "content": "Valid after partial failures",
        "timestamp": time.time()
    }
    valid_result = room.submit_tile(valid_tile)
    assert valid_result["status"] == "accepted", \
        "Valid tiles should work despite previous partial failures"
```

### Test 5: Ghost Tile Integration
```python
def test_ghost_tile_influence(self):
    """Ghost tiles (from dead agents) should inform P0 decisions."""
    room = DeadbandRoom()
    
    # Simulate ghost tile submission (agent marked as inactive)
    ghost_tile = {
        "agent_id": "ghost_agent",
        "priority": "P0",
        "content": "FAILURE: Tried to optimize before mapping. System crashed.",
        "timestamp": time.time() - 3600,  # 1 hour ago
        "agent_status": "inactive"
    }
    
    # Room should accept ghost tiles
    ghost_result = room.submit_tile(ghost_tile)
    assert ghost_result["status"] == "accepted", "Ghost tiles should be accepted"
    
    # Live agent submitting P0 should show awareness of ghost tiles
    live_tile = {
        "agent_id": "live_agent",
        "priority": "P0",
        "content": "Mapping negative space with historical knowledge",
        "timestamp": time.time()
    }
    
    live_result = room.submit_tile(live_tile)
    assert live_result["status"] == "accepted", "Live agent tile should be accepted"
    
    # Room's P0 summary should reference ghost learnings
    p0_summary = room.get_priority_summary("P0")
    assert isinstance(p0_summary, dict), "Should return summary dict"
    
    # Ghost learnings should be incorporated
    if "learnings" in p0_summary:
        # Check if ghost content influenced P0 knowledge
        ghost_content_present = any(
            "ghost" in str(item).lower() or 
            "inactive" in str(item).lower() or
            "failure" in str(item).lower()
            for item in p0_summary.get("learnings", [])
        )
        assert ghost_content_present, \
            "P0 summary should incorporate ghost tile learnings"
```
