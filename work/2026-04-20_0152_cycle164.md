### Test 1: Concurrent Agent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    
    # Simulate 3 agents submitting tiles at same time
    tiles = [
        {"agent_id": "agent1", "priority": 0, "content": "P0: Avoid null pointer"},
        {"agent_id": "agent2", "priority": 1, "content": "P1: Use safe allocator"},
        {"agent_id": "agent3", "priority": 0, "content": "P0: Check bounds"}
    ]
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    
    # Tiles should be processed in priority order (P0 before P1)
    processed = room.get_processed_tiles()
    assert processed[0]["priority"] == 0
    assert processed[1]["priority"] == 0  
    assert processed[2]["priority"] == 1
```

### Test 2: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Test room recovery when agent skips to P2 without P1."""
    room = DeadbandRoom()
    
    # Agent tries to submit P2 tile without P1
    tile = {"agent_id": "rogue_agent", "priority": 2, "content": "P2: Optimize"}
    result = room.submit_tile(tile)
    
    # Should be rejected with guidance
    assert result["status"] == "rejected"
    assert "P1 required before P2" in result["message"]
    
    # Room should still accept valid P0/P1 tiles after rejection
    valid_tile = {"agent_id": "good_agent", "priority": 0, "content": "P0: Test"}
    valid_result = room.submit_tile(valid_tile)
    assert valid_result["status"] == "accepted"
```

### Test 3: Queue Capacity Limits
```python
def test_queue_capacity_limits(self):
    """Test behavior when tile queue reaches capacity."""
    room = DeadbandRoom(max_queue_size=5)
    
    # Fill queue with P0 tiles
    for i in range(5):
        tile = {"agent_id": f"agent{i}", "priority": 0, "content": f"P0 tile {i}"}
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # Next submission should be queued or rejected based on implementation
    overflow_tile = {"agent_id": "overflow", "priority": 0, "content": "P0 overflow"}
    result = room.submit_tile(overflow_tile)
    
    # Either queued with warning or rejected
    assert result["status"] in ["queued", "rejected"]
    if result["status"] == "queued":
        assert "queue_full" in result
```

### Test 4: Malformed Tile Handling
```python
def test_malformed_tile_handling(self):
    """Test room handling of invalid tile formats."""
    room = DeadbandRoom()
    
    test_cases = [
        # Missing required fields
        {"priority": 0},
        {"agent_id": "test", "content": "test"},
        # Invalid priority value
        {"agent_id": "test", "priority": 3, "content": "test"},
        {"agent_id": "test", "priority": -1, "content": "test"},
        # Invalid content type
        {"agent_id": "test", "priority": 0, "content": 123},
        # Empty content
        {"agent_id": "test", "priority": 0, "content": ""},
    ]
    
    for tile in test_cases:
        result = room.submit_tile(tile)
        assert result["status"] == "rejected"
        assert "validation" in result["message"].lower() or "invalid" in result["message"].lower()
```

### Test 5: State Persistence Across Cycles
```python
def test_state_persistence(self):
    """Test room maintains state across training cycles."""
    room = DeadbandRoom()
    
    # First training cycle
    tile1 = {"agent_id": "agent1", "priority": 0, "content": "P0: Initial"}
    room.submit_tile(tile1)
    room.process_tiles()
    
    # Simulate cycle end and new cycle start
    initial_state = room.get_state()
    room.new_training_cycle()
    
    # State should persist certain elements (e.g., learned P0 constraints)
    new_state = room.get_state()
    
    # P0 knowledge should persist
    assert "p0_constraints" in initial_state
    assert "p0_constraints" in new_state
    assert len(new_state["p0_constraints"]) >= len(initial_state["p0_constraints"])
    
    # Queue should be reset for new cycle
    assert len(new_state["queue"]) == 0
```
