```python
def test_deadband_p0_conflicting_constraints(self):
    """Test P0 phase with overlapping/conflicting constraints."""
    room = DeadbandRoom(
        constraints=[
            {"type": "rectangle", "x": 0, "y": 0, "width": 10, "height": 10},
            {"type": "rectangle", "x": 5, "y": 5, "width": 10, "height": 10},
            {"type": "circle", "center_x": 12, "center_y": 12, "radius": 3}
        ]
    )
    
    # P0 should identify all negative spaces
    p0_result = room.execute_phase(0, {})
    self.assertIn("negative_spaces", p0_result)
    self.assertEqual(len(p0_result["negative_spaces"]), 3)
    
    # Verify overlapping areas are properly handled
    self.assertTrue(p0_result["is_valid"])
```

### Test 3: P2 with Non-Linear Objective Function
```python
def test_deadband_p2_nonlinear_optimization(self):
    """Test P2 optimization with a custom non-linear objective."""
    room = DeadbandRoom(
        constraints=[
            {"type": "rectangle", "x": 20, "y": 20, "width": 10, "height": 10}
        ],
        search_bounds={"x": 0, "y": 0, "width": 50, "height": 50}
    )
    
    # Custom objective: minimize distance from point (40, 40)
    def custom_objective(x, y):
        return (x - 40) ** 2 + (y - 40) ** 2
    
    p0_result = room.execute_phase(0, {})
    p1_result = room.execute_phase(1, p0_result)
    
    # P2 with custom objective
    p2_input = {
        **p1_result,
        "objective_function": custom_objective,
        "optimization_type": "minimize"
    }
    p2_result = room.execute_phase(2, p2_input)
    
    self.assertIn("optimal_point", p2_result)
    self.assertAlmostEqual(p2_result["optimal_point"][0], 30, delta=0.1)
    self.assertAlmostEqual(p2_result["optimal_point"][1], 30, delta=0.1)
```

### Test 4: Large Constraint Set Performance
```python
def test_deadband_large_constraint_set(self):
    """Test performance with 1000+ constraints."""
    import time
    
    # Generate 1000 random constraints
    constraints = []
    for i in range(1000):
        constraints.append({
            "type": "rectangle",
            "x": i % 50 * 2,
            "y": i // 50 * 2,
            "width": 1,
            "height": 1
        })
    
    room = DeadbandRoom(constraints=constraints)
    
    start_time = time.time()
    p0_result = room.execute_phase(0, {})
    p1_result = room.execute_phase(1, p0_result)
    p2_result = room.execute_phase(2, p1_result)
    end_time = time.time()
    
    # Should complete within reasonable time
    self.assertLess(end_time - start_time, 5.0)  # 5 seconds max
    self.assertTrue(p2_result["is_valid"])
```

### Test 5: Floating-Point Precision Edge Cases
```python
def test_deadband_floating_point_precision(self):
    """Test handling of floating-point constraints near boundaries."""
    room = DeadbandRoom(
        constraints=[
            {"type": "rectangle", "x": 0.1, "y": 0.1, "width": 0.8, "height": 0.8}
        ],
        search_bounds={"x": 0.0, "y": 0.0, "width": 1.0, "height": 1.0}
    )
    
    p0_result = room.execute_phase(0, {})
    p1_result = room.execute_phase(1, p0_result)
    
    # P1 should find safe channels with floating-point precision
    self.assertGreater(len(p1_result["safe_channels"]), 0)
    
    # Check that channels don't intersect with constraints
    for channel in p1_result["safe_channels"]:
        # Channel should not intersect the 0.1-0.9 rectangle
        self.assertTrue(
            channel["x"] + channel["width"] <= 0.1 or channel["x"] >= 0.9 or
            channel["y"] + channel["height"] <= 0.1 or channel["y"] >= 0.9
        )
```
