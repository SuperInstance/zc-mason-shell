```python
def test_concurrent_phase_conflicts(self):
    """Test multiple agents attempting conflicting phase transitions simultaneously."""
    room = DeadbandRoom()
    
    # Simulate two agents trying to move to different phases
    agent1_state = {"current_phase": "P0", "attempted_phase": "P2"}
    agent2_state = {"current_phase": "P0", "attempted_phase": "P1"}
    
    # Both agents should be prevented from skipping P1 (P0 violation)
    result1 = room.process_phase_transition(agent1_state)
    result2 = room.process_phase_transition(agent2_state)
    
    assert result1["allowed"] == False
    assert "cannot skip P1" in result1["reason"].lower()
    assert result2["allowed"] == True  # P0→P1 is valid
```

### Test 4: Mixed Protocol State Interactions
```python
def test_mixed_protocol_state_interactions(self):
    """Test agents at different protocol phases interacting in same room."""
    room = DeadbandRoom()
    
    # Register agents at different phases
    room.register_agent("agent_p0", {"current_phase": "P0", "knowledge": []})
    room.register_agent("agent_p1", {"current_phase": "P1", "channels": ["safe1"]})
    room.register_agent("agent_p2", {"current_phase": "P2", "optimization": "in_progress"})
    
    # Simulate knowledge sharing (P0 agent learning from P2)
    sharing_result = room.share_knowledge(
        from_agent="agent_p2",
        to_agent="agent_p0",
        knowledge_type="p0_negative_space"
    )
    
    # P0 agent should accept P0 knowledge but not P2 optimization details
    assert sharing_result["accepted"] == True
    assert "p0" in sharing_result["knowledge_type"].lower()
    
    # Attempt invalid sharing (P0 trying to share optimization with P2)
    invalid_result = room.share_knowledge(
        from_agent="agent_p0",
        to_agent="agent_p2",
        knowledge_type="p2_optimization"
    )
    assert invalid_result["accepted"] == False
```

### Test 5: Recovery from Corrupted Persistence State
```python
def test_recovery_from_corrupted_state(self):
    """Test DeadbandRoom recovery when persistence layer returns corrupted data."""
    room = DeadbandRoom()
    
    # Simulate corrupted persistence data
    corrupted_data = {
        "agents": {
            "agent1": {"current_phase": "INVALID_PHASE", "data": b"corrupted\x00bytes"}
        },
        "room_state": "corrupted",
        "timestamp": "invalid_date"
    }
    
    recovery_result = room.recover_from_persistence(corrupted_data)
    
    # Should either recover gracefully or initialize fresh state
    assert recovery_result["status"] in ["recovered", "fresh_start"]
    
    if recovery_result["status"] == "recovered":
        assert "recovered_agents" in recovery_result
        assert "lost_agents" in recovery_result
    else:  # fresh_start
        assert "reason" in recovery_result
        assert "corrupt" in recovery_result["reason"].lower() or "invalid" in recovery_result["reason"].lower()
    
    # Room should be in a valid state after recovery attempt
    assert room.validate_state()["valid"] == True
```
