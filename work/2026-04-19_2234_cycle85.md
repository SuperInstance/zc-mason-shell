```python
"""
DeadbandRoom - A PLATO room preset implementing the Deadband Protocol.

This room enforces the P0/P1/P2 priority system for agent interactions.
Agents submit tiles (work outputs) which are validated against protocol rules.
"""

from typing import Dict, List, Optional, Any
from dataclasses import dataclass
from enum import Enum

class TileStatus(Enum):
    PENDING = "pending"
    ACCEPTED = "accepted"
    REJECTED = "rejected"
    GHOST = "ghost"  # From dead agents

@dataclass
class Tile:
    agent_id: str
    cycle: int
    content: str
    phase: int  # 0, 1, 2, 3, 4
    status: TileStatus = TileStatus.PENDING
    validation_errors: List[str] = None

class DeadbandRoom:
    def __init__(self, room_id: str, max_tiles_per_cycle: int = 100):
        self.room_id = room_id
        self.max_tiles = max_tiles_per_cycle
        self.tiles: List[Tile] = []
        self.agent_stats: Dict[str, Dict] = {}
    
    def submit_tile(self, agent_id: str, cycle: int, content: str, phase: int) -> Tile:
        """Submit a tile for validation."""
        if len(self.tiles) >= self.max_tiles:
            raise ValueError(f"Room at capacity: {self.max_tiles} tiles max")
        
        tile = Tile(
            agent_id=agent_id,
            cycle=cycle,
            content=content,
            phase=phase,
            status=TileStatus.PENDING,
            validation_errors=[]
        )
        self.tiles.append(tile)
        self._validate_tile(tile)
        return tile
    
    def _validate_tile(self, tile: Tile) -> None:
        """Apply Deadband Protocol validation rules."""
        errors = []
        
        # P0: Negative space checks
        if tile.phase < 0 or tile.phase > 4:
            errors.append(f"Invalid phase {tile.phase}: must be 0-4")
        if not tile.content or len(tile.content.strip()) == 0:
            errors.append("Empty content")
        if "guaranteed" in tile.content.lower() or "always" in tile.content.lower() or "never" in tile.content.lower():
            errors.append("Absolute claim detected (avoid 'always', 'never', 'guaranteed')")
        
        # P1: Safe channel checks
        if tile.phase == 0 and "P0" not in tile.content:
            errors.append("Phase 0 tile missing P0 marker")
        if tile.phase == 1 and ("P0" not in tile.content or "P1" not in tile.content):
            errors.append("Phase 1 tile missing P0/P1 markers")
        if tile.phase == 2 and ("P0" not in tile.content or "P1" not in tile.content or "P2" not in tile.content):
            errors.append("Phase 2 tile missing P0/P1/P2 markers")
        
        # Update tile
        if errors:
            tile.status = TileStatus.REJECTED
            tile.validation_errors = errors
        else:
            tile.status = TileStatus.ACCEPTED
    
    def get_agent_stats(self, agent_id: str) -> Dict[str, Any]:
        """Return statistics for an agent."""
        accepted = [t for t in self.tiles if t.agent_id == agent_id and t.status == TileStatus.ACCEPTED]
        rejected = [t for t in self.tiles if t.agent_id == agent_id and t.status == TileStatus.REJECTED]
        return {
            "accepted_count": len(accepted),
            "rejected_count": len(rejected),
            "acceptance_rate": len(accepted) / max(len(accepted) + len(rejected), 1),
            "last_cycle": max([t.cycle for t in accepted + rejected], default=0)
        }
    
    def get_ghost_tiles(self) -> List[Tile]:
        """Return tiles from agents no longer active (simulated)."""
        return [t for t in self.tiles if t.status == TileStatus.GHOST]
```

from plato_torch.presets.deadband_room import DeadbandRoom, TileStatus

def test_submit_tile_basic():
    room = DeadbandRoom("test_room", max_tiles_per_cycle=10)
    tile = room.submit_tile("agent1", 1, "Test content", 2)
    assert tile.status == TileStatus.ACCEPTED
    assert len(room.tiles) == 1

def test_submit_tile_capacity():
    room = DeadbandRoom("test_room", max_tiles_per_cycle=2)
    room.submit_tile("agent1", 1, "Content 1", 2)
    room.submit_tile("agent2", 1, "Content 2", 2)
    with pytest.raises(ValueError, match="Room at capacity"):
        room.submit_tile("agent3", 1, "Content 3", 2)

def test_validation_empty_content():
    room = DeadbandRoom("test_room")

    assert tile.status == TileStatus.REJECTED
