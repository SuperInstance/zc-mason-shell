```python
def test_concurrent_agent_tile_submission():
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"concurrent_agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles_submitted = []
    for agent_id in agent_ids:
        tile = {
            "phase": "P0",
            "content": f"P0 analysis from {agent_id}",
            "agent_id": agent_id,
            "timestamp": datetime.utcnow().isoformat()
        }
        result = room.submit_tile(agent_id, tile)
        tiles_submitted.append((agent_id, result))
    
    # All should be accepted if room handles concurrency
    accepted_count = sum(1 for _, result in tiles_submitted if result["accepted"])
    
    # Either all are accepted (proper locking) or some fail (expected in some designs)
    assert accepted_count == len(agent_ids) or accepted_count > 0
    
    # Room state should be consistent
    room_state = room.get_room_state()
    assert len(room_state["submitted_tiles"]) == accepted_count
```

### Test 4: Room State Recovery After Crash
```python
def test_room_state_recovery():
    """Test that room can recover state after simulated crash/restart."""
    room = DeadbandRoom()
    agent_id = "recovery_agent"
    
    # Submit some tiles
    for phase in ["P0", "P1", "P2"]:
        tile = {
            "phase": phase,
            "content": f"{phase} work completed",
            "agent_id": agent_id
        }
        room.submit_tile(agent_id, tile)
    
    # Simulate crash by capturing state then creating new room instance
    saved_state = room.save_state()
    
    # Create new room instance with saved state
    recovered_room = DeadbandRoom()
    recovered_room.load_state(saved_state)
    
    # Verify recovery
    agent_status = recovered_room.get_agent_status(agent_id)
    assert agent_status["current_phase"] == "P2"
    
    room_state = recovered_room.get_room_state()
    assert len(room_state["submitted_tiles"]) == 3
    
    # Should be able to continue from recovered state
    new_tile = {
        "phase": "P2",
        "content": "Additional optimization after recovery",
        "agent_id": agent_id
    }
    result = recovered_room.submit_tile(agent_id, new_tile)
    assert result["accepted"]
```
