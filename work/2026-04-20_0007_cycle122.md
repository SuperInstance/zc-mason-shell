```python
def test_deadband_protocol_priority_violation():

    agent_id = "test_agent_001"

    violating_tile = {

        "cycle": 1,
        "phase": 4,
        "protocol_step": "P2",  # Skipping P0 and P1

    assert "protocol violation" in result["message"].lower()
    assert room.get_agent_progress(agent_id)["violations"] == 1
```

### Test 3: Concurrent Tile Submission Race Conditions
```python
def test_concurrent_tile_submissions():
    """Test room handling of simultaneous tile submissions from same agent."""
    room = DeadbandRoom()
    agent_id = "test_agent_003"
    
    import threading
    
    results = []
    errors = []
    
    def submit_concurrent_tile(cycle_num):
        tile = {
            "agent_id": agent_id,
            "cycle": cycle_num,
            "phase": 1,
            "protocol_step": "P0",
            "content": f"Concurrent test cycle {cycle_num}"
        }
        try:
            result = room.submit_tile(agent_id, tile)
            results.append((cycle_num, result["status"]))
        except Exception as e:
            errors.append((cycle_num, str(e)))
    
    # Submit 5 tiles concurrently
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_concurrent_tile, args=(100 + i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Room should handle concurrency without data corruption
    assert len(errors) == 0, f"Errors during concurrent submission: {errors}"
    
    # Check that all submissions were processed
    accepted = [r[1] for r in results if r[1] == "accepted"]
    assert len(accepted) > 0, "No tiles accepted during concurrent submission"
    
    # Verify room state consistency
    agent_state = room.get_agent_progress(agent_id)
    assert agent_state["last_cycle"] >= 104  # Should have processed highest cycle
```

```python
def test_malformed_sentiment_recovery():
    """Test room recovery from tiles with invalid sentiment structures."""
    room = DeadbandRoom()
    agent_id = "test_agent_004"
    
    test_cases = [
        # Missing sentiment entirely
        {
            "agent_id": agent_id,
            "cycle": 1,
            "phase": 1,
            "protocol_step": "P0",
            "content": "Tile without sentiment"
        },
        # Sentiment with wrong data types
        {
            "agent_id": agent_id,
            "cycle": 2,
            "phase": 1,
            "protocol_step": "P0",
            "content": "Tile with string sentiment",
            "sentiment": "not a dict"
        },
        # Sentiment with out-of-range values
        {
            "agent_id": agent_id,
            "cycle": 3,
            "phase": 1,
            "protocol_step": "P0",
            "content": "Tile with extreme values",
            "sentiment": {"confidence": 5.0, "valence": -2.0}  # Outside 0-1 range
        },
        # Nested malformed sentiment
        {
            "agent_id": agent_id,
            "cycle": 4,
            "phase": 1,
            "protocol_step": "P0",
            "content": "Tile with nested issues",
            "sentiment": {"confidence": {"nested": "wrong"}, "valence": [0.5]}
        }
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(agent_id, tile)
        
        # Room should handle gracefully, not crash
        assert result["status"] in ["accepted", "rejected"]
        
        if result["status"] == "accepted":
            # If accepted, sentiment should be normalized or defaulted
            stored_tile = room.get_tile_history(agent_id)[-1]
            if "sentiment" in stored_tile:
                sent = stored_tile["sentiment"]
                assert isinstance(sent, dict)
                if "confidence" in sent:
                    assert 0.0 <= sent["confidence"] <= 1.0
                if "valence" in sent:
                    assert -1.0 <= sent["valence"] <= 1.0
```
