#[test]
fn test_deadband_protocol_violation_skipping_to_p2() {
    let mut room = DeadbandRoom::new();
    let agent_id = "test_agent_1".to_string();
    
    // Attempt to submit P2 tile without P0/P1 context
    let tile = TrainingTile {
        agent_id: agent_id.clone(),
        content: "P2 optimization".to_string(),
        priority: 2,
        timestamp: Utc::now(),
    };
    
    let result = room.submit_tile(tile);
    assert!(result.is_err());
    assert_eq!(
        result.unwrap_err(),
        RoomError::ProtocolViolation("Cannot submit P2 without P0/P1 context".to_string())
    );
    
    // Verify agent is blocked from further submissions
    assert!(room.is_agent_blocked(&agent_id));
}

### Test 2: State Persistence Across Resets
```rust
#[test]
fn test_state_persistence_after_room_reset() {
    let mut room = DeadbandRoom::new();
    let agent_id = "test_agent_2".to_string();
    
    // Submit valid P0 tile
    let p0_tile = TrainingTile {
        agent_id: agent_id.clone(),
        content: "P0 negative space".to_string(),
        priority: 0,
        timestamp: Utc::now(),
    };
    
    room.submit_tile(p0_tile).unwrap();
    
    // Reset room (simulating server restart)
    room.reset();
    
    // Agent should still be recognized but context cleared
    assert!(!room.is_agent_blocked(&agent_id));
    assert_eq!(room.get_agent_priority_context(&agent_id), None);
    
    // Agent can start fresh protocol
    let new_p0_tile = TrainingTile {
        agent_id: agent_id.clone(),
        content: "Fresh P0".to_string(),
        priority: 0,
        timestamp: Utc::now(),
    };
    
    assert!(room.submit_tile(new_p0_tile).is_ok());
}
```

### Test 4: Concurrent Tile Submission Race Condition
```rust
#[test]
fn test_concurrent_tile_submission_race_condition() {
    let room = Arc::new(Mutex::new(DeadbandRoom::new()));
    let agent_id = "test_agent_4".to_string();
    
    let mut handles = vec![];
    
    // Simulate 5 concurrent submissions from same agent
    for i in 0..5 {
        let room_clone = Arc::clone(&room);
        let agent_id_clone = agent_id.clone();
        
        let handle = thread::spawn(move || {
            let tile = TrainingTile {
                agent_id: agent_id_clone,
                content: format!("Tile {}", i),
                priority: 0,
                timestamp: Utc::now(),
            };
            
            let mut room_lock = room_clone.lock().unwrap();
            room_lock.submit_tile(tile)
        });
        
        handles.push(handle);
    }
    
    // Collect results
    let mut success_count = 0;
    let mut failure_count = 0;
    
    for handle in handles {
        match handle.join().unwrap() {
            Ok(_) => success_count += 1,
            Err(e) => {
                assert!(matches!(e, RoomError::ConcurrentModification(_)));
                failure_count += 1;
            }
        }
    }
    
    // Only one submission should succeed
    assert_eq!(success_count, 1);
    assert_eq!(failure_count, 4);
}
```

### Test 5: Malformed Tile Data Corruption
```rust
#[test]
fn test_malformed_tile_data_handling() {
    let mut room = DeadbandRoom::new();
    
    // Test with empty agent_id
    let empty_id_tile = TrainingTile {
        agent_id: "".to_string(), // Empty string
        content: "Test content".to_string(),
        priority: 0,
        timestamp: Utc::now(),
    };
    
    let result = room.submit_tile(empty_id_tile);
    assert!(result.is_err());
    assert_eq!(
        result.unwrap_err(),
        RoomError::InvalidTile("Agent ID cannot be empty".to_string())
    );
    
    // Test with extremely large content
    let large_content = "x".repeat(1_000_001); // 1MB+ content
    let large_tile = TrainingTile {
        agent_id: "test_agent_5".to_string(),
        content: large_content,
        priority: 0,
        timestamp: Utc::now(),
    };
    
    let result = room.submit_tile(large_tile);
    assert!(result.is_err());
    assert_eq!(
        result.unwrap_err(),
        RoomError::InvalidTile("Tile content exceeds 1MB limit".to_string())
    );
    
    // Test with future timestamp
    let future_tile = TrainingTile {
        agent_id: "test_agent_5".to_string(),
        content: "Future tile".to_string(),
        priority: 0,
        timestamp: Utc::now() + Duration::days(1), // Tomorrow
    };
    
    let result = room.submit_tile(future_tile);
    assert!(result.is_err());
    assert_eq!(
        result.unwrap_err(),
        RoomError::InvalidTile("Tile timestamp cannot be in the future".to_string())
    );
}
```
