"""
Edge-case tests for DeadbandRoom.
These complement the core test suite in test_deadband_room.py.
"""

import pytest
from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""
    
    def setup_method(self):
        """Fresh room for each test."""
        self.room = DeadbandRoom(room_id="test_edge_room", config={"max_cycles": 100})
        # Simulate room has processed some initial state
        self.room.current_cycle = 10
        self.room.current_phase = 2  # P2 optimization phase
    
    def test_protocol_regression_violation(self):
        """P0->P1->P2->P1 should be rejected (can't go back to P1 after P2)."""
        # First, establish valid P2 context
        valid_p2_tile = {
            "agent_id": "agent_test",
            "cycle": 10,
            "phase": 2,
            "content": {"action": "optimize", "result": "speed+5%"}
        }
        self.room.process_tile(valid_p2_tile)
        
        # Now attempt regression to P1
        invalid_p1_tile = {
            "agent_id": "agent_test", 
            "cycle": 11,
            "phase": 1,
            "content": {"action": "find_channel", "channel": "safe_zone"}
        }
        
        with pytest.raises(ValueError, match="Protocol regression"):
            self.room.process_tile(invalid_p1_tile)
    
    def test_cycle_skip_detection(self):
        """Agent skipping cycles >1 should trigger warning."""
        # Current cycle is 10
        skip_tile = {
            "agent_id": "agent_skip",
            "cycle": 15,  # Skips 11-14
            "phase": 2,
            "content": {"action": "optimize", "result": "jumped"}
        }
        
        # Should process but log warning
        with patch('plato_torch.rooms.deadband_room.logger') as mock_logger:
            result = self.room.process_tile(skip_tile)
            assert result['status'] == 'accepted'
            mock_logger.warning.assert_called_once()
            assert "cycle skip" in mock_logger.warning.call_args[0][0].lower()
    
    def test_phase_mismatch_with_room_state(self):
        """Tile phase doesn't match room's current phase expectation."""
        # Room is in phase 2, but tile claims phase 0
        mismatch_tile = {
            "agent_id": "agent_mismatch",
            "cycle": 10,
            "phase": 0,  # Room expects phase 2
            "content": {"action": "map_negative", "hazards": ["rock"]}
        }
        
        # Room should adjust or reject based on config
        with patch.object(self.room, 'config', {'strict_phase_sync': True}):
            with pytest.raises(ValueError, match="Phase mismatch"):
                self.room.process_tile(mismatch_tile)
        
        # With strict_phase_sync=False, it should adapt
        with patch.object(self.room, 'config', {'strict_phase_sync': False}):
            result = self.room.process_tile(mismatch_tile)
            assert result['status'] == 'adapted'
            assert 'phase_adjusted' in result
    
    def test_concurrent_tile_collision_handling(self):
        """Two agents submitting for same cycle - last write wins with conflict marker."""
        tile_a = {
            "agent_id": "agent_a",
            "cycle": 10,
            "phase": 2,
            "content": {"action": "optimize", "value": 100}
        }
        
        tile_b = {
            "agent_id": "agent_b", 
            "cycle": 10,  # Same cycle
            "phase": 2,
            "content": {"action": "optimize", "value": 200}
        }
        
        # Process both
        result_a = self.room.process_tile(tile_a)
        result_b = self.room.process_tile(tile_b)
        
        # Last should win
        assert result_b['status'] == 'accepted'
        assert 'conflict_with' in result_b
        assert result_b['conflict_with'] == 'agent_a'
        
        # Verify storage shows B's content
        stored = self.room.get_tile(cycle=10)
        assert stored['agent_id'] == 'agent_b'
        assert stored['content']['value'] == 200
    
    def test_malformed_content_structure(self):
        """Content is valid JSON but missing required action/result fields."""
        # Content has wrong structure for phase 2
        malformed_tile = {
            "agent_id": "agent_malformed",
            "cycle": 10,
            "phase": 2,
            "content": {
                "description": "Optimized something",  # Missing 'action' and 'result'
                "timestamp": "2026-04-20"
            }
        }
        
        # Should be rejected with specific field error
        with pytest.raises(ValueError) as exc_info:
            self.room.process_tile(malformed_tile)
        
        assert "missing required field" in str(exc_info.value).lower()
        assert "action" in str(exc_info.value).lower()
