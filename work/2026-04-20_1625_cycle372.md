### Test 1: Tile Capacity Boundary
```python
def test_add_tile_at_capacity(self):
    """Test adding a tile when room is at max_tiles capacity."""
    room = DeadbandRoom(max_tiles=2)
    tile1 = {"id": "t1", "content": "test1", "phase": 1}
    tile2 = {"id": "t2", "content": "test2", "phase": 2}
    tile3 = {"id": "t3", "content": "test3", "phase": 3}
    
    # Fill to capacity
    self.assertTrue(room.add_tile(tile1))
    self.assertTrue(room.add_tile(tile2))
    
    # Attempt to exceed capacity
    with self.assertLogs(level='WARNING') as log:
        result = room.add_tile(tile3)
        self.assertFalse(result)
        self.assertIn("Room at capacity", log.output[0])
    
    # Verify only first two tiles stored
    self.assertEqual(len(room.tiles), 2)
    self.assertEqual(room.tiles[0]["id"], "t1")
    self.assertEqual(room.tiles[1]["id"], "t2")
```

```python
@patch('transformers.AutoModelForCausalLM.from_pretrained')

    self.assertIsNone(room.ensign)
```

### Test 4: Concurrent Tile Addition Race Condition
```python
def test_concurrent_tile_addition(self):
    """Simulate concurrent tile additions to test thread safety."""
    import threading
    
    room = DeadbandRoom(max_tiles=100)
    tiles_added = []
    lock = threading.Lock()
    
    def add_tile_thread(tile_id):
        tile = {"id": f"t{tile_id}", "content": f"content{tile_id}", "phase": 1}
        result = room.add_tile(tile)
        with lock:
            tiles_added.append((tile_id, result))
    
    # Create 10 threads adding tiles simultaneously
    threads = []
    for i in range(10):
        thread = threading.Thread(target=add_tile_thread, args=(i,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all tiles were added successfully
    self.assertEqual(len(room.tiles), 10)
    
    # Check for duplicate IDs (should not happen with proper locking)
    tile_ids = [tile["id"] for tile in room.tiles]
    self.assertEqual(len(set(tile_ids)), 10)
    
    # Verify all threads reported success
    successful_adds = [tid for tid, success in tiles_added if success]
    self.assertEqual(len(successful_adds), 10)
```

```python
        room.export_ensign("")
    self.assertIn("Adapter path cannot be empty", str(context.exception))
    
    # Test with non-string path

    mock_save_pretrained.assert_not_called()
```
