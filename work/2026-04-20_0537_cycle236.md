## Edge-Case Tests Written

Based on reading the actual code, here are 5 new edge-case tests for `DeadbandRoom`:

### 1. Test P0 Violation with Empty Negative Space
**Scenario:** Agent submits a tile claiming P0 (negative space) but provides an empty `negative_space` list.
**Expected:** Room should reject the tile because P0 requires concrete hazards.
**Rationale:** Prevents agents from skipping P0 by submitting empty hazards.

```python
def test_p0_violation_empty_negative_space(self):
    """P0 tile with empty negative_space should be rejected."""
    agent_id = "test_agent_1"
    tile = {
        "phase": "P0",
        "negative_space": [],  # Empty list violates P0 requirement
        "timestamp": time.time(),
        "agent_id": agent_id
    }
    result = self.room.submit_tile(agent_id, tile)
    self.assertFalse(result["accepted"])
    self.assertIn("P0 requires non-empty negative_space", result["message"])
```

### 2. Test P1 Channel Overlap Detection
**Scenario:** Agent submits P1 tile with channels that overlap with previously mapped negative space.
**Expected:** Room should reject channels that intersect known hazards.
**Rationale:** Ensures safe channels don't contain known P0 violations.

```python
def test_p1_channel_overlaps_negative_space(self):
    """P1 channels overlapping with P0 negative_space should be rejected."""
    agent_id = "test_agent_2"
    
    # First submit valid P0
    p0_tile = {
        "phase": "P0",
        "negative_space": [{"x": 0, "y": 0, "radius": 5}],
        "timestamp": time.time(),
        "agent_id": agent_id
    }
    self.room.submit_tile(agent_id, p0_tile)
    
    # Try P1 with overlapping channel
    p1_tile = {
        "phase": "P1",
        "channels": [{"start": {"x": 3, "y": 3}, "end": {"x": 7, "y": 7}}],  # Overlaps hazard at (0,0) radius 5
        "timestamp": time.time(),
        "agent_id": agent_id
    }
    result = self.room.submit_tile(agent_id, p1_tile)
    self.assertFalse(result["accepted"])
    self.assertIn("channel overlaps with known hazard", result["message"].lower())
```

### 3. Test P2 Without Established P1
**Scenario:** Agent tries to submit P2 optimization tile without having established P1 channels first.
**Expected:** Room should reject P2 when no valid P1 channels exist for that agent.
**Rationale:** Enforces strict P0→P1→P2 protocol sequence.

```python
def test_p2_without_p1_channels(self):
    """P2 tile submitted without established P1 channels should be rejected."""
    agent_id = "test_agent_3"
    
    # Submit only P0 (skip P1)
    p0_tile = {
        "phase": "P0",
        "negative_space": [{"x": 10, "y": 10, "radius": 2}],
        "timestamp": time.time(),
        "agent_id": agent_id
    }
    self.room.submit_tile(agent_id, p0_tile)
    
    # Try to submit P2 directly
    p2_tile = {
        "phase": "P2",
        "optimization": {"path": [{"x": 0, "y": 0}, {"x": 5, "y": 5}]},
        "timestamp": time.time(),
        "agent_id": agent_id
    }
    result = self.room.submit_tile(agent_id, p2_tile)
    self.assertFalse(result["accepted"])
    self.assertIn("P1 channels not established", result["message"])
```

### 4. Test Malformed Tile Structure
**Scenario:** Agent submits tile with missing required fields or wrong data types.
**Expected:** Room should reject malformed tiles with specific error messages.
**Rationale:** Robust input validation prevents room corruption.

```python
def test_malformed_tile_structure(self):
    """Tiles with missing required fields or wrong types should be rejected."""
    agent_id = "test_agent_4"
    
    # Missing 'phase' field
    tile1 = {
        "negative_space": [{"x": 0, "y": 0}],
        "timestamp": time.time(),

        "phase": "P0",

        "timestamp": time.time(),
        "agent_id": agent_id
    }
    result2 = self.room.submit_tile(agent_id, tile2)
    self.assertFalse(result2["accepted"])
    self.assertIn("negative_space must be a list", result2["message"].lower())
```

### 5. Test Concurrent Agent State Isolation
**Scenario:** Multiple agents submitting tiles; one agent's P0 shouldn't affect another agent's P1 if channels don't overlap.
**Expected:** Agents maintain independent protocol state; valid non-overlapping channels accepted.
**Rationale:** Supports parallel agent work without unnecessary blocking.

```python
def test_concurrent_agent_state_isolation(self):
    """Agents should have independent protocol state."""
    agent_a = "agent_alpha"
    agent_b = "agent_beta"
    
    # Agent A submits P0
    p0_a = {
        "phase": "P0",
        "negative_space": [{"x": 0, "y": 0, "radius": 3}],
        "timestamp": time.time(),
        "agent_id": agent_a
    }
    self.room.submit_tile(agent_a, p0_a)
    
    # Agent B submits P1 that doesn't overlap with A's negative space
    p1_b = {
        "phase": "P1",
        "channels": [{"start": {"x": 10, "y": 10}, "end": {"x": 20, "y": 20}}],  # Far from A's hazard
        "timestamp": time.time(),
        "agent_id": agent_b
    }
    result = self.room.submit_tile(agent_b, p1_b)
    self.assertTrue(result["accepted"], f"Agent B's valid P1 should be accepted: {result.get('message')}")
    
    # Verify agents have separate state
    self.assertNotEqual(
        self.room.get_agent_state(agent_a)["current_phase"],
        self.room.get_agent_state(agent_b)["current_phase"]
    )
```
