### Test 1: Consecutive P0 Violations Should Not Crash Room
```python
def test_consecutive_p0_violations_preserve_state():
    """Room should handle multiple P0 violations without state corruption."""
    room = DeadbandRoom(room_id="edge-test-1")
    context = {"agent_id": "test-agent"}
    
    # First P0 violation
    action1 = {"type": "move", "direction": "into_rock"}
    result1 = room.process_action(action1, context)
    assert result1["status"] == "blocked"
    assert result1["phase"] == "P0"
    
    # Second P0 violation (different action, same phase)
    action2 = {"type": "build", "material": "unstable"}
    result2 = room.process_action(action2, context)
    assert result2["status"] == "blocked"
    assert result2["phase"] == "P0"
    
    # Room state should still be functional
    assert room.agent_violations["test-agent"] == 2
    # Valid action should still work
    action3 = {"type": "scan", "area": "safe_zone"}
    result3 = room.process_action(action3, context)
    assert result3["status"] == "processed"
```

### Test 3: P1 Channel Exhaustion Scenario
```python
def test_p1_channel_exhaustion_fallback():
    """When all known safe channels are exhausted, room should provide fallback guidance."""
    room = DeadbandRoom(room_id="edge-test-3")
    context = {"agent_id": "test-agent"}
    
    # Simulate exhausting all known safe channels
    for i in range(10):
        action = {"type": "explore", "channel": f"channel_{i}", "exhaust": True}
        room.process_action(action, context)
    
    # Try an action that needs a new channel
    action = {"type": "navigate", "destination": "unknown"}
    result = room.process_action(action, context)
    
    # Should either: 
    # 1) Find a new channel (processed in P1/P2)
    # 2) Provide guidance to wait/scan (blocked with guidance)
    assert result["status"] in ["processed", "blocked"]
    if result["status"] == "blocked":
        assert "channel" in result.get("guidance", "").lower() or "scan" in result.get("guidance", "").lower()
```

### Test 5: P2 Optimization With Conflicting Constraints
```python
def test_p2_conflicting_constraints():
    """P2 optimizer should handle actions with multiple conflicting optimization goals."""
    room = DeadbandRoom(room_id="edge-test-5")
    context = {"agent_id": "test-agent"}
    
    # Action that passes P0 and P1 but has conflicting P2 optimization parameters
    action = {
        "type": "route",
        "channel": "safe_channel_1",
        "optimize_for": ["speed", "energy", "safety"],  # Conflicting priorities
        "constraints": {
            "max_time": 10,
            "min_safety": 0.9,
            "max_energy": 5
        }
    }
    
    result = room.process_action(action, context)
    
    # Should reach P2 (passed P0 and P1)
    if result["phase"] == "P2":
        # P2 should handle the conflict, not crash
        assert result["status"] in ["processed", "optimized", "blocked"]
        if result["status"] == "blocked":
            assert "conflict" in result.get("reason", "").lower() or "constraint" in result.get("reason", "").lower()
    else:
        # If blocked earlier, should have clear reason
        assert "reason" in result
```
