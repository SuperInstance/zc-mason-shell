### Test 2: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission():
    """P1: Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom(room_id="edge-test-2")
    num_agents = 10
    
    def submit_tile_parallel(agent_id):
        tile = {
            "phase": "P0",
            "negative_space": f"test_{agent_id}",
            "timestamp": time.time()
        }
        return room.submit_tile(agent_id, tile)
    
    # Use ThreadPoolExecutor to simulate concurrent submissions
    with concurrent.futures.ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(submit_tile_parallel, f"agent-{i}") 
                  for i in range(num_agents)]
        results = [f.result() for f in futures]
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    # Tile count should match number of agents
    assert room.get_tile_count() == num_agents
    # No duplicate agent entries in state
    assert len(room._agent_states) == num_agents
```

### Test 4: Room Capacity Enforcement
```python
def test_room_capacity_limits():
    """P0: Exceeding MAX_CONCURRENT_AGENTS should reject new agents."""
    room = DeadbandRoom(room_id="edge-test-4")
    
    # Fill room to capacity
    for i in range(room.MAX_CONCURRENT_AGENTS):
        room.submit_tile(f"agent-{i}", {
            "phase": "P0",
            "negative_space": f"test_{i}",
            "timestamp": time.time()
        })
    
    # Next agent should be rejected
    with pytest.raises(RoomCapacityError) as exc_info:
        room.submit_tile("excess-agent", {
            "phase": "P0",
            "negative_space": "should_fail",
            "timestamp": time.time()
        })
    
    assert "capacity exceeded" in str(exc_info.value)
    assert room.get_active_agent_count() == room.MAX_CONCURRENT_AGENTS
    
    # After one agent leaves, new agent should be accepted
    room._agent_states.pop("agent-0")
    result = room.submit_tile("replacement-agent", {
        "phase": "P0",
        "negative_space": "should_succeed",
        "timestamp": time.time()
    })
    assert result["status"] == "accepted"
```

### Test 5: Network Partition Simulation

```python
def test_partial_tile_submission_failure():
    """P1: Simulate network partition where tile submission partially fails."""
    room = DeadbandRoom(room_id="edge-test-5")
    agent_id = "test-agent-5"
    
    # Mock network failure during tile persistence
    original_persist = room._persist_tile
    
    def failing_persist(tile):
        if tile["content"] == "fail_me":
            raise IOError("Network partition simulated")
        return original_persist(tile)
    
    room._persist_tile = failing_persist
    
    # First tile should succeed
    success_result = room.submit_tile(agent_id, {
        "phase": "P0",
        "content": "success",
        "timestamp": time.time()
    })
    assert success_result["status"] == "accepted"
    
    # Second tile should fail due to simulated network issue
    with pytest.raises(IOError) as exc_info:
        room.submit_tile(agent_id, {
            "phase": "P0",
            "content": "fail_me",
            "timestamp": time.time()
        })
    assert "Network partition" in str(exc_info.value)
    
    # Room should maintain consistency despite failure
    assert room.get_tile_count() == 1  # Only first tile persisted
    assert room.get_agent_state(agent_id)["last_activity"] is not None
    
    # Restore original method
    room._persist_tile = original_persist
```
