### Test 4: Concurrent Tile Submission Simulation

def test_concurrent_tile_submission_ordering():
    """Test tile submission order preservation under simulated concurrency."""
    room = DeadbandRoom(room_id="test_concurrent")
    
    # Create tiles with slightly different timestamps
    tiles = []
    for i in range(5):
        tile = {
            "agent_id": f"agent_{i}",
            "timestamp": f"2026-04-20T05:00:{i:02d}Z",
            "content": {
                "phase": "P1",
                "work_output": f"Channel analysis {i}",
                "safe_channels": [f"channel_{i}"],
                "metrics": {"confidence": 0.7 + i*0.05}
            },
            "signature": f"sig_{i}"
        }
        tiles.append(tile)
    
    # Submit in random order
    import random
    random_order = random.sample(tiles, len(tiles))
    
    for tile in random_order:
        room.submit_tile(tile)
    
    # Retrieve tiles - should be in timestamp order
    stored_tiles = room.get_recent_tiles(limit=10)
    timestamps = [t["timestamp"] for t in stored_tiles]
    
    # Verify chronological order
    assert timestamps == sorted(timestamps)
    # Verify all tiles preserved
    assert len(stored_tiles) == 5
```

def test_protocol_phase_skipping_detection():
    """Test detection and rejection of P2 attempts without P0/P1."""
    room = DeadbandRoom(room_id="test_phase_skip")
    
    # Agent trying to jump directly to P2
    mock_agent = Mock()
    mock_agent.execute_protocol_step.side_effect = [
        {
            "phase": "P2",  # Skipping P0 and P1!
            "output": {"optimized_path": "direct_optimization"},
            "confidence": 0.95
        }
    ]
    
    result = room.execute_protocol_for_agent(mock_agent)
    
    # Should be rejected and forced to P0
    assert result["current_phase"] == "P0"
    assert "Protocol violation" in result["log"]
    assert "P0 required before P2" in result["log"]
    
    # Room should record violation
    stats = room.get_statistics()
    assert stats["protocol_violations"] > 0
    assert "phase_skip" in stats["violation_types"]
```
