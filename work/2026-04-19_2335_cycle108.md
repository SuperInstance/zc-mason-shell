        """P0: When all actions are mapped as negative space.

        message about no safe channels, not crash with index error.
        """
        room = DeadbandRoom(
            action_dim=3,
            state_dim=2,

            p1_margin=0.05
        )

        q_values = np.array([0.2, 0.2, 0.2])  # All > 0.1 threshold

            room.select_action(state, q_values)
        
        assert "no safe channels" in str(exc_info.value).lower()
    
    def test_tie_breaking_multiple_p1_channels(self):
        """P1: Multiple equally valid safe channels.
