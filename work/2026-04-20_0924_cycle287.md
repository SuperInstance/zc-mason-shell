### Test 1: Protocol Reset During P2 Optimization
**Scenario:** What happens when the protocol is reset (e.g., due to external intervention) while in P2 optimization phase?
**Edge Case:** P2 operations might have side effects or partial state that needs cleanup.
```python
def test_protocol_reset_during_p2_optimization(self):
    """Test that resetting protocol during P2 optimization cleans up state properly."""
    room = DeadbandRoom()
    # Progress through P0 and P1
    room.process_input("identify_obstacles:rock1,rock2")
    room.process_input("find_safe_channels:channel_a,channel_b")
    room.process_input("optimize_path:channel_a")
    
    # Verify we're in P2
    assert room.current_phase == "P2"
    assert room.current_channel == "channel_a"
    
    # Reset protocol
    room.reset_protocol()
    
    # Verify clean state
    assert room.current_phase == "P0"
    assert room.current_channel is None
    assert room.identified_obstacles == []
    assert room.safe_channels == []
    assert not room.optimization_in_progress
```

### Test 2: Concurrent Input Processing
**Scenario:** Multiple inputs arrive in rapid succession before previous processing completes.
**Edge Case:** Race conditions in state updates.
```python
def test_concurrent_input_handling(self):
    """Test that concurrent inputs are processed sequentially without state corruption."""
    room = DeadbandRoom()
    
    # Simulate rapid input sequence
    inputs = [
        "identify_obstacles:rock1",
        "identify_obstacles:rock2,rock3",  # Overwrites previous?
        "find_safe_channels:channel_x",
        "optimize_path:channel_x"
    ]
    
    for input_str in inputs:
        room.process_input(input_str)
    
    # Verify final state is consistent
    assert "rock2" in room.identified_obstacles
    assert "rock3" in room.identified_obstacles
    assert "rock1" not in room.identified_obstacles  # Was overwritten
    assert room.current_phase == "P2"
```

### Test 5: Protocol Violation Recovery
**Scenario:** What happens when P2 is attempted before P1, but then corrected?
**Edge Case:** Protocol should enforce order but allow recovery.
```python
def test_protocol_violation_and_recovery(self):
    """Test that violating protocol order can be recovered from gracefully."""
    room = DeadbandRoom()
    
    # Attempt P2 before P1 (violation)
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.process_input("optimize_path:channel_x")
    assert "P1" in str(exc_info.value)  # Error mentions missing P1
    
    # Verify state didn't partially update
    assert room.current_phase == "P0"
    assert room.current_channel is None
    
    # Now do correct sequence
    room.process_input("identify_obstacles:rock1")
    room.process_input("find_safe_channels:channel_x")
    room.process_input("optimize_path:channel_x")
    
    # Should succeed after correct sequence
    assert room.current_phase == "P2"
    assert room.current_channel == "channel_x"
```
