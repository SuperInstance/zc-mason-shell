```python
def test_resource_exhaustion_p2(deadband_room):
    """P2 optimization should handle computational limits gracefully."""
    # Setup P0 and P1
    tile_p0 = {
        "phase": 0,
        "negative_space": ["memory > 1GB", "cpu_usage > 90%"],
        "timestamp": time.time(),
        "agent_id": "test_agent_3"
    }
    deadband_room.submit_tile(tile_p0)
    
    tile_p1 = {
        "phase": 1,
        "safe_channels": [{"constraints": ["memory <= 512MB", "cpu_usage <= 70%"]}],
        "timestamp": time.time(),
        "agent_id": "test_agent_3"
    }
    deadband_room.submit_tile(tile_p1)
    
    # P2 with resource-intensive optimization
    tile_p2 = {
        "phase": 2,
        "optimization": {
            "method": "exhaustive_search",
            "parameters": {"max_iterations": 10**6},
            "objective": "minimize_latency"

        "timestamp": time.time(),
        "agent_id": "test_agent_3"

    # Should timeout or throttle, not crash
    result = deadband_room.submit_tile(tile_p2, timeout=0.1)
    assert result["status"] in ["timeout", "throttled", "partial"]
    if result["status"] == "partial":
        assert "best_path" in result
```

### Test 4: Concurrent Tile Collisions
```python
def test_concurrent_tile_collisions(deadband_room):
    """Multiple agents submitting conflicting P1 channels."""
    import threading
    
    results = []
    
    def submit_agent_tile(agent_id, channel):
        tile_p0 = {
            "phase": 0,
            "negative_space": ["conflict_zone"],
            "timestamp": time.time(),
            "agent_id": agent_id
        }
        deadband_room.submit_tile(tile_p0)
        
        tile_p1 = {
            "phase": 1,
            "safe_channels": [channel],
            "timestamp": time.time(),
            "agent_id": agent_id
        }
        results.append(deadband_room.submit_tile(tile_p1))
    
    # Start concurrent submissions
    threads = []
    channels = [
        {"constraints": ["x <= 50"]},
        {"constraints": ["x >= 50"]},
        {"constraints": ["x == 50"]}  # Direct conflict
    ]
    
    for i, channel in enumerate(channels):
        t = threading.Thread(
            target=submit_agent_tile,
            args=(f"agent_{i}", channel)
        )
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # At least one should be rejected due to conflict
    statuses = [r["status"] for r in results]
    assert "rejected" in statuses or "conflict" in statuses
```

```python
def test_malformed_tile_metadata(deadband_room):
    """Invalid timestamps, agent IDs, or missing fields."""
    test_cases = [
        # Missing timestamp
        {
            "phase": 0,
            "negative_space": ["test"],
            "agent_id": "agent_x"
        },
        # Future timestamp
        {
            "phase": 0,
            "negative_space": ["test"],
            "timestamp": time.time() + 3600,
            "agent_id": "agent_x"
        },
        # Invalid agent ID format
        {
            "phase": 0,
            "negative_space": ["test"],
            "timestamp": time.time(),
            "agent_id": ""  # Empty
        },
        # Missing phase
        {
            "negative_space": ["test"],
            "timestamp": time.time(),
            "agent_id": "agent_x"
        }
    ]
    
    for tile in test_cases:
        with pytest.raises((ValidationError, ValueError)) as exc_info:
            deadband_room.submit_tile(tile)
        
        # Ensure helpful error message
        error_msg = str(exc_info.value).lower()
        assert any(keyword in error_msg for keyword in 
                  ["invalid", "missing", "required", "timestamp", "agent", "phase"])
```
