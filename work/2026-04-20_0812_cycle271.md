# plato-torch/tests/test_deadband_room_edge_cases.py
"""
Edge-case tests for DeadbandRoom.
These tests cover scenarios not in the main test suite.
"""

import pytest
import asyncio
from unittest.mock import patch, AsyncMock
import json
from plato_torch.rooms.deadband_room import DeadbandRoom


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""
    
    def setup_method(self):
        """Fresh room for each test."""
        self.room = DeadbandRoom(room_id="edge-test-room")
    
    # Test 1: Protocol Priority Violation
    def test_priority_skip_rejection(self):
        """Test that tiles skipping P0→P1→P2 sequence are rejected."""
        # Agent tries to submit P2 optimization without P0/P1
        invalid_tile = {
            "agent_id": "impatient-agent",
            "timestamp": "2026-04-20T08:15:00Z",
            "content": {
                "priority": "P2",  # Skipping P0 and P1
                "action": "optimize_path",
                "data": {"path": "fastest"}
            }
        }
        
        result = self.room.process_tile(invalid_tile)
        assert result["status"] == "rejected"
        assert "priority violation" in result["reason"].lower()
        assert len(self.room.negative_space) == 0  # No P0 recorded
    
    # Test 2: Deeply Nested Malformed Tile
    def test_deeply_nested_malformed_tile(self):
        """Test handling of complex malformed tile structures."""
        # Tile with circular reference attempt (JSON serialization test)
        malformed_tile = {
            "agent_id": "buggy-agent",
            "timestamp": "2026-04-20T08:16:00Z",
            "content": {
                "priority": "P0",
                "action": "map_negative",
                "data": {
                    "nested": {
                        "deeper": {
                            "self_ref": None  # Will try to create circular ref
                        }
                    }
                }
            }
        }
        
        # Create circular reference
        malformed_tile["content"]["data"]["nested"]["deeper"]["self_ref"] = \
            malformed_tile["content"]["data"]["nested"]
        
        result = self.room.process_tile(malformed_tile)
        # Should either reject or handle gracefully
        assert result["status"] in ["rejected", "error"]
        if "error" in result.get("reason", ""):
            assert "serialization" in result["reason"].lower() or \
                   "circular" in result["reason"].lower()
    
    # Test 3: Concurrent Tile Processing
    @pytest.mark.asyncio
    async def test_concurrent_tile_race_condition(self):
        """Test simultaneous tile submissions from multiple agents."""
        async def submit_tile(agent_id, priority):
            tile = {
                "agent_id": agent_id,
                "timestamp": f"2026-04-20T08:17:{agent_id[-2:]}Z",
                "content": {
                    "priority": priority,
                    "action": f"{priority}_action",
                    "data": {"source": agent_id}
                }
            }
            # Simulate network delay
            await asyncio.sleep(0.01 * int(agent_id[-1]))
            return self.room.process_tile(tile)
        
        # Submit 10 tiles concurrently
        tasks = []
        for i in range(10):
            priority = "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2"
            task = submit_tile(f"agent-{i:02d}", priority)
            tasks.append(task)
        
        results = await asyncio.gather(*tasks)
        
        # All should be processed without corruption
        success_count = sum(1 for r in results if r["status"] == "processed")
        assert success_count == 10
        
        # State should be consistent
        assert len(self.room.interaction_tiles) == 10
        # Verify no duplicate processing
        agent_ids = [tile["agent_id"] for tile in self.room.interaction_tiles]
        assert len(set(agent_ids)) == 10
    
    # Test 4: State Persistence Across Sessions
    def test_state_persistence_and_reset(self):
        """Test that room state can be saved/loaded and reset properly."""
        # Add some state
        self.room.process_tile({
            "agent_id": "persistence-test",
            "timestamp": "2026-04-20T08:18:00Z",
            "content": {
                "priority": "P0",
                "action": "map_negative",
                "data": {"obstacle": "rock"}
            }
        })
        
        # Save state
        saved_state = self.room.get_state()
        assert "negative_space" in saved_state
        assert len(saved_state["negative_space"]) > 0
        
        # Create new room and load state
        new_room = DeadbandRoom(room_id="reloaded-room")
        new_room.load_state(saved_state)
        
        # Verify state transferred
        assert len(new_room.negative_space) == len(self.room.negative_space)
        assert new_room.negative_space[0]["data"]["obstacle"] == "rock"
        
        # Test reset
        new_room.reset()
        assert len(new_room.negative_space) == 0
        assert len(new_room.interaction_tiles) == 0
    
    # Test 5: Tile Flood and Memory Limits
    def test_tile_flood_memory_handling(self):
        """Test behavior when flooded with tiles (memory/performance)."""
        # Room should handle many tiles without crashing
        tile_count = 1000
        processed_count = 0
        
        for i in range(tile_count):
            tile = {
                "agent_id": f"flood-agent-{i:04d}",
                "timestamp": f"2026-04-20T08:19:{i%60:02d}Z",
                "content": {
                    "priority": "P0" if i % 10 == 0 else "P1",
                    "action": "test_action",
                    "data": {"index": i}
                }
            }
            
            result = self.room.process_tile(tile)
            if result["status"] == "processed":
                processed_count += 1
        
        # Should process most tiles (some might be rejected for being duplicates)
        assert processed_count >= tile_count * 0.95  # 95% success rate
        
        # Memory should be managed (optional: check if cleanup happened)
        # This would depend on room implementation - just verify no crash
        assert len(self.room.interaction_tiles) <= tile_count
        
        # Later tiles should still be processable
        final_tile = {
            "agent_id": "final-test",
            "timestamp": "2026-04-20T08:20:00Z",
            "content": {
                "priority": "P0",
                "action": "final_check",
                "data": {"test": "complete"}
            }
        }
        
        final_result = self.room.process_tile(final_tile)
        assert final_result["status"] == "processed"
