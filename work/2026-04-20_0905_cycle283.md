### Test 3: Concurrent Agent State Corruption
```python
def test_concurrent_agent_state_isolation(self):
    """Test that multiple agents' states don't interfere with each other."""
    room = DeadbandRoom()
    agent_a = "agent_alpha"
    agent_b = "agent_beta"
    
    # Agent A completes P0
    tile_a_p0 = {
        "agent_id": agent_a,
        "phase": "P0",
        "content": "A's negative space",
        "timestamp": time.time()
    }
    room.submit_tile(tile_a_p0)
    
    # Agent B is still at P0
    tile_b_p0 = {
        "agent_id": agent_b,
        "phase": "P0",
        "content": "B's negative space",
        "timestamp": time.time()
    }
    room.submit_tile(tile_b_p0)
    
    # Agent A tries P1 (should work)
    tile_a_p1 = {
        "agent_id": agent_a,
        "phase": "P1",
        "content": "A's safe channels",
        "timestamp": time.time()
    }
    result_a = room.submit_tile(tile_a_p1)
    
    # Agent B tries P1 (should work - independent progress)
    tile_b_p1 = {
        "agent_id": agent_b,
        "phase": "P1",
        "content": "B's safe channels",
        "timestamp": time.time()
    }
    result_b = room.submit_tile(tile_b_p1)
    
    assert result_a["status"] == "accepted"
    assert result_b["status"] == "accepted"
    assert room.get_agent_state(agent_a) == "P1_complete"
    assert room.get_agent_state(agent_b) == "P1_complete"
```

### Test 4: Malformed Tile Data Recovery
```python
def test_malformed_tile_recovery(self):
    """Test room recovery after receiving malformed tile data."""
    room = DeadbandRoom()
    agent_id = "test_agent_malformed"
    
    # Submit valid P0 tile
    valid_tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Valid mapping",
        "timestamp": time.time()
    }
    room.submit_tile(valid_tile)
    
    # Submit malformed tile (missing required field)
    malformed_tile = {
        "agent_id": agent_id,
        # Missing 'phase' field
        "content": "Malformed content",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(malformed_tile)
    assert result["status"] == "rejected"
    assert "missing required" in result["message"].lower()
    
    # Room should still accept next valid tile
    next_valid_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Recovery test",
        "timestamp": time.time()
    }
    
    recovery_result = room.submit_tile(next_valid_tile)
    assert recovery_result["status"] == "accepted"
    assert room.get_agent_state(agent_id) == "P1_complete"
```

### Test 5: Corrupted Room State Recovery
```python
def test_corrupted_state_recovery(self):
    """Test room recovery when internal state becomes corrupted."""
    room = DeadbandRoom()
    
    # Simulate corrupted state by directly modifying internal structures
    # (This would normally be protected, but tests edge case)
    room.agent_states["corrupted_agent"] = "invalid_state_xyz"
    room.tile_history.append({"corrupted": "data"})
    
    # Room should detect and repair on next operation
    agent_id = "recovery_agent"
    tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Test after corruption",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(tile)
    
    # Room should either:
    # 1. Reject due to corruption detection, OR
    # 2. Recover and accept
    assert result["status"] in ["accepted", "rejected"]
    
    if result["status"] == "rejected":
        assert "corruption" in result["message"].lower() or "invalid" in result["message"].lower()
    else:
        # If accepted, verify clean state
        assert room.get_agent_state(agent_id) == "P0_complete"
        assert "corrupted_agent" not in room.agent_states
        assert all("corrupted" not in str(tile) for tile in room.tile_history[-10:])
```
