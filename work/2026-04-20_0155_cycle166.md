```python
def test_max_tile_boundary_handling():
    """P1: Test room behavior at maximum tile capacity."""
    room = DeadbandRoom()
    
    # Fill room to 99% capacity
    for i in range(room.MAX_TILES - 10):
        room.submit_tile({
            "phase": "P0",
            "content": f"Negative space mapping {i}",
            "agent_id": "filler_agent"
        })
    
    # Submit batch that would exceed limit
    overflow_results = []
    for j in range(15):
        result = room.submit_tile({
            "phase": "P1",
            "content": f"Channel finding attempt {j}",
            "agent_id": "boundary_agent"
        })
        overflow_results.append(result["accepted"])
    
    # Verify graceful degradation, not crash
    assert room.get_tile_count() <= room.MAX_TILES
    assert any(not accepted for accepted in overflow_results)
```

### Test 3: Concurrent Submission Race Condition
```python
def test_concurrent_tile_race():
    """P0: Simulate multiple agents submitting simultaneously."""
    room = DeadbandRoom()
    
    def submit_concurrent(agent_id, count):
        results = []
        for i in range(count):
            tile = {
                "phase": "P0",
                "content": f"Concurrent test {i}",
                "agent_id": agent_id,
                "timestamp": time.time() + random.random() * 0.01
            }
            results.append(room.submit_tile(tile))
        return results
    
    # Simulate 3 agents submitting concurrently
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = [
            executor.submit(submit_concurrent, f"agent_{i}", 5)
            for i in range(3)
        ]
        all_results = [f.result() for f in futures]
    
    # Verify no data corruption and all tiles accounted for
    total_tiles = sum(len(r) for r in all_results)
    assert room.get_tile_count() == total_tiles
    assert room.validate_state_integrity()
```

### Test 5: Malformed Negative Space Mapping
```python
def test_invalid_p0_mapping():
    """P0: Test handling of logically invalid negative space definitions."""
    room = DeadbandRoom()
    
    invalid_p0_cases = [
        # Self-contradictory mapping
        {
            "phase": "P0",
            "content": "Avoid all channels except safe ones",
            "agent_id": "confused_agent"
        },
        # Empty negative space
        {
            "phase": "P0",
            "content": "",
            "agent_id": "empty_agent"
        },
        # Circular reference
        {
            "phase": "P0",
            "content": "Don't do what you shouldn't avoid",
            "agent_id": "circular_agent"
        },
        # Overly broad prohibition
        {
            "phase": "P0",
            "content": "Avoid everything",
            "agent_id": "paranoid_agent"
        }
    ]
    
    rejection_reasons = []
    for tile in invalid_p0_cases:
        result = room.submit_tile(tile)
        if not result["accepted"]:
            rejection_reasons.append(result.get("reason", ""))
    
    # Should reject at least some invalid P0 mappings
    assert len(rejection_reasons) >= 2
    assert any("contradict" in r.lower() or "invalid" in r.lower() 
               for r in rejection_reasons)
```
