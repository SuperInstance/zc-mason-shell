"""
Edge-case tests for DeadbandRoom preset in plato-torch.
Mason (test builder) - Cycle 71.
Focus on P0 (negative space), boundary conditions, and error paths.
"""

import pytest
import torch
from unittest.mock import Mock, patch
import numpy as np

# Assuming DeadbandRoom is in this module
from plato_torch.rooms.deadband_room import DeadbandRoom


class TestDeadbandRoomEdgeCases:
    """Test suite for edge cases in DeadbandRoom."""

    def test_p0_negative_space_with_empty_tile(self):
        """P0: Map negative space - empty tile should not crash."""
        room = DeadbandRoom()
        # Simulate receiving an empty tile (None or empty dict)
        tile = None
        result = room.process_tile(tile)
        # Expect graceful handling: return None or default
        assert result is None or isinstance(result, dict)
        # Log should indicate P0 violation
        assert "P0" in room.get_logs() or room.phase == "P0"

    def test_p1_safe_channel_boundary_max_capacity(self):
        """P1: Find safe channels - at maximum tile capacity."""
        room = DeadbandRoom(max_tiles=100)
        # Fill room to capacity
        for i in range(100):
            room.add_tile({"id": i, "data": f"tile_{i}"})
        # Attempt to add one more tile
        with pytest.raises(ValueError) as exc_info:
            room.add_tile({"id": 101, "data": "overflow"})
        assert "capacity" in str(exc_info.value).lower()
        # Ensure room still in P1 (safe state)
        assert room.phase == "P1"

    def test_p2_optimize_with_malformed_gradient(self):
        """P2: Optimize within channels - malformed gradient tensor."""
        room = DeadbandRoom()
        room.set_phase("P2")
        # Simulate a gradient with NaN or inf values
        malformed_grad = torch.tensor([1.0, float('nan'), 3.0])
        # Room should handle without crashing
        result = room.apply_gradient(malformed_grad)
        # Expect either a corrected tensor or a fallback
        assert not torch.isnan(result).any()
        # Log should indicate P0 detection
        logs = room.get_logs()
        assert any("invalid gradient" in log.lower() for log in logs)

    def test_phase_transition_invalid_sequence(self):
        """Invalid phase transition (e.g., P0 to P2 skipping P1)."""
        room = DeadbandRoom()
        room.set_phase("P0")
        # Attempt to skip P1 (violates strict priority)
        with pytest.raises(RuntimeError) as exc_info:
            room.set_phase("P2")
        assert "P1" in str(exc_info.value)
        # Room should remain in P0
        assert room.phase == "P0"

    def test_ghost_tile_handling_corrupted_data(self):
        """Ghost tile with corrupted or unexpected data structure."""
        room = DeadbandRoom()
        # Simulate a ghost tile with missing required fields
        ghost_tile = {"type": "ghost", "lesson": "P0: avoid null"}
        # Room should extract P0 knowledge without crashing
        p0_rules = room.extract_p0_knowledge(ghost_tile)
        assert isinstance(p0_rules, list)
        # Even if corrupted, should return empty list or default
        assert len(p0_rules) >= 0
        # Log should note corruption
        logs = room.get_logs()
        assert any("corrupt" in log.lower() or "ghost" in log.lower() for log in logs)

    def test_mirror_play_with_mismatched_screens(self):
        """Mirror Play: two PLATO systems with mismatched screen states."""
        room = DeadbandRoom()
        # Simulate two screens with different dimensions
        screen_a = torch.randn(3, 64, 64)
        screen_b = torch.randn(3, 32, 32)
        # Room should handle size mismatch (e.g., resize or error)
        try:
            result = room.mirror_play(screen_a, screen_b)
            # If no error, result should be a tensor
            assert isinstance(result, torch.Tensor)
        except ValueError as e:
            # Expected error for mismatch
            assert "size" in str(e).lower() or "dimension" in str(e).lower()
        # Ensure room state is still valid
        assert room.phase in ["P0", "P1", "P2"]

    def test_tile_submission_with_high_latency(self):
        """Tile submission simulated under high network latency."""
        room = DeadbandRoom()
        # Mock a delayed tile submission
        delayed_tile = {"id": "delay", "timestamp": 0}
        with patch('time.time', return_value=10.0):  # 10 sec delay
            room.submit_tile(delayed_tile)
        # Room should mark tile as stale or queue it
        assert room.get_pending_tiles() <= 1
        # Log should note latency
        logs = room.get_logs()
        assert any("delay" in log.lower() or "latency" in log.lower() for log in logs)

    def test_ensign_export_with_partial_training(self):
        """Export ensign (LoRA) when room has insufficient tiles."""
        room = DeadbandRoom(min_tiles_for_export=50)
        # Add only 10 tiles
        for i in range(10):
            room.add_tile({"id": i})
        # Attempt export should raise error or return None
        with pytest.raises(RuntimeError) as exc_info:
            room.export_ensign()
        assert "insufficient" in str(exc_info.value).lower()
        # Room should remain in P1 (safe channel)
        assert room.phase == "P1"
