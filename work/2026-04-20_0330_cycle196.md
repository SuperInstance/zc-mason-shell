# plato-torch/tests/test_deadband_room_edge.py
"""
Edge-case tests for DeadbandRoom beyond core protocol validation.
These tests cover failure modes, boundary conditions, and unexpected behaviors.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.tiles import Tile, TileValidationError


class TestDeadbandRoomEdgeCases:
    """Edge case tests for DeadbandRoom."""
    
    @pytest.mark.asyncio
    async def test_concurrent_tile_submission(self):
        """Test multiple agents submitting tiles simultaneously.
        
        Edge: Race conditions in room state updates.
        Expected: All tiles processed, no state corruption.
        """
        room = DeadbandRoom(preset_name="deadband")
        validator = Mock()
        validator.validate_tile = AsyncMock(return_value=True)
        room.tile_validator = validator
        
        # Create multiple tiles from different agents
        tiles = [
            Tile(agent_id=f"agent_{i}", content=f"test_{i}", phase="P0")
            for i in range(5)
        ]
        
        # Submit all tiles concurrently
        tasks = [room.submit_tile(tile) for tile in tiles]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Verify all succeeded
        assert all(r is True for r in results), "Concurrent submission failed"
        assert len(room.room_state.submitted_tiles) == 5, "Tile count mismatch"
        
        # Verify no duplicate processing
        agent_ids = [t.agent_id for t in room.room_state.submitted_tiles]
        assert len(set(agent_ids)) == 5, "Duplicate or missing agent tiles"
    
    @pytest.mark.asyncio
    async def test_malformed_tile_payload(self):
        """Test submission of tiles with invalid structure.
        
        Edge: Missing required fields, wrong data types.
        Expected: Validation error, tile rejected, room state unchanged.
        """
        room = DeadbandRoom(preset_name="deadband")
        
        # Tile missing required 'content' field
        invalid_tile = Tile(agent_id="test_agent", phase="P0")
        invalid_tile.content = None  # Explicitly malformed
        
        with pytest.raises(TileValidationError) as exc_info:
            await room.submit_tile(invalid_tile)
        
        assert "content" in str(exc_info.value).lower(), \
            "Should complain about missing content"
        assert len(room.room_state.submitted_tiles) == 0, \
            "Malformed tile should not be stored"
    
    @pytest.mark.asyncio
    async def test_protocol_regression_attempt(self):
        """Test agent trying to move backward in protocol (P2→P1).
        
        Edge: Protocol allows forward movement only, but code might not check regression.
        Expected: Regression rejected, phase remains at P2.
        """
        room = DeadbandRoom(preset_name="deadband")
        
        # Manually advance room to P2 through valid sequence
        room.room_state.current_phase = "P0"
        await room.record_negative_space("test_negative")
        assert room.room_state.current_phase == "P1"
        
        await room.discover_safe_channel("test_channel")
        assert room.room_state.current_phase == "P2"
        
        # Attempt regression tile
        regression_tile = Tile(
            agent_id="regressor",
            content="Trying to go back to P1",
            phase="P1"  # Attempting backward move
        )
        
        # Mock validator to approve (testing room logic, not validator)
        room.tile_validator.validate_tile = AsyncMock(return_value=True)
        
        result = await room.submit_tile(regression_tile)
        
        # Should be rejected by room logic
        assert result is False, "Protocol regression should be rejected"
        assert room.room_state.current_phase == "P2", \
            "Phase should not regress to P1"
        
        # Tile should not be stored
        assert len(room.room_state.submitted_tiles) == 0, \
            "Regression tile should not be stored"
    
    @pytest.mark.parametrize("memory_limit", [100, 500, 1000])
    @pytest.mark.asyncio
    async def test_memory_limit_enforcement(self, memory_limit):
        """Test room behavior when tile memory limit is reached.
        
        Edge: Resource exhaustion scenario.
        Expected: Oldest tiles evicted, room continues functioning.
        """
        room = DeadbandRoom(preset_name="deadband")
        room.room_state.max_tiles = memory_limit
        
        # Fill room to capacity
        for i in range(memory_limit):
            tile = Tile(
                agent_id=f"agent_{i}",
                content=f"tile_{i}",
                phase="P0"
            )
            room.tile_validator.validate_tile = AsyncMock(return_value=True)
            await room.submit_tile(tile)
        
        assert len(room.room_state.submitted_tiles) == memory_limit
        
        # Add one more tile - should trigger eviction
        eviction_tile = Tile(
            agent_id="eviction_test",
            content="should_evict_oldest",
            phase="P1"
        )
        await room.submit_tile(eviction_tile)
        
        # Verify size remains at limit
        assert len(room.room_state.submitted_tiles) == memory_limit
        
        # Verify oldest tile was evicted
        remaining_ids = [t.agent_id for t in room.room_state.submitted_tiles]
        assert "agent_0" not in remaining_ids, "Oldest tile should be evicted"
        assert "eviction_test" in remaining_ids, "New tile should be present"
        
        # Room should still function
        assert room.room_state.current_phase == "P1", \
            "Room should still track current phase"
    
    @pytest.mark.asyncio
    async def test_validator_unexpected_exception(self):
        """Test TileValidator throwing unexpected exception during validation.
        
        Edge: Dependency failure, not validation failure.
        Expected: Room handles exception gracefully, tile rejected.
        """
        room = DeadbandRoom(preset_name="deadband")
        
        # Configure validator to raise unexpected exception
        room.tile_validator.validate_tile = AsyncMock(
            side_effect=RuntimeError("Validator crashed!")
        )
        
        valid_tile = Tile(
            agent_id="test_agent",
            content="Perfectly valid content",
            phase="P0"
        )
        
        # Should not propagate exception to caller
        result = await room.submit_tile(valid_tile)
        
        assert result is False, "Should return False on validator exception"
        assert len(room.room_state.submitted_tiles) == 0, \
            "Tile should not be stored after validator crash"
        
        # Room should remain in operable state
        assert room.room_state.current_phase == "P0", \
            "Phase should be unchanged"
        # Verify we can still submit after recovery
        room.tile_validator.validate_tile = AsyncMock(return_value=True)
        recovery_result = await room.submit_tile(valid_tile)
        assert recovery_result is True, "Room should recover after validator fix"
