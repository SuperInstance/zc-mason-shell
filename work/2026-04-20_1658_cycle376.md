### Test 2: Tile Submission with Malformed JSON in Content Field
**Scenario:** Agent submits tile where `content` field contains invalid JSON string.
**Expected:** Room should reject tile, increment error count, maintain phase state.
```python
def test_tile_submission_malformed_json_content(mock_llm, tmp_path):
    """Tile with invalid JSON in content field should be rejected."""
    room = DeadbandRoom(
        llm=mock_llm,
        room_id="test_malformed_json",
        storage_path=tmp_path
    )
    
    initial_error_count = room.error_count
    
    # Submit tile with malformed JSON in content
    tile = {
        "agent_id": "test_agent",
        "phase": "p1",
        "content": '{"key": "value", "unclosed: true}',  # Malformed JSON
        "timestamp": "2026-04-20T16:45:00Z"
    }
    
    result = room.submit_tile(tile)
    
    assert result["valid"] is False
    assert "JSON decode error" in result.get("error", "")
    assert room.error_count == initial_error_count + 1
    assert room.current_phase == "p0"  # Should not change phase
```

### Test 3: State File Corruption with Valid Header

```python
def test_load_corrupted_state_file(tmp_path):
    """Room should handle corrupted state files gracefully."""
    state_file = tmp_path / "test_corrupted_state.json"
    
    # Write valid JSON header then corrupt the rest
    with open(state_file, 'w') as f:
        f.write('{"room_id": "test_corrupted", "current_phase": "p0",')
        f.write('"corrupted": true,')  # Missing closing brace
    
    # Should not raise exception
    room = DeadbandRoom(
        llm=None,  # LLM not needed for load test
        room_id="test_corrupted",
        storage_path=tmp_path
    )
    
    # Should fall back to default initialization
    assert room.room_id == "test_corrupted"
    assert room.current_phase == "p0"  # Default phase
    assert room.error_count > 0  # Should record load error
```

### Test 4: Concurrent Tile Submission Simulation
**Scenario:** Simulate rapid sequential tile submissions from same agent.
**Expected:** Room should process tiles in order, maintain consistent state.
```python
def test_rapid_sequential_tile_submission(mock_llm, tmp_path):
    """Room should handle rapid tile submissions without state corruption."""
    room = DeadbandRoom(
        llm=mock_llm,
        room_id="test_rapid_tiles",
        storage_path=tmp_path
    )
    
    tiles = []
    for i in range(5):
        tile = {
            "agent_id": "test_agent",
            "phase": "p1",
            "content": f'{{"channel_{i}": "test_data"}}',
            "timestamp": f"2026-04-20T16:45:{i:02d}Z"
        }
        tiles.append(tile)
    
    results = []
    for tile in tiles:
        results.append(room.submit_tile(tile))
    
    # All tiles should be processed
    assert len(results) == 5
    assert all(r["valid"] for r in results)
    
    # Room state should be consistent
    assert room.tile_count == 5
    assert len(room.tile_history) == 5
    
    # Tiles should be in chronological order
    timestamps = [t["timestamp"] for t in room.tile_history]
    assert timestamps == sorted(timestamps)
```

```python
def test_phase_transition_at_tile_limit(mock_llm, tmp_path):
    """Room should transition phase when reaching tile limit for current phase."""
    room = DeadbandRoom(
        llm=mock_llm,
        room_id="test_tile_limit",
        storage_path=tmp_path,
        max_tiles_per_phase=3
    )
    
    mock_llm.respond.return_value = {
        "p0_map": ["test_constraint"],
        "confidence": 0.9
    }
    
    # Initial P0 execution
    room.execute_phase("p0", agent_id="test_agent")
    
    # Submit tiles up to limit
    for i in range(3):
        tile = {
            "agent_id": "test_agent",
            "phase": "p0",
            "content": f'{{"tile_{i}": "data"}}',
            "timestamp": f"2026-04-20T16:46:{i:02d}Z"
        }
        room.submit_tile(tile)
    
    # Should still be in P0 (at limit)
    assert room.current_phase == "p0"
    assert room.tile_count == 3
    
    # One more tile should trigger phase transition
    mock_llm.respond.return_value = {
        "p1_channels": ["safe_channel"],
        "confidence": 0.8
    }
    
    tile = {
        "agent_id": "test_agent",
        "phase": "p0",
        "content": '{"excess_tile": "data"}',
        "timestamp": "2026-04-20T16:47:00Z"
    }
    
    result = room.submit_tile(tile)
    
    # Should transition to P1
    assert result["valid"] is True
    assert room.current_phase == "p1"
    assert room.tile_count == 0  # Reset for new phase
```
