### Test 1: Nested Constraint Violation
```python
def test_deadband_nested_violation():
    """Test P0 detection within an established P1 safe channel."""
    room = DeadbandRoom()
    # Establish safe channel with valid constraints
    result = room.process([
        {"priority": 0, "constraints": ["no_rock"]},
        {"priority": 1, "channels": ["safe_path"]}
    ])
    # Inject P0 violation within P1 channel
    contaminated = room.process([
        {"priority": 1, "channels": ["safe_path"], "constraints": ["no_rock"]}
    ])
    assert contaminated["phase"] == 0  # Should revert to P0
    assert "contaminated_channel" in contaminated["errors"]
```

### Test 3: Protocol Interruption Recovery
```python
def test_deadband_interruption_recovery():
    """Test protocol recovery after external interruption."""
    room = DeadbandRoom()
    # Start normal flow
    room.process([
        {"priority": 0, "constraints": ["avoid_a"]},
        {"priority": 1, "channels": ["path_b"]}
    ])
    # Simulate interruption (e.g., system reset)
    room.reset_state()
    # Resume with partial memory
    result = room.process([
        {"priority": 0, "constraints": ["avoid_a"]},
        {"priority": 1, "channels": ["path_b", "path_c"]}
    ], resume=True)
    assert result["phase"] == 1  # Should recover to P1
    assert "recovery_attempts" in result["metadata"]
```

```python
def test_deadband_priority_contamination():

        {"priority": 0, "constraints": ["no_rock"]},

    # Attempt to use P2 data in P0 context

    assert "priority_contamination" in contaminated["errors"]
    assert contaminated["valid"] is False
```

### Test 5: Cross-Cycle State Persistence
```python
def test_deadband_cross_cycle_persistence():
    """Test that P0 constraints persist across multiple processing cycles."""
    room = DeadbandRoom()
    # Cycle 1: Establish constraints
    result1 = room.process([
        {"priority": 0, "constraints": ["avoid_zone_a", "avoid_zone_b"]}
    ])
    # Cycle 2: Try to establish channels
    result2 = room.process([
        {"priority": 1, "channels": ["through_zone_a"]}  # Violates persisted constraint
    ])
    assert result2["phase"] == 0
    assert "violates_persisted_constraint" in result2["errors"]
    # Cycle 3: Valid channel that respects persisted constraints
    result3 = room.process([
        {"priority": 1, "channels": ["around_zone_a"]}
    ])
    assert result3["phase"] == 1
    assert len(result3["safe_channels"]) > 0
```
