### Test 3: Concurrent Submissions (Thread Safety)
```python
import threading

def test_concurrent_tile_submissions(deadband_room):
    """P1: Room should handle simultaneous submissions without corruption."""
    tile_template = {"phase": "P1", "data": {"channel": "test"}, "agent_id": "agent_{}"}
    
    def submit_tiles(agent_num, count=10):
        for i in range(count):
            tile = tile_template.copy()
            tile["agent_id"] = tile["agent_id"].format(agent_num)
            deadband_room.submit_tile(tile)
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tiles, args=(i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All tiles should be accepted without data corruption
    assert deadband_room.get_tile_count() == 50
    # Verify no duplicate IDs or missing tiles
    tile_ids = [t["agent_id"] for t in deadband_room.get_all_tiles()]
    assert len(set(tile_ids)) == 50  # All unique
```

```python
def test_malformed_phase_data(deadband_room):
    """P0: Tiles with invalid phase data structures should be rejected."""
    # P1 requires "channels" list, P2 requires "optimizations" dict
    invalid_p1_tile = {"phase": "P1", "data": {"not_channels": []}, "agent_id": "test"}
    invalid_p2_tile = {"phase": "P2", "data": {"not_optimizations": {}}, "agent_id": "test"}
    
    p1_result = deadband_room.submit_tile(invalid_p1_tile)
    p2_result = deadband_room.submit_tile(invalid_p2_tile)
    
    assert p1_result["status"] == "rejected"
    assert "structure" in p1_result["reason"].lower() or "channel" in p1_result["reason"].lower()
    assert p2_result["status"] == "rejected"
    assert "structure" in p2_result["reason"].lower() or "optimization" in p2_result["reason"].lower()
```
