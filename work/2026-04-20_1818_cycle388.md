```python
import asyncio
from unittest.mock import AsyncMock, patch
import pytest

@pytest.mark.asyncio
async def test_concurrent_tile_submissions(deadband_room):
    """Test that room handles multiple agents submitting tiles simultaneously."""
    room = deadband_room
    
    # Mock tile server to track submission order
    submission_times = []
    original_submit = room.tile_server.submit_tile
    
    async def tracked_submit(tile):
        submission_times.append(asyncio.get_event_loop().time())
        return await original_submit(tile)
    
    room.tile_server.submit_tile = tracked_submit
    
    # Create multiple agents
    agents = [
        {"id": f"agent_{i}", "phase": 1, "channel": f"channel_{i % 3}"}
        for i in range(5)
    ]
    
    # Submit tiles concurrently
    tasks = []
    for agent in agents:
        tile = {
            "agent_id": agent["id"],
            "phase": agent["phase"],
            "channel": agent["channel"],
            "content": f"Test from {agent['id']}"
        }
        tasks.append(room.process_tile(tile))
    
    await asyncio.gather(*tasks)
    
    # Verify all submissions were processed
    assert len(submission_times) == 5
    # Verify room state reflects last processed tile
    assert room.state["last_agent"] == "agent_4"
```

```python
@pytest.mark.asyncio
async def test_network_partition_recovery(deadband_room):
    """Test room behavior when tile server becomes temporarily unreachable."""
    room = deadband_room
    
    # First submission succeeds
    tile1 = {
        "agent_id": "test_agent",
        "phase": 0,
        "channel": "negative_space",
        "content": "P0 mapping"
    }
    
    result1 = await room.process_tile(tile1)
    assert result1["status"] == "accepted"
    
    # Simulate network partition - tile server raises connection error
    room.tile_server.submit_tile = AsyncMock(side_effect=ConnectionError("Network unreachable"))
    
    tile2 = {
        "agent_id": "test_agent",
        "phase": 1,
        "channel": "safe_channel",
        "content": "P1 exploration"
    }
    
    # Should handle error gracefully
    with pytest.raises(ConnectionError):
        await room.process_tile(tile2)
    
    # Verify room maintains previous state
    assert room.state["last_successful_tile"]["agent_id"] == "test_agent"
    assert room.state["last_successful_tile"]["phase"] == 0
    
    # Restore connection
    room.tile_server.submit_tile = AsyncMock(return_value={"status": "accepted"})
    
    # Should recover and process new tile
    result3 = await room.process_tile(tile2)
    assert result3["status"] == "accepted"
```

### Test 3: Malformed Tile Data Validation

```python
@pytest.mark.asyncio
async def test_malformed_tile_validation(deadband_room):
    """Test room validation of tile structure and data types."""
    room = deadband_room
    
    test_cases = [
        # Missing required fields
        {"phase": 1, "channel": "test"},  # missing agent_id
        {"agent_id": "test", "channel": "test"},  # missing phase
        {"agent_id": "test", "phase": 1},  # missing channel
        
        # Invalid data types
        {"agent_id": 123, "phase": 1, "channel": "test"},  # agent_id not string
        {"agent_id": "test", "phase": "one", "channel": "test"},  # phase not int
        {"agent_id": "test", "phase": 1, "channel": 42},  # channel not string
        
        # Protocol violations
        {"agent_id": "test", "phase": 3, "channel": "test"},  # invalid phase
        {"agent_id": "test", "phase": 2, "channel": ""},  # empty channel in P2
        {"agent_id": "test", "phase": 0, "channel": "optimized"},  # channel in P0
    ]
    
    for i, tile in enumerate(test_cases):
        with pytest.raises((ValueError, KeyError)):
            await room.process_tile(tile)
        
        # Verify room state unchanged after invalid tile
        if i == 0:
            initial_state = room.state.copy()
        else:
            assert room.state == initial_state
```

### Test 4: Protocol Violation Recovery

```python
@pytest.mark.asyncio
async def test_protocol_violation_recovery(deadband_room):
    """Test agent skipping P0→P2 then correcting with P1."""
    room = deadband_room
    
    # Agent incorrectly submits P2 tile without P0/P1
    bad_tile = {
        "agent_id": "reckless_agent",
        "phase": 2,
        "channel": "optimized_path",
        "content": "Trying to optimize without mapping"
    }
    
    # Room should reject or flag this
    result = await room.process_tile(bad_tile)
    assert result["status"] in ["rejected", "flagged"]
    assert "protocol_violation" in result.get("flags", [])
    
    # Agent corrects with proper P0
    p0_tile = {
        "agent_id": "reckless_agent",
        "phase": 0,
        "channel": None,
        "content": "Mapping negative space"
    }
    
    result = await room.process_tile(p0_tile)
    assert result["status"] == "accepted"
    
    # Now P1 should work
    p1_tile = {
        "agent_id": "reckless_agent",
        "phase": 1,
        "channel": "safe_channel_found",
        "content": "Found safe channel"
    }
    
    result = await room.process_tile(p1_tile)
    assert result["status"] == "accepted"
    
    # Now P2 should be accepted
    p2_tile = {
        "agent_id": "reckless_agent",
        "phase": 2,
        "channel": "optimized_path",
        "content": "Now optimizing properly"
    }
    
    result = await room.process_tile(p2_tile)
    assert result["status"] == "accepted"
```
