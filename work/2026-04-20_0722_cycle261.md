### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """P0: Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    threads = []
    results = []
    
    def submit_tile(agent_id):
        try:
            room.submit_tile(agent_id, {"phase": "P1", "content": f"test from {agent_id}"})
            results.append((agent_id, "success"))
        except Exception as e:
            results.append((agent_id, str(e)))
    
    for agent_id in agent_ids:
        t = threading.Thread(target=submit_tile, args=(agent_id,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    success_count = sum(1 for _, status in results if status == "success")
    self.assertEqual(success_count, len(agent_ids))
    
    # Verify room state consistency
    self.assertEqual(len(room.tiles), len(agent_ids))
    self.assertEqual(len(set(room.tiles.keys())), len(agent_ids))  # All unique agent IDs
```

### Test 2: Protocol Violation with Recovery
```python
def test_skip_p1_recovery(self):
    """P0: Agent skipping P1 should be rejected but room should remain operational."""
    room = DeadbandRoom()
    agent_id = "rogue_agent"
    
    # Initialize at P0
    room.set_agent_phase(agent_id, "P0")
    
    # Attempt illegal transition P0 → P2 (skipping P1)
    with self.assertRaises(ProtocolViolationError) as context:
        room.submit_tile(agent_id, {"phase": "P2", "content": "skipping P1"})
    
    self.assertIn("P1", str(context.exception))  # Error should mention missing P1
    
    # Verify room still accepts valid transitions
    room.set_agent_phase(agent_id, "P1")  # Manual correction
    room.submit_tile(agent_id, {"phase": "P1", "content": "valid after recovery"})
    self.assertEqual(room.get_agent_phase(agent_id), "P1")
```

### Test 3: Malformed Tile Data Handling
```python
def test_malformed_tile_validation(self):
    """P0: Invalid tile structures should be rejected with specific error messages."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    room.set_agent_phase(agent_id, "P1")
    
    test_cases = [
        (None, "Tile data cannot be None"),
        ({}, "Missing required field: phase"),
        ({"phase": "P1"}, "Missing required field: content"),
        ({"phase": "INVALID", "content": "test"}, "Invalid phase: INVALID"),
        ({"phase": "P1", "content": ""}, "Content cannot be empty"),
        ("not a dict", "Tile must be a dictionary"),
    ]
    
    for tile_data, expected_error in test_cases:
        with self.subTest(tile_data=tile_data):
            with self.assertRaises(ValidationError) as context:
                room.submit_tile(agent_id, tile_data)
            self.assertIn(expected_error, str(context.exception))
```

### Test 4: Room Capacity Enforcement
```python
def test_max_agents_enforcement(self):
    """P0: Exceeding MAX_CONCURRENT_AGENTS should queue or reject new agents."""
    room = DeadbandRoom()
    max_agents = room.MAX_CONCURRENT_AGENTS
    
    # Fill room to capacity
    for i in range(max_agents):
        agent_id = f"agent_{i}"
        room.register_agent(agent_id)
        room.set_agent_phase(agent_id, "P0")
    
    # Attempt to exceed capacity
    with self.assertRaises(RoomCapacityError) as context:
        room.register_agent("excess_agent")
    
    self.assertIn(str(max_agents), str(context.exception))
    
    # Remove one agent, should accept new one
    room.remove_agent("agent_0")
    room.register_agent("replacement_agent")  # Should succeed
    self.assertIn("replacement_agent", room.agents)
```

### Test 5: Network Partition During State Transition
```python
def test_partial_state_transition(self):
    """P0: Simulate network failure during P1→P2 transition."""
    room = DeadbandRoom()
    agent_id = "unlucky_agent"
    room.set_agent_phase(agent_id, "P1")
    
    # Mock network failure after phase update but before tile storage
    original_store_tile = room._store_tile
    
    def failing_store_tile(agent_id, tile):
        raise ConnectionError("Network partition")
    
    room._store_tile = failing_store_tile
    
    # Attempt transition should fail
    with self.assertRaises(ConnectionError):
        room.submit_tile(agent_id, {"phase": "P2", "content": "transitioning"})
    
    # Restore original method
    room._store_tile = original_store_tile
    
    # Verify agent is in consistent state (should remain at P1)
    self.assertEqual(room.get_agent_phase(agent_id), "P1")
    
    # Verify no partial tile was stored
    self.assertNotIn(agent_id, room.tiles)
    
    # Should be able to retry successfully
    room.submit_tile(agent_id, {"phase": "P2", "content": "retry after failure"})
    self.assertEqual(room.get_agent_phase(agent_id), "P2")
```
