def test_p2_skip_without_p1_recovery(self):
    """Test that attempting P2 optimization without completing P1 triggers recovery to P1."""
    room = DeadbandRoom()
    agent_id = "test_agent_skip"
    
    # Attempt P2 directly (protocol violation)
    result = room.process_tile(agent_id, {
        "phase": "P2",
        "action": "optimize",
        "parameters": {"target": "max_efficiency"}
    })
    
    # Should reject and guide back to P1
    assert result["status"] == "rejected"
    assert "P1" in result["message"]
    assert "complete P1 first" in result["message"].lower()
    
    # Verify room state preserved agent context
    assert room.get_agent_phase(agent_id) == "P0"  # Reset to start
```

### Test 2: Extreme Channel Boundary Values

def test_channel_boundary_extremes(self):
    """Test safe channel identification with extreme parameter values."""
    room = DeadbandRoom()
    agent_id = "test_agent_extreme"
    
    # Complete P0
    room.process_tile(agent_id, {
        "phase": "P0",
        "action": "map_negative",
        "parameters": {"constraints": ["rock1", "rock2"]}
    })
    
    # P1 with extreme values (near numeric limits)
    result = room.process_tile(agent_id, {
        "phase": "P1",
        "action": "find_channels",
        "parameters": {
            "search_space": 1e-10,  # Extremely small
            "tolerance": 1e308,     # Near float max
            "dimensions": 1000      # High dimensionality
        }
    })
    
    # Should handle gracefully, not crash
    assert result["status"] in ["accepted", "rejected"]
    if result["status"] == "accepted":
        assert "channels" in result
        # Verify channels are valid even with extremes
        for channel in result["channels"]:
            assert "bounds" in channel
            assert channel["bounds"]["lower"] < channel["bounds"]["upper"]
```

### Test 3: Concurrent Agent Phase Collision
```python
def test_concurrent_agent_phase_collision(self):
    """Test multiple agents interacting with room simultaneously causing phase state collisions."""
    room = DeadbandRoom()
    agents = [f"agent_{i}" for i in range(5)]
    
    # All agents start P0
    for agent_id in agents:
        room.process_tile(agent_id, {
            "phase": "P0",
            "action": "map_negative",
            "parameters": {"constraints": ["shared_rock"]}
        })
    
    # Simulate concurrent P1 submissions (thread-safe check)
    results = []
    for agent_id in agents:
        result = room.process_tile(agent_id, {
            "phase": "P1",
            "action": "find_channels",
            "parameters": {"search_space": "shared"}
        })
        results.append((agent_id, result))
    
    # Each agent should get unique channel assignments or graceful handling
    channel_sets = set()
    for agent_id, result in results:
        if result["status"] == "accepted":
            if "channels" in result:
                # Convert to hashable representation
                channel_repr = str(sorted([str(c) for c in result["channels"]]))
                channel_sets.add(channel_repr)
    
    # Either all get same (shared resource) or different channels
    # Key is no crashes and consistent state
    assert room.agent_states_count() == len(agents)
```

### Test 4: Malformed Tile Structure Recovery
```python
def test_malformed_tile_recovery(self):
    """Test room handling of structurally invalid tiles."""
    room = DeadbandRoom()
    agent_id = "test_agent_malformed"
    
    malformed_tiles = [
        {},  # Empty tile
        {"phase": "P0"},  # Missing action
        {"action": "map_negative"},  # Missing phase
        {"phase": "P0", "action": "map_negative", "parameters": "not_a_dict"},  # Wrong type
        None,  # Null tile
        {"phase": "INVALID_PHASE", "action": "invalid_action", "parameters": {}},  # Invalid values
    ]
    
    for i, tile in enumerate(malformed_tiles):
        result = room.process_tile(agent_id, tile)
        
        # Should reject with informative error, not crash
        assert result["status"] == "rejected"
        assert "error" in result or "message" in result
        
        # Room state should remain stable
        assert room.get_agent_phase(agent_id) in ["P0", None]  # Either reset or no state
        
    # After malformed tiles, valid tile should still work
    valid_result = room.process_tile(agent_id, {
        "phase": "P0",
        "action": "map_negative",
        "parameters": {"constraints": ["recovery_test"]}
    })
    assert valid_result["status"] == "accepted"
```

### Test 5: Resource Exhaustion Resilience
```python
def test_memory_exhaustion_resilience(self):
    """Test room behavior under memory pressure from excessive tile history."""
    room = DeadbandRoom()
    agent_id = "test_agent_memory"
    
    # Generate large number of tiles (simulating memory pressure)
    large_history = []
    for i in range(1000):  # Larger than normal operation
        tile = {
            "phase": "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
            "action": f"action_{i}",
            "parameters": {"data": "x" * 1000}  # 1KB per parameter
        }
        large_history.append(tile)
    
    # Process all tiles
    results = []
    for tile in large_history:
        result = room.process_tile(agent_id, tile)
        results.append(result)
        
        # Room should remain responsive, even if rejecting some
        assert result["status"] in ["accepted", "rejected"]
        
        # If accepted, verify phase progression is logical
        if result["status"] == "accepted":
            current_phase = room.get_agent_phase(agent_id)
            # P2 should only follow P1, P1 should only follow P0
            if "P2" in str(tile):
                assert current_phase == "P2"
    
    # Final state should be coherent
    final_phase = room.get_agent_phase(agent_id)
    assert final_phase in ["P0", "P1", "P2", None]
    
    # Room should still accept new tiles after exhaustion test
    recovery_result = room.process_tile(agent_id, {
        "phase": "P0",
        "action": "map_negative",
        "parameters": {"constraints": ["post_exhaustion"]}
    })
    assert recovery_result["status"] == "accepted"
```
