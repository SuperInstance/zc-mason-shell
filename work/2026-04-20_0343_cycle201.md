```python
# File: plato-torch/tests/test_deadband_edge.py
"""
Edge-case tests for DeadbandRoom.
These complement the existing test suite in test_deadband.py.
"""

import pytest
import numpy as np
from unittest.mock import Mock, patch
from plato_torch.rooms.deadband import DeadbandRoom

class TestDeadbandRoomEdgeCases:
    """Edge-case tests for DeadbandRoom protocol enforcement."""
    
    def test_protocol_priority_violation(self):
        """P0: Agent attempts P2 action before P1 channel finding.
        
        Expected: Room should reject the action, remain in P0,
        and return negative reward.
        """
        room = DeadbandRoom()
        state = room.reset()
        
        # Force state to be in P0 (negative space mapping)
        room.current_phase = 0
        # Agent incorrectly tries a P2 optimization action
        invalid_action = {"phase": 2, "optimization": "gradient_descent"}
        
        next_state, reward, done, info = room.step(invalid_action)
        
        assert reward < 0, "P0 violation should yield negative reward"
        assert room.current_phase == 0, "Should remain in P0 after violation"
        assert "protocol_violation" in info
        
    def test_state_persistence_after_reset(self):
        """P1: Room reset should not leak information between episodes.
        
        Expected: New episode should have fresh state with no memory
        of previous episode's internal buffers.
        """
        room = DeadbandRoom()
        
        # First episode
        state1 = room.reset()
        action1 = {"phase": 1, "channel": "safe_path_a"}
        room.step(action1)
        
        # Capture internal state before reset
        pre_reset_phase = room.current_phase
        pre_reset_step = room.step_count
        
        # Second episode
        state2 = room.reset()
        
        assert room.current_phase == 0, "Should reset to P0"
        assert room.step_count == 0, "Step counter should reset"
        assert not np.array_equal(state1, state2), "States should differ"
        
    def test_action_space_mutation_during_step(self):
        """P0: External code mutates action_space during step execution.
        
        Expected: Room should defensively copy or validate action_space
        to prevent external mutation from affecting protocol.
        """
        room = DeadbandRoom()
        room.reset()
        
        # Simulate external mutation
        original_action = {"phase": 1, "channel": "safe_channel"}
        mutated_action = original_action.copy()
        
        with patch.object(room, 'action_space', wraps=room.action_space) as mock_space:
            # External agent modifies action_space during step
            def mutate_during_step(action):
                action["injected"] = "malicious"
                return Mock()
            mock_space.sample.side_effect = mutate_during_step
            
            next_state, reward, done, info = room.step(mutated_action)
            
            # Action should be validated before use
            assert "injected" not in info.get("action_validated", {}), \
                "Should validate action before processing"
                
    def test_sentinel_values_in_state(self):
        """P0: State contains NaN, Infinity, or None values.
        
        Expected: Room should detect invalid state, reset to safe defaults,
        and return negative reward.
        """
        room = DeadbandRoom()
        room.reset()
        
        # Inject sentinel values into state
        invalid_state = np.array([1.0, np.nan, np.inf, -np.inf, 2.0])
        
        # Use reflection to simulate corrupted state
        room._state = invalid_state
        
        # Attempt protocol step
        action = {"phase": 0, "negative_space": "avoid_nan"}
        next_state, reward, done, info = room.step(action)
        
        # Room should either reset or handle gracefully
        assert not np.any(np.isnan(next_state)), "State should not contain NaN"
        assert not np.any(np.isinf(next_state)), "State should not contain Inf"
        assert reward <= 0, "Invalid state should not yield positive reward"
        
    def test_reward_shaping_by_phase(self):
        """P2: Verify reward correlates with protocol phase success.
        
        Expected: P0 violations → negative, P1 channel finding → small positive,
        P2 optimization → larger positive.
        """
        room = DeadbandRoom()
        room.reset()
        
        # Test P0 violation
        room.current_phase = 0
        p0_action = {"phase": 2, "skip": True}  # Violation
        _, reward_p0, _, _ = room.step(p0_action)
        
        # Test P1 success
        room.reset()
        room.current_phase = 0
        p1_action = {"phase": 1, "channel": "valid_channel"}
        _, reward_p1, _, _ = room.step(p1_action)
        
        # Test P2 success
        room.current_phase = 2
        p2_action = {"phase": 2, "optimization": "valid_opt"}
        _, reward_p2, _, _ = room.step(p2_action)
        
        # Assert reward shaping
        assert reward_p0 < 0, "P0 violation should be negative"
        assert reward_p1 >= 0, "P1 success should be non-negative"
        assert reward_p2 > reward_p1, "P2 optimization should reward more than P1"
```
