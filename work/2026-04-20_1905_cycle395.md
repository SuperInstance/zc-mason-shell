```python
def test_concurrent_tile_submissions(self):
    """Test that concurrent tile submissions from multiple agents don't corrupt state."""
    room = DeadbandRoom()
    agent1 = MockAgent(id="agent1")
    agent2 = MockAgent(id="agent2")
    room.register_agent(agent1)
    room.register_agent(agent2)
    
    # Both agents submit valid P0 tiles simultaneously
    tile1 = Tile(phase="P0", content={"negative_space": "avoid_rock"})
    tile2 = Tile(phase="P0", content={"negative_space": "avoid_edge"})
    
    # Simulate concurrent submission
    results = []
    for tile in [tile1, tile2]:
        result = room.submit_tile(agent1.id if tile == tile1 else agent2.id, tile)
        results.append(result)
    
    # Both should be accepted
    assert all(results), "Concurrent valid submissions should all succeed"
    assert len(room.get_tiles(phase="P0")) == 2, "Should have 2 P0 tiles"
```

### Test 2: Malformed Tile Structure Validation
```python
def test_malformed_tile_validation(self):
    """Test that tiles with missing required fields are rejected."""
    room = DeadbandRoom()
    agent = MockAgent(id="test_agent")
    room.register_agent(agent)
    
    # Tile missing 'phase' field
    malformed_tile1 = {"content": {"data": "test"}}  # Missing phase
    # Tile with invalid phase value
    malformed_tile2 = {"phase": "P3", "content": {"data": "test"}}  # Invalid phase
    # Tile missing content entirely
    malformed_tile3 = {"phase": "P0"}  # Missing content
    
    malformed_tiles = [malformed_tile1, malformed_tile2, malformed_tile3]
    
    rejection_count = 0
    for tile_data in malformed_tiles:
        try:
            # Attempt to create Tile object (should fail validation)
            tile = Tile(**tile_data)
            result = room.submit_tile(agent.id, tile)
            if not result:
                rejection_count += 1
        except (ValueError, TypeError):
            rejection_count += 1
    
    assert rejection_count == len(malformed_tiles), "All malformed tiles should be rejected"
```

### Test 5: Agent Disconnection Mid-Cycle

```python
def test_agent_disconnection_recovery(self):
    """Test room recovery when agent disconnects during active cycle."""
    room = DeadbandRoom()
    agent = MockAgent(id="disconnecting_agent")
    room.register_agent(agent)
    
    # Start a cycle
    room.start_cycle()
    
    # Agent submits partial work
    tile1 = Tile(phase="P0", content={"negative_space": "partial_work"})
    room.submit_tile(agent.id, tile1)
    
    # Simulate agent disconnection (deregister mid-cycle)
    room.deregister_agent(agent.id)
    
    # Room should handle missing agent gracefully
    # Attempt to submit from disconnected agent
    tile2 = Tile(phase="P0", content={"negative_space": "after_disconnect"})
    result = room.submit_tile(agent.id, tile2)
    assert not result, "Should reject tiles from disconnected agents"
    
    # Room should still be functional for other agents
    other_agent = MockAgent(id="other_agent")
    room.register_agent(other_agent)
    other_tile = Tile(phase="P0", content={"negative_space": "recovery"})
    other_result = room.submit_tile(other_agent.id, other_tile)
    assert other_result, "Room should accept tiles from newly registered agents"
    
    # Cycle should be able to complete
    room.end_cycle()
    assert room.cycle_complete, "Should be able to complete cycle after disconnection"
```
