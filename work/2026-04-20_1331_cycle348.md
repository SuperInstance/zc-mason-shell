**Key Components:**

- `_check_p1_channels(tile_data)` - checks if tile fits allowed categories
- `_check_p2_optimization(tile_data)` - scores tile quality
- `_record_ghost_tile(agent_id, reason)` - logs failed submissions as P0 knowledge

**Existing Tests Cover:**
- Basic tile submission and acceptance
- P0 rejection for absolute claims
- P1 rejection for out-of-channel content
- P2 scoring mechanics
- Ghost tile recording on failure

```python
def test_submit_empty_tile_creates_ghost_tile(self):
    """P0: Empty tile data should create ghost tile with 'empty_content' reason."""
    agent_id = "test_agent_1"
    empty_tile = {}
    
    result = self.room.submit_tile(agent_id, empty_tile)

    self.assertFalse(result["accepted"])
```

### Test 3: P1 Channel Boundary - Undefined Category
```python
def test_undefined_p1_category(self):
    """P1: Tile with category not in allowed_channels should be rejected."""
    agent_id = "test_agent_3"
    
    # Get current allowed channels
    allowed = self.room.allowed_channels
    
    # Create a category that definitely isn't allowed
    invalid_category = f"not_in_allowed_{hash('random')}"
    while invalid_category in allowed:
        invalid_category = f"not_in_allowed_{hash(invalid_category)}"
    
    valid_tile = {
        "content": "Factual content about fleet operations.",
        "category": invalid_category
    }
    
    result = self.room.submit_tile(agent_id, valid_tile)
    
    self.assertFalse(result["accepted"])
    self.assertIn("P1", result["reason"])
    self.assertIn("channel", result["reason"].lower())
```

### Test 5: P2 Scoring Edge Case - Minimum Viable Tile

```python
def test_p2_scoring_minimum_viable_tile(self):
    """P2: Tile that passes P0 and P1 but has minimal content should get low P2 score."""
    agent_id = "test_agent_5"
    
    # Minimum viable tile - passes P0 (no absolutes), passes P1 (valid category)
    minimal_tile = {
        "content": "Test.",
        "category": "test",
        "metadata": {"test": True}
    }
    
    result = self.room.submit_tile(agent_id, minimal_tile)
    
    # Should be accepted (passes P0 and P1)
    self.assertTrue(result["accepted"])
    
    # But should have low P2 score
    self.assertIn("p2_score", result)
    p2_score = result["p2_score"]
    
    # Score should be low but above minimum threshold
    self.assertLess(p2_score, 0.3)  # Assuming 0.0-1.0 scale
    self.assertGreaterEqual(p2_score, 0.0)
```
