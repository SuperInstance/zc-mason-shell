### Test 2: State Persistence Across Multiple Invalid Actions

```python
def test_state_persistence_after_p0_rejections(self):
    """Ensure room state remains unchanged after P0 rejections."""
    room = DeadbandRoom()
    initial_state = room.get_state()
    
    # Execute series of P0 violations
    for invalid_action in ["invalid_action", "unsafe_move", "out_of_bounds"]:
        result = room.step({"action": invalid_action, "value": 0.5})
        assert result["status"] == "P0_REJECTED"
    
    # State should remain identical to initial
    current_state = room.get_state()
    assert current_state == initial_state
    
    # Valid action should still work after rejections
    valid_result = room.step({"action": "move", "value": 0.5})
    assert valid_result["status"] == "P1_ACCEPTED"
```

### Test 3: Consecutive P2 Optimizations
```python
def test_consecutive_p2_optimizations(self):
    """Test that P2 optimization improves across multiple valid steps."""
    room = DeadbandRoom()
    optimization_scores = []
    
    # Execute series of valid actions
    for i in range(5):
        result = room.step({"action": "optimize", "value": 0.3})
        
        if result["status"] == "P2_OPTIMIZED":
            optimization_scores.append(result.get("improvement_score", 0))
        
        # Ensure we're not hitting P0 or staying at P1
        assert result["status"] in ["P1_ACCEPTED", "P2_OPTIMIZED"]
    
    # If we got P2 optimizations, scores should show improvement trend
    if len(optimization_scores) > 1:
        # Optimization should generally improve or maintain
        assert all(score >= 0 for score in optimization_scores)
```

### Test 4: Resource Exhaustion Edge Cases
```python
def test_resource_limits(self):
    """Test behavior when approaching computational/resource limits."""
    room = DeadbandRoom(max_steps=10)  # Small limit for testing
    
    steps_taken = 0
    last_status = None
    
    # Execute until limit or failure
    while steps_taken < 15:  # Try to exceed limit
        result = room.step({"action": "move", "value": 0.5})
        last_status = result["status"]
        
        if "ERROR" in last_status or "LIMIT" in last_status:
            break
            
        steps_taken += 1
    
    # Should hit limit around max_steps
    assert steps_taken <= 10
    assert last_status in ["P1_ACCEPTED", "LIMIT_REACHED", "ERROR_RESOURCE"]
```

```python
def test_action_pattern_detection(self):
    """Test room's response to repetitive or patterned action sequences."""
    room = DeadbandRoom()
    pattern_results = []
    
    # Execute repeating pattern
    pattern = [0.2, 0.8, 0.2, 0.8, 0.2]
    
    for value in pattern:
        result = room.step({"action": "move", "value": value})
        pattern_results.append(result["status"])
    
    # Analyze pattern response
    p0_count = pattern_results.count("P0_REJECTED")
    p1_count = pattern_results.count("P1_ACCEPTED")
    p2_count = pattern_results.count("P2_OPTIMIZED")
    
    # Valid pattern should have no P0 rejections
    assert p0_count == 0
    
    # Pattern might trigger P2 optimizations on recognition
    if p2_count > 0:
        # P2 should only occur after some P1 steps
        first_p2_index = pattern_results.index("P2_OPTIMIZED") if "P2_OPTIMIZED" in pattern_results else -1
        assert first_p2_index > 0  # Not on first step
```
