```python
def test_concurrent_agents_deadband_protocol():
    """Test multiple agents interacting simultaneously while maintaining P0→P1→P2 priority."""
    room = DeadbandRoom(room_id="concurrent_test")
    
    # Simulate 3 agents submitting tiles in parallel
    agent_tiles = [
        {"agent_id": "agent_1", "cycle": 1, "p0_data": {"negative_space": ["rock_a"]}},
        {"agent_id": "agent_2", "cycle": 1, "p0_data": {"negative_space": ["rock_b"]}},
        {"agent_id": "agent_3", "cycle": 1, "p0_data": {"negative_space": ["rock_c"]}}
    ]
    
    # Submit all tiles (simulating concurrent submission)
    results = []
    for tile in agent_tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all agents received valid P0 responses
    assert all(r["status"] == "accepted" for r in results)
    
    # Verify room state reflects all negative spaces
    room_state = room.get_state()
    assert len(room_state["negative_spaces"]) == 3
    assert all(f"rock_{chr(97+i)}" in room_state["negative_spaces"] for i in range(3))
```

### Test 2: Malformed Tile Structure Recovery

```python
def test_malformed_tile_recovery():
    """Test room's resilience to malformed tile structures with partial data."""
    room = DeadbandRoom(room_id="malformed_test")
    
    # Test cases of malformed tiles
    malformed_tiles = [
        {},  # Empty tile
        {"agent_id": "test_agent"},  # Missing cycle
        {"cycle": 1},  # Missing agent_id
        {"agent_id": "test", "cycle": 1, "p2_data": {"optimization": "skip_p0"}},  # Protocol violation
        {"agent_id": "test", "cycle": 1, "p0_data": {}},  # Empty negative space
        {"agent_id": "test", "cycle": 1, "p0_data": {"negative_space": []}},  # Empty array
        {"agent_id": "test", "cycle": 1, "p0_data": {"negative_space": [None]}},  # Invalid element
    ]
    
    expected_responses = [
        {"status": "rejected", "error": "missing_required_fields"},
        {"status": "rejected", "error": "missing_required_fields"},
        {"status": "rejected", "error": "missing_required_fields"},
        {"status": "rejected", "error": "protocol_violation"},
        {"status": "accepted", "warning": "empty_negative_space"},
        {"status": "accepted", "warning": "empty_negative_space"},
        {"status": "rejected", "error": "invalid_negative_space_element"},
    ]
    
    for tile, expected in zip(malformed_tiles, expected_responses):
        result = room.submit_tile(tile)
        assert result["status"] == expected["status"]
        if "error" in expected:
            assert expected["error"] in result.get("error", "")
        if "warning" in expected:
            assert expected["warning"] in result.get("warning", "")
```

### Test 3: Resource Exhaustion and Cleanup
```python
def test_memory_cleanup_after_tile_overflow():
    """Test room's memory management when exceeding tile capacity."""
    room = DeadbandRoom(room_id="overflow_test", max_tiles=10)
    
    # Fill room to capacity with valid tiles
    for i in range(10):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": i,
            "p0_data": {"negative_space": [f"obstacle_{i}"]}
        }
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # Verify at capacity
    assert len(room.get_state()["tiles"]) == 10
    
    # Submit one more tile (should trigger cleanup)
    overflow_tile = {
        "agent_id": "overflow_agent",
        "cycle": 11,
        "p0_data": {"negative_space": ["overflow_obstacle"]}
    }
    
    result = room.submit_tile(overflow_tile)
    
    # Room should either:
    # 1. Reject with capacity error, OR
    # 2. Accept and apply LRU cleanup
    if result["status"] == "rejected":
        assert "capacity" in result.get("error", "").lower()
    else:
        # If accepted, verify cleanup occurred
        state = room.get_state()
        assert len(state["tiles"]) <= 10
        # Most recent tile should be present
        assert any(t["agent_id"] == "overflow_agent" for t in state["tiles"])
```

```python
    original_state = room.get_state()
    
    # Phase 2: Room continues processing with stale connections
    # Simulate delayed tile arriving out of order
    delayed_tile = {
        "agent_id": "agent_b",
        "cycle": 0,  # Earlier cycle arriving late
        "p0_data": {"negative_space": ["old_obstacle"]}
    }
    
    result_delayed = room.submit_tile(delayed_tile)
    
    # Room should either:
    # 1. Reject out-of-order cycle, OR
    # 2. Incorporate with cycle adjustment
    if result_delayed["status"] == "rejected":
        assert "cycle" in result_delayed.get("error", "").lower()
    else:
        # If accepted, verify state consistency
        updated_state = room.get_state()
        # Negative spaces should include both old and new
        assert "network_failure" in updated_state["negative_spaces"]
        assert "old_obstacle" in updated_state["negative_spaces"]
        # Cycle tracking should reflect adjustment
        assert updated_state["latest_cycle"] >= 1
```
