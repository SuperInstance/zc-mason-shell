### Test 1: Protocol Phase Transition Under Memory Pressure
```rust
#[test]
fn test_deadband_protocol_phase_transition_under_memory_pressure() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use plato_torch::memory::MemoryPressureSimulator;
    
    let mut room = DeadbandRoom::new();
    let mut pressure_sim = MemoryPressureSimulator::new();
    
    // Simulate increasing memory pressure while protocol runs
    for i in 0..100 {
        pressure_sim.apply_pressure(i as f32 / 100.0);
        
        // Protocol should handle pressure without panicking
        let current_phase = room.protocol.current_phase();
        assert!(matches!(current_phase, P0 | P1 | P2));
        
        // Buffer should not overflow under pressure
        assert!(room.tile_buffer.usage() < 0.95);
    }
}
```

### Test 2: Tile Buffer Rollover with Ghost Tiles
```rust
#[test]
fn test_tile_buffer_rollover_with_ghost_tiles() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use plato_torch::tiles::{Tile, GhostTile};
    
    let mut room = DeadbandRoom::new();
    
    // Fill buffer to capacity with mixed live/ghost tiles
    for i in 0..room.tile_buffer.capacity() * 2 {
        let tile = if i % 3 == 0 {
            Tile::ghost(format!("ghost_{}", i), vec![P0; 5])
        } else {
            Tile::new(format!("live_{}", i), vec![P1; 3])
        };
        
        room.add_tile(tile);
        
        // Ghost tiles should persist through rollover
        let ghost_count = room.tile_buffer.ghost_count();
        assert!(ghost_count <= room.tile_buffer.capacity());
    }
}
```

#[test]
fn test_constraint_snapping_degenerate_geometry() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use constraint_theory_core::Geometry;
    
    let room = DeadbandRoom::new();
    
    // Test cases that should not panic
    let degenerate_cases = vec![
        Geometry::Point(0.0, 0.0, 0.0),  // Zero-dimensional
        Geometry::Line((0.0, 0.0), (0.0, 0.0)),  // Zero-length

    for geom in degenerate_cases {
        let result = room.constraint_snapper.snap(&geom);
        
        // Should return Some or None, not panic
        match result {
            Some(snapped) => {
                // Snapped geometry should be valid
                assert!(snapped.is_valid());
            }
            None => {
                // Degenerate case correctly rejected
            }
        }
    }
}

### Test 4: Sentiment Engine Under Rapid NPC Churn
```rust
#[test]
fn test_sentiment_engine_rapid_npc_churn() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use holodeck_rust::npc::NPC;
    
    let mut room = DeadbandRoom::new();
    
    // Create and destroy NPCs rapidly
    for i in 0..50 {
        let npc = NPC::new(format!("npc_{}", i));
        room.add_npc(npc.clone());
        
        // Immediate interaction
        room.record_interaction(&npc.id(), "test", 0.5);
        
        // Remove half of NPCs immediately
        if i % 2 == 0 {
            room.remove_npc(&npc.id());
        }
        
        // Sentiment map should handle missing NPCs gracefully
        let sentiment = room.get_sentiment_for(&npc.id());
        assert!(sentiment.is_none() || (sentiment.unwrap() >= -1.0 && sentiment.unwrap() <= 1.0));
    }
}
```

### Test 5: Protocol Reset During Active Training
```rust
#[test]
fn test_protocol_reset_during_active_training() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    
    let mut room = DeadbandRoom::new();
    
    // Start training
    room.start_training();
    assert!(room.is_training());
    
    // Add some tiles
    for i in 0..10 {
        room.add_tile(Tile::new(format!("tile_{}", i), vec![P1; 2]));
    }
    
    // Reset protocol while training
    room.protocol.reset();
    
    // Training should continue or handle reset gracefully
    // Buffer should retain tiles
    assert!(room.tile_buffer.count() >= 10);
    
    // Protocol should be in initial state
    assert_eq!(room.protocol.current_phase(), P0);
    
    // Can resume operations
    room.add_tile(Tile::new("post_reset", vec![P2; 1]));
    assert!(room.tile_buffer.count() >= 11);
}
```
