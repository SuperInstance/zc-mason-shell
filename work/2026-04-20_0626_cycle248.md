```python
def test_concurrent_channel_collision(self):
    """Test multiple agents attempting to write to same channel simultaneously."""
    room = DeadbandRoom()
    
    # Simulate two agents finding same safe channel
    agent1 = MockAgent(agent_id="agent_1")
    agent2 = MockAgent(agent_id="agent_2")
    
    # Both complete P0 (negative space mapping)
    room.process_agent_step(agent1, {"phase": "P0", "negative_space": ["rock1", "rock2"]})
    room.process_agent_step(agent2, {"phase": "P0", "negative_space": ["rock1", "rock2"]})
    
    # Both find same safe channel in P1
    safe_channel = {"id": "channel_alpha", "capacity": 1}  # Capacity of 1
    room.process_agent_step(agent1, {"phase": "P1", "safe_channels": [safe_channel]})
    
    # Second agent should receive collision error or alternative channel
    result = room.process_agent_step(agent2, {"phase": "P1", "safe_channels": [safe_channel]})
    
    # Verify collision handling
    assert "collision" in result or "alternative" in result
    assert room.get_channel_occupancy("channel_alpha") <= 1
```

```python
def test_protocol_violation_recovery(self):
    """Test agent skipping P1, getting rejected, then recovering properly."""
    room = DeadbandRoom()

    # Agent attempts to skip from P0 directly to P2 (violation)

    violation_result = room.process_agent_step(

    recovery_result = room.process_agent_step(

    )
    
    # Should now accept P1 and allow progression
    assert "accepted" in recovery_result
    assert agent.current_phase == "P1"
```

### Test 3: Ghost Tile Influence on P0 Mapping
```python
def test_ghost_tile_influence(self):
    """Test dead agents' ghost tiles influencing current agent's P0 decisions."""
    room = DeadbandRoom()
    
    # First agent dies after mapping negative space
    dead_agent = MockAgent(agent_id="ghost_agent", alive=False)
    room.process_agent_step(
        dead_agent,
        {"phase": "P0", "negative_space": ["hidden_rock", "trap_zone"]}
    )
    room.mark_agent_dead(dead_agent.agent_id)
    
    # Create ghost tile from dead agent's experience
    ghost_tile = room.create_ghost_tile(dead_agent.agent_id)
    room.add_ghost_tile(ghost_tile)
    
    # New agent starts P0, should receive ghost warnings
    new_agent = MockAgent(agent_id="new_agent")
    p0_result = room.process_agent_step(
        new_agent,
        {"phase": "P0", "negative_space": []}
    )
    
    # Verify ghost tile influence appears in results
    assert "ghost_warnings" in p0_result
    assert "hidden_rock" in str(p0_result["ghost_warnings"])
    assert "trap_zone" in str(p0_result["ghost_warnings"])
    
    # Agent's negative space should incorporate ghost knowledge
    assert len(new_agent.negative_space) > 0
```

### Test 4: Channel Saturation and Resource Exhaustion
```python
def test_channel_saturation(self):
    """Test behavior when all safe channels reach capacity."""
    room = DeadbandRoom()
    
    # Create limited channel set
    channels = [
        {"id": "chan1", "capacity": 2},
        {"id": "chan2", "capacity": 1},
        {"id": "chan3", "capacity": 1}
    ]
    room.initialize_channels(channels)
    
    agents = []
    # Fill all channels to capacity
    for i in range(4):  # Total capacity = 4
        agent = MockAgent(agent_id=f"agent_{i}")
        agents.append(agent)
        room.process_agent_step(agent, {"phase": "P0", "negative_space": []})
        room.process_agent_step(
            agent,
            {"phase": "P1", "safe_channels": [channels[i % 3]]}
        )
    
    # Fifth agent should find no available channels
    fifth_agent = MockAgent(agent_id="agent_5")
    room.process_agent_step(fifth_agent, {"phase": "P0", "negative_space": []})
    saturation_result = room.process_agent_step(
        fifth_agent,
        {"phase": "P1", "safe_channels": channels}
    )
    
    # Should indicate saturation and suggest waiting or P0 re-evaluation
    assert "saturated" in saturation_result
    assert "capacity" in saturation_result
    assert fifth_agent.current_phase == "P0"  # Should remain in P0
```

```python
def test_temporal_deadband_violation(self):
    """Test time-based constraint violations in optimization phase."""
    room = DeadbandRoom()
    agent = MockAgent(agent_id="temporal_test_agent")
    
    # Complete P0 and P1 normally
    room.process_agent_step(agent, {"phase": "P0", "negative_space": ["obstacle"]})
    room.process_agent_step(
        agent,
        {"phase": "P1", "safe_channels": [{"id": "timed_channel", "capacity": 3}]}
    )
    
    # Set up temporal constraint: optimization must complete within time window
    room.set_temporal_constraint(
        channel="timed_channel",
        max_duration=5.0  # 5 second window
    )
    
    # Simulate delayed optimization (exceeds time window)
    room.simulate_time_passage(6.0)  # 6 seconds pass
    
    # Attempt P2 optimization after deadline
    timed_result = room.process_agent_step(
        agent,
        {
            "phase": "P2", 
            "optimization": "delayed_path",
            "channel": "timed_channel",
            "duration": 6.5
        }
    )
    
    # Should violate temporal deadband
    assert "temporal_violation" in timed_result
    assert "deadline" in timed_result
    assert agent.optimization_attempts == 0  # Should not count as valid attempt
    
    # Verify agent gets reset to P1 for new channel selection
    assert agent.current_phase == "P1"
```
