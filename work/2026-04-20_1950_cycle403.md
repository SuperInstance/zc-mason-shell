### Test 1: Agent Submits Tile with Future Timestamp
**Scenario:** An agent submits a tile with a `timestamp` field set far in the future.
**Expected Behavior:** Room should reject tile as invalid (P0 violation of temporal logic).
**Rationale:** Prevents training data contamination with impossible timestamps.

```python
def test_future_timestamp_rejection(room, agent):
    """Tile with future timestamp should be rejected."""
    future_tile = {
        "agent_id": agent,
        "timestamp": "2030-01-01T00:00:00Z",  # Far future
        "content": "Test content",
        "priority": 1,
        "phase": 4
    }
    result = room.submit_tile(future_tile)
    assert result["status"] == "rejected"
    assert "timestamp" in result["reason"].lower()
```

### Test 4: Concurrent Tile Submission from Multiple Agents
**Scenario:** Two agents submit tiles simultaneously (simulated via threading).
**Expected Behavior:** Room should handle concurrent access without data corruption.
**Rationale:** Tests thread safety for high-volume training rooms.

```python
import threading

def test_concurrent_submissions(room):
    """Multiple agents submitting concurrently should not corrupt room state."""
    results = []
    
    def submit_agent_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "timestamp": "2026-04-20T19:48:00Z",
            "content": f"Test from {agent_id}",
            "priority": 0,
            "phase": 4
        }
        results.append(room.submit_tile(tile))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_agent_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    # Room tile count should match
    assert room.get_tile_count() == 5
```

### Test 5: Agent Submits Tile After Being Marked Inactive
**Scenario:** Agent submits tile after room has marked agent as inactive (due to timeout).
**Expected Behavior:** Room should reject tile or trigger agent reactivation flow.
**Rationale:** Tests lifecycle management of agent sessions.

```python
def test_tile_after_inactive(room, agent):
    """Tile from inactive agent should trigger reactivation or rejection."""
    # First tile to establish activity
    tile1 = {
        "agent_id": agent,
        "timestamp": "2026-04-20T19:49:00Z",
        "content": "Active tile",
        "priority": 0,
        "phase": 4
    }
    room.submit_tile(tile1)
    
    # Force agent inactive (simulate timeout)
    room.mark_agent_inactive(agent)
    
    # Try to submit another tile
    tile2 = {
        "agent_id": agent,
        "timestamp": "2026-04-20T19:50:00Z",
        "content": "Tile after inactive",
        "priority": 1,
        "phase": 4
    }
    result = room.submit_tile(tile2)
    
    # Either rejected or accepted with reactivation
    assert result["status"] in ["rejected", "accepted"]
    if result["status"] == "accepted":
        assert room.is_agent_active(agent) == True
```
