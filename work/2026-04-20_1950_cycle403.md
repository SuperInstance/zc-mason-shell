## Observed Room Behavior (from `deadband_room.py`)
- `DeadbandRoom` inherits from `BaseRoom`
- Primary method: `process_input(self, input_text: str) -> Dict[str, Any]`
- Protocol steps:
  1. **P0 (Negative Space)**: Validate input, check for disallowed patterns
  2. **P1 (Safe Channels)**: Find valid transformation paths
  3. **P2 (Optimization)**: Apply optimal transformation
- Returns dict with `status`, `output`, `tile_data` (if training tile generated)
- Training tiles submitted via `self.trainer.submit_tile(tile_data)`

## 5 Additional Edge-Case Tests

### Test 1: Protocol Violation — Skipping Directly to P2
**Scenario:** Input attempts to force P2 optimization without P0/P1 validation.
```python
def test_skip_to_p2_violation(deadband_room):
    """P0: Input that tries to jump directly to optimization should be rejected."""
    input_text = '{"phase": "P2", "action": "optimize", "data": {"target": "max_speed"}}'
    result = deadband_room.process_input(input_text)
    
    assert result["status"] == "error"
    assert "P0" in result["output"]  # Should mention P0 violation
    assert "skip" in result["output"].lower() or "violation" in result["output"].lower()
```

### Test 2: Malformed P1 Channel Specification
**Scenario:** P1 channel data is structurally valid but semantically impossible.
```python
def test_invalid_p1_channel_semantics(deadband_room):
    """P1: Channel specification that is JSON-valid but describes contradictory constraints."""
    input_text = '''{
        "phase": "P1",
        "channels": [
            {"constraint": "x > 10", "constraint": "x < 5"}
        ]
    }'''
    result = deadband_room.process_input(input_text)
    
    # Should either reject at P0 (contradiction detection) or return empty P1 result
    assert result["status"] in ["error", "success"]
    if result["status"] == "success":
        # If processed, should have empty or warning-filled output
        assert len(result.get("output", "")) > 0
```

### Test 4: Partial Protocol Completion
**Scenario:** Valid P0 and P1, but P2 optimization returns no viable solution.
```python
def test_p2_no_solution(deadband_room):
    """P2: Valid channels but no optimization path exists."""
    input_text = '''{
        "phase": "P0",
        "checks": ["no_errors"],
        "then": {
            "phase": "P1",
            "channels": [{"constraint": "x == 10"}]
        },
        "then": {
            "phase": "P2",
            "objective": "minimize y",
            "constraints": ["y < 0", "y > 10"]
        }
    }'''
    result = deadband_room.process_input(input_text)
    
    # Should complete P0 and P1 successfully, but P2 returns empty
    assert result["status"] == "success"
    output_data = result.get("output", {})
    # Output should indicate P2 produced no viable solution
    assert "no solution" in str(output_data).lower() or "empty" in str(output_data).lower()
```

### Test 5: Tile Submission with Invalid Metadata
**Scenario:** Training tile generated but with missing required metadata fields.
```python
def test_tile_submission_invalid_metadata(deadband_room, mocker):
    """Tile submission should validate metadata before accepting."""
    mock_submit = mocker.patch.object(deadband_room.trainer, 'submit_tile')
    
    # Room tries to submit tile with missing 'phase' field
    input_text = '{"generate_tile": true, "tile_data": {"content": "test", "agent": "mason"}}'
    result = deadband_room.process_input(input_text)
    
    # submit_tile should not be called with invalid data
    # Either room validates internally, or trainer rejects
    if mock_submit.called:
        call_args = mock_submit.call_args[0][0]
        # Check if call was made but with error status
        assert call_args.get("status") == "error"
    else:
        # Or room prevented submission entirely
        assert "tile" not in result.get("output", "").lower()
```
