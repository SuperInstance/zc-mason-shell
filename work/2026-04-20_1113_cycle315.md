### Test 1: Concurrent Tile Submission Race Condition
```python
import asyncio
import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom

@pytest.mark.asyncio
async def test_concurrent_tile_submission():
    """Test that concurrent tile submissions don't corrupt room state."""
    room = DeadbandRoom(room_id="edge-test-1")
    
    # Simulate 10 agents submitting tiles simultaneously
    async def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": asyncio.get_event_loop().time()
        }
        return await room.submit_tile(tile)
    
    # Submit concurrently
    tasks = [submit_tile(f"agent-{i}") for i in range(10)]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # Verify all submissions succeeded
    successes = [r for r in results if not isinstance(r, Exception)]
    assert len(successes) == 10, f"Expected 10 successes, got {len(successes)}"
    
    # Verify room state consistency
    assert room.tile_count == 10
    assert len(room.recent_tiles) == 10
```

### Test 2: Malformed Tile Payload Handling

```python
import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom, TileValidationError

def test_malformed_tile_payload():
    """Test that malformed tile payloads are rejected with appropriate errors."""
    room = DeadbandRoom(room_id="edge-test-2")
    
    test_cases = [
        # (tile_payload, expected_error)
        ({}, "Missing required field: agent_id"),
        ({"agent_id": "test"}, "Missing required field: phase"),
        ({"agent_id": "test", "phase": "P5"}, "Invalid phase: P5"),
        ({"agent_id": "test", "phase": "P1", "content": ""}, "Empty content"),
        ({"agent_id": "test", "phase": "P1", "content": "x" * 10001}, "Content too large"),
    ]
    
    for tile_payload, expected_error in test_cases:
        with pytest.raises(TileValidationError) as exc_info:
            room._validate_tile(tile_payload)
        assert expected_error in str(exc_info.value)
```

### Test 4: Resource Exhaustion During Batch Processing
```python
import pytest
import psutil
import os
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_resource_exhaustion_handling():
    """Test graceful handling when system resources are constrained."""
    room = DeadbandRoom(room_id="edge-test-4")
    
    # Simulate memory pressure
    original_memory_limit = room.memory_limit_mb
    room.memory_limit_mb = 1  # Artificially low limit
    
    large_tile = {
        "agent_id": "memory-test",
        "phase": "P1",
        "content": "x" * 1000000,  # 1MB content
    }
    
    # Should fail gracefully, not crash
    with pytest.raises(ResourceWarning) as exc_info:
        room.submit_tile(large_tile)
    assert "Memory limit exceeded" in str(exc_info.value)
    
    # Restore and verify normal operation resumes
    room.memory_limit_mb = original_memory_limit
    normal_tile = {**large_tile, "content": "normal size"}
    result = room.submit_tile(normal_tile)
    assert result["status"] == "accepted"
```

### Test 5: Network Partition Affecting Ghost Tile Sync

```python
import pytest
from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_ghost_tile_sync_network_partition():
    """Test behavior when ghost tile synchronization fails due to network issues."""
    room = DeadbandRoom(room_id="edge-test-5")
    
    # Mock network failure
    with patch('plato_torch.rooms.deadband_room.GhostTileSync.sync') as mock_sync:
        mock_sync.side_effect = ConnectionError("Network partition")
        
        # Room should still operate in degraded mode
        tile = {
            "agent_id": "partition-test",
            "phase": "P2",
            "content": "Optimizing despite network issues"
        }
        
        result = room.submit_tile(tile)
        
        # Verify tile accepted but ghost sync failed
        assert result["status"] == "accepted"
        assert result["ghost_sync"] == "failed"
        assert "Network partition" in result.get("warning", "")
        
        # Verify room continues to accept tiles
        tile2 = {**tile, "content": "Second tile after partition"}
        result2 = room.submit_tile(tile2)
        assert result2["status"] == "accepted"
```
