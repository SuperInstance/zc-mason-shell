### Test 1: Concurrent Tile Submission

def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3", "agent_4", "agent_5"]
    
    # Simulate concurrent submissions using threads
    import threading
    results = []
    
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "priority": "P0",
            "content": f"Negative observation from {agent_id}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        results.append((agent_id, result))
    
    threads = [threading.Thread(target=submit_tile, args=(aid,)) for aid in agent_ids]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    self.assertEqual(len(results), 5)
    for agent_id, result in results:
        self.assertTrue(result["success"], f"Agent {agent_id} submission failed")
    
    # Verify room state consistency
    self.assertEqual(room.tile_count, 5)
    self.assertEqual(len(room.p0_tiles), 5)
```

### Test 2: P0 Exhaustion Scenario

def test_p0_exhaustion_handling(self):
    """Room should handle scenario where all negative space is mapped."""
    room = DeadbandRoom()
    
    # Submit maximum P0 tiles (configurable limit)
    max_p0_tiles = room.config.get("max_p0_tiles", 1000)
    for i in range(max_p0_tiles):
        tile = {
            "agent_id": f"scout_{i}",
            "priority": "P0",
            "content": f"Negative space observation {i}",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # Attempt one more P0 tile
    overflow_tile = {
        "agent_id": "overflow_agent",
        "priority": "P0",
        "content": "This should trigger exhaustion handling",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(overflow_tile)
    
    # Expected behavior: either reject or trigger phase transition
    if not result["success"]:
        self.assertEqual(result["error"], "P0_EXHAUSTED")
    else:
        # If accepted, room should have transitioned phases
        self.assertTrue(room.current_phase in ["P1", "COMPLETE"])
```

### Test 3: P1 Channel Collapse During P2
```python
def test_p1_channel_collapse_during_p2(self):
    """Safe channel becoming unsafe during optimization should revert to P1."""
    room = DeadbandRoom()
    
    # Setup: Complete P0, establish P1 channels
    room.submit_tile({"agent_id": "scout", "priority": "P0", "content": "test"})
    room.current_phase = "P1"
    room.safe_channels = ["channel_a", "channel_b", "channel_c"]
    
    # Move to P2 optimization
    room.current_phase = "P2"
    room.active_channel = "channel_b"
    
    # Simulate channel collapse (external event or new P0 tile)
    collapse_tile = {
        "agent_id": "sentinel",
        "priority": "P0",
        "content": "Channel_b is now unsafe!",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(collapse_tile)
    
    # Room should revert to P1 and remove unsafe channel
    self.assertEqual(room.current_phase, "P1")
    self.assertNotIn("channel_b", room.safe_channels)
    self.assertIsNone(room.active_channel)
```

### Test 4: Malformed Tile Data Resilience
```python
def test_malformed_tile_handling(self):
    """Room should gracefully handle various malformed tile submissions."""
    room = DeadbandRoom()
    
    test_cases = [
        # (tile_data, expected_error)
        (None, "INVALID_JSON"),
        ("not json at all", "INVALID_JSON"),
        ({}, "MISSING_AGENT_ID"),  # Missing required field
        ({"agent_id": "test"}, "MISSING_PRIORITY"),
        ({"agent_id": "test", "priority": "INVALID_PRIO"}, "INVALID_PRIORITY"),
        ({"agent_id": "test", "priority": "P0", "content": "x" * 10000}, "CONTENT_TOO_LARGE"),
        ({"agent_id": "test", "priority": "P0", "content": "", "timestamp": "not-a-number"}, "INVALID_TIMESTAMP"),
    ]
    
    for tile_data, expected_error in test_cases:
        result = room.submit_tile(tile_data)
        self.assertFalse(result["success"])
        self.assertEqual(result["error"], expected_error)
        # Room state should remain unchanged
        self.assertEqual(room.tile_count, 0)
```

### Test 5: Room State Persistence and Recovery
```python
def test_state_persistence_recovery(self):
    """Room should recover state after simulated crash/restart."""
    room = DeadbandRoom()
    
    # Initial state with some tiles
    for i in range(3):
        room.submit_tile({
            "agent_id": f"agent_{i}",
            "priority": ["P0", "P1", "P2"][i],
            "content": f"Test tile {i}",
            "timestamp": time.time()
        })
    
    # Capture serialized state
    saved_state = room.serialize_state()
    
    # Simulate crash and restart
    new_room = DeadbandRoom()
    new_room.restore_state(saved_state)
    
    # Verify state recovery
    self.assertEqual(new_room.tile_count, 3)
    self.assertEqual(new_room.current_phase, "P2")  # Last tile was P2
    self.assertEqual(len(new_room.all_tiles), 3)
    
    # Verify tiles maintain integrity
    for original, restored in zip(room.all_tiles, new_room.all_tiles):
        self.assertEqual(original["agent_id"], restored["agent_id"])
        self.assertEqual(original["priority"], restored["priority"])
        self.assertEqual(original["content"], restored["content"])
```
