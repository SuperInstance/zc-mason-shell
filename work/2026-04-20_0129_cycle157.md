```python
def test_concurrent_agent_interactions(self):
    """Test that multiple agents can interact with DeadbandRoom simultaneously without state corruption."""
    room = DeadbandRoom(room_id="test_concurrent")
    
    # Simulate two agents submitting tiles at same time
    agent1_tile = {"priority": "P0", "channels": ["avoid_crash"], "data": "agent1"}
    agent2_tile = {"priority": "P0", "channels": ["avoid_memory_leak"], "data": "agent2"}
    
    # Process in rapid succession (simulating concurrent access)
    result1 = room.process_interaction(agent1_tile)
    result2 = room.process_interaction(agent2_tile)
    
    # Both should succeed
    self.assertTrue(result1["success"])
    self.assertTrue(result2["success"])
    
    # State should reflect both interactions
    self.assertIn("agent1", room.state["processed_tiles"])
    self.assertIn("agent2", room.state["processed_tiles"])
    
    # Tile queue should contain both
    self.assertEqual(len(room.get_pending_tiles()), 2)
```

### Test 3: Malformed JSON Recovery
```python
def test_malformed_json_recovery(self):
    """Test that DeadbandRoom handles malformed tile data without crashing."""
    room = DeadbandRoom(room_id="test_malformed")
    
    # Test various malformed inputs
    malformed_inputs = [
        "{invalid json",  # Unclosed brace
        '{"priority": "P0", "channels": []',  # Missing closing brace
        "",  # Empty string
        None,  # Null input
        b"binary data",  # Bytes instead of string
    ]
    
    for i, malformed in enumerate(malformed_inputs):
        try:
            result = room.process_interaction(malformed)
            # Should either fail gracefully or handle appropriately
            if result["success"]:
                self.assertIn("processed_tiles", room.state)
            else:
                self.assertIn("error", result)
        except Exception as e:
            # Room should not crash on malformed input
            self.fail(f"Room crashed on malformed input {i}: {str(e)}")
    
    # Room should still be functional after malformed inputs
    valid_tile = {"priority": "P0", "channels": ["recovery_test"], "data": "test"}
    result = room.process_interaction(valid_tile)
    self.assertTrue(result["success"])
```

### Test 4: Network Partition Simulation
```python
def test_network_partition_handling(self):
    """Test that DeadbandRoom handles partial write failures (simulating network partitions)."""
    room = DeadbandRoom(room_id="test_network")
    
    # Mock a storage failure after first write
    original_save = room._save_state
    call_count = 0
    
    def failing_save():
        nonlocal call_count
        call_count += 1
        if call_count == 2:  # Fail on second save attempt
            raise IOError("Simulated network partition")
        original_save()
    
    room._save_state = failing_save
    
    # First tile should succeed
    tile1 = {"priority": "P0", "channels": ["first"], "data": "tile1"}
    result1 = room.process_interaction(tile1)
    self.assertTrue(result1["success"])
    
    # Second tile should fail due to simulated partition
    tile2 = {"priority": "P1", "channels": ["second"], "data": "tile2"}
    result2 = room.process_interaction(tile2)
    self.assertFalse(result2["success"])
    self.assertIn("Simulated network partition", result2["error"])
    
    # Room should maintain consistency - first tile persisted, second not
    self.assertIn("tile1", room.state["processed_tiles"])
    self.assertNotIn("tile2", room.state["processed_tiles"])
```

```python
def test_priority_inversion_protection(self):
    """Test that P2 optimizations cannot proceed when P1 channels are unresolved."""
    room = DeadbandRoom(room_id="test_priority_inversion")
    
    # Start with P0 only
    p0_tile = {
        "priority": "P0",
        "channels": ["avoid_rock"],
        "data": "identified hazard"
    }
    room.process_interaction(p0_tile)
    
    # Attempt P2 optimization without P1 channel definition
    p2_tile = {
        "priority": "P2",
        "channels": ["optimize_path"],
        "data": "trying to optimize"
    }
    
    result = room.process_interaction(p2_tile)
    
    # Should fail - cannot optimize (P2) without safe channels (P1)
    self.assertFalse(result["success"])
    self.assertIn("P1 channels required", result["error"])
    
    # Now define P1 channel
    p1_tile = {
        "priority": "P1",
        "channels": ["safe_passage"],
        "data": "found safe channel"
    }
    room.process_interaction(p1_tile)
    
    # Now P2 should succeed
    result = room.process_interaction(p2_tile)
    self.assertTrue(result["success"])
```
