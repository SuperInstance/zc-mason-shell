### Test 1: Protocol Violation — Skipping to P2

```rust
#[test]
fn test_deadband_room_skipping_to_p2_violation() {
    let mut room = DeadbandRoom::new();
    // Attempt to submit a P2 tile without P0/P1
    let tile = TrainingTile {
        protocol_phase: ProtocolPhase::P2,
        content: "Optimizing path".to_string(),
        sentiment: 0.5,
        metadata: HashMap::new(),
    };
    
    let result = room.submit_tile(tile);
    assert!(result.is_err());
    assert_eq!(
        result.unwrap_err(),
        RoomError::ProtocolViolation("Cannot submit P2 without completing P0 and P1".to_string())
    );
    
    // Verify room state unchanged
    assert_eq!(room.current_phase(), ProtocolPhase::P0);
    assert_eq!(room.tile_count(), 0);
}
```

### Test 3: Concurrent Submission Race Condition
```rust
#[test]
fn test_deadband_room_concurrent_submission_handling() {
    let room = Arc::new(Mutex::new(DeadbandRoom::new()));
    let mut handles = vec![];
    
    // Simulate 10 concurrent agents submitting P0 tiles
    for i in 0..10 {
        let room_clone = Arc::clone(&room);
        handles.push(thread::spawn(move || {
            let mut room = room_clone.lock().unwrap();
            let tile = TrainingTile {
                protocol_phase: ProtocolPhase::P0,
                content: format!("P0 tile from agent {}", i),
                sentiment: -0.5,
                metadata: HashMap::new(),
            };
            room.submit_tile(tile)
        }));
    }
    
    // Collect results
    let mut success_count = 0;
    for handle in handles {
        match handle.join().unwrap() {
            Ok(_) => success_count += 1,
            Err(e) => println!("Concurrent submission error: {:?}", e),
        }
    }
    
    // All should succeed or fail gracefully
    let final_room = room.lock().unwrap();
    assert!(success_count > 0); // At least some succeeded
    assert!(final_room.tile_count() <= 10); // No duplicates
    assert!(final_room.current_phase() == ProtocolPhase::P0 || 
            final_room.current_phase() == ProtocolPhase::P1);
}
```

### Test 4: Memory Exhaustion Protection
```rust
#[test]
fn test_deadband_room_memory_exhaustion_handling() {
    let mut room = DeadbandRoom::new();
    let mut submitted = 0;
    
    // Attempt to submit extremely large tiles
    for i in 0..100 {
        let large_content = "x".repeat(10_000_000); // 10MB per tile
        let tile = TrainingTile {
            protocol_phase: ProtocolPhase::P0,
            content: large_content,
            sentiment: 0.0,
            metadata: HashMap::new(),
        };
        
        match room.submit_tile(tile) {
            Ok(_) => submitted += 1,
            Err(RoomError::ResourceExhausted(_)) => {
                // Expected after certain point
                break;
            }
            Err(e) => panic!("Unexpected error: {:?}", e),
        }
        
        // Room should protect itself from OOM
        assert!(room.memory_usage() < 1_000_000_000); // Under 1GB
    }
    
    // Should have accepted some tiles but protected itself
    assert!(submitted > 0);
    assert!(submitted < 100);
    println!("Accepted {} large tiles before protection kicked in", submitted);
}
```

### Test 5: State Recovery After Protocol Error
```rust
#[test]
fn test_deadband_room_state_recovery_after_error() {
    let mut room = DeadbandRoom::new();
    
    // 1. Submit valid P0 tile
    let tile_p0 = TrainingTile {
        protocol_phase: ProtocolPhase::P0,
        content: "Valid P0".to_string(),
        sentiment: -0.8,
        metadata: HashMap::new(),
    };
    assert!(room.submit_tile(tile_p0).is_ok());
    assert_eq!(room.current_phase(), ProtocolPhase::P1);
    
    // 2. Attempt invalid P2 (should fail)
    let tile_p2_invalid = TrainingTile {
        protocol_phase: ProtocolPhase::P2,
        content: "Invalid P2".to_string(),
        sentiment: 0.5,
        metadata: HashMap::new(),
    };
    assert!(room.submit_tile(tile_p2_invalid).is_err());
    
    // 3. Room should still accept valid P1 after error
    let tile_p1 = TrainingTile {
        protocol_phase: ProtocolPhase::P1,
        content: "Valid P1 after error".to_string(),
        sentiment: 0.3,
        metadata: HashMap::new(),
    };
    assert!(room.submit_tile(tile_p1).is_ok());
    assert_eq!(room.current_phase(), ProtocolPhase::P2);
    
    // 4. Now P2 should work
    let tile_p2_valid = TrainingTile {
        protocol_phase: ProtocolPhase::P2,
        content: "Valid P2 after recovery".to_string(),
        sentiment: 0.7,
        metadata: HashMap::new(),
    };
    assert!(room.submit_tile(tile_p2_valid).is_ok());
    
    // Verify final state
    assert_eq!(room.tile_count(), 3);
    assert!(room.is_complete());
}
```
