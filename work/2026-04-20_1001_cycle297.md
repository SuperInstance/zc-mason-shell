```python
def test_concurrent_agent_tile_submission(self):
    """Test DeadbandRoom handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(room_id="concurrent_test")
    
    # Simulate 3 agents submitting tiles at same cycle
    agent_tiles = [
        {"agent_id": "agent_1", "priority": 0, "action": "avoid_rock"},
        {"agent_id": "agent_2", "priority": 1, "action": "find_channel"},
        {"agent_id": "agent_3", "priority": 0, "action": "map_negative"}
    ]
    
    # Submit all tiles
    results = []
    for tile in agent_tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all tiles processed without corruption
    assert len(results) == 3
    assert all(r["status"] == "accepted" for r in results)
    
    # Verify P0 tiles processed before P1 (Deadband Protocol)
    p0_tiles = [t for t in room.get_tile_history() if t["priority"] == 0]
    p1_tiles = [t for t in room.get_tile_history() if t["priority"] == 1]
    assert len(p0_tiles) >= 2  # Both P0 tiles should be present
    assert len(p1_tiles) == 1
```

```python
def test_priority_inversion_rejection(self):
    """Test that P2 actions are rejected when P0 mapping incomplete."""
    room = DeadbandRoom(room_id="priority_test")
    
    # Submit P2 tile first (violates protocol)
    p2_result = room.submit_tile({
        "agent_id": "eager_agent",
        "priority": 2,
        "action": "optimize_path",
        "channel_id": "unmapped_channel"
    })
    
    # Should be rejected
    assert p2_result["status"] == "rejected"
    assert "P0 mapping required" in p2_result["message"]
    
    # Now submit proper P0 tile
    p0_result = room.submit_tile({
        "agent_id": "eager_agent",
        "priority": 0,
        "action": "map_negative_space"
    })
    
    assert p0_result["status"] == "accepted"
    
    # Now P2 should work
    p2_retry = room.submit_tile({
        "agent_id": "eager_agent",
        "priority": 2,
        "action": "optimize_path",
        "channel_id": "safe_channel_1"
    })
    
    assert p2_retry["status"] == "accepted"
```

### Test 3: Malformed Tile Payload Recovery
```python
def test_malformed_tile_recovery(self):
    """Test room handles JSON structure violations gracefully."""
    room = DeadbandRoom(room_id="malformed_test")
    
    # Missing required fields
    bad_tiles = [
        {"priority": 0},  # Missing agent_id and action
        {"agent_id": "test_agent"},  # Missing priority and action
        {"action": "test", "extra": "field"},  # Missing agent_id and priority
        None,  # Null payload
        "not_a_dict",  # String instead of dict
    ]
    
    expected_errors = []
    for tile in bad_tiles:
        try:
            result = room.submit_tile(tile)
            # If we get here, tile should be rejected
            assert result["status"] == "rejected"
            expected_errors.append(result["message"])
        except (TypeError, AttributeError) as e:
            # Room should not crash on bad input
            expected_errors.append(str(e))
    
    # Room state should still be valid
    assert room.room_id == "malformed_test"
    assert room.get_cycle_count() > 0
    
    # Valid tile should still work after bad ones
    valid_result = room.submit_tile({
        "agent_id": "recovery_agent",
        "priority": 0,
        "action": "test_recovery"
    })
    
    assert valid_result["status"] == "accepted"
```

### Test 5: Partial Network Failure Handling
```python
def test_partial_submission_failure(self):
    """Test behavior when tile submission partially fails."""
    room = DeadbandRoom(room_id="network_test")
    
    # Simulate network partition mid-processing
    original_process = room._process_tile_internal
    
    def failing_process(tile):
        """Simulate network failure after validation but before commit."""
        # Validate tile
        if not tile.get("agent_id"):
            return {"status": "rejected", "message": "Missing agent_id"}
        
        # Simulate network failure
        if tile.get("action") == "cause_failure":
            raise ConnectionError("Network partition simulated")
        
        # Normal processing
        return original_process(tile)
    
    room._process_tile_internal = failing_process
    
    # Tile that causes failure
    try:
        result = room.submit_tile({
            "agent_id": "network_agent",
            "priority": 0,
            "action": "cause_failure"
        })
        # Should not reach here
        assert False, "Expected ConnectionError"
    except ConnectionError:
        # Expected failure
        pass
    
    # Room should maintain consistency - no partial state
    assert len(room.get_tile_history()) == 0
    
    # Normal tile should still work
    normal_result = room.submit_tile({
        "agent_id": "network_agent",
        "priority": 0,
        "action": "recover_after_failure"
    })
    
    assert normal_result["status"] == "accepted"
    assert len(room.get_tile_history()) == 1
```
