```python
def test_concurrent_p0_violations(self):
    """Test room handling when multiple agents submit P0-violating tiles simultaneously."""
    room = DeadbandRoom()
    
    # Simulate concurrent violations from different agents
    violation_tiles = [
        {"agent_id": "agent1", "phase": "P2", "content": "Optimizing without P0"},  # Skip P0
        {"agent_id": "agent2", "phase": "P2", "content": "Another P0 skip"},  # Another skip
        {"agent_id": "agent3", "phase": "P1", "content": "Valid P1 but after violations"}
    ]
    
    results = []
    for tile in violation_tiles:
        try:
            result = room.process_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify room maintains consistency
    assert room.get_state()["violation_count"] == 2
    assert "agent3" in room.get_state()["valid_agents"]
    assert len(results) == 3
```

### Test 3: Network Partition Simulation
```python
def test_partial_tile_submission_with_timeout(self):
    """Test protocol behavior when tile submission times out mid-process."""
    room = DeadbandRoom()
    
    # Start a valid P0 tile
    tile = {
        "agent_id": "network_agent",
        "phase": "P0",
        "content": "Mapping negative space",
        "requires_ack": True  # Simulate synchronous requirement
    }
    
    # Simulate timeout after partial processing
    original_process = room._process_phase
    call_count = 0
    
    def timeout_after_one_call(*args, **kwargs):
        nonlocal call_count
        call_count += 1
        if call_count == 1:
            raise TimeoutError("Network partition simulated")
        return original_process(*args, **kwargs)
    
    room._process_phase = timeout_after_one_call
    
    # Should handle timeout gracefully
    try:
        result = room.process_tile(tile)
        # If we get here, room recovered from timeout
        assert result.get("recovered", False) == True
    except TimeoutError:
        # Room didn't handle timeout - this is a failure
        assert False, "Room should handle network timeouts gracefully"
    
    # Verify room state is consistent despite timeout
    assert room.get_state()["last_agent"] == "network_agent"
```

```python
    test_cases = [
        # Valid JSON, invalid phase progression
        {"agent_id": "agent1", "phase": "P2", "content": "", "previous_phase": "P0"},

        # Valid JSON, missing required context
        {"agent_id": "agent3", "phase": "P0", "content": "test", "context": {}},
        # Valid JSON, circular references (simulated)
        {"agent_id": "agent4", "phase": "P1", "content": "test", "reference": {"self_ref": None}}
    ]
    
    for i, tile in enumerate(test_cases):
        # Create circular reference for last test case

            tile["reference"]["self_ref"] = tile["reference"]
        
        result = room.process_tile(tile)
        
        # Room should reject semantically invalid tiles
        assert "error" in result or "invalid" in str(result).lower()
        assert room.get_state()["rejection_count"] > 0
```

```python
def test_p1_p2_oscillation_detection(self):
    """Test detection of infinite oscillation between P1 and P2."""
    room = DeadbandRoom()
    
    # Simulate agent oscillating between P1 and P2
    oscillation_pattern = [
        {"agent_id": "osc_agent", "phase": "P1", "content": "Channel A"},
        {"agent_id": "osc_agent", "phase": "P2", "content": "Optimize A"},
        {"agent_id": "osc_agent", "phase": "P1", "content": "Channel B"},  # New channel
        {"agent_id": "osc_agent", "phase": "P2", "content": "Optimize B"},
        {"agent_id": "osc_agent", "phase": "P1", "content": "Channel A"},  # Back to A
        {"agent_id": "osc_agent", "phase": "P2", "content": "Optimize A"},
    ]
    
    results = []
    for tile in oscillation_pattern:
        result = room.process_tile(tile)
        results.append(result)
    
    # Room should detect oscillation pattern
    state = room.get_state()
    assert "oscillation_detected" in state
    assert state["oscillation_detected"] == True
    
    # Last result should indicate oscillation was detected
    assert "stabilize" in results[-1] or "reset" in results[-1]
    
    # Verify room can recover from oscillation
    recovery_tile = {
        "agent_id": "osc_agent",
        "phase": "P0",  # Go back to P0 to break cycle
        "content": "Breaking oscillation"
    }
    recovery_result = room.process_tile(recovery_tile)
    assert recovery_result.get("phase") == "P0"
```
