### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(preset="deadband_basic")
    
    # Simulate 5 agents submitting tiles at same time
    tiles = []
    for i in range(5):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P0",
            "content": f"Negative space mapping {i}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles concurrently (simulated with threads)
    results = []
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append(("success", result))
        except Exception as e:
            results.append(("error", str(e)))
    
    threads = []
    for tile in tiles:
        thread = threading.Thread(target=submit_tile, args=(tile,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all tiles were processed
    success_count = sum(1 for r in results if r[0] == "success")
    self.assertEqual(success_count, 5, "All concurrent submissions should succeed")
    
    # Verify room state reflects all submissions
    self.assertEqual(len(room.state["processed_tiles"]), 5)
```

### Test 2: State Corruption Recovery
```python
def test_state_corruption_recovery(self):
    """Test room recovery when state file becomes corrupted."""
    room = DeadbandRoom(preset="deadband_basic")
    
    # Submit a valid tile first
    valid_tile = {
        "agent_id": "test_agent",
        "phase": "P1",
        "content": "Safe channel identified",
        "timestamp": time.time()
    }
    room.submit_tile(valid_tile)
    
    # Corrupt the state file (simulate disk corruption)
    state_path = room.state_file_path
    with open(state_path, 'w') as f:
        f.write("CORRUPTED_JSON{invalid")
    
    # Attempt to load room - should recover gracefully
    try:
        recovered_room = DeadbandRoom(preset="deadband_basic")
        # Should either initialize fresh state or restore from backup
        self.assertIsNotNone(recovered_room.state)
        self.assertIn("processed_tiles", recovered_room.state)
    except Exception as e:
        self.fail(f"Room should handle corrupted state gracefully: {e}")
```

### Test 3: Protocol Phase Skipping Detection
```python
def test_protocol_phase_skipping_detection(self):
    """Test that room rejects tiles attempting to skip P0→P2 directly."""
    room = DeadbandRoom(preset="deadband_strict")
    
    # Agent attempts to submit P2 tile without P0/P1 history
    skipping_tile = {
        "agent_id": "impatient_agent",
        "phase": "P2",
        "content": "Optimized path found!",
        "timestamp": time.time()
    }
    
    # Should be rejected with appropriate error
    with self.assertRaises(ProtocolViolationError) as context:
        room.submit_tile(skipping_tile)
    
    self.assertIn("P0", str(context.exception))
    self.assertIn("required", str(context.exception))
    
    # Verify room state unchanged
    self.assertEqual(len(room.state["processed_tiles"]), 0)
```

### Test 4: Resource Exhaustion Handling
```python
def test_resource_exhaustion_handling(self):
    """Test room behavior when system resources are exhausted."""
    room = DeadbandRoom(preset="deadband_basic")
    
    # Simulate memory pressure by mocking psutil
    original_virtual_memory = psutil.virtual_memory
    psutil.virtual_memory = lambda: type('obj', (object,), {
        'percent': 99.8,  # 99.8% memory used
        'available': 1024 * 1024  # 1MB available
    })()
    
    try:
        tile = {
            "agent_id": "test_agent",
            "phase": "P0",
            "content": "Testing under memory pressure",
            "timestamp": time.time()
        }
        
        # Room should handle gracefully, not crash
        result = room.submit_tile(tile)
        
        # Either process successfully or queue with warning
        self.assertIn(result["status"], ["processed", "queued"])
        if result["status"] == "queued":
            self.assertIn("resource_warning", result)
            
    finally:
        psutil.virtual_memory = original_virtual_memory  # Restore
```
