```python
def test_concurrent_tile_submission(self):
    """Test room handles simultaneous submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Negative space map from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (in rapid succession)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All should be accepted if within capacity
    accepted = sum(1 for r in results if r["status"] == "accepted")
    self.assertGreaterEqual(accepted, 2)  # At least 2 should succeed
    
    # Room state should reflect submissions
    self.assertEqual(len(room.tiles), accepted)
```

### Test 3: Storage Capacity Limits
```python
def test_storage_capacity_limits(self):
    """Test room behavior when hitting tile storage limits."""
    room = DeadbandRoom()
    max_tiles = room.config.get("max_tiles", 1000)
    
    # Fill room to capacity
    for i in range(max_tiles):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P0",
            "content": f"Test tile {i}",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    self.assertEqual(len(room.tiles), max_tiles)
    
    # Attempt to submit one more tile
    overflow_tile = {
        "agent_id": "overflow_agent",
        "phase": "P1",
        "content": "This should trigger capacity limit",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(overflow_tile)
    
    # Room should either reject or implement LRU eviction
    if result["status"] == "rejected":
        self.assertIn("capacity", result["reason"].lower())
        self.assertEqual(len(room.tiles), max_tiles)
    else:
        # If accepted, room should have evicted oldest tile
        self.assertEqual(len(room.tiles), max_tiles)
        # Verify oldest tile was removed
        oldest_agent = f"agent_0"
        self.assertNotIn(oldest_agent, [t["agent_id"] for t in room.tiles])
```

```python
def test_p2_without_p0_p1(self):
    """Test room detects and rejects P2 optimization without P0/P1 completion."""
    room = DeadbandRoom()
    
    # Agent tries to submit P2 tile without prior P0/P1
    p2_tile = {
        "agent_id": "impatient_agent",
        "phase": "P2",
        "content": "Trying to optimize without mapping negative space",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(p2_tile)
    self.assertEqual(result["status"], "rejected")
    self.assertIn("protocol violation", result["reason"].lower())
    
    # Now submit proper P0 tile
    p0_tile = {
        "agent_id": "impatient_agent",
        "phase": "P0",
        "content": "Mapping negative space properly",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(p0_tile)
    self.assertEqual(result["status"], "accepted")
    
    # Submit P1 tile
    p1_tile = {
        "agent_id": "impatient_agent",
        "phase": "P1",
        "content": "Finding safe channels",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(p1_tile)
    self.assertEqual(result["status"], "accepted")
    
    # Now P2 should be accepted
    p2_tile_2 = {
        "agent_id": "impatient_agent",
        "phase": "P2",
        "content": "Optimizing within channels",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(p2_tile_2)
    self.assertEqual(result["status"], "accepted")
```

```python
def test_corrupted_sentiment_data(self):
    """Test room handles malformed sentiment data from NPC interactions."""
    room = DeadbandRoom()
    
    # Tile with valid structure but corrupted sentiment field
    corrupted_tile = {
        "agent_id": "npc_interaction_agent",
        "phase": "P1",
        "content": "Interaction with NPC in holodeck",
        "timestamp": time.time(),
        "metadata": {
            "npc_sentiment": "NOT_A_VALID_SENTIMENT",  # Corrupted value
            "interaction_id": "npc_123"
        }
    }
    
    result = room.submit_tile(corrupted_tile)
    
    # Room should either reject or sanitize the data
    if result["status"] == "rejected":
        self.assertIn("sentiment", result["reason"].lower())
    else:
        # If accepted, sentiment should be normalized or defaulted
        stored_tile = room.tiles[-1]
        sentiment = stored_tile.get("metadata", {}).get("npc_sentiment")
        self.assertIn(sentiment, ["positive", "neutral", "negative", "unknown"])
    
    # Test with missing sentiment but valid interaction
    valid_tile = {
        "agent_id": "npc_interaction_agent",
        "phase": "P1",
        "content": "Another NPC interaction",
        "timestamp": time.time(),
        "metadata": {
            "interaction_id": "npc_456"
            # No sentiment field
        }
    }
    
    result = room.submit_tile(valid_tile)
    self.assertEqual(result["status"], "accepted")
```
