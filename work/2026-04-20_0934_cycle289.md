## 5 Additional Edge-Case Tests

### Test 1: Consecutive P0 Tiles with Contradictory Constraints
**Scenario:** Agent submits two P0 tiles where the second contradicts the first (e.g., "don't use GPU" then "must use GPU").
**Expected:** Room should accept both as valid P0 observations but flag potential contradiction in metadata.
**Rationale:** Real agents might observe conflicting negative spaces; the room should log this without crashing.

```python
def test_consecutive_p0_tiles_with_contradiction(self, deadband_room):
    """Test that contradictory P0 tiles are accepted but flagged."""
    tile1 = {
        "priority": 0,
        "content": "P0: Do not use GPU for inference under 10ms latency",
        "agent_id": "test_agent_1"
    }
    tile2 = {
        "priority": 0, 
        "content": "P0: Must use GPU for all tensor operations",
        "agent_id": "test_agent_2"
    }
    
    result1 = deadband_room.submit_tile(tile1)
    result2 = deadband_room.submit_tile(tile2)
    
    assert result1["status"] == "accepted"
    assert result2["status"] == "accepted"
    # Room should add contradiction flag to metadata
    assert "contradiction_check" in result2.get("metadata", {})
```

### Test 2: P2 Tile Submitted Before Any P1
**Scenario:** Agent tries to submit a P2 (optimization) tile without establishing P1 (safe channels) first.
**Expected:** Room should reject with specific error about protocol violation.
**Rationale:** Deadband Protocol requires strict priority order; skipping P1 breaks the doctrine.

```python
def test_p2_tile_without_p1_channel(self, deadband_room):
    """Test rejection of P2 tile when no P1 channel exists."""
    # Submit only P0 tile first
    deadband_room.submit_tile({
        "priority": 0,
        "content": "P0: Avoid memory leaks in tensor operations",
        "agent_id": "test_agent"
    })
    
    # Try to submit P2 directly
    p2_tile = {
        "priority": 2,

    with pytest.raises(ValueError) as exc_info:
```

### Test 5: Concurrent Tile Submissions from Multiple Agents
**Scenario:** Two agents submit tiles simultaneously to the same room instance.
**Expected:** Room should handle concurrent submissions without data corruption.
**Rationale:** Real fleet usage involves multiple agents; thread safety is essential.

```python
def test_concurrent_tile_submissions(self, deadband_room):
    """Test that concurrent submissions don't corrupt internal state."""
    import threading
    
    results = []
    errors = []
    
    def submit_tile_thread(agent_id, priority):
        try:
            tile = {
                "priority": priority,
                "content": f"Test from {agent_id}",
                "agent_id": agent_id
            }
            result = deadband_room.submit_tile(tile)
            results.append((agent_id, result["status"]))
        except Exception as e:
            errors.append((agent_id, str(e)))
    
    # Create multiple threads
    threads = []
    for i in range(5):
        t = threading.Thread(
            target=submit_tile_thread,
            args=(f"agent_{i}", i % 3)  # Mix of P0, P1, P2
        )
        threads.append(t)
        t.start()
    
    # Wait for all threads
    for t in threads:
        t.join()
    
    # Verify no errors and all submissions processed
    assert len(errors) == 0
    assert len(results) == 5
    assert all(status == "accepted" for _, status in results)
    
    # Verify tile count matches
    assert deadband_room.get_tile_count() == 5
```
