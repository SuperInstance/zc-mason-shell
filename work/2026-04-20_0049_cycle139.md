### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(capacity=10)
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (simulated)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results if isinstance(r, dict))
    assert room.get_tile_count() == len(agent_ids)
```

### Test 2: Malformed Tile Structure Validation
```python
def test_malformed_tile_rejection(self):
    """Test room rejection of tiles with invalid structure."""
    room = DeadbandRoom()
    
    malformed_tiles = [
        {},  # Empty tile
        {"agent_id": "test_agent"},  # Missing required fields
        {"agent_id": "test", "phase": "P5", "content": "test"},  # Invalid phase
        {"agent_id": "test", "phase": "P1", "content": ""},  # Empty content
        {"agent_id": "test", "phase": "P1", "content": "x" * 10001},  # Content too large
        {"agent_id": "", "phase": "P1", "content": "test"},  # Empty agent_id
    ]
    
    for tile in malformed_tiles:
        with pytest.raises((ValueError, TypeError)):
            room.submit_tile(tile)
    
    # Room should remain empty
    assert room.get_tile_count() == 0
```

### Test 3: Protocol State Timeout Handling
```python
def test_protocol_state_timeout(self):
    """Test agents timing out in P0/P1 states."""
    room = DeadbandRoom(state_timeout=60)  # 60 second timeout
    
    # Agent gets stuck in P0
    agent_state = room.initialize_agent("stuck_agent")
    assert agent_state["phase"] == "P0"
    
    # Simulate timeout by advancing internal clock
    room._last_state_update = time.time() - 120  # 2 minutes ago
    
    # Attempt transition should fail or reset
    with pytest.raises(StateTimeoutError):
        room.transition_phase("stuck_agent", "P1")
    
    # Agent should be marked as timed out
    assert room.get_agent_state("stuck_agent")["status"] == "timed_out"
```

### Test 4: Resource Exhaustion During Training
```python
def test_resource_exhaustion_handling(self):
    """Test room behavior when system resources are exhausted."""
    room = DeadbandRoom(max_memory_mb=10)  # Very low memory limit
    
    # Fill room with tiles until near limit
    for i in range(100):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P1",
            "content": "x" * 1000,  # 1KB per tile
            "timestamp": time.time()
        }
        try:
            room.submit_tile(tile)
        except ResourceExhaustedError:
            break
    
    # Verify room stopped accepting tiles before crash
    tile_count = room.get_tile_count()
    assert tile_count < 100  # Should have stopped early
    assert room.get_status()["memory_usage_mb"] <= 10
    
    # Attempt training should fail gracefully
    with pytest.raises(ResourceExhaustedError):
        room.train_ensign()
```

### Test 5: Network Partition Simulation
```python
def test_network_partition_recovery(self):
    """Test room recovery after network partition."""
    room = DeadbandRoom()
    
    # Normal operation
    tile = {
        "agent_id": "test_agent",
        "phase": "P1",
        "content": "Before partition",
        "timestamp": time.time()
    }
    room.submit_tile(tile)
    
    # Simulate network partition
    room._network_connected = False
    
    # Tile submission during partition should queue
    partitioned_tile = {
        "agent_id": "test_agent",
        "phase": "P2",
        "content": "During partition",
        "timestamp": time.time()
    }
    
    # Should not raise, but queue internally
    room.submit_tile(partitioned_tile)
    assert len(room._queued_tiles) == 1
    
    # Restore connectivity
    room._network_connected = True
    room._process_queued_tiles()
    
    # Verify both tiles are present
    assert room.get_tile_count() == 2
    assert room._queued_tiles == []
```
