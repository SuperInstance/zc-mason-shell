### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test that concurrent tile submissions from multiple agents don't corrupt state."""
    room = DeadbandRoom(room_id="race_test", capacity=100)
    
    # Simulate 10 agents submitting tiles simultaneously
    tiles = [
        Tile(agent_id=f"agent_{i}", content=f"test_content_{i}", cycle=i)
        for i in range(10)
    ]
    
    # Submit all tiles concurrently (simulated with threads)
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            room.submit_tile(tile)
            results.append(True)
        except Exception as e:
            results.append(False)
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all tiles were accepted
    assert all(results), "Some concurrent tile submissions failed"
    assert len(room.get_tiles()) == 10, "Tile count mismatch after concurrent submissions"
```

```python
def test_memory_pressure_large_tiles(self):
    """Test room behavior when tiles exceed typical memory boundaries."""
    room = DeadbandRoom(room_id="memory_test", capacity=10)
    
    # Generate tile with content size that might stress memory (10MB simulated)
    large_content = "x" * (10 * 1024 * 1024)  # 10MB string
    
    tile = Tile(
        agent_id="memory_agent",
        content=large_content,
        cycle=1,
        metadata={"size": len(large_content)}
    )
    
    # Room should handle large content appropriately
    # Options: accept, reject, or compress
    try:
        accepted = room.submit_tile(tile)
        
        if accepted:
            # If accepted, verify it's stored correctly
            stored_tile = room.get_tiles()[0]
            assert len(stored_tile.content) <= len(large_content)
            # Content might be compressed or truncated
        else:
            # If rejected, error should be informative
            assert room.get_last_error() is not None
            assert "memory" in room.get_last_error().lower() or \
                   "size" in room.get_last_error().lower()
            
    except MemoryError:
        # System-level memory error should be caught and handled
        assert room.get_agent_state("memory_agent").last_error is not None
```

### Test 4: Protocol Violation Recovery (Skipping P0 to P2)

```python
def test_protocol_violation_recovery(self):
    """Test that skipping P0 (negative space) and going directly to P2 triggers recovery."""
    room = DeadbandRoom(room_id="protocol_test", capacity=100)
    agent_id = "impatient_agent"
    
    # Create agent state that attempts to skip P0
    agent_state = AgentState(
        agent_id=agent_id,
        current_step="P2",  # Skipping P0 and P1!
        completed_steps=[],
        last_error=None
    )
    
    # Room should detect protocol violation
    try:
        result = room.execute_protocol_step(agent_state, step="P2")
        # If execution continues, it should be with corrective action
        assert result.get("corrective_action") is True
        assert result.get("forced_step") == "P0"
    except ProtocolViolationError as e:
        # Expected error for strict protocol enforcement
        assert "P0" in str(e)
        assert "required" in str(e).lower()
    
    # Verify agent state updated with violation record
    updated_state = room.get_agent_state(agent_id)
    assert updated_state.protocol_violations > 0
    assert updated_state.current_step == "P0"  # Reset to beginning
```

### Test 5: Cross-Cycle State Persistence Under Load
```python
def test_cross_cycle_state_persistence(self):
    """Test that agent state persists correctly across many cycles with load."""
    room = DeadbandRoom(room_id="persistence_test", capacity=1000)
    agent_id = "long_running_agent"
    
    # Simulate 100 cycles of activity
    for cycle in range(100):
        # Vary the protocol step each cycle
        step = ["P0", "P1", "P2"][cycle % 3]
        
        # Get or create agent state
        agent_state = room.get_agent_state(agent_id)
        if agent_state is None:
            agent_state = AgentState(agent_id=agent_id, current_step=step, cycle=cycle)
        
        # Execute protocol step
        result = room.execute_protocol_step(agent_state, step=step)
        
        # Submit a tile every 10 cycles
        if cycle % 10 == 0:
            tile = Tile(
                agent_id=agent_id,
                content=f"Cycle {cycle} report",
                cycle=cycle,
                step_completed=step
            )
            room.submit_tile(tile)
        
        # Update agent state
        agent_state.current_step = step
        agent_state.cycle = cycle
        agent_state.last_result = result
        room.update_agent_state(agent_state)
    
    # Verify final state
    final_state = room.get_agent_state(agent_id)
    assert final_state.cycle == 99
    assert final_state.current_step in ["P0", "P1", "P2"]
    
    # Verify tile count (should be 10 tiles: cycles 0, 10, 20, ..., 90)
    tiles = room.get_tiles(agent_id=agent_id)
    assert len(tiles) == 10
    
    # Verify tiles are in correct order
    for i, tile in enumerate(tiles):
        expected_cycle = i * 10
        assert tile.cycle == expected_cycle
        assert f"Cycle {expected_cycle}" in tile.content
```
