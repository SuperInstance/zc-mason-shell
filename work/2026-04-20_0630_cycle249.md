### Test 3: Concurrent Agent Tile Collision
```python
def test_concurrent_tile_submissions(self):
    """P1: Room should handle simultaneous submissions from multiple agents"""
    room = DeadbandRoom()
    num_agents = 10
    
    def submit_tile_async(agent_num):
        tile = {
            "agent_id": f"agent_{agent_num}",
            "content": f"Test content from agent {agent_num}",
            "phase": agent_num % 3,
            "timestamp": time.time()
        }
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(submit_tile_async, i) for i in range(num_agents)]
        results = [f.result() for f in futures]
    
    # All submissions should be processed
    self.assertEqual(len(results), num_agents)
    
    # Check no data corruption occurred
    accepted_tiles = [r for r in results if r["accepted"]]
    self.assertGreater(len(accepted_tiles), 0)
    
    # Verify tile ordering or conflict resolution
    room_state = room.get_state()
    self.assertEqual(len(room_state["tiles"]), len(accepted_tiles))
```

### Test 5: State Corruption Recovery
```python
def test_corrupted_state_recovery(self):
    """P1: Room should recover from corrupted internal state"""
    room = DeadbandRoom()
    
    # First, establish normal operation
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"Normal tile {i}",
            "phase": i,
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    initial_state = room.get_state()
    initial_tile_count = len(initial_state["tiles"])
    
    # Simulate state corruption (direct attribute manipulation)
    # This tests the room's resilience to internal data corruption
    try:
        # Corrupt the tiles list
        room.tiles = "corrupted_data"
        
        # Attempt normal operation - should trigger recovery
        recovery_tile = {
            "agent_id": "recovery_agent",
            "content": "Recovery attempt",
            "phase": 0,
            "timestamp": time.time()
        }
        
        result = room.submit_tile(recovery_tile)
        
        # Room should either recover or fail gracefully
        if result["accepted"]:
            # Recovery succeeded
            recovered_state = room.get_state()
            self.assertIsInstance(recovered_state["tiles"], list)
        else:
            # Graceful failure
            self.assertIn("recovery", result["reason"].lower())
            
    except Exception as e:
        # Room should not crash entirely
        self.fail(f"Room crashed on corrupted state: {e}")
    
    # Final sanity check
    final_state = room.get_state()
    self.assertIsInstance(final_state, dict)
    self.assertIn("tiles", final_state)
```
