### Test 3: Concurrent Tile Submission Race
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom()
    num_agents = 10
    tiles = []
    
    # Create tiles from different agents for same phase
    for i in range(num_agents):
        tiles.append({
            "phase": "P0",
            "content": f"Negative space mapping {i}",
            "agent_id": f"agent_{i}"
        })
    
    # Simulate concurrent submission
    import threading
    results = []
    
    def submit_tile(tile):
        results.append(room.submit_tile(tile))
    
    threads = []
    for tile in tiles:
        t = threading.Thread(target=submit_tile, args=(tile,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions processed
    assert len(results) == num_agents
    accepted = [r for r in results if r["status"] == "accepted"]
    assert len(accepted) > 0  # At least some should be accepted
    
    # Final state should be consistent
    final_count = room.get_tile_count()
    assert final_count == len(accepted)
    assert final_count <= num_agents  # Cannot exceed submissions
```

### Test 4: Resource Limit Enforcement
```python
def test_tile_storage_limits(self):
    """Room should enforce storage limits and handle overflow gracefully."""
    room = DeadbandRoom()
    
    # Configure small limit for testing
    room.MAX_TILES = 5
    
    # Fill room to capacity
    for i in range(5):
        tile = {
            "phase": "P0",
            "content": f"Test tile {i}",
            "agent_id": "test_agent"
        }
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    assert room.get_tile_count() == 5
    
    # Attempt to exceed limit
    overflow_tile = {
        "phase": "P1",
        "content": "Should be rejected",
        "agent_id": "test_agent"
    }
    result = room.submit_tile(overflow_tile)
    assert result["status"] == "rejected"
    assert "limit" in result["message"].lower()
    assert room.get_tile_count() == 5  # Unchanged
    
    # After reset, should accept new tiles
    room.reset()
    assert room.get_tile_count() == 0
    result = room.submit_tile(overflow_tile)
    assert result["status"] == "accepted"
    assert room.get_tile_count() == 1
```
