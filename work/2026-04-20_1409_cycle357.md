```python
# plato-torch/tests/test_deadband_room_edge.py
"""
Edge-case tests for DeadbandRoom.
These complement the core tests in test_deadband_room.py.
"""

import pytest
import numpy as np
from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.exceptions import DeadbandViolationError, ChannelExhaustedError

class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""

    def test_apply_deadband_with_nan_and_inf(self):
        """P0: Test deadband application with NaN and infinity in input vector."""
        room = DeadbandRoom(deadband=0.1, tolerance=[0.05, 0.05])
        # Input with NaN and inf
        test_vector = np.array([1.0, np.nan, np.inf, -np.inf, 2.0])
        # Expect NaN and inf to propagate or raise? Current behavior: may raise ValueError.
        with pytest.raises(ValueError):
            room.apply_deadband(test_vector)

    def test_repeated_find_safe_channels_state_mutation(self):
        """P1: Test that repeated calls to find_safe_channels do not corrupt internal state."""
        room = DeadbandRoom(deadband=0.5, tolerance=[0.1, 0.1])
        constraints = [{'min': 0, 'max': 10}, {'min': 0, 'max': 10}]
        # First call
        channels1 = room.find_safe_channels(constraints)
        assert len(channels1) > 0
        initial_state_hash = hash(str(room.safe_channels))
        # Second call with same constraints
        channels2 = room.find_safe_channels(constraints)
        # Should be identical (deterministic) and not mutate safe_channels unexpectedly
        assert channels1 == channels2
        assert hash(str(room.safe_channels)) == initial_state_hash

    def test_optimize_with_channel_exhaustion(self):
        """P2: Test optimization when channel is exhausted (no viable points)."""
        room = DeadbandRoom(deadband=0.2, tolerance=[0.01, 0.01])
        # Artificially set safe_channels to a single very narrow channel
        room.safe_channels = [{'min': 5.0, 'max': 5.001}]  # Near-zero width channel
        with pytest.raises(ChannelExhaustedError):
            room.optimize_within_channel(objective='maximize')

    def test_mixed_type_tolerance_array(self):
        """Test initialization with tolerance array containing mixed numeric types."""
        # This should be normalized to float or raise TypeError
        with pytest.raises(TypeError):
            DeadbandRoom(deadband=0.3, tolerance=[0.1, '0.2', 0.3])
        # With valid mixed but convertible types (int, float)
        room = DeadbandRoom(deadband=0.3, tolerance=[0.1, 2, 3.0])  # int 2 should convert
        assert all(isinstance(t, float) for t in room.tolerance)

    def test_tile_submission_with_corrupted_validation_token(self):
        """Test tile submission when the validation token is malformed (simulating network corruption)."""
        room = DeadbandRoom(deadband=0.4)
        mock_tile = Mock()
        mock_tile.to_dict.return_value = {'data': 'test', 'hash': 'abc123'}
        # Simulate a corrupted token (e.g., empty string, wrong format)
        with patch('plato_torch.rooms.deadband_room.validate_tile_token') as mock_validate:
            mock_validate.side_effect = ValueError("Invalid token format")
            result = room.submit_tile(mock_tile, validation_token='')
            # Expect submission to be rejected, return False
            assert result is False
```

## Rationale for Each Test
1. **NaN/Inf test**: Real-world data from sensors or simulations can produce NaN/inf. The deadband logic should handle or reject these explicitly.
2. **State mutation test**: Repeated calls should be idempotent for same inputs; internal `safe_channels` should not be corrupted.
3. **Channel exhaustion test**: When a channel is impossibly narrow, optimization should raise a clear error rather than silently fail.
4. **Mixed-type tolerance**: Tolerance arrays may be constructed from heterogeneous sources; type handling should be robust.
5. **Corrupted token test**: Network issues or malformed tokens should be gracefully rejected without crashing the room.
