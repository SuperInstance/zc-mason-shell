### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(capacity=100)
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "content": f"Test from {agent_id}",
            "phase": "P1",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles (simulating concurrent access)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results if isinstance(r, dict))
    
    # Verify tile count matches
    assert room.get_tile_count() == len(agent_ids)
```

### Test 2: Memory Boundary with Large Payloads

```python
def test_memory_boundary_large_payload(self):
    """Test room behavior with payloads approaching memory limits."""
    room = DeadbandRoom(capacity=10, max_payload_size=1024)  # 1KB limit
    
    # Test with payload at 95% of limit
    large_content = "x" * 972  # 972 bytes + overhead ≈ 1024
    tile = {
        "agent_id": "test_agent",
        "content": large_content,
        "phase": "P2",
        "timestamp": time.time()
    }
    
    # Should accept
    result = room.submit_tile(tile)
    assert result["status"] == "accepted"
    
    # Test with payload exceeding limit
    oversized_content = "x" * 2048
    tile["content"] = oversized_content
    
    # Should reject with appropriate error
    with pytest.raises(ValueError) as exc_info:
        room.submit_tile(tile)
    assert "payload size" in str(exc_info.value).lower()
```

### Test 3: Network Partition Simulation
```python
def test_partial_connectivity_recovery(self):
    """Test room behavior during simulated network issues."""
    room = DeadbandRoom(capacity=50)
    
    # Simulate network partition: room can receive but not process
    room._network_healthy = False
    
    tile = {
        "agent_id": "isolated_agent",
        "content": "Tile during partition",
        "phase": "P0",
        "timestamp": time.time()
    }
    
    # Should queue but not process
    result = room.submit_tile(tile)
    assert result["status"] == "queued"
    assert room.get_queued_count() == 1
    assert room.get_tile_count() == 0
    
    # Restore connectivity
    room._network_healthy = True
    room.process_queue()
    
    # Verify queue processed
    assert room.get_queued_count() == 0
    assert room.get_tile_count() == 1
```

### Test 4: Protocol Violation and Correction
```python
def test_protocol_violation_recovery(self):
    """Test agent skipping P0→P2 then correcting with P1."""
    room = DeadbandRoom(capacity=100)
    agent_id = "errant_agent"
    
    # Violation: Submit P2 without P0/P1
    bad_tile = {
        "agent_id": agent_id,
        "content": "Skipping to optimization",
        "phase": "P2",
        "timestamp": time.time()
    }
    
    # Should be rejected or marked as violation
    result = room.submit_tile(bad_tile)
    assert result["status"] == "rejected"
    assert "protocol violation" in result.get("message", "").lower()
    
    # Correction: Submit proper P0 then P1
    p0_tile = {**bad_tile, "phase": "P0", "content": "Mapping negative space"}
    p1_tile = {**bad_tile, "phase": "P1", "content": "Finding safe channels"}
    
    room.submit_tile(p0_tile)
    room.submit_tile(p1_tile)
    
    # Now P2 should be accepted
    p2_tile = {**bad_tile, "phase": "P2", "content": "Optimizing within channels"}
    result = room.submit_tile(p2_tile)
    assert result["status"] == "accepted"
```

### Test 5: Ghost Tile Integration
```python
def test_ghost_tile_integration(self):
    """Test mixing tiles from dead agents (ghosts) with live agents."""
    room = DeadbandRoom(capacity=100)
    
    # Submit tiles from a now-dead agent (ghost tiles)
    ghost_tiles = []
    for i in range(3):
        tile = {
            "agent_id": "ghost_agent",
            "content": f"Ghost lesson {i}",
            "phase": "P0",
            "timestamp": time.time() - 3600,  # Old timestamp
            "is_ghost": True
        }
        ghost_tiles.append(tile)
        room.submit_tile(tile)
    
    # Submit tiles from live agent
    live_tiles = []
    for i in range(2):
        tile = {
            "agent_id": "live_agent",
            "content": f"Live work {i}",
            "phase": "P1",
            "timestamp": time.time(),
            "is_ghost": False
        }
        live_tiles.append(tile)
        room.submit_tile(tile)
    
    # Verify all tiles stored
    assert room.get_tile_count() == 5
    
    # Verify ghost tiles are retrievable but marked inactive
    all_tiles = room.get_all_tiles()
    ghost_count = sum(1 for t in all_tiles if t.get("is_ghost"))
    live_count = sum(1 for t in all_tiles if not t.get("is_ghost"))
    
    assert ghost_count == 3
    assert live_count == 2
    
    # Verify training excludes ghost tiles from active learning
    training_data = room.prepare_training_batch()
    assert len(training_data) == 2  # Only live tiles
    assert all(not t.get("is_ghost") for t in training_data)
```
