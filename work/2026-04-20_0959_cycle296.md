### Test 1: Extreme Environment - No Safe Channels

```python
def test_p1_no_safe_channels(self):
    """Test P1 behavior when environment has zero safe channels."""
    room = DeadbandRoom(room_id="edge_extreme")
    # Mock environment where all positions are blocked
    room.environment = MockEnvironment(all_blocked=True)
    
    agent_state = {"phase": 1, "position": (0, 0)}
    result = room.process_agent_step(agent_state)
    
    # Should remain in P1, not progress to P2
    assert result["phase"] == 1
    assert "safe_channels" in result
    assert len(result["safe_channels"]) == 0
    assert "blocked_reason" in result  # Should explain why
```

### Test 2: Concurrent Agent Tile Collision
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(room_id="edge_concurrent")
    
    # Simulate two agents reaching P2 optimization simultaneously
    agent1_tile = {"agent_id": "agent1", "phase": 2, "optimization": "path_a"}
    agent2_tile = {"agent_id": "agent2", "phase": 2, "optimization": "path_b"}
    
    # Submit tiles in rapid succession (simulated)
    result1 = room.submit_tile(agent1_tile)
    result2 = room.submit_tile(agent2_tile)
    
    # Both should be accepted if they don't conflict
    assert result1["accepted"] is True
    assert result2["accepted"] is True
    # Room should track both submissions
    assert len(room.get_recent_tiles()) == 2
```

### Test 3: Tile Storage Limit Enforcement
```python
def test_tile_storage_limits(self):
    """Test behavior when tile storage reaches configured limit."""
    room = DeadbandRoom(room_id="edge_storage", max_tiles=5)
    
    # Fill storage to capacity
    for i in range(5):
        tile = {"agent_id": f"agent{i}", "phase": 1, "channel": f"channel{i}"}
        room.submit_tile(tile)
    
    # Attempt to submit one more tile
    overflow_tile = {"agent_id": "overflow", "phase": 2, "optimization": "test"}
    result = room.submit_tile(overflow_tile)
    
    # Should be rejected or trigger cleanup
    assert result["accepted"] is False
    assert "storage_full" in result.get("reason", "")
    # Verify oldest tile may be removed (if LRU implemented)
    assert len(room.get_recent_tiles()) <= 5
```

### Test 4: Malformed Tile Data Recovery
```python
def test_malformed_tile_recovery(self):
    """Test room handling of invalid tile structures."""
    room = DeadbandRoom(room_id="edge_malformed")
    
    test_cases = [
        {"invalid": "json"},  # Missing required fields
        {"agent_id": "test", "phase": "not_a_number"},  # Wrong type
        None,  # Null input
        "{invalid json string}",  # Bad JSON
    ]
    
    for malformed_tile in test_cases:
        result = room.submit_tile(malformed_tile)
        # Should reject with clear error, not crash
        assert result["accepted"] is False
        assert "error" in result
        assert "validation" in result["error"].lower() or "invalid" in result["error"].lower()
```

### Test 5: Protocol Violation and Recovery Path
```python
def test_p2_violation_recovery(self):
    """Test agent skipping P1 to P2, then recovering correctly."""
    room = DeadbandRoom(room_id="edge_recovery")
    
    # Agent attempts illegal P0→P2 skip
    agent_state = {"phase": 0, "attempted_phase": 2, "optimization": "premature"}
    result = room.process_agent_step(agent_state)
    
    # Should be rejected and reset to P1
    assert result["phase"] == 1
    assert result.get("violation") is True
    assert "must_complete_p1" in result.get("message", "")
    
    # Now provide valid P1 data
    agent_state = {"phase": 1, "environment_scan": {"safe_channels": ["channel1"]}}
    result = room.process_agent_step(agent_state)
    
    # Should accept and progress normally
    assert result["phase"] == 1 or result["phase"] == 2
    assert result.get("violation") is None
```
