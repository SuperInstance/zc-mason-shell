### Test 1: Concurrent Agent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # All agents join simultaneously
    for agent_id in agent_ids:
        room.join(agent_id)
    
    # Submit tiles concurrently (simulated with threads)
    import threading
    results = []
    
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        results.append((agent_id, result))
    
    threads = []
    for agent_id in agent_ids:
        thread = threading.Thread(target=submit_tile, args=(agent_id,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all submissions were processed
    self.assertEqual(len(results), len(agent_ids))
    for agent_id, result in results:
        self.assertTrue(result["accepted"], f"Tile from {agent_id} should be accepted")
    
    # Verify room state consistency
    self.assertEqual(room.get_tile_count(), len(agent_ids))
```

### Test 2: Malformed Tile Data (P0 Violation)

```python
def test_malformed_tile_rejection(self):
    """Test that room rejects tiles violating P0 constraints."""
    room = DeadbandRoom()
    room.join("test_agent")
    
    malformed_tiles = [
        # Missing required fields
        {"agent_id": "test_agent"},
        {"phase": "P1", "content": "test"},
        # Invalid phase skipping (P0 -> P2)
        {"agent_id": "test_agent", "phase": "P2", "content": "Skipped P1"},
        # Negative timestamp
        {"agent_id": "test_agent", "phase": "P1", "content": "test", "timestamp": -1},
        # Excessively large content
        {"agent_id": "test_agent", "phase": "P1", "content": "x" * 1000000},
        # Invalid agent ID (not joined)
        {"agent_id": "unknown_agent", "phase": "P1", "content": "test"},
    ]
    
    for i, tile in enumerate(malformed_tiles):
        with self.subTest(tile=i):
            result = room.submit_tile(tile)
            self.assertFalse(result["accepted"], f"Tile {i} should be rejected")
            self.assertIn("error", result)
            self.assertIn("P0", result["error"])  # Should reference P0 violation
```

```python
def test_room_capacity_limits(self):
    """Test that room enforces maximum agent capacity."""
    room = DeadbandRoom()
    max_capacity = room.MAX_AGENTS  # Assuming this attribute exists
    
    # Fill room to capacity
    for i in range(max_capacity):
        result = room.join(f"agent_{i}")
        self.assertTrue(result["success"], f"Should join agent_{i}")
    
    # Try to exceed capacity
    result = room.join("extra_agent")
    self.assertFalse(result["success"])
    self.assertIn("capacity", result.get("error", "").lower())
    
    # Ensure existing agents can still submit tiles
    for i in range(max_capacity):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P1",
            "content": f"Tile from agent_{i}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        self.assertTrue(result["accepted"], f"Agent_{i} should still submit")
    
    # After an agent leaves, new agent can join
    room.leave("agent_0")
    result = room.join("replacement_agent")
    self.assertTrue(result["success"], "Should join after capacity freed")
```

### Test 4: Network Failure Simulation
```python
def test_network_failure_recovery(self):
    """Test room behavior during simulated network failures."""
    room = DeadbandRoom()
    room.join("test_agent")
    
    # Simulate network failure during tile submission
    original_submit = room._submit_to_storage
    
    failed_attempts = []
    def failing_submit(tile):
        failed_attempts.append(tile)
        raise ConnectionError("Network failure")
    
    room._submit_to_storage = failing_submit
    
    tile = {
        "agent_id": "test_agent",
        "phase": "P1",
        "content": "Test tile",
        "timestamp": time.time()
    }
    
    # Should raise on network failure
    with self.assertRaises(ConnectionError):
        room.submit_tile(tile)
    
    self.assertEqual(len(failed_attempts), 1)
    
    # Restore connection and retry
    room._submit_to_storage = original_submit
    
    # Room should maintain state and allow resubmission
    result = room.submit_tile(tile)
    self.assertTrue(result["accepted"], "Should accept after recovery")
    
    # Verify tile persistence
    self.assertEqual(room.get_tile_count(), 1)
```

### Test 5: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Test that agents can recover from protocol violations."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    room.join(agent_id)
    
    # Submit valid P0 tile
    p0_tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Negative space mapped",
        "timestamp": time.time()
    }
    result = room.submit_tile(p0_tile)
    self.assertTrue(result["accepted"])
    
    # Attempt to skip to P2 (violation)
    p2_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Trying to optimize prematurely",
        "timestamp": time.time()
    }
    result = room.submit_tile(p2_tile)
    self.assertFalse(result["accepted"])
    self.assertIn("P1", result.get("error", ""))  # Should indicate missing P1
    
    # Submit proper P1 tile (recovery)
    p1_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Finding safe channels",
        "timestamp": time.time()
    }
    result = room.submit_tile(p1_tile)
    self.assertTrue(result["accepted"], "Should accept P1 after violation")
    
    # Now P2 should be accepted
    p2_tile["timestamp"] = time.time()
    result = room.submit_tile(p2_tile)
    self.assertTrue(result["accepted"], "Should accept P2 after proper sequence")
    
    # Verify complete protocol sequence
    tiles = room.get_agent_tiles(agent_id)
    phases = [t["phase"] for t in tiles]
    self.assertEqual(phases, ["P0", "P1", "P2"], "Should have complete sequence")
```
