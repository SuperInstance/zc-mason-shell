```python
def test_protocol_violation_direct_p2():
    """P0: Attempting P2 optimization before P1 channel finding should fail."""
    room = DeadbandRoom()
    agent_id = "test_agent_1"
    
    # Skip P0 and P1, try P2 directly
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.submit_action(agent_id, {"phase": "P2", "action": "optimize_path"})
    
    assert "P1 channels not established" in str(exc_info.value)
    assert room.get_agent_state(agent_id)["violations"] == 1
```

```python
def test_resource_exhaustion_high_concurrency():
    """P0: Room should handle 1000 concurrent agents without memory explosion."""
    room = DeadbandRoom()
    agent_ids = [f"load_agent_{i}" for i in range(1000)]
    
    # All agents submit P0 actions simultaneously
    for agent_id in agent_ids:
        room.submit_action(agent_id, {"phase": "P0", "action": "map_negative_space"})
    
    # Verify memory usage hasn't exploded
    import psutil
    process = psutil.Process()
    memory_mb = process.memory_info().rss / 1024 / 1024
    
    # Should stay under 500MB for 1000 agents
    assert memory_mb < 500, f"Memory usage {memory_mb}MB exceeds limit"
    
    # All agents should be in valid states
    for agent_id in agent_ids:
        state = room.get_agent_state(agent_id)
        assert state["current_phase"] == "P0_COMPLETE"
```

### Test 4: Concurrent Access Race Conditions
```python
def test_concurrent_access_race_condition():
    """P0: Multiple threads accessing same agent state should not corrupt data."""
    room = DeadbandRoom()
    agent_id = "race_agent"
    
    import threading
    results = []
    
    def submit_concurrent_action(thread_id):
        try:
            # All threads try to advance from P0 to P1
            room.submit_action(
                agent_id, 
                {"phase": "P1", "action": f"find_channels_{thread_id}"}
            )
            results.append(("success", thread_id))
        except Exception as e:
            results.append(("error", str(e), thread_id))
    
    # Start 10 concurrent threads
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_concurrent_action, args=(i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Only one thread should succeed, others should get concurrency errors
    successes = [r for r in results if r[0] == "success"]
    errors = [r for r in results if r[0] == "error"]
    
    assert len(successes) == 1, f"Expected 1 success, got {len(successes)}"
    assert len(errors) == 9, f"Expected 9 errors, got {len(errors)}"
    
    # Verify state is consistent
    state = room.get_agent_state(agent_id)
    assert state["current_phase"] in ["P0_COMPLETE", "P1_ACTIVE"]
```

### Test 5: Recovery From Corrupted/Crashed State
```python
def test_recovery_from_crashed_state():
    """P1: Room should recover gracefully when agent crashes mid-phase."""
    room = DeadbandRoom()
    agent_id = "crash_agent"
    
    # Start normal protocol
    room.submit_action(agent_id, {"phase": "P0", "action": "map_negative_space"})
    room.submit_action(agent_id, {"phase": "P1", "action": "find_channels"})
    
    # Simulate crash: agent disappears without cleanup
    # Room should detect stale agent after timeout
    import time
    time.sleep(room._stale_agent_timeout + 1)
    
    # New agent with same ID should be able to resume
    recovery_action = {
        "phase": "P1",
        "action": "recover_session",
        "session_token": room._generate_recovery_token(agent_id)
    }
    
    result = room.submit_action(agent_id, recovery_action)
    
    # Should recover previous P1 progress
    assert result["status"] == "recovered"
    assert "previous_channels" in result
    assert len(result["previous_channels"]) > 0
    
    # Should be able to continue from recovered state
    room.submit_action(agent_id, {"phase": "P2", "action": "optimize_path"})
    state = room.get_agent_state(agent_id)
    assert state["current_phase"] == "P2_ACTIVE"
```
