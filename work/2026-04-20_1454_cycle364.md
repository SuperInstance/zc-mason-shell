## Analysis of Existing Tests
The `DeadbandRoom` class implements the fleet's Deadband Protocol (P0: negative space, P1: safe channels, P2: optimization). Existing tests in `test_deadband_room.py` cover:
- Basic protocol adherence (P0, P1, P2 sequence)
- Input validation (rejects invalid actions)
- State transitions (room correctly moves between phases)
- Tile submission and validation

## Identified Edge Cases (P0 Mapping)
Based on reading the code, I identified these untested negative spaces and boundary conditions:

1. **Protocol Violation Recovery** – What happens when an agent submits a P2 action before P1? Does the room reject it, log it, and allow correction?
2. **Concurrent Agent Interference** – Multiple agents in the same room submitting conflicting phase actions. Does the room handle state locks or race conditions?
3. **Malformed Tile Payloads** – Tiles with missing required fields, incorrect types, or oversized data.
4. **Phase Timeout Scenarios** – If an agent stalls in P0 or P1 indefinitely, does the room have a timeout or escalation path?
5. **Resource Exhaustion** – Many tiles submitted rapidly; does the room handle memory/queue limits gracefully?

import pytest
from unittest.mock import patch, MagicMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.exceptions import ProtocolViolationError, TileValidationError

def test_protocol_sequence_violation_rejection():
    """P0: Room must reject P2 action submitted before P1 safe channel is established."""
    room = DeadbandRoom(room_id="test_edge_1")
    agent_id = "agent_alpha"
    
    # Simulate agent joining and immediately trying to optimize (P2) without P1
    room.join(agent_id)
    # Force room state to P0 (mapping negative space)
    room.current_phase[agent_id] = "P0"
    
    # Attempt P2 action
    invalid_action = {"phase": "P2", "action": "optimize", "payload": {"target": "efficiency"}}
    
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.submit_action(agent_id, invalid_action)
    
    assert "P2 not allowed before P1" in str(exc_info.value)
    # Verify room state remains P0 for this agent
    assert room.current_phase[agent_id] == "P0"

def test_concurrent_agent_phase_conflict():
    """P0: Room must handle multiple agents in different phases without cross-talk interference."""
    room = DeadbandRoom(room_id="test_edge_2")
    agent_a = "agent_a"
    agent_b = "agent_b"
    
    room.join(agent_a)
    room.join(agent_b)
    
    # Agent A in P1, Agent B in P0
    room.current_phase[agent_a] = "P1"
    room.current_phase[agent_b] = "P0"
    
    # Agent B tries to submit a P1 action (should be rejected)
    action_b = {"phase": "P1", "action": "explore_channel", "payload": {"channel": "test"}}
    
    with pytest.raises(ProtocolViolationError):
        room.submit_action(agent_b, action_b)
    
    # Agent A's P1 action should succeed
    action_a = {"phase": "P1", "action": "explore_channel", "payload": {"channel": "valid"}}
    result = room.submit_action(agent_a, action_a)
    assert result["status"] == "accepted"
    
    # Verify phases remain separate
    assert room.current_phase[agent_a] == "P1"
    assert room.current_phase[agent_b] == "P0"

def test_malformed_tile_validation():
    """P0: Room must reject tiles with missing fields, type mismatches, or oversized payloads."""
    room = DeadbandRoom(room_id="test_edge_3")
    agent_id = "agent_malformed"
    room.join(agent_id)
    room.current_phase[agent_id] = "P0"
    
    # Missing required 'action' field
    tile_missing_field = {"phase": "P0", "payload": {"note": "test"}}
    with pytest.raises(TileValidationError) as exc_info:
        room.submit_tile(agent_id, tile_missing_field)
    assert "Missing required field" in str(exc_info.value)
    
    # Incorrect type for 'phase' (int instead of str)
    tile_wrong_type = {"phase": 0, "action": "map", "payload": {}}
    with pytest.raises(TileValidationError):
        room.submit_tile(agent_id, tile_wrong_type)
    
    # Oversized payload (simulate > 10KB)
    oversized_payload = {"data": "x" * 15000}
    tile_oversized = {"phase": "P0", "action": "map", "payload": oversized_payload}
    with pytest.raises(TileValidationError):
        room.submit_tile(agent_id, tile_oversized)

def test_phase_timeout_escalation():
    """P0: Room should detect stalled agents and escalate or reset phase after timeout."""
    room = DeadbandRoom(room_id="test_edge_4", phase_timeout_seconds=0.1)  # Short timeout for test
    agent_id = "agent_stalled"
    room.join(agent_id)
    room.current_phase[agent_id] = "P0"
    room.phase_start_time[agent_id] = time.time() - 0.2  # Simulate started 0.2s ago
    
    # Attempt to submit action after timeout
    action = {"phase": "P0", "action": "map", "payload": {"obstacle": "rock"}}
    
    # Room should either reject or auto-advance phase
    with patch.object(room, '_escalate_phase') as mock_escalate:
        result = room.submit_action(agent_id, action)
        # Expect escalation call
        mock_escalate.assert_called_once_with(agent_id)
        # Or expect rejection with timeout error
        if result["status"] == "rejected":
            assert "phase timeout" in result["message"].lower()

def test_high_volume_tile_submission():
    """P0: Room must handle rapid tile submissions without resource exhaustion."""
    room = DeadbandRoom(room_id="test_edge_5", max_tiles_per_agent=100)
    agent_id = "agent_flood"
    room.join(agent_id)
    room.current_phase[agent_id] = "P0"
    
    # Submit tiles up to limit
    for i in range(100):
        tile = {
            "phase": "P0",
            "action": "map",
            "payload": {"obstacle": f"rock_{i}"},
            "timestamp": time.time()
        }
        result = room.submit_tile(agent_id, tile)
        assert result["status"] == "accepted"
    
    # 101st tile should be rejected or queued
    tile_overflow = {
        "phase": "P0",
        "action": "map",
        "payload": {"obstacle": "overflow"},
        "timestamp": time.time()
    }
    result = room.submit_tile(agent_id, tile_overflow)
    assert result["status"] == "rejected"
    assert "limit" in result["message"].lower()
    
    # Verify room still operational for other agents
    other_agent = "agent_normal"
    room.join(other_agent)
    room.current_phase[other_agent] = "P0"
    normal_tile = {"phase": "P0", "action": "map", "payload": {"obstacle": "test"}}
    result = room.submit_tile(other_agent, normal_tile)
    assert result["status"] == "accepted"
