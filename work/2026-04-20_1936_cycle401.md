### Test 1: Concurrent Phase Transition Race Condition
```python
import threading
import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_concurrent_phase_transition_race():
    """Test that phase transitions handle concurrent agent requests safely."""
    room = DeadbandRoom(max_agents=5)
    
    def agent_transition(agent_id, target_phase):
        try:
            room._transition_phase(agent_id, target_phase)
        except Exception as e:
            return str(e)
        return "success"
    
    # Simulate 3 agents attempting to transition simultaneously
    threads = []
    results = []
    
    for i in range(3):
        t = threading.Thread(
            target=lambda idx=i: results.append(agent_transition(f"agent_{idx}", "P1"))
        )
        threads.append(t)
    
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # At least one should succeed, none should crash
    assert any(r == "success" for r in results)
    assert not any("deadlock" in str(r).lower() for r in results if r != "success")
```

### Test 3: Malformed Tile Structure Handling

import json
import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_malformed_tile_submission():
    """Test room handles various malformed tile structures gracefully."""
    room = DeadbandRoom()
    
    malformed_tiles = [
        None,  # Null tile
        {"phase": "P1"},  # Missing required fields
        {"phase": "P3", "action": "invalid"},  # Invalid phase
        "not_a_dict",  # Wrong type
        {"phase": "P1", "action": "", "metadata": {"depth": -1}},  # Invalid metadata
        json.dumps({"phase": "P1"}),  # JSON string instead of dict
    ]
    
    for tile in malformed_tiles:
        result = room.submit_tile("test_agent", tile)
        assert result["status"] in ["rejected", "error"]
        assert "validation" in result["message"].lower() or "invalid" in result["message"].lower()
```

### Test 4: Network Partition Resilience

import pytest
from unittest.mock import patch, MagicMock
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_network_partition_recovery():
    """Test room recovers from simulated network partitions."""
    room = DeadbandRoom()
    
    # Simulate normal operation
    room.submit_tile("agent1", {"phase": "P0", "action": "map_negative"})
    
    # Simulate network partition - connection drops
    with patch('plato_torch.rooms.deadband_room._check_connectivity', return_value=False):
        partition_result = room.submit_tile("agent1", {"phase": "P1", "action": "find_channel"})
        assert partition_result["status"] == "buffered"
        assert "offline" in partition_result["message"]
    
    # Simulate reconnection
    with patch('plato_torch.rooms.deadband_room._check_connectivity', return_value=True):
        recovery_result = room._flush_buffered_tiles()
        assert len(recovery_result["processed"]) > 0
        assert recovery_result["status"] == "recovered"
```

### Test 5: Protocol Rollback from P2 to P1
```python
import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_protocol_rollback_p2_to_p1():
    """Test rollback when P1 channels become invalid during P2 optimization."""
    room = DeadbandRoom()
    
    # Start in P1 with valid channels
    room._current_phase = "P1"
    room._safe_channels = ["channel_a", "channel_b"]
    
    # Begin P2 optimization
    room._current_phase = "P2"
    
    # Simulate external event that invalidates channels
    room._safe_channels = []  # Channels become unavailable
    
    # Attempt to continue P2 optimization
    result = room.process_phase("P2", {"channels": ["channel_a"]})
    
    # Should trigger rollback to P1
    assert result["status"] == "rollback"
    assert room._current_phase == "P1"
    assert "channels invalid" in result["message"]
    assert room._rollback_count > 0
```
