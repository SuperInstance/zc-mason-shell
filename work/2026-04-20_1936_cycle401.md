## Proposed Edge-Case Tests
Based on the Deadband Protocol doctrine (P0→P1→P2 strict priority) and the room's API, I will write tests for the following scenarios:

### 1. **Empty Negative Space (P0)**
**Scenario:** Agent submits an empty list for negative space.
**Expected:** Room should handle gracefully (return empty safe channels or specific error state).
**Rationale:** Tests initialization boundary where no constraints exist.

### 2. **Overlapping Safe Channels (P1)**
**Scenario:** Safe channels returned from P1 have overlapping ranges.
**Expected:** Room should deduplicate or merge channels before P2 optimization.
**Rationale:** Ensures channel list is valid for optimization phase.

### 3. **No Viable Channels After P1**
**Scenario:** P1 returns an empty list of safe channels.
**Expected:** Room should not proceed to P2; should return appropriate "no path" state.
**Rationale:** Tests dead-end scenario per protocol logic.

### 4. **Malformed Optimization Goal (P2)**
**Scenario:** P2 receives a valid channel but an invalid optimization target (e.g., None, wrong type).
**Expected:** Room should raise a clear error or return a fallback result.
**Rationale:** Tests input validation in the optimization phase.

### 5. **State Persistence Across Protocol Steps**
**Scenario:** Call P0, then P1, then P2 with intermediate state checks.
**Expected:** Room should maintain internal state (negative space, channels) between steps.
**Rationale:** Validates that the room correctly sequences the protocol.

# plato-torch/tests/test_deadband_room.py (additional tests)

def test_p0_empty_negative_space(deadband_room):
    """Test DeadbandRoom handles empty negative space list."""
    # Arrange
    negative_space = []
    
    # Act
    result = deadband_room.map_negative_space(negative_space)
    
    # Assert
    assert result is not None
    # Depending on implementation, could be empty dict or list
    assert len(result) == 0

def test_p1_overlapping_channels(deadband_room):
    """Test DeadbandRoom merges overlapping safe channels."""
    # Arrange
    negative_space = [{"x": (0, 10), "y": (0, 10)}]
    deadband_room.map_negative_space(negative_space)
    
    # Mock find_safe_channels to return overlapping ranges
    overlapping_channels = [
        {"x": (5, 15), "y": (5, 15)},
        {"x": (10, 20), "y": (10, 20)},
        {"x": (8, 12), "y": (8, 12)}  # Contained within first
    ]
    deadband_room.find_safe_channels = lambda: overlapping_channels
    
    # Act
    channels = deadband_room.find_safe_channels()
    
    # Assert - channels should be deduplicated/merged
    assert len(channels) <= len(overlapping_channels)
    # Additional check: no channel should be fully contained within another
    for i, chan1 in enumerate(channels):
        for j, chan2 in enumerate(channels):
            if i != j:
                # Simple overlap check for x range
                x1_start, x1_end = chan1["x"]
                x2_start, x2_end = chan2["x"]
                assert not (x1_start >= x2_start and x1_end <= x2_end)

def test_p1_no_channels(deadband_room):
    """Test DeadbandRoom handles case with no safe channels."""
    # Arrange
    negative_space = [{"x": (0, 100), "y": (0, 100)}]  # Entire space blocked
    deadband_room.map_negative_space(negative_space)
    
    # Mock find_safe_channels to return empty list
    deadband_room.find_safe_channels = lambda: []
    
    # Act
    channels = deadband_room.find_safe_channels()
    
    # Assert
    assert channels == []
    # P2 should not be callable or should return empty
    with pytest.raises(ValueError) as exc_info:
        deadband_room.optimize_within_channels(channels, target={"x": 50, "y": 50})
    assert "no channels" in str(exc_info.value).lower()

def test_p2_malformed_target(deadband_room):
    """Test DeadbandRoom handles invalid optimization target."""
    # Arrange
    negative_space = [{"x": (0, 10), "y": (0, 10)}]
    deadband_room.map_negative_space(negative_space)
    
    channels = [{"x": (20, 30), "y": (20, 30)}]
    deadband_room.find_safe_channels = lambda: channels
    
    invalid_targets = [None, "not_a_dict", {"x": 50}, {}]
    
    for target in invalid_targets:
        # Act & Assert
        with pytest.raises((ValueError, TypeError)):
            deadband_room.optimize_within_channels(channels, target=target)

def test_state_persistence(deadband_room):
    """Test DeadbandRoom maintains state across P0→P1→P2 sequence."""
    # Arrange
    negative_space = [{"x": (0, 10), "y": (0, 10)}]
    
    # Act P0
    p0_result = deadband_room.map_negative_space(negative_space)
    assert deadband_room.negative_space == negative_space
    
    # Act P1
    channels = deadband_room.find_safe_channels()
    assert deadband_room.safe_channels == channels
    
    # Act P2
    target = {"x": 50, "y": 50}
    p2_result = deadband_room.optimize_within_channels(channels, target=target)
    
    # Assert final state contains all protocol data
    assert hasattr(deadband_room, 'negative_space')
    assert hasattr(deadband_room, 'safe_channels')
    assert hasattr(deadband_room, 'optimization_result')
    assert deadband_room.optimization_result == p2_result
