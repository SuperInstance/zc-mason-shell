### Test 3: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_handling(self):
    """Simulate multiple agents submitting tiles simultaneously."""
    from threading import Thread
    
    results = []
    errors = []
    
    def submit_tile_worker(agent_id, tile_data):
        try:
            tile = Tile(**tile_data)
            result = self.room.submit_tile(tile, agent_id=agent_id)
            results.append((agent_id, result["accepted"]))
        except Exception as e:
            errors.append((agent_id, str(e)))
    
    # Create 5 agents submitting simultaneously
    workers = []
    for i in range(5):
        tile_data = {
            "phase": "P1",
            "content": f"Safe channel {i}",
            "score": 0.5 + (i * 0.1)
        }
        worker = Thread(target=submit_tile_worker, args=(f"agent_{i}", tile_data))
        workers.append(worker)
    
    # Start all workers
    for w in workers:
        w.start()
    
    # Wait for completion
    for w in workers:
        w.join()
    
    # Verify no data corruption occurred
    self.assertEqual(len(errors), 0, f"Errors during concurrent access: {errors}")
    self.assertEqual(len(results), 5, "Not all submissions completed")
    
    # Check room state consistency
    room_state = self.room.get_state()
    self.assertEqual(room_state["total_tiles"], 5)
    self.assertEqual(len(room_state["agent_stats"]), 5)
```

```python
def test_corrupted_state_recovery(self):
    """Test room's ability to recover from malformed internal state."""
    # Manually corrupt room's internal data structure
    self.room._agent_states["corrupt_agent"] = {
        "current_phase": "INVALID_PHASE",  # Invalid phase value
        "tile_count": -5,  # Negative count
        "last_submission": "not-a-datetime"
    }
    
    # Add malformed tile to history
    self.room._tile_history.append({
        "agent_id": "corrupt_agent",
        "content": None,  # Missing required fields
        "timestamp": "invalid"
    })
    
    # Attempt normal operation - room should sanitize or reset corrupted state
    valid_tile = Tile(phase="P0", content="Recovery test", score=0.3)
    result = self.room.submit_tile(valid_tile, agent_id="corrupt_agent")
    
    # Room should either:
    # 1. Reject due to corrupted state and provide reset instructions, OR
    # 2. Reset agent state and accept the tile
    self.assertTrue(
        result["accepted"] or "reset" in result["feedback"].lower(),
        f"Room didn't handle corruption properly: {result}"
    )
    
    # Verify agent state is now valid
    agent_state = self.room.get_agent_state("corrupt_agent")
    self.assertIn(agent_state["current_phase"], ["P0", "P1", "P2"])
    self.assertGreaterEqual(agent_state["tile_count"], 0)
```
