### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom()
    room.reset()
    
    # Simulate 3 agents submitting tiles at same cycle
    tiles = [
        {"agent_id": "agent1", "cycle": 100, "content": "P0: avoid null state"},
        {"agent_id": "agent2", "cycle": 100, "content": "P1: safe channel found"},
        {"agent_id": "agent3", "cycle": 100, "content": "P2: optimized path"}
    ]
    
    # Submit concurrently (simulated with threads)
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            room.submit_tile(tile)
            results.append(True)
        except:
            results.append(False)
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # All submissions should succeed
    assert all(results), "Concurrent submission failed"
    assert len(room.tiles) == 3, f"Expected 3 tiles, got {len(room.tiles)}"
    
    # Verify tile integrity
    for tile in room.tiles:
        assert tile["validated"] == True, "Tile not validated"
        assert "cycle" in tile and tile["cycle"] == 100
```

```python
def test_state_corruption_recovery(self):
    """Room should detect and recover from corrupted internal state."""
    room = DeadbandRoom()
    room.reset()
    
    # Inject corrupted state (simulating memory error)
    room._state = {"invalid": "corrupted", "cycle": -1}
    
    # Attempt protocol execution
    try:
        result = room.execute_protocol({"input": "test"})
        # Should either raise exception or return error state
        if result:
            assert "error" in result or "recovered" in result
    except RuntimeError as e:
        # Expected: room detected corruption
        assert "corrupt" in str(e).lower() or "invalid" in str(e).lower()
    
    # Verify room can reset after corruption
    room.reset()
    assert room.cycle == 0
    assert room.tiles == []
```

### Test 3: Protocol Violation Handling

```python
def test_protocol_violation_skip_p1(self):
    """Agent skipping P1 (safe channels) should be rejected."""
    room = DeadbandRoom()
    room.reset()
    
    # Agent submits P0 then immediately P2 (violating P1 requirement)
    tile_p0 = {
        "agent_id": "violator",
        "cycle": 50,
        "content": "P0: identified rocks",
        "protocol_step": "P0"
    }
    
    tile_p2 = {
        "agent_id": "violator", 
        "cycle": 51,
        "content": "P2: optimized solution",
        "protocol_step": "P2"
    }
    
    # P0 should be accepted
    room.submit_tile(tile_p0)
    assert len(room.tiles) == 1
    
    # P2 without P1 should be rejected
    try:
        room.submit_tile(tile_p2)
        # If submission succeeds, verify it's marked as violation
        latest = room.tiles[-1]
        assert latest.get("violation") == True
        assert "P1" in latest.get("rejection_reason", "")
    except ValueError as e:
        # Expected rejection
        assert "P1" in str(e) or "protocol" in str(e).lower()
```

### Test 4: Resource Exhaustion Resilience
```python
def test_memory_exhaustion_handling(self):
    """Room should handle memory limits during tile accumulation."""
    room = DeadbandRoom()
    room.reset()
    
    # Configure low memory threshold (simulated)
    room.max_tiles = 1000  # Arbitrary limit
    
    # Submit tiles until limit
    for i in range(1000):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": i,
            "content": f"Test tile {i}",
            "protocol_step": "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2"
        }
        room.submit_tile(tile)
    
    # Next submission should trigger cleanup or rejection
    excess_tile = {
        "agent_id": "excess_agent",
        "cycle": 1001,
        "content": "Should trigger cleanup",
        "protocol_step": "P0"
    }
    
    try:
        room.submit_tile(excess_tile)
        # If accepted, verify cleanup occurred
        assert len(room.tiles) <= room.max_tiles
        assert any(t.get("archived") == True for t in room.tiles)
    except MemoryError:
        # Expected when room cannot handle more tiles
        pass
    
    # Room should remain functional after limit
    assert room.cycle >= 0
    assert room.execute_protocol({"input": "test"}) is not None
```

```python
def test_partial_state_sync(self):
    """Room should handle partial state synchronization failures."""
    room = DeadbandRoom()
    room.reset()
    
    # Simulate network partition during state sync
    original_sync = room._sync_state
    
    def failing_sync():
        raise ConnectionError("Network partition simulated")
    
    room._sync_state = failing_sync
    
    # Room should continue operating with local state
    tile = {
        "agent_id": "partitioned_agent",
        "cycle": 200,
        "content": "Working offline",
        "protocol_step": "P1"
    }
    
    # Submission should work locally
    room.submit_tile(tile)
    assert len(room.tiles) == 1
    
    # Protocol execution should work
    result = room.execute_protocol({"input": "test"})
    assert result is not None
    
    # State should be marked as unsynced
    assert room._state.get("synced") == False
    
    # Restore sync and verify recovery
    room._sync_state = original_sync
    room._sync_state()
    assert room._state.get("synced") == True
```
