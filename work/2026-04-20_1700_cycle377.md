```python
def test_p2_before_p1_rejection():
    """Submit P2 optimization tile before P1 safe channel tile."""
    room = DeadbandRoom()
    agent_id = "test_agent_1"
    
    # Attempt P2 tile first (should fail)
    p2_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "payload": {"optimization": "best_path"},
        "timestamp": time.time()
    }
    result = room.submit_tile(p2_tile)
    assert not result["accepted"]
    assert "P1 required before P2" in result["error"]
    
    # Submit P1 tile (should succeed)
    p1_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "payload": {"channel": "safe_zone"},
        "timestamp": time.time()
    }
    result = room.submit_tile(p1_tile)
    assert result["accepted"]
    
    # Now P2 should work
    p2_tile["timestamp"] = time.time()
    result = room.submit_tile(p2_tile)
    assert result["accepted"]
```

```python
def test_partial_write_recovery():
    """Simulate partial write during tile submission."""
    room = DeadbandRoom()
    agent_id = "test_agent_3"
    
    # Mock a partial write scenario
    original_save = room._save_tile_to_storage
    
    def failing_save(tile):
        raise IOError("Simulated disk full")
    
    room._save_tile_to_storage = failing_save
    
    tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "payload": {"negative_space": "rock"},
        "timestamp": time.time()
    }
    
    result = room.submit_tile(tile)
    assert not result["accepted"]
    assert "storage error" in result["error"].lower()
    
    # Restore and verify room still functional
    room._save_tile_to_storage = original_save
    result = room.submit_tile(tile)
    assert result["accepted"]
```

### Test 5: Ghost Tile References

```python
def test_ghost_tile_references():
    """Test tiles that reference dead agents (ghost tiles)."""
    room = DeadbandRoom()
    live_agent = "live_agent_1"
    dead_agent = "ghost_agent_999"  # Simulated dead agent
    
    # First, simulate a tile from now-dead agent
    ghost_tile = {
        "agent_id": dead_agent,
        "phase": "P0",
        "payload": {"warning": "avoid_this"},
        "timestamp": time.time() - 3600  # 1 hour ago
    }
    room.submit_tile(ghost_tile)
    
    # Live agent references ghost agent's tile
    reference_tile = {
        "agent_id": live_agent,
        "phase": "P1",
        "payload": {
            "channel": "safe",
            "references": [dead_agent]  # Reference to ghost
        },
        "timestamp": time.time()
    }
    
    result = room.submit_tile(reference_tile)
    # Should be accepted but flagged for review
    assert result["accepted"]
    assert "ghost_reference" in result.get("warnings", [])
    
    # Verify ghost tile is still accessible
    history = room.get_agent_history(dead_agent)
    assert len(history) == 1
    assert history[0]["agent_id"] == dead_agent
```
