### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous submissions from multiple agents."""
    import threading
    
    room = DeadbandRoom(room_id="concurrent_test")
    results = []
    errors = []
    
    def submit_tile(agent_id):
        try:
            tile = {
                "agent_id": agent_id,
                "phase": "P1",
                "content": f"Test from {agent_id}",
                "timestamp": time.time()
            }
            result = room.submit_tile(tile)
            results.append((agent_id, result))
        except Exception as e:
            errors.append((agent_id, str(e)))
    
    # Simulate 5 agents submitting simultaneously
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    self.assertEqual(len(results), 5)
    self.assertEqual(len(errors), 0)
    
    # Verify room state reflects all submissions
    self.assertEqual(len(room.get_tile_history()), 5)
    
    # Verify no duplicate processing
    agent_ids = [r[0] for r in results]
    self.assertEqual(len(set(agent_ids)), 5)
```

```python
def test_validator_network_failure(self):
    """Test room behavior when TileValidator throws connection errors."""
    room = DeadbandRoom(room_id="network_failure_test")
    
    # Mock validator to simulate network failure
    mock_validator = Mock(spec=TileValidator)
    mock_validator.validate_tile.side_effect = ConnectionError("Network unreachable")
    room.validator = mock_validator
    
    tile = {
        "agent_id": "test_agent",
        "phase": "P0",
        "content": "Test tile during network outage",
        "timestamp": time.time()
    }
    
    # Should raise specific error, not generic exception
    with self.assertRaises(ConnectionError) as context:
        room.submit_tile(tile)
    
    self.assertIn("Network unreachable", str(context.exception))
    
    # Verify tile was NOT added to history
    self.assertEqual(len(room.get_tile_history()), 0)
    
    # Verify room state unchanged
    self.assertEqual(room.current_phase, "P0")  # Default phase
```

### Test 4: State Rollback on Failed Submission
```python
def test_state_rollback_on_failed_submission(self):
    """Test that room state remains consistent after failed submission."""
    room = DeadbandRoom(room_id="rollback_test")
    
    # First, submit a valid tile to establish state
    valid_tile = {
        "agent_id": "agent_1",
        "phase": "P1",
        "content": "Valid initial tile",
        "timestamp": time.time()
    }
    room.submit_tile(valid_tile)
    
    initial_state = {
        "tile_count": len(room.get_tile_history()),
        "last_agent": room.get_tile_history()[-1]["agent_id"],
        "current_phase": room.current_phase
    }
    
    # Now attempt submission that will fail mid-process
    mock_validator = Mock(spec=TileValidator)
    
    # Simulate validator that fails after room has processed tile locally
    call_count = 0
    def failing_validate(tile):
        nonlocal call_count
        call_count += 1
        if call_count == 2:  # Fail on second validation attempt
            raise RuntimeError("Simulated processing failure")
        return True
    
    mock_validator.validate_tile.side_effect = failing_validate
    room.validator = mock_validator
    
    failing_tile = {
        "agent_id": "agent_2",
        "phase": "P2",
        "content": "This should fail",
        "timestamp": time.time()
    }
    
    try:
        room.submit_tile(failing_tile)
    except RuntimeError:
        pass  # Expected failure
    
    # Verify room state rolled back to initial state
    final_state = {
        "tile_count": len(room.get_tile_history()),
        "last_agent": room.get_tile_history()[-1]["agent_id"] if room.get_tile_history() else None,
        "current_phase": room.current_phase
    }
    
    self.assertEqual(initial_state["tile_count"], final_state["tile_count"])
    self.assertEqual(initial_state["last_agent"], final_state["last_agent"])
    # Note: current_phase might change based on room logic
```
