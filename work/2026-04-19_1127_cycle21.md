### Test 1: Concurrent Channel Occupation
```rust
#[test]
fn test_concurrent_p1_channel_occupation() {
    let config = RoomConfig {
        max_agents: 3,
        channel_capacity: 1, // Single agent per channel
        ..Default::default()
    };
    
    let mut room = DeadbandRoom::new(config);
    let agent1 = AgentId::new("agent1");
    let agent2 = AgentId::new("agent2");
    
    // Both agents complete P0 simultaneously
    room.process_tile(agent1, Tile::new(P0, "avoid_rock"));
    room.process_tile(agent2, Tile::new(P0, "avoid_rock"));
    
    // Both attempt same P1 channel
    let tile1 = Tile::new(P1, "channel_a");
    let tile2 = Tile::new(P1, "channel_a");
    
    let result1 = room.process_tile(agent1, tile1);
    let result2 = room.process_tile(agent2, tile2);
    
    // One should succeed, one should be rejected
    assert!(result1.is_ok() ^ result2.is_ok());
    assert_eq!(room.get_agent_state(&agent1).channel, 
               room.get_agent_state(&agent2).channel);
}
```

### Test 2: Tile Flooding Protection
```rust
#[test]
fn test_tile_flooding_protection() {
    let config = RoomConfig {
        max_tiles_per_cycle: 3,
        ..Default::default()
    };
    
    let mut room = DeadbandRoom::new(config);
    let agent = AgentId::new("flooder");
    
    // Submit 4 tiles in rapid succession (exceeds limit)
    let tiles = vec![
        Tile::new(P0, "avoid_a"),
        Tile::new(P0, "avoid_b"),
        Tile::new(P0, "avoid_c"),
        Tile::new(P0, "avoid_d"),
    ];
    
    let mut accepted = 0;
    for tile in tiles {
        if room.process_tile(agent, tile).is_ok() {
            accepted += 1;
        }
    }
    
    // Should accept at most max_tiles_per_cycle
    assert!(accepted <= config.max_tiles_per_cycle);
    assert!(room.get_agent_state(&agent).is_throttled());
}
```

### Test 3: Malformed Tile Payload
```rust
#[test]
fn test_malformed_tile_payload() {
    let mut room = DeadbandRoom::default();
    let agent = AgentId::new("agent");
    
    // Tile missing required 'content' field
    let malformed_tile = Tile {
        priority: P0,
        content: None, // Missing content
        metadata: HashMap::new(),
    };
    
    let result = room.process_tile(agent, malformed_tile);
    assert!(result.is_err());
    assert!(matches!(result.unwrap_err(), RoomError::InvalidTileFormat));
    
    // Agent should remain in P0 state (no progress)
    assert_eq!(room.get_agent_state(&agent).current_priority, P0);
}
```

### Test 5: Agent Disconnection State Persistence
```rust
#[test]
fn test_agent_disconnection_reconnection() {
    let mut room = DeadbandRoom::default();
    let agent = AgentId::new("agent");
    
    // Agent progresses to P1
    room.process_tile(agent, Tile::new(P0, "avoid_rock")).unwrap();
    room.process_tile(agent, Tile::new(P1, "channel_x")).unwrap();
    
    let state_before = room.get_agent_state(&agent).clone();
    
    // Simulate disconnection (remove from active agents)
    room.disconnect_agent(&agent);
    assert!(!room.is_agent_active(&agent));
    
    // Reconnect with same agent ID
    room.connect_agent(agent);
    let state_after = room.get_agent_state(&agent);
    
    // Should retain P1 state and channel
    assert_eq!(state_before.current_priority, state_after.current_priority);
    assert_eq!(state_before.channel, state_after.channel);
    assert_eq!(state_before.tiles_submitted, state_after.tiles_submitted);
}
```
