#[test]
fn test_deadband_room_empty_constraint_space() {
    use plato_torch::presets::DeadbandRoom;
    
    let mut room = DeadbandRoom::new();
    // Empty negative and positive constraints
    let result = room.apply_protocol(&vec![], &vec![]);
    
    // P0: With no constraints, everything is "safe" (no negative space)
    // P1: Should return all possible states (empty filter)

### Test 2: Overlapping Negative/Positive Constraints
```rust
#[test]
fn test_deadband_room_overlapping_constraints() {
    use plato_torch::presets::DeadbandRoom;
    
    let mut room = DeadbandRoom::new();
    // Same constraint appears in both negative and positive sets
    let negative = vec!["collision".to_string(), "overheat".to_string()];
    let positive = vec!["collision".to_string()]; // Overlap!
    
    let result = room.apply_protocol(&negative, &positive);
    
    // P0 should filter out "collision" first
    // P1 then tries to find "collision" in positive set but it's already removed
    // Result should be empty (no safe channels with overlapping constraint)
    assert!(result.is_ok());
    let safe_states = result.unwrap();
    assert_eq!(safe_states.len(), 0);
}
```

### Test 3: Protocol Reset After Partial P1
```rust
#[test]
fn test_deadband_room_reset_after_partial_completion() {
    use plato_torch::presets::DeadbandRoom;
    
    let mut room = DeadbandRoom::new();
    let negative = vec!["rock".to_string()];
    let positive = vec!["channel_a".to_string(), "channel_b".to_string()];
    
    // Complete P0 and P1
    let phase1_result = room.apply_protocol(&negative, &positive);
    assert!(phase1_result.is_ok());
    
    // Reset room state (simulating new training cycle)
    room.reset();
    
    // Apply different constraints
    let negative2 = vec!["wall".to_string()];
    let positive2 = vec!["channel_c".to_string()];
    let phase2_result = room.apply_protocol(&negative2, &positive2);
    
    // Should work independently without carryover
    assert!(phase2_result.is_ok());
    let safe_states = phase2_result.unwrap();
    assert_eq!(safe_states, vec!["channel_c".to_string()]);
}
```

### Test 4: Memory Persistence Across Multiple Cycles
```rust
#[test]
fn test_deadband_room_memory_persistence() {
    use plato_torch::presets::DeadbandRoom;
    
    let mut room = DeadbandRoom::new();
    let mut all_violations = Vec::new();
    
    // Run 3 training cycles with same negative constraint
    for i in 0..3 {
        let negative = vec![format!("violation_{}", i)];
        let positive = vec!["safe_channel".to_string()];
        
        let result = room.apply_protocol(&negative, &positive);
        assert!(result.is_ok());
        
        // Record that this violation was processed
        all_violations.push(format!("violation_{}", i));
    }
    
    // Room should remember total violations processed
    // (Assuming DeadbandRoom has a metrics method)
    // This tests that training tiles accumulate
    let metrics = room.get_metrics();
    assert_eq!(metrics.total_cycles, 3);
    assert!(metrics.avg_violations_per_cycle > 0.0);
}
```
