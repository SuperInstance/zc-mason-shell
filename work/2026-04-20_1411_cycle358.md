## New Edge-Case Tests

### Test 1: Protocol Priority Violation (Skip P1)
**Scenario:** Agent attempts to optimize (P2) without first establishing a safe channel (P1).
**Expected:** Room should reject the P2 action and return a P1 requirement error.
**Rationale:** Tests strict priority enforcement from Deadband Protocol doctrine.

```python
def test_protocol_priority_violation_skip_p1(self):
    """Test that P2 optimization is rejected if P1 channel not established."""
    room = DeadbandRoom()
    # Attempt P2 optimization without P1 channel
    result = room.process_action({
        'phase': 'P2',
        'action': 'optimize_path',
        'parameters': {'target': 'efficiency'}
    })
    self.assertFalse(result['success'])
    self.assertIn('P1', result['error'])  # Error should mention P1 requirement
```

### Test 2: Channel Exhaustion
**Scenario:** All identified P1 channels become invalid during session.
**Expected:** Room should reset to P0 and require new negative space mapping.
**Rationale:** Tests recovery when no safe channels remain.

```python
def test_channel_exhaustion_recovery(self):
    """Test recovery when all P1 channels become invalid."""
    room = DeadbandRoom()
    # Establish a P1 channel
    room.process_action({
        'phase': 'P1',
        'action': 'identify_channel',
        'parameters': {'channel_id': 'safe_1'}
    })
    # Invalidate the channel (simulate external change)
    room.invalidate_channel('safe_1')
    # Attempt P1 action with invalid channel
    result = room.process_action({
        'phase': 'P1',
        'action': 'validate_channel',
        'parameters': {'channel_id': 'safe_1'}
    })
    self.assertFalse(result['success'])
    self.assertEqual(room.current_phase, 'P0')  # Should fall back to P0
```

### Test 3: Concurrent Session Interference
**Scenario:** Multiple agents interacting with same room instance.
**Expected:** Room should maintain separate state per session or handle conflicts gracefully.
**Rationale:** Tests thread safety and session isolation.

```python
def test_concurrent_session_state_isolation(self):
    """Test that room maintains separate state for different sessions."""
    room = DeadbandRoom()
    session_a = room.create_session('agent_a')
    session_b = room.create_session('agent_b')
    
    # Agent A establishes P1 channel

        'parameters': {'channel_id': 'channel_a'}
    })
    
    # Agent B should not see Agent A's channel
    result = room.process_action({
        'session_id': session_b,

        'action': 'list_channels',
        'parameters': {}
    })
    self.assertEqual(result['channels'], [])  # Should be empty for Agent B
```

### Test 4: Malformed Phase Transition
**Scenario:** Agent sends phase transition that doesn't follow P0→P1→P2 sequence.
**Expected:** Room should reject invalid transition and maintain current phase.
**Rationale:** Tests state machine integrity.

```python
def test_malformed_phase_transition_p2_to_p0(self):
    """Test that direct P2→P0 transition is rejected."""
    room = DeadbandRoom()
    # Setup: Move through P0→P1→P2 legitimately
    room.process_action({'phase': 'P0', 'action': 'map_negative_space'})
    room.process_action({'phase': 'P1', 'action': 'identify_channel'})
    room.process_action({'phase': 'P2', 'action': 'optimize_path'})
    
    # Attempt invalid transition
    result = room.process_action({
        'phase': 'P0',  # Trying to go back to P0 from P2
        'action': 'map_negative_space',
        'parameters': {}
    })
    self.assertFalse(result['success'])
    self.assertEqual(room.current_phase, 'P2')  # Should remain in P2
```

### Test 5: Resource Constraint Simulation
**Scenario:** P2 optimization hits computational or memory limits.

**Rationale:** Tests robustness under resource pressure.

```python
def test_p2_optimization_resource_constraint(self):
    """Test P2 optimization behavior when resources are constrained."""
    room = DeadbandRoom()
    # Setup: Reach P2 phase
    room.process_action({'phase': 'P0', 'action': 'map_negative_space'})
    room.process_action({'phase': 'P1', 'action': 'identify_channel'})
    
    # Simulate resource constraint via parameter
    result = room.process_action({
        'phase': 'P2',
        'action': 'optimize_path',
        'parameters': {
            'target': 'efficiency',
            'resource_constraint': 'high'  # Simulate limited resources
        }
    })
    
    # Should still succeed but with degraded optimization
    self.assertTrue(result['success'])
    self.assertIn('optimization_level', result)
    self.assertEqual(result['optimization_level'], 'degraded')
    self.assertIn('fallback_strategy', result)
```
