### Test 1: Concurrent Agent Submission Storm
**Scenario:** 50 agents submit tiles simultaneously to a room with capacity 20.
**Edge:** Thread safety, lock contention, queue overflow.
```python
def test_concurrent_agent_submission_storm(self):
    """P0: Room must not crash or corrupt state under concurrent load."""
    room = DeadbandRoom(capacity=20)
    agents = [MockAgent(f"agent_{i}") for i in range(50)]
    
    def submit_tile(agent):
        tile = TrainingTile(agent_id=agent.id, content=f"test from {agent.id}")
        return room.submit_tile(tile)
    
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = [executor.submit(submit_tile, agent) for agent in agents]
        results = [f.result() for f in futures]
    
    # Room should accept only up to capacity
    accepted = [r for r in results if r.status == "accepted"]
    assert len(accepted) <= room.capacity
    # No tile should be partially written
    assert room.state_file_exists()
    assert room.validate_state_integrity()
```

### Test 3: File Descriptor Exhaustion
**Scenario:** Room opens many log files without closing them.
**Edge:** Resource leak, OS limits.
```python
def test_file_descriptor_exhaustion(self):
    """P0: Room must handle file resource limits gracefully."""
    room = DeadbandRoom()
    # Simulate many tile submissions with file operations
    for i in range(1025):  # Exceeds typical ulimit of 1024
        tile = TrainingTile(
            agent_id=f"agent_{i}",
            content=f"test content {i}"
        )
        # Room should handle OS errors without crashing
        try:
            room.submit_tile(tile)
        except OSError as e:
            # Should log error, not crash
            assert room.error_log is not None
            break
    
    # Room should still be operational for new submissions
    recovery_tile = TrainingTile(agent_id="recovery", content="test")
    result = room.submit_tile(recovery_tile)
    assert result.status in ["accepted", "rejected"]  # Defined state
```

### Test 4: Network Partition During Write
**Scenario:** Room writes tile to disk but fails mid-write due to simulated network error.
**Edge:** Partial writes, state corruption recovery.
```python
def test_network_partition_during_write(self):
    """P0: Room must detect and recover from partial tile writes."""
    room = DeadbandRoom()
    agent = MockAgent("partition_agent")
    
    # Monkey-patch the file write to fail halfway
    original_write = room._write_tile_to_disk
    call_count = 0
    
    def failing_write(tile, path):
        nonlocal call_count
        call_count += 1
        if call_count == 1:
            # First write fails after partial write
            with open(path, 'w') as f:
                f.write(tile.content[:10])  # Partial write
                raise ConnectionError("Network partition")
        return original_write(tile, path)
    
    room._write_tile_to_disk = failing_write
    
    tile = TrainingTile(agent_id=agent.id, content="complete tile content")
    result = room.submit_tile(tile)
    
    # Room should either:
    # 1. Reject the tile due to write failure
    # 2. Retry and succeed on second attempt
    assert result.status in ["rejected", "accepted"]
    if result.status == "accepted":
        # Verify complete write on retry
        assert call_count == 2
        assert room.get_tile(agent.id).content == "complete tile content"
```

### Test 5: Ghost Tile Re-integration After Agent Death
**Scenario:** Agent dies mid-task, leaves partial tile. New agent resumes same task.
**Edge:** State reconciliation, duplicate prevention.
```python
def test_ghost_tile_reintegration(self):
    """P1: Room should allow new agents to continue dead agents' work."""
    room = DeadbandRoom()
    dead_agent = MockAgent("dead_agent")
    new_agent = MockAgent("new_agent")
    
    # Simulate dead agent's partial tile
    ghost_tile = TrainingTile(
        agent_id=dead_agent.id,
        content="P0 analysis incomplete...",
        protocol_steps=["P0"],
        metadata={"task_id": "task_123", "status": "partial"}
    )
    room._store_ghost_tile(ghost_tile)
    
    # New agent submits continuation tile for same task
    continuation_tile = TrainingTile(
        agent_id=new_agent.id,
        content="P0 complete. P1 safe channel found.",
        protocol_steps=["P0", "P1"],
        metadata={"task_id": "task_123", "continues": dead_agent.id}
    )
    
    result = room.submit_tile(continuation_tile)
    # Room should accept continuation of same task
    assert result.status == "accepted"
    # Ghost tile should be marked as superseded
    ghost = room.get_ghost_tile(dead_agent.id)
    assert ghost.metadata.get("superseded_by") == new_agent.id
```
