### Test 1: Concurrent Tile Processing Race Condition
```rust
#[tokio::test]
async fn test_concurrent_tile_processing_race_condition() {
    use std::sync::Arc;
    use tokio::task;
    
    let room = Arc::new(tokio::sync::Mutex::new(DeadbandRoom::new()));
    let mut handles = vec![];
    
    // Spawn 10 concurrent tile processing tasks
    for i in 0..10 {
        let room_clone = Arc::clone(&room);
        handles.push(task::spawn(async move {
            let tile = Tile {
                id: format!("tile-{}", i),
                content: format!("content-{}", i),
                priority: i % 3,
                timestamp: chrono::Utc::now(),
            };
            
            let mut room_guard = room_clone.lock().await;
            room_guard.process_tile(tile).await
        }));
    }
    
    // Verify all tiles processed without deadlock or data corruption
    let results = futures::future::join_all(handles).await;
    let mut success_count = 0;
    let mut error_count = 0;
    
    for result in results {
        match result {
            Ok(Ok(_)) => success_count += 1,
            Ok(Err(_)) => error_count += 1,
            Err(_) => panic!("Task panicked"),
        }
    }
    
    // Should process all tiles successfully
    assert_eq!(success_count, 10);
    assert_eq!(error_count, 0);
    
    // Verify final room state is consistent
    let room_guard = room.lock().await;
    let state = room_guard.get_state();
    assert_eq!(state.processed_tiles, 10);
}
```

### Test 2: Memory Pressure with Large Tile Batches
```rust
#[test]
fn test_memory_pressure_large_tile_batch() {
    let mut room = DeadbandRoom::new();
    let mut tiles = Vec::with_capacity(10_000);
    
    // Generate 10,000 tiles with varying sizes
    for i in 0..10_000 {
        let content_size = if i % 100 == 0 {
            // Every 100th tile has 1MB content (simulating large payload)
            String::from_utf8(vec![b'X'; 1_000_000]).unwrap()
        } else {
            format!("tile-{}", i)
        };
        
        tiles.push(Tile {
            id: format!("batch-tile-{}", i),
            content: content_size,
            priority: (i % 5) as u8,
            timestamp: chrono::Utc::now(),
        });
    }
    
    // Process in batches of 1000 to avoid immediate OOM
    let mut processed_count = 0;
    for chunk in tiles.chunks(1000) {
        for tile in chunk {
            match room.process_tile(tile.clone()) {
                Ok(_) => processed_count += 1,
                Err(RoomError::ResourceExhausted(_)) => {
                    // Acceptable if room protects itself
                    println!("Room rejected tile {} due to memory pressure", tile.id);
                    continue;
                }
                Err(e) => panic!("Unexpected error: {:?}", e),
            }
        }
        
        // Force garbage collection if supported
        if let Some(gc_method) = room.as_any().downcast_ref::<GarbageCollector>() {
            gc_method.collect();
        }
    }
    
    // Room should either process all or fail gracefully
    assert!(processed_count > 0, "Room should process at least some tiles");
    let state = room.get_state();
    assert!(state.memory_usage < state.memory_limit, "Should stay within limits");
}
```

### Test 3: Circular Constraint Dependency Detection
```rust
#[test]
fn test_circular_constraint_dependency() {
    let mut room = DeadbandRoom::new();
    
    // Create constraints that form a circular dependency: A → B → C → A
    let constraints = vec![
        Constraint {
            id: "constraint-a".to_string(),
            depends_on: vec!["constraint-c".to_string()],  // Circular: A depends on C
            rule: ConstraintRule::MustPrecede { target: "constraint-b".to_string() },
        },
        Constraint {
            id: "constraint-b".to_string(),
            depends_on: vec!["constraint-a".to_string()],  // B depends on A
            rule: ConstraintRule::MustPrecede { target: "constraint-c".to_string() },
        },
        Constraint {
            id: "constraint-c".to_string(),
            depends_on: vec!["constraint-b".to_string()],  // C depends on B
            rule: ConstraintRule::MustPrecede { target: "constraint-a".to_string() },  // Completes circle
        },
    ];
    
    // Room should detect and reject circular dependencies
    match room.apply_constraints(constraints) {
        Err(RoomError::ConstraintError(ConstraintError::CircularDependency(details))) => {
            // Verify error contains information about the cycle
            assert!(details.contains("constraint-a"));
            assert!(details.contains("constraint-b"));
            assert!(details.contains("constraint-c"));
            println!("Correctly detected circular dependency: {}", details);
        }
        Ok(_) => panic!("Should have rejected circular constraints"),
        Err(e) => panic!("Wrong error type: {:?}", e),
    }
    
    // Room state should remain unchanged
    let state = room.get_state();
    assert_eq!(state.active_constraints, 0);
}
```

### Test 4: Network Partition Simulation (Partial State)
```rust
#[test]
fn test_network_partition_recovery() {
    let mut room = DeadbandRoom::new();
    
    // Simulate normal operation
    for i in 0..5 {
        room.process_tile(Tile {
            id: format!("pre-partition-{}", i),
            content: "normal".to_string(),
            priority: 1,
            timestamp: chrono::Utc::now(),
        }).unwrap();
    }
    
    let pre_partition_state = room.get_state().clone();
    
    // Simulate network partition: Room loses connection to storage
    // This would normally be triggered by a network failure event
    room.simulate_network_partition(true);
    
    // During partition, tiles should be queued or rejected
    let partition_result = room.process_tile(Tile {
        id: "during-partition".to_string(),
        content: "queued".to_string(),
        priority: 2,
        timestamp: chrono::Utc::now(),
    });
    
    // Either queued locally or rejected with appropriate error
    match partition_result {
        Ok(processed) if processed.status == TileStatus::QueuedLocally => {
            // Acceptable: tile queued for later sync
        }
        Err(RoomError::NetworkError(_)) => {
            // Acceptable: rejected due to partition
        }
        _ => panic!("Unexpected behavior during partition"),
    }
    
    // Simulate network recovery
    room.simulate_network_partition(false);
    
    // Room should attempt to sync queued tiles
    let recovery_result = room.recover_from_partition();
    assert!(recovery_result.is_ok(), "Should recover from partition");
    
    // State should be consistent post-recovery
    let post_recovery_state = room.get_state();
    assert!(
        post_recovery_state.last_sync >= pre_partition_state.last_sync,
        "Should have synced after recovery"
    );
}
