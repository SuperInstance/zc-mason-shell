### Test 1: Concurrent Tile Submission
```rust
#[test]
fn test_concurrent_tile_submission() {
    let room = DeadbandRoom::new("test-room", 10);
    let tile1 = Tile::new("agent-1", "P0 constraint: avoid null pointers");
    let tile2 = Tile::new("agent-2", "P0 constraint: validate input bounds");
    
    // Simulate concurrent submission
    let room_clone = room.clone();
    let handle1 = thread::spawn(move || room_clone.submit_tile(tile1));
    let handle2 = thread::spawn(move || room.submit_tile(tile2));
    
    let result1 = handle1.join().expect("Thread 1 panicked");
    let result2 = handle2.join().expect("Thread 2 panicked");
    
    // Both submissions should succeed or be properly serialized
    assert!(result1.is_ok() || result2.is_ok());
    // At least one should succeed if room capacity allows
    assert!(!(result1.is_err() && result2.is_err()));
}
```

### Test 3: No Safe Channels in P1
```rust
#[test]
fn test_channel_exhaustion() {
    let mut room = DeadbandRoom::new("exhausted-room", 3);
    
    // Submit P0 constraints that eliminate all possible channels
    let constraints = vec![
        "P0: Avoid all network I/O",
        "P0: Avoid all file I/O", 
        "P0: Avoid all memory allocation",
    ];
    
    for constraint in constraints {
        let tile = Tile::new("sentinel", constraint);
        room.submit_tile(tile).expect("Failed to submit P0 constraint");
    }
    
    // Attempt to transition to P1 should fail
    match room.current_protocol_phase() {
        ProtocolPhase::P0 => {
            let result = room.advance_phase();
            assert!(result.is_err());
            assert!(result.unwrap_err().contains("no safe channels"));
        }
        _ => panic!("Should still be in P0 phase"),
    }
}
```

### Test 4: Protocol Regression Attempt
```rust
#[test]
fn test_skip_to_p2_attempt() {
    let mut room = DeadbandRoom::new("regression-test", 5);
    
    // Submit one P0 constraint
    let p0_tile = Tile::new("eager-agent", "P0: Validate inputs");
    room.submit_tile(p0_tile).unwrap();
    
    // Agent attempts to submit P2 optimization directly
    let p2_tile = Tile::new("eager-agent", "P2: Use SIMD instructions");
    
    let result = room.submit_tile(p2_tile);
    assert!(result.is_err());
    assert!(result.unwrap_err().contains("P1 not completed"));
    
    // Verify room remains in P0
    assert_eq!(room.current_protocol_phase(), ProtocolPhase::P0);
}
```

### Test 5: Resource Exhaustion During Optimization
```rust
#[test]
fn test_p2_resource_limits() {
    let mut room = DeadbandRoom::new("resource-test", 100);
    
    // Setup normal P0 → P1 progression
    room.submit_tile(Tile::new("agent", "P0: Check bounds")).unwrap();
    room.submit_tile(Tile::new("agent", "P1: Use channel A")).unwrap();
    room.advance_phase().unwrap(); // To P1
    room.advance_phase().unwrap(); // To P2
    
    // Submit P2 tile that would exceed resource limits
    let large_optimization = "P2: " .repeat(1000); // Artificially large
    let tile = Tile::new("optimizer", &large_optimization);
    
    let result = room.submit_tile(tile);
    
    // Room should either reject or truncate, not crash
    assert!(result.is_ok() || result.is_err());
    // If it succeeds, verify resource tracking
    if result.is_ok() {
        assert!(room.current_memory_usage() < room.memory_limit());
    }
}
```
