"""
Edge-case tests for DeadbandRoom beyond basic protocol validation.
These tests target scenarios identified as gaps in existing coverage.
"""

import pytest
import asyncio
from datetime import datetime, timedelta
from plato_torch.rooms.deadband import DeadbandRoom
from plato_torch.models import AgentState, Tile, ProtocolPhase

class TestDeadbandRoomEdgeCases:
    
    def test_concurrent_agent_tile_submissions(self):
        """Test multiple agents submitting tiles simultaneously.
        
        Edge-case: Race conditions in state updates when multiple agents
        interact with the same room concurrently.
        """
        room = DeadbandRoom(room_id="concurrent_test")
        
        # Create multiple agent states
        agents = [
            AgentState(agent_id=f"agent_{i}", phase=ProtocolPhase.P1, progress=0.0)
            for i in range(5)
        ]
        
        # Simulate concurrent submissions
        tiles = []
        for agent in agents:
            tile = Tile(
                agent_id=agent.agent_id,
                content=f"Test from {agent.agent_id}",
                phase=ProtocolPhase.P1,
                timestamp=datetime.utcnow()
            )
            tiles.append(tile)
        
        # Process submissions (simulating concurrent access)
        results = []
        for tile in tiles:
            try:
                result = room.submit_tile(tile)
                results.append((tile.agent_id, result))
            except Exception as e:
                results.append((tile.agent_id, f"Error: {str(e)}"))
        
        # Verify all submissions were processed without corruption
        assert len(results) == 5
        successful = [r for r in results if r[1] == "accepted"]
        assert len(successful) == 5, f"Expected 5 accepted, got {len(successful)}"
        
        # Verify room state consistency
        assert room.get_agent_count() == 5
        for agent_id in [f"agent_{i}" for i in range(5)]:
            assert room.get_agent_state(agent_id) is not None
    
    def test_protocol_priority_boundary_conditions(self):
        """Test edge cases at protocol phase boundaries.
        
        Edge-case: Agents attempting to operate exactly at phase boundaries
        or with ambiguous phase transitions.
        """
        room = DeadbandRoom(room_id="boundary_test")
        
        # Test P0 with maximum negative space mapping (boundary of completeness)
        agent = AgentState(agent_id="boundary_agent", phase=ProtocolPhase.P0, progress=0.99)
        room.add_agent(agent)
        
        # Submit tile that would complete P0
        tile = Tile(
            agent_id="boundary_agent",
            content="Final P0 negative space mapping",
            phase=ProtocolPhase.P0,
            timestamp=datetime.utcnow()
        )
        
        result = room.submit_tile(tile)
        assert result == "accepted"
        
        # Verify phase transition logic handles boundary correctly
        updated_agent = room.get_agent_state("boundary_agent")
        assert updated_agent.progress >= 1.0
        # Room should allow P1 submission after P0 completion
        # but not before - this tests that boundary condition
        
        # Test ambiguous phase (e.g., progress = 1.0 but phase not updated)
        ambiguous_agent = AgentState(
            agent_id="ambiguous_agent",
            phase=ProtocolPhase.P1,
            progress=1.0  # Completed but phase not advanced
        )
        room.add_agent(ambiguous_agent)
        
        # Room should handle this state gracefully
        # Either reject further P1 tiles or auto-advance phase
        tile2 = Tile(
            agent_id="ambiguous_agent",
            content="Post-completion P1 attempt",
            phase=ProtocolPhase.P1,
            timestamp=datetime.utcnow()
        )
        
        result2 = room.submit_tile(tile2)
        # Acceptable outcomes: rejected with specific error, or accepted with auto-advance
        assert result2 in ["accepted", "rejected"]
        if result2 == "rejected":
            assert "phase" in result2.lower() or "complete" in result2.lower()
    
    def test_resource_exhaustion_scenarios(self):
        """Test room behavior under resource constraints.
        
        Edge-case: Memory exhaustion, storage limits, connection limits.
        """
        room = DeadbandRoom(room_id="resource_test", max_tiles=100, max_agents=50)
        
        # Exhaust tile storage
        for i in range(100):
            agent_id = f"exhaust_agent_{i % 10}"
            tile = Tile(
                agent_id=agent_id,
                content=f"Tile {i} to exhaust storage",
                phase=ProtocolPhase.P1,
                timestamp=datetime.utcnow()
            )
            # Add agent if not exists
            if not room.get_agent_state(agent_id):
                room.add_agent(AgentState(agent_id=agent_id, phase=ProtocolPhase.P1, progress=0.0))
            
            result = room.submit_tile(tile)
            assert result == "accepted", f"Failed at tile {i}"
        
        # 101st tile should be rejected or trigger cleanup
        tile_overflow = Tile(
            agent_id="exhaust_agent_0",
            content="Overflow tile",
            phase=ProtocolPhase.P1,
            timestamp=datetime.utcnow()
        )
        
        result_overflow = room.submit_tile(tile_overflow)
        # Room should either reject or implement LRU eviction
        assert result_overflow in ["rejected", "accepted"]
        if result_overflow == "accepted":
            # Verify eviction occurred (tile count <= max_tiles)
            assert room.get_tile_count() <= 100
