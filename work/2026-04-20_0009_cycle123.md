### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3", "agent_4", "agent_5"]
    
    # Simulate concurrent submissions using threads
    results = []
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "priority": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        results.append((agent_id, result))
    
    threads = []
    for agent_id in agent_ids:
        thread = threading.Thread(target=submit_tile, args=(agent_id,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # All submissions should be processed without data loss
    self.assertEqual(len(results), len(agent_ids))
    
    # Verify each agent's tile was recorded
    for agent_id, result in results:
        self.assertTrue(result["success"])
        self.assertIn(agent_id, room.tile_log)
```

### Test 2: Network Partition with Partial Tile Loss
```python
def test_network_partition_recovery(self):
    """Test room recovery when tiles arrive out-of-order or with gaps."""
    room = DeadbandRoom()
    
    # Simulate network partition: tiles 1, 3, 5 arrive, then 2, 4
    tiles = [
        {"id": 1, "priority": "P0", "content": "Tile 1", "sequence": 1},
        {"id": 3, "priority": "P1", "content": "Tile 3", "sequence": 3},
        {"id": 5, "priority": "P2", "content": "Tile 5", "sequence": 5},
    ]
    
    # Submit out-of-order tiles
    for tile in tiles:
        room.submit_tile(tile)
    
    # Room should buffer incomplete sequences
    self.assertEqual(len(room.pending_tiles), 3)
    
    # Later tiles arrive
    late_tiles = [
        {"id": 2, "priority": "P0", "content": "Tile 2", "sequence": 2},
        {"id": 4, "priority": "P1", "content": "Tile 4", "sequence": 4},
    ]
    
    for tile in late_tiles:
        room.submit_tile(tile)
    
    # All tiles should now be processed in correct order
    self.assertEqual(len(room.pending_tiles), 0)
    self.assertEqual(len(room.processed_tiles), 5)
    
    # Verify sequence order
    sequences = [t["sequence"] for t in room.processed_tiles]
    self.assertEqual(sequences, [1, 2, 3, 4, 5])
```

### Test 3: Priority Protocol Violation Detection
```python
def test_priority_inversion_detection(self):
    """Test that room detects and rejects P2 attempts before P1 completion."""
    room = DeadbandRoom()
    
    # Start with P0 tile (valid)
    p0_tile = {
        "agent_id": "test_agent",
        "priority": "P0",
        "content": "Negative space mapping",
        "phase": "initial"
    }
    p0_result = room.submit_tile(p0_tile)
    self.assertTrue(p0_result["success"])
    
    # Attempt P2 before P1 (should be rejected)
    p2_tile = {
        "agent_id": "test_agent",
        "priority": "P2",
        "content": "Optimization attempt",
        "phase": "premature"
    }
    p2_result = room.submit_tile(p2_tile)
    
    # Should fail with protocol violation error
    self.assertFalse(p2_result["success"])
    self.assertIn("protocol", p2_result.get("error", "").lower())
    self.assertIn("p1", p2_result.get("error", "").lower())
    
    # Now submit valid P1 tile
    p1_tile = {
        "agent_id": "test_agent",
        "priority": "P1",
        "content": "Safe channel finding",
        "phase": "correct"
    }
    p1_result = room.submit_tile(p1_tile)
    self.assertTrue(p1_result["success"])
    
    # Now P2 should be accepted
    p2_tile["phase"] = "valid"
    p2_result = room.submit_tile(p2_tile)
    self.assertTrue(p2_result["success"])
```

### Test 4: Malformed Tile Payload Resilience
```python
def test_malformed_tile_handling(self):
    """Test room resilience against various malformed tile payloads."""
    room = DeadbandRoom()
    
    test_cases = [
        # (tile_payload, expected_error_contains)
        (None, "invalid"),
        ({}, "missing"),
        ({"priority": "P3"}, "invalid priority"),  # Invalid priority level
        ({"priority": "P0"}, "missing content"),  # Missing required fields
        ({"priority": "P1", "content": 12345}, "invalid content"),  # Wrong type
        ({"priority": "P0", "content": "x" * 1000000}, "too large"),  # Size limit
        ({"priority": "P1", "content": "<script>alert()</script>"}, "invalid"),  # XSS attempt
        ({"priority": "P2", "content": "", "metadata": {"nested": {"deep": "x"*5000}}}, "too deep"),  # Deep nesting
    ]
    
    for tile_payload, expected_error in test_cases:
        result = room.submit_tile(tile_payload)
        
        # Should fail gracefully without crashing
        self.assertFalse(result["success"])
        
        # Should provide meaningful error
        if expected_error:
            error_msg = result.get("error", "").lower()
            self.assertIn(expected_error, error_msg)
        
        # Room state should remain stable
        self.assertIsNotNone(room.tile_log)
        self.assertIsNotNone(room.pending_tiles)
```
