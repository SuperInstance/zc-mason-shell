def test_deadband_boundary_constraints():
    """Test P1 channel detection at exact constraint boundaries."""
    config = {
        "constraints": {"max_energy": 100.0, "min_confidence": 0.5},
        "agents": [{"energy": 100.0, "confidence": 0.5}]  # Exactly at boundaries
    }
    room = DeadbandRoom(config)
    result = room.run_protocol()
    # Should find safe channels (P1) since values are inclusive
    assert result["phase"] == "P1"
    assert len(result["channels"]) > 0
```

### Test 2: Mixed Valid/Invalid Input Sequence

def test_mixed_input_sequence():
    """Test protocol with alternating valid/invalid steps."""
    config = {
        "steps": [
            {"type": "P0", "data": {"obstacles": ["rock1", "rock2"]}},  # Valid
            {"type": "P2", "data": {"path": "fast"}},  # Invalid skip
            {"type": "P1", "data": {"constraints": ["safe1"]}},  # Valid
            {"type": "P0", "data": {}}  # Valid empty
        ]
    }
    room = DeadbandRoom(config)
    result = room.run_protocol()
    # Should handle invalid step gracefully, not crash
    assert "errors" in result
    assert len(result["completed_steps"]) == 3  # All but invalid P2
```

### Test 3: Resource Exhaustion Protection
```python
def test_infinite_loop_prevention():
    """Test that P1 channel search has iteration limits."""
    config = {
        "constraints": {"impossible": True},  # No valid channels
        "max_iterations": 1000
    }
    room = DeadbandRoom(config)
    result = room.run_protocol()
    # Should exit P1 after max_iterations, not hang
    assert result["phase"] == "P0"  # Back to mapping negative space
    assert result["iterations"] <= config["max_iterations"]
```

### Test 4: Concurrent Agent Interactions
```python
def test_concurrent_agent_deadband():
    """Test multiple agents interacting with shared constraints."""
    config = {
        "agents": [
            {"id": "agent1", "constraints": {"area": "north"}},
            {"id": "agent2", "constraints": {"area": "south"}},
            {"id": "agent3", "constraints": {"area": "north"}}  # Conflict with agent1
        ]
    }
    room = DeadbandRoom(config)
    result = room.run_protocol()
    # Should identify conflict in P0, find non-overlapping channels in P1
    assert "conflicts" in result.get("P0_output", {})
    assert all(channel["agent_count"] == 1 for channel in result.get("channels", []))
```

### Test 5: Partial Protocol State Persistence
```python
def test_partial_protocol_state():
    """Test resuming from partial P0/P1 completion."""
    partial_state = {
        "P0_complete": True,
        "negative_space": ["rock1", "rock2"],
        "P1_channels_found": ["channel_a"],
        "P2_not_started": True
    }
    config = {"initial_state": partial_state}
    room = DeadbandRoom(config)
    result = room.run_protocol(start_phase="P1")  # Resume from P1
    # Should use existing negative space, not redo P0
    assert result["phase"] == "P2" or result["phase"] == "P1"
    assert "rock1" in result.get("negative_space", [])
    assert not result.get("P0_repeated", False)
```
