### Test 1: Deadband Threshold Boundary Conditions

```python
def test_deadband_threshold_boundaries(self):
    """Test P0/P1 transition at exact deadband boundaries."""
    room = DeadbandRoom(deadband_threshold=0.1)
    
    # Exactly at threshold (should remain in P0)
    state = {"error": 0.1}
    result = room.evaluate(state)
    self.assertEqual(result["phase"], "P0", "Should stay in P0 at threshold")
    
    # Just beyond threshold (should transition to P1)
    state = {"error": 0.1001}
    result = room.evaluate(state)
    self.assertEqual(result["phase"], "P1", "Should transition to P1 beyond threshold")
    
    # Negative boundary (should handle gracefully)
    state = {"error": -0.1}
    result = room.evaluate(state)
    self.assertIn(result["phase"], ["P0", "P1"], "Should handle negative thresholds")
```

### Test 2: Concurrent Agent State Isolation
```python
def test_concurrent_agent_isolation(self):
    """Test multiple agents in same room maintain isolated states."""
    room = DeadbandRoom()
    
    # Agent A enters P1
    state_a = {"agent_id": "A", "error": 0.5, "constraints": ["avoid_rock"]}
    result_a = room.evaluate(state_a)
    
    # Agent B with different state
    state_b = {"agent_id": "B", "error": 0.05, "constraints": ["avoid_edge"]}
    result_b = room.evaluate(state_b)
    
    self.assertNotEqual(result_a["phase"], result_b["phase"],
                       "Agents should have independent phase evaluations")
    self.assertEqual(result_a["agent_id"], "A", "Agent ID should persist")
    self.assertEqual(result_b["agent_id"], "B", "Agent ID should persist")
```

### Test 3: Ghost Tile Persistence Across Cycles
```python
def test_ghost_tile_persistence(self):
    """Test ghost tiles from failed agents persist and influence P0."""
    room = DeadbandRoom()
    
    # Simulate agent failure with P0 violation
    ghost_tile = {
        "agent_id": "ghost_001",
        "phase": "P0",
        "violation": "skipped_to_p2",
        "timestamp": time.time() - 3600  # 1 hour ago
    }
    room.add_ghost_tile(ghost_tile)
    
    # New agent evaluation should consider ghost tile
    state = {"agent_id": "new_agent", "error": 0.3}
    result = room.evaluate(state)
    
    self.assertIn("ghost_warnings", result,
                  "Result should include ghost warnings")
    self.assertGreater(len(result.get("ghost_warnings", [])), 0,
                      "Should have at least one ghost warning")
    
    # Verify ghost tile still exists after multiple cycles
    room.cycle()
    self.assertIn(ghost_tile, room.get_ghost_tiles(),
                  "Ghost tiles should persist across cycles")
```

### Test 4: Protocol Violation Detection

```python
def test_protocol_violation_detection(self):
    """Test detection of P0→P2 skipping violations."""
    room = DeadbandRoom()
    
    # Attempt to skip from P0 directly to P2
    state = {
        "phase": "P0",
        "next_phase": "P2",  # Violation: skipping P1
        "error": 0.8,
        "constraints": []
    }
    
    result = room.evaluate(state)
    
    self.assertTrue(result.get("violation_detected", False),
                   "Should detect protocol violation")
    self.assertEqual(result.get("corrected_phase"), "P1",
                    "Should correct to P1")
    self.assertIn("protocol_error", result,
                  "Should include protocol error details")
    
    # Verify violation is logged
    self.assertGreater(len(room.get_violation_log()), 0,
                      "Violation should be logged")
```

### Test 5: Resource Exhaustion Recovery
```python
def test_resource_exhaustion_recovery(self):
    """Test room behavior when system resources are exhausted."""
    room = DeadbandRoom(max_tiles=100)
    
    # Fill room with tiles to capacity
    for i in range(100):
        room.add_tile({
            "agent_id": f"agent_{i}",
            "phase": "P1",
            "content": f"test_tile_{i}"
        })
    
    # Attempt to add one more tile (should trigger cleanup)
    result = room.add_tile({
        "agent_id": "overflow_agent",
        "phase": "P2",
        "content": "overflow_tile"
    })
    
    self.assertFalse(result["success"],
                    "Should fail to add tile at capacity")
    self.assertEqual(result["reason"], "capacity_exceeded",
                    "Should indicate capacity issue")
    
    # Verify cleanup mechanism activates
    room.cleanup_old_tiles(max_age=60)  # Remove tiles older than 60s
    self.assertLess(len(room.get_tiles()), 100,
                   "Cleanup should reduce tile count")
    
    # Now should be able to add new tile
    result = room.add_tile({
        "agent_id": "recovery_agent",
        "phase": "P1",
        "content": "recovery_tile"
    })
    self.assertTrue(result["success"],
                   "Should succeed after cleanup")
```
