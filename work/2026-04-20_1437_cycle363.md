### Test 1: Concurrent Agent Protocol Adherence
```python
def test_concurrent_agents_protocol_priority():
    """Test that multiple agents in same room maintain strict P0→P1→P2 priority."""
    room = DeadbandRoom(room_id="concurrent_test")
    
    # Simulate two agents submitting tiles simultaneously
    agent1_tiles = [
        {"cycle": 1, "phase": "P0", "content": "Avoid invalid inputs"},
        {"cycle": 2, "phase": "P1", "content": "Found safe API path"}
    ]
    
    agent2_tiles = [
        {"cycle": 1, "phase": "P0", "content": "Don't exceed rate limits"},
        {"cycle": 2, "phase": "P2", "content": "Optimized query"}  # Violation: P2 before P1
    ]
    
    # Process tiles
    results1 = [room.process_tile(tile) for tile in agent1_tiles]
    results2 = [room.process_tile(tile) for tile in agent2_tiles]
    
    # Agent1 should succeed through P0→P1
    assert results1[0]["status"] == "accepted"
    assert results1[1]["status"] == "accepted"
    
    # Agent2's P2 tile should be rejected (protocol violation)
    assert results2[1]["status"] == "rejected"
    assert "protocol violation" in results2[1]["reason"].lower()
```

### Test 3: Boundary Condition Channel Detection
```python
def test_channel_boundary_conditions():
    """Test channel detection at threshold boundaries."""
    room = DeadbandRoom(room_id="boundary_test")
    
    # Test with input at exact channel threshold
    boundary_cases = [
        {"input": 0.0, "expected_channels": ["zero_crossing"]},
        {"input": 1.0, "expected_channels": ["unit_boundary"]},
        {"input": -1.0, "expected_channels": ["negative_unit"]},
        {"input": float("inf"), "expected_channels": []},  # No safe channels
        {"input": float("nan"), "expected_channels": []},  # Invalid input
    ]
    
    for case in boundary_cases:
        tile = {
            "cycle": 1,
            "phase": "P1",
            "content": f"Testing boundary {case['input']}",
            "input_value": case["input"]
        }
        
        result = room.process_tile(tile)
        
        if case["input"] in [float("inf"), float("nan")]:
            assert result["status"] == "rejected"
        else:
            assert result["status"] == "accepted"
            detected_channels = result.get("detected_channels", [])
            assert set(detected_channels) == set(case["expected_channels"])
```

### Test 4: Resource Exhaustion Protection
```python
def test_resource_limits_enforced():
    """Test that room enforces cycle and memory limits."""
    room = DeadbandRoom(room_id="resource_test", max_cycles_per_agent=3, max_memory_kb=10)
    
    # Submit tiles up to cycle limit
    for cycle in range(1, 5):  # 4 cycles, exceeding limit of 3
        tile = {
            "cycle": cycle,
            "phase": "P0",
            "content": f"Cycle {cycle} test",
            "memory_usage": 2  # KB per tile
        }
        
        result = room.process_tile(tile)
        
        if cycle <= 3:
            assert result["status"] == "accepted"
        else:
            assert result["status"] == "rejected"
            assert "cycle limit" in result["reason"].lower()
    
    # Test memory limit
    large_tile = {
        "cycle": 5,
        "phase": "P1",
        "content": "A" * 10240,  # 10KB content
        "memory_usage": 15  # Exceeds 10KB limit
    }
    
    result = room.process_tile(large_tile)
    assert result["status"] == "rejected"
    assert "memory limit" in result["reason"].lower()
```

```python
def test_malformed_tile_handling():
    """Test room handling of tiles with missing/invalid fields."""
    room = DeadbandRoom(room_id="malformed_test")
    
    malformed_tiles = [
        {},  # Empty tile
        {"cycle": 1},  # Missing phase
        {"phase": "P0"},  # Missing cycle
        {"cycle": "invalid", "phase": "P0"},  # Wrong cycle type
        {"cycle": 1, "phase": "INVALID_PHASE"},  # Invalid phase
        {"cycle": 1, "phase": "P0", "content": 123},  # Wrong content type
        {"cycle": -1, "phase": "P0", "content": "Negative cycle"},  # Invalid cycle
    ]
    
    for i, tile in enumerate(malformed_tiles):
        result = room.process_tile(tile)
        assert result["status"] == "rejected"
        assert "invalid" in result["reason"].lower() or "missing" in result["reason"].lower()
        
        # Room should remain operational after malformed tile
        valid_tile = {"cycle": i + 10, "phase": "P0", "content": "Recovery test"}
        recovery_result = room.process_tile(valid_tile)
        assert recovery_result["status"] == "accepted"
```
