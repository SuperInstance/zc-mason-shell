### Test 3: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """P1: Multiple agents submitting tiles simultaneously should not corrupt state."""
    import threading
    
    results = []
    errors = []
    
    def submit_tile(agent_id):
        try:
            tile = Tile(
                content=f"Tile from {agent_id}",
                agent_id=agent_id,
                timestamp=time.time()
            )
            self.room.submit_tile(tile)
            results.append(agent_id)
        except Exception as e:
            errors.append(str(e))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All tiles should be accepted without errors
    self.assertEqual(len(errors), 0)
    self.assertEqual(len(self.room.tiles), 5)
    # State should remain consistent
    self.assertIn(self.room.state, ["active", "processing"])
```

### Test 4: Maximum Tile Capacity Limit
```python
def test_max_tile_capacity(self):
    """P0: Room should reject tiles when at maximum capacity."""
    # Fill room to capacity
    for i in range(self.room.MAX_TILES):
        tile = Tile(
            content=f"Tile {i}",
            agent_id=f"agent_{i}",
            timestamp=time.time()
        )
        self.room.submit_tile(tile)
    
    # Attempt to submit one more tile
    overflow_tile = Tile(
        content="Overflow tile",
        agent_id="overflow_agent",
        timestamp=time.time()
    )
    with self.assertRaises(RuntimeError) as context:
        self.room.submit_tile(overflow_tile)
    self.assertIn("Room at maximum capacity", str(context.exception))
    self.assertEqual(len(self.room.tiles), self.room.MAX_TILES)
```

### Test 5: State Transition Deadlock Detection
```python
def test_state_transition_deadlock_detection(self):
    """P0: Invalid state transition sequence should be detected and prevented."""
    # Force room into processing state
    self.room.state = "processing"
    self.room._processing_lock = True
    
    # Attempt illegal transition back to active
    with self.assertRaises(RuntimeError) as context:
        self.room.transition_state("active")
    self.assertIn("Cannot transition from processing to active", str(context.exception))
    
    # Verify state unchanged
    self.assertEqual(self.room.state, "processing")
```
