### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race():
    """Test that room handles simultaneous tile submissions without data corruption."""
    room = DeadbandRoom(room_id="race_test")
    
    def submit_tile(agent_id, content):
        tile = {
            "agent_id": agent_id,
            "content": content,
            "timestamp": time.time(),
            "phase": "P1"
        }
        return room.submit_tile(tile)
    
    # Simulate 10 concurrent submissions
    threads = []
    results = []
    
    for i in range(10):
        t = threading.Thread(
            target=lambda idx=i: results.append(
                submit_tile(f"agent_{idx}", f"content_{idx}")
            )
        )
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all tiles were accepted
    assert len(results) == 10
    assert all(r["status"] == "accepted" for r in results)
    
    # Verify tile count matches
    assert room.get_tile_count() == 10
    
    # Verify no duplicate or missing content
    contents = [r["content"] for r in results if "content" in r]
    assert len(set(contents)) == 10
```

### Test 3: Corrupted Binary Data Handling
```python
def test_corrupted_binary_data():
    """Test room rejects tiles with corrupted or non-UTF-8 data."""
    room = DeadbandRoom(room_id="binary_test")
    
    # Generate various corrupted data patterns
    corrupted_data = [
        b"\xff\xfe\x00\x00",  # Invalid UTF-8 BOM
        b"\x80\x80\x80",  # Continuation bytes without start
        b"\xc0\xaf",  # Overlong encoding
        b"\xed\xa0\x80",  # Surrogate pair (invalid in UTF-8)
        b"\x00\x00\x00",  # Null bytes
    ]
    
    for i, binary_data in enumerate(corrupted_data):
        tile = {
            "agent_id": f"agent_{i}",
            "content": binary_data,  # Intentionally pass binary
            "timestamp": time.time(),
            "phase": "P1"
        }
        
        result = room.submit_tile(tile)
        assert result["status"] == "rejected", \
            f"Should reject binary data: {binary_data.hex()}"
        
        # Verify error message indicates encoding issue
        if "error" in result:
            assert "encoding" in result["error"].lower() or \
                   "utf-8" in result["error"].lower()
```

### Test 4: Network Partition Simulation
```python
def test_partial_write_recovery():
    """Test room recovers from partial writes (simulated network partition)."""
    room = DeadbandRoom(room_id="partition_test")
    
    # Simulate partial write by mocking file system failure
    original_write = room._write_tile_to_storage
    
    def failing_write(tile_data, attempt=0):
        if attempt < 2:  # Fail first two attempts
            raise IOError("Simulated network partition")
        return original_write(tile_data)
    
    room._write_tile_to_storage = failing_write
    
    tile = {
        "agent_id": "partition_agent",
        "content": "survive partition",
        "timestamp": time.time(),
        "phase": "P2"
    }
    
    # Should succeed on third attempt
    result = room.submit_tile(tile)
    assert result["status"] == "accepted"
    
    # Verify tile actually persisted
    assert room.get_tile_count() == 1
    
    # Restore original method
    room._write_tile_to_storage = original_write
```

    # 3. Raise specific corruption error
