def test_concurrent_tile_submission_race_condition():
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "content": f"Test from {agent_id}",
            "phase": "P1",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (simulated)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results)
    
    # Verify room state consistency
    assert room.tile_count == len(agent_ids)
    assert len(room.agent_sessions) == len(agent_ids)
```

### Test 2: Malformed Tile Structure Recovery

def test_malformed_tile_structure_recovery():
    """Test room's resilience against malformed tile data structures."""
    room = DeadbandRoom()
    
    test_cases = [
        # Missing required fields
        {"agent_id": "test_agent"},
        {"content": "test", "phase": "P1"},
        {"agent_id": "test_agent", "phase": "P1"},
        
        # Invalid data types
        {"agent_id": 123, "content": "test", "phase": "P1"},
        {"agent_id": "test_agent", "content": 456, "phase": "P1"},
        {"agent_id": "test_agent", "content": "test", "phase": 789},
        
        # Invalid phase values
        {"agent_id": "test_agent", "content": "test", "phase": "P3"},
        {"agent_id": "test_agent", "content": "test", "phase": "invalid"},
        
        # Excessively large content
        {"agent_id": "test_agent", "content": "x" * 1000000, "phase": "P1"},
        
        # Nested malicious structures
        {"agent_id": "test_agent", "content": "test", "phase": "P1", "extra": {"nested": "malicious"}},
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(tile)
        # Should reject malformed tiles gracefully
        assert result.get("status") == "rejected"
        assert "error" in result
        # Room state should remain stable
        assert room.tile_count == 0
        assert not room.agent_sessions
```

def test_protocol_phase_skipping_detection():
    """Test detection of agents attempting to skip from P0 directly to P2."""
    room = DeadbandRoom()
    agent_id = "skip_agent"
    
    # Agent tries to submit P2 tile without P1
    tile_p2 = {
        "agent_id": agent_id,
        "content": "Trying to optimize without safe channels",
        "phase": "P2",
        "timestamp": time.time()
    }
    
    result_p2 = room.submit_tile(tile_p2)
    assert result_p2.get("status") == "rejected"
    assert "P1" in result_p2.get("error", "").lower() or "skip" in result_p2.get("error", "").lower()
    
    # Now submit proper P0 tile
    tile_p0 = {
        "agent_id": agent_id,
        "content": "Mapping negative space",
        "phase": "P0",
        "timestamp": time.time()
    }
    
    result_p0 = room.submit_tile(tile_p0)
    assert result_p0.get("status") == "accepted"
    
    # Try P2 again - should still fail without P1

    assert result_p2_again.get("status") == "rejected"
    
    # Submit P1, then P2 should work
    tile_p1 = {
        "agent_id": agent_id,
        "content": "Finding safe channels",
        "phase": "P1",
        "timestamp": time.time()
    }

    assert result_p1.get("status") == "accepted"
    
    result_p2_final = room.submit_tile(tile_p2)

### Test 4: Resource Exhaustion and Cleanup
```python
def test_resource_exhaustion_and_cleanup():
    """Test room behavior under resource constraints and cleanup mechanisms."""
    room = DeadbandRoom()
    
    # Simulate many tile submissions
    max_tiles = 1000
    accepted_count = 0
    
    for i in range(max_tiles):
        tile = {
            "agent_id": f"agent_{i % 10}",  # 10 agents cycling
            "content": f"Tile {i}",
            "phase": "P1",
            "timestamp": time.time()
        }
        
        result = room.submit_tile(tile)
        if result.get("status") == "accepted":
            accepted_count += 1
    
    # Verify room didn't crash
    assert room.tile_count == accepted_count
    
    # Test cleanup of stale agent sessions
    stale_agent = "stale_agent"
    tile_stale = {
        "agent_id": stale_agent,
        "content": "Old tile",
        "phase": "P0",
        "timestamp": time.time() - 3600  # 1 hour old
    }
    
    room.submit_tile(tile_stale)
    assert stale_agent in room.agent_sessions
    
    # Trigger cleanup (simulate room maintenance)
    room._cleanup_stale_sessions(max_age_seconds=1800)  # 30 minutes
    
    # Stale session should be removed
    assert stale_agent not in room.agent_sessions
    
    # Active sessions should remain
    assert len(room.agent_sessions) > 0
```
