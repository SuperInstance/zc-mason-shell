```python
def test_concurrent_agent_interactions():
    """Test DeadbandRoom handling of simultaneous agent submissions."""
    room = DeadbandRoom(room_id="concurrent_test")
    
    # Register multiple agents
    agents = [f"agent_{i}" for i in range(5)]
    for agent_id in agents:
        room.register_agent(agent_id)
    
    # Simulate concurrent submissions
    submissions = []
    for agent_id in agents:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        submissions.append(room.process_interaction(agent_id, tile))
    
    # Verify all submissions processed without corruption
    assert len(room.accumulated_tiles) == 5
    assert all(s["status"] == "accepted" for s in submissions)
    
    # Verify agent-specific state maintained
    for agent_id in agents:
        assert agent_id in room.agent_states
```

### Test 2: Malformed Tile Data Recovery
```python
def test_malformed_tile_recovery():
    """Test room recovery after receiving invalid tile structures."""
    room = DeadbandRoom(room_id="malformed_test")
    room.register_agent("test_agent")
    
    # Test various malformed tile structures
    malformed_tiles = [
        {},  # Empty tile
        {"agent_id": "test_agent"},  # Missing required fields
        {"phase": "P1", "content": "test"},  # Missing agent_id
        {"agent_id": "test_agent", "phase": "P5", "content": "test"},  # Invalid phase
        {"agent_id": "wrong_agent", "phase": "P1", "content": "test"},  # Wrong agent
        None,  # Null tile
        "not_a_dict",  # Wrong type
    ]
    
    expected_errors = []
    for tile in malformed_tiles:
        result = room.process_interaction("test_agent", tile)
        if result["status"] == "rejected":
            expected_errors.append(result["error"])
    
    # Verify room still functional after errors
    assert len(expected_errors) == len(malformed_tiles)
    
    # Valid submission should still work
    valid_tile = {
        "agent_id": "test_agent",
        "phase": "P1",
        "content": "Valid after errors",
        "timestamp": time.time()
    }
    result = room.process_interaction("test_agent", valid_tile)
    assert result["status"] == "accepted"
    assert len(room.accumulated_tiles) == 1
```

### Test 4: Resource Exhaustion Resilience
```python
def test_resource_exhaustion_resilience():
    """Test room behavior under resource constraints."""
    room = DeadbandRoom(room_id="resource_test", max_tiles=10)  # Small limit
    room.register_agent("resource_agent")
    
    # Fill room to capacity
    for i in range(10):
        tile = {
            "agent_id": "resource_agent",
            "phase": "P1",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        result = room.process_interaction("resource_agent", tile)
        assert result["status"] == "accepted"
    
    # Attempt to exceed capacity
    overflow_tile = {
        "agent_id": "resource_agent",
        "phase": "P1",
        "content": "Should be rejected",
        "timestamp": time.time()
    }
    result = room.process_interaction("resource_agent", overflow_tile)
    assert result["status"] == "rejected"
    assert "capacity" in result["error"].lower()
    
    # Verify existing tiles preserved
    assert len(room.accumulated_tiles) == 10
    
    # Test memory-efficient tile storage
    tile_sizes = []
    for tile in room.accumulated_tiles:
        import sys
        tile_sizes.append(sys.getsizeof(str(tile)))
    
    # Ensure no exponential memory growth
    assert max(tile_sizes) - min(tile_sizes) < 1000  # Reasonable variance
```
