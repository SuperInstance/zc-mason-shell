### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race_condition():
    """P0: Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(room_id="race_test", validator=mock_validator)
    
    def submit_tile(agent_id, content):
        return room.submit_tile(
            agent_id=agent_id,
            tile_content=content,
            protocol_phase="P1"
        )
    
    # Simulate 10 concurrent submissions
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = [
            executor.submit(submit_tile, f"agent_{i}", f"content_{i}")
            for i in range(10)
        ]
        results = [f.result() for f in futures]
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    # Room tile count should be accurate
    assert room.get_tile_count() == 10
    # No duplicate tile IDs
    tile_ids = [r["tile_id"] for r in results]
    assert len(set(tile_ids)) == 10
```

### Test 2: Validator Service Partition Mid-Validation
```python
def test_validator_partition_during_validation():
    """P0: Room should handle validator becoming unavailable during tile validation."""
    validator = Mock(spec=TileValidator)
    validator.validate_tile.side_effect = [
        {"valid": True, "score": 0.8},
        ConnectionError("Validator service unreachable"),
        {"valid": True, "score": 0.7}
    ]
    
    room = DeadbandRoom(room_id="partition_test", validator=validator)
    
    # First submission should succeed
    result1 = room.submit_tile("agent1", "content1", "P1")
    assert result1["status"] == "accepted"
    
    # Second should fail gracefully
    result2 = room.submit_tile("agent2", "content2", "P1")
    assert result2["status"] == "rejected"
    assert "validator_unavailable" in result2["reason"]
    
    # Third should succeed after validator returns
    result3 = room.submit_tile("agent3", "content3", "P1")
    assert result3["status"] == "accepted"
    
    # Room should maintain consistent state
    assert room.get_tile_count() == 2
```

### Test 4: Session State Persistence After Disconnect
```python
def test_session_state_persistence_reconnection():
    """P1: Agent disconnects and reconnects, room should maintain session state."""
    room = DeadbandRoom(room_id="session_test", validator=mock_validator)
    session_id = "agent_alpha_session_001"
    
    # Initial connection and P1 submission
    room.register_agent("agent_alpha", session_id)
    result1 = room.submit_tile(
        agent_id="agent_alpha",
        tile_content={"p1_mapping": "safe_channel_identified"},
        protocol_phase="P1",
        session_id=session_id
    )
    assert result1["status"] == "accepted"
    
    # Simulate disconnect (session remains in room)
    # Agent reconnects with same session ID
    room.register_agent("agent_alpha", session_id)
    
    # Should be able to continue from P2 (since P1 established)
    result2 = room.submit_tile(
        agent_id="agent_alpha",
        tile_content={"p2_optimization": "improved_path"},
        protocol_phase="P2",
        session_id=session_id
    )
    assert result2["status"] == "accepted"
    
    # Session should track protocol phase progression
    session = room.get_session(session_id)
    assert session["current_phase"] == "P2"
    assert session["tile_count"] == 2
```

### Test 5: Memory Pressure During Batch Processing
```python
def test_memory_pressure_large_batch():
    """P0: Room should handle memory constraints during large tile batch processing."""
    room = DeadbandRoom(
        room_id="memory_test",
        validator=mock_validator,
        max_tiles_per_batch=1000,
        max_memory_mb=50
    )
    
    large_content = {"data": "x" * 1024 * 1024}  # 1MB per tile
    
    results = []
    for i in range(100):
        try:
            result = room.submit_tile(
                agent_id=f"batch_agent_{i}",
                tile_content=large_content.copy(),
                protocol_phase="P1"
            )
            results.append(result)
        except MemoryError:
            # Room should fail gracefully when hitting memory limits
            assert room.get_tile_count() < 100
            assert room._memory_usage_mb > 45  # Should be near limit
            break
    
    # Should have processed some tiles before hitting limit
    assert len(results) > 0
    assert len(results) < 100
    
    # Room should still be operational for smaller tiles
    small_result = room.submit_tile(
        agent_id="recovery_agent",
        tile_content={"data": "small"},
        protocol_phase="P1"
    )
    assert small_result["status"] == "accepted"
```
