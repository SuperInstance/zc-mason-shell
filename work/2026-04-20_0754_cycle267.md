### Test 1: P0 Negative Space Overflow
**Scenario:** Agent attempts to map negative space with excessively large state that exceeds memory limits.
**Purpose:** Test memory boundary conditions and graceful degradation.
```python
def test_p0_negative_space_overflow(self):
    """Test P0 phase with state size exceeding configured limits."""
    room = DeadbandRoom(max_state_size=1000)
    # Create state larger than limit
    oversized_state = {"negative_space": ["rock"] * 10000}
    
    result = room.process_cycle(agent_id="test_agent", state=oversized_state)
    
    # Should handle gracefully, not crash
    assert "error" in result or "warning" in result
    assert result.get("phase") == "P0"  # Should still attempt P0
    assert len(result.get("negative_space", [])) <= 1000  # Should be truncated
```

### Test 2: P1 Channel Boundary Collision
**Scenario:** Agent finds safe channels that intersect with previously mapped negative space.
**Purpose:** Test protocol's ability to detect and handle contradictory information.
```python
def test_p1_channel_boundary_collision(self):
    """Test when safe channels intersect with previously mapped negative space."""
    room = DeadbandRoom()
    
    # First cycle: Map negative space
    state1 = {"negative_space": ["rock_at_0_0", "rock_at_1_1"]}
    result1 = room.process_cycle(agent_id="test_agent", state=state1)
    
    # Second cycle: Try to find safe channel that includes negative space
    state2 = {
        "negative_space": result1.get("negative_space", []),
        "proposed_channels": [{"coordinates": [0, 0], "type": "safe"}]  # Collision!
    }
    result2 = room.process_cycle(agent_id="test_agent", state=state2)
    
    # Should reject the colliding channel
    assert "rejected_channels" in result2
    assert len(result2.get("rejected_channels", [])) > 0
    assert result2.get("phase") == "P1"
```

### Test 3: P2 Optimization with Degrading Resources
**Scenario:** Agent optimizes path within valid channel while resources (energy, time) degrade.
**Purpose:** Test optimization under constraints and failure modes.
```python
def test_p2_optimization_with_degrading_resources(self):
    """Test P2 optimization when resources are critically low."""
    room = DeadbandRoom()
    
    state = {
        "phase": "P2",
        "safe_channels": [{"path": [[0,0], [1,1], [2,2]], "cost": 10}],
        "resources": {"energy": 0.1, "time": 0.05},  # Critically low
        "optimization_target": "minimize_cost"
    }
    
    result = room.process_cycle(agent_id="test_agent", state=state)
    
    # Should either:
    # 1. Return emergency protocol
    # 2. Return conservative optimization
    # 3. Signal resource exhaustion
    assert "optimized_path" in result or "emergency_protocol" in result
    if "optimized_path" in result:
        assert result["optimized_path"]["estimated_cost"] <= 10  # Should not exceed original
```

### Test 4: Concurrent Agent Priority Conflict
**Scenario:** Multiple agents attempt to use the same channel simultaneously with conflicting priorities.
**Purpose:** Test room's handling of multi-agent contention.
```python
def test_concurrent_agent_priority_conflict(self):
    """Test when multiple agents conflict on channel usage."""
    room = DeadbandRoom()
    
    # Simulate concurrent requests (in reality would be async)
    agent1_state = {
        "phase": "P2",
        "safe_channels": [{"id": "channel_a", "priority": 1}],
        "requested_channel": "channel_a"
    }
    
    agent2_state = {
        "phase": "P2", 
        "safe_channels": [{"id": "channel_a", "priority": 2}],
        "requested_channel": "channel_a"
    }
    
    result1 = room.process_cycle(agent_id="agent_1", state=agent1_state)
    result2 = room.process_cycle(agent_id="agent_2", state=agent2_state)
    
    # Room should mediate conflict
    # Higher priority should win, or both should get conflict resolution
    assert result1.get("channel_assignment") != result2.get("channel_assignment")
    # Or conflict flag should be set
    assert any(r.get("conflict_detected", False) for r in [result1, result2])
```

### Test 5: Protocol Regression from P2 to P0
**Scenario:** Agent successfully reaches P3 (completion), then encounters new negative space requiring regression to P0.
**Purpose:** Test protocol's ability to handle regression and re-evaluation.
```python
def test_protocol_regression_p2_to_p0(self):
    """Test when agent must regress from P2 back to P0 due to new obstacles."""
    room = DeadbandRoom()
    
    # Complete a full cycle
    state1 = {"negative_space": ["rock1"], "phase": "P0"}
    result1 = room.process_cycle(agent_id="test_agent", state=state1)
    
    state2 = {"safe_channels": ["channel1"], "phase": "P1", **result1}
    result2 = room.process_cycle(agent_id="test_agent", state=state2)
    
    state3 = {"optimized_path": "path1", "phase": "P2", **result2}
    result3 = room.process_cycle(agent_id="test_agent", state=state3)
    
    # Now encounter new negative space requiring regression
    state4 = {
        **result3,
        "phase": "P2",
        "new_negative_space": ["rock2"],  # New obstacle in current path
        "current_position": "mid_path"
    }
    
    result4 = room.process_cycle(agent_id="test_agent", state=state4)
    
    # Should trigger regression to P0 or P1
    assert result4.get("phase") in ["P0", "P1"]
    assert "regression_required" in result4
    assert "rock2" in str(result4.get("negative_space", []))
```
