## Identified Edge Cases for Additional Tests
Following the Deadband Protocol's strict priority (P0→P1→P2, never skip), I've identified 5 edge cases not covered in the existing test suite:

### Edge Case 1: Overlapping Constraint Spaces
**Scenario:** P1 constraints overlap with P0 negative space (contradictory constraints)
**Test Purpose:** Verify the room handles contradictory constraints without crashing
**Expected Behavior:** Room should raise a validation error during initialization or prioritize P0 over P1

### Edge Case 2: Empty P1 Channel
**Scenario:** P0 constraints defined, but P1 returns empty list (no safe channels)
**Test Purpose:** Test protocol behavior when no P1 path exists
**Expected Behavior:** Room should remain at P0 phase indefinitely or raise specific error

### Edge Case 3: Maximum Constraint Limits
**Scenario:** Extremely large constraint sets (10,000+ constraints per phase)
**Test Purpose:** Test performance and memory handling
**Expected Behavior:** Room should initialize and execute within reasonable time/memory bounds

### Edge Case 4: Concurrent Protocol Execution
**Scenario:** Multiple agents attempting to advance phases simultaneously
**Test Purpose:** Test thread safety in multi-agent environments
**Expected Behavior:** Protocol state should remain consistent; no race conditions

### Edge Case 5: Invalid Constraint Types
**Scenario:** Constraints that aren't callable or don't return boolean
**Test Purpose:** Test input validation and error handling
**Expected Behavior:** Room should raise TypeError during initialization

# plato-torch/tests/test_deadband_room_edge_cases.py
"""
Edge-case tests for DeadbandRoom preset.
These tests cover scenarios beyond basic protocol execution.
"""

import pytest
import time
from threading import Thread
from plato_torch.rooms.deadband_room import DeadbandRoom


def test_overlapping_constraint_spaces():
    """Edge Case 1: P1 constraints overlap with P0 negative space."""
    
    # Define contradictory constraints
    def p0_constraint(state):
        return state.get("x", 0) > 5  # Negative space: x > 5
    
    def p1_constraint(state):
        return state.get("x", 0) > 3  # Safe channel includes x > 5 (overlaps with P0!)
    
    # Room should reject contradictory constraints
    with pytest.raises(ValueError, match="contradict|overlap"):
        room = DeadbandRoom(
            p0_constraints=[p0_constraint],
            p1_constraints=[p1_constraint],
            p2_constraints=[lambda s: True]
        )


def test_empty_p1_channel():
    """Edge Case 2: No safe channels after P0 mapping."""
    
    def p0_constraint(state):
        return state.get("x", 0) < 10  # Negative space: x < 10
    
    def p1_constraint(state):
        return False  # No safe channels exist!
    
    room = DeadbandRoom(
        p0_constraints=[p0_constraint],
        p1_constraints=[p1_constraint],
        p2_constraints=[lambda s: True]
    )
    
    # Initial state violates P0 (x=0, which is <10)
    room.step({"x": 0})
    
    # Should be stuck at P0 phase since no P1 path exists
    assert room.current_phase == "P0"
    
    # Attempting to advance should fail or do nothing
    room.step({"x": 15})  # Now satisfies P0 (x=15 > 10)
    assert room.current_phase == "P0"  # Still stuck because P1 returns False


def test_maximum_constraint_limits():
    """Edge Case 3: Performance with large constraint sets."""
    
    # Generate 10,000 simple constraints
    large_p0 = [lambda s, i=i: s.get("x", 0) > i for i in range(10000)]
    large_p1 = [lambda s, i=i: s.get("x", 0) < 20000 - i for i in range(10000)]
    large_p2 = [lambda s, i=i: s.get("x", 0) == 10000 for i in range(10000)]
    
    start_time = time.time()
    
    room = DeadbandRoom(
        p0_constraints=large_p0,
        p1_constraints=large_p1,
        p2_constraints=large_p2
    )
    
    initialization_time = time.time() - start_time
    
    # Should initialize within 2 seconds (adjust based on fleet hardware)
    assert initialization_time < 2.0, f"Initialization too slow: {initialization_time}s"
    
    # Test execution time
    start_time = time.time()
    room.step({"x": 15000})
    execution_time = time.time() - start_time
    
    assert execution_time < 1.0, f"Step execution too slow: {execution_time}s"


def test_concurrent_protocol_execution():
    """Edge Case 4: Thread safety with multiple agents."""
    
    room = DeadbandRoom(
        p0_constraints=[lambda s: s.get("x", 0) > 5],
        p1_constraints=[lambda s: s.get("x", 0) < 20],
        p2_constraints=[lambda s: s.get("x", 0) == 15]
    )
    
    results = []
    errors = []
    
    def agent_thread(agent_id, state_value):
        try:
            # Each agent attempts to advance the protocol
            room.step({"x": state_value})
            results.append((agent_id, room.current_phase))
        except Exception as e:
            errors.append((agent_id, str(e)))
    
    # Create multiple threads simulating concurrent agents
    threads = []
    test_cases = [("agent1", 10), ("agent2", 12), ("agent3", 15)]
    
    for agent_id, state_value in test_cases:
        thread = Thread(target=agent_thread, args=(agent_id, state_value))
        threads.append(thread)
        thread.start()
    
    # Wait for all threads
    for thread in threads:
        thread.join()
    
    # Verify no errors occurred
    assert len(errors) == 0, f"Concurrency errors: {errors}"
    
    # Verify protocol state is consistent
    # All agents should see the same final phase
    final_phases = [phase for _, phase in results]
    assert len(set(final_phases)) == 1, f"Inconsistent phases: {final_phases}"


def test_invalid_constraint_types():
    """Edge Case 5: Non-callable or malformed constraints."""
    
    # Test with non-callable constraint
    with pytest.raises(TypeError, match="callable"):
        room = DeadbandRoom(
            p0_constraints=["not a function"],  # String instead of callable
            p1_constraints=[lambda s: True],
            p2_constraints=[lambda s: True]
        )
    
    # Test with constraint that doesn't return boolean
    def bad_constraint(state):
        return "not a boolean"
    
    room = DeadbandRoom(
        p0_constraints=[bad_constraint],
        p1_constraints=[lambda s: True],
        p2_constraints=[lambda s: True]
    )
    
    # Room should handle this gracefully or raise error during execution
    with pytest.raises(TypeError, match="bool"):
        room.step({"x": 10})
