"""
Edge-case tests for DeadbandRoom.
Focus on boundary conditions, error paths, and protocol violations.
"""

import pytest
import numpy as np
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import TrainingTile

class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""

    def test_p0_violation_with_valid_p1_channel(self):
        """Test: Agent submits P2 tile while P0 violation exists but P1 channel is available.
        
        Scenario: Room has active P0 violation (negative space), but agent identifies
        a safe P1 channel. Protocol requires P1 before P2, even if P0 exists.
        """
        room = DeadbandRoom(room_id="edge_test_1")
        
        # Simulate P0 violation detection
        room.negative_space = ["rock:avoid_absolute_claims"]
        room.current_priority = "P0"
        
        # Agent submits P1 tile (safe channel identification)
        p1_tile = TrainingTile(
            agent_id="test_agent",
            content="P1: Found safe channel in documentation review",
            priority="P1",
            metadata={"channel_type": "doc_review"}
        )
        
        result = room.submit_tile(p1_tile)
        assert result["accepted"] == True
        assert room.current_priority == "P1"
        assert len(room.tiles) == 1

    def test_tile_with_malformed_priority_sequence(self):
        """Test: Agent submits tiles out of P0→P1→P2 sequence.
        
        Scenario: Agent tries to jump from P0 directly to P2, skipping P1.
        Protocol should reject the P2 tile.
        """
        room = DeadbandRoom(room_id="edge_test_2")
        room.current_priority = "P0"
        
        p2_tile = TrainingTile(
            agent_id="test_agent",
            content="P2: Optimizing channel without P1",
            priority="P2",
            metadata={"optimization": "premature"}
        )
        
        result = room.submit_tile(p2_tile)
        assert result["accepted"] == False
        assert "P1" in result.get("reason", "")  # Rejection should mention P1 requirement

    def test_concurrent_tile_submission_race(self):
        """Test: Multiple agents submit tiles simultaneously at priority boundaries.
        
        Scenario: Two agents identify different P1 channels at same time.
        Room should handle concurrent updates without corruption.
        """
        room = DeadbandRoom(room_id="edge_test_3")
        room.current_priority = "P0"
        
        # Simulate concurrent submissions
        tiles = [
            TrainingTile(
                agent_id="agent_a",
                content="P1: Channel A - file analysis",
                priority="P1",
                metadata={"channel": "file_analysis"}
            ),
            TrainingTile(
                agent_id="agent_b",
                content="P1: Channel B - api_review",
                priority="P1",
                metadata={"channel": "api_review"}
            )
        ]
        
        # Submit both (simulating near-concurrent timing)
        results = []
        for tile in tiles:
            results.append(room.submit_tile(tile))
        
        # Only one should be accepted (first mover)
        accepted = [r for r in results if r["accepted"]]
        assert len(accepted) == 1
        
        # Room should now be at P1
        assert room.current_priority == "P1"

    def test_ghost_tile_influence_on_current_priority(self):
        """Test: Ghost tiles (P0 knowledge from dead agents) affect new agent's submissions.
        
        Scenario: Room contains ghost tiles with P0 violations. New agent's
        P1 submission should still be accepted if it avoids those violations.
        """
        room = DeadbandRoom(room_id="edge_test_4")
        
        # Add ghost tile (simulated from dead agent)
        ghost_tile = TrainingTile(
            agent_id="ghost_agent",
            content="P0: Avoid hallucinating file contents",
            priority="P0",
            metadata={"violation": "hallucination", "ghost": True}
        )
        room.ghost_tiles.append(ghost_tile)
        room.negative_space.append("hallucination")
        
        # New agent submits P1 tile that respects ghost P0
        p1_tile = TrainingTile(
            agent_id="live_agent",
            content="P1: Reading actual files from fleet-knowledge/",
            priority="P1",
            metadata={"action": "file_reading"}
        )
        
        result = room.submit_tile(p1_tile)
        assert result["accepted"] == True
        assert "hallucination" not in p1_tile.content

    def test_tile_with_boundary_metadata_size(self):
        """Test: Tile metadata at size limits (very small, very large).
        
        Scenario: Agent submits tile with minimal metadata (empty dict) and
        maximal metadata (large nested structure). Room should handle both.
        """
        room = DeadbandRoom(room_id="edge_test_5")
        room.current_priority = "P1"
        
        # Minimal metadata
        tile_min = TrainingTile(
            agent_id="min_agent",
            content="P2: Optimize with no metadata",
            priority="P2",
            metadata={}
        )
        
        # Large metadata (simulating complex state)
        large_metadata = {
            "channel": "integration_test",
            "dependencies": ["repo_a", "repo_b", "repo_c"] * 10,
            "config": {f"key_{i}": f"value_{i}" for i in range(50)},
            "metrics": {"coverage": 0.95, "latency": 120.5, "errors": []}
        }
        tile_max = TrainingTile(
            agent_id="max_agent",
            content="P2: Optimize with large metadata",
            priority="P2",
            metadata=large_metadata
        )
        
        # Both should be accepted if protocol allows P2
        result_min = room.submit_tile(tile_min)
        result_max = room.submit_tile(tile_max)
        
        assert result_min["accepted"] == True
        assert result_max["accepted"] == True
        assert len(room.tiles) == 2

if __name__ == "__main__":
    pytest.main([__file__, "-v"])
