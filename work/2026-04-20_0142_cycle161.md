```python
def test_memory_exhaustion_protection():
    """Test room behavior when tile storage approaches limits."""
    room = DeadbandRoom(max_tiles=100)
    
    # Flood with valid tiles
    for i in range(150):
        tile = {
            "phase": "P0",
            "constraint": f"test_constraint_{i}",
            "timestamp": time.time()
        }
        
        if i < 100:
            # First 100 should succeed
            assert room.process_tile(tile) is True
        else:
            # After 100, should reject or trigger cleanup

            if result is False:
```

### Test 4: Network Partition Simulation
```python
def test_partial_connectivity_scenario():
    """Test room behavior when agents have intermittent connectivity."""
    room = DeadbandRoom()
    
    # Simulate network partition
    room.network_status = "partitioned"
    
    # Agents submit tiles during partition
    agent_tiles = [
        {"phase": "P1", "channel": "safe_path_a", "agent_id": "agent1"},
        {"phase": "P1", "channel": "safe_path_b", "agent_id": "agent2"},
    ]
    
    # Tiles should queue during partition
    for tile in agent_tiles:
        room.queue_tile(tile)
    
    assert len(room.pending_tiles) == len(agent_tiles)
    
    # Network recovery
    room.network_status = "connected"
    room.process_pending_tiles()
    
    # Queued tiles should process in order
    assert len(room.pending_tiles) == 0
    assert room.state["phase"] == "P1"
```

### Test 5: State Persistence After Crash

```python
def test_state_recovery_after_crash():
    """Test room can recover state from persistent storage after failure."""
    # Initial room state
    room1 = DeadbandRoom()
    room1.process_tile({"phase": "P1", "channel": "test_channel"})
    room1.process_tile({"phase": "P2", "optimization": "test_optimization"})
    
    # Simulate crash and persistence
    saved_state = room1.save_state()
    
    # New room instance loading saved state
    room2 = DeadbandRoom()
    room2.load_state(saved_state)
    
    # Should maintain phase and channel
    assert room2.state["phase"] == "P2"
    assert room2.state["current_channel"] == "test_channel"
    assert len(room2.processed_tiles) == 2
    
    # Should continue processing from saved state
    result = room2.process_tile({"phase": "P2", "optimization": "continued"})
    assert result is True
```
