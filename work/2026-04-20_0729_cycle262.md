### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_processing(self):
    """P1: Room should handle simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    tiles = [
        {"agent_id": aid, "content": f"test from {aid}", "type": "observation"}
        for aid in agent_ids
    ]
    
    # Simulate concurrent submission (thread pool)
    from concurrent.futures import ThreadPoolExecutor
    results = []
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = [executor.submit(room.process_tile, tile) for tile in tiles]
        results = [f.result() for f in futures]
    
    # All tiles should be accepted
    assert all(r["status"] == "accepted" for r in results)
    # State should reflect all three tiles
    state = room.get_state()
    assert len(state["processed_tiles"]) == 3
    assert set(t["agent_id"] for t in state["processed_tiles"]) == set(agent_ids)
```

```python
def test_tile_flood_performance(self):
    """P0: Room should not degrade significantly under high tile volume."""
    room = DeadbandRoom()
    start_time = time.time()
    
    # Submit 1000 valid tiles
    for i in range(1000):
        tile = {
            "agent_id": f"stress_agent_{i}",
            "content": f"Stress test tile {i}",
            "type": "observation",
            "timestamp": time.time()
        }
        result = room.process_tile(tile)
        assert result["status"] == "accepted"
    
    end_time = time.time()
    processing_time = end_time - start_time
    
    # Performance check: < 0.1 seconds per tile on average
    assert processing_time < 100.0, f"Processing too slow: {processing_time}s for 1000 tiles"
    
    # State integrity check
    state = room.get_state()
    assert len(state["processed_tiles"]) == 1000
    assert state["metrics"]["tiles_processed"] == 1000
```

### Test 3: Protocol Violation Recovery
```python
def test_p0_violation_recovery(self):
    """P0: Room should reject P0-violating tiles but continue operating."""
    room = DeadbandRoom()
    
    # First, a valid tile to establish baseline
    valid_tile = {
        "agent_id": "test_agent",
        "content": "Valid observation",
        "type": "observation"
    }
    valid_result = room.process_tile(valid_tile)
    assert valid_result["status"] == "accepted"
    
    # P0 violation: tile claiming to skip to P2 directly
    p0_violation_tile = {
        "agent_id": "reckless_agent",
        "content": "Skipping to optimization without mapping negative space",
        "type": "protocol_violation",
        "priority": "P2",  # Missing P0 and P1
        "violation": True
    }
    violation_result = room.process_tile(p0_violation_tile)
    assert violation_result["status"] == "rejected"
    assert "P0" in violation_result.get("reason", "")
    
    # Room should still accept new valid tiles after rejection
    another_valid = {
        "agent_id": "cautious_agent",
        "content": "Following protocol properly",
        "type": "observation"
    }
    final_result = room.process_tile(another_valid)
    assert final_result["status"] == "accepted"
    
    # Only two tiles should be in processed state (first and last)
    state = room.get_state()
    assert len(state["processed_tiles"]) == 2
    assert state["rejected_tiles"] == 1
```

```python
def test_semantically_invalid_tile(self):
    """P0: Tiles with valid schema but invalid semantics should be rejected."""
    room = DeadbandRoom()
    
    # Schema-valid but semantically nonsense
    semantic_violations = [
        {
            "agent_id": "confused_agent",
            "content": "",  # Empty content
            "type": "observation"
        },
        {
            "agent_id": "looping_agent",
            "content": "This is a test " * 1000,  # Excessive repetition
            "type": "observation"
        },
        {
            "agent_id": "contradictory_agent",
            "content": "P0: Do nothing. P1: Do everything.",
            "type": "protocol_guidance",
            "contradiction": True
        }
    ]
    
    rejection_count = 0
    for tile in semantic_violations:
        result = room.process_tile(tile)
        if result["status"] == "rejected":
            rejection_count += 1
            assert any(keyword in result.get("reason", "").lower() 
                      for keyword in ["semantic", "invalid", "empty", "contradict"])
    
    # At least 2 of 3 should be rejected for semantic issues
    assert rejection_count >= 2, f"Only {rejection_count}/3 semantic violations caught"
```

### Test 5: State Persistence Boundary
```python
def test_state_persistence_boundary(self):
    """P1: Room state should maintain integrity at configured limits."""
    room = DeadbandRoom()
    
    # Check default configuration
    config = room.get_config()
    max_tiles = config.get("max_tiles_memory", 10000)
    
    # Fill to just below limit
    for i in range(max_tiles - 10):
        tile = {
            "agent_id": f"boundary_agent_{i}",
            "content": f"Tile {i} for boundary testing",
            "type": "observation"
        }
        room.process_tile(tile)
    
    state = room.get_state()
    assert len(state["processed_tiles"]) == max_tiles - 10
    
    # Add 20 more tiles - room should handle overflow according to policy
    overflow_results = []
    for i in range(20):
        tile = {
            "agent_id": f"overflow_agent_{i}",
            "content": f"Overflow tile {i}",
            "type": "observation"
        }
        overflow_results.append(room.process_tile(tile))
    
    # Check overflow handling (should be defined in room config)
    overflow_policy = config.get("overflow_policy", "reject_oldest")
    
    if overflow_policy == "reject_oldest":
        # Oldest tiles should be removed, newest accepted
        accepted = sum(1 for r in overflow_results if r["status"] == "accepted")
        assert accepted > 0, "No overflow tiles accepted"
        
        final_state = room.get_state()
        # Should be at or near max_tiles
        assert len(final_state["processed_tiles"]) <= max_tiles + 5
    elif overflow_policy == "reject_new":
        # New overflow tiles should be rejected
        rejected = sum(1 for r in overflow_results if r["status"] == "rejected")
        assert rejected >= 15, f"Only {rejected}/20 overflow tiles rejected"
```
