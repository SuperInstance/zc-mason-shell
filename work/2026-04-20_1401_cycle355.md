### 1. Test Deadband Threshold Boundary (Float Precision)

```python
def test_deadband_threshold_float_precision(self):
    """Test that deadband thresholds handle floating-point precision correctly."""
    # Set thresholds at extreme precision boundaries
    room = DeadbandRoom(
        room_id="edge_precision",
        deadband_lower=0.000001,  # 1e-6
        deadband_upper=0.999999,
        max_tiles=1000
    )
    
    # Add tile with value just outside lower deadband
    tile = {"value": 0.0000009, "phase": "P0", "timestamp": "2026-04-20T14:00:00Z"}
    result = room.add_tile(tile)
    # Should be accepted (outside deadband)
    self.assertTrue(result["accepted"])
    
    # Add tile with value just inside deadband
    tile = {"value": 0.0000011, "phase": "P0", "timestamp": "2026-04-20T14:00:01Z"}
    result = room.add_tile(tile)
    # Should be rejected (inside deadband)
    self.assertFalse(result["accepted"])
    self.assertEqual(result["reason"], "value_within_deadband")
```

### 2. Test Max Tiles Enforcement with Concurrent Adds
```python
def test_max_tiles_concurrent_enforcement(self):
    """Test that max_tiles limit is strictly enforced under rapid tile addition."""
    room = DeadbandRoom(
        room_id="edge_concurrent",
        deadband_lower=0.1,
        deadband_upper=0.9,
        max_tiles=5  # Small limit for testing
    )
    
    # Rapidly add 7 tiles (exceeds limit)
    tiles = []
    for i in range(7):
        tile = {
            "value": 0.2 + (i * 0.1),  # Values outside deadband
            "phase": "P1",
            "timestamp": f"2026-04-20T14:00:{i:02d}Z"
        }
        tiles.append(tile)
    
    accepted_count = 0
    for tile in tiles:
        result = room.add_tile(tile)
        if result["accepted"]:
            accepted_count += 1
    
    # Should accept exactly max_tiles
    self.assertEqual(accepted_count, 5)
    self.assertEqual(len(room.tiles), 5)
    
    # Verify oldest tiles are preserved (FIFO behavior)
    self.assertEqual(room.tiles[0]["value"], 0.2)
    self.assertEqual(room.tiles[4]["value"], 0.6)
```

```python
def test_tile_nested_injection_attempt(self):

    room = DeadbandRoom(
        room_id="edge_injection",
        deadband_lower=0.25,
        deadband_upper=0.75,

    # Should be rejected due to invalid value type (not numeric)
    self.assertFalse(result["accepted"])
    self.assertIn("invalid", result["reason"].lower())

    self.assertEqual(len(room.tiles), 0)
```
