## New Edge-Case Tests

### Test 1: Agent Submits P2 Tile Without P1 Completion
**Scenario:** An agent attempts to submit a P2-optimization tile before the room has recorded any P1-safe-channel tiles.
**Expected:** Tile rejected, room remains in P1, error logged.
**Rationale:** Ensures strict priority enforcement (P0 → P1 → P2).

```python
def test_p2_tile_without_p1_completion(room, agent):
    """Submit P2 tile before any P1 tiles are accepted."""
    # Room starts at P0
    assert room.current_phase == "P0"
    
    # Submit P0 tile (negative space mapping)
    p0_tile = {"phase": "P0", "content": "Avoid rock X"}
    room.submit_tile(agent, p0_tile)
    assert room.current_phase == "P1"  # Room advances to P1
    
    # Attempt P2 tile directly
    p2_tile = {"phase": "P2", "content": "Optimize path Y"}
    result = room.submit_tile(agent, p2_tile)
    
    # Should be rejected
    assert result["accepted"] is False
    assert "P1" in result.get("error", "")  # Error mentions missing P1
    assert room.current_phase == "P1"  # Phase unchanged
    assert room.tiles_accepted == 1  # Only P0 tile accepted
```

### Test 2: Malformed Tile (Missing Phase Key)
**Scenario:** Agent submits a tile dictionary lacking the required "phase" key.
**Expected:** Tile rejected, room state unchanged, validation error.
**Rationale:** Tests input validation robustness.

```python
def test_malformed_tile_missing_phase(room, agent):
    """Tile missing 'phase' key should be rejected."""
    initial_phase = room.current_phase
    initial_count = room.tiles_accepted
    
    malformed_tile = {"content": "Some content"}  # No phase
    result = room.submit_tile(agent, malformed_tile)
    
    assert result["accepted"] is False
    assert "phase" in result.get("error", "").lower()
    assert room.current_phase == initial_phase
    assert room.tiles_accepted == initial_count
```

### Test 3: Rapid-Fire Tile Submission from Multiple Agents
**Scenario:** Two agents submit tiles in quick succession during phase transition.
**Expected:** Room handles concurrency, maintains consistent state, no race conditions.
**Rationale:** Tests thread safety in simulated multi-agent environments.

```python
def test_concurrent_tile_submission(room, agent_factory):
    """Two agents submitting tiles during phase transition."""
    agent1 = agent_factory()
    agent2 = agent_factory()
    
    # Both submit P0 tiles
    p0_tile1 = {"phase": "P0", "content": "Avoid A"}
    p0_tile2 = {"phase": "P0", "content": "Avoid B"}
    
    # Simulate near-simultaneous submission
    result1 = room.submit_tile(agent1, p0_tile1)
    result2 = room.submit_tile(agent2, p0_tile2)
    
    # One should be accepted, the other may be rejected or accepted based on timing
    # But room should remain consistent
    assert room.tiles_accepted >= 1
    assert room.current_phase in ["P0", "P1"]
    # Ensure no crash or state corruption
    assert room._validate_state()  # Internal consistency check
```

### Test 5: Room Reset After Protocol Completion
**Scenario:** Room completes P0→P1→P2 cycle, then receives reset signal.
**Expected:** Room returns to P0, tiles_accepted reset to 0, clean state.
**Rationale:** Tests room lifecycle and reset functionality.

```python
def test_room_reset_after_completion(room, agent):
    """Complete protocol cycle then reset room."""
    # Complete all phases
    p0_tile = {"phase": "P0", "content": "Avoid X"}
    room.submit_tile(agent, p0_tile)
    assert room.current_phase == "P1"
    
    p1_tile = {"phase": "P1", "content": "Safe channel Y"}
    room.submit_tile(agent, p1_tile)
    assert room.current_phase == "P2"
    
    p2_tile = {"phase": "P2", "content": "Optimize Z"}
    room.submit_tile(agent, p2_tile)
    assert room.current_phase == "P2"  # Stays in P2 after completion
    assert room.tiles_accepted == 3
    
    # Reset
    room.reset()
    assert room.current_phase == "P0"
    assert room.tiles_accepted == 0
    assert len(room._tile_history) == 0
```
