# plato-torch/tests/test_deadband_room_edge.py
"""
Edge-case tests for DeadbandRoom preset.
These tests validate behavior under exceptional conditions.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
import numpy as np
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import TrainingTile
from plato_torch.core.exceptions import ProtocolViolationError, ResourceExhaustedError


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""
    
    @pytest.fixture
    def room(self):
        """Create a fresh DeadbandRoom instance for each test."""
        return DeadbandRoom(
            room_id="edge-test-room",
            max_agents=10,
            tile_capacity=1000,
            training_cycles=50
        )
    
    @pytest.mark.asyncio
    async def test_protocol_priority_violation(self, room):
        """Test that attempting P2 before P1 raises ProtocolViolationError.
        
        Edge case: Agent tries to optimize (P2) without first finding safe channels (P1).
        Expected: Clear error indicating protocol violation.
        """
        agent_id = "impatient-agent"
        
        # Skip P0 and P1, directly attempt P2
        with pytest.raises(ProtocolViolationError) as exc_info:
            await room.submit_tile(
                agent_id=agent_id,
                tile=TrainingTile(
                    phase="P2",
                    content={"optimization": "premature"},
                    metadata={"attempted_skip": True}
                )
            )
        
        assert "P1" in str(exc_info.value)  # Error should mention missing P1
        assert room.get_agent_state(agent_id)["violations"] == 1
    
    @pytest.mark.asyncio
    async def test_concurrent_tile_submission(self, room):
        """Test concurrent tile submission from multiple agents.
        
        Edge case: 5 agents submit tiles simultaneously.
        Expected: All tiles processed, no data corruption, order preserved.
        """
        num_agents = 5
        agents = [f"concurrent-agent-{i}" for i in range(num_agents)]
        
        # Create tiles with unique content
        tiles = [
            TrainingTile(
                phase="P0",
                content={"negative_space": f"avoid-{i}"},
                metadata={"sequence": i}
            )
            for i in range(num_agents)
        ]
        
        # Submit all tiles concurrently
        tasks = [
            room.submit_tile(agent_id=agent, tile=tile)
            for agent, tile in zip(agents, tiles)
        ]
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Verify all submissions succeeded
        assert all(r is True for r in results)
        
        # Verify all tiles are in room state
        room_tiles = room.get_tile_history()
        assert len(room_tiles) == num_agents
        
        # Verify no duplicate or missing sequences
        sequences = [t.metadata.get("sequence") for t in room_tiles]
        assert set(sequences) == set(range(num_agents))
    
    @pytest.mark.asyncio
    async def test_resource_exhaustion_during_training(self, room):
        """Test behavior when system resources are exhausted.
        
        Edge case: Simulate memory exhaustion during training cycle.
        Expected: Graceful degradation or clear error, not crash.
        """
        agent_id = "resource-test-agent"
        
        # Submit valid P0 tile
        await room.submit_tile(
            agent_id=agent_id,
            tile=TrainingTile(
                phase="P0",
                content={"negative_space": "memory_exhaustion"},
                metadata={"test": "resource"}
            )
        )
        
        # Mock resource exhaustion during training
        with patch('plato_torch.rooms.deadband_room.psutil.virtual_memory') as mock_mem:
            mock_mem.return_value.percent = 99.8  # Simulate 99.8% memory usage
            
            # Attempt to run training cycle
            with pytest.raises(ResourceExhaustedError) as exc_info:
                await room.run_training_cycle()
            
            assert "memory" in str(exc_info.value).lower()
            
            # Verify room state is preserved despite error
            assert room.room_id == "edge-test-room"
            assert len(room.get_tile_history()) == 1
    
    @pytest.mark.asyncio
    async def test_network_partition_simulation(self, room):
        """Test behavior during partial network connectivity.
        
        Edge case: Fleet sync fails for some agents but succeeds for others.
        Expected: Partial success, clear logging, retry mechanism.
        """
        agent_ids = ["connected-agent", "partitioned-agent", "flaky-agent"]
        
        # Mock fleet sync with partial failures
        with patch('plato_torch.rooms.deadband_room.FleetSyncClient') as mock_sync:
            mock_instance = AsyncMock()
            
            # Configure different responses for different agents
            def side_effect(agent_id, tile):
                if agent_id == "partitioned-agent":
                    raise ConnectionError("Network partition detected")
                elif agent_id == "flaky-agent":
                    # Succeeds on second attempt
                    if not hasattr(side_effect, 'flaky_attempts'):
                        side_effect.flaky_attempts = 0
                    side_effect.flaky_attempts += 1
                    if side_effect.flaky_attempts == 1:
                        raise TimeoutError("Connection timeout")
                    return True
                else:
                    return True
            
            mock_instance.sync_tile.side_effect = side_effect
            mock_sync.return_value = mock_instance
            
            # Submit tiles from all agents
            results = []
            for agent_id in agent_ids:
                tile = TrainingTile(
                    phase="P1",
                    content={"safe_channel": f"channel-{agent_id}"},
                    metadata={"network_test": True}
                )
                
                try:
                    result = await room.submit_tile(agent_id=agent_id, tile=tile)
                    results.append((agent_id, result, None))
                except Exception as e:
                    results.append((agent_id, False, e))
            
            # Verify mixed results
            success_count = sum(1 for _, success, _ in results if success)
            error_count = sum(1 for _, _, error in results if error is not None)
            
            assert success_count >= 1  # At least one succeeded
            assert error_count >= 1    # At least one failed
            assert room.get_sync_metrics()["retry_attempts"] > 0

                    },
                    metadata={"contradicts_previous": True}
                )
            )
        
        assert "contradict" in str(exc_info.value).lower()
        assert "P0" in str
