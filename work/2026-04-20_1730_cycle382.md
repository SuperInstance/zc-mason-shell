import unittest
from unittest.mock import Mock, patch, MagicMock
import json
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import Tile

class TestDeadbandRoomEdgeCases(unittest.TestCase):
    
    def setUp(self):
        """Set up fresh mocks for each test"""
        self.mock_db = Mock()
        self.mock_exporter = Mock()
        self.room = DeadbandRoom(
            room_id="test-edge-room",
            tile_db=self.mock_db,
            ensign_exporter=self.mock_exporter
        )
    
    def test_concurrent_tile_submission(self):
        """Test that concurrent tile submissions from multiple agents don't corrupt state"""
        # Simulate two agents submitting tiles simultaneously
        agent1_tile = Tile(
            agent_id="agent-1",
            phase="P1",
            content={"channel": "safe-path-1"},
            timestamp=1234567890
        )
        
        agent2_tile = Tile(
            agent_id="agent-2", 
            phase="P0",
            content={"negative_space": "rock-1"},
            timestamp=1234567891
        )
        
        # Mock DB to handle concurrent writes
        self.mock_db.store_tile.side_effect = lambda tile: tile
        
        # Submit both tiles
        result1 = self.room.submit_tile(agent1_tile)
        result2 = self.room.submit_tile(agent2_tile)
        
        # Verify both were processed
        self.assertEqual(self.mock_db.store_tile.call_count, 2)
        self.assertIsNotNone(result1)
        self.assertIsNotNone(result2)
        
        # Verify room state reflects latest valid protocol position
        self.assertIn("agent-1", self.room.agent_states)
        self.assertIn("agent-2", self.room.agent_states)
    
    def test_malformed_tile_json_structure(self):
        """Test handling of tiles with invalid JSON structures beyond basic validation"""
        # Create tile with nested malformed data
        malformed_tile = Tile(
            agent_id="malformed-agent",
            phase="P1",
            content={
                "channel": "safe-path",
                "nested": {
                    "circular": None  # Will be set to circular reference
                }
            },
            timestamp=1234567890
        )
        
        # Create actual circular reference (can't be JSON serialized)
        malformed_tile.content["nested"]["circular"] = malformed_tile.content
        
        # Mock DB to raise JSON serialization error
        self.mock_db.store_tile.side_effect = TypeError("Circular reference detected")
        
        # Should handle gracefully, not crash
        with self.assertRaises(TypeError):
            self.room.submit_tile(malformed_tile)
        
        # Room should maintain operational state
        self.assertEqual(self.room.status, "active")
    
    def test_ensign_export_during_active_processing(self):
        """Test attempting to export ensign while room is actively processing tiles"""
        # Put room in processing state
        self.room.current_batch_size = 5
        self.room.is_processing = True
        
        # Mock exporter to validate it won't be called during processing
        self.mock_exporter.export.side_effect = RuntimeError("Should not be called during processing")
        
        # Attempt export - should raise or return None based on implementation
        # Based on code, export() checks is_processing
        result = self.room.export_ensign()
        
        # Verify exporter wasn't called
        self.mock_exporter.export.assert_not_called()
        self.assertIsNone(result)
        
        # Change to non-processing state and try again
        self.room.is_processing = False
        self.room.current_batch_size = 0
        self.mock_exporter.export.return_value = "ensign-lora-adapter"
        
        result = self.room.export_ensign()
        self.assertEqual(result, "ensign-lora-adapter")
    
    def test_protocol_violation_recovery(self):
        """Test agent that violates protocol (P0→P2) then recovers with valid sequence"""
        agent_id = "recovering-agent"
        
        # Agent incorrectly submits P2 tile without P1
        p2_tile = Tile(
            agent_id=agent_id,
            phase="P2",
            content={"optimization": "fast-path"},
            timestamp=1234567890
        )
        
        # Should be rejected (based on protocol enforcement in submit_tile)
        self.mock_db.store_tile.return_value = None
        
        result = self.room.submit_tile(p2_tile)
        
        # Implementation may reject or accept with warning
        # Check that agent state reflects the violation
        if result is None:
            # Tile was rejected
            self.assertNotIn(agent_id, self.room.agent_states)
        else:
            # Tile was accepted but protocol violation logged
            self.assertIn(agent_id, self.room.agent_states)
            self.assertIn("protocol_violation", self.room.agent_states[agent_id])
        
        # Now submit correct P0 → P1 → P2 sequence
        p0_tile = Tile(
            agent_id=agent_id,
            phase="P0",
            content={"negative_space": "avoid-rock"},
            timestamp=1234567891
        )
        
        p1_tile = Tile(
            agent_id=agent_id,
            phase="P1",
            content={"channel": "safe-channel"},
            timestamp=1234567892
        )
        
        p2_tile_correct = Tile(
            agent_id=agent_id,
            phase="P2",
            content={"optimization": "efficient-path"},
            timestamp=1234567893
        )
        
        # Reset mock
        self.mock_db.store_tile.reset_mock()
        self.mock_db.store_tile.side_effect = lambda tile: tile
        
        # Submit correct sequence
        self.room.submit_tile(p0_tile)
        self.room.submit_tile(p1_tile)
        result = self.room.submit_tile(p2_tile_correct)
        
        # Should succeed
        self.assertIsNotNone(result)
        self.assertEqual(self.mock_db.store_tile.call_count, 3)
    
    def test_resource_exhaustion_handling(self):
        """Test room behavior when underlying storage fails (disk full, memory error)"""
        agent_id = "agent-resource-test"
        
        # Create valid tile
        valid_tile = Tile(
            agent_id=agent_id,
            phase="P0",
            content={"negative_space": "memory-exhaustion"},
            timestamp=1234567890
        )
        
        # Mock DB to simulate various resource failures
        failure_modes = [
            MemoryError("Out of memory"),
            OSError("Disk full"),
            ConnectionError("Database connection lost"),
            RuntimeError("Unexpected storage error")
        ]
        
        for failure in failure_modes:
            with self.subTest(failure_mode=str(failure)):
                self.mock_db.store_tile.side_effect = failure
                
                # Should raise the exception
                with self.assertRaises(type(failure)):
                    self.room.submit_tile(valid_tile)
                
                # Room should remain in recoverable state
                self.assertEqual(self.room.status, "active")
                # Agent state may not be updated due to failure
                
                # Reset mock for next iteration
                self.mock_db.store_tile.reset_mock()

if __name__ == "__main__":
    unittest.main()
