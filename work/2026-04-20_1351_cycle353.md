### Test 1: Concurrent Tile Submission Race Condition
```python
import asyncio
from unittest.mock import AsyncMock, patch
import pytest

@pytest.mark.asyncio
async def test_concurrent_tile_submission():
    """Test that room handles simultaneous tile submissions without corruption."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom(room_id="race_test")
    validator_mock = AsyncMock()
    validator_mock.validate_tile.return_value = {"valid": True, "score": 0.8}
    
    room.validator = validator_mock
    
    # Simulate 5 agents submitting tiles at the same time
    tiles = [
        {"agent_id": f"agent_{i}", "content": f"test_content_{i}", "phase": "P1"}
        for i in range(5)
    ]
    
    tasks = [room.submit_tile(tile) for tile in tiles]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # All submissions should complete without exceptions
    assert not any(isinstance(r, Exception) for r in results)
    
    # Validator should be called exactly 5 times
    assert validator_mock.validate_tile.call_count == 5
    
    # Room state should reflect all submissions
    assert len(room.state.submitted_tiles) == 5
    assert all(tile["agent_id"] in [f"agent_{i}" for i in range(5)] 
               for tile in room.state.submitted_tiles)
```

### Test 2: Malformed Tile Payload Handling
```python
import pytest
from pydantic import ValidationError

def test_malformed_tile_payload():
    """Test room's response to structurally invalid tile data."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom(room_id="malformed_test")
    
    # Missing required 'agent_id' field
    invalid_tile_1 = {"content": "test", "phase": "P1"}
    
    # Invalid phase value
    invalid_tile_2 = {"agent_id": "test_agent", "content": "test", "phase": "P99"}
    
    # Non-JSON serializable content
    invalid_tile_3 = {"agent_id": "test_agent", "content": object(), "phase": "P1"}
    
    with pytest.raises(ValidationError):
        room.submit_tile(invalid_tile_1)
    
    with pytest.raises(ValueError, match="Invalid protocol phase"):
        room.submit_tile(invalid_tile_2)
    
    with pytest.raises(TypeError, match="not JSON serializable"):
        room.submit_tile(invalid_tile_3)
    
    # Room state should remain unchanged
    assert len(room.state.submitted_tiles) == 0
```

### Test 5: Validator Service Unavailability
```python
import pytest
import aiohttp
from unittest.mock import AsyncMock, patch
from asyncio import TimeoutError

@pytest.mark.asyncio
async def test_validator_service_outage():
    """Test room behavior when tile validator service is unavailable."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom(room_id="outage_test")
    
    # Mock validator to simulate different failure modes
    validator_mock = AsyncMock()
    
    # Test 5a: Network timeout
    validator_mock.validate_tile.side_effect = TimeoutError("Connection timeout")
    
    tile = {"agent_id": "test_agent", "content": "test", "phase": "P1"}
    room.validator = validator_mock
    
    with pytest.raises(TimeoutError, match="Connection timeout"):
        await room.submit_tile(tile)
    
    # Test 5b: HTTP 503 Service Unavailable
    validator_mock.validate_tile.side_effect = aiohttp.ClientResponseError(
        request_info=None,
        history=None,
        status=503,
        message="Service Unavailable"
    )
    
    with pytest.raises(aiohttp.ClientResponseError):
        await room.submit_tile(tile)
    
    # Test 5c: Circuit breaker pattern - after N failures, use cached validation
    room.state.circuit_breaker_failures = 5  # Simulate previous failures
    
    # Room should bypass validator and use cached/default validation
    result = await room.submit_tile(tile, bypass_validation=True)
    
    assert result["validated"] is True
    assert result["source"] == "cache"
    assert room.state.circuit_breaker_tripped is True
```
