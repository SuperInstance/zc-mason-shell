### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(capacity=5)
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Create valid P0 tiles for each agent
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Negative space map for {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles concurrently (simulated with threads)
    results = []
    def submit_tile(tile):
        results.append(room.submit_tile(tile))
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for thread in threads:
        thread.start()
    for thread in threads:
        thread.join()
    
    # Verify all submissions were processed
    assert len(results) == 3
    assert all(result["status"] == "accepted" for result in results)
    assert room.get_agent_count() == 3
```

### Test 2: Network Partition - Partial Tile Data
```python
def test_partial_tile_recovery(self):
    """Test room's handling of tiles missing required fields due to network issues."""
    room = DeadbandRoom(capacity=3)
    
    # Tile missing 'phase' field (simulating network drop)
    partial_tile = {
        "agent_id": "network_agent",
        "content": "Valid content but missing phase",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(partial_tile)
    assert result["status"] == "rejected"
    assert "missing required field" in result["error"].lower()
    
    # Verify agent wasn't registered despite partial data
    assert room.get_agent_count() == 0
    
    # Resubmit complete tile should work
    complete_tile = {**partial_tile, "phase": "P0"}
    result = room.submit_tile(complete_tile)
    assert result["status"] == "accepted"
```

def test_contradictory_p0_tiles(self):

### Test 4: Phase Regression Prevention
```python
def test_phase_regression_blocked(self):
    """Test that agents cannot regress from higher to lower phases."""
    room = DeadbandRoom(capacity=1)
    agent_id = "progressive_agent"
    
    # Submit P0, P1, P2 in sequence (normal progression)
    for phase in ["P0", "P1", "P2"]:
        tile = {
            "agent_id": agent_id,
            "phase": phase,
            "content": f"{phase} content",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # Attempt to submit P1 tile after reaching P2
    regression_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Trying to go back to P1",
        "timestamp": time.time() + 10
    }
    
    result = room.submit_tile(regression_tile)
    assert result["status"] == "rejected"
    assert "phase regression" in result["error"].lower()
    
    # P2 tiles should still be accepted
    p2_continuation = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Continuing P2 optimization",
        "timestamp": time.time() + 11
    }
    result = room.submit_tile(p2_continuation)
    assert result["status"] == "accepted"
```
