### Test 2: Concurrent Agent Access Race Condition
```python
def test_concurrent_agent_access():
    """Test multiple agents accessing same room simultaneously."""
    room = DeadbandRoom()
    
    def agent_worker(agent_id):
        for _ in range(5):
            room.submit_tile(agent_id, {"phase": "P0", "content": f"test_{agent_id}"})
    
    # Simulate concurrent access
    import threading
    threads = []
    for i in range(3):
        t = threading.Thread(target=agent_worker, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Room should maintain consistent state
    assert room.tile_count >= 15  # 3 agents × 5 tiles each
    assert len(room.agent_sessions) == 3
```

### Test 5: Protocol State Persistence Across Restarts
```python
def test_state_persistence_simulation():
    """Test that protocol state can be serialized/deserialized correctly."""
    room = DeadbandRoom()
    
    # Advance protocol through several steps
    room.current_phase = 1  # P1
    room.current_step = 3
    room.agent_sessions = {
        "agent_1": {"tiles_submitted": 5, "last_phase": "P0"},
        "agent_2": {"tiles_submitted": 2, "last_phase": "P1"}
    }
    room.tile_count = 7
    
    # Simulate serialization
    state = {
        'current_phase': room.current_phase,
        'current_step': room.current_step,
        'agent_sessions': room.agent_sessions,
        'tile_count': room.tile_count
    }
    
    # Simulate deserialization (new room instance)
    new_room = DeadbandRoom()
    new_room.current_phase = state['current_phase']
    new_room.current_step = state['current_step']
    new_room.agent_sessions = state['agent_sessions']
    new_room.tile_count = state['tile_count']
    
    # Verify state integrity
    assert new_room.current_phase == 1
    assert new_room.current_step == 3
    assert len(new_room.agent_sessions) == 2
    assert new_room.tile_count == 7
    
    # Protocol should continue from saved state
    result = new_room.execute_protocol_step("agent_1")
    assert result is not None
```
