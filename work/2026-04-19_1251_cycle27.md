### Test 1: Empty Negative Space (P0) with Immediate P2 Attempt

#[test]
fn test_empty_negative_space_with_immediate_p2() {
    let mut room = DeadbandRoom::new();
    // Configure with empty P0 constraints
    room.set_negative_space(vec![]);
    
    // Agent attempts to optimize (P2) without finding safe channels (P1)

### Test 4: Maximum Tile Capacity with Mixed Priority Tiles
```rust
#[test]
fn test_max_tile_capacity_mixed_priority() {
    let mut room = DeadbandRoom::with_capacity(100);
    
    // Fill with mixed priority tiles (simulating real training load)
    for i in 0..100 {
        let priority = match i % 3 {
            0 => TilePriority::P0,
            1 => TilePriority::P1,
            2 => TilePriority::P2,
            _ => unreachable!(),
        };
        let tile = TrainingTile::new(
            format!("tile_{}", i),
            TileType::Interaction,
            priority,
        );
        assert!(room.add_tile(tile).is_ok());
    }
    
    // 101st tile should be rejected
    let overflow_tile = TrainingTile::new(
        "overflow".to_string(),
        TileType::Interaction,
        TilePriority::P0,
    );
    let result = room.add_tile(overflow_tile);
    assert!(result.is_err());
    assert!(result.unwrap_err().contains("capacity"));
    
    // Room should maintain protocol integrity despite full capacity
    assert!(room.can_advance_phase());
}
```

### Test 5: Protocol Reset with Persistent Ghost Tiles
```rust
#[test]
fn test_protocol_reset_persistent_ghosts() {
    let mut room = DeadbandRoom::new();
    
    // Run full protocol cycle
    room.start_protocol();
    room.advance_through_phases(); // P0 → P1 → P2
    
    // Add ghost tiles during execution
    let ghost_tile = TrainingTile::ghost(
        "critical_failure".to_string(),
        TileType::P0Constraint,
        "Critical failure mode discovered".to_string(),
    );
    room.inject_tile(ghost_tile);
    
    // Reset protocol
    room.reset_protocol();
    
    // Protocol state should reset
    assert_eq!(room.current_phase(), Phase::P0);
    assert!(room.get_agent_states().is_empty());
    
    // Ghost tiles should persist across reset
    let constraints = room.get_current_constraints();
    assert!(constraints.iter().any(|c| c.contains("failure")));
    
    // New protocol run should incorporate ghost knowledge
    room.start_protocol();
    let initial_constraints = room.get_current_constraints();
    assert!(!initial_constraints.is_empty());
}
```

## Test Rationale
These edge-case tests address:
1. **Protocol integrity** - Ensuring strict P0→P1→P2 progression
2. **Resource boundaries** - Testing capacity limits and overflow handling
3. **State persistence** - Verifying ghost tiles survive protocol resets
4. **Conflict resolution** - Handling overlapping/ambiguous safe channels
5. **Dynamic adaptation** - Incorporating new knowledge mid-execution

Each test probes a specific boundary condition that could cause failures in production but might not be caught by standard happy-path testing.
