```rust
fn test_p2_skip_recovery() {
    // P0: Agent attempts to skip directly to P2 optimization
    // Expected: Room rejects tile, forces agent back to P1 channel finding
    let mut room = DeadbandRoom::new();
    let agent_id = "test_agent_1".to_string();

    assert_eq!(agent_state.current_phase, Phase::P1);
}
```

```rust
fn test_concurrent_phase_collision() {

    // Expected: Room maintains protocol integrity for each agent independently
    let mut room = DeadbandRoom::new();
    let agent_a = "agent_a".to_string();
    let agent_b = "agent_b".to_string();

    let tile_a = Tile::new(agent_a.clone(), Phase::P0, "Mapping negative space");
    let tile_b = Tile::new(agent_b.clone(), Phase::P2, "Optimizing within channel");
    
    let result_a = room.submit_tile(tile_a);
    let result_b = room.submit_tile(tile_b);

    assert!(result_a.is_ok());
    assert!(result_b.is_ok());
    
    // Verify states remain independent
    assert_eq!(room.get_agent_state(&agent_a).current_phase, Phase::P1);
    assert_eq!(room.get_agent_state(&agent_b).current_phase, Phase::P2);
}
```

```rust
        new_agent.clone(),
        Phase::P1,
        "Found safe channel: use bounded tensors"
    );
    
    let result = room.submit_tile(p1_tile);
    assert!(result.is_ok());
    assert_eq!(room.get_agent_state(&new_agent).current_phase, Phase::P1);
```

### Test 5: Protocol Phase Transition Edge Cases
```rust
#[test]
fn test_phase_transition_edge_cases() {
    // Test unusual phase transitions
    let mut room = DeadbandRoom::new();
    let agent_id = "edge_agent".to_string();
    room.initialize_agent(&agent_id, Phase::P0);
    
    // Normal progression: P0 → P1
    let p0_tile = Tile::new(agent_id.clone(), Phase::P0, "Negative space mapped");
    assert!(room.submit_tile(p0_tile).is_ok());
    assert_eq!(room.get_agent_state(&agent_id).current_phase, Phase::P1);
    
    // Attempt regression: P1 → P0 (should fail)
    let regress_tile = Tile::new(agent_id.clone(), Phase::P0, "Trying to go back");
    assert!(room.submit_tile(regress_tile).is_err());
    
    // P1 → P2 with insufficient channel data (should fail)
    let weak_p1_tile = Tile::new(agent_id.clone(), Phase::P1, "Vague channel");
    assert!(room.submit_tile(weak_p1_tile).is_ok());
    
    let premature_p2 = Tile::new(agent_id.clone(), Phase::P2, "Optimizing weak channel");
    let result = room.submit_tile(premature_p2);
    assert!(result.is_err());
    assert_eq!(result.unwrap_err(), RoomError::InsufficientChannelData);
    
    // Proper P1 with detailed channel, then P2
    let detailed_p1 = Tile::new(
        agent_id.clone(),
        Phase::P1,
        "Safe channel: use bounded tensors < 1024, validate inputs, timeout after 5s"
    );
    assert!(room.submit_tile(detailed_p1).is_ok());
    
    let valid_p2 = Tile::new(
        agent_id.clone(),
        Phase::P2,
        "Optimized: batch size 32, LR 0.001, early stopping"
    );
    assert!(room.submit_tile(valid_p2).is_ok());
    assert_eq!(room.get_agent_state(&agent_id).current_phase, Phase::P2);
}
```
