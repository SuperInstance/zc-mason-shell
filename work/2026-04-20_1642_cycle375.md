## New Edge-Case Tests

### Test 1: Protocol Priority Violation (P2 before P1)
**Scenario:** Agent submits a tile with P2 optimization content but has no prior P1 safe channel established in session.
**Expected:** Room rejects tile with appropriate error.
**Rationale:** Tests strict P0→P1→P2 priority enforcement.

```python
def test_reject_p2_without_p1_context(self, deadband_room):
    """P2 tile submitted without preceding P1 context should fail."""
    # Start fresh session
    deadband_room.reset_session("test_agent_1")
    
    # Submit P0 tile (valid)
    p0_tile = Tile(
        agent_id="test_agent_1",
        protocol_phase=0,
        content="P0: Avoid null pointer dereference",
        metadata={"domain": "c++"}
    )
    assert deadband_room.submit_tile(p0_tile) is True
    
    # Attempt P2 tile without P1 (should fail)
    p2_tile = Tile(
        agent_id="test_agent_1",
        protocol_phase=2,
        content="P2: Use SIMD optimization here",
        metadata={"domain": "c++"}
    )
    result = deadband_room.submit_tile(p2_tile)
    assert result is False
    assert "P1 context required" in deadband_room.last_error
```

### Test 2: Cross-Domain Protocol Contamination
**Scenario:** Agent establishes P1 safe channel in one domain (e.g., "rust"), then submits P2 optimization in unrelated domain (e.g., "python").
**Expected:** Room rejects or flags the tile as protocol violation.
**Rationale:** Tests domain consistency in protocol progression.

```python
def test_reject_cross_domain_p2_optimization(self, deadband_room):
    """P2 optimization must be in same domain as P1 safe channel."""
    deadband_room.reset_session("test_agent_2")
    
    # Establish P1 in Rust domain
    p0_tile = Tile(
        agent_id="test_agent_2",
        protocol_phase=0,
        content="P0: Don't use unsafe blocks without review",
        metadata={"domain": "rust"}
    )
    deadband_room.submit_tile(p0_tile)
    
    p1_tile = Tile(
        agent_id="test_agent_2",
        protocol_phase=1,
        content="P1: Safe channel: Use Result<> for error handling",
        metadata={"domain": "rust"}
    )
    deadband_room.submit_tile(p1_tile)
    
    # Attempt P2 in Python domain (should fail)
    p2_tile = Tile(
        agent_id="test_agent_2",
        protocol_phase=2,
        content="P2: Use list comprehension instead of loop",
        metadata={"domain": "python"}
    )
    result = deadband_room.submit_tile(p2_tile)
    assert result is False
    assert "domain mismatch" in deadband_room.last_error.lower()
```

### Test 3: Ghost Tile Integration
**Scenario:** Agent submits a tile that references a ghost tile (dead agent's P0 lesson).
**Expected:** Room accepts tile and links it to ghost tile in training data.
**Rationale:** Tests ghost tile integration as specified in architecture.

```python
def test_ghost_tile_reference_validation(self, deadband_room):
    """Tiles can reference ghost tiles if they exist."""
    # Add a ghost tile to room state
    ghost_tile = GhostTile(
        original_agent_id="dead_agent_001",
        protocol_phase=0,
        content="P0: Never call free() twice",
        metadata={"domain": "c", "fatal_error": "double_free"}
    )
    deadband_room.room_state.add_ghost_tile(ghost_tile)
    
    # Submit tile referencing ghost tile
    tile = Tile(
        agent_id="test_agent_3",
        protocol_phase=1,
        content="P1: Safe channel: Use RAII wrapper for malloc/free",
        metadata={
            "domain": "c",
            "references_ghost": "dead_agent_001",
            "applies_lesson": "double_free"
        }
    )
    
    result = deadband_room.submit_tile(tile)
    assert result is True
    
    # Verify reference was recorded
    training_data = deadband_room.get_training_tiles()
    latest = training_data[-1]
    assert latest.metadata.get("references_ghost") == "dead_agent_001"
```

### Test 4: Malformed Metadata Structure
**Scenario:** Agent submits tile with nested metadata structures that exceed validation limits.
**Expected:** Room rejects tile with validation error.
**Rationale:** Tests robustness against complex malformed inputs.

```python
def test_reject_excessively_nested_metadata(self, deadband_room):
    """Metadata with deep nesting should be rejected."""
    # Create deeply nested metadata
    nested_meta = {"level1": {}}
    current = nested_meta["level1"]
    for i in range(2, 50):  # Exceeds reasonable depth
        current[f"level{i}"] = {}
        current = current[f"level{i}"]
    
    tile = Tile(
        agent_id="test_agent_4",
        protocol_phase=0,
        content="P0 test",
        metadata=nested_meta
    )
    
    result = deadband_room.submit_tile(tile)
    assert result is False
    assert "metadata too deep" in deadband_room.last_error.lower()
```

### Test 5: Concurrent Session Isolation
**Scenario:** Multiple agents submit tiles in parallel sessions.
**Expected:** Each agent's protocol progression is isolated; one agent's P1 doesn't satisfy another's P2 requirement.
**Rationale:** Tests session isolation in multi-agent scenarios.

```python
def test_concurrent_session_isolation(self, deadband_room):
    """Protocol progression is per-agent, not global."""
    # Agent A establishes P1
    deadband_room.reset_session("agent_a")
    deadband_room.submit_tile(Tile(
        agent_id="agent_a", protocol_phase=0,
        content="P0: Don't use eval()", metadata={"domain": "js"}
    ))
    deadband_room.submit_tile(Tile(
        agent_id="agent_a", protocol_phase=1,
        content="P1: Use Function() constructor instead",
        metadata={"domain": "js"}
    ))
    
    # Agent B attempts P2 without own P1 (should fail)
    deadband_room.reset_session("agent_b")
    deadband_room.submit_tile(Tile(
        agent_id="agent_b", protocol_phase=0,
        content="P0: Avoid memory leaks", metadata={"domain": "c++"}
    ))
    
    p2_tile = Tile(
        agent_id="agent_b",
        protocol_phase=2,
        content="P2: Use smart pointers",
        metadata={"domain": "c++"}
    )
    
    result = deadband_room.submit_tile(p2_tile)
    assert result is False
    assert "P1 context required" in deadband_room.last_error
    
    # Agent A should still be able to submit P2
    deadband_room.reset_session("agent_a")
    p2_tile_a = Tile(
        agent_id="agent_a",
        protocol_phase=2,
        content="P2: Use strict mode",
        metadata={"domain": "js"}
    )
    assert deadband_room.submit_tile(p2_tile_a) is True
```
