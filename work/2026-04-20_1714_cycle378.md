### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race_condition():
    """Test that room handles multiple simultaneous tile submissions without corruption."""
    room = DeadbandRoom(room_id="race_test", max_tiles=1000)
    
    # Simulate 10 agents submitting tiles at same time
    tiles = []
    for i in range(10):
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"Test tile {i}",
            "timestamp": datetime.utcnow().isoformat(),
            "phase": "P1"
        }
        tiles.append(tile)
    
    # Submit all tiles concurrently (simulated with threads)
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append(("success", tile["agent_id"]))
        except Exception as e:
            results.append(("error", str(e)))
    
    threads = []
    for tile in tiles:
        thread = threading.Thread(target=submit_tile, args=(tile,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all tiles were processed
    assert len(results) == 10
    assert all(r[0] == "success" for r in results)
    assert room.tile_count == 10
```

### Test 2: Network Partition Recovery
```python
def test_network_partition_recovery():
    """Test room recovery after tile server becomes temporarily unreachable."""
    room = DeadbandRoom(room_id="network_test")
    
    # First, normal operation
    tile = {
        "agent_id": "test_agent",
        "content": "Before partition",
        "timestamp": datetime.utcnow().isoformat(),
        "phase": "P0"
    }
    assert room.submit_tile(tile) is True
    
    # Simulate network partition (server unreachable)
    original_send_tile = room._send_tile_to_server
    partition_attempts = []
    
    def partitioned_send(tile_data):
        partition_attempts.append(tile_data)
        raise ConnectionError("Tile server unreachable")
    
    room._send_tile_to_server = partitioned_send
    
    # Try to submit during partition
    tile2 = {
        "agent_id": "test_agent",
        "content": "During partition",
        "timestamp": datetime.utcnow().isoformat(),
        "phase": "P1"
    }
    
    # Should queue tile locally but not fail
    assert room.submit_tile(tile2) is True
    assert len(partition_attempts) == 1
    
    # Restore connection
    room._send_tile_to_server = original_send_tile
    
    # Trigger sync of queued tiles
    room._sync_queued_tiles()
    
    # Verify both tiles are in final state
    assert room.tile_count == 2
```

### Test 3: Protocol Violation - Skipping P0
```python
def test_protocol_violation_skip_p0():
    """Test that room rejects attempts to skip P0 phase."""
    room = DeadbandRoom(room_id="protocol_test")
    
    # Agent tries to submit P1 tile without completing P0
    tile = {
        "agent_id": "impatient_agent",
        "content": "Trying to optimize before mapping negatives",
        "timestamp": datetime.utcnow().isoformat(),
        "phase": "P2"  # Skipping P0 and P1!
    }
    
    # This should fail validation
    with pytest.raises(ValueError) as exc_info:
        room.submit_tile(tile)
    
    assert "P0" in str(exc_info.value).lower()
    assert room.tile_count == 0
```

```python
def test_clock_skew_timestamp_validation():
    """Test handling of tiles with future/past timestamps beyond tolerance."""
    room = DeadbandRoom(room_id="clock_test", max_clock_skew_seconds=300)
    
    # Tile with timestamp 10 minutes in future (beyond 5min tolerance)
    future_time = (datetime.utcnow() + timedelta(minutes=10)).isoformat()
    tile_future = {
        "agent_id": "time_traveler",
        "content": "From the future",
        "timestamp": future_time,
        "phase": "P0"
    }
    
    with pytest.raises(ValueError) as exc_info:
        room.submit_tile(tile_future)
    
    assert "timestamp" in str(exc_info.value).lower()
    
    # Tile with timestamp 1 hour in past (should be fine for historical data)
    past_time = (datetime.utcnow() - timedelta(hours=1)).isoformat()
    tile_past = {
        "agent_id": "historian",
        "content": "Historical data",
        "timestamp": past_time,
        "phase": "P0"
    }
    
    # Past timestamps should be accepted for archival
    assert room.submit_tile(tile_past) is True
    assert room.tile_count == 1
```
