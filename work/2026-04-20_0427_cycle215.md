### Test 1: Protocol Reset During Active P2 Optimization
**Scenario**: When the agent is in `OPTIMIZING` state (P2), a protocol reset command is issued.
**Edge Case**: This tests interruption of optimization and proper state cleanup.
**Expected**: Room should reset to `IDLE` state, clear all channel data, and allow fresh P0 mapping.

```python
def test_protocol_reset_during_p2_optimization(self):
    """Test that protocol reset during P2 optimization cleans up state correctly."""
    room = DeadbandRoom()
    
    # Simulate full protocol progression to P2
    room.process_input({"phase": "P0", "action": "map", "data": {"hazards": ["rock1", "rock2"]}})
    room.process_input({"phase": "P1", "action": "find_channels", "data": {"safe_zones": ["zone1"]}})
    room.process_input({"phase": "P2", "action": "optimize", "data": {"channel": "zone1", "path": "A->B"}})
    
    # Verify we're in OPTIMIZING state
    self.assertEqual(room.state, "OPTIMIZING")
    
    # Issue protocol reset
    result = room.process_input({"action": "reset_protocol"})
    
    # Should reset to IDLE and clear all phase data
    self.assertEqual(room.state, "IDLE")
    self.assertEqual(room.current_phase, "P0")
    self.assertEqual(room.p0_hazards, [])
    self.assertEqual(room.p1_channels, [])
    self.assertIsNone(room.p2_current_channel)
    
    # Reset should be acknowledged
    self.assertIn("reset_complete", result)
```

### Test 2: Concurrent Hazard Addition During P1 Channel Finding
**Scenario**: New hazards are reported while the agent is in `FINDING_CHANNELS` state (P1).
**Edge Case**: Tests whether new P0 data invalidates current P1 analysis.
**Expected**: Room should either reject the new P0 data (maintaining protocol priority) or incorporate it and restart P1 analysis.

```python
def test_concurrent_hazard_addition_during_p1(self):
    """Test handling of new P0 hazards reported during P1 channel finding."""
    room = DeadbandRoom()
    
    # Start with initial P0 mapping
    room.process_input({"phase": "P0", "action": "map", "data": {"hazards": ["rock1"]}})
    self.assertEqual(room.state, "MAPPING")
    
    # Transition to P1
    room.process_input({"phase": "P1", "action": "find_channels", "data": {"safe_zones": ["zone1"]}})
    self.assertEqual(room.state, "FINDING_CHANNELS")
    
    # Attempt to add new hazard during P1
    with self.assertRaises(ProtocolViolationError) as context:
        room.process_input({"phase": "P0", "action": "map", "data": {"hazards": ["rock2"]}})
    
    # Should reject with appropriate error
    self.assertIn("P0 mapping not allowed during P1", str(context.exception))
    # P1 state should remain unchanged
    self.assertEqual(room.state, "FINDING_CHANNELS")
```

### Test 5: Channel Validation Against Previously Mapped Hazards
**Scenario**: P1 identifies a channel that overlaps with previously mapped P0 hazards.
**Edge Case**: Tests that "safe" channels are actually validated against known hazards.
**Expected**: Should reject channels that intersect with hazard zones.

```python
def test_p1_channel_validation_against_p0_hazards(self):
    """Test that P1 channels are validated against P0 hazard data."""
    room = DeadbandRoom()
    
    # Map hazards with spatial coordinates
    room.process_input({
        "phase": "P0", 
        "action": "map", 
        "data": {
            "hazards": [
                {"id": "rock1", "x": 0, "y": 0, "radius": 5},
                {"id": "rock2", "x": 20, "y": 20, "radius": 3}
            ]
        }
    })
    
    # Attempt to identify channel that overlaps with hazard
    with self.assertRaises(ValidationError) as context:
        room.process_input({
            "phase": "P1", 
            "action": "find_channels", 
            "data": {
                "safe_zones": [
                    {"id": "unsafe_zone", "x": 2, "y": 2, "radius": 4}  # Overlaps with rock1
                ]
            }
        })
    
    self.assertIn("overlaps with hazard", str(context.exception).lower())
    self.assertIn("rock1", str(context.exception))
    
    # Valid non-overlapping channel should be accepted
    result = room.process_input({
        "phase": "P1", 
        "action": "find_channels", 
        "data": {
            "safe_zones": [
                {"id": "safe_zone", "x": 30, "y": 30, "radius": 5}  # Clear of hazards
            ]
        }
    })
    
    self.assertIn("channels_identified", result)
    self.assertEqual(len(room.p1_channels), 1)
```
