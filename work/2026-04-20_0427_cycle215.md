```python
import asyncio
from unittest.mock import AsyncMock, patch
import pytest

@pytest.mark.asyncio
async def test_concurrent_tile_submission_race_condition(deadband_room):
    """Test that concurrent tile submissions from multiple agents don't corrupt state."""
    # Simulate 5 agents submitting tiles simultaneously
    tile_data = {
        "agent_id": "test_agent",
        "phase": "P1",
        "content": "Test tile content",
        "timestamp": "2026-04-20T04:30:00Z"
    }
    
    async def submit_tile(agent_id):
        tile = tile_data.copy()
        tile["agent_id"] = agent_id
        return await deadband_room.submit_tile(tile)
    
    # Submit tiles concurrently
    tasks = [submit_tile(f"agent_{i}") for i in range(5)]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # Verify all submissions succeeded
    assert all(isinstance(r, dict) for r in results), "Some submissions failed"
    assert len(deadband_room.tile_buffer) == 5, "Not all tiles were buffered"
    
    # Verify tile IDs are unique
    tile_ids = [r.get("tile_id") for r in results]
    assert len(set(tile_ids)) == 5, "Duplicate tile IDs detected"
```

```python
import numpy as np

def test_phase_transition_boundary_condition(deadband_room):
    """Test when P0 constraints exactly align with P1 channel boundaries."""
    # Create a constraint space that exactly matches a channel boundary
    constraints = {
        "bounds": {
            "x": {"min": 0.0, "max": 1.0},
            "y": {"min": 0.0, "max": 1.0}
        }
    }
    
    # Set up room with these constraints
    deadband_room.p0_constraints = constraints
    
    # Define channels that exactly match constraint boundaries
    channels = [
        {"id": "channel_1", "bounds": constraints["bounds"]},
        {"id": "channel_2", "bounds": {"x": {"min": 1.0, "max": 2.0}, "y": {"min": 0.0, "max": 1.0}}}
    ]
    
    # This should identify channel_1 as valid (exact match)
    valid_channels = deadband_room._find_valid_channels(channels)
    
    assert len(valid_channels) == 1, "Should find exactly one valid channel"
    assert valid_channels[0]["id"] == "channel_1", "Should match the exactly aligned channel"
    
    # Test edge case where constraint is at floating-point precision limit
    epsilon = np.finfo(float).eps
    tight_constraints = {
        "bounds": {
            "x": {"min": 0.0, "max": 1.0 + epsilon},
            "y": {"min": 0.0, "max": 1.0}
        }
    }
    
    deadband_room.p0_constraints = tight_constraints
    valid_channels = deadband_room._find_valid_channels(channels)
    
    # With epsilon expansion, channel_1 should still be valid
    assert len(valid_channels) >= 1, "Should tolerate floating-point precision issues"
```
