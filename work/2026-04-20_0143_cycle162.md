import pytest
import numpy as np
from plato_torch.rooms.deadband import DeadbandRoom
from plato_torch.core.agent import Agent

def test_deadband_concurrent_agent_interference():
    """P0: Two agents submitting actions simultaneously should not corrupt state."""
    room = DeadbandRoom(room_id="edge_concurrent")
    agent1 = Agent(agent_id="agent_1")
    agent2 = Agent(agent_id="agent_2")
    
    # Both agents attempt to submit actions in same step
    action1 = {"type": "move", "direction": "north"}
    action2 = {"type": "move", "direction": "south"}
    
    # Room should process sequentially without state corruption
    result1 = room.submit_action(agent1, action1)
    result2 = room.submit_action(agent2, action2)
    
    assert result1 is not None, "First agent action should be processed"
    assert result2 is not None, "Second agent action should be processed"
    assert room.step_count == 2, "Step count should increment for each action"
    # State integrity check
    assert room.current_state["last_agent"] in ["agent_1", "agent_2"], "State should track last agent correctly"

def test_deadband_malformed_state_recovery():
    """P0: Room should recover gracefully when internal state becomes malformed."""
    room = DeadbandRoom(room_id="edge_malformed")
    
    # Inject malformed state (simulating a bug or corruption)
    room.current_state = {"invalid": "data", "nested": {"broken": True}}
    
    # Attempt normal protocol operation
    agent = Agent(agent_id="recovery_agent")
    action = {"type": "sense", "target": "environment"}
    
    # Room should either reset state or raise specific error
    try:
        result = room.submit_action(agent, action)
        # If it succeeds, state should be repaired
        assert isinstance(room.current_state, dict), "State should be a valid dict"
        assert "step" in room.current_state, "State should have standard keys"
    except (ValueError, RuntimeError) as e:
        # Expected to raise a specific error type for malformed state
        assert "state" in str(e).lower() or "corrupt" in str(e).lower()

def test_deadband_extreme_numeric_inputs():
    """P0: Protocol should handle extreme numeric values without crashing."""
    room = DeadbandRoom(room_id="edge_numeric")
    agent = Agent(agent_id="numeric_agent")
    
    # Actions with extreme values
    extreme_actions = [
        {"type": "adjust", "value": float("inf")},
        {"type": "adjust", "value": float("-inf")},
        {"type": "adjust", "value": float("nan")},
        {"type": "adjust", "value": 1e100},
        {"type": "adjust", "value": -1e100},
    ]
    
    for action in extreme_actions:
        result = room.submit_action(agent, action)
        # Should either reject (P0) or handle with clamping
        assert result is not None, f"Action {action} should not crash room"
        if result.get("status") == "rejected":
            assert "P0" in result.get("reason", ""), "Rejection should cite P0 violation"

def test_deadband_protocol_skip_detection():
    """P0: Room should detect and reject attempts to skip from P0 directly to P2."""
    room = DeadbandRoom(room_id="edge_skip")
    agent = Agent(agent_id="skip_agent")
    
    # Mock agent that tries to bypass P1 (find safe channels)
    # Simulate by tampering with internal flags
    room._current_phase = "P0"
    # Attempt to force P2 optimization without P1
    room._force_phase_skip = True  # Hypothetical vulnerability
    
    action = {"type": "optimize", "bypass": True}
    result = room.submit_action(agent, action)
    
    # Room should detect skip and either correct or reject
    assert result is not None
    if result.get("status") == "rejected":
        assert "P1" in result.get("reason", ""), "Rejection should mention missing P1 phase"
    else:
        # If processed, verify phases were followed
        assert room._current_phase in ["P0", "P1", "P2"], "Phase should be valid"
        assert not hasattr(room, '_force_phase_skip') or not room._force_phase_skip

def test_deadband_empty_safe_channels():
    """P1: When no safe channels exist, room should handle gracefully (dead end)."""
    room = DeadbandRoom(room_id="edge_no_channels")
    agent = Agent(agent_id="stuck_agent")
    
    # Configure room with restrictive P0 that eliminates all channels
    room.p0_constraints = ["no_move", "no_sense", "no_adjust"]  # Hypothetical restrictive constraints
    
    action = {"type": "move", "direction": "east"}
    result = room.submit_action(agent, action)
    
    # Expected: P0 should reject, or P1 should return empty set
    assert result is not None
    if result.get("status") == "rejected":
        assert "P0" in result.get("reason", ""), "Should be P0 violation"
    elif result.get("status") == "processed":
        # If processed, safe channels list should be empty
        safe_channels = room._find_safe_channels(agent, action)
        assert len(safe_channels) == 0, "No safe channels should be found"
        # Room might have fallback behavior (e.g., wait, reset)
        assert "fallback" in result or "wait" in result.get("action", "")
