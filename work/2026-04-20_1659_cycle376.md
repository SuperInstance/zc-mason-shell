```python
def test_concurrent_agent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(room_id="concurrent_test")
    agent_ids = ["agent_alpha", "agent_beta", "agent_gamma"]
    
    # Simulate concurrent submissions using threads
    submission_results = []
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "cycle": 1,
            "phase": "P0",
            "content": f"Negative space map from {agent_id}",
            "validation_hash": "abc123"
        }
        result = room.submit_tile(tile)
        submission_results.append((agent_id, result))
    
    threads = [threading.Thread(target=submit_tile, args=(aid,)) for aid in agent_ids]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    self.assertEqual(len(submission_results), 3)
    for agent_id, result in submission_results:
        self.assertTrue(result["accepted"], f"Tile from {agent_id} should be accepted")
        self.assertIn(agent_id, room.tile_history)
```

### Test 2: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Test room handling when agent attempts P2 without completing P0/P1."""
    room = DeadbandRoom(room_id="protocol_violation_test")
    agent_id = "impatient_agent"
    
    # Agent attempts P2 directly (violation)
    violation_tile = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": "P2",
        "content": "Trying to optimize without mapping negative space",
        "validation_hash": "violation123"
    }
    
    result = room.submit_tile(violation_tile)
    self.assertFalse(result["accepted"])
    self.assertEqual(result["error_code"], "PROTOCOL_VIOLATION")
    self.assertIn("P0 required before P2", result["message"])
    
    # Verify agent can recover by submitting correct P0 tile
    recovery_tile = {
        "agent_id": agent_id,
        "cycle": 2,
        "phase": "P0",
        "content": "Mapping negative space after violation",
        "validation_hash": "recovery456"
    }
    
    recovery_result = room.submit_tile(recovery_tile)
    self.assertTrue(recovery_result["accepted"])
    self.assertEqual(room.get_agent_phase(agent_id), "P0")
```

### Test 3: Resource Exhaustion Handling
```python
def test_tile_queue_overflow(self):
    """Test room behavior when tile queue exceeds capacity."""
    room = DeadbandRoom(room_id="overflow_test", max_tiles=5)
    agent_id = "flood_agent"
    
    # Submit tiles until capacity
    for i in range(5):
        tile = {
            "agent_id": agent_id,
            "cycle": i,
            "phase": "P0",
            "content": f"Tile {i}",
            "validation_hash": f"hash{i}"
        }
        result = room.submit_tile(tile)
        self.assertTrue(result["accepted"])
    
    # Attempt overflow
    overflow_tile = {
        "agent_id": agent_id,
        "cycle": 6,
        "phase": "P1",
        "content": "This should trigger overflow",
        "validation_hash": "overflow"
    }
    
    result = room.submit_tile(overflow_tile)
    self.assertFalse(result["accepted"])
    self.assertEqual(result["error_code"], "QUEUE_FULL")
    
    # Verify oldest tile was removed (FIFO behavior)
    self.assertEqual(len(room.tile_history[agent_id]), 5)
    self.assertEqual(room.tile_history[agent_id][0]["cycle"], 1)  # Oldest remains
```

### Test 4: Malformed Sentiment Data
```python
def test_malformed_sentiment_injection(self):
    """Test room handling of invalid sentiment data from NPC system."""
    room = DeadbandRoom(room_id="sentiment_test")
    
    # Attempt to inject sentiment outside valid range [-1.0, 1.0]
    invalid_sentiments = [
        {"npc_id": "npc1", "sentiment": -2.5},  # Below minimum
        {"npc_id": "npc2", "sentiment": 1.5},   # Above maximum
        {"npc_id": "npc3", "sentiment": "happy"},  # Wrong type
        {"npc_id": "npc4", "sentiment": None},     # Null value
    ]
    
    for sentiment_data in invalid_sentiments:
        with self.assertRaises(ValueError) as context:
            room.inject_sentiment(sentiment_data)
        self.assertIn("sentiment", str(context.exception).lower())
    
    # Verify valid sentiment is accepted
    valid_result = room.inject_sentiment({"npc_id": "valid_npc", "sentiment": 0.7})
    self.assertTrue(valid_result["success"])
    self.assertIn("valid_npc", room.npc_sentiments)
```

```python
def test_agent_disconnection_recovery(self):
    """Test room handling of agent disconnection during protocol execution."""
    room = DeadbandRoom(room_id="network_test")
    agent_id = "unstable_agent"
    
    # Agent starts P0
    tile1 = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": "P0",
        "content": "Starting negative space mapping",
        "validation_hash": "start123"
    }
    room.submit_tile(tile1)
    
    # Simulate disconnection (agent state remains)
    room.mark_agent_disconnected(agent_id)
    self.assertFalse(room.is_agent_connected(agent_id))
    
    # Verify protocol state is preserved
    self.assertEqual(room.get_agent_phase(agent_id), "P0")
    
    # Agent reconnects and continues
    room.mark_agent_connected(agent_id)
    tile2 = {
        "agent_id": agent_id,
        "cycle": 2,
        "phase": "P1",  # Can proceed to P1
        "content": "Finding safe channels after reconnect",
        "validation_hash": "reconnect456"
    }
    
    result = room.submit_tile(tile2)
    self.assertTrue(result["accepted"])
    self.assertEqual(room.get_agent_phase(agent_id), "P1")
```
