### Test 1: Concurrent Agent Interactions
**Scenario:** Multiple agents submitting interactions simultaneously
**Edge Case:** Race conditions in tile generation and state management
```python
def test_concurrent_agent_interactions(self):
    """Test that DeadbandRoom handles concurrent agent interactions without data corruption."""
    room = DeadbandRoom(room_id="test_concurrent")
    
    # Simulate 5 agents submitting interactions at the same time
    agents = [f"agent_{i}" for i in range(5)]
    interactions = [
        {"agent_id": agent, "input": f"P0 test from {agent}", "timestamp": time.time()}
        for agent in agents
    ]
    
    # Process all interactions (simulating concurrent access)
    results = []
    for interaction in interactions:
        result = room.process_interaction(interaction)
        results.append(result)
    
    # Verify all interactions were processed
    assert len(results) == 5
    # Verify no duplicate tile IDs were generated
    tile_ids = [r.get("tile_id") for r in results if r and "tile_id" in r]
    assert len(tile_ids) == len(set(tile_ids)), "Duplicate tile IDs detected"
```

### Test 2: Malformed Protocol Transitions
**Scenario:** Agent attempts illegal P2→P0 transition
**Edge Case:** Protocol violation that should be caught but not crash
```python
def test_malformed_protocol_transition(self):
    """Test that DeadbandRoom handles illegal protocol transitions gracefully."""
    room = DeadbandRoom(room_id="test_malformed")
    
    # Agent tries to go from P2 directly to P0 (violates strict priority)
    interaction = {
        "agent_id": "test_agent",
        "input": "P2 optimization complete, now doing P0 mapping",
        "current_phase": "P2",
        "target_phase": "P0",
        "timestamp": time.time()
    }
    
    result = room.process_interaction(interaction)
    
    # Should return error, not crash
    assert "error" in result
    assert "protocol violation" in result["error"].lower()
    # Should not generate a tile for invalid transitions
    assert "tile_id" not in result
```

### Test 5: Protocol Phase Boundary Conditions
**Scenario:** Agent operates at exact boundary between protocol phases
**Edge Case:** Precision issues in phase transition logic
```python
def test_protocol_phase_boundary_conditions(self):
    """Test DeadbandRoom handling of boundary conditions in protocol phase transitions."""
    room = DeadbandRoom(room_id="test_boundaries")
    
    # Test cases at boundary conditions
    test_cases = [
        # (current_state, input, expected_next_phase)
        ({"P0_complete": True, "P1_channels": []}, "P1 exploration", "P1"),
        ({"P0_complete": True, "P1_channels": ["channel_a"], "P1_explored": True}, "P2 optimization", "P2"),
        ({"P0_complete": False, "P0_attempts": 99}, "P0 mapping attempt 100", "P0"),  # High count boundary
    ]
    
    for current_state, agent_input, expected_phase in test_cases:
        interaction = {
            "agent_id": "boundary_test_agent",
            "input": agent_input,
            "current_state": current_state,
            "timestamp": time.time()
        }
        
        result = room.process_interaction(interaction)
        
        # Verify phase transition logic handles boundaries correctly
        if "error" not in result:
            assert "next_phase" in result
            # Allow for P0->P1 transition even with empty P1_channels (exploration needed)
            if expected_phase == "P1" and current_state.get("P1_channels") == []:
                assert result["next_phase"] in ["P0", "P1"]  # Could stay in P0 or move to P1
            else:
                assert result["next_phase"] == expected_phase
```
