### Test 2: Concurrent Tile Submission Race
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom(room_id="race_condition_test")
    room.set_phase("P2")
    
    import threading
    results = []
    
    def submit_tile(agent_id):
        tile = {"agent": agent_id, "phase": "P2", "data": f"work_{agent_id}"}
        success = room.submit_tile(tile)
        results.append((agent_id, success))
    
    threads = [threading.Thread(target=submit_tile, args=(i,)) for i in range(10)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # All submissions should succeed without state corruption
    successful = sum(1 for _, success in results if success)
    self.assertEqual(successful, 10)
    self.assertEqual(len(room.tiles), 10)  # All tiles stored
```

### Test 3: Malformed Tile Payloads

```python
def test_malformed_tile_payloads(self):
    """Room should reject tiles with invalid structure."""
    room = DeadbandRoom(room_id="malformed_payload_test")
    room.set_phase("P2")
    
    test_cases = [
        None,  # Null payload
        "not a dict",  # String instead of dict
        {"missing_phase": "data"},  # Missing required 'phase' field
        {"phase": "P2", "data": "x" * 1000000},  # Extremely large payload
        {"phase": "INVALID_PHASE", "data": "test"},  # Invalid phase name
    ]
    
    for tile in test_cases:
        with self.subTest(tile=tile):
            success = room.submit_tile(tile)
            self.assertFalse(success, f"Should reject malformed tile: {type(tile)}")
```

### Test 4: Phase Regression Validation
```python
def test_phase_regression_constraint(self):
    """Moving backward in phases (P2→P0) may be restricted."""
    room = DeadbandRoom(room_id="regression_test")
    room.set_phase("P0")
    room.set_phase("P1")
    room.set_phase("P2")
    
    # Attempt regression to P1 (should work if allowed)
    try:
        room.set_phase("P1")
        regression_allowed = True
    except ValueError:
        regression_allowed = False
    
    # Document behavior for protocol clarification
    if regression_allowed:
        self.assertEqual(room.current_phase, "P1")
    else:
        self.assertEqual(room.current_phase, "P2")  # Stays at P2
```

```python
def test_large_batch_tile_submission(self):
    """Room should handle large tile batches without crashing."""
    room = DeadbandRoom(room_id="batch_stress_test")
    room.set_phase("P2")
    
    # Submit 1000 tiles (simulating high-throughput training)
    for i in range(1000):
        tile = {
            "agent": f"agent_{i % 10}",
            "phase": "P2",
            "data": {"iteration": i, "work": "x" * 100}  # 100-char payload
        }
        success = room.submit_tile(tile)
        self.assertTrue(success, f"Tile {i} failed")
    
    # Verify state integrity
    self.assertEqual(len(room.tiles), 1000)
    self.assertLess(room._get_memory_usage_mb(), 50)  # Should use <50MB
```
