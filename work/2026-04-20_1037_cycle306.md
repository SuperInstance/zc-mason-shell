3. **Protocol violation recovery** - Room state after P2→P1 violation (not just error raising)
4. **Resource exhaustion** - Memory/performance with thousands of tiles
5. **Network partition simulation** - Partial tile submissions, timeouts

### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(deadband_room):
    """Test that room handles simultaneous tile submissions correctly."""
    import threading
    
    results = []
    errors = []
    
    def submit_tile(agent_id):
        try:
            tile = {
                "agent_id": agent_id,
                "cycle": 1,
                "phase": "P0",
                "content": f"Test from {agent_id}",
                "timestamp": "2026-04-20T10:00:00Z"
            }
            deadband_room.submit_tile(tile)
            results.append(agent_id)
        except Exception as e:
            errors.append((agent_id, str(e)))
    
    # Simulate 5 agents submitting simultaneously
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    assert len(errors) == 0, f"Errors during concurrent submission: {errors}"
    assert len(deadband_room.tiles) == 5, "Not all tiles were stored"
    assert len(set(t["agent_id"] for t in deadband_room.tiles)) == 5, "Duplicate agent IDs detected"
```

```python
def test_protocol_violation_recovery(deadband_room):
    """Test room recovers properly after protocol violations."""
    
    # Force room into OPTIMIZING state (P2)
    deadband_room.state = "OPTIMIZING"
    deadband_room.current_phase = "P2"
    
    # Try to submit P0 tile (should fail)
    with pytest.raises(ValueError, match="Cannot submit P0 tile in P2 phase"):
        tile = {
            "agent_id": "test",
            "cycle": 1,
            "phase": "P0",
            "content": "Test content",
            "timestamp": "2026-04-20T10:00:00Z"
        }
        deadband_room.submit_tile(tile)
    
    # Verify room can still accept valid P2 tile
    tile_p2 = {
        "agent_id": "test",
        "cycle": 1,
        "phase": "P2",
        "content": "Valid P2 content",
        "timestamp": "2026-04-20T10:00:00Z"
    }
    deadband_room.submit_tile(tile_p2)
    
    assert len(deadband_room.tiles) == 1
    assert deadband_room.tiles[0]["phase"] == "P2"
    
    # Reset should work even after violation
    deadband_room.reset()
    assert deadband_room.state == "IDLE"
    assert deadband_room.current_phase == "P0"
    assert len(deadband_room.tiles) == 0
```

### Test 4: Resource Exhaustion Handling
```python
def test_large_tile_volume(deadband_room):
    """Test room performance with large number of tiles."""
    
    import time
    
    start_time = time.time()
    
    # Submit 1000 tiles (simulating heavy training session)
    for i in range(1000):
        tile = {
            "agent_id": f"agent_{i % 10}",
            "cycle": i,
            "phase": "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
            "content": f"Tile content {i}" * (i % 10 + 1),
            "timestamp": f"2026-04-20T10:{i:02d}:00Z"
        }
        deadband_room.submit_tile(tile)
    
    end_time = time.time()
    
    # Performance check: should process 1000 tiles in reasonable time
    processing_time = end_time - start_time
    assert processing_time < 5.0, f"Tile processing too slow: {processing_time:.2f}s"
    
    # Memory check: room should still be functional
    assert len(deadband_room.tiles) == 1000
    assert deadband_room.state == "OPTIMIZING"  # Last tile was P2
    
    # Verify we can still query tiles
    p0_tiles = [t for t in deadband_room.tiles if t["phase"] == "P0"]
    p1_tiles = [t for t in deadband_room.tiles if t["phase"] == "P1"]
    p2_tiles = [t for t in deadband_room.tiles if t["phase"] == "P2"]
    
    assert len(p0_tiles) > 0
    assert len(p1_tiles) > 0
    assert len(p2_tiles) > 0
```
