### Test 1: Concurrent Tile Processing

```python
def test_concurrent_tile_processing(self):
    """Test DeadbandRoom handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(config=self.default_config)
    
    # Simulate 10 agents submitting tiles at same cycle
    tiles = []
    for i in range(10):
        tile = {
            'agent_id': f'agent_{i}',
            'cycle': 1,
            'content': f'Test content from agent {i}',
            'protocol_phase': 'P0' if i % 3 == 0 else 'P1' if i % 3 == 1 else 'P2'
        }
        tiles.append(tile)
    
    # Process all tiles concurrently (simulated)
    results = []
    for tile in tiles:
        result = room.process_tile(tile)
        results.append(result)
    
    # Verify all tiles processed without state corruption
    assert len(results) == 10
    assert all(r['status'] == 'accepted' for r in results)
    assert room.state['processed_count'] == 10
```

### Test 3: Resource Exhaustion Protection
```python
def test_resource_exhaustion_protection(self):
    """Test DeadbandRoom protects against memory/CPU exhaustion during high-volume training."""
    room = DeadbandRoom(config=self.default_config)
    
    # Submit large volume of tiles
    large_tile_batch = []
    for i in range(1000):
        large_tile_batch.append({
            'agent_id': f'batch_agent_{i // 100}',
            'cycle': i,
            'content': 'x' * 10000,  # Large content
            'protocol_phase': 'P0'
        })
    
    # Process with memory monitoring
    import psutil
    process = psutil.Process()
    initial_memory = process.memory_info().rss
    
    processed_count = 0
    for tile in large_tile_batch[:100]:  # Process subset to test scaling
        result = room.process_tile(tile)
        if result['status'] == 'accepted':
            processed_count += 1
    
    final_memory = process.memory_info().rss
    memory_increase = final_memory - initial_memory
    
    # Memory increase should be reasonable (< 50MB for 100 large tiles)
    assert memory_increase < 50 * 1024 * 1024
    assert processed_count > 0
```

### Test 4: Network Partition Simulation
```python
def test_network_partition_recovery(self):
    """Test room recovers from partial state updates during network partitions."""
    room = DeadbandRoom(config=self.default_config)
    
    # Simulate normal operation
    for i in range(5):
        room.process_tile({
            'agent_id': 'stable_agent',
            'cycle': i,
            'content': f'Normal tile {i}',
            'protocol_phase': 'P0'
        })
    
    # Simulate network partition - save partial state
    partial_state = room.state.copy()
    partial_state['last_cycle'] = 3  # Only through cycle 3 persisted
    
    # Create new room with partial state (simulating recovery)
    recovered_room = DeadbandRoom(config=self.default_config)
    recovered_room.state = partial_state
    
    # Verify recovery logic
    assert recovered_room.state['last_cycle'] == 3
    assert recovered_room.state['processed_count'] == 4  # cycles 0-3
    
    # Should accept new tiles after recovery
    result = recovered_room.process_tile({
        'agent_id': 'stable_agent',
        'cycle': 5,  # Skipping cycle 4 due to partition
        'content': 'Post-recovery tile',
        'protocol_phase': 'P1'
    })
    
    assert result['status'] == 'accepted'
```

```python
def test_malformed_state_recovery(self):
    """Test room recovers from corrupted state file on initialization."""
    # Create corrupted state file
    import json
    corrupted_state = {
        'processed_count': 'not_a_number',  # Wrong type
        'last_cycle': 42,
        'agents': {'test_agent': {'phase': 'P2'}},  # Missing required fields
        'unexpected_field': [1, 2, 3]  # Extra field
    }
    
    # Save corrupted state
    state_path = '/tmp/corrupted_deadband_state.json'
    with open(state_path, 'w') as f:
        json.dump(corrupted_state, f)
    
    # Configure room to load from corrupted state
    config = self.default_config.copy()
    config['state_file'] = state_path
    config['recovery_mode'] = True
    
    # Should initialize with defaults, not crash
    room = DeadbandRoom(config=config)
    
    # Verify default state loaded
    assert room.state['processed_count'] == 0
    assert room.state['last_cycle'] == 0
    assert len(room.state['agents']) == 0
    
    # Should log corruption event
    assert hasattr(room, 'recovery_log')
    assert 'corrupted' in room.recovery_log[0] or 'recovered' in room.recovery_log[0]
    
    # Cleanup
    import os
    os.remove(state_path)
```
