### Test 3: Concurrent Tile Submission
```python
def test_concurrent_tile_submission():
    """P1: Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom(room_id="edge_test_3", buffer_size=50)
    
    def agent_worker(agent_id, tile_count):
        for i in range(tile_count):
            tile = {
                "agent_id": agent_id,
                "content": f"Tile from {agent_id}-{i}",
                "metadata": {"thread_safe": True}
            }
            room.submit_tile(agent_id, tile)
    
    # Simulate concurrent submissions
    import threading
    threads = []
    for j in range(3):
        t = threading.Thread(target=agent_worker, args=(f"agent_{j}", 10))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify no data corruption
    buffer = room.get_tile_buffer()
    agent_counts = {}
    for tile in buffer:
        agent_id = tile["agent_id"]
        agent_counts[agent_id] = agent_counts.get(agent_id, 0) + 1
    
    # All agents should have some tiles accepted
    assert len(agent_counts) == 3
    assert all(count > 0 for count in agent_counts.values())
    
    # Total should not exceed buffer capacity
    assert sum(agent_counts.values()) <= 50
```

def test_extreme_parameter_values():
    """P0: Room initialization with boundary parameter values."""
    
    # Test minimum buffer size (edge case: zero or negative)
    with pytest.raises(ValueError):
        DeadbandRoom(room_id="test_min", buffer_size=0)
    
    with pytest.raises(ValueError):
        DeadbandRoom(room_id="test_neg", buffer_size=-1)
    
    # Test very large buffer size (memory consideration)
    large_room = DeadbandRoom(room_id="test_large", buffer_size=10000)
    assert large_room.buffer_size == 10000
    
    # Test extreme learning rates
    tiny_lr = DeadbandRoom(room_id="test_tiny_lr", buffer_size=10, learning_rate=0.000001)
    assert tiny_lr.learning_rate == 0.000001
    
    huge_lr = DeadbandRoom(room_id="test_huge_lr", buffer_size=10, learning_rate=10.0)
    assert huge_lr.learning_rate == 10.0
    
    # Verify protocol still works with extreme values
    result = tiny_lr.execute_protocol_step("agent", "P0", {"map": "test"})
    assert result["status"] in ["success", "error"]  # Should complete without crash
