"""
Edge-case tests for DeadbandRoom.
These complement the existing test suite in test_deadband_room.py.

from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.models import Tile, AgentState

class TestDeadbandRoomEdgeCases:
    """Edge-case tests focusing on protocol adherence and state transitions."""
    
    @pytest.fixture
    def room(self):
        """Fixture providing a fresh DeadbandRoom instance."""
        return DeadbandRoom(room_id="test_edge_room")
    
    @pytest.fixture
    def agent_state(self):
        """Fixture providing a base agent state."""
        return AgentState(
            agent_id="test_agent_edge",
            current_phase=1,
            tiles_submitted=0,
            protocol_step="P0"

    def test_protocol_priority_enforcement_p2_before_p1(self, room, agent_state):
        """P0: Agent attempts to submit P2 tile before completing P1.
        
        Expected: Tile rejected, agent remains at P0/P1.
        """
        # Setup agent at P0
        agent_state.protocol_step = "P0"
        room.agent_states = {agent_state.agent_id: agent_state}
        
        # Attempt to submit P2 tile directly
        tile = Tile(
            agent_id=agent_state.agent_id,
            content="P2 optimization attempt",
            protocol_step="P2",
            phase=1
        )
        
        result = room.submit_tile(tile)
        
        # Should be rejected
        assert not result["success"]
        assert "P1" in result["message"] or "priority" in result["message"].lower()
        # Agent should not progress to P2
        assert room.agent_states[agent_state.agent_id].protocol_step != "P2"
    
    def test_phase_boundary_transition(self, room, agent_state):
        """Agent at phase boundary (e.g., end of phase 3) submits final tile.
        
        Expected: Agent transitions to next phase, protocol resets to P0.
        """
        # Setup agent at end of phase 3
        agent_state.current_phase = 3
        agent_state.tiles_submitted = 9  # Assuming 10 tiles per phase
        agent_state.protocol_step = "P2"
        room.agent_states = {agent_state.agent_id: agent_state}
        
        # Submit final tile for phase 3
        tile = Tile(
            agent_id=agent_state.agent_id,
            content="Phase 3 completion",
            protocol_step="P2",
            phase=3
        )
        
        with patch.object(room, 'get_agent_state', return_value=agent_state):
            result = room.submit_tile(tile)
        
        # Should succeed
        assert result["success"]
        # Agent should transition to phase 4, protocol reset to P0
        updated_state = room.agent_states[agent_state.agent_id]
        assert updated_state.current_phase == 4
        assert updated_state.protocol_step == "P0"
        assert updated_state.tiles_submitted == 0
    
    def test_tile_content_boundary_values(self, room, agent_state):
        """Tile content at boundaries: empty string, very long, special chars.
        
        Expected: Handled per validation rules (may accept or reject).
        """
        test_cases = [
            ("", "empty string"),
            ("A" * 10000, "very long content"),
            ("\x00\n\t\r\\\"'", "control chars and escapes"),
            ("🎉🔬📊", "Unicode emoji"),
        ]
        
        for content, description in test_cases:
            room.agent_states = {agent_state.agent_id: agent_state}
            tile = Tile(
                agent_id=agent_state.agent_id,
                content=content,
                protocol_step="P0",
                phase=1
            )
            
            result = room.submit_tile(tile)
            # Document behavior without assuming pass/fail
            assert "success" in result
            print(f"Content {description}: success={result['success']}")
    
    def test_sequential_agents_interference(self, room):
        """Multiple agents submitting tiles sequentially.
        
        Expected: Each agent's state is isolated and persists correctly.
        """
        agent1_state = AgentState(agent_id="agent1", current_phase=1, tiles_submitted=0, protocol_step="P0")
        agent2_state = AgentState(agent_id="agent2", current_phase=2, tiles_submitted=5, protocol_step="P1")
        
        room.agent_states = {
            agent1_state.agent_id: agent1_state,
            agent2_state.agent_id: agent2_state

        result1 = room.submit_tile(tile1)

        tile2 = Tile(agent_id="agent2", content="Agent2 P1", protocol_step="P1", phase=2)
        result2 = room.submit_tile(tile2)

        assert room.agent_states["agent2"].tiles_submitted == 6

        assert room.agent_states["agent2"].protocol_step == "P2"  # Progressed
    
    def test_error_recovery_after_invalid_tile(self, room, agent_state):

            agent_id=agent_state.agent_id,
            content="Valid recovery",
            protocol_step="P0",
            phase=1
        )
        result = room.submit_tile(valid_tile)
        
        # Should succeed
        assert result["success"]
        assert room.agent_states[agent_state.agent_id].tiles_submitted == 1
        assert room.agent_states[agent_state.agent_id].protocol_step == "P1"
