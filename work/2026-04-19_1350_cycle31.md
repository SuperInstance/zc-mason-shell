### Test 3: Concurrent Agent Interactions
```rust
#[test]
fn test_deadband_room_concurrent_tile_processing() {
    use std::sync::{Arc, Mutex};
    use std::thread;
    
    let room = Arc::new(Mutex::new(DeadbandRoom::new(0.1, 0.3, 1000)));
    let mut handles = vec![];
    
    // Simulate 10 agents submitting tiles simultaneously
    for i in 0..10 {
        let room_clone = Arc::clone(&room);
        handles.push(thread::spawn(move || {
            let tile = Tile {
                id: format!("agent_{}_tile", i),
                protocol_step: ProtocolStep::P0,
                content: format!("Negative space mapping from agent {}", i),
                metadata: HashMap::new(),
            };
            
            let mut room = room_clone.lock().unwrap();
            room.process_tile(tile).unwrap()
        }));
    }
    
    // Verify all tiles processed without state corruption
    for handle in handles {
        let result = handle.join().unwrap();
        assert!(result.success);
    }
    
    let final_room = room.lock().unwrap();
    assert_eq!(final_room.get_tile_count(), 10);
}
```

### Test 4: State Persistence Under Load
```rust
#[test]
fn test_deadband_room_state_persistence_under_high_load() {
    let mut room = DeadbandRoom::new(0.1, 0.3, 5); // Small capacity: 5 tiles
    
    // Fill room to capacity with P0 tiles
    for i in 0..5 {
        let tile = Tile {
            id: format!("load_tile_{}", i),
            protocol_step: ProtocolStep::P0,
            content: "Mapping negative space".to_string(),
            metadata: HashMap::new(),
        };
        room.process_tile(tile).unwrap();
    }
    
    // Verify state is preserved at capacity
    let state_at_capacity = room.get_state();
    assert_eq!(state_at_capacity.tile_count, 5);
    assert_eq!(state_at_capacity.current_protocol_step, ProtocolStep::P0);
    
    // Attempt to add one more tile (should trigger cleanup/eviction)
    let overflow_tile = Tile {
        id: "overflow_tile".to_string(),
        protocol_step: ProtocolStep::P1,
        content: "Finding safe channel".to_string(),
        metadata: HashMap::new(),
    };
    
    let result = room.process_tile(overflow_tile);
    assert!(result.is_ok()); // Room should handle overflow gracefully
    
    // Verify state transition occurred
    let final_state = room.get_state();
    assert!(final_state.tile_count <= 5); // Should not exceed capacity
    assert_eq!(final_state.current_protocol_step, ProtocolStep::P1);
}
```

        id: "large_tile".to_string(),
        protocol_step: ProtocolStep::P0,

        metadata: HashMap::new(),

    // Room should either process or reject based on configured limits

        Ok(_) => println!("Large tile accepted"),
        Err(e) => assert!(e.contains("size limit") || e.contains("too large")),

        content: "Valid content".to_string(),

    };

}
