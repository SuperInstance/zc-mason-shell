### Test 1: Concurrent Tile Submission Race Condition
**Scenario**: Multiple agents attempt to submit tiles simultaneously during phase transition
**Edge Case**: Race condition when P0 completes and P1 begins
**Test Purpose**: Ensure thread safety during phase transitions
```python
def test_concurrent_tile_submission_during_phase_transition():
    """Test that concurrent tile submissions during phase transition don't corrupt state."""
    room = DeadbandRoom()
    room.start_session()
    
    # Complete P0
    room.submit_tile(Priority.P0, {"type": "negative_space", "data": "avoid_x"})
    
    # Simulate multiple agents trying to submit as phase transitions
    def submit_p1_tile():
        return room.submit_tile(Priority.P1, {"type": "safe_channel", "data": "channel_y"})
    
    # Use threading to simulate concurrency
    import threading
    threads = [threading.Thread(target=submit_p1_tile) for _ in range(5)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # State should be consistent
    assert room.current_phase == Priority.P1
    assert len(room.get_tiles(Priority.P1)) >= 1  # At least one successful submission
```

### Test 3: Maximum Tile Capacity Per Phase
**Scenario**: Agent submits excessive tiles in a single phase
**Edge Case**: System resource exhaustion or performance degradation
**Test Purpose**: Verify graceful handling of high-volume tile submissions
```python
def test_max_tile_capacity_per_phase():
    """Test behavior when approaching system limits for tile storage."""
    room = DeadbandRoom()
    room.start_session()
    
    # Submit large number of P0 tiles
    max_tiles = 1000  # Configurable limit
    successful = 0
    
    for i in range(max_tiles + 100):  # Try to exceed limit
        try:
            room.submit_tile(Priority.P0, {
                "type": "negative_space",
                "data": f"avoid_pattern_{i}",
                "timestamp": i
            })
            successful += 1
        except (MemoryError, RuntimeError) as e:
            # System should raise appropriate error, not crash
            assert "capacity" in str(e).lower() or "limit" in str(e).lower()
            break
    
    # System should handle gracefully - either accept all or raise clear error
    tiles = room.get_tiles(Priority.P0)
    assert len(tiles) <= max_tiles  # Should not exceed reasonable limit
    if successful < max_tiles + 100:
        assert room.current_phase == Priority.P0  # Should not advance if error
```

```python
def test_agent_death_mid_protocol():
    """Test handling when agent stops participating after submitting tiles."""
    room = DeadbandRoom()
    room.start_session()
    
    # Agent 1 submits P0 tile then "dies" (stops responding)
    room.submit_tile(Priority.P0, {
        "type": "negative_space",
        "data": "critical_avoidance",
        "agent_id": "agent_1",
        "is_ghost_candidate": True
    })
    
    # Simulate time passing without agent 1
    import time
    time.sleep(0.1)  # Simulate delay
    
    # Agent 2 joins and should see P0 as complete (ghost tile present)
    # System should allow phase advancement based on ghost tiles
    p0_tiles = room.get_tiles(Priority.P0)
    assert len(p0_tiles) == 1
    assert p0_tiles[0].get("is_ghost_candidate") is True
    
    # Agent 2 should be able to start P1 work
    # This tests that ghost tiles don't block protocol progression
    room.submit_tile(Priority.P1, {
        "type": "safe_channel",
        "data": "recovery_channel",
        "agent_id": "agent_2"
    })
    
    assert room.current_phase == Priority.P1
    # Ghost tile from agent 1 should be preserved in history
    history = room.get_session_history()
    assert any(tile.get("agent_id") == "agent_1" for tile in history)
```
