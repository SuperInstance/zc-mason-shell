def test_protocol_violation_skip_p0_to_p2():
    """P0: Agent attempts to submit P2 tile without completing P1."""
    room = DeadbandRoom()
    agent_id = "test_agent_skip"
    
    # Agent submits P0 tile (valid)
    p0_tile = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": 0,
        "content": "P0: Don't skip phases",
        "type": "negative_space"
    }
    result = room.submit_tile(p0_tile)
    assert result["accepted"] == True
    
    # Agent immediately tries to submit P2 tile (should be rejected)
    p2_tile = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": 2,
        "content": "P2: Optimizing prematurely",
        "type": "optimization"
    }
    result = room.submit_tile(p2_tile)
    assert result["accepted"] == False
    assert "P1 not completed" in result["reason"]
    
    # Verify agent remains in P0 state
    agent_state = room.get_agent_state(agent_id)
    assert agent_state["current_phase"] == 0
    assert agent_state["completed_phases"] == [0]

    room = DeadbandRoom()
    agent_ids = ["agent_a", "agent_b", "agent_c"]
    
    # Submit tiles from multiple agents
    for i, agent_id in enumerate(agent_ids):

### Test 3: Concurrent Tile Submission Race Condition

def test_concurrent_tile_submission():
    """P0: Multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    num_agents = 10
    tiles = []
    
    # Prepare tiles from multiple agents for same cycle/phase
    for i in range(num_agents):
        tiles.append({
            "agent_id": f"concurrent_agent_{i}",
            "cycle": 1,
            "phase": 0,
            "content": f"Concurrent test {i}",
            "type": "negative_space"
        })
    
    # Simulate concurrent submission (threaded in real scenario)
    results = []
    for tile in tiles:
        # In actual implementation, this would be threaded
        result = room.submit_tile(tile)
        results.append(result)
    
    # All should be accepted (no phase violation)
    accepted_count = sum(1 for r in results if r["accepted"])
    assert accepted_count == num_agents
    
    # Verify all agents in correct state
    for i in range(num_agents):
        state = room.get_agent_state(f"concurrent_agent_{i}")
        assert state["current_phase"] == 0
        assert state["cycle"] == 1
```

def test_tile_storage_limits():
    """P0: Exceeding configured tile storage capacity."""
    room = DeadbandRoom()
    
    # Configure low storage limit for testing
    room.max_tiles = 5
    
    # Submit tiles up to limit
    for i in range(5):
        tile = {
            "agent_id": "storage_test_agent",
            "cycle": 1,
            "phase": 0,
            "content": f"Tile {i}",
            "type": "negative_space"
        }
        result = room.submit_tile(tile)
        assert result["accepted"] == True
    
    # Next tile should be rejected due to storage limit
    overflow_tile = {
        "agent_id": "storage_test_agent",
        "cycle": 1,
        "phase": 1,  # Valid phase progression
        "content": "Overflow tile",
        "type": "safe_channel"
    }
    result = room.submit_tile(overflow_tile)
    assert result["accepted"] == False
    assert "storage limit" in result["reason"].lower()
    
    # Verify agent can still progress after cleanup
    room.cleanup_old_tiles(keep_last=3)
    result = room.submit_tile(overflow_tile)
    assert result["accepted"] == True  # Should work after cleanup
```
