### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    room.initialize()
    
    # Simulate 3 agents submitting tiles at same time
    tiles = [
        {"agent_id": "agent1", "phase": "P0", "content": "Avoid rocks"},
        {"agent_id": "agent2", "phase": "P0", "content": "Map negative space"},
        {"agent_id": "agent3", "phase": "P0", "content": "Identify obstacles"}
    ]
    
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions processed
    assert len(results) == 3
    assert all("accepted" in str(r).lower() or "queued" in str(r).lower() for r in results)
    # Verify room state consistent
    assert room.get_tile_count() == 3 or room.get_queue_size() == 3
```

### Test 3: Malformed Content Injection
```python
def test_malformed_content_injection(self):
    """Test handling of potentially malicious tile content."""
    room = DeadbandRoom()
    room.initialize()
    
    injection_attempts = [
        {"agent_id": "bad_actor", "phase": "P0", "content": "<script>alert('xss')</script>"},
        {"agent_id": "bad_actor2", "phase": "P1", "content": "../../etc/passwd"},
        {"agent_id": "bad_actor3", "phase": "P2", "content": '"; DROP TABLE tiles; --'}
    ]
    
    for attempt in injection_attempts:
        result = room.submit_tile(attempt)
        # Should either sanitize or reject
        assert result.get("status") in ["rejected", "sanitized", "accepted"]
        if result.get("status") == "accepted":
            # Verify content was sanitized
            submitted_content = room.get_recent_tiles(1)[0]["content"]
            assert "<script>" not in submitted_content
            assert "../" not in submitted_content
            assert "DROP TABLE" not in submitted_content
```

### Test 4: Protocol Phase Rollback
```python
def test_phase_rollback_scenario(self):
    """Test P2 → P1 rollback when new P0 constraints emerge."""
    room = DeadbandRoom()
    room.initialize()
    
    # Advance to P2 normally
    room.submit_tile({"agent_id": "agent1", "phase": "P0", "content": "Constraint A"})
    room.submit_tile({"agent_id": "agent2", "phase": "P1", "content": "Channel B"})
    assert room.current_phase == "P2"
    
    # Submit new P0 constraint (should trigger rollback)
    rollback_tile = {
        "agent_id": "sentinel",
        "phase": "P0", 
        "content": "NEW CONSTRAINT: System overload detected",
        "force_rollback": True
    }
    
    result = room.submit_tile(rollback_tile)
    assert result.get("requires_rollback", False) == True
    
    # Room should revert to P1 or P0 based on implementation
    assert room.current_phase in ["P0", "P1"]
    # Previous P2 tiles should be marked for review
    assert room.get_pending_review_count() > 0
```

### Test 5: Resource Exhaustion Protection
```python
def test_resource_limits_enforced(self):
    """Test tile processing under resource constraints."""
    room = DeadbandRoom(max_tiles=100, max_tile_size=1024)  # Small limits for test
    room.initialize()
    
    # Fill room to capacity
    for i in range(100):
        room.submit_tile({
            "agent_id": f"agent_{i}",
            "phase": "P0",
            "content": f"Test tile {i}"
        })
    
    # Attempt to exceed tile count limit
    with pytest.raises(RuntimeError) as exc_info:
        room.submit_tile({
            "agent_id": "overflow_agent",
            "phase": "P1",
            "content": "Should be rejected"
        })
    assert "limit" in str(exc_info.value).lower() or "full" in str(exc_info.value).lower()
    
    # Attempt to exceed size limit
    large_content = "X" * 2048  # Exceeds 1024 limit
    with pytest.raises(ValueError) as exc_info:
        room.submit_tile({
            "agent_id": "large_agent",
            "phase": "P2",
            "content": large_content
        })
    assert "size" in str(exc_info.value).lower() or "too large" in str(exc_info.value).lower()
```
