def test_p2_before_p1_violation(self):
    """Test that agent cannot execute P2 optimization before completing P1 channel finding."""
    room = DeadbandRoom(room_id="edge-test-1")
    agent_id = room.register_agent("test-agent")
    
    # Attempt P2 directly (should fail)
    with pytest.raises(ProtocolViolationError) as exc_info:

    assert room.get_agent_state(agent_id)["current_phase"] == "P0"

### Test 2: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(room_id="edge-test-2")
    agents = [room.register_agent(f"agent-{i}") for i in range(3)]
    
    # All agents complete P0 and submit tiles concurrently
    tiles = []
    for agent_id in agents:
        room.execute_protocol_step(agent_id, "P0", {"negative_space": "test"})
        tile = room.submit_tile(agent_id, {"data": f"tile-from-{agent_id}"})
        tiles.append(tile)
    
    # Verify all tiles were accepted and stored
    room_state = room.get_state()
    assert len(room_state["tiles"]) == 3
    assert all(t["status"] == "accepted" for t in tiles)
```

### Test 4: Room Capacity Limits
```python
def test_room_capacity_limits(self):
    """Test that room enforces maximum agent capacity."""
    room = DeadbandRoom(room_id="edge-test-4", max_agents=2)
    
    # Register first two agents (should succeed)
    agent1 = room.register_agent("agent-1")
    agent2 = room.register_agent("agent-2")
    
    # Attempt to register third agent (should fail)
    with pytest.raises(RoomCapacityError) as exc_info:
        room.register_agent("agent-3")
    
    assert "Room at maximum capacity" in str(exc_info.value)
    assert len(room.get_state()["agents"]) == 2
    
    # Remove an agent and try again
    room.remove_agent(agent1)
    agent3 = room.register_agent("agent-3")
    assert agent3 is not None
```

### Test 5: Network Partition Simulation
```python
def test_network_partition_recovery(self):
    """Test agent disconnection and reconnection during protocol execution."""
    room = DeadbandRoom(room_id="edge-test-5")
    agent_id = room.register_agent("test-agent")
    
    # Start P0
    room.execute_protocol_step(agent_id, "P0", {"negative_space": "test"})
    
    # Simulate network partition (agent disappears)
    room._simulate_network_partition(agent_id)
    
    # Agent tries to continue (should fail)
    with pytest.raises(AgentDisconnectedError) as exc_info:
        room.execute_protocol_step(agent_id, "P1", {"channels": "test"})
    
    assert "Agent disconnected" in str(exc_info.value)
    
    # Agent reconnects and recovers state
    room._reconnect_agent(agent_id)
    state = room.get_agent_state(agent_id)
    assert state["current_phase"] == "P0"
    assert state["disconnected"] is False
    
    # Can continue from where left off
    result = room.execute_protocol_step(agent_id, "P1", {"channels": "test"})
    assert result["phase"] == "P1"
```
