"""
Edge-case tests for DeadbandRoom protocol implementation.
Tests scenarios beyond normal operation to validate robustness.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
import sys
from pathlib import Path

# Add parent directory to path for imports
sys.path.insert(0, str(Path(__file__).parent.parent))

from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.protocol import ProtocolViolationError
from plato_torch.core.tiles import Tile, GhostTile, TileValidationError


class TestDeadbandRoomEdgeCases:
    """Edge-case tests for DeadbandRoom protocol implementation."""
    
    @pytest.fixture
    def room(self):
        """Create a fresh DeadbandRoom instance for each test."""
        return DeadbandRoom(room_id="edge-test-room", capacity=10)
    
    @pytest.fixture
    def sample_tile(self):
        """Create a valid sample tile for testing."""
        return Tile(
            agent_id="test-agent",
            content="Test tile content",
            phase="P1",
            timestamp="2026-04-20T06:00:00Z",
            metadata={"test": True}
        )
    
    @pytest.fixture
    def sample_ghost_tile(self):
        """Create a valid ghost tile for testing."""
        return GhostTile(
            original_agent_id="ghost-agent",

            phase="P0",
            timestamp="2026-04-19T12:00:00Z",
            cause_of_death="timeout",
            lessons=["avoid infinite loops"]
        )
    
    # Test 1: Protocol Violation Recovery
    def test_protocol_violation_p2_before_p1(self, room):
        """
        Edge Case 1: Agent attempts P2 optimization before completing P1 channel finding.
        Expected: ProtocolViolationError raised, agent reset to P0.
        """
        # Simulate agent trying to submit P2 tile without P1 completion
        p2_tile = Tile(
            agent_id="violating-agent",
            content="Trying to optimize before finding channels",
            phase="P2",
            timestamp="2026-04-20T06:01:00Z"
        )
        
        # Agent should not have P1 completion flag
        room.agent_states["violating-agent"] = {"current_phase": "P0", "p1_complete": False}
        
        with pytest.raises(ProtocolViolationError) as exc_info:
            room.submit_tile(p2_tile)
        
        assert "P2 attempted before P1 completion" in str(exc_info.value)
        # Verify agent was reset to P0
        assert room.agent_states["violating-agent"]["current_phase"] == "P0"
    
    # Test 2: Concurrent Tile Submission
    @pytest.mark.asyncio
    async def test_concurrent_tile_submission(self, room, sample_tile):
        """
        Edge Case 2: Multiple agents submit tiles simultaneously.
        Expected: All tiles processed without data corruption or race conditions.
        """
        num_agents = 5
        tiles = []
        
        # Create unique tiles from different agents
        for i in range(num_agents):
            tile = Tile(
                agent_id=f"agent-{i}",
                content=f"Concurrent test content {i}",
                phase="P1",
                timestamp=f"2026-04-20T06:02:{i:02d}Z"
            )
            tiles.append(tile)
        
        # Submit all tiles concurrently
        tasks = [asyncio.create_task(self._submit_tile_async(room, tile)) for tile in tiles]
        results = await asyncio.gather(*tasks)
        
        # Verify all submissions succeeded
        assert all(results), "All concurrent submissions should succeed"
        
        # Verify tiles were stored correctly
        stored_tiles = room.get_recent_tiles(limit=num_agents)
        assert len(stored_tiles) == num_agents
        
        # Verify no duplicate agent IDs in stored tiles
        agent_ids = [t.agent_id for t in stored_tiles]
        assert len(set(agent_ids)) == num_agents, "No agent ID collisions during concurrent submission"
    
    async def _submit_tile_async(self, room, tile):
        """Helper method to submit tile asynchronously."""
        try:
            room.submit_tile(tile)
            return True
        except Exception:
            return False
    
    # Test 3: Resource Exhaustion During Large Batch
    def test_large_tile_batch_memory_pressure(self, room):
        """
        Edge Case 3: Submit large batch of tiles to test memory limits.
        Expected: Room handles batch within configured limits, may throttle or reject.
        """
        batch_size = 1000  # Large batch to simulate memory pressure
        rejected_count = 0
        
        for i in range(batch_size):
            tile = Tile(
                agent_id=f"batch-agent-{i % 10}",  # 10 agents submitting repeatedly
                content=f"Large batch content {i}" + "x" * 1000,  # 1KB content
                phase="P1",
                timestamp=f"2026-04-20T06:03:{i % 60:02d}Z"
            )
            
            try:
                room.submit_tile(tile)
            except (TileValidationError, ProtocolViolationError) as e:
                if "memory limit" in str(e).lower() or "capacity" in str(e).lower():
                    rejected_count += 1
                else:
                    raise
        
        # Verify room is still functional
        recent_tiles = room.get_recent_tiles(limit=10)
        assert len(recent_tiles) > 0, "Room should still have tiles after large batch"
        
        # Log rejection rate for analysis
        rejection_rate = rejected_count / batch_size
        print(f"Large batch test: {rejected_count}/{batch_size} tiles rejected ({rejection_rate:.1%})")
        
        # Room should handle most tiles unless explicitly configured with limits
        assert rejection_rate < 0.5, "Should not reject more than 50% of large batch"
