### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results if isinstance(r, dict))
    
    # Verify room state integrity
    assert room.get_tile_count() == len(agent_ids)
    assert len(set(room.get_agent_ids())) == len(agent_ids)
```

### Test 2: Malformed Tile Payload Recovery
```python
def test_malformed_tile_recovery(self):
    """Room should reject malformed tiles but remain operational."""
    room = DeadbandRoom()
    
    malformed_tiles = [
        None,  # null payload
        "not a dict",  # wrong type
        {},  # empty dict
        {"agent_id": "test"},  # missing required fields
        {"agent_id": "test", "phase": "P5", "content": "test"},  # invalid phase
        {"agent_id": "test", "phase": "P1", "content": ""},  # empty content
        {"agent_id": "test", "phase": "P1", "content": "x" * 10001},  # content too large
    ]
    
    valid_tile = {
        "agent_id": "test_agent",
        "phase": "P1",
        "content": "Valid test tile",
        "timestamp": time.time()
    }
    
    # Submit malformed tiles (should reject)
    rejections = []
    for tile in malformed_tiles:
        try:
            room.submit_tile(tile)
            rejections.append(False)  # Should not reach here
        except (ValueError, TypeError, KeyError) as e:
            rejections.append(True)
    
    # Submit valid tile (should accept)
    result = room.submit_tile(valid_tile)
    
    assert all(rejections), "All malformed tiles should be rejected"
    assert result["status"] == "accepted", "Valid tile should be accepted"
    assert room.get_tile_count() == 1, "Only valid tile should be stored"
```

### Test 3: Protocol Violation Detection and Correction

```python
def test_protocol_violation_correction(self):
    """Agent skipping P0→P2 should be corrected to P0."""
    room = DeadbandRoom()
    agent_id = "violating_agent"
    
    # Agent tries to skip from P0 directly to P2
    violation_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Trying to optimize without mapping negative space",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(violation_tile)
    
    # Room should detect violation and correct to P0
    assert result["status"] == "corrected"
    assert result["corrected_phase"] == "P0"
    assert "Protocol violation" in result["message"]
    
    # Agent's state should now be P0
    agent_state = room.get_agent_state(agent_id)
    assert agent_state["current_phase"] == "P0"
    assert agent_state["violation_count"] == 1
```

### Test 4: Resource Boundary Testing
```python
def test_resource_limits(self):
    """Room should handle resource constraints gracefully."""
    room = DeadbandRoom()
    
    # Test with maximum allowed tiles
    max_tiles = room.MAX_TILES_PER_SESSION
    for i in range(max_tiles):
        tile = {
            "agent_id": f"agent_{i % 10}",  # 10 agents rotating
            "phase": "P1",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # Should accept max tiles
    assert room.get_tile_count() == max_tiles
    
    # Next submission should be rejected
    excess_tile = {
        "agent_id": "excess_agent",
        "phase": "P1",
        "content": "One tile too many",
        "timestamp": time.time()
    }
    
    with pytest.raises(ResourceLimitError) as exc_info:
        room.submit_tile(excess_tile)
    
    assert "tile limit" in str(exc_info.value).lower()
    
    # Reset should clear resources
    room.reset()
    assert room.get_tile_count() == 0
    assert room.get_session_duration() < 1.0  # Fresh session
```

### Test 5: Partial Failure Recovery
```python
def test_partial_failure_recovery(self):
    """Room should recover from partial state corruption."""
    room = DeadbandRoom()
    
    # Create normal session state
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P1",
            "content": f"Normal tile {i}",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # Simulate partial corruption (e.g., disk failure during write)
    original_state = room._state.copy()
    
    # Corrupt one agent's state
    room._state["agents"]["agent_1"]["current_phase"] = "CORRUPTED"
    
    # Room should detect and repair on next access
    try:
        state = room.get_agent_state("agent_1")
        # If we get here, repair happened
        assert state["current_phase"] in ["P0", "P1", "P2"]
        assert "repaired" in state.get("flags", [])
    except StateCorruptionError:
        # Or should raise specific error
        room.repair_state()
        state = room.get_agent_state("agent_1")
        assert state["current_phase"] == "P0"  # Default safe state
    
    # Verify other agents unaffected
    assert room.get_agent_state("agent_0")["current_phase"] == "P1"
    assert room.get_agent_state("agent_2")["current_phase"] == "P1"
```
