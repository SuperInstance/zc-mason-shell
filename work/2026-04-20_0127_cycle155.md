```python
def test_priority_violation_skips_to_p2():
    """P0: Room should reject agents that skip directly to optimization."""
    room = DeadbandRoom(config=standard_config)
    agent_id = "violator_agent"
    
    # Agent tries to submit P2 optimization without P0/P1
    violation_tile = {
        "cycle": 10,
        "agent_id": agent_id,
        "phase": "P2",  # Skipping protocol
        "content": {"optimization": "fast_path"},
        "sentiment": 0.8
    }
    
    result = room.submit_tile(agent_id, violation_tile)
    assert not result["accepted"]
    assert "protocol violation" in result["message"].lower()
    assert agent_id in room._violation_log
```

### Test 3: Concurrent Agent Collision
```python
def test_concurrent_agent_tile_collision():
    """P1: Room should handle simultaneous submissions safely."""
    import threading
    
    room = DeadbandRoom(config=standard_config)
    results = []
    
    def submit_agent_tile(agent_id):
        tile = {
            "cycle": 1,
            "agent_id": agent_id,
            "phase": "P0",
            "content": {"negative_space": ["test"]},
            "sentiment": 0.5
        }
        results.append(room.submit_tile(agent_id, tile))
    
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_agent_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Should have exactly 10 results, all accepted
    assert len(results) == 10
    accepted_count = sum(1 for r in results if r["accepted"])
    assert accepted_count == 10
    
    # Room should have 10 tiles in history
    assert len(room.tile_history) == 10
```

### Test 4: Resource Exhaustion Protection
```python
def test_memory_exhaustion_safety():
    """P0: Room should limit tile history to prevent OOM."""
    room = DeadbandRoom(config=standard_config)
    
    # Submit more tiles than max_history allows
    max_history = room.config.get("max_history", 1000)
    for i in range(max_history + 100):
        tile = {
            "cycle": i,
            "agent_id": f"agent_{i % 10}",
            "phase": "P0",
            "content": {"negative_space": [f"item_{j}" for j in range(100)]},  # Large content
            "sentiment": 0.5
        }
        room.submit_tile(f"agent_{i % 10}", tile)
    
    # History should be trimmed to max_history
    assert len(room.tile_history) <= max_history
    
    # Oldest tiles should be removed first
    oldest_cycle = min(t["cycle"] for t in room.tile_history)
    assert oldest_cycle > 0  # Some early cycles were evicted
```

### Test 5: Ghost Tile Sentiment Boundary
```python
def test_ghost_tile_sentiment_boundaries():
    """P1: Ghost tiles (sentiment <= -0.8) should be stored but not trained on."""
    room = DeadbandRoom(config=standard_config)
    
    boundary_tiles = [
        {"cycle": 1, "agent_id": "ghost", "phase": "P0", 
         "content": {"negative_space": ["fatal_error"]}, "sentiment": -0.8},  # Boundary
        {"cycle": 2, "agent_id": "ghost", "phase": "P0", 
         "content": {"negative_space": ["fatal_error"]}, "sentiment": -0.9},  # Below
        {"cycle": 3, "agent_id": "active", "phase": "P0", 
         "content": {"negative_space": ["minor_issue"]}, "sentiment": -0.7},  # Above
    ]
    
    for tile in boundary_tiles:
        room.submit_tile(tile["agent_id"], tile)
    
    # All tiles should be in history
    assert len(room.tile_history) == 3
    
    # Only non-ghost tiles should be in training set
    training_tiles = room.get_training_batch(batch_size=10)
    training_cycles = {t["cycle"] for t in training_tiles}
    
    assert 1 not in training_cycles  sentiment -0.8 (ghost boundary)
    assert 2 not in training_cycles  sentiment -0.9 (ghost)
    assert 3 in training_cycles  sentiment -0.7 (trainable)
    
    # Ghost tiles should be marked in metadata
    ghost_tiles = [t for t in room.tile_history if t.get("metadata", {}).get("ghost")]
    assert len(ghost_tiles) == 2
```
