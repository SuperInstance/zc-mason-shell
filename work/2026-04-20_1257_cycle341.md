## Edge-Case Test Proposals

### Test 1: Empty Tile Submission
**Scenario:** Agent submits a tile with empty content (zero-length string).
**Edge:** Should this be rejected as invalid input or accepted as a "null operation"?
**Expected:** Rejection with appropriate error code (P0 violation - no useful work).

### Test 2: Protocol Priority Violation
**Scenario:** Agent submits a tile that attempts P2 optimization without establishing P1 safe channels.
**Edge:** The tile claims to optimize a path in undefined space.
**Expected:** Rejection with "P1 not satisfied before P2" error.

### Test 3: Maximum Tile Size Boundary
**Scenario:** Agent submits a tile at exactly the maximum allowed size (e.g., 1MB if that's the limit).
**Edge:** Testing boundary condition for tile storage/processing.
**Expected:** Acceptance but with potential performance warning if processing is slow.

### Test 4: Concurrent Tile Submission Race Condition
**Scenario:** Two agents submit tiles simultaneously that would update the same room state variable.
**Edge:** Testing thread safety in room state management.
**Expected:** Either sequential processing with locks or conflict resolution mechanism.

### Test 5: Malformed JSON in Tile Metadata
**Scenario:** Agent submits tile with valid content but malformed metadata JSON.
**Edge:** Testing robustness of parsing logic.
**Expected:** Graceful rejection with parsing error, not a crash.

import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.models.tile import Tile

class TestDeadbandRoomEdgeCases:
    
    def test_empty_tile_rejection(self):
        """Test that empty content tiles are rejected."""
        room = DeadbandRoom(room_id="test-edge-1")
        empty_tile = Tile(
            agent_id="test-agent",
            content="",  # Empty string
            protocol_phase="P1",
            metadata={"task": "empty test"}
        )
        
        result = room.submit_tile(empty_tile)
        assert not result["accepted"]
        assert "empty" in result["reason"].lower() or "invalid" in result["reason"].lower()
    
    def test_protocol_priority_violation(self):
        """Test that P2 without P1 is rejected."""
        room = DeadbandRoom(room_id="test-edge-2")
        
        # First, verify room has no P1 channels established
        assert len(room.safe_channels) == 0
        
        # Attempt P2 optimization tile
        p2_tile = Tile(
            agent_id="test-agent",
            content="Optimizing path X with factor Y",
            protocol_phase="P2",  # Skipping P1
            metadata={"optimization": "premature"}
        )
        
        result = room.submit_tile(p2_tile)
        assert not result["accepted"]
        assert "P1" in result["reason"] and "P2" in result["reason"]
    
    def test_max_tile_size_boundary(self):
        """Test tile at maximum size limit."""
        room = DeadbandRoom(room_id="test-edge-3")

            agent_id="test-agent",

            protocol_phase="P1",
            metadata={"size_test": "boundary"}

        result = room.submit_tile(boundary_tile)
        # Should be accepted but might have performance implications
        assert result["accepted"] == True
        # Optional: check processing time warning if implemented

            assert "performance" in result["warning"].lower()
    
    def test_concurrent_tile_submission(self, mocker):
        """Test race condition handling for concurrent submissions."""
        room = DeadbandRoom(room_id="test-edge-4")
        
        # Mock two agents submitting simultaneously
        tile1 = Tile(
            agent_id="agent-1",
            content="Channel A is safe",
            protocol_phase="P1",
            metadata={"channel": "A"}
        )
        
        tile2 = Tile(
            agent_id="agent-2", 
            content="Channel B is safe",
            protocol_phase="P1",
            metadata={"channel": "B"}
        )
        
        # Use threading to simulate concurrency
        import threading
        results = []
        
        def submit_tile(tile):
            result = room.submit_tile(tile)
            results.append(result)
        
        thread1 = threading.Thread(target=submit_tile, args=(tile1,))
        thread2 = threading.Thread(target=submit_tile, args=(tile2,))
        
        thread1.start()
        thread2.start()
        thread1.join()
        thread2.join()
        
        # Both should be accepted
        assert len(results) == 2
        assert all(r["accepted"] for r in results)
        
        # Room should have both channels
        assert len(room.safe_channels) == 2
        assert "A" in room.safe_channels and "B" in room.safe_channels
