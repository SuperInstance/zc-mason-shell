def test_concurrent_tile_submission(self):
    """Test DeadbandRoom handling of simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    
    # Simulate 10 agents submitting tiles at the same time
    tiles = [
        Tile(agent_id=f"agent_{i}", 
             content=f"Test tile {i}",
             protocol_phase="P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2")
        for i in range(10)
    ]
    
    # Submit all tiles concurrently (simulated)
    results = []
    for tile in tiles:
        result = room.process_tile(tile)
        results.append(result)
    
    # Verify all tiles were processed without corruption
    assert len(results) == 10
    assert all(r.get('status') in ['accepted', 'rejected'] for r in results)
    
    # Verify protocol phase consistency was maintained
    accepted_tiles = [r for r in results if r.get('status') == 'accepted']
    if accepted_tiles:
        phases = [tile.protocol_phase for tile in tiles 
                 if any(r.get('tile_id') == tile.id for r in accepted_tiles)]
        assert len(set(phases)) <= 3  # Should only contain P0, P1, P2
```

### Test 2: Malformed Tile Structure Handling

def test_malformed_tile_handling(self):
    """Test room's resilience against corrupted or incomplete tile data."""
    room = DeadbandRoom()
    
    test_cases = [
        # (tile_data, expected_handling)
        (None, "reject"),  # Null tile
        ({}, "reject"),  # Empty dict
        ({"agent_id": "test_agent"}, "reject"),  # Missing content
        ({"content": "test"}, "reject"),  # Missing agent_id
        ({"agent_id": "test", "content": "", "protocol_phase": "P0"}, "accept"),  # Empty content (valid)
        ({"agent_id": "test", "content": "x" * 10000, "protocol_phase": "P3"}, "reject"),  # Invalid phase
    ]
    
    for tile_data, expected in test_cases:
        try:
            if isinstance(tile_data, dict):
                tile = Tile(**tile_data)
            else:
                tile = tile_data
                
            result = room.process_tile(tile)
            
            if expected == "reject":
                assert result.get('status') == 'rejected'
                assert 'error' in result
            else:
                assert result.get('status') == 'accepted'
        except (TypeError, ValueError) as e:
            # Constructor validation failed - room should handle gracefully
            if expected == "reject":
                pass  # Expected failure
            else:
                raise AssertionError(f"Unexpected error: {e}")
```

### Test 4: Network Partition Scenario
```python
def test_partial_connectivity_scenario(self):
    """Test protocol behavior during simulated network issues."""
    room = DeadbandRoom()
    
    # Simulate intermittent connectivity
    connectivity_states = [True, False, True, False, True]  # Flapping connection
    
    tiles_submitted = 0
    tiles_processed = 0
    
    for i, connected in enumerate(connectivity_states):
        tile = Tile(
            agent_id=f"agent_{i}",
            content=f"Tile during connectivity state: {connected}",
            protocol_phase="P0" if i == 0 else "P1"
        )
        
        if connected:
            # Normal processing
            result = room.process_tile(tile)
            tiles_processed += 1
            assert result.get('status') in ['accepted', 'rejected']
        else:
            # Simulate network failure
            # Room should either queue or reject with network error
            try:
                result = room.process_tile(tile, simulate_network_failure=True)
                # If it returns, it should indicate network issue
                assert result.get('status') == 'rejected'
                assert 'network' in result.get('error', '').lower()
            except ConnectionError:
                # Expected exception for network failure
                pass
        
        tiles_submitted += 1
    
    # Verify some tiles were processed despite network issues
    assert tiles_processed > 0
    assert tiles_processed < tiles_submitted  # Some should have failed
```
