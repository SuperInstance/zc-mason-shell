```rust
#[test]
fn test_p0_to_p2_skip_violation() {
    let mut room = DeadbandRoom::new("test-skip".to_string(), 100);
    
    // Force P0 completion
    room.current_phase = ProtocolPhase::P0;
    room.p0_complete = true;
    
    // Attempt to execute P2 without P1
    let result = room.execute_protocol_step(ProtocolPhase::P2);
    
    assert!(result.is_err());
    assert!(result.unwrap_err().contains("P1 not complete"));
    
    // Verify room didn't advance to P2
    assert_eq!(room.current_phase, ProtocolPhase::P0);
}
```

### Test 3: Tile Capacity Limits
```rust
#[test]
fn test_tile_capacity_enforcement() {
    let max_tiles = 10;
    let mut room = DeadbandRoom::new("test-capacity".to_string(), max_tiles);
    
    // Fill to capacity
    for i in 0..max_tiles {
        let tile = format!(
            r#"{{"agent_id": "agent-{}", "content": "tile-{}", "timestamp": {}}}"#,
            i, i, 1234567890 + i
        );
        room.add_interaction_tile(tile).unwrap();
    }
    
    assert_eq!(room.interaction_tiles.len(), max_tiles);
    
    // Attempt to exceed capacity
    let extra_tile = r#"{"agent_id": "extra", "content": "overflow", "timestamp": 1234567900}"#;
    
    let result = room.add_interaction_tile(extra_tile.to_string());
    assert!(result.is_err());
    assert!(result.unwrap_err().contains("capacity"));
    
    // Verify count unchanged
    assert_eq!(room.interaction_tiles.len(), max_tiles);
}
```

### Test 4: Concurrent Tile Submission Simulation
```rust
#[test]
fn test_concurrent_tile_race_condition() {
    let mut room = DeadbandRoom::new("test-concurrent".to_string(), 100);
    let room_ptr = Arc::new(Mutex::new(room));
    
    let mut handles = vec![];
    
    // Simulate 5 agents submitting tiles simultaneously
    for agent_id in 0..5 {
        let room_clone = Arc::clone(&room_ptr);
        let handle = thread::spawn(move || {
            let tile = format!(
                r#"{{"agent_id": "agent-{}", "content": "concurrent-test", "timestamp": {}}}"#,
                agent_id, SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs()
            );
            
            let mut room = room_clone.lock().unwrap();
            room.add_interaction_tile(tile)
        });
        handles.push(handle);
    }
    
    // Collect results
    let mut successes = 0;
    for handle in handles {
        if handle.join().unwrap().is_ok() {
            successes += 1;
        }
    }
    
    let final_room = room_ptr.lock().unwrap();
    
    // All submissions should succeed (thread-safe implementation)
    assert_eq!(successes, 5);
    assert_eq!(final_room.interaction_tiles.len(), 5);
    
    // Verify all agent IDs are present
    let agent_ids: HashSet<_> = final_room.interaction_tiles
        .iter()
        .map(|t| t.agent_id.clone())
        .collect();
    
    for i in 0..5 {
        assert!(agent_ids.contains(&format!("agent-{}", i)));
    }
}
```

### Test 5: Corrupted State Recovery
```rust
#[test]
fn test_partial_state_recovery() {
    // Create a valid room state
    let mut room = DeadbandRoom::new("test-recovery".to_string(), 50);
    room.add_interaction_tile(
        r#"{"agent_id": "test", "content": "valid", "timestamp": 1234567890}"#.to_string()
    ).unwrap();
    
    // Serialize to JSON
    let serialized = serde_json::to_string(&room).unwrap();
    
    // Corrupt the JSON (truncate it)
    let corrupted = &serialized[0..serialized.len() / 2];
    
    // Attempt deserialization
    let result: Result<DeadbandRoom, _> = serde_json::from_str(corrupted);
    
    // Should fail gracefully
    assert!(result.is_err());
    
    // Alternative: Test with missing required fields in serialized form
    let mut partial_state: HashMap<String, serde_json::Value> = serde_json::from_str(&serialized).unwrap();
    partial_state.remove("room_id"); // Remove required field
    
    let corrupted_json = serde_json::to_string(&partial_state).unwrap();
    let result: Result<DeadbandRoom, _> = serde_json::from_str(&corrupted_json);
    
    assert!(result.is_err());
    assert!(result.unwrap_err().to_string().contains("missing field"));
}
```
