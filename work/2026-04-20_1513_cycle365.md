```python
def test_mixed_validity_tiles():
    """P1: Room should process valid tiles even when some are invalid."""
    room = DeadbandRoom(
        room_id="test-room",
        deadband_width=1.0,
        learning_rate=0.01,
        max_tiles=1000
    )
    
    tiles = [
        {"content": "valid tile 1", "metadata": {"phase": "P1"}},  # Valid
        {"invalid": "missing content field"},  # Invalid
        {"content": "valid tile 2", "metadata": {"phase": "P2"}},  # Valid
        {},  # Empty, invalid
    ]
    
    # Should process valid tiles, skip invalid ones
    loss = room.train_step(tiles)
    
    # Loss should be computed only from valid tiles
    assert isinstance(loss, float)
    assert loss >= 0.0
    # Room state should reflect only valid tiles processed
    assert room.tiles_processed == 2  # Only the two valid tiles
```

### Test 4: Large Batch Handling
```python
def test_large_tile_batch():
    """P1: Room should handle batches up to max_tiles without crashing."""
    max_tiles = 1000
    room = DeadbandRoom(
        room_id="test-room",
        deadband_width=1.0,
        learning_rate=0.01,
        max_tiles=max_tiles
    )
    
    # Create batch at maximum capacity
    tiles = [
        {"content": f"tile_{i}", "metadata": {"phase": "P1"}}
        for i in range(max_tiles)
    ]
    
    loss = room.train_step(tiles)
    assert isinstance(loss, float)
    assert room.tiles_processed == max_tiles
    
    # Attempt to exceed capacity (should handle gracefully)
    extra_tiles = tiles + [{"content": "extra", "metadata": {"phase": "P1"}}]
    # Implementation may truncate or raise error - test documents actual behavior
    try:
        loss = room.train_step(extra_tiles)
        # If no error, verify processing didn't exceed max
        assert room.tiles_processed <= max_tiles * 2  # Reasonable upper bound
    except ValueError as e:
        assert "exceeds maximum" in str(e) or "capacity" in str(e)
```

### Test 5: Protocol Phase Tracking
```python
def test_protocol_phase_transitions():
    """P2: Room should track and optimize based on protocol phase."""
    room = DeadbandRoom(
        room_id="test-room",
        deadband_width=1.0,
        learning_rate=0.01,
        max_tiles=1000
    )
    
    # Initial state should be P0 (negative space mapping)
    assert room.current_phase == "P0"
    
    # P0 tiles (negative examples)
    p0_tiles = [
        {"content": "avoid this pattern", "metadata": {"phase": "P0"}},
        {"content": "invalid approach", "metadata": {"phase": "P0"}},
    ]
    
    loss_p0 = room.train_step(p0_tiles)
    # After processing P0 tiles, room may transition or stay in P0
    # Document actual behavior
    
    # P1 tiles (safe channels)
    p1_tiles = [
        {"content": "safe approach A", "metadata": {"phase": "P1"}},
        {"content": "valid method B", "metadata": {"phase": "P1"}},
    ]
    
    loss_p1 = room.train_step(p1_tiles)
    
    # P2 tiles (optimization)
    p2_tiles = [
        {"content": "optimize parameter X", "metadata": {"phase": "P2"}},
        {"content": "improve efficiency Y", "metadata": {"phase": "P2"}},
    ]
    
    loss_p2 = room.train_step(p2_tiles)
    
    # Verify all phases processed
    assert room.tiles_processed == 6
    # Loss should generally decrease P0→P1→P2 (optimization)
    # But not guaranteed - document actual pattern
    losses = [loss_p0, loss_p1, loss_p2]
    for i in range(1, len(losses)):
        if losses[i] < losses[i-1]:
            print(f"Optimization observed: phase transition reduced loss")
```
