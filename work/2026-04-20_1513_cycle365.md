```python
def test_concurrent_tile_submission_ordering():
    """Test that tiles maintain submission order when submitted in rapid succession."""
    room = DeadbandRoom(
        room_id="concurrent_test",
        max_cycles=10,
        max_tiles=5,
        protocol_steps=["P0", "P1", "P2"]
    )
    
    # Simulate concurrent submissions with timestamps
    import time
    tiles = []
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"test_content_{i}",
            "timestamp": time.time() + i * 0.001  # Slight offset
        }
        tiles.append(tile)
    
    # Submit all tiles
    for tile in tiles:
        room.submit_tile(tile)
    
    # Verify tiles are stored in submission order
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == 3
    for i, stored_tile in enumerate(stored_tiles):
        assert stored_tile["agent_id"] == f"agent_{i}"
        assert stored_tile["content"] == f"test_content_{i}"
```

### Test 3: Partial Protocol Failure Recovery

```python
def test_partial_protocol_failure_recovery():
    """Test room state when one agent fails mid-protocol but others continue."""
    room = DeadbandRoom(
        room_id="failure_recovery_test",
        max_cycles=5,
        max_tiles=10,
        protocol_steps=["P0", "P1", "P2"]
    )
    
    # Start protocol normally
    room.execute_protocol_step("P0")
    assert room.current_protocol_step == "P0"
    
    # Submit tiles from multiple agents
    room.submit_tile({"agent_id": "agent_a", "content": "P0 work"})
    room.submit_tile({"agent_id": "agent_b", "content": "P0 work"})
    
    # Simulate agent_b failing (no further submissions)
    # Move to P1 with only agent_a participating
    room.execute_protocol_step("P1")
    room.submit_tile({"agent_id": "agent_a", "content": "P1 work"})
    
    # Room should continue despite partial participation
    assert room.current_protocol_step == "P1"
    assert len(room.get_tiles()) == 3
    assert not room.is_complete()
```

### Test 5: Protocol Skipping Detection
```python
def test_protocol_skipping_detection():
    """Test that attempting to skip protocol steps raises appropriate errors."""
    room = DeadbandRoom(
        room_id="skip_test",
        max_cycles=3,
        max_tiles=5,
        protocol_steps=["P0", "P1", "P2"]
    )
    
    # Normal progression
    room.execute_protocol_step("P0")
    assert room.current_protocol_step == "P0"
    
    # Attempt to jump to P2 (should fail based on Deadband Protocol)
    # Implementation may either raise error or ignore
    try:
        room.execute_protocol_step("P2")
        # If no error, verify state didn't change incorrectly
        assert room.current_protocol_step != "P2"
        assert room.current_protocol_step in ["P0", "P1"]
    except ValueError as e:
        assert "protocol" in str(e).lower() or "skip" in str(e).lower() or "order" in str(e).lower()
    
    # Complete protocol normally
    room.execute_protocol_step("P1")
    room.execute_protocol_step("P2")
    assert room.is_complete() == True
```
