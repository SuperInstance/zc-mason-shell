### 2. Test: Concurrent Tile Submissions
```python
def test_concurrent_tile_submissions(self):
    """Multiple agents submitting tiles simultaneously"""
    config = {"room_id": "concurrent-test"}
    room = DeadbandRoom(config)
    
    def submit_tile(agent_id):
        tile = {
            "agent": agent_id,
            "p0": {"avoid": [f"rock-{agent_id}"]},
            "timestamp": time.time()
        }
        return room.submit_tile(tile)
    
    # Simulate 10 concurrent submissions
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = [executor.submit(submit_tile, i) for i in range(10)]
        results = [f.result() for f in futures]
    
    # All submissions should be accepted
    assert all(r["status"] == "accepted" for r in results)
    # Room should track all agents
    assert len(room.active_agents) == 10
```

### 3. Test: Malformed P0 Data Recovery
```python
def test_malformed_p0_recovery(self):
    """Room should handle corrupted negative space data"""
    config = {"room_id": "recovery-test"}
    room = DeadbandRoom(config)
    
    # Intentionally malformed P0 data
    test_cases = [
        {"p0": {"avoid": None}},  # None instead of list
        {"p0": {"avoid": [{}]}},  # Empty dict in list
        {"p0": {"avoid": "not-a-list"}},  # Wrong type
        {"p0": {}},  # Missing avoid key
    ]
    
    for i, data in enumerate(test_cases):
        try:
            result = room.process(data)
            # Should either handle gracefully or raise specific error
            assert "error" in result or "p1" in result
        except (TypeError, ValueError) as e:
            # Expected for some malformed cases
            assert "invalid" in str(e).lower() or "missing" in str(e).lower()
```

### 4. Test: Resource Exhaustion During P2
```python
def test_p2_optimization_resource_limits(self):
    """P2 optimization with memory/CPU constraints"""
    config = {
        "room_id": "resource-test",
        "resource_limits": {
            "max_memory_mb": 50,
            "max_time_seconds": 1
        }
    }
    room = DeadbandRoom(config)
    
    # Create optimization problem that could be expensive
    data = {
        "p0": {"avoid": ["overflow"]},
        "p1": {"channels": [{"range": [0, 1000000]}]},
        "p2": {
            "optimize": "grid_search",
            "resolution": 10000  # High resolution grid
        }
    }
    
    start_time = time.time()
    start_memory = psutil.Process().memory_info().rss / 1024 / 1024
    
    result = room.process(data)
    
    elapsed = time.time() - start_time
    end_memory = psutil.Process().memory_info().rss / 1024 / 1024
    memory_used = end_memory - start_memory
    
    # Should respect resource limits
    assert elapsed < 2.0  # Allow some buffer
    assert memory_used < 55  # MB, with buffer
    
    # Should still produce valid result or graceful degradation
    assert "p2" in result or "degraded" in result["status"]
```

### 5. Test: Network Partition Simulation
```python
def test_partial_data_availability(self):
    """Processing with incomplete data due to network issues"""
    config = {"room_id": "partition-test"}
    room = DeadbandRoom(config)
    
    # Simulate staggered data arrival
    partial_data = [
        {"p0": {"avoid": ["node1-down"]}},  # First packet
        {"p1": {"channels": []}},  # Second packet, no P0
        {"p2": {"optimize": "nearest"}},  # Third packet, no P0/P1
    ]
    
    results = []
    for data in partial_data:
        try:
            result = room.process(data)
            results.append(result)
        except Exception as e:
            results.append({"error": str(e)})
    
    # Analyze results
    # First should work (has P0)
    assert "p1" in results[0] or "error" not in results[0]
    
    # Second might fail (missing P0) or use cached
    assert isinstance(results[1], dict)
    
    # Third should fail or use cached channels
    assert "p2" not in results[2] or "error" in results[2]
    
    # Room should maintain state across partial updates
    assert hasattr(room, 'last_valid_state')
```
