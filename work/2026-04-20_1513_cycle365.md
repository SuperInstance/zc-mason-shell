```python
# File: plato-torch/tests/test_deadband_room_edge.py
import unittest
import numpy as np
from plato_torch.rooms.deadband_room import DeadbandRoom

class TestDeadbandRoomEdgeCases(unittest.TestCase):
    
    def setUp(self):
        self.room = DeadbandRoom()
    
    def test_state_with_nan_values(self):
        """Test P0 detection when state contains NaN values"""
        # State with NaN in position
        state = {
            'position': np.array([np.nan, 0.0]),
            'velocity': np.array([1.0, 0.0]),
            'obstacles': []
        }
        action = 0
        
        # Should detect as P0 (negative space) due to invalid state
        result = self.room._apply_deadband_protocol(state, action)
        self.assertEqual(result['phase'], 'P0', 
                        "NaN in state should trigger P0 negative space")
        self.assertTrue(result['is_negative_space'],
                       "State with NaN should be marked as negative space")
    
    def test_action_out_of_bounds(self):
        """Test behavior when action index is outside valid range"""
        valid_state = {
            'position': np.array([0.0, 0.0]),
            'velocity': np.array([1.0, 0.0]),
            'obstacles': []
        }
        
        # Action index beyond maximum (valid: 0-7)
        invalid_action = 10
        
        # Should handle gracefully, likely P0 or specific error
        result = self.room._apply_deadband_protocol(valid_state, invalid_action)
        
        # Either P0 (invalid action) or raises IndexError
        self.assertIn(result['phase'], ['P0', 'ERROR'],
                     "Out-of-bounds action should not proceed to P1/P2")
    
    def test_concurrent_state_modification(self):
        """Test protocol stability when state dict is modified during processing"""
        state = {
            'position': np.array([0.0, 0.0]),
            'velocity': np.array([1.0, 0.0]),
            'obstacles': []
        }
        action = 0
        
        # Simulate concurrent modification by callback
        original_p1_method = self.room._p1_safe_channels
        
        def modifying_p1(state):
            # Modify state during P1 calculation
            state['position'][0] = 999.0  # Concurrent modification
            return original_p1_method(state)
        
        self.room._p1_safe_channels = modifying_p1
        
        try:
            result = self.room._apply_deadband_protocol(state.copy(), action)
            # Should complete without crash
            self.assertIn(result['phase'], ['P0', 'P1', 'P2'],
                         "Should handle concurrent modifications gracefully")
        finally:
            # Restore original method
            self.room._p1_safe_channels = original_p1_method
    
    def test_protocol_phase_transition_edge(self):
        """Test edge case where state is exactly at P0/P1 boundary"""
        # Create state that's barely not in negative space
        state = {
            'position': np.array([0.499, 0.0]),  # Just inside safe boundary
            'velocity': np.array([0.0, 0.0]),
            'obstacles': [{'position': np.array([0.5, 0.0]), 'radius': 0.1}]
        }
        action = 0
        
        # Obstacle at 0.5, state at 0.499 with radius 0.1
        # Distance = 0.001, which is less than radius (0.1)
        # This should be P0 (collision), but tests boundary precision
        
        result = self.room._apply_deadband_protocol(state, action)
        
        # Verify consistent behavior
        if result['phase'] == 'P0':
            self.assertTrue(result['is_negative_space'],
                           "Boundary collision should be P0")
        else:
            # If not P0, should have safe channels
            self.assertGreater(len(result.get('safe_actions', [])), 0,
                              "If not P0, should have P1 safe channels")
    
    def test_high_velocity_state_overflow(self):
        """Test state with extremely high velocity values"""
        # State with velocity that could cause numerical issues
        state = {
            'position': np.array([0.0, 0.0]),
            'velocity': np.array([1e10, 1e10]),  # Extremely high velocity
            'obstacles': []
        }
        action = 0
        
        # Should handle without overflow or crash
        result = self.room._apply_deadband_protocol(state, action)
        
        self.assertIn(result['phase'], ['P0', 'P1', 'P2'],
                     "Should handle extreme velocity values")
        
        # Check that velocity didn't get corrupted
        if 'next_state' in result:
            self.assertFalse(np.any(np.isnan(result['next_state'].get('velocity', []))),
                           "Velocity should not become NaN")
            self.assertFalse(np.any(np.isinf(result['next_state'].get('velocity', []))),
                           "Velocity should not become infinite")

if __name__ == '__main__':
    unittest.main()
```
