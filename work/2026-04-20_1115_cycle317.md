## New Edge-Case Tests

### Test 1: Tile Buffer Overflow Protection
**Scenario:** The `_tile_buffer` has a maximum capacity (implied by memory constraints). Add tiles until limit, verify oldest tiles are discarded or error is raised.
**Rationale:** Prevents memory exhaustion in long-running rooms.
```python
def test_tile_buffer_overflow(self):
    """Adding tiles beyond buffer capacity should trigger cleanup or rejection."""
    room = DeadbandRoom(max_buffer_size=10)
    # Add 15 tiles
    for i in range(15):
        tile = {"id": f"tile_{i}", "content": f"test_{i}", "priority": 1}
        room.add_tile(tile)
    # Buffer should not exceed max_buffer_size
    assert len(room._tile_buffer) <= 10
    # Oldest tiles (tile_0..tile_4) should be removed
    remaining_ids = {t["id"] for t in room._tile_buffer}
    assert "tile_0" not in remaining_ids
    assert "tile_14" in remaining_ids  # Newest tile retained
```

### Test 2: Priority Zero Tile Handling
**Scenario:** Add a tile with `priority=0` (P0: negative space mapping). Ensure it's stored but not used in training cycles.
**Rationale:** P0 tiles are for knowledge preservation, not training.
```python
def test_priority_zero_tile_skipped_in_training(self):
    """P0 tiles should be archived but excluded from training cycles."""
    room = DeadbandRoom()
    p0_tile = {"id": "p0_ghost", "content": "avoid_this_pattern", "priority": 0}
    p1_tile = {"id": "p1_safe", "content": "safe_channel", "priority": 1}
    
    room.add_tile(p0_tile)
    room.add_tile(p1_tile)
    
    # Train cycle should only process P1+ tiles
    result = room.train_cycle()
    assert result["tiles_processed"] == 1  # Only p1_safe
    assert "p0_ghost" in room._archive  # P0 tile moved to archive
```

### Test 3: Concurrent Tile Addition During Training
**Scenario:** Simulate a tile being added while `train_cycle()` is executing.
**Rationale:** Real-world rooms may receive new tiles mid-cycle; ensure thread safety or proper locking.
```python
def test_concurrent_tile_addition_during_training(self):
    """Adding tiles during train_cycle should not corrupt buffer."""
    room = DeadbandRoom()
    # Start with some tiles
    for i in range(3):
        room.add_tile({"id": f"initial_{i}", "content": "data", "priority": 1})
    
    # Mock train_cycle to simulate mid-execution addition
    original_train = room.train_cycle
    def mocked_train():
        # Add a tile while training
        room.add_tile({"id": "mid_cycle", "content": "new", "priority": 2})
        return original_train()
    room.train_cycle = mocked_train
    
    result = room.train_cycle()
    # Buffer should be consistent
    assert len(room._tile_buffer) >= 0
    # The mid-cycle tile might be processed next cycle, but no crash
    assert "mid_cycle" in [t["id"] for t in room._tile_buffer]
```

### Test 4: Malformed Tile Recovery
**Scenario:** Add a tile with a valid structure but extreme values (e.g., huge content string, negative priority).
**Rationale:** Robustness against data corruption or adversarial inputs.
```python
def test_malformed_tile_extreme_values(self):
    """Room should handle tiles with extreme field values gracefully."""
    room = DeadbandRoom()
    # Extremely large content
    large_tile = {
        "id": "big",
        "content": "x" * 10_000_000,  # 10MB string
        "priority": 1
    }
    # This should not crash; room may truncate or reject
    try:
        room.add_tile(large_tile)
        # If accepted, check it's handled
        assert len(room._tile_buffer) == 1
        # Content may be truncated
        assert len(room._tile_buffer[0]["content"]) <= 10_000_000
    except (MemoryError, ValueError) as e:
        # Acceptable to raise error for resource protection
        assert "exceeds limit" in str(e).lower() or "memory" in str(e).lower()
    
    # Negative priority
    neg_tile = {"id": "neg", "content": "test", "priority": -5}
    # Should be clamped to 0 or raise error
    try:
        room.add_tile(neg_tile)
        assert room._tile_buffer[-1]["priority"] >= 0
    except ValueError as e:
        assert "priority" in str(e).lower()
```

### Test 5: Export Ensign with Partial Training History
**Scenario:** Call `export_ensign()` after only a few training cycles, with mixed success/failure results.
**Rationale:** Ensign adapters should be exportable even with incomplete training.
```python
def test_export_ensign_partial_history(self):
    """Export should work with incomplete training history (some cycles failed)."""
    room = DeadbandRoom()
    # Add some tiles
    for i in range(5):
        room.add_tile({"id": f"t{i}", "content": f"c{i}", "priority": 1})
    
    # Mock training to simulate partial success
    room.train_cycle = MagicMock(side_effect=[
        {"success": True, "loss": 0.5},
        {"success": False, "error": "gradient overflow"},
        {"success": True, "loss": 0.3}
    ])
    # Run three cycles (two success, one failure)
    for _ in range(3):
        try:
            room.train_cycle()
        except:
            pass  # Failure is expected
    
    # Export should still work, maybe with warnings
    exporter_mock = MagicMock()
    room._exporter = exporter_mock
    result = room.export_ensign("partial_test")
    
    # Exporter called with available history
    assert exporter_mock.export.called
    call_args = exporter_mock.export.call_args
    # History should include both successful cycles
    history = call_args[0][1]  # Second argument is training_history
    assert len(history) == 2  # Only successful cycles
    assert all("loss" in h for h in history)
```
