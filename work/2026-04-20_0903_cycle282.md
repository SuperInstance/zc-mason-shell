### Test 1: Concurrent Agent Tile Collision
```python
def test_concurrent_tile_submission_collision(self):
    """Test that concurrent tile submissions from multiple agents don't corrupt room state."""
    room = DeadbandRoom(capacity=5)
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 282,
            "content": f"Test from {agent_id}",
            "protocol_phase": "P1"
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (in rapid succession)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions succeeded
    assert all(r["status"] == "accepted" for r in results)
    
    # Verify room state integrity
    assert len(room.get_tiles()) == 3
    assert set(t["agent_id"] for t in room.get_tiles()) == set(agent_ids)
```

```python
def test_partial_network_failure_recovery(self):
    """Test room handles partial tile data due to network failure."""
    room = DeadbandRoom(capacity=10)
    
    # Simulate truncated tile data (missing required fields)
    malformed_tile = {
        "agent_id": "test_agent",
        "cycle": 282
        # Missing "content" and "protocol_phase"
    }
    
    # Should reject with specific error
    with pytest.raises(ValueError) as exc_info:
        room.submit_tile(malformed_tile)
    
    assert "Missing required fields" in str(exc_info.value)
    
    # Room state should remain unchanged
    assert len(room.get_tiles()) == 0
```

### Test 4: Room State Persistence Simulation
```python
def test_state_persistence_across_restart(self):
    """Test that room can serialize/deserialize state correctly."""
    room = DeadbandRoom(capacity=3)
    
    # Add some tiles
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": 282 + i,
            "content": f"Test content {i}",
            "protocol_phase": "P0" if i == 0 else "P1"
        }
        room.submit_tile(tile)
    
    # Simulate persistence
    serialized = room.serialize_state()
    
    # Create new room with serialized state
    new_room = DeadbandRoom(capacity=3)
    new_room.deserialize_state(serialized)
    
    # Verify state matches
    original_tiles = room.get_tiles()
    restored_tiles = new_room.get_tiles()
    
    assert len(original_tiles) == len(restored_tiles)
    for orig, rest in zip(original_tiles, restored_tiles):
        assert orig["agent_id"] == rest["agent_id"]
        assert orig["content"] == rest["content"]
        assert orig["protocol_phase"] == rest["protocol_phase"]
```

### Test 5: Protocol Violation Recovery
```python
def test_p2_violation_with_p0_recovery(self):
    """Test agent skipping to P2 gets redirected to P0."""
    room = DeadbandRoom(capacity=5)
    
    # Agent tries to submit P2 tile without P0/P1 history
    tile = {
        "agent_id": "eager_agent",
        "cycle": 282,
        "content": "Optimizing without mapping!",
        "protocol_phase": "P2"
    }
    
    result = room.submit_tile(tile)
    
    # Room should either:
    # 1. Reject with guidance (preferred)
    # 2. Accept but flag violation
    # 3. Auto-correct to P0
    
    if result["status"] == "rejected":
        assert "P0" in result.get("message", "").lower()
    elif result["status"] == "accepted":
        # Check if room added corrective metadata
        stored = room.get_tiles()[-1]
        assert "protocol_correction" in stored.get("metadata", {})
        assert stored["metadata"]["protocol_correction"] == "P0"
```
