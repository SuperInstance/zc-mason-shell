### Test 4: Concurrent Tile Processing Race Condition
```python
def test_concurrent_tile_processing():
    """Test handling of rapid, concurrent tile submissions."""
    import threading
    import time
    
    room = DeadbandRoom()
    processed_tiles = []
    errors = []
    
    def submit_tile(tile_id):
        try:
            tile = {
                'agent_id': f'agent_{tile_id}',
                'timestamp': time.time(),
                'content': f'Test content {tile_id}'
            }
            result = room.process_tile(tile)
            processed_tiles.append((tile_id, result))
        except Exception as e:
            errors.append((tile_id, str(e)))
    
    # Submit 10 tiles concurrently
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_tile, args=(i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all tiles were processed without corruption
    assert len(errors) == 0
    assert len(processed_tiles) == 10
    
    # Check tile storage integrity
    assert len(room.tiles) == 10
    tile_ids = [t['agent_id'] for t in room.tiles]
    assert all(f'agent_{i}' in tile_ids for i in range(10))
```
