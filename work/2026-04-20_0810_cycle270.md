```python
def test_mixed_valid_invalid_p0_constraints(self):
    """Test agent behavior when P0 contains both valid and invalid constraints."""
    room = DeadbandRoom(
        p0_constraints=["avoid_rock", "invalid_constraint_123"],
        p1_constraints=["find_channel"],
        p2_constraints=["optimize_path"]
    )
    agent = TestAgent()
    
    # Agent should process valid constraints and log invalid ones
    room.add_agent(agent)
    room.process_agent_step(agent.id)
    
    # Should still progress to P1 if at least one valid constraint processed
    self.assertEqual(agent.current_phase, "P1")
    self.assertIn("invalid_constraint_123", room.invalid_constraints_log)
```

### Test 3: Maximum Agent Capacity Stress Test
```python
def test_max_agent_capacity(self):
    """Test room behavior at maximum agent capacity."""
    room = DeadbandRoom(
        p0_constraints=["avoid_rock"],
        p1_constraints=["find_channel"],
        p2_constraints=["optimize_path"]
    )
    
    # Add agents up to capacity (default: 10)
    agents = []
    for i in range(10):
        agent = TestAgent(agent_id=f"agent_{i}")
        room.add_agent(agent)
        agents.append(agent)
    
    # Attempt to add one more agent beyond capacity
    extra_agent = TestAgent(agent_id="extra_agent")
    with self.assertRaises(RoomCapacityError):
        room.add_agent(extra_agent)
    
    # Verify all existing agents can still process
    for agent in agents:
        room.process_agent_step(agent.id)
        self.assertEqual(agent.current_phase, "P1")
```

### Test 4: Concurrent Agent Phase Transitions
```python
def test_concurrent_phase_transitions(self):
    """Test multiple agents transitioning phases simultaneously."""
    room = DeadbandRoom(
        p0_constraints=["avoid_rock"],
        p1_constraints=["find_channel"],
        p2_constraints=["optimize_path"]
    )
    
    agents = [TestAgent(agent_id=f"concurrent_{i}") for i in range(5)]
    for agent in agents:
        room.add_agent(agent)
    
    # Process all agents in "parallel" (simulated)
    for agent in agents:
        room.process_agent_step(agent.id)
    
    # All should be in P1
    for agent in agents:
        self.assertEqual(agent.current_phase, "P1")
    
    # Process all to P2
    for agent in agents:
        room.process_agent_step(agent.id)
    
    # All should be in P2
    for agent in agents:
        self.assertEqual(agent.current_phase, "P2")
```

### Test 5: Recovery From Corrupted Agent State
```python
def test_corrupted_agent_state_recovery(self):
    """Test room recovery when agent state becomes corrupted."""
    room = DeadbandRoom(
        p0_constraints=["avoid_rock"],
        p1_constraints=["find_channel"],
        p2_constraints=["optimize_path"]
    )
    
    agent = TestAgent()
    room.add_agent(agent)
    
    # Simulate corrupted state (e.g., invalid phase value)
    agent.current_phase = "CORRUPTED_PHASE"
    
    # Room should detect and reset agent to P0
    room.validate_agent_state(agent.id)
    
    self.assertEqual(agent.current_phase, "P0")
    self.assertIn(agent.id, room.recovery_log)
```
