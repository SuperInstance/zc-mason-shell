```python
def test_concurrent_agent_submissions():
    """Test that room handles multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(room_id="concurrent_test", max_agents=5)
    
    # Simulate 3 agents submitting P0 tiles at same time
    agent_tiles = [
        {"agent_id": "agent1", "priority": 0, "content": "P0: Avoid null pointers"},
        {"agent_id": "agent2", "priority": 0, "content": "P0: Don't divide by zero"},
        {"agent_id": "agent3", "priority": 0, "content": "P0: Validate input ranges"}
    ]
    
    results = []
    for tile in agent_tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All P0 tiles should be accepted
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent P0 submissions should all be accepted"
    
    # Verify tiles are stored with correct agent context
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == 3, "Should store all concurrent submissions"
    assert len(set(t["agent_id"] for t in stored_tiles)) == 3, \
        "Should preserve distinct agent identities"
```

### Test 2: Protocol Regression Prevention
```python
def test_protocol_regression_rejection():
    """Test that room rejects attempts to regress to earlier priorities."""
    room = DeadbandRoom(room_id="regression_test")
    agent_id = "test_agent"
    
    # Valid P0 → P1 progression
    p0_tile = {"agent_id": agent_id, "priority": 0, "content": "P0: Map rocks"}
    p1_tile = {"agent_id": agent_id, "priority": 1, "content": "P1: Find channels", "channels": ["safe_path"]}
    
    assert room.submit_tile(p0_tile)["status"] == "accepted"
    assert room.submit_tile(p1_tile)["status"] == "accepted"
    
    # Attempt to regress to P0 (should be rejected)
    regression_tile = {"agent_id": agent_id, "priority": 0, "content": "Trying to go back to P0"}
    result = room.submit_tile(regression_tile)
    
    assert result["status"] == "rejected", \
        "Should reject protocol regression attempts"
    assert "cannot regress" in result.get("message", "").lower(), \
        "Rejection message should indicate regression violation"
```

```python
def test_invalid_priority_values():
    """Test handling of non-standard priority values."""
    room = DeadbandRoom(room_id="priority_edge_test")
    
    test_cases = [
        {"priority": -1, "content": "Negative priority"},
        {"priority": 3, "content": "Priority beyond P2"},
        {"priority": "zero", "content": "String priority"},  # Non-integer
        {"priority": None, "content": "Null priority"},
        {"priority": 1.5, "content": "Float priority"},
    ]
    
    for i, test_case in enumerate(test_cases):
        tile = {"agent_id": f"agent_{i}", **test_case}
        result = room.submit_tile(tile)
        
        # All invalid priorities should be rejected
        assert result["status"] == "rejected", \
            f"Should reject invalid priority: {test_case['priority']}"
        
        # Room state should remain unchanged
        assert len(room.get_tiles()) == 0, \
            "Invalid tiles should not affect room state"
```

```python
def test_resource_limits():
    """Test room behavior at resource boundaries."""
    # Create room with artificially low limits for testing
    room = DeadbandRoom(
        room_id="resource_test",
        max_tiles=5,  # Small limit for testing
        max_agents=2
    )
    
    # Fill room to capacity with P0 tiles
    for i in range(5):
        result = room.submit_tile({
            "agent_id": f"agent_{i % 2}",  # Only 2 agents allowed
            "priority": 0,
            "content": f"P0 tile {i}"
        })
        assert result["status"] == "accepted", \
            f"Should accept tile {i} within limits"
    
    # Next tile should be rejected (max_tiles exceeded)
    overflow_result = room.submit_tile({
        "agent_id": "agent_0",
        "priority": 0,
        "content": "P0 overflow tile"
    })
    
    assert overflow_result["status"] == "rejected", \
        "Should reject tiles beyond max_tiles limit"
    assert "capacity" in overflow_result.get("message", "").lower(), \
        "Rejection should mention capacity limit"
    
    # Test agent limit
    third_agent_result = room.submit_tile({
        "agent_id": "agent_2",  # Third unique agent
        "priority": 0,
        "content": "P0 from new agent"
    })
    
    # Note: This depends on implementation - some rooms might reject,
    # others might accept but warn. Current behavior appears to accept.
    # We'll test both possibilities:
    if third_agent_result["status"] == "rejected":
        assert "agent limit" in third_agent_result.get("message", "").lower()
```
