```python
    self.assertIn("JSON", result["error"].lower())
```

### Test 2: Concurrent Tile Addition Race Condition
```python
def test_concurrent_tile_addition(self):
    """Simulate concurrent tile addition from multiple agents."""
    room = DeadbandRoom(room_id="edge-test-2", protocol_version="1.0")
    
    tiles = []
    for i in range(10):
        tile = {
            "tile_id": f"concurrent-{i:03d}",
            "agent_id": f"agent-{i % 3}",
            "timestamp": f"2026-04-20T13:2{i:02d}:00Z",
            "data": json.dumps({"cycle": i, "value": i * 10}),
            "metadata": {"test": "concurrent"}
        }
        tiles.append(tile)
    
    # Simulate near-simultaneous addition
    results = []
    for tile in tiles:
        results.append(room.add_tile(tile))
    
    # All should succeed
    success_count = sum(1 for r in results if r["success"])
    self.assertEqual(success_count, 10)
    
    # Verify all tiles are stored
    self.assertEqual(len(room.get_tiles()), 10)
    
    # Verify no duplicate tile_ids were accepted
    tile_ids = [t["tile_id"] for t in room.get_tiles()]
    self.assertEqual(len(tile_ids), len(set(tile_ids)))
```

### Test 3: Protocol Violation with Mixed Priority Levels

```python
def test_protocol_mixed_priority_violation(self):
    """Test tiles that claim multiple priority levels simultaneously."""
    room = DeadbandRoom(room_id="edge-test-3", protocol_version="1.0")
    
    # Tile claiming to be both P0 and P2 (impossible per protocol)
    mixed_priority_tile = {
        "tile_id": "mixed-prio-001",
        "agent_id": "confused-agent",
        "timestamp": "2026-04-20T13:25:00Z",
        "data": json.dumps({
            "p0_violation": "something dangerous",
            "p2_optimization": "efficiency gain",
            "priority": ["p0", "p2"]  # Mixed priorities
        }),
        "metadata": {"priority_flags": ["p0", "p2"]}
    }
    
    result = room.add_tile(mixed_priority_tile)
    # According to Deadband Protocol, mixed priorities should be rejected
    # or default to P0 (safety first)
    self.assertFalse(result["success"])
    self.assertIn("priority", result["error"].lower())
```

### Test 5: Training Cycle with Corrupted Model State
```python
def test_train_cycle_corrupted_state(self):
    """Test training cycle recovery when model state is corrupted."""
    room = DeadbandRoom(room_id="edge-test-5", protocol_version="1.0")
    
    # First add some valid tiles
    for i in range(5):
        tile = {
            "tile_id": f"valid-{i:03d}",
            "agent_id": "test-agent",
            "timestamp": f"2026-04-20T13:3{i:02d}:00Z",
            "data": json.dumps({"cycle": i, "observation": f"test_{i}"}),
            "metadata": {"valid": True}
        }
        room.add_tile(tile)
    
    # Manually corrupt the model state (simulating disk/memory corruption)
    # This tests the room's resilience
    room.model_state = b"corrupted\x00\xFF\xFEmodel data"
    
    # Attempt training cycle - should handle corruption gracefully
    try:
        result = room.train_cycle()
        # Either succeeds with recovery or fails cleanly
        self.assertIn("status", result)
        if not result.get("success", False):
            self.assertIn("corrupt", result.get("error", "").lower() or 
                         "recover" in result.get("error", "").lower())
    except Exception as e:
        # Exception should be specific and informative
        self.assertIn("corrupt", str(e).lower() or "state" in str(e).lower())
```
