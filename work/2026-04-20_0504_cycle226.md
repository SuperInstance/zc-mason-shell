### Test 1: P0 Negative Space Persistence Across Resets
**Scenario:** A user submits a P0 tile (negative space), then resets the room. The P0 knowledge should persist as a "ghost tile" (per fleet doctrine) and influence subsequent P1 channel finding.
```python
def test_p0_ghost_tile_persistence(self):
    """P0 negative space should persist across room resets as ghost knowledge."""
    room = DeadbandRoom()
    # Submit a P0 tile
    tile = {"phase": "P0", "content": "Avoid using absolute claims."}
    result = room.submit_tile(tile)
    self.assertTrue(result["accepted"])
    # Reset room state (simulating a new session)
    room.reset()
    # Verify P0 ghost tile is retained in room memory
    self.assertIn("ghost_tiles", room.room_state)
    self.assertEqual(len(room.room_state["ghost_tiles"]), 1)
    self.assertEqual(room.room_state["ghost_tiles"][0]["content"], tile["content"])
    # New P1 submission should be influenced by ghost tile
    p1_tile = {"phase": "P1", "content": "Find safe channels in documentation."}
    result = room.submit_tile(p1_tile)
    self.assertTrue(result["accepted"])
    # Room should log that P1 considered P0 ghost
    self.assertIn("considered_ghost_tiles", room.room_state["last_evaluation"])
```

### Test 3: Concurrent Tile Submission Race Condition
**Scenario:** Two simulated users submit tiles simultaneously while the room is evaluating. This tests thread safety and state integrity.
```python
def test_concurrent_tile_submission(self):
    """Room should handle concurrent tile submissions without state corruption."""
    import threading
    room = DeadbandRoom()
    results = []
    def submit_tile(tile_content):
        tile = {"phase": "P0", "content": tile_content}
        result = room.submit_tile(tile)
        results.append(result)
    # Simulate concurrent submissions
    threads = [
        threading.Thread(target=submit_tile, args=("Avoid rock A",)),
        threading.Thread(target=submit_tile, args=("Avoid rock B",))
    ]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    # Exactly one tile should be accepted (first-come)
    accepted_count = sum(1 for r in results if r.get("accepted"))
    self.assertEqual(accepted_count, 1)
    # Room state should be consistent
    self.assertIsInstance(room.room_state["tiles"], list)
    self.assertEqual(len(room.room_state["tiles"]), 1)
```

### Test 5: Phase Regression Detection
**Scenario:** A user attempts to submit a tile for a previous phase after having advanced. The protocol is strictly linear; regression should be blocked.
```python
def test_phase_regression_blocked(self):
    """Room should block tile submission for a phase earlier than current."""
    room = DeadbandRoom()
    # Submit P0, advance to P1
    room.submit_tile({"phase": "P0", "content": "P0 tile"})
    room.advance_phase()
    self.assertEqual(room.room_state["current_phase"], "P1")
    # Attempt to submit another P0 tile (regression)
    tile = {"phase": "P0", "content": "Another P0 tile"}
    result = room.submit_tile(tile)
    self.assertFalse(result["accepted"])
    self.assertEqual(result["error"], "Phase regression not allowed: current P1, submitted P0")
    # P1 tile should still be accepted
    p1_tile = {"phase": "P1", "content": "P1 tile"}
    result = room.submit_tile(p1_tile)
    self.assertTrue(result["accepted"])
```
