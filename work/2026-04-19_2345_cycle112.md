### Test 4: Protocol Reset After Completion
```python
def test_protocol_reset():
    """Test agent can reset from P2 back to P0 with new constraints."""
    room = DeadbandRoom(room_id="test_reset")
    agent = MockAgent()
    
    # Complete full protocol
    room.execute_phase(agent, phase="P0", constraints=["rock"])
    room.execute_phase(agent, phase="P1", safe_channels=[{"x": (0, 10), "y": (0, 10)}])
    room.execute_phase(agent, phase="P2", optimization_target="speed")
    
    # Reset to P0 with new constraints
    result = room.execute_phase(agent, phase="P0", constraints=["rock", "pit", "wall"])
    assert result["status"] == "complete"
    assert result["constraints_mapped"] == 3
    assert room.get_agent_phase(agent) == "P0"
```

### Test 5: Tile Validation Failure Recovery

```python
def test_tile_validation_failure():
    """Test room handles server rejection of invalid tile."""
    room = DeadbandRoom(room_id="test_tile_fail")
    agent = MockAgent()
    
    # Mock server rejection
    room.tile_server = MockTileServer(reject_next_tile=True)
    
    result = room.submit_tile(agent, tile_data={"test": "invalid"})
    assert result["status"] == "rejected"
    assert "validation_error" in result
    assert room.get_agent_status(agent) == "retry_pending"
    
    # Agent should be able to retry with corrected tile
    retry_result = room.submit_tile(agent, tile_data={"phase": "P0", "constraints": ["rock"]})
    assert retry_result["status"] == "accepted"
```
