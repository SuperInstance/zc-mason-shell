## 5 Edge-Case Tests for DeadbandRoom

### 1. Test: `test_invalid_phase_transition_skips_p1`
**Scenario:** Agent attempts to move from P0 directly to P2, violating strict priority.
**Expected:** Room rejects action, returns error, agent remains in P0.
**Purpose:** Enforce Deadband Protocol P0 rule: "Never skip to P2."

```python
def test_invalid_phase_transition_skips_p1(self):
    """Agent cannot skip P1 phase."""
    room = DeadbandRoom(config=self.valid_config)
    agent_id = room.add_agent()
    # Force P0 completion
    room.agents[agent_id]["phase"] = "P0"
    room.agents[agent_id]["p0_complete"] = True
    # Attempt illegal transition to P2
    with self.assertRaises(ValueError) as context:
        room.process_agent_action(agent_id, {"phase": "P2"})
    self.assertIn("must complete P1 first", str(context.exception))
    self.assertEqual(room.agents[agent_id]["phase"], "P0")
```

### 2. Test: `test_malformed_tile_submission`
**Scenario:** Agent submits tile with missing required fields or invalid JSON.
**Expected:** Tile rejected, not added to training set, error logged.
**Purpose:** Ensure robustness against corrupted input.

```python
def test_malformed_tile_submission(self):
    """Room rejects tiles with missing fields."""
    room = DeadbandRoom(config=self.valid_config)
    agent_id = room.add_agent()
    # Tile missing 'content' field
    invalid_tile = {"agent_id": agent_id, "timestamp": "2026-04-19T15:30:00Z"}
    result = room.submit_tile(invalid_tile)
    self.assertFalse(result["accepted"])
    self.assertIn("missing required field", result["error"])
    self.assertEqual(len(room.training_tiles), 0)
```

### 3. Test: `test_max_agents_enforced`
**Scenario:** Attempt to add agents beyond room capacity.
**Expected:** Addition fails after limit, returns appropriate error.
**Purpose:** Prevent resource exhaustion.

```python
def test_max_agents_enforced(self):
    """Room enforces maximum agent limit."""
    room = DeadbandRoom(config={"max_agents": 2})
    agent1 = room.add_agent()
    agent2 = room.add_agent()
    # Third agent should fail
    with self.assertRaises(RuntimeError) as context:
        room.add_agent()
    self.assertIn("max agents reached", str(context.exception).lower())
    self.assertEqual(len(room.agents), 2)
```

### 4. Test: `test_concurrent_state_corruption`
**Scenario:** Simulate two agents attempting to modify same channel simultaneously.
**Expected:** Room uses locking or conflict resolution; state remains consistent.
**Purpose:** Detect race conditions in multi-agent scenarios.

```python
def test_concurrent_state_corruption(self):
    """Room handles concurrent channel claims without corruption."""
    room = DeadbandRoom(config=self.valid_config)
    agent1 = room.add_agent()
    agent2 = room.add_agent()
    # Both agents attempt to claim same channel in P1
    def claim_channel(aid, channel):
        room.agents[aid]["phase"] = "P1"
        return room.process_agent_action(aid, {"claim_channel": channel})
    # Simulate near-concurrent claims (threading would be used in real test)
    result1 = claim_channel(agent1, "channel_a")
    result2 = claim_channel(agent2, "channel_a")
    # Only one should succeed
    successes = [r.get("success", False) for r in (result1, result2)]
    self.assertEqual(sum(successes), 1)
    # Channel owner should be consistent
    channel_owner = room.channels.get("channel_a")
    self.assertIn(channel_owner, (agent1, agent2))
```

### 5. Test: `test_recovery_from_corrupted_state`
**Scenario:** Room state becomes corrupted (e.g., agent in P2 without P1 completion); recovery method invoked.
**Expected:** Room can reset/repair state, or raise actionable error.
**Purpose:** Ensure fault tolerance and debuggability.

```python
def test_recovery_from_corrupted_state(self):
    """Room can detect and recover from invalid internal state."""
    room = DeadbandRoom(config=self.valid_config)
    agent_id = room.add_agent()
    # Manually corrupt state: agent in P2 without P1 complete
    room.agents[agent_id]["phase"] = "P2"
    room.agents[agent_id]["p1_complete"] = False
    # Recovery method should reset to last valid phase
    room.recover_state()
    self.assertEqual(room.agents[agent_id]["phase"], "P0")
    self.assertFalse(room.agents[agent_id]["p0_complete"])
```
