### Test 1: Network Partition During Tile Submission
**Scenario:** Agent submits tile, network drops before ACK.
**Edge:** Partial state persistence.
```python
def test_tile_submission_network_partition(self):
    """P0: Tile submission with simulated network failure before ACK."""
    agent = DeadbandRoom.enter_room("test_agent")
    tile = {"phase": "P1", "content": "test", "timestamp": time.time()}
    
    # Simulate network failure after send, before ACK
    with patch.object(agent.connection, 'send_tile', side_effect=ConnectionError("Network partition")):
        with self.assertRaises(ConnectionError):
            agent.submit_tile(tile)
    
    # Verify tile not in accepted set but remains in pending buffer
    self.assertNotIn(tile['timestamp'], agent.accepted_tiles)
    self.assertIn(tile, agent.pending_buffer)
    self.assertEqual(agent.state, "RECONNECTING")
```

def test_malformed_tile_circular_reference(self):
    """P0: Tile with circular reference that breaks JSON serialization."""
    agent = DeadbandRoom.enter_room("test_agent")

    circular_dict = {"key": "value"}
    circular_dict["self"] = circular_dict  # Circular reference

        "phase": "P1",

### Test 3: Concurrent Agent ID Collision
**Scenario:** Two agents attempt to enter room with same ID simultaneously.
**Edge:** Race condition in agent registry.
```python
def test_concurrent_agent_id_collision(self):
    """P1: Simulate concurrent room entry with same agent ID."""
    room = DeadbandRoom()
    
    # Simulate concurrent entry
    def enter_agent(agent_id):
        return room.enter_room(agent_id)
    
    with ThreadPoolExecutor(max_workers=2) as executor:
        futures = [executor.submit(enter_agent, "duplicate_id") for _ in range(2)]
        results = [f.result() for f in futures]
    
    # Only one agent should get the ID, other should get modified ID
    agent_ids = [agent.agent_id for agent in results]
    self.assertEqual(len(set(agent_ids)), 2)  # Both should have unique IDs
    self.assertTrue(any("duplicate_id" in id for id in agent_ids))
    self.assertTrue(any("duplicate_id_" in id for id in agent_ids))
```

### Test 4: Memory Exhaustion from Tile Flood
**Scenario:** Malicious agent submits thousands of tiles rapidly.
**Edge:** Memory protection and rate limiting.
```python
def test_tile_flood_memory_protection(self):
    """P0: Submit tiles until memory protection triggers."""
    agent = DeadbandRoom.enter_room("test_agent")
    initial_memory = agent.get_memory_usage()
    
    # Submit tiles until rejection
    rejected_count = 0
    for i in range(10000):  # Excessive count
        tile = {
            "phase": "P1",
            "content": f"tile_{i}",
            "timestamp": time.time()
        }
        try:
            agent.submit_tile(tile)
        except MemoryError:
            rejected_count += 1
            break
    
    # Verify protection triggered
    self.assertGreater(rejected_count, 0)
    self.assertLess(agent.get_memory_usage(), initial_memory * 2)  # Less than 2x growth
    self.assertEqual(agent.state, "THROTTLED")
```

### Test 5: Protocol Violation Recovery (Skip P1 to P2)
**Scenario:** Agent attempts to optimize (P2) without finding safe channels (P1).
**Edge:** Protocol enforcement and recovery path.
```python
def test_protocol_violation_skip_p1_to_p2(self):
    """P0: Agent tries to submit P2 tile without completing P1 phase."""
    agent = DeadbandRoom.enter_room("test_agent")
    
    # Start with P0 tile (correct)
    p0_tile = {"phase": "P0", "content": "negative space", "timestamp": time.time()}
    agent.submit_tile(p0_tile)
    
    # Violation: Skip P1, try P2 directly
    p2_tile = {"phase": "P2", "content": "optimization", "timestamp": time.time()}
    
    # Should raise protocol violation
    with self.assertRaises(ProtocolViolationError) as context:
        agent.submit_tile(p2_tile)
    
    self.assertIn("P1 not completed", str(context.exception))
    
    # Agent should be reset to P1 phase, not kicked
    self.assertEqual(agent.current_phase, "P1")
    self.assertEqual(agent.state, "READY")
    
    # Now P1 tile should work
    p1_tile = {"phase": "P1", "content": "safe channels", "timestamp": time.time()}
    result = agent.submit_tile(p1_tile)
    self.assertTrue(result["accepted"])
```
