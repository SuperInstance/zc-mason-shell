### Test 1: **Converging Obstacles (Dead-End Channel)**
```python
def test_p1_safe_channels_converging_obstacles():
    """Test P1 when obstacles converge, creating a dead-end that appears safe initially."""
    room = DeadbandRoom(room_id="edge-converge", max_iterations=100)
    # Create V-shaped obstacles that converge
    obstacles = [
        {"x": 0, "y": 0, "radius": 5},
        {"x": 10, "y": 0, "radius": 5},
        {"x": 5, "y": 8, "radius": 3}  # Tip of the V
    ]
    room.set_obstacles(obstacles)
    
    # Agent starts in the V opening
    agent_state = {"x": 5, "y": -5, "velocity": 0.5, "heading": 90}
    
    result = room.run_protocol(agent_state)
    
    # P1 should identify the narrowing channel
    assert "p1_channels" in result
    channels = result["p1_channels"]
    # Should have at least one channel but with warning about convergence
    assert len(channels) > 0
    # The channel width should decrease toward the tip
    if len(channels) > 1:
        assert channels[0]["width"] > channels[-1]["width"]
```

### Test 2: **Dynamic Obstacle Expansion During Protocol**
```python
def test_p0_obstacle_expansion_mid_protocol():
    """Test P0 when obstacles expand during negative space mapping."""
    room = DeadbandRoom(room_id="edge-expand", max_iterations=50)
    
    # Start with small obstacles
    obstacles = [
        {"x": 3, "y": 3, "radius": 2, "expansion_rate": 0.1},
        {"x": 7, "y": 7, "radius": 2, "expansion_rate": 0.2}
    ]
    room.set_obstacles(obstacles)
    
    agent_state = {"x": 0, "y": 0, "velocity": 0.3, "heading": 45}
    
    # Run protocol but simulate obstacle expansion between phases
    result = room.run_protocol(agent_state, dynamic_obstacles=True)
    
    # P0 map should account for expansion
    assert "p0_map" in result
    p0_map = result["p0_map"]
    # Expanded obstacles should have larger effective radius
    for obs in p0_map.get("obstacles", []):
        if "expansion_rate" in obs:
            assert obs["effective_radius"] > obs["radius"]
```

### Test 3: **Agent With Erratic Velocity (Oscillating Heading)**
```python
def test_p2_optimization_oscillating_agent():
    """Test P2 optimization for agent with oscillating heading/velocity."""
    room = DeadbandRoom(room_id="edge-oscillate", max_iterations=200)
    obstacles = [{"x": 5, "y": 5, "radius": 4}]
    room.set_obstacles(obstacles)
    
    # Agent with sine-wave heading oscillation
    def oscillating_agent(t):
        return {
            "x": 0,
            "y": 0,
            "velocity": 0.5 + 0.2 * math.sin(t * 0.1),
            "heading": 45 + 30 * math.sin(t * 0.05)
        }
    
    # Run protocol with time-varying agent state
    results = []
    for t in range(10):
        agent_state = oscillating_agent(t)
        result = room.run_protocol(agent_state)
        results.append(result)
    
    # P2 should still find stable path despite oscillations
    successful_p2 = [r for r in results if r.get("p2_path") is not None]
    assert len(successful_p2) > 5  # Majority should succeed
```

### Test 4: **Boundary Conditions (Agent at World Edge)**

```python
def test_p1_channels_world_boundary():
    """Test P1 channel identification when agent is at world boundary."""
    room = DeadbandRoom(
        room_id="edge-boundary",
        max_iterations=100,
        world_bounds={"x_min": -10, "x_max": 10, "y_min": -10, "y_max": 10}
    )
    
    # Obstacle near boundary
    obstacles = [{"x": 8, "y": 8, "radius": 3}]
    room.set_obstacles(obstacles)
    
    # Agent pinned against top-right boundary
    agent_state = {"x": 9.5, "y": 9.5, "velocity": 0.1, "heading": 0}
    
    result = room.run_protocol(agent_state)
    
    # P1 should identify channels that respect world bounds
    assert "p1_channels" in result
    channels = result["p1_channels"]
    for channel in channels:
        # Channel should not extend beyond world bounds
        assert channel["start"]["x"] >= -10
        assert channel["start"]["y"] >= -10
        assert channel["end"]["x"] <= 10
        assert channel["end"]["y"] <= 10
```

### Test 5: **Protocol Interruption and Resume**
```python
def test_protocol_interruption_resume():
    """Test protocol can be interrupted after P0/P1 and resumed later."""
    room = DeadbandRoom(room_id="edge-interrupt", max_iterations=100)
    obstacles = [
        {"x": 2, "y": 2, "radius": 1},
        {"x": 6, "y": 6, "radius": 2}
    ]
    room.set_obstacles(obstacles)
    
    agent_state = {"x": 0, "y": 0, "velocity": 0.4, "heading": 60}
    
    # Run only P0
    phase_result = room.run_phase("p0", agent_state)
    assert "p0_map" in phase_result
    p0_map = phase_result["p0_map"]
    
    # Save P0 state
    saved_state = {
        "agent": agent_state,
        "p0_map": p0_map,
        "obstacles": obstacles
    }
    
    # Simulate interruption (room reset)
    room2 = DeadbandRoom(room_id="edge-resume", max_iterations=100)
    room2.set_obstacles(saved_state["obstacles"])
    
    # Resume from P1 using saved P0 map
    resume_result = room2.run_protocol(
        saved_state["agent"],
        initial_p0_map=saved_state["p0_map"]
    )
    
    # Should complete P1 and P2 successfully
    assert "p1_channels" in resume_result
    assert "p2_path" in resume_result
    # Should be faster than full protocol run
    assert resume_result["iterations_used"] < 100
```
