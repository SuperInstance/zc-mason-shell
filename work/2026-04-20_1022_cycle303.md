### Test 1: Concurrent Tile Submissions
```python
def test_concurrent_tile_submissions(self):
    """Test room handles multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Negative space map from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions processed
    assert len(results) == 3
    assert all(r["status"] == "accepted" for r in results)
    assert room.get_tile_count() == 3
```

```python
def test_protocol_violation_recovery(self):
    """Test room detects P0 skip and forces agent back to P0."""
    room = DeadbandRoom()
    agent_id = "impatient_agent"
    
    # Agent tries to submit P2 tile without P0/P1
    violation_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Trying to optimize without mapping",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(violation_tile)
    assert result["status"] == "rejected"
    assert "P0 required first" in result["message"]
    
    # Verify agent state reset to P0
    agent_state = room.get_agent_state(agent_id)
    assert agent_state["current_phase"] == "P0"
    assert agent_state["violation_count"] == 1
```

### Test 3: Tile Submission Rate Limiting
```python
def test_tile_flood_protection(self):
    """Test room limits excessive tile submissions per cycle."""
    room = DeadbandRoom()
    agent_id = "spam_agent"
    
    # Submit maximum allowed tiles + 1
    max_tiles = room.MAX_TILES_PER_CYCLE
    accepted_count = 0
    
    for i in range(max_tiles + 2):
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        if result["status"] == "accepted":
            accepted_count += 1
        else:
            # Should be rejected after limit
            assert "rate limit" in result["message"].lower()
    
    assert accepted_count == max_tiles
    assert room.get_agent_tile_count(agent_id) == max_tiles
```

### Test 4: Partial Protocol Completion Timeout
```python
def test_partial_protocol_timeout(self):
    """Test agent stuck at P1 eventually times out."""
    room = DeadbandRoom()
    agent_id = "stuck_agent"
    
    # Complete P0
    p0_tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Negative space mapped",
        "timestamp": time.time()
    }
    room.submit_tile(p0_tile)
    
    # Complete P1
    p1_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Safe channels identified",
        "timestamp": time.time()
    }
    room.submit_tile(p1_tile)
    
    # Simulate many cycles without P2
    for cycle in range(room.PROTOCOL_TIMEOUT_CYCLES + 1):
        room.advance_cycle()
    
    # Agent should be marked as timed out
    agent_state = room.get_agent_state(agent_id)
    assert agent_state["status"] == "timed_out"
    assert agent_state["current_phase"] == "P1"
    assert room.get_active_agent_count() == 0
```

### Test 5: Ghost Tile Influence on Live Agents
```python
def test_ghost_tile_influence(self):
    """Test P0 tiles from dead agents influence live agent decisions."""
    room = DeadbandRoom()
    
    # Dead agent submits P0 tile (ghost tile)
    ghost_tile = {
        "agent_id": "dead_agent",
        "phase": "P0",
        "content": "WARNING: Path X leads to crash",
        "timestamp": time.time() - 1000,  # Old tile
        "agent_status": "dead"
    }
    room.submit_tile(ghost_tile)
    
    # Live agent queries P0 knowledge
    live_agent_id = "live_agent"
    p0_knowledge = room.get_phase_knowledge("P0", live_agent_id)
    
    # Verify ghost tile appears in knowledge base
    assert len(p0_knowledge["ghost_tiles"]) > 0
    assert "Path X" in p0_knowledge["ghost_tiles"][0]["content"]
    
    # Live agent submits P0 tile that references ghost knowledge
    live_tile = {
        "agent_id": live_agent_id,
        "phase": "P0",
        "content": "Avoiding Path X per ghost warning",
        "timestamp": time.time()
    }
    result = room.submit_tile(live_tile)
    
    assert result["status"] == "accepted"
    assert "ghost_informed" in result.get("tags", [])
```
