    room = DeadbandRoom(room_id="race_test", max_tiles=1000)

    tiles = [
        {"agent_id": f"agent_{i}", "phase": "P0", "content": f"test_{i}"}
        for i in range(10)
    ]
    
    submit_coroutines = [room.submit_tile(tile) for tile in tiles]
    results = await asyncio.gather(*submit_coroutines, return_exceptions=True)
    
    # All submissions should succeed without exceptions
    assert all(r is True for r in results), "Some concurrent submissions failed"
    # Room tile count should match number of submissions

def test_phase_transition_with_incomplete_p0():
    """
    P0/P1 Boundary: Agent attempting P1 without completing P0 mapping should be rejected.
    """
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom(room_id="phase_boundary_test")
    
    # Agent submits P1 tile without any P0 history
    result = room.submit_tile({
        "agent_id": "eager_agent",
        "phase": "P1",
        "content": "Trying to optimize without mapping negatives"
    })
    
    # Should be rejected per Deadband Protocol strict priority
    assert result is False
    assert "P0 not completed" in room.get_last_validation_error()
    
    # After submitting P0 tile, P1 should be accepted
    room.submit_tile({
        "agent_id": "eager_agent",
        "phase": "P0",
        "content": "Mapping negative space"
    })
    result = room.submit_tile({
        "agent_id": "eager_agent",
        "phase": "P1",
        "content": "Now finding safe channels"
    })
    assert result is True
