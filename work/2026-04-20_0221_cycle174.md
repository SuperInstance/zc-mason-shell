### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission():
    """Test that room handles multiple simultaneous tile submissions correctly."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3", "agent_4", "agent_5"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 174,
            "content": f"Test tile from {agent_id}",
            "priority": "P1",
            "channel": "safe_channel_1"
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results)
    
    # Verify tiles are stored separately
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == len(agent_ids)
```

```python
def test_malformed_tile_data():
    """Test room's handling of invalid tile structures."""
    room = DeadbandRoom()
    
    test_cases = [
        # Missing required fields
        {"agent_id": "test_agent"},
        {"content": "test content"},
        # Invalid priority
        {"agent_id": "test_agent", "cycle": 174, "content": "test", "priority": "P3"},
        # Invalid JSON structure
        "not a dict",
        # Empty tile
        {},
        # Cycle as string instead of int
        {"agent_id": "test_agent", "cycle": "174", "content": "test", "priority": "P1"}
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(tile)
        assert result.get("status") == "rejected", f"Test case {i} should have been rejected"
        assert "error" in result, f"Test case {i} should include error message"
```

```python
def test_protocol_state_persistence():
    """Test that protocol state is maintained across room operations."""
    room = DeadbandRoom()
    
    # Initial state
    initial_state = room.get_protocol_state()
    assert initial_state["current_phase"] == "P0"
    
    # Perform P0 operations
    negative_space = {"rock_1": "avoid", "rock_2": "danger"}
    room.map_negative_space(negative_space)
    
    # Verify state update
    state_after_p0 = room.get_protocol_state()
    assert state_after_p0["current_phase"] == "P1"
    assert "negative_space" in state_after_p0
    
    # Reset room (simulating new cycle)
    room.reset_for_cycle(175)
    
    # Verify protocol resets to P0 but retains learned constraints
    reset_state = room.get_protocol_state()
    assert reset_state["current_phase"] == "P0"
    assert "learned_constraints" in reset_state
    assert len(reset_state["learned_constraints"]) > 0
```

```python
def test_channel_boundary_conditions():
    """Test safe channel discovery at numerical extremes."""
    room = DeadbandRoom()
    
    # Test with boundary values
    boundary_cases = [
        {"min": 0, "max": 0},           # Zero-width channel
        {"min": -1000, "max": 1000},    # Large range
        {"min": 1e-6, "max": 1e-6},     # Very small positive
        {"min": -1e-6, "max": -1e-6},   # Very small negative
        {"min": float("inf"), "max": float("inf")},  # Infinite
    ]
    
    for bounds in boundary_cases:
        channels = room.find_safe_channels(
            constraints=[], 
            search_space=bounds
        )
        
        # Room should handle boundaries gracefully
        assert isinstance(channels, list)
        
        # For valid channels, verify properties
        if channels:
            for channel in channels:
                assert "min" in channel
                assert "max" in channel
                assert channel["min"] <= channel["max"]
```

### Test 5: Negative Space Overlap
```python
def test_negative_space_overlap():
    """Test handling of conflicting P0 constraints."""
    room = DeadbandRoom()
    
    # Conflicting constraints: same space marked by different agents
    constraints_agent1 = [
        {"id": "rock_1", "bounds": {"x": [0, 10], "y": [0, 10]}, "danger": "high"}
    ]
    
    constraints_agent2 = [
        {"id": "safe_zone", "bounds": {"x": [5, 15], "y": [5, 15]}, "danger": "low"}
    ]
    
    # Map both constraints
    room.map_negative_space(constraints_agent1)
    room.map_negative_space(constraints_agent2)
    
    # Get consolidated negative space
    negative_space = room.get_negative_space()
    
    # Verify overlap is detected and handled
    assert "overlaps" in negative_space
    assert len(negative_space["overlaps"]) > 0
    
    # For overlapping regions, higher danger should dominate
    for overlap in negative_space["overlaps"]:
        assert overlap["resolved_danger"] == "high"
```
