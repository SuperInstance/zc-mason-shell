### Test 1: Empty Action Space (No Safe Channels)
**Scenario:** When P1 analysis returns an empty set of safe channels, the room should not proceed to P2 optimization.
```python
def test_empty_safe_channels_prevents_p2():
    """P2 optimization should not run when P1 returns empty safe channels."""
    room = DeadbandRoom(config={"max_iterations": 10})
    # Mock P1 to return empty list
    with patch.object(room, '_analyze_p1_channels', return_value=[]):
        result = room.process_action({"type": "navigate", "target": "goal"})
        # Should return P1 failure, not attempt P2
        assert result["phase"] == "P1"
        assert "no_safe_channels" in result["status"]
        assert "P2" not in result
```

### Test 2: Concurrent P0 Violation Detection
**Scenario:** Multiple agents submitting actions that each violate P0 in different ways.
```python
def test_concurrent_p0_violations():
    """Room should handle multiple simultaneous P0 violations correctly."""
    room = DeadbandRoom(config={"max_agents": 3})
    
    actions = [
        {"agent": "A", "action": {"type": "move", "direction": "into_wall"}},
        {"agent": "B", "action": {"type": "use", "item": "forbidden_tool"}},
        {"agent": "C", "action": {"type": "query", "target": "restricted_data"}}
    ]
    
    results = room.process_batch(actions)
    
    # All should be caught as P0 violations
    assert all(r["phase"] == "P0" for r in results)
    assert len(set(r["violation_type"] for r in results)) == 3  # Different violation types
```

### Test 3: P1 Channel Boundary Conditions
**Scenario:** Safe channels with minimal/maximal boundaries that might cause floating-point or integer overflow issues.
```python
def test_p1_channel_boundary_conditions():
    """P1 should handle channels at numerical limits correctly."""
    room = DeadbandRoom(config={"precision": 1e-6})
    
    # Test with channels at extreme values
    extreme_channels = [
        {"min": -float("inf"), "max": 0},
        {"min": 0, "max": float("inf")},
        {"min": -1e308, "max": 1e308},
        {"min": 0, "max": 0},  # Zero-width channel
    ]
    
    with patch.object(room, '_detect_p0_violations', return_value=[]):
        with patch.object(room, '_analyze_p1_channels', return_value=extreme_channels):
            result = room.process_action({
                "type": "select_channel",
                "position": 0.0
            })
            
            # Should handle extremes without crashing
            assert "selected_channel" in result
            assert result["phase"] == "P1"
```

### Test 5: State Persistence Across Protocol Phases
**Scenario:** Ensure room state is preserved correctly when an action transitions through P0→P1→P2 phases.
```python
def test_state_persistence_across_phases():
    """Room state should persist correctly through P0-P1-P2 progression."""
    room = DeadbandRoom(config={"stateful": True})
    initial_state = room.get_state()
    
    # Sequence: Pass P0, pass P1, reach P2
    action_sequence = [
        {"phase_check": "P0", "should_violate": False},
        {"phase_check": "P1", "channels": [{"min": 0, "max": 10}]},
        {"phase_check": "P2", "optimization_data": {"step": 1}}
    ]
    
    for i, action in enumerate(action_sequence):
        with patch.object(room, '_detect_p0_violations', 
                         return_value=[] if not action.get("should_violate") else ["violation"]):
            with patch.object(room, '_analyze_p1_channels', 
                             return_value=action.get("channels", [])):
                result = room.process_action({
                    "type": "progressive_action",
                    "step": i
                })
                
                # State should be updated but not corrupted
                current_state = room.get_state()
                assert current_state["last_phase"] == action["phase_check"]
                assert current_state["action_count"] == i + 1
                
                # Critical state from previous phases should persist
                if i > 0:
                    assert "P0_passed" in current_state
```
