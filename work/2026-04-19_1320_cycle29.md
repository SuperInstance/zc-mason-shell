### Test 5: Concurrent Channel Modification
```rust
#[test]
fn test_channel_lock_mechanism() {
    let room = DeadbandRoom::new();
    let channel_id = "test_channel".to_string();
    
    // Simulate concurrent access
    let room_arc = Arc::new(RwLock::new(room));
    let mut handles = vec![];
    
    for i in 0..5 {
        let room_clone = room_arc.clone();
        let channel_clone = channel_id.clone();
        handles.push(thread::spawn(move || {
            let room = room_clone.write().unwrap();
            room.modify_channel(&channel_clone, i as f64)
        }));
    }
    
    // Collect results
    let results: Vec<_> = handles.into_iter().map(|h| h.join().unwrap()).collect();
    
    // Only one should succeed, others should get contention errors
    let success_count = results.iter().filter(|r| r.is_ok()).count();
    assert_eq!(success_count, 1);
}
```
