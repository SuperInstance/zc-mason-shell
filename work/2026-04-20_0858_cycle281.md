### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race():
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(capacity=5)
    agents = [MockAgent(f"agent_{i}") for i in range(3)]
    
    # All agents join simultaneously
    for agent in agents:
        room.join(agent)
    
    # Concurrent tile submissions
    tiles = []
    for agent in agents:
        tile = {
            "agent_id": agent.id,
            "phase": "P0",
            "content": f"Negative space map from {agent.id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles in rapid succession
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all tiles processed without corruption
    assert len(results) == 3
    assert all(r["status"] == "accepted" for r in results)
    assert room.tile_count == 3
    
    # Verify tile ordering preserves protocol sequence
    submitted_tiles = room.get_recent_tiles(3)
    assert len(submitted_tiles) == 3
    # Each tile should have unique agent_id
    agent_ids = {t["agent_id"] for t in submitted_tiles}
    assert len(agent_ids) == 3
```

```python
def test_network_partition_mid_protocol():
    """Test agent disconnection during P1 phase, then reconnection."""
    room = DeadbandRoom(capacity=2)
    agent = MockAgent("test_agent")
    room.join(agent)
    
    # Start with P0 tile
    p0_tile = {
        "agent_id": agent.id,
        "phase": "P0",
        "content": "Negative space: avoid absolute claims",
        "timestamp": time.time()
    }
    room.submit_tile(p0_tile)
    
    # Simulate network partition - agent disappears
    room.leave(agent)
    assert agent.id not in room.active_agents
    
    # Agent reconnects with same ID
    reconnected_agent = MockAgent("test_agent")  # Same ID
    room.join(reconnected_agent)
    
    # Submit P1 tile after reconnection
    p1_tile = {
        "agent_id": reconnected_agent.id,
        "phase": "P1",
        "content": "Safe channel: write factual tests",
        "timestamp": time.time()
    }
    result = room.submit_tile(p1_tile)
    
    # Should accept P1 despite partition
    assert result["status"] == "accepted"
    assert "reconnected_agent" in result.get("notes", "")
    
    # Verify protocol state maintained
    agent_state = room.get_agent_state(reconnected_agent.id)
    assert agent_state["current_phase"] == "P1"
    assert agent_state["tile_count"] == 2
```

### Test 3: Memory Spike During Large Tile Processing
```python
def test_large_tile_memory_pressure():
    """Test processing of excessively large tile payloads."""
    room = DeadbandRoom(capacity=1)
    agent = MockAgent("memory_test_agent")
    room.join(agent)
    
    # Create tile with massive nested structure
    large_payload = {"data": []}
    current = large_payload["data"]
    for i in range(1000):  # Deep nesting
        new_layer = {"nested": [], "index": i}
        current.append(new_layer)
        current = new_layer["nested"]
    
    large_tile = {
        "agent_id": agent.id,
        "phase": "P0",
        "content": large_payload,
        "timestamp": time.time(),
        "metadata": {"size_estimate": "large"}
    }
    
    # Monitor memory during processing
    import psutil
    import os
    process = psutil.Process(os.getpid())
    memory_before = process.memory_info().rss
    
    result = room.submit_tile(large_tile)
    
    memory_after = process.memory_info().rss
    memory_increase = memory_after - memory_before
    
    # Tile should be rejected or processed with memory limits
    if result["status"] == "rejected":
        assert "memory" in result.get("reason", "").lower() or \
               "size" in result.get("reason", "").lower()
    else:
        # If accepted, memory increase should be bounded
        assert memory_increase < 100 * 1024 * 1024  # < 100MB
        assert room.tile_count == 1
```

### Test 4: Protocol Violation with Self-Correction

```python
def test_skip_p0_then_correct():
    """Test agent skipping P0, getting rejected, then submitting proper P0."""
    room = DeadbandRoom(capacity=1)
    agent = MockAgent("impatient_agent")
    room.join(agent)
    
    # Try to skip directly to P2 (violation)
    p2_tile = {
        "agent_id": agent.id,
        "phase": "P2",
        "content": "Optimizing without mapping negative space",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(p2_tile)
    assert result["status"] == "rejected"
    assert "P0" in result.get("reason", "")
    
    # Check agent state shows violation
    violations = room.get_protocol_violations(agent.id)
    assert len(violations) == 1
    assert violations[0]["violation"] == "phase_skip"
    
    # Now submit proper P0
    p0_tile = {
        "agent_id": agent.id,
        "phase": "P0",
        "content": "Correcting: map negative space first",
        "timestamp": time.time(),
        "correction": True
    }
    
    result = room.submit_tile(p0_tile)
    assert result["status"] == "accepted"
    assert "correction_accepted" in result.get("notes", "")
    
    # Should now allow P1
    agent_state = room.get_agent_state(agent.id)
    assert agent_state["current_phase"] == "P0"
    assert agent_state["can_advance"] == True
```
