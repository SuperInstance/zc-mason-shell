### Test 1: Concurrent Agent Tile Submission

```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "content": f"Test from {agent_id}",
            "phase": "P1",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") in ["accepted", "rejected"] for r in results)
    
    # Verify room state consistency
    assert room.get_agent_count() == len(agent_ids)
```

### Test 2: Network Partition Recovery
```python
def test_network_partition_recovery(self):
    """Test room recovery after simulated network partition."""
    room = DeadbandRoom()
    
    # Normal operation
    tile = {"agent_id": "agent1", "content": "P0 mapping", "phase": "P0"}
    result1 = room.submit_tile(tile)
    assert result1["status"] == "accepted"
    
    # Simulate network partition (connection loss)
    room._simulate_network_partition = True
    
    # Attempt submission during partition
    tile2 = {"agent_id": "agent1", "content": "P1 channel", "phase": "P1"}
    result2 = room.submit_tile(tile2)
    assert result2["status"] == "queued"
    assert "retry_after" in result2
    
    # Simulate network recovery
    room._simulate_network_partition = False
    room._recover_from_partition()
    
    # Verify queued tile processed
    assert room.get_queued_tile_count() == 0
    assert room.get_last_tile()["content"] == "P1 channel"
```

### Test 3: Malformed Tile Payload Handling
```python
def test_malformed_tile_handling(self):
    """Test room handling of various malformed tile payloads."""
    room = DeadbandRoom()
    
    malformed_tiles = [
        None,  # Null tile
        {},  # Empty dict
        {"agent_id": "agent1"},  # Missing required fields
        {"content": "test", "phase": "INVALID"},  # Invalid phase
        {"agent_id": "a" * 1000, "content": "x", "phase": "P0"},  # Oversized ID
        {"agent_id": "agent1", "content": "", "phase": "P0"},  # Empty content
        {"agent_id": "agent1", "content": "test", "phase": "P0", "extra": "x" * 10000},  # Oversized payload
    ]
    
    for i, tile in enumerate(malformed_tiles):
        try:
            result = room.submit_tile(tile)
            # Should either reject or sanitize
            assert result["status"] in ["rejected", "sanitized"]
            if result["status"] == "sanitized":
                assert "warnings" in result
        except Exception as e:
            # Room should not crash on malformed input
            assert isinstance(e, ValueError) or isinstance(e, TypeError)
```

### Test 4: Resource Exhaustion Resilience
```python
def test_memory_exhaustion_resilience(self):
    """Test room behavior under memory pressure."""
    room = DeadbandRoom()
    
    # Submit large number of tiles to simulate memory pressure
    large_tiles = []
    for i in range(1000):
        tile = {
            "agent_id": f"agent_{i}",
            "content": "x" * 1000,  # 1KB per tile
            "phase": "P1",
            "metadata": {"index": i}
        }
        large_tiles.append(tile)
    
    accepted = 0
    rejected = 0
    
    for tile in large_tiles:
        result = room.submit_tile(tile)
        if result["status"] == "accepted":
            accepted += 1
        else:
            rejected += 1
            assert "reason" in result
            assert "memory" in result["reason"].lower() or "quota" in result["reason"].lower()
    
    # Room should remain operational
    assert room.get_state() == "operational"
    assert accepted > 0  # Some tiles should be accepted
    assert room.get_memory_usage() < room.MAX_MEMORY_MB
```

### Test 5: Protocol Violation Recovery
```python
def test_p2_before_p1_recovery(self):
    """Test recovery when agent attempts P2 optimization before P1 channel finding."""
    room = DeadbandRoom()
    agent_id = "eager_agent"
    
    # Agent incorrectly starts with P2 (protocol violation)
    tile_p2 = {
        "agent_id": agent_id,
        "content": "Trying to optimize without channel",
        "phase": "P2"
    }
    
    result = room.submit_tile(tile_p2)
    
    # Room should reject P2 without prior P1
    assert result["status"] == "rejected"
    assert "P1" in result.get("message", "")
    
    # Room should provide guidance
    assert "expected_phase" in result
    assert result["expected_phase"] == "P1"
    
    # Agent corrects and submits P1
    tile_p1 = {
        "agent_id": agent_id,
        "content": "Mapping safe channels",
        "phase": "P1"
    }
    
    result_p1 = room.submit_tile(tile_p1)
    assert result_p1["status"] == "accepted"
    
    # Now P2 should be accepted
    tile_p2_corrected = {
        "agent_id": agent_id,
        "content": "Optimizing within channel",
        "phase": "P2"
    }
    
    result_p2 = room.submit_tile(tile_p2_corrected)
    assert result_p2["status"] == "accepted"
```
