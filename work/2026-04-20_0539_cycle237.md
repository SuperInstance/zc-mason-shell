```python
def test_concurrent_tile_processing(self):
    """Test DeadbandRoom handles simultaneous tile submissions from multiple agents."""
    import threading
    from queue import Queue
    
    room = DeadbandRoom(room_id="concurrent_test")
    results = Queue()
    
    def submit_tile(agent_id, priority):
        tile = {
            "agent_id": agent_id,
            "timestamp": time.time(),
            "content": f"Test from {agent_id}",
            "priority": priority,
            "protocol_phase": "P0" if priority == 0 else "P1" if priority == 1 else "P2"
        }
        try:
            result = room.process_tile(tile)
            results.put((agent_id, "success", result))
        except Exception as e:
            results.put((agent_id, "error", str(e)))
    
    # Create 10 agent threads submitting tiles
    threads = []
    for i in range(10):
        priority = i % 3  # Mix of P0, P1, P2
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}", priority))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all tiles were processed
    success_count = 0
    while not results.empty():
        agent_id, status, _ = results.get()
        if status == "success":
            success_count += 1
    
    assert success_count == 10, f"Expected 10 successful submissions, got {success_count}"
    assert room.get_tile_count() == 10, f"Room should have 10 tiles, has {room.get_tile_count()}"
```

```python
def test_memory_pressure_handling(self):
    """Test room behavior when processing large volumes of tiles."""
    room = DeadbandRoom(room_id="memory_pressure_test", max_tiles_in_memory=100)
    
    # Generate 150 tiles (exceeding memory limit)
    large_tiles = []
    for i in range(150):
        large_tiles.append({
            "agent_id": f"load_agent_{i % 5}",
            "priority": i % 3,
            "protocol_phase": ["P0", "P1", "P2"][i % 3],
            "content": "x" * 1024,  # 1KB per tile
            "timestamp": time.time() + i
        })
    
    processed_count = 0
    errors = []
    
    for tile in large_tiles:
        try:
            room.process_tile(tile)
            processed_count += 1
        except MemoryError as e:
            errors.append(str(e))
        except Exception as e:
            errors.append(f"Unexpected error: {e}")
    
    # Room should handle memory pressure gracefully
    assert processed_count >= 100, f"Should process at least up to limit, processed {processed_count}"
    
    if errors:
        # Errors should be MemoryError, not crashes
        assert all("memory" in err.lower() or "limit" in err.lower() for err in errors), \
            f"Unexpected error types: {errors}"
    
    # Verify persistence kicked in
    assert room.get_persisted_tile_count() > 0, "Should have persisted excess tiles to disk"
```

```python
def test_invalid_priority_handling(self):
    """Test room handling of malformed priority values."""
    room = DeadbandRoom(room_id="priority_edge_test")
    
    edge_cases = [
        {"priority": -1, "expected": "error"},
        {"priority": 3, "expected": "error"},
        {"priority": "P0", "expected": "error"},  # string instead of int
        {"priority": None, "expected": "error"},
        {"priority": 2.5, "expected": "error"},  # float instead of int
        {"priority": 0, "expected": "success"},
        {"priority": 1, "expected": "success"},
        {"priority": 2, "expected": "success"},
    ]
    
    results = []
    for case in edge_cases:
        tile = {
            "agent_id": "test_agent",
            "priority": case["priority"],
            "protocol_phase": "P0",  # Will be invalid for non-integer priorities
            "content": f"Test with priority {case['priority']}",
            "timestamp": time.time()
        }
        
        try:
            room.process_tile(tile)
            results.append((case["priority"], "success"))
        except (ValueError, TypeError, ProtocolViolationError) as e:
            results.append((case["priority"], "error", str(e)))
        except Exception as e:
            results.append((case["priority"], "unexpected", str(e)))
    
    # Validate expected outcomes
    for i, (priority, expected) in enumerate(edge_cases):
        result = results[i]
        if expected == "success":
            assert result[1] == "success", f"Priority {priority} should succeed, got {result[1]}"
        else:
            assert result[1] == "error", f"Priority {priority} should error, got {result[1]}"
```
