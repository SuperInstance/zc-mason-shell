# test_deadband_room_edge_cases.py
import pytest
import json
from datetime import datetime
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.models.tile import Tile

class TestDeadbandRoomEdgeCases:
    
    def test_protocol_sequence_violation(self):
        """Test that P2 actions are rejected when P1 not complete."""
        room = DeadbandRoom()
        
        # Force room to P0 state
        room.current_protocol = "P0"
        room.p1_complete = False
        
        # Attempt P2-style tile (optimization content)
        tile = Tile(
            content='{"action": "optimize", "data": "test"}',
            agent_id="test_agent",
            timestamp=datetime.utcnow().isoformat()
        )
        
        result = room.process_tile(tile)
        assert not result["accepted"]
        assert "P1 not complete" in result.get("reason", "")
    
    def test_concurrent_tile_processing(self):
        """Test handling of simultaneous tile submissions."""
        room = DeadbandRoom()
        
        # Create multiple tiles with same timestamp
        timestamp = datetime.utcnow().isoformat()
        tiles = [
            Tile(
                content=f'{{"action": "map", "data": "test_{i}"}}',
                agent_id=f"agent_{i}",
                timestamp=timestamp
            )
            for i in range(5)
        ]
        
        # Process all tiles
        results = []
        for tile in tiles:
            results.append(room.process_tile(tile))
        
        # All should be accepted if within limits
        accepted_count = sum(1 for r in results if r["accepted"])
        assert accepted_count == len(tiles)
        
        # Check processing order preserved (FIFO)
        processed_ids = [tile.agent_id for tile in tiles]
        room_agent_ids = [t.agent_id for t in room.processed_tiles[-5:]]
        assert processed_ids == room_agent_ids
    
    def test_state_persistence_after_reset(self):
        """Test room maintains protocol state after reset."""
        room = DeadbandRoom()
        
        # Advance to P1
        room.current_protocol = "P1"
        room.p1_complete = True
        room.p0_lessons = ["avoid_rock_1", "avoid_rock_2"]
        
        # Simulate reset (new instance with same config)
        config = room.get_config()
        new_room = DeadbandRoom.from_config(config)
        
        # State should be preserved
        assert new_room.current_protocol == "P1"
        assert new_room.p1_complete == True
        assert len(new_room.p0_lessons) == 2
    
    def test_semantically_invalid_tile(self):
        """Test tile with valid JSON but invalid semantics."""
        room = DeadbandRoom()
        
        # JSON is valid but action doesn't match protocol state
        tile = Tile(
            content='{"action": "optimize_channel", "channel": "invalid_channel_123"}',
            agent_id="test_agent",
            timestamp=datetime.utcnow().isoformat()
        )
        
        result = room.process_tile(tile)
        
        # Should be rejected with specific reason
        assert not result["accepted"]
        assert "invalid" in result.get("reason", "").lower() or \
               "not allowed" in result.get("reason", "").lower()
    
    def test_resource_boundary_conditions(self):
        """Test room behavior at tile capacity limits."""
        room = DeadbandRoom()
        
        # Fill room to near capacity
        max_tiles = room.max_tiles_per_cycle
        for i in range(max_tiles - 1):
            tile = Tile(
                content=f'{{"action": "map", "data": "fill_{i}"}}',
                agent_id=f"filler_{i}",
                timestamp=datetime.utcnow().isoformat()
            )
            room.process_tile(tile)
        
        # One more should succeed
        tile = Tile(
            content='{"action": "map", "data": "last_one"}',
            agent_id="last_agent",
            timestamp=datetime.utcnow().isoformat()
        )
        result1 = room.process_tile(tile)
        assert result1["accepted"]
        
        # Next one should fail
        tile = Tile(
            content='{"action": "map", "data": "one_too_many"}',
            agent_id="excess_agent",
            timestamp=datetime.utcnow().isoformat()
        )
        result2 = room.process_tile(tile)
        assert not result2["accepted"]
        assert "capacity" in result2.get("reason", "").lower()
