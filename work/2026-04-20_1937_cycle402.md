def test_protocol_violation_skip_to_p2():

    room = DeadbandRoom(room_id="edge_test_1")

### Test 3: Concurrent Agent Submissions
```python
def test_concurrent_agent_submissions():
    """DeadbandRoom should handle multiple agents submitting tiles in the same cycle."""
    room = DeadbandRoom(room_id="edge_test_3")
    
    # Simulate concurrent submissions
    agents = [f"agent_{i}" for i in range(5)]
    tiles = []
    
    for i, agent_id in enumerate(agents):
        tile = {
            "agent_id": agent_id,
            "cycle": 1,
            "phase": "P0",
            "content": f"Negative space mapping from {agent_id}",
            "protocol_step": 0
        }
        tiles.append((agent_id, tile))
    
    # Submit all tiles
    results = []
    for agent_id, tile in tiles:
        result = room.submit_tile(agent_id, tile)
        results.append(result)
    
    # All should be accepted if protocol is followed
    accepted_count = sum(1 for r in results if r["accepted"])
    assert accepted_count == len(agents)
    
    # Verify each agent's state is tracked separately
    for agent_id in agents:
        agent_state = room.get_agent_state(agent_id)
        assert agent_state is not None
        assert agent_state["last_cycle"] == 1
```

def test_cycle_number_validation():

    tile1 = {

        "content": "First tile",
        "protocol_step": 0
    }

    assert result1["accepted"]
    
    # Submit cycle 3 (skipping cycle 2)
    tile3 = {
        "agent_id": agent_id,
        "cycle": 3,
        "phase": "P1",
        "content": "Skipped cycle 2",
        "protocol_step": 1
    }
    result3 = room.submit_tile(agent_id, tile3)
    assert not result3["accepted"]
    assert "cycle" in result3.get("reason", "").lower() or "sequence" in result3.get("reason", "").lower()
    
    # Submit cycle 2 (should be accepted)
    tile2 = {
        "agent_id": agent_id,
        "cycle": 2,
        "phase": "P1",
        "content": "Correct cycle",
        "protocol_step": 1
    }
    result2 = room.submit_tile(agent_id, tile2)
    assert result2["accepted"]
