"""
Edge-case tests for DeadbandRoom preset.
Focus on protocol violations, adversarial inputs, and concurrency scenarios.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
import json
from datetime import datetime, timedelta

from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import TrainingTile


class TestDeadbandRoomEdgeCases:
    """Edge-case tests that go beyond basic protocol validation."""
    
    @pytest.mark.asyncio
    async def test_concurrent_p0_violation_detection(self):
        """Test that multiple agents violating P0 simultaneously are handled correctly."""
        room = DeadbandRoom(room_id="edge-concurrent")
        
        # Create multiple agents with P0 violations
        agents = []
        for i in range(5):
            agent = Mock()
            agent.agent_id = f"agent-{i}"
            agent.submit_tile = AsyncMock()
            await room.join_room(agent)
            agents.append(agent)
        
        # All agents submit P0-violating tiles simultaneously
        violating_tiles = []
        for i, agent in enumerate(agents):
            tile = TrainingTile(
                agent_id=agent.agent_id,
                content=f"P0 violation {i}: Skipping to P2 optimization",
                protocol_phase="P2",  # Violation: P2 without P0/P1
                timestamp=datetime.utcnow()
            )
            violating_tiles.append(tile)
        
        # Submit all tiles concurrently
        tasks = [room.process_tile(agent.agent_id, tile) for agent, tile in zip(agents, violating_tiles)]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Verify all violations are caught
        assert all(isinstance(r, dict) and r.get("status") == "rejected" for r in results)
        assert all("P0 violation" in r.get("reason", "") for r in results)
        
        # Verify room state remains consistent
        assert len(room.active_agents) == 5
        assert room.room_state == "training"
    
    @pytest.mark.asyncio
    async def test_malformed_tile_injection(self):
        """Test handling of deeply malformed tile data."""
        room = DeadbandRoom(room_id="edge-malformed")
        agent = Mock()
        agent.agent_id = "malicious-agent"
        await room.join_room(agent)
        
        # Test various malformed payloads
        malformed_payloads = [
            None,  # Null tile
            "not a dict",  # String instead of dict
            {"agent_id": 123},  # Wrong type for agent_id
            {"agent_id": "test", "content": None, "protocol_phase": "P0"},  # Null content
            {"agent_id": "test", "content": "", "protocol_phase": "INVALID"},  # Invalid phase
            {"agent_id": "test", "content": "x" * 1000000, "protocol_phase": "P0"},  # Excessive size
            {"agent_id": "test", "content": "<script>alert('xss')</script>", "protocol_phase": "P0"},  # XSS attempt
            {"agent_id": "test", "content": {"nested": "object"}, "protocol_phase": "P0"},  # Non-string content
        ]
        
        for payload in malformed_payloads:
            try:
                # Try to create tile with malformed data
                if isinstance(payload, dict):
                    tile = TrainingTile(**payload)
                else:
                    tile = payload
                
                result = await room.process_tile(agent.agent_id, tile)
                
                # Should either reject or raise appropriate exception
                if isinstance(result, dict):
                    assert result.get("status") == "rejected"
                else:
                    # Exception should be caught and logged
                    assert isinstance(result, Exception) or result is None
            except (TypeError, ValueError) as e:
                # Expected for truly invalid constructions
                pass
        
        # Agent should still be in room after malformed attempts
        assert agent.agent_id in room.active_agents
    
    @pytest.mark.asyncio
    async def test_resource_exhaustion_tile_flood(self):
        """Test room behavior under high-volume tile submission."""
        room = DeadbandRoom(room_id="edge-flood")
        agent = Mock()
        agent.agent_id = "flood-agent"
        await room.join_room(agent)
        
        # Submit large number of valid tiles rapidly
        num_tiles = 1000
        submission_times = []
        
        for i in range(num_tiles):
            tile = TrainingTile(
                agent_id=agent.agent_id,
                content=f"Valid tile {i} with proper protocol",
                protocol_phase="P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
                timestamp=datetime.utcnow()
            )
            
            start = datetime.utcnow()
            result = await room.process_tile(agent.agent_id, tile)
            end = datetime.utcnow()
            
            submission_times.append((end - start).total_seconds())
            
            # First few should succeed, later ones might be rate-limited
            if i < 100:  # Initial burst allowance
                assert result.get("status") in ["accepted", "rejected"]
            else:
                # After burst, might get rate-limited or queued
                assert result.get("status") in ["accepted", "rejected", "queued"] or \
                       "rate limit" in result.get("reason", "").lower()
        
        # Calculate performance metrics
        avg_time = sum(submission_times) / len(submission_times)
        max_time = max(submission_times)
        
        # Room should still be functional
        assert room.room_state == "training"
        assert agent.agent_id in room.active_agents
        
        # Log performance characteristics (not an assertion, for monitoring)
        print(f"Tile flood test: {num_tiles} tiles, avg={avg_time:.4f}s, max={max_time:.4f}s")
