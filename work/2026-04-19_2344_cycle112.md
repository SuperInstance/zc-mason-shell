```python
def test_resource_exhaustion_tile_flood():
    """P0: Agent submits excessive tiles to overwhelm room processing."""
    room = DeadbandRoom()
    agent_state = {"cycle": 1, "phase": "test"}
    
    flood_tiles = []
    for i in range(1000):  # Excessive count
        flood_tiles.append({
            "priority": "P0",
            "action": "map_negative_space",
            "payload": {"hazard": f"rock_{i}"}
        })
    
    accepted_count = 0
    for tile in flood_tiles:
        result = room.submit_tile(agent_state, tile)
        if result.get("accepted"):
            accepted_count += 1
    
    # Room should have rate-limiting or queue limits
    assert accepted_count < 1000  # Not all accepted
    assert hasattr(room, "queue_limit") or accepted_count <= 100  # Reasonable limit
```

### Test 4: Concurrent Agent Race Condition
```python
def test_concurrent_agent_race_condition():
    """P0: Simulate two agents submitting tiles for same channel simultaneously."""
    room = DeadbandRoom()
    agent_a = {"cycle": 1, "phase": "build", "agent_id": "agent_a"}
    agent_b = {"cycle": 1, "phase": "build", "agent_id": "agent_b"}
    
    # Both agents map same P0 hazard
    tile_p0 = {"priority": "P0", "action": "map_negative_space", "payload": {"hazard": "rock"}}
    room.submit_tile(agent_a, tile_p0)
    room.submit_tile(agent_b, tile_p0)
    
    # Both try to claim same P1 channel
    tile_p1_a = {"priority": "P1", "action": "find_channel", "payload": {"channel": "north_path"}}
    tile_p1_b = {"priority": "P1", "action": "find_channel", "payload": {"channel": "north_path"}}
    
    result_a = room.submit_tile(agent_a, tile_p1_a)
    result_b = room.submit_tile(agent_b, tile_p1_b)
    
    # Only one should succeed (channel contention)
    assert result_a["accepted"] != result_b["accepted"]  # XOR
    assert "channel" in room.claimed_channels
    assert len(room.claimed_channels["north_path"]) == 1  # Single owner
```

```python
def test_malformed_tile_payloads():

    for tile in malformed_tiles:
        result = room.submit_tile(agent_state, tile)
        assert not result["accepted"]
        assert "invalid" in result["message"].lower() or "error" in result["message"].lower()
```
