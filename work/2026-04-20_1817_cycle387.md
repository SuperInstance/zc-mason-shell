```python
async def test_process_tile_extreme_numeric_values(tmp_path, event_loop):
    """Test tile processing with very large, small, or negative numeric metadata."""
    config = {"room_id": "edge-test-1", "storage_path": str(tmp_path)}
    room = DeadbandRoom(config)
    
    # Tile with extreme values
    tile_data = {
        "tile_id": "tile_extreme",
        "agent_id": "agent_test",
        "content": "test content",
        "metadata": {
            "score": 1e308,  # Very large float
            "neg_score": -1e308,
            "zero": 0.0,
            "tiny": 1e-308,
            "int_max": 2**63 - 1,
        }
    }
    
    result = await room.process_tile(json.dumps(tile_data))
    assert result["status"] == "accepted"
    # Verify metadata preserved
    stored = await room.get_state()
    assert stored["tiles"][0]["metadata"]["score"] == 1e308
    assert stored["tiles"][0]["metadata"]["int_max"] == 9223372036854775807
```

**Test 3: Rapid concurrent tile processing with same tile_id**
```python
async def test_concurrent_duplicate_tile_ids(tmp_path, event_loop):
    """Test race condition when multiple threads submit tiles with same ID."""
    config = {"room_id": "edge-test-3", "storage_path": str(tmp_path)}
    room = DeadbandRoom(config)
    
    tile_template = {
        "tile_id": "duplicate_id",  # Same ID for all
        "agent_id": "agent_test",
        "content": "",
        "metadata": {}
    }
    
    async def submit_tile(index):
        tile = tile_template.copy()
        tile["content"] = f"Content {index}"
        return await room.process_tile(json.dumps(tile))
    
    # Submit 5 tiles concurrently with same tile_id
    tasks = [submit_tile(i) for i in range(5)]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # Should accept first, reject duplicates or handle gracefully
    accepted = [r for r in results if not isinstance(r, Exception) and r["status"] == "accepted"]
    assert len(accepted) >= 1  # At least first accepted
    # Verify only one tile stored or all stored with deduplication
    stored = await room.get_state()
    assert len(stored["tiles"]) <= 5  # Should not crash
```

**Test 4: Malformed JSON that's partially valid**
```python
async def test_process_tile_partial_json(tmp_path, event_loop):
    """Test tile processing with truncated or malformed JSON that's partially parseable."""
    config = {"room_id": "edge-test-4", "storage_path": str(tmp_path)}
    room = DeadbandRoom(config)
    
    # Valid JSON but missing closing brace
    partial_json = '{"tile_id": "partial", "agent_id": "test", "content": "test"'
    
    with pytest.raises(ValueError, match="JSON decode error"):
        await room.process_tile(partial_json)
    
    # JSON with extra trailing comma (invalid per strict JSON)
    trailing_comma = '{"tile_id": "trailing", "agent_id": "test", "content": "test",}'
    
    # Depending on JSON parser, may or may not raise
    try:
        result = await room.process_tile(trailing_comma)
        # If accepted, verify it worked
        if result["status"] == "accepted":
            stored = await room.get_state()
            assert stored["tiles"][0]["tile_id"] == "trailing"
    except ValueError as e:
        assert "JSON" in str(e)  # Should mention JSON parsing
```

```python
async def test_storage_path_permissions(tmp_path, event_loop):
    """Test behavior with read-only storage path and path traversal attempts."""
    config = {"room_id": "edge-test-5", "storage_path": str(tmp_path)}
    room = DeadbandRoom(config)
    
    # First, normal write should work
    tile_data = {
        "tile_id": "normal",
        "agent_id": "agent_test",
        "content": "test",
        "metadata": {}
    }
    
    result = await room.process_tile(json.dumps(tile_data))
    assert result["status"] == "accepted"
    
    # Make directory read-only (Unix only)
    if hasattr(os, 'chmod'):
        os.chmod(tmp_path, 0o444)  # Read-only
        
        tile_data["tile_id"] = "readonly_test"
        with pytest.raises((IOError, OSError, PermissionError)):
            await room.process_tile(json.dumps(tile_data))
        
        os.chmod(tmp_path, 0o755)  # Restore
    
    # Test path traversal attempt in room_id
    config_traversal = {"room_id": "../malicious", "storage_path": str(tmp_path)}
    room2 = DeadbandRoom(config_traversal)
    # Should either sanitize or raise error
    state = await room2.get_state()
    assert "room_id" in state
    # Verify room_id doesn't actually traverse
    assert "malicious" not in str(tmp_path.parent)
```
