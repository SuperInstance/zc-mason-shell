```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom()
    tiles = [
        Tile(agent_id="agent1", content="P0: test", priority=0),
        Tile(agent_id="agent2", content="P1: test", priority=1),
        Tile(agent_id="agent3", content="P2: test", priority=2)
    ]
    
    # Simulate concurrent submission
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(results), "Concurrent submissions failed"
    assert len(room.get_tiles()) == 3, "Tile count mismatch after concurrent submission"
```

### Test 2: Network Partition During Sync
```python
def test_partial_write_recovery(self):
    """Room should recover gracefully when tile storage fails mid-write."""
    room = DeadbandRoom()
    
    # Mock storage failure after first tile
    original_save = room._save_tile
    call_count = 0
    
    def failing_save(tile):
        nonlocal call_count
        call_count += 1
        if call_count == 2:
            raise IOError("Simulated storage failure")
        return original_save(tile)
    
    room._save_tile = failing_save
    
    # First tile should succeed
    tile1 = Tile(agent_id="test", content="P0: valid", priority=0)
    assert room.submit_tile(tile1), "First tile should succeed"
    
    # Second tile should fail gracefully
    tile2 = Tile(agent_id="test", content="P1: should fail", priority=1)
    assert not room.submit_tile(tile2), "Second tile should fail during storage"
    
    # Room state should remain consistent
    assert len(room.get_tiles()) == 1, "Room should contain only successfully saved tiles"
```

```python
def test_priority_sequence_violation(self):
    """Room should reject tiles that violate P0→P1→P2 sequence."""
    room = DeadbandRoom()
    
    # Submit P2 without P1
    tile_p2 = Tile(agent_id="test", content="P2: optimize", priority=2)
    assert not room.submit_tile(tile_p2), "Should reject P2 without preceding P1"
    
    # Submit P1 without P0
    tile_p1 = Tile(agent_id="test", content="P1: find channel", priority=1)
    assert not room.submit_tile(tile_p1), "Should reject P1 without preceding P0"
    
    # Valid sequence
    tile_p0 = Tile(agent_id="test", content="P0: avoid rocks", priority=0)
    assert room.submit_tile(tile_p0), "Valid P0 should be accepted"
    assert room.submit_tile(tile_p1), "Valid P1 after P0 should be accepted"
    assert room.submit_tile(tile_p2), "Valid P2 after P1 should be accepted"
```

### Test 5: Corrupted State Recovery
```python
def test_corrupted_state_recovery(self):
    """Room should recover from corrupted persistent state."""
    # Create room with valid state
    room = DeadbandRoom()
    tile = Tile(agent_id="test", content="P0: valid", priority=0)
    room.submit_tile(tile)
    room.save_state()
    
    # Corrupt the saved state
    state_file = room._get_state_file()
    with open(state_file, 'wb') as f:
        f.write(b'CORRUPTED_DATA')
    
    # Attempt to load corrupted state
    try:
        room.load_state()
        # If load succeeds, it should have fallback behavior
        assert room.get_tiles() == [], "Should have empty state after corruption"
    except Exception as e:
        # Room should raise specific corruption error
        assert "corrupt" in str(e).lower() or "invalid" in str(e).lower()
    
    # Room should be usable after corruption
    new_tile = Tile(agent_id="test", content="P0: recovery", priority=0)
    assert room.submit_tile(new_tile), "Should accept tiles after corruption recovery"
```
