### Test 1: Concurrent Tile Submission During Protocol Step
**Scenario:** What happens when a tile is submitted while the room is processing a protocol step?
**Edge Case:** Race condition between training data ingestion and state transition.
**Expected Behavior:** Either serialized processing or clear error about busy state.

```python
def test_concurrent_tile_during_step():
    """Test tile submission while room is in step() processing."""
    room = DeadbandRoom()
    room.step("map_p0")  # Start P0 mapping
    
    # Try to submit tile during active step
    tile_data = {"lesson": "test", "priority": "P0"}
    
    # This could either:
    # 1. Queue the tile for after step completes
    # 2. Raise an error about busy state
    # 3. Process immediately (if thread-safe)
    
    # Based on current implementation, it likely processes immediately
    # but we should verify no state corruption
    result = room.submit_tile(tile_data)
    assert result["status"] in ["queued", "accepted", "rejected"]
    
    # Verify protocol state unchanged
    assert room.current_phase == "P0"
```

### Test 2: Malformed Tile Data Types
**Scenario:** Submit tiles with incorrect data types (non-dict, wrong value types).
**Edge Case:** Type validation and error handling for training data.
**Expected Behavior:** Clear error messages with specific validation failures.

```python
def test_malformed_tile_data_types():
    """Test tile submission with various malformed data types."""
    room = DeadbandRoom()
    
    test_cases = [
        (None, "Tile data cannot be None"),
        ("string_not_dict", "Tile data must be a dictionary"),
        (123, "Tile data must be a dictionary"),
        ([], "Tile data must be a dictionary"),
        ({"priority": 123}, "Priority must be string"),
        ({"lesson": None}, "Lesson cannot be None"),
        ({"priority": "P5"}, "Priority must be P0, P1, or P2"),
        ({}, "Tile must contain at least 'lesson' or 'priority'"),
    ]
    
    for tile_data, expected_error in test_cases:
        try:
            room.submit_tile(tile_data)
            assert False, f"Should have raised error for {tile_data}"
        except (ValueError, TypeError) as e:
            assert expected_error in str(e), f"Wrong error for {tile_data}: {e}"
```

### Test 3: Protocol State Persistence Across Resets
**Scenario:** Multiple reset cycles with different protocol paths.
**Edge Case:** Ensure reset() truly clears all state, not just partial.
**Expected Behavior:** Complete state reset to initial conditions.

```python
def test_protocol_state_persistence_across_resets():
    """Test that reset() fully clears protocol state."""
    room = DeadbandRoom()
    
    # Take through full protocol
    room.step("map_p0")
    room.step("find_p1_channels")
    room.step("optimize_p2")
    room.submit_tile({"lesson": "full cycle", "priority": "P2"})
    
    # Verify state changed
    assert room.current_phase == "P2"
    assert len(room.tiles) > 0
    assert room.cycle_count > 0
    
    # Reset
    room.reset()
    
    # Verify complete reset
    assert room.current_phase == "P0"
    assert len(room.tiles) == 0
    assert room.cycle_count == 0
    assert room.protocol_history == []
    
    # Verify can start fresh protocol
    room.step("map_p0")
    assert room.current_phase == "P0"
    
    # Submit tile after reset
    result = room.submit_tile({"lesson": "fresh start", "priority": "P0"})
    assert result["status"] == "accepted"
```

### Test 4: Maximum Tile Capacity
**Scenario:** Submit large number of tiles (beyond expected daily use).
**Edge Case:** Memory/performance limits of tile storage.
**Expected Behavior:** Either graceful degradation or clear capacity limits.

```python
def test_maximum_tile_capacity():
    """Test behavior when submitting many tiles."""
    room = DeadbandRoom()
    
    # Submit many tiles
    max_reasonable = 10000  # Arbitrary large number
    accepted_count = 0
    
    for i in range(max_reasonable):
        tile_data = {
            "lesson": f"Test tile {i}",
            "priority": "P0",
            "timestamp": i,
            "agent": "mason"
        }
        
        try:
            result = room.submit_tile(tile_data)
            if result["status"] == "accepted":
                accepted_count += 1
            else:
                # Might start rejecting after certain limit
                break
        except MemoryError:
            # Hit system limit
            break
    
    # Verify either:
    # 1. All accepted (no practical limit)
    # 2. Some rejected after limit
    # 3. Memory error raised
    
    assert accepted_count > 0, "Should accept at least some tiles"
    
    if accepted_count < max_reasonable:
        # There's a limit - verify it's documented or consistent
        print(f"Tile limit found: {accepted_count} accepted")
        
    # Verify room still functional after many tiles
    room.step("map_p0")
    assert room.current_phase == "P0"
```

### Test 5: Invalid Protocol Skipping (P0→P2 Direct)
**Scenario:** Attempt to skip P1 entirely by going from P0 directly to P2.
**Edge Case:** Protocol enforcement for strict P0→P1→P2 priority.
**Expected Behavior:** Clear error about protocol violation.

```python
def test_invalid_protocol_skipping():
    """Test attempting to skip P1 phase entirely."""
    room = DeadbandRoom()
    
    # Start in P0
    room.step("map_p0")
    assert room.current_phase == "P0"
    
    # Try to jump directly to P2 (should fail)
    try:
        room.step("optimize_p2")
        assert False, "Should have raised protocol violation error"
    except ValueError as e:
        assert "P1" in str(e), f"Error should mention P1 requirement: {e}"
        assert "skip" in str(e) or "violat" in str(e).lower()
    
    # Verify still in P0
    assert room.current_phase == "P0"
    
    # Now do proper P0→P1→P2
    room.step("find_p1_channels")
    assert room.current_phase == "P1"
    
    room.step("optimize_p2")
    assert room.current_phase == "P2"
    
    # Try to go back to P0 from P2 (should also fail)
    try:
        room.step("map_p0")
        assert False, "Should have raised error about going backward"
    except ValueError as e:
        assert "back" in str(e).lower() or "reverse" in str(e).lower()
```
