### Test 2: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission():
    """Test that concurrent tile submissions don't corrupt room state."""
    room = DeadbandRoom()
    
    def submit_tile_thread(priority, content):
        return room.submit_tile(priority=priority, content=content)
    
    # Create multiple threads submitting tiles simultaneously
    threads = []
    results = []
    
    for i in range(10):
        t = threading.Thread(
            target=lambda i=i: results.append(
                submit_tile_thread(priority=i % 3, content={"thread": i})
            )
        )
        threads.append(t)
    
    # Start all threads
    for t in threads:
        t.start()
    
    # Wait for completion
    for t in threads:
        t.join()
    
    # Verify all submissions were successful
    assert len(results) == 10
    for result in results:
        assert result["success"] is True
    
    # Verify room state consistency
    assert len(room.get_tiles()) == 10
    assert room.get_stats()["total_tiles"] == 10
```

```python
def test_resource_exhaustion_recovery():
    """Test behavior when room reaches resource limits."""
    room = DeadbandRoom(max_tiles=5)  # Small limit for testing
    
    # Fill room to capacity
    for i in range(5):
        result = room.submit_tile(priority=i % 3, content={"index": i})
        assert result["success"] is True
    
    # Try to exceed capacity
    with pytest.raises(RuntimeError, match="Room at capacity"):
        room.submit_tile(priority=0, content={"index": "overflow"})
    
    # Clear some tiles
    room.clear_tiles(priority=0)  # Clear all P0 tiles
    
    # Should be able to submit again
    result = room.submit_tile(priority=0, content={"new": "tile"})
    assert result["success"] is True
    
    # Test memory usage tracking
    stats = room.get_stats()
    assert "memory_usage_mb" in stats
    assert stats["memory_usage_mb"] > 0
```
