def test_p1_to_p0_regression():
    """Test that room can handle regression from P1 back to P0."""
    room = DeadbandRoom(
        room_id="regression-test",
        max_cycles=5,
        p0_threshold=0.5,
        p1_threshold=0.8,
        tile_store=Mock()
    )
    
    # Start with high-quality actions (should be P2)
    actions_high = [{"score": 0.9}, {"score": 0.95}]
    result1 = room.process_actions(actions_high)
    assert result1["current_phase"] == "P2"
    
    # Switch to very poor actions (should regress to P0)
    actions_low = [{"score": 0.1}, {"score": 0.2}]
    result2 = room.process_actions(actions_low)
    
    # Should detect regression and return to P0
    assert result2["current_phase"] == "P0"
    assert "regression_detected" in result2.get("metadata", {})

def test_tile_store_exception_recovery():
    """Test recovery when TileStore raises exception."""
    mock_store = Mock()
    mock_store.save_tile.side_effect = [
        Exception("Storage error"),
        None,  # Second call succeeds
    ]
    
    room = DeadbandRoom(
        room_id="recovery-test",
        max_cycles=3,
        tile_store=mock_store
    )
    
    actions = [{"score": 0.8}, {"score": 0.9}]
    
    # Should not crash on first storage failure
    result = room.process_actions(actions)
    
    # Should indicate partial success
    assert result["tiles_saved"] == 1
    assert "storage_errors" in result.get("metadata", {})
    assert mock_store.save_tile.call_count == 2
```

### Test 5: Integration - Real TileStore with Memory Backend

def test_integration_memory_tile_store():
    """Integration test with actual MemoryTileStore."""
    from plato_torch.storage.memory_tile_store import MemoryTileStore
    
    tile_store = MemoryTileStore()
    room = DeadbandRoom(
        room_id="integration-test",
        max_cycles=5,
        tile_store=tile_store
    )
    
    actions = [
        {"score": 0.4, "content": "P0 action"},
        {"score": 0.6, "content": "P1 action"},
        {"score": 0.9, "content": "P2 action"},
    ]
    
    result = room.process_actions(actions)
    
    # Verify tiles were actually stored
    stored_tiles = tile_store.get_tiles(room_id="integration-test")
    assert len(stored_tiles) == len(actions)
    
    # Verify tile contents match inputs
    for i, tile in enumerate(stored_tiles):
        assert tile["content"] == actions[i]["content"]
        assert tile["phase"] == result["phases"][i]
```
