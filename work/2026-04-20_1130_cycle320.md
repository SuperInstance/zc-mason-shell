### Test 3: Concurrent Tile Submissions
```python
def test_deadband_room_concurrent_submissions():
    """Simulate concurrent tile submissions from multiple agents."""
    room = DeadbandRoom(room_id="concurrent_test")
    
    def submit_tile_safe(agent_id, tile_num):
        """Thread-safe tile submission."""
        try:
            return room.submit_tile(
                agent_id=agent_id,
                tile_data={
                    "phase": "P0",
                    "timestamp": 1713619200 + tile_num,
                    "content": f"Test tile {tile_num} from {agent_id}"
                }
            )
        except Exception as e:
            return str(e)
    
    # Simulate 10 agents submitting tiles concurrently
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = []
        for i in range(10):
            future = executor.submit(
                submit_tile_safe, 
                f"agent_{i}", 
                i
            )
            futures.append(future)
        
        results = [f.result() for f in futures]
    
    # Verify all submissions were processed
    successful = [r for r in results if r is not None]
    assert len(successful) == 10, f"Expected 10 successful submissions, got {len(successful)}"
    
    # Verify tile ordering (should maintain submission order per agent)
    tiles = room.get_tiles(agent_id="agent_0")
    timestamps = [t["timestamp"] for t in tiles]
    assert timestamps == sorted(timestamps), "Tiles should be in timestamp order"
```

def test_deadband_room_phase_boundaries():
    """Test protocol execution at exact threshold boundaries."""
    room = DeadbandRoom(
        room_id="boundary_test",
        p0_threshold=0.3,
        p1_threshold=0.6,
        p2_threshold=0.9
    )
    
    # Test at exact P0 threshold
    result = room.execute_protocol(current_value=0.3)
    assert result["phase"] == "P0", f"Expected P0 at threshold, got {result['phase']}"
    assert result["action"] == "map_negative_space"
    
    # Test just above P0 threshold (should still be P0)
    result = room.execute_protocol(current_value=0.3001)
    assert result["phase"] == "P0"
    
    # Test at exact P1 threshold
    result = room.execute_protocol(current_value=0.6)
    assert result["phase"] == "P1", f"Expected P1 at threshold, got {result['phase']}"
    assert result["action"] == "find_safe_channels"
    
    # Test at exact P2 threshold
    result = room.execute_protocol(current_value=0.9)
    assert result["phase"] == "P2", f"Expected P2 at threshold, got {result['phase']}"
    assert result["action"] == "optimize_within_channels"
    
    # Test above P2 threshold
    result = room.execute_protocol(current_value=1.0)
    assert result["phase"] == "P2", "Values above P2 threshold should still be P2"

### Test 5: Recovery From Partial Protocol Failure
```python
def test_deadband_room_partial_failure_recovery():
    """Test room recovery after partial protocol execution failure."""
    room = DeadbandRoom(room_id="recovery_test")
    
    # Start a multi-step protocol
    room.start_protocol_session(session_id="test_session")
    
    # Simulate partial execution
    room.execute_protocol_step(session_id="test_session", step_data={"phase": "P0"})
    room.execute_protocol_step(session_id="test_session", step_data={"phase": "P1"})
    
    # Simulate failure (e.g., agent disconnect)
    # Room should maintain session state
    
    # Attempt to resume with invalid session
    with pytest.raises(ValueError):
        room.execute_protocol_step(session_id="invalid_session", step_data={"phase": "P2"})
    
    # Resume valid session
    result = room.execute_protocol_step(session_id="test_session", step_data={"phase": "P2"})
    assert result["completed"] is True
    
    # Verify session history
    history = room.get_session_history(session_id="test_session")
    assert len(history) == 3
    assert history[0]["phase"] == "P0"
    assert history[1]["phase"] == "P1"
    assert history[2]["phase"] == "P2"
    
    # Test session cleanup
    room.cleanup_session(session_id="test_session")
    with pytest.raises(ValueError):
        room.get_session_history(session_id="test_session")
```
