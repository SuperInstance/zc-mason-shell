### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race_condition():
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(capacity=10)
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "phase": "P1",
            "content": f"Safe channel analysis from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append((agent_id, tile))
    
    # Submit all tiles rapidly
    results = []
    for agent_id, tile in tiles:
        result = room.submit_tile(agent_id, tile)
        results.append(result)
    
    # All submissions should be processed without corruption
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent submissions should not corrupt room state"
    
    # Verify each agent's state is correctly updated
    for agent_id in agent_ids:
        state = room.get_agent_state(agent_id)
        assert state["phase"] == "P1", \
            f"Agent {agent_id} should be in P1 phase after submission"
```

### Test 3: Protocol Violation with Mixed Priority Signals
```python
def test_mixed_priority_protocol_violation():
    """Test tile that contains both P0 (negative) and P2 (optimization) content."""
    room = DeadbandRoom(capacity=3)
    
    # Tile attempts to skip from P0 directly to P2
    tile = {
        "phase": "P0",  # Starting phase
        "content": "Avoid rocks AND optimize path",  # Mixed priority
        "p0_warnings": ["rock ahead"],
        "p2_optimizations": ["shortest path found"],  # Should not be in P0
        "timestamp": time.time()
    }
    
    result = room.submit_tile("violating_agent", tile)
    
    # Should be rejected for protocol violation
    assert result["status"] == "rejected", \
        "Mixed priority content should be rejected"
    assert "protocol_violation" in result.get("errors", []), \
        "Should flag protocol violation error"
    
    # Agent should remain in P0 phase
    state = room.get_agent_state("violating_agent")
    assert state["phase"] == "P0", \
        "Agent should not advance after protocol violation"
```

### Test 4: Network Partition Recovery
```python
def test_network_partition_recovery():
    """Test room recovery after simulated network partition."""
    room = DeadbandRoom(capacity=8)
    
    # Normal operation
    tile1 = {"phase": "P1", "content": "Channel analysis", "timestamp": time.time()}
    result1 = room.submit_tile("agent_a", tile1)
    assert result1["status"] == "accepted"
    
    # Simulate network partition - room loses connection to ensign storage
    room._ensign_connection = False
    
    # Submission during partition
    tile2 = {"phase": "P1", "content": "More analysis", "timestamp": time.time()}
    result2 = room.submit_tile("agent_a", tile2)
    
    # Should queue tile but not apply to ensign
    assert result2["status"] == "queued", \
        "Tile should be queued during network partition"
    assert room._tile_queue.qsize() > 0, \
        "Queue should contain unprocessed tiles"
    
    # Restore connection
    room._ensign_connection = True
    room._process_queued_tiles()
    
    # Verify queued tile was processed
    state = room.get_agent_state("agent_a")
    assert state["score"] > 0, \
        "Score should reflect processed queued tile"
    assert room._tile_queue.qsize() == 0, \
        "Queue should be empty after recovery"
```

### Test 5: Memory Exhaustion Edge Case
```python
def test_memory_exhaustion_handling():
    """Test room behavior when approaching memory limits."""
    room = DeadbandRoom(capacity=2, memory_limit_mb=1)  # 1MB limit
    
    agents = ["agent_x", "agent_y"]
    
    # Fill room with large tiles
    for i, agent_id in enumerate(agents):
        large_content = "data" * 50000  # ~200KB per tile
        tile = {
            "phase": "P1",
            "content": large_content,
            "timestamp": time.time()
        }
        result = room.submit_tile(agent_id, tile)
        assert result["status"] == "accepted"
    
    # Attempt to exceed memory limit
    agent_z = "agent_z"
    oversized_content = "data" * 300000  # ~1.2MB - should exceed limit
    tile = {
        "phase": "P1",
        "content": oversized_content,
        "timestamp": time.time()
    }
    
    result = room.submit_tile(agent_z, tile)
    
    # Should reject or trigger garbage collection
    if result["status"] == "rejected":
        assert "memory_limit" in result.get("errors", []), \
            "Should flag memory limit error"
    elif result["status"] == "accepted":
        # If accepted, verify garbage collection occurred
        assert room._memory_usage_mb < 1.0, \
            "Memory should be below limit after GC"
        # Verify oldest tile was evicted
        assert len(room._tile_history) <= 2, \
            "Tile history should be pruned"
```
