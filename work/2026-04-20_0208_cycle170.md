## Analysis of Existing Tests
The `DeadbandRoom` class implements the fleet's Deadband Protocol (P0 negative space, P1 safe channels, P2 optimization). Existing tests in `test_deadband.py` cover:
- Basic protocol flow (P0→P1→P2 progression)
- Valid state transitions
- Input validation (rejecting invalid priorities)
- Simple constraint satisfaction

Missing edge cases identified:
1. **Protocol violation recovery** – What happens when an agent tries to skip from P0 directly to P2?
2. **Concurrent agent interference** – Multiple agents in same room with conflicting constraints.
3. **Constraint exhaustion** – No safe channels found after P0 mapping.
4. **State persistence across cycles** – Room state maintenance between agent interactions.
5. **Malformed tile submission** – Invalid JSON, missing required fields.

### Test 1: Protocol Violation Recovery
```python
def test_protocol_violation_skip_p1():
    """Agent attempts to skip P1 and go directly from P0 to P2.
    Expected: Room rejects transition, returns to P0 or P1."""
    room = DeadbandRoom()
    agent_id = "test_agent_1"
    
    # Start at P0
    room.enter_agent(agent_id)
    room.submit_tile(agent_id, {"phase": "P0", "constraints": ["rock1", "rock2"]})
    
    # Attempt illegal P2 submission
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.submit_tile(agent_id, {"phase": "P2", "optimization": "fastest"})
    
    assert "Cannot transition from P0 to P2" in str(exc_info.value)
    assert room.get_agent_phase(agent_id) in ["P0", "P1"]  # Should revert or stay
```

### Test 2: Concurrent Agent Constraint Conflict
```python
def test_concurrent_agents_conflicting_constraints():
    """Two agents map conflicting negative spaces in P0.
    Expected: Room should handle both constraint sets without deadlock."""
    room = DeadbandRoom()
    agent_a = "agent_a"
    agent_b = "agent_b"
    
    room.enter_agent(agent_a)
    room.enter_agent(agent_b)
    
    # Agent A maps constraints
    room.submit_tile(agent_a, {
        "phase": "P0",
        "constraints": ["avoid_region_x", "avoid_time_window_t1"]
    })
    
    # Agent B maps overlapping but different constraints
    room.submit_tile(agent_b, {
        "phase": "P0", 
        "constraints": ["avoid_region_x", "avoid_resource_r1"]
    })
    
    # Both should be able to proceed to P1
    room.submit_tile(agent_a, {
        "phase": "P1",
        "safe_channels": ["channel_alpha", "channel_beta"]
    })
    
    room.submit_tile(agent_b, {
        "phase": "P1",
        "safe_channels": ["channel_gamma"]
    })
    
    # Verify both agents have independent state
    assert room.get_agent_phase(agent_a) == "P1"
    assert room.get_agent_phase(agent_b) == "P1"
    assert len(room.get_agent_channels(agent_a)) == 2
    assert len(room.get_agent_channels(agent_b)) == 1
```

### Test 4: State Persistence Across Cycles
```python
def test_state_persistence_between_interactions():
    """Agent leaves and re-enters room, state should persist.
    Expected: Agent resumes from last completed phase."""
    room = DeadbandRoom()
    agent_id = "test_agent_4"
    
    # First session
    room.enter_agent(agent_id)
    room.submit_tile(agent_id, {
        "phase": "P0",
        "constraints": ["constraint_a"]
    })
    room.submit_tile(agent_id, {
        "phase": "P1", 
        "safe_channels": ["channel_1"]
    })
    room.leave_agent(agent_id)
    
    # Second session
    room.enter_agent(agent_id)
    
    # Should resume at P1 or P2 depending on implementation
    current_phase = room.get_agent_phase(agent_id)
    assert current_phase in ["P1", "P2"]
    
    # If at P1, channels should be preserved
    if current_phase == "P1":
        assert room.get_agent_channels(agent_id) == ["channel_1"]
```

### Test 5: Malformed Tile Submission
```python
def test_malformed_tile_submission():
    """Submit tiles with invalid structure, missing fields, wrong types.
    Expected: Room rejects with descriptive error, maintains agent state."""
    room = DeadbandRoom()
    agent_id = "test_agent_5"
    
    room.enter_agent(agent_id)
    
    # Test cases
    malformed_tiles = [
        {},  # Empty dict
        {"phase": "P0"},  # Missing constraints
        {"constraints": ["c1"]},  # Missing phase
        {"phase": "P3", "constraints": []},  # Invalid phase
        {"phase": "P0", "constraints": "not_a_list"},  # Wrong type
        {"phase": "P1", "safe_channels": [1, 2, 3]},  # Wrong channel type
    ]
    
    for tile in malformed_tiles:
        with pytest.raises((ValidationError, ValueError)) as exc_info:
            room.submit_tile(agent_id, tile)
        
        # Ensure agent state unchanged
        assert room.get_agent_phase(agent_id) == "P0"  # Never left initial
    
    # Valid tile should still work after errors
    valid_result = room.submit_tile(agent_id, {
        "phase": "P0",
        "constraints": ["valid_constraint"]
    })
    assert valid_result.get("status") == "accepted"
```
