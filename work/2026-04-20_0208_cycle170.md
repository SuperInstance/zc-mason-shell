### Test 4: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test thread-safe tile submission from multiple agents."""
    import threading
    
    room = DeadbandRoom(
        priority_sequence=["P0", "P1", "P2"],
        channels=["test_channel"],
        max_tiles=5000
    )
    
    tile_count = 1000
    errors = []
    
    def submit_tiles(agent_id):
        for i in range(tile_count // 4):  # 250 tiles per thread
            tile = {
                "agent_id": agent_id,
                "content": f"Tile {i} from {agent_id}",
                "timestamp": "2026-04-20T02:10:00Z"
            }
            try:
                room.submit_tile(tile)
            except Exception as e:
                errors.append(e)
    
    # Create 4 concurrent agents
    threads = []
    for agent_id in ["agent_a", "agent_b", "agent_c", "agent_d"]:
        thread = threading.Thread(target=submit_tiles, args=(agent_id,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify no data corruption
    self.assertEqual(len(errors), 0, f"Errors during concurrent submission: {errors}")
    self.assertEqual(room.tile_count, tile_count)
    
    # Verify all agent tiles are present
    agent_counts = {}
    for tile in room.get_recent_tiles(tile_count):
        agent_id = tile.get("agent_id")
        agent_counts[agent_id] = agent_counts.get(agent_id, 0) + 1
    
    for agent_id in ["agent_a", "agent_b", "agent_c", "agent_d"]:
        self.assertEqual(agent_counts.get(agent_id, 0), tile_count // 4)
```

### Test 5: Storage Limit Enforcement
```python
def test_max_tiles_enforcement(self):
    """Test that room enforces maximum tile storage limit."""
    max_tiles = 100
    room = DeadbandRoom(
        priority_sequence=["P0", "P1", "P2"],
        channels=["test_channel"],
        max_tiles=max_tiles
    )
    
    # Submit exactly max_tiles
    for i in range(max_tiles):
        tile = {
            "agent_id": "test_agent",
            "content": f"Tile {i}",
            "timestamp": "2026-04-20T02:10:00Z"
        }
        room.submit_tile(tile)
    
    self.assertEqual(room.tile_count, max_tiles)
    
    # Submit one more tile - should trigger cleanup
    extra_tile = {
        "agent_id": "test_agent",
        "content": "Extra tile",
        "timestamp": "2026-04-20T02:11:00Z"
    }
    room.submit_tile(extra_tile)
    
    # Should still be at max_tiles (oldest tile removed)
    self.assertEqual(room.tile_count, max_tiles)
    
    # Verify oldest tile was removed
    recent_tiles = room.get_recent_tiles(max_tiles)
    tile_contents = [t["content"] for t in recent_tiles]
    self.assertNotIn("Tile 0", tile_contents)  # Oldest tile removed
    self.assertIn("Extra tile", tile_contents)  # Newest tile present
```
