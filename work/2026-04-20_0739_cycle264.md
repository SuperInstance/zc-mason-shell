### Test 1: Empty Tile Content
**Scenario:** Agent submits a tile with valid priority structure but empty content field.
**Expected:** Tile should be rejected with appropriate error message.
**Rationale:** Ensures content validation prevents empty training data.

```python
def test_submit_tile_empty_content(self):
    """Test tile submission with empty content field."""
    room = DeadbandRoom()
    tile = {
        "tile_id": "empty_content_001",
        "priorities": {
            "p0": "Avoid null pointers",
            "p1": "Use optional types",
            "p2": "Implement early returns"
        },
        "content": ""  # Empty content

    assert "content" in result["message"].lower()
```

### Test 3: Concurrent Tile Submission
**Scenario:** Multiple agents attempt to submit tiles with the same tile_id simultaneously.
**Expected:** Only first submission succeeds, subsequent submissions fail.
**Rationale:** Tests thread safety and duplicate detection under concurrency.

```python
def test_concurrent_duplicate_submission(self):
    """Test concurrent submission of tiles with same ID."""
    room = DeadbandRoom()
    tile_template = {
        "tile_id": "concurrent_test_001",
        "priorities": {
            "p0": "Test P0",
            "p1": "Test P1",
            "p2": "Test P2"
        },
        "content": "Test content"
    }
    
    # Simulate concurrent submissions
    import threading
    
    results = []
    def submit():
        results.append(room.submit_tile(tile_template.copy()))
    
    threads = [threading.Thread(target=submit) for _ in range(5)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Only one should be accepted
    accepted_count = sum(1 for r in results if r["accepted"])
    assert accepted_count == 1
```

### Test 5: Tile Submission After Room Reset
**Scenario:** Agent submits tile, room is reset, same tile_id is submitted again.
**Expected:** Second submission should succeed (fresh state).
**Rationale:** Tests state isolation and reset functionality.

```python
def test_tile_submission_after_reset(self):
    """Test that tile submission works after room reset."""
    room = DeadbandRoom()
    
    # First submission
    tile = {
        "tile_id": "reset_test_001",
        "priorities": {
            "p0": "Initial P0",
            "p1": "Initial P1",
            "p2": "Initial P2"
        },
        "content": "Initial content"
    }
    result1 = room.submit_tile(tile)
    assert result1["accepted"]
    
    # Reset room
    room.reset()
    
    # Same tile_id, different content
    tile["content"] = "New content after reset"
    result2 = room.submit_tile(tile)
    assert result2["accepted"], "Should accept after reset"
    assert result2["tile"]["content"] == "New content after reset"
```
