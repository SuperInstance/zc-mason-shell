### 4. Test: Concurrent Tile Submission Race Condition
**Scenario:** Multiple agents submit tiles simultaneously
**Expected Behavior:** Room should handle concurrent submissions without data corruption
**Rationale:** Tests thread safety in simulation environments
```python
def test_deadband_room_concurrent_submission(self):
    """Test concurrent tile submission from multiple agents."""
    room = DeadbandRoom(
        room_id="test_concurrent",
        agent_ids=["agent1", "agent2", "agent3"],
        protocol_steps=["P0"],
        max_cycles=100
    )
    room.initialize()
    
    import threading
    
    results = []
    lock = threading.Lock()
    
    def submit_tile(agent_id):
        tile = {"agent": agent_id, "data": "test"}
        success = room.submit_tile(agent_id=agent_id, tile_data=tile)
        with lock:
            results.append((agent_id, success))
    
    threads = []
    for agent_id in room.agent_ids:
        thread = threading.Thread(target=submit_tile, args=(agent_id,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # All submissions should succeed
    self.assertEqual(len(results), 3)
    self.assertTrue(all(success for _, success in results))
```

### 5. Test: Invalid Tile Data Types
**Scenario:** Agent submits tile with unsupported data types (None, integer, malformed JSON)
**Expected Behavior:** Room should validate tile data and reject invalid types
**Rationale:** Tests input validation and error messaging
```python
def test_deadband_room_invalid_tile_data(self):
    """Test submission of invalid tile data types."""
    room = DeadbandRoom(
        room_id="test_invalid_data",
        agent_ids=["agent1"],
        protocol_steps=["P0"],
        max_cycles=100
    )
    room.initialize()
    
    test_cases = [
        (None, "None value"),
        (123, "Integer instead of dict"),
        ("string", "String instead of dict"),
        ([], "Empty list"),
        ({"invalid": object()}, "Non-serializable object"),
    ]
    
    for invalid_data, description in test_cases:
        with self.subTest(description=description):
            with self.assertRaises((TypeError, ValueError)):
                room.submit_tile(agent_id="agent1", tile_data=invalid_data)
```
