### Test 1: Concurrent Tile Submission
**Scenario:** Two agents submit tiles simultaneously to the same room.
**Edge:** Race condition in tile queue handling.
```python
def test_concurrent_tile_submission(self):
    """DeadbandRoom should handle simultaneous tile submissions without data loss."""
    room = DeadbandRoom(room_id="test_concurrent")
    agent_a = MockAgent(agent_id="agent_a")
    agent_b = MockAgent(agent_id="agent_b")
    
    tile_a = {"priority": "P0", "content": "Avoid rocks", "agent": "agent_a"}
    tile_b = {"priority": "P1", "content": "Safe channel found", "agent": "agent_b"}
    
    # Simulate concurrent submission
    import threading
    results = []
    def submit(tile, agent):
        results.append(room.submit_tile(tile, agent))
    
    thread_a = threading.Thread(target=submit, args=(tile_a, agent_a))
    thread_b = threading.Thread(target=submit, args=(tile_b, agent_b))
    thread_a.start()
    thread_b.start()
    thread_a.join()
    thread_b.join()
    
    # Both tiles should be accepted
    assert len(results) == 2
    assert all(r["status"] == "accepted" for r in results)
    assert len(room.tiles) == 2
```

### Test 3: Protocol Violation with Correction
**Scenario:** Agent tries to submit P2 tile before P1, then corrects.
**Edge:** Room should enforce strict priority but allow correction path.
```python
def test_protocol_violation_correction(self):
    """DeadbandRoom should reject out-of-order priorities but accept corrected sequence."""
    room = DeadbandRoom(room_id="test_violation")
    agent = MockAgent(agent_id="test_agent")
    
    # Try to skip P1 (violation)
    tile_p2 = {"priority": "P2", "content": "Optimizing", "agent": "test_agent"}
    response1 = room.submit_tile(tile_p2, agent)
    assert response1["status"] == "rejected"
    assert "out of order" in response1["error"].lower()
    
    # Submit correct P0
    tile_p0 = {"priority": "P0", "content": "Negative space mapped", "agent": "test_agent"}
    response2 = room.submit_tile(tile_p0, agent)
    assert response2["status"] == "accepted"
    
    # Now P1 should work
    tile_p1 = {"priority": "P1", "content": "Safe channel found", "agent": "test_agent"}
    response3 = room.submit_tile(tile_p1, agent)
    assert response3["status"] == "accepted"
    
    # Now P2 should work
    response4 = room.submit_tile(tile_p2, agent)
    assert response4["status"] == "accepted"
```

### Test 5: Network Failure Simulation
**Scenario:** Tile submission succeeds but acknowledgment fails.
**Edge:** Room should maintain consistency despite partial failure.
```python
def test_partial_network_failure(self):
    """DeadbandRoom should handle cases where tile storage succeeds but acknowledgment fails."""
    room = DeadbandRoom(room_id="test_network")
    agent = MockAgent(agent_id="test_agent")
    
    # Simulate storage success but network failure on response
    original_store = room._store_tile
    network_failed = False
    
    def failing_store(tile, agent):
        # Store succeeds
        original_store(tile, agent)
        # But simulate network failure
        nonlocal network_failed
        if not network_failed:
            network_failed = True
            raise ConnectionError("Network timeout")
    
    room._store_tile = failing_store
    
    tile = {"priority": "P0", "content": "Test tile", "agent": "test_agent"}
    
    # First submission: network failure
    try:
        response = room.submit_tile(tile, agent)
    except ConnectionError:
        # Tile was stored but agent didn't get acknowledgment
        pass
    
    # Tile should still be in room storage
    assert len(room.tiles) == 1
    assert room.tiles[0]["content"] == "Test tile"
    
    # Agent retries (idempotency check)
    response = room.submit_tile(tile, agent)
    # Should either reject as duplicate or accept idempotently
    assert response["status"] in ["accepted", "rejected"]
    if response["status"] == "rejected":
        assert "duplicate" in response["error"].lower()
```
