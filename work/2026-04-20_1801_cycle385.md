```python
# File: plato-torch/tests/test_deadband_room_edge.py
import pytest
import json
import copy
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.tile import Tile

class TestDeadbandRoomEdgeCases:
    """Edge-case tests for DeadbandRoom beyond core functionality."""

    def test_tile_with_extreme_sentiment(self):
        """P0: Adding a tile with sentiment value outside [-1, 1] should not crash."""
        room = DeadbandRoom(room_id="edge_sentiment", max_tiles=100)
        # Sentiment values beyond typical range
        extreme_tile = Tile(
            tile_id="extreme_1",
            content="Sentiment = 5.0",
            metadata={"sentiment": 5.0, "phase": "P0"}
        )
        # Room should handle without exception
        room.add_tile(extreme_tile)
        stats = room.get_statistics()
        assert stats["tile_count"] == 1
        # Verify sentiment is stored as provided
        stored = room._tiles["extreme_1"]
        assert stored.metadata["sentiment"] == 5.0

    def test_rapid_sequential_tile_addition(self):
        """P1: Adding many tiles rapidly should maintain consistency."""
        room = DeadbandRoom(room_id="rapid_add", max_tiles=50)
        tiles = []
        for i in range(30):
            tile = Tile(
                tile_id=f"rapid_{i}",
                content=f"Content {i}",
                metadata={"cycle": i, "phase": "P1"}
            )
            tiles.append(tile)
        # Add all tiles in quick succession (simulating concurrent-like behavior)
        for tile in tiles:
            room.add_tile(tile)
        stats = room.get_statistics()
        assert stats["tile_count"] == 30
        # Verify all tiles are present and accessible
        for i in range(30):
            assert f"rapid_{i}" in room._tiles

    def test_train_with_single_tile(self):
        """P2: Training with exactly one tile should not error."""
        room = DeadbandRoom(room_id="single_tile_train", max_tiles=10)
        tile = Tile(
            tile_id="sole_tile",
            content="Only one tile",
            metadata={"sentiment": 0.8, "phase": "P2"}
        )
        room.add_tile(tile)
        # Training with minimal data should complete
        result = room.train_ensign(epochs=1)
        assert result["status"] == "trained"
        assert result["tiles_used"] == 1
        # Ensign should be exportable
        ensign = room.export_ensign()
        assert ensign is not None
        assert "adapters" in ensign

    def test_export_after_partial_training_interruption(self):
        """P0: Export after interrupted training (simulated) should handle gracefully."""
        room = DeadbandRoom(room_id="partial_train", max_tiles=20)
        # Add some tiles
        for i in range(5):
            tile = Tile(
                tile_id=f"partial_{i}",
                content=f"Tile {i}",
                metadata={"phase": "P0"}
            )
            room.add_tile(tile)
        # Start training but simulate interruption (e.g., exception caught)
        try:
            # Force an interruption by mocking an error after one epoch
            # For this test, we'll just train one epoch and stop
            room.train_ensign(epochs=1)
        except Exception:
            pass  # Simulated interruption
        # Export should still work, possibly with a "partially_trained" state
        ensign = room.export_ensign()
        # The room should not crash and should return something
        assert ensign is not None
        # It may be an empty dict or contain partial adapters
        assert isinstance(ensign, dict)

    def test_room_serialization_roundtrip(self):
        """P1: Room state should survive serialization/deserialization."""
        room = DeadbandRoom(room_id="serial_test", max_tiles=15)
        # Add a few tiles with varied metadata
        tiles = [
            Tile("t1", "Content A", {"phase": "P0", "score": 0.1}),
            Tile("t2", "Content B", {"phase": "P1", "score": 0.9}),
            Tile("t3", "Content C", {"phase": "P2", "score": -0.5}),
        ]
        for tile in tiles:
            room.add_tile(tile)
        # Train a little to have some state
        room.train_ensign(epochs=2)
        # Capture full state
        state_before = room.get_statistics()
        # Serialize (simulate saving to disk)
        serialized = json.dumps(room.to_dict())
        # Deserialize into a new room instance
        room_data = json.loads(serialized)
        new_room = DeadbandRoom.from_dict(room_data)
        # Compare key properties
        state_after = new_room.get_statistics()
        assert state_before["tile_count"] == state_after["tile_count"]
        assert state_before["last_trained"] == state_after["last_trained"]
        # Verify tiles are present
        for tile_id in ["t1", "t2", "t3"]:
            assert tile_id in new_room._tiles
            # Content should match
            assert new_room._tiles[tile_id].content == room._tiles[tile_id].content
```

## Rationale for Edge Cases

1. **Extreme sentiment**: Real-world data may have outliers; the room should not crash.
2. **Rapid sequential addition**: Simulates high-load scenarios without full concurrency logic.
3. **Single tile training**: Tests the lower bound of training data requirements.
4. **Partial training interruption**: Ensures robustness when training is interrupted (e.g., server restart).
5. **Serialization roundtrip**: Critical for persistence across sessions; state must be preserved.

These tests align with the Deadband Protocol:
- **P0**: Avoid crashes (tests 1, 4).
- **P1**: Ensure safe operation under edge conditions (tests 2, 5).
- **P2**: Optimize for minimal viable training (test 3).
