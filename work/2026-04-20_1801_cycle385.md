### Test 1: Concurrent Agent Tile Submission
**Scenario:** Multiple agents submit tiles simultaneously during P1 channel finding.
**Edge Case:** Race condition in channel occupancy tracking.
```python
def test_concurrent_p1_tile_submission(self):
    """Test that channel occupancy is correctly tracked with concurrent submissions."""
    room = DeadbandRoom()
    room.state = "channel_finding"
    
    # Simulate 3 agents finding channels simultaneously
    tiles = [
        {"agent": "agent1", "phase": "P1", "channel": "channel_a", "content": "test"},
        {"agent": "agent2", "phase": "P1", "channel": "channel_b", "content": "test"},
        {"agent": "agent3", "phase": "P1", "channel": "channel_a", "content": "test"}  # Duplicate channel
    ]
    
    results = []
    for tile in tiles:
        results.append(room.submit_tile(tile))
    
    # Channel_a should only accept first submission
    assert results[0] == True  # agent1 accepted
    assert results[1] == True  # agent2 accepted
    assert results[2] == False  # agent3 rejected (channel occupied)
    
    assert len(room.occupied_channels) == 2
    assert "channel_a" in room.occupied_channels
    assert "channel_b" in room.occupied_channels
```

### Test 3: Network Partition During State Transition
**Scenario:** Network failure occurs between P0 completion and P1 start.
**Edge Case:** Room left in inconsistent state requiring recovery.
```python
def test_network_partition_mid_transition(self):
    """Test recovery from failed state transition due to network issues."""
    room = DeadbandRoom()
    room.state = "mapping"
    
    # Simulate successful P0 completion
    p0_tile = {
        "agent": "agent1",
        "phase": "P0",
        "negative_space": ["invalid_json", "timeout_error"],
        "content": "P0 mapping complete"
    }
    assert room.submit_tile(p0_tile) == True
    
    # Simulate network partition before P1 starts
    room._simulate_network_failure = True
    
    # Attempt P1 transition (should fail gracefully)
    try:
        room.transition_to_p1()
        transition_ok = True
    except ConnectionError:
        transition_ok = False
    
    assert transition_ok == False
    assert room.state == "mapping"  # Should remain in previous state
    assert hasattr(room, 'recovery_required') == True
```
