def test_deadband_room_extreme_parameters():
    """Test room initialization with extreme parameter values."""
    # Zero values should be rejected
    with pytest.raises(ValueError):
        DeadbandRoom(p0_threshold=0, p1_range=10, p2_optimization="min")
    
    # Negative values should be rejected
    with pytest.raises(ValueError):
        DeadbandRoom(p0_threshold=-5, p1_range=10, p2_optimization="min")
    
    # Very large values should be accepted but may have performance implications
    room = DeadbandRoom(p0_threshold=1e6, p1_range=1e6, p2_optimization="min")
    assert room.p0_threshold == 1e6
    assert room.p1_range == 1e6
    
    # Float precision edge case
    room = DeadbandRoom(p0_threshold=0.000001, p1_range=0.000001, p2_optimization="min")
    assert room.p0_threshold == 0.000001
```

### Test 2: Concurrent Agent Interactions

def test_concurrent_tile_submissions():
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(p0_threshold=10, p1_range=20, p2_optimization="min")
    
    # Simulate concurrent submissions
    agent_ids = [f"agent_{i}" for i in range(5)]
    tiles = []
    
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results if isinstance(r, dict))
    
    # Verify room state reflects all agents
    assert len(room.active_agents) == len(agent_ids)
```

def test_malformed_tile_validation():
    """Test rejection of tiles with invalid structure."""
    room = DeadbandRoom(p0_threshold=10, p1_range=20, p2_optimization="min")
    
    test_cases = [
        # Missing required fields
        {"agent_id": "test_agent"},  # missing phase, content
        {"phase": "P0", "content": "test"},  # missing agent_id
        
        # Wrong field types
        {"agent_id": 123, "phase": "P0", "content": "test"},  # agent_id not string
        {"agent_id": "test", "phase": 123, "content": "test"},  # phase not string
        
        # Invalid phase values
        {"agent_id": "test", "phase": "P3", "content": "test"},  # P3 doesn't exist
        {"agent_id": "test", "phase": "", "content": "test"},  # empty phase
        
        # Content too large
        {"agent_id": "test", "phase": "P0", "content": "x" * 10001},  # exceeds limit
        
        # Invalid timestamp
        {"agent_id": "test", "phase": "P0", "content": "test", "timestamp": "not-a-number"},
    ]
    
    for tile in test_cases:
        with pytest.raises((ValueError, TypeError)):
            room.submit_tile(tile)
```

def test_protocol_state_transitions():
    """Test edge cases in protocol state transitions."""
    room = DeadbandRoom(p0_threshold=10, p1_range=20, p2_optimization="min")
    agent_id = "edge_case_agent"
    
    # Test: Agent tries to submit P1 without completing P0
    p1_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Trying to skip P0",
        "timestamp": time.time()
    }
    
    with pytest.raises(ProtocolViolationError, match="P0 not completed"):
        room.submit_tile(p1_tile)
    
    # Test: Agent completes P0, then submits another P0
    p0_tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Valid P0 tile",
        "timestamp": time.time()
    }
    
    room.submit_tile(p0_tile)  # First P0 should succeed
    
    with pytest.raises(ProtocolViolationError, match="P0 already completed"):
        room.submit_tile(p0_tile)  # Second P0 should fail
    
    # Test: Agent completes all phases, then tries to go back
    p1_tile["content"] = "Valid P1 after P0"
    room.submit_tile(p1_tile)
    
    p2_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Valid P2",
        "timestamp": time.time()
    }
    room.submit_tile(p2_tile)
    
    # After completing P2, should not accept more tiles for this agent
    with pytest.raises(ProtocolViolationError, match="Protocol completed"):
        room.submit_tile(p0_tile)
```
