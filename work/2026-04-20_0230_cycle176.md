### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent submissions using threads
    def submit_tile(agent_id):
        room.submit_tile(agent_id, "P0", {"rock": "avoid_this"})
        room.submit_tile(agent_id, "P1", {"channel": "safe_path"})
        room.submit_tile(agent_id, "P2", {"optimization": "best_path"})
    
    threads = [threading.Thread(target=submit_tile, args=(aid,)) for aid in agent_ids]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all agents completed protocol without state corruption
    for aid in agent_ids:
        assert room.get_agent_state(aid) == "COMPLETED"
    assert len(room.tiles) == 9  # 3 agents × 3 tiles each
```

### Test 2: P0 Violation Recovery
```python
def test_p0_violation_recovery(self):
    """Agent submitting P2 before P0 should be reset to P0 with warning."""
    room = DeadbandRoom()
    agent_id = "reckless_agent"
    
    # Attempt to skip to P2 (violation)
    result = room.submit_tile(agent_id, "P2", {"optimization": "premature"})
    assert result["status"] == "REJECTED"
    assert "P0 required first" in result["message"]
    
    # Verify agent state is reset to require P0
    assert room.get_agent_state(agent_id) == "AWAITING_P0"
    
    # Now follow correct protocol
    room.submit_tile(agent_id, "P0", {"rock": "avoid"})
    room.submit_tile(agent_id, "P1", {"channel": "safe"})
    room.submit_tile(agent_id, "P2", {"optimization": "valid"})
    assert room.get_agent_state(agent_id) == "COMPLETED"
```

### Test 3: Channel Exhaustion Scenario
```python
def test_channel_exhaustion(self):
    """When all P1 channels become invalid, agent should report dead-end."""
    room = DeadbandRoom()
    agent_id = "explorer_agent"
    
    # Submit P0 that maps all channels as negative space
    room.submit_tile(agent_id, "P0", {
        "rock_1": "entire_east_path",
        "rock_2": "entire_west_path", 
        "rock_3": "central_corridor"
    })
    
    # Attempt P1 - should return empty channels
    result = room.submit_tile(agent_id, "P1", {"channel": "anywhere"})
    assert result["status"] == "BLOCKED"
    assert "no safe channels" in result["message"]
    assert room.get_agent_state(agent_id) == "STUCK_AT_P1"
```

### Test 4: Ghost Tile Integration
```python
def test_ghost_tile_integration(self):
    """Dead agents' P0 knowledge should be automatically applied to new agents."""
    room = DeadbandRoom()
    
    # Simulate dead agent leaving ghost tile
    room.add_ghost_tile("dead_agent_42", "P0", {"hidden_rock": "unmarked_hazard"})
    
    # New agent starts protocol
    agent_id = "new_agent"
    room.submit_tile(agent_id, "P0", {"visible_rock": "marked_hazard"})
    
    # Verify ghost P0 is integrated into agent's knowledge
    state = room.get_agent_state(agent_id)
    assert "hidden_rock" in state["p0_knowledge"]
    assert "visible_rock" in state["p0_knowledge"]
    
    # P1 should avoid both hazards
    result = room.submit_tile(agent_id, "P1", {"channel": "safe_path"})
    assert "avoiding hidden_rock" in result["path_notes"]
    assert "avoiding visible_rock" in result["path_notes"]
```

### Test 5: Multi-Cycle State Persistence
```python
def test_multi_cycle_state_persistence(self):
    """Room should maintain agent states across training cycles."""
    room = DeadbandRoom()
    agent_id = "persistent_agent"
    
    # Cycle 1: Start protocol
    room.submit_tile(agent_id, "P0", {"rock": "first_hazard"})
    room.submit_tile(agent_id, "P1", {"channel": "first_safe_path"})
    
    # Simulate cycle end and restart
    room.end_training_cycle()
    room.start_training_cycle()
    
    # Agent should resume at P2 (where they left off)
    assert room.get_agent_state(agent_id) == "AWAITING_P2"
    
    # Complete protocol
    room.submit_tile(agent_id, "P2", {"optimization": "continued_path"})
    assert room.get_agent_state(agent_id) == "COMPLETED"
    
    # Verify all tiles from both cycles are preserved
    assert len([t for t in room.tiles if t["agent_id"] == agent_id]) == 3
```
