### Test 1: Concurrent Tile Submission Race Condition

```python
def test_concurrent_tile_submission_race():
    """Test that room handles simultaneous tile submissions without corruption."""
    room = DeadbandRoom()
    tiles = [
        Tile(agent_id="agent1", content="P0: Avoid null input", phase=0),
        Tile(agent_id="agent2", content="P1: Find valid channel", phase=1),
        Tile(agent_id="agent3", content="P2: Optimize path", phase=2)
    ]
    
    # Simulate concurrent submissions
    results = []
    for tile in tiles:
        # In real implementation, this would use threading
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all tiles were processed
    assert len(results) == 3
    assert all(r['status'] == 'accepted' for r in results)
    
    # Verify room state consistency
    assert room.get_tile_count() == 3
    assert len(room.get_phase_tiles(0)) == 1
    assert len(room.get_phase_tiles(1)) == 1
    assert len(room.get_phase_tiles(2)) == 1
```

```python
def test_memory_exhaustion_handling():
    """Test room behavior when tile storage approaches limits."""
    room = DeadbandRoom(max_tiles=1000)
    
    # Submit tiles until near limit
    for i in range(999):
        tile = Tile(
            agent_id=f"agent_{i}",
            content=f"Test tile {i}",
            phase=i % 3
        )
        room.submit_tile(tile)
    
    # Verify room still accepts tiles at limit
    final_tile = Tile(
        agent_id="final_agent",
        content="Final test tile",
        phase=0

    assert result['status'] == 'accepted'
    assert room.get_tile_count() == 1000
    
    # Test rejection beyond limit
    overflow_tile = Tile(
        agent_id="overflow_agent",
        content="Should be rejected",
        phase=1
    )
    result = room.submit_tile(overflow_tile)

    assert 'exceeded' in result['reason'].lower()
```

### Test 4: State Persistence Across Room Reset
```python
def test_state_persistence_after_reset():
    """Test that critical room state persists after soft reset."""
    room = DeadbandRoom()
    
    # Populate room with tiles
    for i in range(5):
        tile = Tile(
            agent_id=f"pre_reset_agent_{i}",
            content=f"Pre-reset tile {i}",
            phase=i % 3
        )
        room.submit_tile(tile)
    
    # Capture pre-reset state
    pre_reset_tile_count = room.get_tile_count()
    pre_reset_phase_counts = {
        0: len(room.get_phase_tiles(0)),
        1: len(room.get_phase_tiles(1)),
        2: len(room.get_phase_tiles(2))
    }
    
    # Perform soft reset (simulating room restart)
    room.soft_reset()
    
    # Verify tile count persists
    assert room.get_tile_count() == pre_reset_tile_count
    
    # Verify phase distributions persist
    for phase in [0, 1, 2]:
        assert len(room.get_phase_tiles(phase)) == pre_reset_phase_counts[phase]
    
    # Verify new tiles can be added after reset
    post_reset_tile = Tile(
        agent_id="post_reset_agent",
        content="Post-reset validation",
        phase=0
    )
    result = room.submit_tile(post_reset_tile)
    assert result['status'] == 'accepted'
    assert room.get_tile_count() == pre_reset_tile_count + 1
```

```python
def test_mixed_phase_priority_violation():
    """Test detection when agent submits tiles violating priority protocol."""
    room = DeadbandRoom()
    
    # Agent submits P0 tile
    tile_p0 = Tile(
        agent_id="mixed_phase_agent",
        content="P0: Identify constraint",
        phase=0
    )
    room.submit_tile(tile_p0)
    
    # Agent tries to submit P2 while P1 is incomplete (should be rejected)
    tile_p2_premature = Tile(
        agent_id="mixed_phase_agent",
        content="P2: Optimize prematurely",
        phase=2
    )
    result_p2 = room.submit_tile(tile_p2_premature)
    assert result_p2['status'] == 'rejected'
    assert 'priority' in result_p2['reason'].lower()
    
    # Another agent submits P1 (should be accepted)
    tile_p1_other = Tile(
        agent_id="other_agent",
        content="P1: Safe channel found",
        phase=1
    )
    result_p1 = room.submit_tile(tile_p1_other)
    assert result_p1['status'] == 'accepted'
    
    # Original agent now tries P2 again (should still be rejected - needs own P1)
    tile_p2_retry = Tile(
        agent_id="mixed_phase_agent",
        content="P2: Retry optimization",
        phase=2
    )
    result_p2_retry = room.submit_tile(tile_p2_retry)
    assert result_p2_retry['status'] == 'rejected'
    assert 'p1' in result_p2_retry['reason'].lower()
```
