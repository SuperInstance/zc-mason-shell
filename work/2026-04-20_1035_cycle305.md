```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions without corruption."""
    room = DeadbandRoom(room_id="race_test")
    
    # Simulate concurrent submissions using threads
    import threading
    results = []
    
    def submit_tile(priority, content):
        tile = {
            "tile_id": f"tile_{priority}_{threading.get_ident()}",
            "priority": priority,
            "content": content,
            "agent_id": f"agent_{priority}",
            "timestamp": time.time()
        }
        try:
            room.process_tile(tile)
            results.append(("success", priority))
        except Exception as e:
            results.append(("error", str(e)))
    
    threads = []
    for i in range(10):  # 10 concurrent submissions
        t = threading.Thread(target=submit_tile, args=(i % 3, f"content_{i}"))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify state consistency
    assert room.state["total_tiles"] == 10
    # Check that P0 tiles were processed first in the final state
    p0_count = sum(1 for r in results if r[1] == 0)
    assert p0_count > 0  # At least one P0 tile
    
    # State should be valid JSON serializable
    import json
    json.dumps(room.state)
```

```python
def test_duplicate_tile_id_handling(self):
    """Test idempotency when same tile ID is submitted multiple times."""
    room = DeadbandRoom(room_id="duplicate_test")
    
    tile = {
        "tile_id": "duplicate_id_123",
        "priority": 1,
        "content": "original content",
        "agent_id": "test_agent",
        "timestamp": time.time()
    }
    
    # First submission should succeed
    room.process_tile(tile)
    initial_count = room.state["total_tiles"]
    
    # Second submission with same ID
    tile2 = tile.copy()
    tile2["content"] = "modified content"  # Try to change content
    
    # Should either reject or accept as idempotent update
    try:
        room.process_tile(tile2)
        # If accepted, check if content was updated or ignored
        if room.state["total_tiles"] == initial_count + 1:
            # New tile counted - check content difference
            pass
        else:
            # Same count - idempotent behavior
            assert room.state["total_tiles"] == initial_count
    except ValueError as e:
        # Expected rejection due to duplicate ID
        assert "duplicate" in str(e).lower() or "already" in str(e).lower()
    
    # Third submission with identical everything
    room.process_tile(tile)
    # State should be consistent regardless of outcome
    assert room.state["total_tiles"] in [initial_count, initial_count + 1]
```
