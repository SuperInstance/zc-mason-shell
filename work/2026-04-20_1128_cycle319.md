```python
"""
Edge-case tests for DeadbandRoom.
Focus on P0 negative space: invalid inputs, boundary conditions, failure modes.
"""
import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.tiles import TrainingTile

def test_empty_constraint_dict():
    """P0: Room should reject empty constraint dictionary."""
    with pytest.raises(ValueError, match="must contain at least P0"):
        DeadbandRoom(constraints={})

def test_non_callable_p1_constraint():
    """P0: P1 constraint must be callable."""
    constraints = {
        "P0": lambda x: x is not None,
        "P1": "not a function",  # Invalid
        "P2": lambda x: x > 0
    }
    with pytest.raises(TypeError, match="P1 constraint must be callable"):
        DeadbandRoom(constraints=constraints)

def test_priority_inversion():
    """P0: P2 constraint that violates P0 should be detected."""
    constraints = {
        "P0": lambda x: x >= 0,      # Must be non-negative
        "P1": lambda x: x < 100,
        "P2": lambda x: x == -5      # Violates P0
    }
    room = DeadbandRoom(constraints=constraints)
    tile = TrainingTile(data={"value": -5}, metadata={"cycle": 1})
    # Tile should fail P0 validation
    assert not room._validate_constraint(tile, "P0")
    # Room should reject tile entirely
    assert not room.submit_tile(tile)

def test_corrupted_tile_metadata():
    """P0: Tile missing required metadata fields."""
    constraints = {
        "P0": lambda x: isinstance(x, dict),
        "P1": lambda x: "value" in x,
        "P2": lambda x: x.get("value", 0) > 10
    }
    room = DeadbandRoom(constraints=constraints)
    # Tile with missing 'data' field
    tile = TrainingTile(metadata={"cycle": 1})  # No data
    with pytest.raises(AttributeError):
        room.submit_tile(tile)

def test_duplicate_tile_submission():
    """P1: Same tile submitted twice should be idempotent or rejected."""
    constraints = {
        "P0": lambda x: True,
        "P1": lambda x: True,
        "P2": lambda x: True
    }
    room = DeadbandRoom(constraints=constraints)
    tile = TrainingTile(data={"test": 1}, metadata={"cycle": 1, "id": "abc"})
    # First submission succeeds
    assert room.submit_tile(tile) is True
    initial_count = len(room.tiles)
    # Second submission should be ignored or rejected
    room.submit_tile(tile)
    assert len(room.tiles) == initial_count  # No duplicate added
```
