### Test 4: Concurrent Tile Submission Race Condition
```rust
#[test]
fn test_concurrent_tile_submission() {
    let room = Arc::new(Mutex::new(DeadbandRoom::new(DeadbandConfig::default())));
    let mut handles = vec![];
    
    // Simulate 10 agents submitting simultaneously
    for agent_num in 0..10 {
        let room_clone = Arc::clone(&room);
        let handle = thread::spawn(move || {
            let agent_id = format!("agent_{}", agent_num);
            let tile = Tile {
                id: format!("tile_from_{}", agent_id),
                content: format!("Content from {}", agent_id),
                protocol_phase: ProtocolPhase::P1,
                timestamp: Utc::now(),
            };
            
            let mut room_lock = room_clone.lock().unwrap();
            room_lock.submit_tile(agent_id, tile)
        });
        handles.push(handle);
    }
    
    // Collect results
    let mut successes = 0;
    let mut errors = 0;
    for handle in handles {
        match handle.join().unwrap() {
            Ok(_) => successes += 1,
            Err(_) => errors += 1,
        }
    }
    
    // All submissions should complete without panic
    // Some may fail due to concurrency control (acceptable)
    assert_eq!(successes + errors, 10);
    
    // Room should be in valid state
    let room_lock = room.lock().unwrap();
    assert!(room_lock.get_status().is_healthy());
}
```

#[should_panic(expected = "out of memory")]  // Or should handle gracefully
fn test_training_memory_exhaustion() {
    let mut room = DeadbandRoom::new(DeadbandConfig {
        max_tiles: 1_000_000,  // Unreasonably large
        ..Default::default()
    });
    
    // Attempt to trigger training with excessive data
    // This tests how room handles system limits
    
    // Method 1: Fill with massive tiles
    for i in 0..10000 {
        let tile = Tile {
            id: format!("huge_tile_{}", i),
            content: "x".repeat(10000),  // 10KB per tile
            protocol_phase: ProtocolPhase::P1,
            timestamp: Utc::now(),
        };
        let _ = room.submit_tile(format!("agent_{}", i), tile);
        
        // Break if we detect memory protection
        if room.get_status().memory_usage > 0.9 {
            break;
        }
    }
    
    // Attempt training cycle
    let result = room.run_training_cycle();
    
    // Room should either:
    // 1. Gracefully handle with error message
    // 2. Panic with specific memory error (test expects this)
    // 3. Implement swap/throttling
    
    if result.is_err() {
        let error = result.unwrap_err();
        assert!(error.to_string().contains("memory") || 
                error.to_string().contains("resource"));
    }
}
