## 5 Additional Edge-Case Tests

### Test 1: Protocol Violation with Partial P1 Completion
**Scenario:** Agent attempts P2 optimization after only partial P1 completion (some safe channels found but not all constraints processed).
```python
def test_protocol_violation_partial_p1_completion(deadband_room):
    """Test that attempting P2 optimization with incomplete P1 raises error."""
    # Arrange: Run P0 successfully
    state = {"constraints": ["avoid_rock", "avoid_edge"]}
    p0_result = deadband_room.map_negative_space(state)
    
    # Act & Assert: Attempt P2 with only P0 completed (no P1)
    with pytest.raises(ProtocolViolationError) as exc_info:
        deadband_room.optimize_within_channels(state)
    assert "P1 not completed" in str(exc_info.value)
```

### Test 3: Empty Safe Channels List from P1
**Scenario:** P1 returns empty list of safe channels (no viable paths).
```python
def test_empty_safe_channels_from_p1(deadband_room):
    """Test P2 behavior when P1 returns no safe channels."""
    # Arrange: State that leads to no safe channels
    state = {
        "constraints": ["impossible_constraint_a", "impossible_constraint_b"],
        "negative_space": ["entire_space"]  # P0 mapped everything as negative
    }
    
    # Act: Run P1
    p1_result = deadband_room.find_safe_channels(state)
    
    # Assert: P1 should return empty channels list, not raise error
    assert p1_result.get("channels") == []
    
    # Act & Assert: P2 should raise NoViablePathError
    with pytest.raises(NoViablePathError) as exc_info:
        deadband_room.optimize_within_channels(state)
    assert "No safe channels" in str(exc_info.value)
```
