"""
Edge-case tests for DeadbandRoom.
These complement the core tests in test_deadband_room.py.

import pytest
import asyncio
from unittest.mock import patch, AsyncMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.models.ensign import LoRAAdapter

class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""

    @pytest.mark.asyncio
    async def test_concurrent_tile_addition_during_training(self):
        """
        Edge: Multiple agents attempt to add tiles while training is in progress.
        Expected: Room should handle concurrency, either via locking or
                  queueing, without dropping tiles or crashing.
        """
        room = DeadbandRoom(room_id="edge_concurrent", protocol_state="P1")
        
        # Simulate training in progress
        room.training_in_progress = True
        tile_queue_before = len(room.pending_tiles)
        
        # Simulate concurrent tile additions
        tile = {
            "agent_id": "test_agent",
            "content": "concurrent test tile",
            "sentiment": 0.5,
            "protocol_phase": "P1"
        }
        
        tasks = []
        for i in range(5):
            task = asyncio.create_task(room.add_tile(tile.copy()))
            tasks.append(task)
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Verify no exceptions occurred
        for r in results:
            assert not isinstance(r, Exception), f"Concurrent add raised: {r}"
        
        # Tiles should be queued or processed based on implementation
        # This test documents the current behavior for future optimization
        tile_queue_after = len(room.pending_tiles)
        assert tile_queue_after >= tile_queue_before, "Tiles should not be lost"

    def test_tile_with_extreme_sentiment_values(self):

            "sentiment": 10.0,  # Beyond normal range
            "protocol_phase": "P1"
        }
        
        # Depending on implementation, this may clamp or raise
        try:
            result = room.add_tile(extreme_tile)
            # If it succeeds, check that sentiment was clamped
            if hasattr(room, 'tiles') and room.tiles:
                stored_sentiment = room.tiles[-1].get('sentiment')
                assert -1.0 <= stored_sentiment <= 1.0, \
                    f"Sentiment should be clamped, got {stored_sentiment}"
        except ValueError as e:
            # Alternative valid behavior: rejection with error
            assert "sentiment" in str(e).lower(), \
                f"Error should mention sentiment: {e}"

        # Test extreme negative sentiment
        extreme_negative_tile = extreme_tile.copy()
        extreme_negative_tile["sentiment"] = -10.0

                stored_sentiment = room.tiles[-1].get('sentiment')
                assert -1.0 <= stored_sentiment <= 1.0
        except ValueError as e:
            assert "sentiment" in str(e).lower()

    def test_protocol_regression_attempt(self):
        """
        Edge: Agent attempts to submit P2 tile before P1 is complete.
        Expected: Room should reject or flag protocol violation according to

        """
        room = DeadbandRoom(room_id="edge_regression")
        room.protocol_state = "P0"  # Still mapping negative space
        
        # Attempt P2 tile while in P0 state
        p2_tile = {
            "agent_id": "eager_agent",
            "content": "Trying to optimize too early",
            "sentiment": 0.3,

        }
        
        # Should raise error or return False

            result = room.add_tile(p2_tile)
