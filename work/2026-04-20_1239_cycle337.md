```python
def test_p2_failure_falls_back_to_p1():

    room = DeadbandRoom(config={"max_retries": 3})

    assert room.current_state == "P1_SAFE_CHANNELS"
```

### Test 2: Concurrent Agent Tile Submission
```python
def test_concurrent_tile_submission():
    """Multiple agents should be able to submit tiles without corruption."""
    room = DeadbandRoom(config={"max_concurrent": 5})
    
    import threading
    results = []
    
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": {"negative_space": f"test_{agent_id}"}
        }
        result = room.process_tile(tile)
        results.append((agent_id, result["status"]))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All submissions should succeed
    statuses = [status for _, status in results]
    assert all(s == "ACCEPTED" for s in statuses)
    assert len(set([agent for agent, _ in results])) == 5  # All agents unique
```

### Test 3: Malformed Tile Data Handling
```python
def test_malformed_tile_rejection():
    """Room should reject tiles with invalid structure and provide clear error."""
    room = DeadbandRoom()
    
    test_cases = [
        {"agent_id": "test"},  # Missing phase and content
        {"phase": "P0", "content": {}},  # Missing agent_id
        "not_a_dict",  # Not even a dict
        {"agent_id": "test", "phase": "INVALID_PHASE", "content": {}},  # Invalid phase
        {"agent_id": "test", "phase": "P0", "content": None},  # Null content
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.process_tile(tile)
        assert result["status"] == "REJECTED"
        assert "error" in result
        assert "validation" in result["error"].lower() or "invalid" in result["error"].lower()
```

### Test 4: Resource Limit Enforcement
```python
def test_tile_storage_limits():
    """Room should enforce storage limits and handle overflow gracefully."""
    room = DeadbandRoom(config={"max_tiles": 3, "storage_path": "/tmp/test_tiles"})
    
    # Fill to capacity
    for i in range(3):
        room.process_tile({
            "agent_id": f"agent_{i}",
            "phase": "P0",
            "content": {"negative_space": f"test_{i}"}
        })
    
    # Fourth tile should trigger cleanup or rejection
    result = room.process_tile({
        "agent_id": "agent_4",
        "phase": "P0",
        "content": {"negative_space": "test_4"}
    })
    
    # Either old tiles are archived or new tile is rejected
    assert result["status"] in ["ACCEPTED_WITH_CLEANUP", "REJECTED_LIMIT_EXCEEDED"]
    
    if result["status"] == "ACCEPTED_WITH_CLEANUP":
        assert "archived" in result
        assert len(room.get_stored_tiles()) <= 3
    else:
        assert "limit" in result["message"].lower()
```

```python
def test_session_expiry_with_state_preservation():
    """Expired sessions should preserve P0 knowledge but require P1 re-evaluation."""
    room = DeadbandRoom(config={"session_timeout": 60})
    
    # Establish session with P0 knowledge
    room.process_tile({
        "agent_id": "session_agent",
        "phase": "P0",
        "content": {"negative_space": ["rock1", "rock2"]},
        "session_id": "test_session"
    })
    
    # Simulate session expiry
    import time
    room.sessions["test_session"]["last_active"] = time.time() - 120  # 2 minutes ago
    
    # New tile from same agent should trigger session renewal
    result = room.process_tile({
        "agent_id": "session_agent",
        "phase": "P1",
        "content": {"safe_channels": ["channel1"]},
        "session_id": "test_session"
    })
    
    assert result["status"] == "SESSION_RENEWED"
    assert "P0_knowledge_preserved" in result
    assert room.sessions["test_session"]["last_active"] > time.time() - 10
```
