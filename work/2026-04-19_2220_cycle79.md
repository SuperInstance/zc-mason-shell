### Test 2: Ghost Tile Persistence
```python
def test_ghost_tile_retrieval():
    """Test that P0 tiles from terminated agents persist as ghost tiles."""
    room = DeadbandRoom()
    # Agent submits P0 tile then terminates
    room.submit_tile(agent_id="ghost_agent", tile_data={"phase": "P0", "content": "Don't crash"})
    room.terminate_agent("ghost_agent")
    
    # New agent should see ghost tile
    ghost_tiles = room.get_ghost_tiles()
    assert len(ghost_tiles) == 1
    assert ghost_tiles[0]["content"] == "Don't crash"
    assert ghost_tiles[0]["source"] == "ghost_agent"
```

    """Test room handles malformed tiles without crashing."""
    room = DeadbandRoom()

    # Empty content
    result = room.submit_tile(agent_id="test", tile_data={"phase": "P0", "content": ""})
    assert result["status"] == "rejected"
    
    # Room should still be functional after rejections
    valid_result = room.submit_tile(agent_id="test", tile_data={"phase": "P0", "content": "Valid"})
    assert valid_result["status"] == "accepted"

def test_reward_boundary_conditions():
    """Test reward calculation at boundary conditions."""
    room = DeadbandRoom()
    
    # Test minimum reward for P0 compliance
    for _ in range(10):
        room.submit_tile(agent_id="agent", tile_data={"phase": "P0", "content": f"Avoid {_}"})

    assert score > 0  # Should have some reward

    # Transition to P1 and get higher rewards
    room.submit_tile(agent_id="agent", tile_data={"phase": "P1", "content": "Channel X"})

    assert p1_score > score  # P1 should give higher reward
    
    # Test reward ceiling
    for _ in range(100):  # Flood with high-quality P2 tiles

            "content": f"Optimize {_}",

        })
    
    final_score = room.get_agent_score("agent")
    assert final_score <= 100.0  # Should have reasonable ceiling
