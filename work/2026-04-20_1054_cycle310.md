### 1. Test Concurrent Tile Submission
```python
def test_concurrent_tile_submission(deadband_room, sample_agent):
    """P0: Ensure room handles simultaneous submissions without corruption."""
    import threading
    from queue import Queue
    
    results = Queue()
    def submit_tile(agent_id):
        tile = {"agent_id": agent_id, "phase": "P1", "content": f"tile from {agent_id}"}
        success, message = deadband_room.submit_tile(tile)
        results.put((agent_id, success, message))
    
    # Create multiple agent threads
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions processed
    successful = 0
    while not results.empty():
        agent_id, success, message = results.get()
        if success:
            successful += 1
        else:
            # Should fail gracefully, not crash
            assert "capacity" in message or "invalid" in message
    
    # At least some should succeed (capacity permitting)
    assert successful > 0
    assert deadband_room.tile_count == successful
```

### 2. Test Network Partition During Submission

```python
def test_network_partition_mid_submission(deadband_room, sample_agent, mocker):
    """P0: Simulate tile server becoming unreachable during submission."""
    # Mock tile server to fail after first call
    call_count = 0
    def mock_submit(tile):
        nonlocal call_count
        call_count += 1
        if call_count == 2:
            raise ConnectionError("Network partition")
        return {"status": "accepted"}
    
    mocker.patch.object(deadband_room.tile_server, 'submit', side_effect=mock_submit)
    
    # First submission should work
    tile1 = {"agent_id": "agent1", "phase": "P0", "content": "initial tile"}
    success1, msg1 = deadband_room.submit_tile(tile1)
    assert success1 is True
    
    # Second should handle error gracefully
    tile2 = {"agent_id": "agent1", "phase": "P1", "content": "follow-up tile"}
    success2, msg2 = deadband_room.submit_tile(tile2)
    
    assert success2 is False
    assert "ConnectionError" in msg2 or "network" in msg2.lower()
    assert deadband_room.get_agent_state("agent1") == "ERROR"
```

```python
def test_agent_state_persistence(deadband_room, sample_agent, tmp_path):
    """P1: Agent mid-protocol state survives room serialization/deserialization."""
    # Put agent in mid-P1 state
    deadband_room.register_agent("agent_x", sample_agent)
    deadband_room.update_agent_protocol("agent_x", current_phase="P1", channel="safe_channel_3")
    
    # Serialize room state
    state_file = tmp_path / "room_state.json"
    deadband_room.save_state(str(state_file))
    
    # Create new room instance and load state
    from plato_torch.rooms.deadband_room import DeadbandRoom
    new_room = DeadbandRoom(capacity=10)
    new_room.load_state(str(state_file))
    
    # Verify agent state preserved
    agent_state = new_room.get_agent_state("agent_x")
    assert agent_state["current_phase"] == "P1"
    assert agent_state["current_channel"] == "safe_channel_3"
    assert agent_state["status"] == "ACTIVE"
    
    # Agent should be able to continue protocol
    tile = {"agent_id": "agent_x", "phase": "P1", "content": "continuing after restart"}
    success, _ = new_room.submit_tile(tile)
    assert success is True
```

```python
def test_repeated_p2_violation_escalation(deadband_room, sample_agent):
    """P0: Agent skipping to P2 repeatedly triggers escalating enforcement."""
    deadband_room.register_agent("violator", sample_agent)
    
    violations = []
    for i in range(4):
        tile = {
            "agent_id": "violator",
            "phase": "P2",  # Skipping P0/P1
            "content": f"direct optimization {i}",
            "justification": "I know better"  # Invalid without P0/P1
        }
        success, message = deadband_room.submit_tile(tile)
        violations.append((success, message))
    
    # First violation: warning
    assert violations[0][0] is False
    assert "P0 first" in violations[0][1] or "protocol" in violations[0][1]
    
    # Subsequent violations: escalating responses
    # Should see stronger rejections, possible cooldown periods
    for i in range(1, 4):
        assert violations[i][0] is False
        # Later messages might reference prior violations
        if i >= 2:
            assert "repeat" in violations[i][1].lower() or "cooldown" in violations[i][1].lower()
    
    # Agent should eventually be marked for review
    agent_state = deadband_room.get_agent_state("violator")
    assert agent_state.get("violation_count", 0) >= 3
    assert "REVIEW" in agent_state.get("flags", []) or agent_state["status"] == "IDLE"
```

    """P0: Room handles memory exhaustion gracefully during large tile processing."""
    # Mock process_tile to simulate memory error
    original_process = deadband_room._process_tile
    
    def memory_hungry_process(tile):
        if len(tile.get("content", "")) > 1000:
            raise MemoryError("Not enough memory to process tile")
        return original_process(tile)
    
    mocker.patch.object(deadband_room, '_process_tile', side_effect=memory_hungry_process)
