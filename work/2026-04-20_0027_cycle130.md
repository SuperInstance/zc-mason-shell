### Test 3: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom()
    
    def submit_tile(agent_id):
        tile = {"agent": agent_id, "content": f"test_{agent_id}"}
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    import threading
    results = []
    threads = []
    
    for i in range(5):
        t = threading.Thread(target=lambda x=i: results.append(submit_tile(x)))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All tiles should be accepted without corruption
    self.assertEqual(len(room.pending_tiles), 5)
    self.assertEqual(len(set(results)), 5)  # All unique acceptance IDs
```

### Test 4: Malformed Tile Data

```python
def test_malformed_tile_rejection(self):
    """P0: Invalid tile structures should be rejected before processing."""
    room = DeadbandRoom()
    
    test_cases = [
        None,  # Null tile
        "not_a_dict",  # Wrong type
        {},  # Empty dict
        {"content": "test"},  # Missing required fields
        {"agent": "test", "content": 123},  # Wrong content type
        {"agent": "", "content": "test"},  # Empty agent ID
    ]
    
    for tile in test_cases:
        with self.subTest(tile=tile):
            with self.assertRaises((InvalidTileError, TypeError)):
                room.submit_tile(tile)
```

### Test 5: State Persistence Across Resets
```python
def test_state_persistence(self):
    """Room should maintain protocol state through soft resets."""
    room = DeadbandRoom()
    
    # Complete P0 and P1
    room.set_state("negative_space_mapped", True)
    room.find_safe_channels(["channel_a", "channel_b"])
    
    # Simulate reset (new instance with same ID)
    room_id = room.room_id
    room2 = DeadbandRoom(room_id=room_id)
    
    # State should persist or be reconstructable
    self.assertTrue(room2.get_state("negative_space_mapped"))
    self.assertIn("channel_a", room2.safe_channels)
    
    # P2 should work with persisted state
    result = room2.optimize("channel_a")
    self.assertIsNotNone(result)
```
