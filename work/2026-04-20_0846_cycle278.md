### 5. Test: `test_concurrent_tile_submission_race`
**Scenario:** Multiple agents submit tiles simultaneously.
**Rationale:** Tests thread safety in tile accumulation.
```python
def test_concurrent_tile_submission_race():
    """DeadbandRoom should handle concurrent tile submissions safely."""
    import threading
    
    room = DeadbandRoom()
    results = []
    
    def submit_tile_thread(tile_id):
        tile = {
            "phase": "P1",
            "content": f"Channel found from thread {tile_id}",
            "timestamp": f"2026-04-20T08:{tile_id:02d}:00Z"
        }
        result = room.submit_tile(tile)
        results.append((tile_id, result))
    
    # Create 10 concurrent submissions
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_tile_thread, args=(i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all tiles were processed
    assert len(results) == 10
    assert all(r[1]["status"] == "accepted" for r in results)
    assert len(room.tiles) == 10
```
