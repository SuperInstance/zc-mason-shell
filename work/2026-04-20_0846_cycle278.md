### Test 4: Concurrent Tile Submission
```python
async def test_concurrent_tile_submission():
    """P1: Room should handle simultaneous submissions from multiple agents."""
    room = DeadbandRoom(room_id="test-edge-4")
    
    async def submit_tile_async(agent_id):
        tile = {
            "agent_id": agent_id,
            "content": {"test": "concurrent"},
            "timestamp": asyncio.get_event_loop().time()
        }
        return await room.submit_tile(tile)
    
    # Submit from 3 agents concurrently
    tasks = [
        submit_tile_async("agent-1"),
        submit_tile_async("agent-2"), 
        submit_tile_async("agent-3")
    ]
    
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # All should succeed without data corruption
    successful = [r for r in results if not isinstance(r, Exception)]
    assert len(successful) == 3
    
    # Verify all tiles stored
    stored = await room.tile_store.get_recent_tiles(limit=10)
    assert len(stored) >= 3
```
