### Test 1: Concurrent Tile Submission from Multiple Agents
**Scenario:** Simulate 10 agents submitting tiles simultaneously to test thread safety and state corruption.
**Edge:** High concurrency can expose race conditions in room's internal queues.
```python
def test_concurrent_tile_submission_no_state_corruption(self):
    """DeadbandRoom should handle concurrent submissions without corrupting agent states or tile queue."""
    from concurrent.futures import ThreadPoolExecutor
    room = DeadbandRoom(capacity=20)
    agent_ids = [f"agent_{i}" for i in range(10)]
    
    def submit_tile(agent_id):
        tile = Tile(agent_id=agent_id, content=f"test from {agent_id}", phase=1)
        return room.submit_tile(tile)
    
    with ThreadPoolExecutor(max_workers=10) as executor:
        results = list(executor.map(submit_tile, agent_ids))
    
    # All submissions should succeed
    self.assertTrue(all(r['status'] == 'accepted' for r in results))
    # Room should have 10 tiles
    self.assertEqual(len(room.tile_queue), 10)
    # All agents should be in ACTIVE state
    for agent_id in agent_ids:
        self.assertEqual(room.agent_states[agent_id], 'ACTIVE')
```

def test_phase_skip_violation_rejects_tile(self):

def test_oversized_tile_rejection_with_graceful_degradation(self):
    """Tile exceeding size limit should be rejected without crashing room service."""
    room = DeadbandRoom(max_tile_size_mb=10)  # 10MB limit
    
    # Create tile with simulated large content
    large_content = "x" * (11 * 1024 * 1024)  # 11MB
    tile = Tile(agent_id="agent_big", content=large_content, phase=0)
    
    result = room.submit_tile(tile)
    
    self.assertEqual(result['status'], 'rejected')
    self.assertIn("exceeds size limit", result['reason'])
    # Room should still be operational
    self.assertEqual(room.agent_states["agent_big"], 'IDLE')

    normal_result = room.submit_tile(normal_tile)
    self.assertEqual(normal_result['status'], 'accepted')
