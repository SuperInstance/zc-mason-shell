### Test 1: Empty Tile Content
**Scenario:** A tile with empty or whitespace-only content is submitted.
**Edge Case:** The protocol should handle this as a P0 violation (no meaningful content) rather than crashing.
```python
def test_empty_tile_content(deadband_room):
    """Test handling of tiles with empty or whitespace-only content."""
    empty_tile = {"content": "", "metadata": {"source": "test"}}
    whitespace_tile = {"content": "   \n\t  ", "metadata": {"source": "test"}}
    
    # Both should be rejected at P0 (no meaningful content)
    result_empty = deadband_room.process_tile(empty_tile)
    result_whitespace = deadband_room.process_tile(whitespace_tile)
    
    assert result_empty["status"] == "rejected"
    assert result_empty["violation_level"] == "P0"
    assert "empty" in result_empty["reason"].lower()
    
    assert result_whitespace["status"] == "rejected"
    assert result_whitespace["violation_level"] == "P0"
```

### Test 3: Concurrent Tile Processing
**Scenario:** Multiple agents submit tiles simultaneously.
**Edge Case:** The room should handle concurrent access without race conditions or state corruption.
```python
import threading
import time

def test_concurrent_tile_processing(deadband_room):
    """Test that room handles concurrent tile submissions correctly."""
    results = []
    errors = []
    
    def submit_tile(tile_id):
        try:
            tile = {
                "content": f"Test tile {tile_id}",
                "metadata": {"source": f"agent_{tile_id}", "timestamp": time.time()}
            }
            result = deadband_room.process_tile(tile)
            results.append((tile_id, result))
        except Exception as e:
            errors.append((tile_id, str(e)))
    
    # Submit 10 tiles concurrently
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_tile, args=(i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify no errors occurred
    assert len(errors) == 0, f"Errors during concurrent processing: {errors}"
    
    # Verify all tiles were processed
    assert len(results) == 10
    
    # Verify protocol state is consistent
    # (Assuming the room tracks some state like processed_count)
    if hasattr(deadband_room, 'processed_count'):
        assert deadband_room.processed_count == 10
```

### Test 5: Protocol State Rollback on Error
**Scenario:** A tile passes P0 and P1 but causes an error during P2 optimization.
**Edge Case:** The room should roll back any partial state changes and maintain consistency.
```python
def test_state_rollback_on_p2_error(deadband_room):
    """Test that room rolls back state when P2 processing fails."""
    # First, get initial state
    initial_state = deadband_room.get_state()
    
    # Create a tile that will pass P0 and P1 but fail in P2
    # This requires knowledge of what causes P2 failures
    # Example: A tile that references non-existent optimization parameters
    problematic_tile = {
        "content": "Valid content that passes P0 and P1",
        "metadata": {
            "source": "test",
            "optimization_params": {"invalid_param": "causes_error"}
        }
    }
    
    try:
        result = deadband_room.process_tile(problematic_tile)
        # If we get here, the room might handle P2 errors gracefully
        if result["status"] == "rejected":
            assert result["violation_level"] == "P2"
    except Exception:
        # If an exception is raised, that's also acceptable for this edge case
        pass
    
    # Verify room state is unchanged or properly rolled back
    final_state = deadband_room.get_state()
    
    # The state should either be unchanged or only reflect successful prior operations
    # For example, processed_count shouldn't increment for failed tiles
    if hasattr(deadband_room, 'processed_count'):
        # Count successful tiles before and after
        successful_before = initial_state.get('successful_tiles', 0)
        successful_after = final_state.get('successful_tiles', 0)
        assert successful_after == successful_before
```
