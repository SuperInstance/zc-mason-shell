```python
def test_concurrent_tile_submission_race_condition():
    """Test that room handles simultaneous submissions from multiple agents."""
    room = DeadbandRoom(config=valid_config)
    
    # Simulate 10 agents submitting tiles at same "cycle"
    tiles = []
    for i in range(10):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": 100,
            "phase": "P1",
            "content": f"Test content from agent {i}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles (simulating concurrent access)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All should be accepted or have clear rejection reasons
    accepted = [r for r in results if r.get("status") == "accepted"]
    rejected = [r for r in results if r.get("status") == "rejected"]
    
    # State should remain consistent
    assert room.state["total_tiles"] == len(accepted)
    assert len(rejected) == 0 or all("reason" in r for r in rejected)
```

### Test 2: Mixed Protocol Phase Batch Processing

```python
def test_mixed_protocol_phase_batch():
    """Test batch containing tiles from all three protocol phases."""
    room = DeadbandRoom(config=valid_config)
    
    mixed_batch = [
        {"phase": "P0", "content": "Negative space mapping", "agent_id": "agent_1"},
        {"phase": "P1", "content": "Safe channel finding", "agent_id": "agent_2"},
        {"phase": "P2", "content": "Optimization attempt", "agent_id": "agent_3"},
        {"phase": "P0", "content": "More P0 work", "agent_id": "agent_1"},
        {"phase": "P2", "content": "Another P2", "agent_id": "agent_4"},  # Should fail if no P1 first
    ]
    
    results = []
    for tile in mixed_batch:
        result = room.submit_tile(tile)
        results.append(result)
        if result["status"] == "accepted":
            print(f"Accepted: {tile['phase']} from {tile['agent_id']}")
    
    # Check protocol enforcement
    p2_without_p1 = any(
        r["status"] == "accepted" 
        for r, tile in zip(results, mixed_batch) 
        if tile["phase"] == "P2" and tile["agent_id"] not in ["agent_1", "agent_2"]
    )
    assert not p2_without_p1, "P2 tiles accepted without preceding P1 from same agent"
```

### Test 3: Storage Quota Exhaustion

```python
def test_storage_quota_exhaustion():
    """Test behavior when tile storage limit is reached."""
    config = valid_config.copy()
    config["max_tiles"] = 5  # Small limit for testing
    room = DeadbandRoom(config=config)
    
    # Fill up to limit
    for i in range(5):
        tile = {
            "agent_id": "test_agent",
            "cycle": i,
            "phase": "P1",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # Attempt to exceed limit
    excess_tile = {
        "agent_id": "test_agent",
        "cycle": 6,
        "phase": "P1",
        "content": "Should be rejected",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(excess_tile)
    assert result["status"] == "rejected"
    assert "quota" in result.get("reason", "").lower() or "limit" in result.get("reason", "").lower()
    
    # Verify existing tiles preserved
    assert room.state["total_tiles"] == 5
```

### Test 4: Malformed Metadata Recovery
```python
def test_malformed_metadata_recovery():
    """Test room handling of tiles with corrupted metadata."""
    room = DeadbandRoom(config=valid_config)
    
    problematic_tiles = [
        {"agent_id": "", "phase": "P1", "content": "Empty agent ID"},  # Empty required field
        {"agent_id": "agent_x", "phase": "P3", "content": "Invalid phase"},  # Non-existent phase
        {"agent_id": "agent_y", "phase": "P1", "content": "", "timestamp": -1},  # Negative timestamp
        {"agent_id": "agent_z", "phase": "P1", "content": "X" * 10000},  # Extremely large content
        {"phase": "P1", "content": "Missing agent_id entirely"},  # Missing required field
    ]
    
    rejection_reasons = []
    for tile in problematic_tiles:
        result = room.submit_tile(tile)
        if result["status"] == "rejected":
            rejection_reasons.append(result.get("reason", "No reason given"))
    
    # All should be rejected with specific reasons
    assert len(rejection_reasons) == len(problematic_tiles)
    assert all(reason for reason in rejection_reasons)
    
    # Room should still accept valid tiles after errors
    valid_tile = {
        "agent_id": "recovery_agent",
        "cycle": 100,
        "phase": "P0",
        "content": "Valid after errors",
        "timestamp": time.time()
    }
    result = room.submit_tile(valid_tile)
    assert result["status"] == "accepted"
```
