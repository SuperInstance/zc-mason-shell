### Test 2: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """P1: Test handling of simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    tiles = []
    
    def submit_concurrent(agent_id):
        tile = {
            "phase": "P1",
            "content": f"Channel from agent {agent_id}",
            "agent_id": agent_id,
            "timestamp": time.time()
        }
        try:
            room.submit_tile(tile)
            tiles.append(tile)
        except Exception as e:
            tiles.append({"error": str(e)})
    
    # Simulate 10 agents submitting simultaneously
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_concurrent, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all tiles were processed without corruption
    self.assertEqual(len([t for t in tiles if "error" not in t]), 10)
    self.assertEqual(len(room.get_tiles()), 10)
```

### Test 4: Resource Exhaustion via Tile Flood
```python
def test_tile_flood_protection(self):
    """P0: Test rate limiting and resource protection against tile flooding."""
    room = DeadbandRoom()
    agent_id = "flood_agent"
    rejected_count = 0
    
    # Submit tiles at maximum rate
    for i in range(1000):  # Attempt 1000 rapid submissions
        tile = {
            "phase": "P1",
            "content": f"Flood tile {i}",
            "agent_id": agent_id,
            "timestamp": time.time()
        }
        try:
            room.submit_tile(tile)
        except RateLimitError:
            rejected_count += 1
        time.sleep(0.001)  # Minimal delay
    
    # Should reject after reasonable limit
    self.assertGreater(rejected_count, 0)
    self.assertLess(len(room.get_tiles()), 1000)
    
    # Verify agent-specific rate limiting
    agent_tiles = [t for t in room.get_tiles() if t["agent_id"] == agent_id]
    self.assertLessEqual(len(agent_tiles), 100)  # Reasonable per-agent limit
```

### Test 5: Integration Failure Fallback
```python
def test_ensign_export_failure_fallback(self):
    """P1: Test graceful degradation when ensign export dependencies fail."""
    room = DeadbandRoom()
    
    # Add valid tiles
    for i in range(5):
        tile = {
            "phase": "P1",
            "content": f"Test tile {i}",
            "agent_id": f"agent_{i}",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # Simulate torch export failure
    original_export = room.export_to_ensign
    room.export_to_ensign = MagicMock(side_effect=RuntimeError("CUDA unavailable"))
    
    # Should fall back to CPU export or raise specific error
    with self.assertRaises(ExportError) as context:
        room.export_to_ensign(lora_rank=8)
    
    self.assertIn("fallback", str(context.exception).lower())
    
    # Restore and verify successful export
    room.export_to_ensign = original_export
    adapter = room.export_to_ensign(lora_rank=8)
    self.assertIsNotNone(adapter)
    self.assertEqual(adapter["tile_count"], 5)
```
