### Test 2: Concurrent Submissions Thread Safety
```python
def test_concurrent_tile_submissions(self):
    """Test that concurrent tile submissions don't corrupt internal state."""
    room = DeadbandRoom()
    tiles = [
        {"agent_id": "agent1", "phase": "P0", "content": "test1", "timestamp": time.time()},
        {"agent_id": "agent2", "phase": "P0", "content": "test2", "timestamp": time.time()},
    ]
    
    def submit_tile(tile):
        try:
            room.submit_tile(tile)
            return True
        except:
            return False
    
    with ThreadPoolExecutor(max_workers=2) as executor:
        results = list(executor.map(submit_tile, tiles))
    
    self.assertTrue(all(results), "All concurrent submissions should succeed")
    self.assertEqual(len(room.get_tiles()), 2, "Both tiles should be stored")
```

def test_protocol_state_validation(self):
    """Test that protocol state machine enforces P0→P1→P2 sequence."""
    room = DeadbandRoom()
    
    # Attempt P1 before P0
    with self.assertRaises(RuntimeError) as context:
        room.execute_phase("P1", {})
    self.assertIn("p0", str(context.exception).lower())
    
    # Complete P0
    room.execute_phase("P0", {"constraints": ["test"]})
    
    # Attempt P2 before P1
    with self.assertRaises(RuntimeError) as context:
        room.execute_phase("P2", {})
    self.assertIn("p1", str(context.exception).lower())
```

def test_large_batch_processing(self):
    """Test behavior when processing large batches that approach memory limits."""
    room = DeadbandRoom()
    
    # Create a batch that's large but reasonable
    large_batch = []
    for i in range(1000):  # Boundary condition
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P0",
            "content": "x" * 1000,  # 1KB per tile
            "timestamp": time.time()
        }
        large_batch.append(tile)
    
    # Process batch
    results = []
    for tile in large_batch:
        try:
            room.submit_tile(tile)
            results.append(True)
        except MemoryError:
            results.append(False)
            break
    
    # Verify either all succeeded or failed gracefully
    if not all(results):
        self.assertIsInstance(room.last_error, MemoryError, 
                            "Should raise MemoryError not crash")
```
