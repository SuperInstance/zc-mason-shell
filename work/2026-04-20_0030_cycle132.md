### Test 5: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_handling():
    """P2: Simulated race condition for multiple agents submitting simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    results = []
    for agent_id in agent_ids:
        result = room.submit_tile(agent_id, f"Tile from {agent_id}")
        results.append(result)
    
    # All should succeed
    success_count = sum(1 for r in results if r["status"] == "success")
    
    # Either all succeed or room has locking mechanism
    assert success_count == len(agent_ids) or room.concurrent_submissions_handled
    
    # Verify all tiles stored with unique IDs
    stored_tiles = room.tile_store.tiles
    assert len(stored_tiles) == len(agent_ids)
    assert len(set(t["id"] for t in stored_tiles)) == len(agent_ids)  # All unique
```
