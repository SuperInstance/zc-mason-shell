```python
"""
Edge-case tests for DeadbandRoom.
Focuses on protocol violations, invalid inputs, and system state resilience.
"""
import pytest
import sys
import os
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), '..')))

from rooms.deadband_room import DeadbandRoom
from exceptions import ProtocolViolationError, InvalidTileError

class TestDeadbandRoomEdgeCases:
    """Test suite for DeadbandRoom edge cases."""

    def test_process_tile_missing_required_keys(self):
        """P0: Tile missing required 'phase' or 'content' keys should raise InvalidTileError."""
        room = DeadbandRoom(room_id="edge_test_1")
        # Tile missing 'phase'
        invalid_tile = {"content": "some content", "metadata": {}}
        with pytest.raises(InvalidTileError):
            room.process_tile(invalid_tile)
        # Tile missing 'content'
        invalid_tile2 = {"phase": "P0", "metadata": {}}
        with pytest.raises(InvalidTileError):
            room.process_tile(invalid_tile2)
        # Tile missing 'metadata' (optional in some cases, but room expects it)
        invalid_tile3 = {"phase": "P0", "content": "test"}
        with pytest.raises(InvalidTileError):
            room.process_tile(invalid_tile3)

    def test_process_tile_invalid_phase_value(self):
        """P0: Tile with phase not in ['P0','P1','P2'] should raise InvalidTileError."""
        room = DeadbandRoom(room_id="edge_test_2")
        invalid_tile = {"phase": "P5", "content": "invalid phase", "metadata": {}}
        with pytest.raises(InvalidTileError):
            room.process_tile(invalid_tile)
        # Edge: empty string phase
        invalid_tile2 = {"phase": "", "content": "empty phase", "metadata": {}}
        with pytest.raises(InvalidTileError):
            room.process_tile(invalid_tile2)
        # Edge: None phase
        invalid_tile3 = {"phase": None, "content": "none phase", "metadata": {}}
        with pytest.raises(InvalidTileError):
            room.process_tile(invalid_tile3)

    def test_protocol_sequence_violation_p1_before_p0(self):
        """P0: Submitting P1 tile before any P0 tile should raise ProtocolViolationError."""
        room = DeadbandRoom(room_id="edge_test_3")
        # First tile is P1 (violation)
        tile_p1 = {"phase": "P1", "content": "skip P0", "metadata": {}}
        with pytest.raises(ProtocolViolationError):
            room.process_tile(tile_p1)
        # Room state should still be at P0
        state = room.get_state()
        assert state["current_phase"] == "P0"
        assert len(state["processed_tiles"]) == 0

    def test_protocol_sequence_violation_p2_before_p1(self):
        """P0: Submitting P2 tile before P1 tile should raise ProtocolViolationError."""
        room = DeadbandRoom(room_id="edge_test_4")
        # Valid P0 first
        tile_p0 = {"phase": "P0", "content": "negative space", "metadata": {}}
        room.process_tile(tile_p0)
        # Jump to P2 (violation) - P1 not done
        tile_p2 = {"phase": "P2", "content": "optimize", "metadata": {}}
        with pytest.raises(ProtocolViolationError):
            room.process_tile(tile_p2)
        # State should reflect only P0 processed
        state = room.get_state()
        assert state["current_phase"] == "P0"
        assert len(state["processed_tiles"]) == 1
        assert state["processed_tiles"][0]["phase"] == "P0"

    def test_room_state_resilience_after_multiple_errors(self):
        """P1: Room should maintain consistent state after multiple invalid submissions."""
        room = DeadbandRoom(room_id="edge_test_5")
        initial_state = room.get_state()
        # Series of invalid submissions
        invalid_tiles = [
            {"phase": "P5", "content": "bad1", "metadata": {}},
            {"phase": "P1", "content": "bad2", "metadata": {}},  # P1 before P0
            {"phase": "P0", "content": "", "metadata": {}},  # Empty content (allowed? assume yes)
        ]
        error_count = 0
        for tile in invalid_tiles:
            try:
                room.process_tile(tile)
            except (InvalidTileError, ProtocolViolationError):
                error_count += 1
        # Should have caught 2 errors (P5 invalid, P1 before P0). Empty content may be valid.
        assert error_count >= 2
        # Room state should be unchanged from initial except maybe internal counters
        final_state = room.get_state()
        assert final_state["current_phase"] == "P0"
        assert len(final_state["processed_tiles"]) == 0
        # Submit a valid P0 tile after errors - should work
        valid_tile = {"phase": "P0", "content": "recovery", "metadata": {}}
        room.process_tile(valid_tile)
        recovered_state = room.get_state()
        assert recovered_state["current_phase"] == "P0"
        assert len(recovered_state["processed_tiles"]) == 1
        assert recovered_state["processed_tiles"][0]["content"] == "recovery"
```

## Test Rationale & Coverage
1. **Missing Required Keys**: Ensures the room validates tile structure before processing. This is a P0 safety check.
2. **Invalid Phase Values**: Tests that only the exact phase strings "P0", "P1", "P2" are accepted. Guards against typos or malformed data.
3. **P1 Before P0**: Direct test of the Deadband Protocol's strict phase ordering. Skipping P0 is a fundamental violation.
4. **P2 Before P1**: Tests the middle step of the protocol. You cannot optimize (P2) before finding safe channels (P1).
5. **State Resilience**: Verifies that the room's internal state remains consistent after multiple errors and can recover to accept valid tiles. This is a P1 "safe channel" for system robustness.

These tests increase coverage of error paths and protocol logic, which are critical for a training room that enforces fleet doctrine.
