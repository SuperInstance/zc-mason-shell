# plato-torch/tests/test_deadband_room.py (additional tests)

import pytest
from unittest.mock import patch, MagicMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.exceptions import ProtocolViolationError, TileRejectedError

def test_p0_violation_with_valid_p2_data(room_instance, sample_tile):
    """Edge Case 1: Tile with valid P2 optimization but P0 constraint violation."""
    # Mock validate_tile to pass P2 checks
    with patch('plato_torch.rooms.deadband_room.validate_tile') as mock_validate:
        mock_validate.return_value = {'p0': False, 'p1': True, 'p2': True}  # P0 false = violation
        with pytest.raises(ProtocolViolationError, match="P0 constraint violated"):
            room_instance.submit_tile(sample_tile)
        # Ensure P2 optimization logic was never called
        assert not room_instance._optimize_p2.called if hasattr(room_instance, '_optimize_p2') else True

def test_concurrent_tile_submission(room_instance, sample_tile):
    """Edge Case 2: Simulate concurrent submissions from multiple agents."""
    import threading
    results = []
    errors = []
    
    def submit():
        try:
            result = room_instance.submit_tile(sample_tile.copy())
            results.append(result)
        except Exception as e:
            errors.append(e)
    
    threads = [threading.Thread(target=submit) for _ in range(10)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # State should remain consistent; no crashes or corrupted logs
    assert len(errors) == 0 or all(isinstance(e, TileRejectedError) for e in errors)
    assert room_instance.state['submission_count'] == len(results) + len([e for e in errors if isinstance(e, TileRejectedError)])

def test_malformed_tile_metadata(room_instance):
    """Edge Case 3: Tiles with missing or invalid required metadata."""
    # Missing agent_id
    tile1 = {'cycle': 336, 'phase': 4, 'content': 'test'}
    with pytest.raises(TileRejectedError, match="Missing required field"):
        room_instance.submit_tile(tile1)
    
    # Invalid cycle (negative)
    tile2 = {'agent_id': 'mason', 'cycle': -1, 'phase': 4, 'content': 'test'}
    with pytest.raises(TileRejectedError, match="Invalid cycle"):
        room_instance.submit_tile(tile2)
    
    # Phase out of bounds (beyond 4)
    tile3 = {'agent_id': 'mason', 'cycle': 336, 'phase': 5, 'content': 'test'}
    with pytest.raises(TileRejectedError, match="Invalid phase"):
        room_instance.submit_tile(tile3)

def test_room_capacity_limits(room_instance, sample_tile):
    """Edge Case 4: Submit tiles until room buffer is full."""
    max_capacity = room_instance.config.get('max_tiles_buffer', 100)
    
    # Fill buffer
    for i in range(max_capacity):
        tile = sample_tile.copy()
        tile['cycle'] = i
        room_instance.submit_tile(tile)
    
    # Next submission should be rejected or queued
    tile_overflow = sample_tile.copy()
    tile_overflow['cycle'] = max_capacity + 1
    
    if room_instance.config.get('reject_on_full', True):
        with pytest.raises(TileRejectedError, match="Buffer full"):
            room_instance.submit_tile(tile_overflow)
    else:
        # Queue behavior: submission succeeds but tile is queued
        result = room_instance.submit_tile(tile_overflow)
        assert result.get('queued', False) == True
        assert len(room_instance._tile_queue) > 0

def test_protocol_phase_rollback(room_instance, sample_tile):
    """Edge Case 5: Tile attempts to move room backward in protocol phase."""
    # Set room to P2 phase
    room_instance.state['current_phase'] = 'P2'
    room_instance.state['phase_progress'] = 0.8
    
    # Submit tile with P1 data (attempting rollback)
    tile = sample_tile.copy()
    tile['phase'] = 1  # P1 phase
    tile['content'] = {'protocol_step': 'P1', 'action': 'find_safe_channel'}
    
    with pytest.raises(ProtocolViolationError, match="Protocol phase rollback"):
        room_instance.submit_tile(tile)
    
    # Room phase should remain unchanged
    assert room_instance.state['current_phase'] == 'P2'
