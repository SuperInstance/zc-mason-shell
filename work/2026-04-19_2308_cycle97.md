"""
Edge-case tests for DeadbandRoom preset in plato-torch.
Focuses on protocol adherence, state transitions, and error handling.
"""

import pytest
from unittest.mock import Mock, patch, AsyncMock
import asyncio

# Assuming DeadbandRoom class exists in plato_torch.presets.deadband
from plato_torch.presets.deadband import DeadbandRoom


class TestDeadbandRoomEdgeCases:
    """Test suite for DeadbandRoom edge cases."""

    @pytest.mark.asyncio
    async def test_p0_violation_blocks_p2(self):
        """
        Edge: Agent attempts P2 optimization while P0 negative space is unmapped.
        Expected: Room rejects tile, returns to P0 state.
        """
        room = DeadbandRoom(config={"channels": [], "thresholds": {"p0": 0.9}})
        agent_id = "test_agent_1"
        
        # Simulate agent skipping to P2 without completing P0
        room.agent_states[agent_id] = {"phase": "P2", "p0_mapped": False}
        
        tile = {"phase": "P2", "content": "Optimizing channel X", "priority": 2}
        
        # Submit tile
        result = await room.submit_tile(agent_id, tile)
        
        # Should be rejected, state reset to P0
        assert not result["accepted"]
        assert "P0 not complete" in result.get("reason", "")
        assert room.agent_states[agent_id]["phase"] == "P0"

    @pytest.mark.asyncio
    async def test_channel_capacity_overflow(self):
        """
        Edge: More agents enter safe channel than configured capacity.
        Expected: Room rejects new agents, suggests alternative channels.
        """
        config = {
            "channels": [{"id": "safe_1", "capacity": 2}],
            "max_agents_per_channel": 2
        }
        room = DeadbandRoom(config=config)
        
        # Fill channel to capacity
        await room.enter_room("agent_a", {"channel": "safe_1"})
        await room.enter_room("agent_b", {"channel": "safe_1"})
        
        # Third agent tries to enter same channel
        result = await room.enter_room("agent_c", {"channel": "safe_1"})
        
        assert not result["success"]
        assert "capacity" in result.get("message", "").lower()
        assert "agent_c" not in room.agent_states

    @pytest.mark.asyncio
    async def test_tile_validation_malformed_structure(self):
        """
        Edge: Agent submits tile with missing required fields.
        Expected: Room rejects tile with specific validation error.
        """
        room = DeadbandRoom(config={})
        agent_id = "test_agent_3"
        await room.enter_room(agent_id, {})
        
        # Tile missing 'content' field
        malformed_tile = {"phase": "P1", "priority": 1}
        
        result = await room.submit_tile(agent_id, malformed_tile)
        
        assert not result["accepted"]
        assert "validation" in result.get("reason", "").lower()
        assert "missing" in result.get("reason", "").lower()

    @pytest.mark.asyncio
    async def test_ghost_tile_inheritance_on_agent_removal(self):
        """
        Edge: Agent removed (e.g., timeout), their P0 lessons persist as ghost tiles.
        Expected: New agent can access ghost tiles for P0 mapping.
        """
        room = DeadbandRoom(config={"ghost_tile_ttl": 3600})
        first_agent = "agent_old"
        
        # First agent submits P0 tile
        await room.enter_room(first_agent, {})
        p0_tile = {
            "phase": "P0",
            "content": "Avoid rock X in channel Y",
            "priority": 0,
            "type": "negative_space"
        }
        await room.submit_tile(first_agent, p0_tile)
        
        # Simulate agent removal (e.g., timeout)
        del room.agent_states[first_agent]
        
        # New agent enters
        new_agent = "agent_new"
        await room.enter_room(new_agent, {})
        
        # Check if ghost tiles are available
        ghost_tiles = room.get_ghost_tiles(phase="P0")
        assert len(ghost_tiles) > 0
        assert ghost_tiles[0]["content"] == "Avoid rock X in channel Y"

    @pytest.mark.asyncio
    async def test_concurrent_tile_submission_race_condition(self):
        """
        Edge: Multiple agents submit tiles simultaneously for same channel.
        Expected: Room processes sequentially, state remains consistent.
        """
        room = DeadbandRoom(config={"channels": [{"id": "fast_channel", "capacity": 10}]})
        agent_ids = [f"agent_race_{i}" for i in range(5)]
        
        # Enter all agents
        for aid in agent_ids:
            await room.enter_room(aid, {"channel": "fast_channel"})
        
        # Concurrent tile submissions
        tiles = [
            {"phase": "P1", "content": f"Channel mapping {i}", "priority": 1}
            for i in range(5)
        ]
        
        # Submit concurrently
        tasks = [
            room.submit_tile(aid, tile)
            for aid, tile in zip(agent_ids, tiles)
        ]
        results = await asyncio.gather(*tasks)
        
        # All should be accepted
        assert all(r["accepted"] for r in results)
        
        # State should reflect all submissions
        channel_tiles = room.get_channel_tiles("fast_channel")
        assert len(channel_tiles) == 5
