### Test 1: Concurrent Tile Submission Race Condition

def test_concurrent_tile_submission(self):
    """Test DeadbandRoom handles concurrent tile submissions from multiple agents."""
    room = DeadbandRoom()
    
    # Simulate 5 agents submitting tiles simultaneously
    tiles = [
        Tile(agent_id=f"agent_{i}", content=f"test_{i}", priority="P0")
        for i in range(5)
    ]
    
    # Submit all tiles concurrently (simulated)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All tiles should be accepted or properly queued
    accepted_count = sum(1 for r in results if r["status"] == "accepted")
    queued_count = sum(1 for r in results if r.get("queued", False))
    
    # Room should handle concurrency without crashing
    assert room.state["total_tiles"] == 5
    assert accepted_count + queued_count == 5
    assert room.state["active_agents"] == 5
```

### Test 3: Memory Pressure During Batch Processing

def test_memory_pressure_handling(self):
    """Test room behavior when processing large tile batches under memory constraints."""
    room = DeadbandRoom()
    
    # Submit large batch of tiles
    large_batch = []
    for i in range(1000):
        tile = Tile(
            agent_id=f"batch_agent_{i}",
            content="x" * 1000,  # 1KB content
            priority="P1" if i % 2 == 0 else "P2"
        )
        large_batch.append(tile)
    
    # Process with simulated memory limit
    processed = 0
    memory_warnings = 0
    
    for tile in large_batch:
        try:
            result = room.submit_tile_with_memory_check(tile, max_memory_mb=50)
            if result.get("memory_warning"):
                memory_warnings += 1
            processed += 1
        except MemoryError:
            # Room should handle gracefully, not crash
            room.log_memory_event("threshold_exceeded")
            break
    
    # Should process some tiles and log memory events
    assert processed > 0
    assert memory_warnings >= 0
    assert "memory_events" in room.metrics
    assert room.state["tiles_processed"] == processed
```

### Test 4: Protocol State Corruption Recovery

def test_corrupted_state_recovery(self):
    """Test room can recover from corrupted protocol state."""
    room = DeadbandRoom()
    
    # First, establish normal state
    for i in range(3):
        tile = Tile(agent_id="recovery_agent", content=f"normal_{i}", priority="P0")
        room.submit_tile(tile)
    
    # Simulate state corruption (e.g., from faulty persistence)
    room.state["current_priority"] = "INVALID_PRIORITY"
    room.state["phase"] = 999
    
    # Room should detect and attempt recovery
    recovery_result = room.attempt_state_recovery()
    
    # Recovery should either fix state or enter safe mode
    assert recovery_result["attempted"] is True
    assert recovery_result["success"] in [True, False]
    
    if not recovery_result["success"]:
        # Should enter safe/fallback mode
        assert room.state["safe_mode"] is True
        assert room.state["current_priority"] == "P0"  # Reset to base
    else:
        # State should be valid
        assert room.state["current_priority"] in ["P0", "P1", "P2"]
        assert 0 <= room.state["phase"] <= 3
```

def test_priority_inversion_detection(self):
    """Test detection of priority inversion across multiple cycles."""
    room = DeadbandRoom()
    
    # Simulate agent attempting priority inversion
    # P0 → P2 (skipping P1) across cycles
    cycle_states = []
    
    # Cycle 1: Submit P0 tile
    tile1 = Tile(agent_id="inversion_agent", content="cycle1_p0", priority="P0")
    result1 = room.submit_tile(tile1)
    cycle_states.append(room.state["current_priority"])
    
    # Simulate cycle advance
    room.advance_cycle()
    
    # Cycle 2: Attempt P2 directly (should be detected)
    tile2 = Tile(agent_id="inversion_agent", content="cycle2_p2_attempt", priority="P2")
    
    # Room should detect this as potential inversion
    inversion_detected = room.check_priority_inversion(
        previous_priority="P0",
        attempted_priority="P2",
        agent_history=["P0"]  # Only P0 in history
    )
    
    assert inversion_detected is True
    
    # Submission should be rejected or downgraded
    result2 = room.submit_tile(tile2)
    assert result2["status"] in ["rejected", "downgraded"]
    
    if result2["status"] == "downgraded":
        assert result2["effective_priority"] == "P1"  # Auto-correct to P1
```
