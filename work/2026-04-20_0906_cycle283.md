```python
def test_concurrent_tile_processing(self):
    """Test that DeadbandRoom handles simultaneous tiles from multiple agents without priority violations."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3", "agent_4"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 283,
            "content": f"Test from {agent_id}",
            "priority_layer": "P0",  # All start at P0
            "validation_hash": f"hash_{agent_id}"
        }
        tiles.append(tile)
    
    # Process concurrently (simulated)
    results = []
    for tile in tiles:
        result = room.process_tile(tile)
        results.append(result)
    
    # Verify all processed without deadlock or state corruption
    assert all(r["status"] == "accepted" for r in results), "Concurrent processing failed"
    assert len(room.get_processed_tiles()) == len(agent_ids), "Tile count mismatch"
    
    # Verify P0→P1→P2 sequence maintained per agent
    for agent_id in agent_ids:
        agent_tiles = room.get_agent_tiles(agent_id)
        priorities = [t["priority_layer"] for t in agent_tiles]
        assert priorities == ["P0"], f"Priority sequence broken for {agent_id}"
```

### Test 2: Protocol State Recovery
```python
def test_state_recovery_after_corruption(self):
    """Test DeadbandRoom recovers from corrupted priority state."""
    room = DeadbandRoom()
    
    # First, establish normal state
    normal_tile = {
        "agent_id": "test_agent",
        "cycle": 283,
        "content": "Normal P0 tile",
        "priority_layer": "P0",
        "validation_hash": "hash_normal"
    }
    room.process_tile(normal_tile)
    
    # Corrupt internal state (simulate bug/edge case)
    room._priority_state["test_agent"] = "P2"  # Direct manipulation
    
    # Submit P1 tile (should be rejected due to corruption)
    p1_tile = {
        "agent_id": "test_agent",
        "cycle": 284,
        "content": "P1 tile after corruption",
        "priority_layer": "P1",
        "validation_hash": "hash_p1"
    }
    
    result = room.process_tile(p1_tile)
    
    # Room should detect corruption and either:
    # 1. Reject with specific error, OR
    # 2. Reset agent state and accept
    assert result["status"] in ["rejected", "accepted"], "Unexpected result"
    
    if result["status"] == "rejected":
        assert "priority_sequence" in result.get("error", "").lower(), \
            "Should mention priority sequence violation"
    else:
        # If accepted, verify state was reset
        assert room._priority_state.get("test_agent") == "P1", \
            "State not properly reset after corruption"
```

### Test 4: Resource Exhaustion Scenarios
```python
def test_resource_exhaustion_handling(self):
    """Test behavior when system resources are constrained."""
    room = DeadbandRoom()
    
    # Simulate memory pressure by processing many tiles
    processed_count = 0
    rejection_count = 0
    
    for i in range(1000):  # Large batch
        tile = {
            "agent_id": f"stress_agent_{i % 10}",  # 10 agents
            "cycle": 283 + i,
            "content": f"Stress test content {i}",
            "priority_layer": "P0" if i % 3 == 0 else "P1",
            "validation_hash": f"hash_{i}"
        }
        
        result = room.process_tile(tile)
        
        if result["status"] == "accepted":
            processed_count += 1
        else:
            rejection_count += 1
            # Verify rejections are for valid reasons, not arbitrary
            assert "error" in result, "Rejection should include error"
    
    # Room should handle gracefully without crash
    assert processed_count > 0, "Should process some tiles"
    
    # Check memory/state hasn't grown unbounded
    state_size = len(str(room.__dict__))
    assert state_size < 1000000, "State size appears excessive"  # 1MB heuristic
    
    # Room should still accept new tiles after stress
    recovery_tile = {
        "agent_id": "recovery_agent",
        "cycle": 1283,
        "content": "Post-stress test",
        "priority_layer": "P0",
        "validation_hash": "hash_recovery"
    }
    
    recovery_result = room.process_tile(recovery_tile)
    assert recovery_result["status"] == "accepted", \
        "Room should recover and accept new tiles after stress"
```
