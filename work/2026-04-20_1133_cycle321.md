### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(room_id="concurrent_test")
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Join all agents
    for agent_id in agent_ids:
        room.join_agent(agent_id)
    
    # Submit tiles concurrently (simulated)
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "protocol_phase": "P0",
            "content": f"Negative space map from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Process tiles
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results if isinstance(r, dict))
    
    # Verify room tile count matches
    assert room.get_tile_count() == len(agent_ids)
```

### Test 2: Protocol State Persistence
```python
def test_protocol_state_persistence(self):
    """Room should maintain protocol phase across multiple training cycles."""
    room = DeadbandRoom(room_id="persistence_test")
    agent_id = "test_agent"
    room.join_agent(agent_id)
    
    # Initial P0 phase
    tile_p0 = {
        "agent_id": agent_id,
        "protocol_phase": "P0",
        "content": "Initial negative space mapping",
        "timestamp": time.time()
    }
    room.submit_tile(tile_p0)
    assert room.get_current_phase() == "P0"
    
    # Transition to P1
    tile_p1 = {
        "agent_id": agent_id,
        "protocol_phase": "P1",
        "content": "Safe channel identification",
        "timestamp": time.time()
    }
    room.submit_tile(tile_p1)
    assert room.get_current_phase() == "P1"
    
    # Simulate room persistence (save/load)
    room_state = room.save_state()
    new_room = DeadbandRoom(room_id="persistence_test_restored")
    new_room.load_state(room_state)
    
    # Verify restored state
    assert new_room.get_current_phase() == "P1"
    assert new_room.get_tile_count() == 2
    assert new_room.has_agent(agent_id)
```

def test_malformed_tile_handling(self):
    """Room should reject malformed tiles with appropriate errors."""
    room = DeadbandRoom(room_id="malformed_test")
    agent_id = "test_agent"
    room.join_agent(agent_id)
    
    test_cases = [
        # (tile_payload, expected_error_pattern)
        ({"agent_id": agent_id}, "Missing required field: protocol_phase"),
        ({"protocol_phase": "P0"}, "Missing required field: agent_id"),
        ({"agent_id": agent_id, "protocol_phase": "P5"}, "Invalid protocol phase"),
        ({"agent_id": "unknown_agent", "protocol_phase": "P0"}, "Agent not joined"),
        ("not_a_dict", "Tile must be a dictionary"),
        (None, "Tile cannot be None"),
        ({"agent_id": agent_id, "protocol_phase": "P0", "content": ""}, "Empty content"),
    ]
    
    for tile_payload, expected_error in test_cases:
        with pytest.raises(ValueError) as exc_info:
            room.submit_tile(tile_payload)
        assert expected_error in str(exc_info.value)

### Test 4: Agent Disconnection Recovery
```python
def test_agent_reconnection_state(self):
    """Agent should maintain their submitted tiles after disconnection/reconnection."""
    room = DeadbandRoom(room_id="reconnect_test")
    agent_id = "reconnecting_agent"
    
    # First session
    room.join_agent(agent_id)
    tile1 = {
        "agent_id": agent_id,
        "protocol_phase": "P0",
        "content": "First session tile",
        "timestamp": time.time()
    }
    room.submit_tile(tile1)
    
    # Simulate disconnection
    room.leave_agent(agent_id)
    assert not room.has_agent(agent_id)
    
    # Reconnection
    room.join_agent(agent_id)
    tile2 = {
        "agent_id": agent_id,
        "protocol_phase": "P1",
        "content": "Second session tile",
        "timestamp": time.time()
    }
    room.submit_tile(tile2)
    
    # Verify both tiles present and associated with agent
    agent_tiles = room.get_agent_tiles(agent_id)
    assert len(agent_tiles) == 2
    assert tile1["content"] in [t["content"] for t in agent_tiles]
    assert tile2["content"] in [t["content"] for t in agent_tiles]
```

### Test 5: High Volume Tile Performance
```python
def test_high_volume_tile_performance(self):
    """Room should handle high tile volume without performance degradation."""
    room = DeadbandRoom(room_id="volume_test")
    agent_id = "volume_agent"
    room.join_agent(agent_id)
    
    tile_count = 1000
    start_time = time.time()
    
    for i in range(tile_count):
        tile = {
            "agent_id": agent_id,
            "protocol_phase": "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
            "content": f"Tile {i}: {'P0' if i % 3 == 0 else 'P1' if i % 3 == 1 else 'P2'} content",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    end_time = time.time()
    processing_time = end_time - start_time
    
    # Performance assertion: < 0.1ms per tile on average
    avg_time_per_tile = (processing_time / tile_count) * 1000  # Convert to ms
    assert avg_time_per_tile < 0.1, f"Average tile processing too slow: {avg_time_per_tile:.3f}ms"
    
    # Memory check (approximate)
    import sys
    room_size = sys.getsizeof(room) + sum(sys.getsizeof(t) for t in room.get_all_tiles())
    assert room_size < 10 * 1024 * 1024, f"Room memory usage too high: {room_size / 1024 / 1024:.1f}MB"
```
