# plato-torch/tests/test_deadband_room_edge.py
import pytest
from unittest.mock import patch, MagicMock
import threading
import time
from plato_torch.rooms.deadband_room import DeadbandRoom

class TestDeadbandRoomEdgeCases:
    
    def test_tile_invalid_priority_value(self):
        """Edge case: Tile has priority field but value is not P0/P1/P2"""
        room = DeadbandRoom()
        invalid_tile = {
            "agent_id": "test_agent",
            "content": "Test content",
            "priority": "P3",  # Invalid priority
            "timestamp": "2026-04-20T19:10:00Z"
        }
        
        result = room.process_tile(invalid_tile)
        assert not result["valid"]
        assert "priority" in result["errors"]
        assert "P3" in result["errors"]["priority"]
    
    def test_tile_empty_content(self):
        """Edge case: Tile content field is empty string"""
        room = DeadbandRoom()
        empty_tile = {
            "agent_id": "test_agent",
            "content": "",  # Empty content
            "priority": "P1",
            "timestamp": "2026-04-20T19:11:00Z"
        }
        
        result = room.process_tile(empty_tile)
        # According to room logic, empty content should be rejected
        assert not result["valid"]
        assert "content" in result["errors"]
        assert "empty" in result["errors"]["content"].lower()
    
    def test_concurrent_tile_processing(self):
        """Edge case: Multiple threads submitting tiles simultaneously"""
        room = DeadbandRoom()
        results = []
        errors = []
        
        def submit_tile(agent_id):
            try:
                tile = {
                    "agent_id": agent_id,
                    "content": f"Content from {agent_id}",
                    "priority": "P1",
                    "timestamp": time.time()
                }
                result = room.process_tile(tile)
                results.append((agent_id, result))
            except Exception as e:
                errors.append((agent_id, str(e)))
        
        threads = []
        for i in range(10):
            t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
            threads.append(t)
            t.start()
        
        for t in threads:
            t.join()
        
        # No exceptions should occur
        assert len(errors) == 0
        # All tiles should be processed
        assert len(results) == 10
        # Check all were valid
        for agent_id, result in results:
            assert result["valid"], f"Tile from {agent_id} was invalid"
    
    def test_training_batch_mixed_priorities(self):
        """Edge case: Training batch maintains P0 > P1 > P2 ordering"""
        room = DeadbandRoom()
        
        # Submit tiles in random priority order
        tiles = [
            {"agent_id": "a1", "content": "P2 content", "priority": "P2", "timestamp": "2026-04-20T19:12:00Z"},
            {"agent_id": "a2", "content": "P0 content", "priority": "P0", "timestamp": "2026-04-20T19:12:01Z"},
            {"agent_id": "a3", "content": "P1 content", "priority": "P1", "timestamp": "2026-04-20T19:12:02Z"},
            {"agent_id": "a4", "content": "P1 content 2", "priority": "P1", "timestamp": "2026-04-20T19:12:03Z"},
            {"agent_id": "a5", "content": "P0 content 2", "priority": "P0", "timestamp": "2026-04-20T19:12:04Z"},
        ]
        
        for tile in tiles:
            room.process_tile(tile)
        
        batch = room.get_training_batch(batch_size=5)
        
        # Verify ordering: P0 tiles first, then P1, then P2
        priorities = [tile["priority"] for tile in batch]
        
        # Count occurrences
        p0_count = priorities.count("P0")
        p1_count = priorities.count("P1")
        p2_count = priorities.count("P2")
        
        # P0 tiles should be first
        assert priorities[0] == "P0"
        assert priorities[1] == "P0"
        # Then P1 tiles
        assert priorities[2] == "P1"
        assert priorities[3] == "P1"
        # Then P2 tile
        assert priorities[4] == "P2"
        
        # Verify counts match
        assert p0_count == 2
        assert p1_count == 2
        assert p2_count == 1
    
    def test_room_state_persistence_after_reset(self):
        """Edge case: Room can be reused after reset without corruption"""
        room = DeadbandRoom()
        
        # Phase 1: Submit some tiles
        for i in range(3):
            tile = {
                "agent_id": f"phase1_agent_{i}",
                "content": f"Phase 1 content {i}",
                "priority": "P1",
                "timestamp": f"2026-04-20T19:13:{i:02d}Z"
            }
            room.process_tile(tile)
        
        # Verify tiles were stored
        batch1 = room.get_training_batch(batch_size=5)
        assert len(batch1) == 3
        
        # Reset room
        room.reset_room()
        
        # Verify room is empty
        batch2 = room.get_training_batch(batch_size=5)
        assert len(batch2) == 0
        
        # Phase 2: Submit new tiles after reset
        for i in range(2):
            tile = {
                "agent_id": f"phase2_agent_{i}",
                "content": f"Phase 2 content {i}",
                "priority": "P2",
                "timestamp": f"2026-04-20T19:14:{i:02d}Z"
            }
            result = room.process_tile(tile)
            assert result["valid"]
        
        # Verify only phase 2 tiles exist
        batch3 = room.get_training_batch(batch_size=5)
        assert len(batch3) == 2
        assert all("phase2_agent" in tile["agent_id"] for tile in batch3)
        assert all(tile["priority"] == "P2" for tile in batch3)
