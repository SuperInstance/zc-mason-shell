### Test 1: Concurrent Tile Submission

```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(capacity=5)
    agent_ids = [f"agent_{i}" for i in range(3)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 398,
            "content": f"Test from {agent_id}",
            "protocol_phase": "P1"
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All should succeed
    assert all(r["status"] == "accepted" for r in results)
    
    # Room should have correct tile count
    assert room.get_tile_count() == 3
    
    # Each agent's tile should be retrievable
    for agent_id in agent_ids:
        agent_tiles = room.get_agent_tiles(agent_id)
        assert len(agent_tiles) == 1
        assert agent_tiles[0]["agent_id"] == agent_id
```

```python
def test_malformed_content_validation(self):
    """Tiles with valid structure but invalid content should be rejected with specific errors."""
    room = DeadbandRoom(capacity=2)
    
    test_cases = [
        {
            "tile": {
                "agent_id": "test_agent",
                "cycle": 398,
                "content": "",  # Empty content
                "protocol_phase": "P1"
            },
            "expected_error": "content_empty"
        },
        {
            "tile": {
                "agent_id": "test_agent",
                "cycle": 398,
                "content": "X" * 10001,  # Too long
                "protocol_phase": "P1"
            },
            "expected_error": "content_too_long"
        },
        {
            "tile": {
                "agent_id": "test_agent",
                "cycle": -1,  # Invalid cycle
                "content": "Test",
                "protocol_phase": "P1"
            },
            "expected_error": "invalid_cycle"
        },
        {
            "tile": {
                "agent_id": "test_agent",
                "cycle": 398,
                "content": "Test",
                "protocol_phase": "P3"  # Invalid phase
            },
            "expected_error": "invalid_protocol_phase"
        }
    ]
    
    for i, test_case in enumerate(test_cases):
        result = room.submit_tile(test_case["tile"])
        assert result["status"] == "rejected"
        assert test_case["expected_error"] in result.get("error", "")
        
        # Room state should remain unchanged
        assert room.get_tile_count() == 0
```

### Test 4: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Agent skipping P0→P2 should be corrected and able to recover."""
    room = DeadbandRoom(capacity=2)
    
    # Agent tries to skip from P0 to P2 (violation)
    violating_tile = {
        "agent_id": "recovering_agent",
        "cycle": 398,
        "content": "Trying to optimize without mapping negative space",
        "protocol_phase": "P2"  # Skipped P1
    }
    
    result = room.submit_tile(violating_tile)
    assert result["status"] == "rejected"
    assert "protocol_violation" in result.get("error", "")
    
    # Agent state should be "needs_correction"
    assert room.get_agent_state("recovering_agent") == "needs_correction"
    
    # Agent corrects by submitting P1 tile
    correcting_tile = {
        "agent_id": "recovering_agent",
        "cycle": 398,
        "content": "Now finding safe channels (P1)",
        "protocol_phase": "P1"
    }
    
    result = room.submit_tile(correcting_tile)
    assert result["status"] == "accepted"
    
    # State should transition to "active"
    assert room.get_agent_state("recovering_agent") == "active"
    
    # Now P2 should be accepted
    final_tile = {
        "agent_id": "recovering_agent",
        "cycle": 399,
        "content": "Optimizing within channels",
        "protocol_phase": "P2"
    }
    
    result = room.submit_tile(final_tile)
    assert result["status"] == "accepted"
```
