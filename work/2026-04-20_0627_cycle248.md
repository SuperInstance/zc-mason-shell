### Test 1: Protocol Violation – Skip from P0 to P2
```python
def test_skip_p0_to_p2_violation():
    """DeadbandRoom must reject actions that skip from P0 directly to P2, logging a violation."""
    room = DeadbandRoom()
    agent_id = "test_agent_skip"
    room.register_agent(agent_id)

    # Agent submits a P0 tile (mapping negative space)
    room.submit_tile(agent_id, {"priority": 0, "content": "Avoid null pointers."})
    # Agent immediately tries to submit a P2 tile (optimization) without P1 channel finding
    with pytest.raises(ProtocolViolationError) as excinfo:
        room.submit_tile(agent_id, {"priority": 2, "content": "Optimize loop unrolling."})
    assert "P1 channel required before P2" in str(excinfo.value)
    # Room state should remain at P0 for this agent
    assert room.get_agent_state(agent_id)["current_priority"] == 0
```

### Test 2: State Persistence Under Malformed Tile Flood
```python
def test_state_persistence_malformed_flood():
    """Room must maintain coherent state when flooded with malformed tiles."""
    room = DeadbandRoom()
    agent_id = "flood_agent"
    room.register_agent(agent_id)

    # Submit 100 malformed tiles (missing required fields)
    for i in range(100):
        try:
            room.submit_tile(agent_id, {"bad_key": i})
        except ValidationError:
            pass  # Expected

    # Valid tile submission should still work
    room.submit_tile(agent_id, {"priority": 0, "content": "Recovered."})
    state = room.get_agent_state(agent_id)
    assert state["current_priority"] == 0
    assert state["tiles_submitted"] == 1  # Only the valid tile counted
```

### Test 3: Concurrent Agent Priority Collision

def test_concurrent_agent_priority_collision():
    """Simulate two agents interacting, one at P1, one at P0, ensuring no state corruption."""
    room = DeadbandRoom()
    agent_a = "agent_a"
    agent_b = "agent_b"
    room.register_agent(agent_a)
    room.register_agent(agent_b)

    # Agent A works through P0 -> P1
    room.submit_tile(agent_a, {"priority": 0, "content": "P0 tile."})
    room.submit_tile(agent_a, {"priority": 1, "content": "P1 channel found."})

    # Agent B is still at P0
    room.submit_tile(agent_b, {"priority": 0, "content": "Another P0."})

    # Verify states are isolated
    assert room.get_agent_state(agent_a)["current_priority"] == 1
    assert room.get_agent_state(agent_b)["current_priority"] == 0
    # Room's global tile count should be 3
    assert room.get_global_tile_count() == 3
```

### Test 4: Ghost Tile Integration from Dead Agent

def test_ghost_tile_integration():
    """Room should accept and apply P0 ghost tiles from a 'dead' agent (simulated)."""
    room = DeadbandRoom()
    live_agent = "live_agent"
    dead_agent = "dead_agent"
    room.register_agent(live_agent)
    room.register_agent(dead_agent)

    # Simulate dead agent submitting a P0 ghost tile
    ghost_tile = {"priority": 0, "content": "Ghost: Avoid infinite recursion.", "ghost": True}
    room.submit_tile(dead_agent, ghost_tile)
    # Mark agent as dead (simulated via internal method)
    room._mark_agent_dead(dead_agent)

    # Live agent should see ghost tile as part of P0 knowledge base
    p0_knowledge = room.get_p0_knowledge()
    assert "Avoid infinite recursion." in p0_knowledge
    # Live agent can now submit P0 based on that knowledge
    room.submit_tile(live_agent, {"priority": 0, "content": "Also avoid recursion."})
    assert room.get_agent_state(live_agent)["current_priority"] == 0
```

### Test 5: Resource Exhaustion – Tile Buffer Limits
```python
def test_tile_buffer_limits():
    """Room should enforce configurable tile buffer limits per agent."""
    room = DeadbandRoom(max_tiles_per_agent=5)
    agent_id = "buffer_agent"
    room.register_agent(agent_id)

    # Submit 5 valid tiles (limit)
    for i in range(5):
        room.submit_tile(agent_id, {"priority": 0, "content": f"Tile {i}"})

    # 6th tile should raise BufferLimitError
    with pytest.raises(BufferLimitError) as excinfo:
        room.submit_tile(agent_id, {"priority": 0, "content": "Tile 6"})
    assert "tile buffer full" in str(excinfo.value)
    # Agent state should still be accessible
    assert room.get_agent_state(agent_id)["tiles_submitted"] == 5
```
