### Test 2: Concurrent Tile Submission Race Condition
```python
import threading

def test_concurrent_tile_submission():
    """Test that concurrent tile submissions don't corrupt room state."""
    room = DeadbandRoom(
        name="concurrency_test",
        channels=[{"name": "test_channel", "threshold": 1.0}],
        priority_sequence=["P0", "P1", "P2"]
    )
    
    results = []
    errors = []
    
    def submit_tile_thread(tile_data):
        try:
            result = room.submit_tile(tile_data)
            results.append(result)
        except Exception as e:
            errors.append(str(e))
    
    threads = []
    for i in range(10):
        tile = {"priority": "P0", "channel": "test_channel", "value": 0.5, "id": i}
        thread = threading.Thread(target=submit_tile_thread, args=(tile,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    assert len(errors) == 0, f"Errors during concurrent submission: {errors}"
    assert len(results) == 10, f"Expected 10 results, got {len(results)}"
    assert len(set(r["tile_id"] for r in results)) == 10, "Duplicate tile IDs detected"
```

```python
def test_memory_limit_enforcement():
    """Test that room respects memory limits for tile storage."""
    room = DeadbandRoom(
        name="memory_test",
        channels=[{"name": "test_channel", "threshold": 1.0}],
        priority_sequence=["P0", "P1", "P2"],
        max_tiles=5  # Small limit for testing
    )
    
    # Submit 5 tiles (should all succeed)
    for i in range(5):
        tile = {"priority": "P0", "channel": "test_channel", "value": 0.1, "id": i}
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # 6th tile should trigger cleanup or rejection
    tile = {"priority": "P0", "channel": "test_channel", "value": 0.1, "id": 5}
    result = room.submit_tile(tile)
    
    # Either old tiles are evicted or new tile is rejected
    assert result["status"] in ["accepted", "rejected"]
    if result["status"] == "accepted":
        assert len(room.get_tiles()) <= 5, "Memory limit exceeded"
```

```python
def test_recovery_from_corrupted_state():
    """Test that room can recover when underlying storage returns corrupted data."""
    room = DeadbandRoom(
        name="recovery_test",
        channels=[{"name": "test_channel", "threshold": 1.0}],
        priority_sequence=["P0", "P1", "P2"]
    )
    
    # Simulate corrupted storage by directly modifying internal state
    room._storage["tiles"] = "not a list"  # Corrupt the tile storage
    
    # Room should handle this gracefully on next operation
    try:
        tiles = room.get_tiles()
        # Either returns empty list or raises specific error
        assert tiles == [] or isinstance(tiles, list)
    except Exception as e:
        # Should raise a specific, catchable error, not crash
        assert "corrupt" in str(e).lower() or "invalid" in str(e).lower()
    
    # Should be able to continue after corruption
    tile = {"priority": "P0", "channel": "test_channel", "value": 0.5}
    result = room.submit_tile(tile)
    assert result["status"] in ["accepted", "rejected"]
```

```python
def test_integration_with_fleet_simulator():
    """Test that DeadbandRoom works with fleet simulator tile format."""
    # Simulate a tile from fleet simulator
    simulator_tile = {
        "priority": "P1",
        "channel": "navigation",
        "value": 0.7,
        "metadata": {
            "simulator_id": "fleet-sim-001",
            "timestamp": "2026-04-20T02:30:00Z",
            "agent_type": "navigator",
            "cycle": 175
        },
        "constraints": [
            {"type": "deadband", "min": 0.0, "max": 1.0},
            {"type": "priority", "level": "P1"}
        ]
    }
    
    room = DeadbandRoom(
        name="integration_test",
        channels=[{"name": "navigation", "threshold": 0.8}],
        priority_sequence=["P0", "P1", "P2"]
    )
    
    result = room.submit_tile(simulator_tile)
    
    # Should accept valid simulator tiles
    assert result["status"] == "accepted"
    
    # Should preserve simulator metadata
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == 1
    assert stored_tiles[0]["metadata"]["simulator_id"] == "fleet-sim-001"
    assert stored_tiles[0]["metadata"]["agent_type"] == "navigator"
```

5. **Integration**: Ensures compatibility with fleet simulator output format
