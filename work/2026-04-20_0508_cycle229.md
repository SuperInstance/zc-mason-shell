### Test 1: Empty Tile Processing
**Scenario:** What happens when a tile contains no content or null data?
**Rationale:** Ensures room handles malformed input without crashing.
```python
def test_process_empty_tile(self, deadband_room):
    """Test processing tile with empty content."""
    empty_tile = {
        'id': 'test_empty',
        'content': '',
        'priority': 'P0'
    }
    result = deadband_room.process_tile(empty_tile)
    # Should not crash; may return error or skip
    assert result is not None
    # Either processed or rejected cleanly
    assert 'error' in result or 'skipped' in result or 'processed' in result
```

### Test 2: Invalid Priority Level
**Scenario:** Tile has priority 'P3' or 'PX' (non-existent protocol level).
**Rationale:** Protocol only defines P0-P2. Tests boundary validation.
```python
def test_invalid_priority_level(self, deadband_room):
    """Test tile with priority outside P0-P2 range."""
    invalid_tile = {
        'id': 'test_invalid_pri',
        'content': 'Test content',
```

### Test 3: Concurrent Tile Processing
**Scenario:** Multiple tiles submitted simultaneously.
**Rationale:** DeadbandRoom may process tiles sequentially; need to verify no race conditions.
```python
def test_concurrent_tile_processing(self, deadband_room):
    """Test submitting multiple tiles in rapid succession."""
    tiles = [
        {'id': f'tile_{i}', 'content': f'Content {i}', 'priority': 'P0'}
        for i in range(5)
    ]
    results = []
    for tile in tiles:
        result = deadband_room.process_tile(tile)
        results.append(result)
    
    # All should be processed without interference
    assert len(results) == 5
    for i, result in enumerate(results):
        assert result['processed'] == True
        assert result['tile_id'] == f'tile_{i}'
```
