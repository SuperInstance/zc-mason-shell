### Test 4: Concurrent Tile Processing
```python
def test_concurrent_tile_processing(self):
    """Test processing multiple tiles in rapid succession."""
    import threading
    
    room = DeadbandRoom()
    results = []
    errors = []
    
    def process_tile_thread(tile_data):
        try:
            result = room.process_tile(tile_data)
            results.append(result)
        except Exception as e:
            errors.append(str(e))
    
    threads = []
    for i in range(10):
        tile = {
            'agent_id': f'test_agent_{i}',
            'content': f'P{i%3} test content for concurrent processing',
            'timestamp': f'2026-04-20T03:15:{i:02d}Z'
        }
        thread = threading.Thread(target=process_tile_thread, args=(tile,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Should process all tiles without race conditions
    assert len(errors) == 0
    assert len(results) == 10
    # All results should have valid structure
    for result in results:
        assert 'score' in result
        assert 0 <= result['score'] <= 100
```
