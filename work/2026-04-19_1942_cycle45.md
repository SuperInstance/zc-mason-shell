### Test 1: Concurrent Tile Submission Race Condition
```rust
#[test]
fn test_concurrent_tile_submission_race() {
    use std::sync::{Arc, Barrier};
    use std::thread;
    
    let room = DeadbandRoom::new();
    let room_arc = Arc::new(std::sync::Mutex::new(room));
    let barrier = Arc::new(Barrier::new(5));
    let mut handles = vec![];
    
    for i in 0..5 {
        let room_clone = Arc::clone(&room_arc);
        let barrier_clone = Arc::clone(&barrier);
        handles.push(thread::spawn(move || {
            barrier_clone.wait();
            let tile = Tile {
                agent_id: format!("agent_{}", i),
                content: format!("Test tile {}", i),
                priority: i % 3, // Mix of P0, P1, P2
                timestamp: SystemTime::now(),
            };
            let mut room = room_clone.lock().unwrap();
            room.submit_tile(tile)
        }));
    }
    
    let results: Vec<Result<(), DeadbandError>> = handles
        .into_iter()
        .map(|h| h.join().unwrap())
        .collect();
    
    // All submissions should succeed or fail gracefully
    assert!(results.iter().all(|r| r.is_ok() || matches!(r, Err(DeadbandError::QueueFull))));
    
    // Final state should be consistent
    let room = room_arc.lock().unwrap();
    assert!(room.validate_state());
}
```

### Test 3: Network Partition with Partial Tile Loss
```rust
#[test]
fn test_network_partition_tile_recovery() {
    let mut room = DeadbandRoom::new();
    
    // Simulate network partition mid-batch
    let tiles = vec![
        Tile { agent_id: "a1".to_string(), content: "tile1".to_string(), priority: 0, timestamp: SystemTime::now() },
        Tile { agent_id: "a2".to_string(), content: "tile2".to_string(), priority: 1, timestamp: SystemTime::now() },
        Tile { agent_id: "a3".to_string(), content: "tile3".to_string(), priority: 2, timestamp: SystemTime::now() },
    ];
    
    // Submit first tile successfully
    assert!(room.submit_tile(tiles[0].clone()).is_ok());
    
    // Simulate network failure
    room.simulate_network_partition(true);
    
    // Subsequent submissions should fail or queue
    let result2 = room.submit_tile(tiles[1].clone());
    assert!(result2.is_err() || matches!(result2, Ok(_)));
    
    // Restore network
    room.simulate_network_partition(false);
    
    // Room should recover and accept remaining tiles
    assert!(room.submit_tile(tiles[2].clone()).is_ok());
    
    // State should be consistent after recovery
    assert!(room.validate_state());
    assert_eq!(room.get_tile_count(), 3);
}
```

### Test 4: Malformed Priority Value Handling

```rust
#[test]
fn test_malformed_priority_handling() {
    let mut room = DeadbandRoom::new();
    
    // Test cases for priority field anomalies
    let test_cases = vec![
        (i32::MAX, DeadbandError::PriorityOutOfBounds),     // Overflow
        (i32::MIN, DeadbandError::PriorityOutOfBounds),     // Underflow
        (-1, DeadbandError::PriorityOutOfBounds),           // Negative
        (3, DeadbandError::PriorityOutOfBounds),            // Above P2
        (0, Ok(())),                                        // Valid P0
        (1, Ok(())),                                        // Valid P1
        (2, Ok(())),                                        // Valid P2
    ];
    
    for (priority, expected) in test_cases {
        let tile = Tile {
            agent_id: "test_agent".to_string(),
            content: format!("Priority {}", priority),
            priority,
            timestamp: SystemTime::now(),
        };
        
        let result = room.submit_tile(tile);
        match expected {
            Ok(_) => assert!(result.is_ok(), "Priority {} should be valid", priority),
            Err(expected_err) => {
                assert!(result.is_err(), "Priority {} should error", priority);
                assert_eq!(result.unwrap_err(), expected_err);
            }
        }
    }
    
    // Room state should remain valid
    assert!(room.validate_state());
}
```
