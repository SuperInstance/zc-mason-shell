### Test 1: Concurrent Protocol Phase Conflicts

```python
def test_concurrent_protocol_conflict(self):
    """Test multiple agents attempting conflicting protocol phases simultaneously."""
    room = DeadbandRoom()
    
    # Simulate two agents: one in P0, one trying to jump to P2
    agent1_state = {"phase": "P0", "agent_id": "agent1"}
    agent2_state = {"phase": "P2", "agent_id": "agent2"}
    
    # Both submit tiles concurrently
    tile1 = {"content": "P0 mapping", "phase": "P0"}
    tile2 = {"content": "P2 optimization", "phase": "P2"}
    
    # Room should reject agent2's tile (P2 without P1)
    result1 = room.submit_tile(agent1_state, tile1)
    result2 = room.submit_tile(agent2_state, tile2)
    
    assert result1["accepted"] == True
    assert result2["accepted"] == False
    assert "P1" in result2.get("error", "")  # Error should mention missing P1
```

### Test 4: Malformed Tile Content Types
```python
def test_malformed_tile_content_types(self):
    """Test submission of tiles with various malformed content types."""
    room = DeadbandRoom()
    agent_state = {"phase": "P0", "agent_id": "test_agent"}
    
    test_cases = [
        ("binary_data", b"raw_bytes_content"),
        ("none_content", None),
        ("integer_content", 42),
        ("list_content", ["not", "a", "dict"]),
        ("circular_reference", {}),
    ]
    
    # Create circular reference
    circular = {"self": None}
    circular["self"] = circular
    test_cases.append(("circular_dict", circular))
    
    for name, content in test_cases:
        tile = {"content": content, "phase": "P0"}
        result = room.submit_tile(agent_state, tile)
        
        # Room should not crash on any content type
        assert "error" in result or result["accepted"] == False or result["accepted"] == True
        # Specifically, binary and circular should likely be rejected
        if name in ["binary_data", "circular_reference"]:
            assert result["accepted"] == False
```

### Test 5: Protocol Phase Rollback Attempt
```python
def test_protocol_phase_rollback_rejection(self):
    """Test attempting to return to previous protocol phase after advancement."""
    room = DeadbandRoom()
    agent_state = {"phase": "P0", "agent_id": "agent1"}
    
    # Normal progression P0 → P1
    tile_p0 = {"content": "P0 work", "phase": "P0"}
    result_p0 = room.submit_tile(agent_state, tile_p0)
    assert result_p0["accepted"] == True
    
    # Advance to P1
    agent_state["phase"] = "P1"
    tile_p1 = {"content": "P1 work", "phase": "P1"}
    result_p1 = room.submit_tile(agent_state, tile_p1)
    assert result_p1["accepted"] == True
    
    # Attempt to submit P0 tile again (rollback)
    rollback_tile = {"content": "More P0 work", "phase": "P0"}
    result_rollback = room.submit_tile(agent_state, rollback_tile)
    
    # According to Deadband Protocol strict priority, rollback should be rejected
    assert result_rollback["accepted"] == False
    assert "rollback" in result_rollback.get("error", "").lower() or \
           "P0" in result_rollback.get("error", "").lower()
```
