## 5 Additional Edge-Case Tests

### Test 1: `test_empty_tile_rejection`
**Scenario:** Agent submits a tile with empty or null `interaction_data`.
**Edge:** The room should handle missing data without crashing and return a P0 violation.
```python
def test_empty_tile_rejection(self):
    """DeadbandRoom should reject tiles with empty interaction data."""
    room = DeadbandRoom()
    empty_tile = InteractionTile(agent_id="test_agent", interaction_data={}, timestamp=time.time())
    result = room.process_tile(empty_tile)
    self.assertEqual(result["status"], "rejected")
    self.assertEqual(result["phase"], "P0")
    self.assertIn("empty interaction data", result["reason"].lower())
```

### Test 2: `test_boundary_channel_detection`
**Scenario:** Tile data contains channel parameters at numerical boundaries (e.g., `confidence=0.0` or `confidence=1.0`).
**Edge:** P1 logic must correctly classify boundary values as safe/unsafe per protocol thresholds.
```python
def test_boundary_channel_detection(self):
    """P1 should correctly evaluate channels at confidence boundaries."""
    room = DeadbandRoom()
    boundary_tile = InteractionTile(
        agent_id="test_agent",
        interaction_data={"confidence": 0.0, "channel_type": "low_confidence_edge"},
        timestamp=time.time()
    )
    result = room.process_tile(boundary_tile)
    # Expect P0 rejection if confidence <= threshold, else P1 evaluation
    if room.config["confidence_threshold"] > 0.0:
        self.assertEqual(result["phase"], "P0")
    else:
        self.assertEqual(result["phase"], "P1")
```

### Test 3: `test_concurrent_tile_processing`
**Scenario:** Multiple agents submit tiles simultaneously.
**Edge:** Room's internal state (channel map, negative space) should remain consistent; no race conditions.
```python
def test_concurrent_tile_processing(self):
    """Room should handle concurrent tile submissions safely."""
    import threading
    room = DeadbandRoom()
    results = []
    def submit_tile(tile_data):
        tile = InteractionTile(agent_id="concurrent_agent", interaction_data=tile_data, timestamp=time.time())
        results.append(room.process_tile(tile))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=({"channel_id": f"channel_{i}"},))
        threads.append(t)
        t.start()
    for t in threads:
        t.join()
    
    # All tiles should be processed without exception
    self.assertEqual(len(results), 5)
    # State consistency check: negative_space should be a valid dict
    self.assertIsInstance(room.negative_space, dict)
```

### Test 5: `test_protocol_skip_detection`
**Scenario:** Agent tile attempts to skip P0 or P1 (e.g., claims "optimized path" without safe channel).
**Edge:** Room should detect protocol violation, reject tile at earliest possible phase, and log violation.
```python
def test_protocol_skip_detection(self):
    """Room should reject tiles that skip P0 or P1 phases."""
    room = DeadbandRoom()
    skip_tile = InteractionTile(
        agent_id="impatient_agent",
        interaction_data={
            "phase": "P2",  # Attempting to jump to optimization
            "optimized_path": {"steps": [1, 2, 3]},
            "channel_validated": False  # Missing P1 validation
        },
        timestamp=time.time()
    )
    result = room.process_tile(skip_tile)
    self.assertEqual(result["status"], "rejected")
    self.assertEqual(result["phase"], "P0")
    self.assertIn("protocol violation", result["reason"].lower())
    # Ensure negative_space is updated with this violation pattern
    self.assertIn("skip_p1", room.negative_space.get("protocol_violations", []))
```
