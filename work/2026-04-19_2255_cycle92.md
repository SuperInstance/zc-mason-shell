## 5 Additional Edge-Case Tests

### 1. Test: Empty Action List with Valid Protocol
**Scenario:** Agent submits empty action list but follows P0-P2 protocol structure.
**Expected:** Should pass P0 validation (no negative actions) but fail P1 (no safe channels identified).
```python
def test_empty_action_list_with_protocol(self):
    """Edge case: Empty action list with proper protocol structure."""
    agent_state = {
        "protocol_phase": "P0",
        "actions": [],
        "negative_space_mapped": True
    }
    result = self.room.validate_agent_step(agent_state)
    self.assertFalse(result["valid"])
    self.assertEqual(result["error"], "P1_FAILED: No safe channels identified")
```

### 2. Test: Concurrent Protocol Phase Violation
**Scenario:** Agent attempts P2 optimization while P1 safe channels are still being evaluated.
**Expected:** Should reject with P1_INCOMPLETE error before allowing P2.
```python
def test_concurrent_phase_violation(self):
    """Edge case: Agent tries P2 while P1 evaluation is ongoing."""
    agent_state = {
        "protocol_phase": "P2",
        "actions": ["optimize_channel_throughput"],
        "p1_evaluation_complete": False,
        "negative_space_mapped": True
    }
    result = self.room.validate_agent_step(agent_state)
    self.assertFalse(result["valid"])
    self.assertIn("P1_INCOMPLETE", result["error"])
```

### 3. Test: Maximum Tile Submission Rate
**Scenario:** Agent submits tiles at the maximum allowed rate (boundary condition).
**Expected:** Should accept valid tiles but reject if exceeding rate limit.
```python
def test_max_tile_submission_rate(self):
    """Edge case: Testing tile submission at rate limit boundary."""
    self.room.reset_submission_tracker()
    
    # Submit at exactly the rate limit
    for i in range(self.room.MAX_TILES_PER_MINUTE):
        tile = {
            "content": f"Test tile {i}",
            "protocol_phase": "P0",
            "timestamp": time.time()
        }
        result = self.room.submit_tile(tile)
        self.assertTrue(result["accepted"])
    
    # One more should be rejected
    tile = {
        "content": "Excess tile",
        "protocol_phase": "P0",
        "timestamp": time.time()
    }
    result = self.room.submit_tile(tile)
    self.assertFalse(result["accepted"])
    self.assertEqual(result["reason"], "RATE_LIMIT_EXCEEDED")
```

### 4. Test: Protocol Rollback Recovery
**Scenario:** Agent fails P2, needs to roll back to P1 without losing P0 knowledge.
**Expected:** Should preserve P0 negative space mapping while re-evaluating P1 channels.
```python
def test_protocol_rollback_recovery(self):
    """Edge case: Recovery from P2 failure with P0 preservation."""
    # Simulate P2 failure
    agent_state = {
        "protocol_phase": "P2",
        "actions": ["invalid_optimization"],
        "negative_space_mapped": True,
        "safe_channels": ["channel_a", "channel_b"]
    }
    result = self.room.validate_agent_step(agent_state)
    self.assertFalse(result["valid"])
    
    # Attempt recovery - roll back to P1
    recovery_state = {
        "protocol_phase": "P1",
        "actions": ["reevaluate_channel_b"],
        "negative_space_mapped": True,  # P0 preserved
        "safe_channels": ["channel_a"]  # One channel removed
    }
    result = self.room.validate_agent_step(recovery_state)
    self.assertTrue(result["valid"])
    self.assertTrue(recovery_state["negative_space_mapped"])
```

### 5. Test: Malformed Tile Content Injection
**Scenario:** Agent submits tiles with embedded escape sequences or malformed JSON.
**Expected:** Should sanitize content and reject dangerously malformed tiles.
```python
def test_malformed_tile_content(self):
    """Edge case: Tiles with injection attempts or malformed content."""
    malicious_tiles = [
        {"content": '"; DROP TABLE tiles; --', "protocol_phase": "P0"},
        {"content": '{"malicious": true}\x00', "protocol_phase": "P1"},
        {"content": "", "protocol_phase": "P2"},  # Empty content
        {"content": None, "protocol_phase": "P0"}  # Null content
    ]
    
    for tile in malicious_tiles:
        result = self.room.submit_tile(tile)
        # Should either sanitize or reject
        self.assertIn(result["status"], ["REJECTED", "SANITIZED"])
        if result["status"] == "SANITIZED":
            self.assertNotIn("\x00", result["sanitized_content"])
            self.assertNotIn("DROP TABLE", result["sanitized_content"])
```
