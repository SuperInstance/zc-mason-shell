### Test 2: Concurrent Submissions Race Condition
```python
def test_concurrent_tile_submissions(self):
    """P1: Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom()
    
    def submit_tile_parallel(agent_id, content, priority):
        tile = {"agent_id": agent_id, "content": content, "priority": priority}
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = [
            executor.submit(submit_tile_parallel, f"agent_{i}", f"Test content {i}", "P0")
            for i in range(3)
        ]
        results = [f.result() for f in futures]
    
    # All should be accepted if room handles concurrency
    accepted_count = sum(1 for r in results if r["accepted"])
    assert accepted_count == 3, f"Expected 3 accepted, got {accepted_count}"
    
    # Room state should be consistent
    assert len(room.get_tiles()) == 3
```

### Test 3: Ambiguous Safe Channel Definition

```python
def test_tile_with_ambiguous_p1_channel(self):
    """P1: Tile that identifies a safe channel but with conflicting boundary definitions should be flagged for review."""
    room = DeadbandRoom()
    
    # First, establish P0 negative space
    p0_tile = {
        "agent_id": "scout_agent",
        "content": "Avoid: Direct API calls without validation",
        "priority": "P0"
    }
    room.submit_tile(p0_tile)
    
    # P1 tile with ambiguous channel definition
    ambiguous_p1_tile = {
        "agent_id": "navigator_agent",
        "content": "Safe channel: Use either REST or GraphQL endpoints (conflicting)",
        "priority": "P1",
        "channel_boundaries": ["REST", "GraphQL"]  # Ambiguous - which one?
    }
    
    result = room.submit_tile(ambiguous_p1_tile)
    # Should be accepted but marked for review, not for direct training
    assert result["accepted"]
    assert result.get("needs_review", False)
    assert "ambiguous" in result.get("flags", [])
```

### Test 4: P2 Optimization Outside Established Channel
```python
def test_p2_outside_p1_channel(self):
    """P2: Optimization tile that references a channel not previously established in P1 should be rejected."""
    room = DeadbandRoom()
    
    # Establish P0 and P1 for "database queries"
    room.submit_tile({
        "agent_id": "sentinel",
        "content": "Avoid: Raw SQL concatenation",
        "priority": "P0"
    })
    room.submit_tile({
        "agent_id": "sentinel",
        "content": "Safe channel: Parameterized queries with connection pooling",
        "priority": "P1",
        "channel": "database_queries"
    })
    
    # P2 optimization for a different channel
    invalid_p2_tile = {
        "agent_id": "optimizer",
        "content": "Optimize: Use connection pooling for file I/O operations",
        "priority": "P2",
        "channel": "file_io"  # Not established in P1
    }
    
    result = room.submit_tile(invalid_p2_tile)
    assert not result["accepted"]
    assert "channel not established" in result["reason"].lower()
```

### Test 5: Resource Constraint Handling
```python
def test_tile_submission_under_resource_constraints(self):
    """P0: Room should handle tile submission when system resources are constrained (simulated)."""
    room = DeadbandRoom()
    
    # Simulate resource constraint by mocking memory check
    original_check = room._check_system_resources
    room._check_system_resources = lambda: {"memory_available": False, "cpu_available": True}
    
    try:
        tile = {
            "agent_id": "resource_aware_agent",
            "content": "P0: Avoid memory-intensive operations during peak load",
            "priority": "P0"
        }
        result = room.submit_tile(tile)
        
        # Room should either queue the tile or reject with resource reason
        if not result["accepted"]:
            assert "resource" in result["reason"].lower()
        else:
            assert result.get("queued", False)
    finally:
        room._check_system_resources = original_check
```
