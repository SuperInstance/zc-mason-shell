def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(capacity=10)
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (simulated with threads)
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append((tile["agent_id"], result))
        except Exception as e:
            results.append((tile["agent_id"], str(e)))
    
    threads = []
    for tile in tiles:
        thread = threading.Thread(target=submit_tile, args=(tile,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all tiles were processed
    self.assertEqual(len(results), 5)
    
    # Check for duplicates or lost tiles
    accepted = [r[0] for r in results if r[1] == "accepted"]
    self.assertEqual(len(accepted), 5)
    
    # Verify room state consistency
    self.assertEqual(room.tile_count, 5)
    self.assertEqual(len(room.get_tiles()), 5)
```

def test_protocol_violation_recovery(self):
    """Test agent skipping from P0 directly to P2 and room recovery."""
    room = DeadbandRoom()
    agent_id = "violating_agent"
    
    # Start at P0
    room.set_agent_phase(agent_id, "P0")
    
    # Violation: Submit P2 tile without P1
    violation_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Optimizing without safe channel",
        "timestamp": time.time()
    }
    
    # Should reject or downgrade
    result = room.submit_tile(violation_tile)
    self.assertIn(result, ["rejected", "downgraded"])
    
    if result == "downgraded":
        # Check tile was stored as P1
        tiles = room.get_agent_tiles(agent_id)
        self.assertEqual(tiles[-1]["phase"], "P1")
    
    # Verify agent can continue with correct protocol
    room.set_agent_phase(agent_id, "P1")
    valid_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Finding safe channels",
        "timestamp": time.time()
    }
    result = room.submit_tile(valid_tile)
    self.assertEqual(result, "accepted")
    
    # Agent state should reflect recovery
    self.assertEqual(room.get_agent_phase(agent_id), "P1")
```
