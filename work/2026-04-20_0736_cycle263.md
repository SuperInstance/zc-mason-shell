def test_concurrent_tile_submission_race():

    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "content": f"Test from {agent_id}",
            "priority": "P1",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (simulated)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(f"Error: {e}")
    
    # Verify: All tiles should be processed without corruption
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" or "Error" in r for r in results)
    # Room state should be consistent
    assert room.get_state()["tile_count"] == len([r for r in results if r.get("status") == "accepted"])
