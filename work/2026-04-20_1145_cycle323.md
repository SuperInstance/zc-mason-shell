### Test 1: Concurrent Tile Submission Stress Test

```python
def test_concurrent_tile_submission(self):
    """Test room handling of simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # All agents join and reach P1 state
    for agent_id in agent_ids:
        room.join(agent_id)
        room.submit_tile(agent_id, {"phase": "P0", "content": "negative space map"})
    
    # Concurrent P1 submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"safe channel from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "simultaneously" (in rapid succession)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile["agent_id"], tile)
        results.append(result)
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") in ["accepted", "rejected"] for r in results)
    
    # Room state should remain consistent
    assert room.get_state()["active_agents"] == len(agent_ids)
```

### Test 2: Protocol Regression Detection
```python
def test_protocol_regression_rejection(self):
    """Test that agents cannot regress from P2 to P0 without room reset."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    
    room.join(agent_id)
    
    # Normal progression: P0 → P1 → P2
    room.submit_tile(agent_id, {"phase": "P0", "content": "P0 work"})
    room.submit_tile(agent_id, {"phase": "P1", "content": "P1 work"})
    room.submit_tile(agent_id, {"phase": "P2", "content": "P2 work"})
    
    # Attempt regression: P2 → P0 (should be rejected)
    result = room.submit_tile(agent_id, {"phase": "P0", "content": "attempted regression"})
    
    assert result["status"] == "rejected"
    assert "protocol violation" in result.get("message", "").lower()
    assert room.get_agent_state(agent_id)["current_phase"] == "P2"
```

### Test 3: Malformed Metadata Handling

```python
def test_malformed_tile_metadata(self):
    """Test room handling of tiles with corrupted or missing metadata."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    room.join(agent_id)
    
    test_cases = [
        # Missing required fields
        {"content": "valid content"},  # missing phase
        {"phase": "P0"},  # missing content
        
        # Invalid data types
        {"phase": 123, "content": "content"},  # phase as int
        {"phase": "P0", "content": 456},  # content as int
        
        # Corrupted nested structures
        {"phase": "P0", "content": {"nested": "object"}},  # content as dict
        {"phase": "P0", "content": "valid", "metadata": "not_a_dict"},
        
        # Extreme values
        {"phase": "P0", "content": "x" * 10000},  # very large content
        {"phase": "P0" * 100, "content": "normal"},  # very large phase
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(agent_id, tile)
        assert result["status"] == "rejected", f"Test case {i} should be rejected"
        assert "invalid" in result.get("message", "").lower() or "missing" in result.get("message", "").lower()
```

### Test 4: Maximum Capacity Enforcement
```python
def test_room_capacity_limits(self):
    """Test room behavior when maximum agent capacity is reached."""
    room = DeadbandRoom()
    max_capacity = room.MAX_AGENTS  # Assuming this attribute exists
    
    # Fill room to capacity
    for i in range(max_capacity):
        agent_id = f"agent_{i}"
        result = room.join(agent_id)
        assert result["success"] == True, f"Should accept agent {i}"
    
    # Attempt to exceed capacity
    excess_agent = "excess_agent"
    result = room.join(excess_agent)
    
    assert result["success"] == False
    assert "capacity" in result.get("message", "").lower() or "full" in result.get("message", "").lower()
    
    # Verify existing agents can still operate
    existing_agent = "agent_0"
    tile_result = room.submit_tile(existing_agent, {"phase": "P0", "content": "still works"})
    assert tile_result["status"] == "accepted"
    
    # After one agent leaves, room should accept new agent
    room.leave("agent_0")
    result = room.join("replacement_agent")
    assert result["success"] == True
```

```python
def test_skip_p1_detection(self):
    """Test detection of agents skipping P1 and going directly to P2."""
    room = DeadbandRoom()
    agent_id = "skip_agent"
    
    room.join(agent_id)
    
    # Submit P0 work
    room.submit_tile(agent_id, {"phase": "P0", "content": "valid P0 work"})
    
    # Attempt to skip P1 and submit P2 work directly
    result = room.submit_tile(agent_id, {"phase": "P2", "content": "skipping P1"})
    
    assert result["status"] == "rejected"
    assert "p1" in result.get("message", "").lower() or "protocol" in result.get("message", "").lower()
    
    # Agent should remain in P0 state
    assert room.get_agent_state(agent_id)["current_phase"] == "P0"
    
    # Now submit proper P1 work
    result = room.submit_tile(agent_id, {"phase": "P1", "content": "proper P1 work"})
    assert result["status"] == "accepted"
    assert room.get_agent_state(agent_id)["current_phase"] == "P1"
```
