### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race_condition():
    """Test that concurrent tile submissions from multiple agents don't corrupt room state."""
    room = DeadbandRoom(tile_server=Mock(), agent_registry=Mock())
    room.initialize()
    
    # Simulate 5 agents submitting tiles simultaneously
    tiles = [
        {"agent_id": f"agent_{i}", "content": f"test_tile_{i}", "timestamp": time.time()}
        for i in range(5)
    ]
    
    # Use ThreadPoolExecutor to simulate concurrency
    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(room.submit_tile, tile) for tile in tiles]
        results = [f.result() for f in futures]
    
    # Verify all submissions were processed without corruption
    assert all(r["status"] == "accepted" for r in results)
    assert len(room.get_tile_history()) == 5
    assert len(set(t["agent_id"] for t in room.get_tile_history())) == 5
```

```python
def test_tile_server_network_partition_recovery():
    """Test room behavior when tile server becomes temporarily unreachable."""
    mock_tile_server = Mock()
    room = DeadbandRoom(tile_server=mock_tile_server, agent_registry=Mock())
    room.initialize()
    
    # First submission succeeds
    mock_tile_server.validate_tile.return_value = {"valid": True, "reason": "ok"}
    tile1 = {"agent_id": "agent_1", "content": "test", "timestamp": time.time()}
    result1 = room.submit_tile(tile1)
    assert result1["status"] == "accepted"
    
    # Simulate network partition - server raises ConnectionError
    mock_tile_server.validate_tile.side_effect = ConnectionError("Network unreachable")
    tile2 = {"agent_id": "agent_1", "content": "test2", "timestamp": time.time()}
    
    # Submission should fail gracefully
    with pytest.raises(RuntimeError, match="Tile server unreachable"):
        room.submit_tile(tile2)
    
    # Server recovers
    mock_tile_server.validate_tile.side_effect = None
    mock_tile_server.validate_tile.return_value = {"valid": True, "reason": "ok"}
    
    # Should accept new tiles after recovery
    result3 = room.submit_tile(tile2)
    assert result3["status"] == "accepted"
```

```python
def test_memory_pressure_tile_processing():
    """Test room behavior under memory pressure from large tile batches."""
    room = DeadbandRoom(tile_server=Mock(), agent_registry=Mock())
    room.initialize()
    
    # Generate 1000 large tiles (simulating memory pressure scenario)
    large_tiles = []
    for i in range(1000):
        large_content = "x" * 10000  # 10KB per tile
        large_tiles.append({
            "agent_id": f"agent_{i % 10}",
            "content": large_content,
            "timestamp": time.time() + i
        })
    
    processed_count = 0
    memory_samples = []
    
    for tile in large_tiles:
        # Sample memory usage before processing
        import psutil
        memory_samples.append(psutil.Process().memory_info().rss)
        
        try:
            result = room.submit_tile(tile)
            if result["status"] == "accepted":
                processed_count += 1
        except MemoryError:
            # Room should handle memory errors gracefully
            room._emergency_flush_tile_buffer()
            continue
    
    # Verify reasonable processing despite memory pressure
    assert processed_count > 500  # At least half processed
    assert len(room.get_tile_history()) == processed_count
    
    # Memory growth should be bounded (not exponential)
    if len(memory_samples) > 10:
        memory_increase = memory_samples[-1] - memory_samples[0]
        assert memory_increase < 100 * 1024 * 1024  # < 100MB increase for 1000 tiles
```

```python
def test_clock_skew_timestamp_validation():
    """Test tile validation with significant clock skew between agents."""
    mock_tile_server = Mock()
    room = DeadbandRoom(tile_server=mock_tile_server, agent_registry=Mock())
    room.initialize()
    
    current_time = time.time()
    
    # Test cases with various clock skew scenarios
    test_cases = [
        {"offset": -3600, "expected": "rejected", "reason": "future_tile"},  # 1 hour future
        {"offset": 3600, "expected": "accepted", "reason": "within_grace"},   # 1 hour past
        {"offset": -86400, "expected": "rejected", "reason": "distant_future"},  # 1 day future
        {"offset": 86400, "expected": "rejected", "reason": "too_old"},  # 1 day past
    ]
    
    for i, tc in enumerate(test_cases):
        skewed_time = current_time + tc["offset"]
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"test_{i}",
            "timestamp": skewed_time
        }
        
        # Configure mock based on expected outcome
        if tc["expected"] == "accepted":
            mock_tile_server.validate_tile.return_value = {"valid": True, "reason": "ok"}
        else:
            mock_tile_server.validate_tile.return_value = {
                "valid": False, 
                "reason": tc["reason"]
            }
        
        result = room.submit_tile(tile)
        assert result["status"] == tc["expected"]
        
        if tc["expected"] == "rejected":
            assert tc["reason"] in result.get("reason", "")
```
