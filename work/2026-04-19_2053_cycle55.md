### Test 1: Concurrent Tile Submission
```rust
#[tokio::test]
async fn test_concurrent_tile_submission() {
    let room = DeadbandRoom::new();
    let tile = Tile::new("agent_1", "Test concurrent tile");
    
    // Simulate 10 agents submitting simultaneously
    let mut handles = vec![];
    for i in 0..10 {
        let room_clone = room.clone();
        let tile_clone = tile.clone_with_agent(format!("agent_{}", i));
        handles.push(tokio::spawn(async move {
            room_clone.submit_tile(tile_clone).await
        }));
    }
    
    // All submissions should complete without deadlock
    let results = futures::future::join_all(handles).await;
    let successes = results.iter().filter(|r| r.is_ok()).count();
    
    // At least one should succeed (first to acquire lock)
    assert!(successes >= 1, "Concurrent submissions caused deadlock");
    
    // Verify state consistency
    assert!(room.current_state().is_ok(), "Room state corrupted by concurrent access");
}
```

### Test 2: State Persistence After Simulated Restart
```rust
#[test]
fn test_state_persistence() {
    let mut room = DeadbandRoom::new();
    
    // Advance through protocol phases
    room.process_tile(Tile::new("agent_1", "P0 mapping")).unwrap();
    assert_eq!(room.current_phase(), Phase::P0);
    
    room.process_tile(Tile::new("agent_2", "P1 channel finding")).unwrap();
    assert_eq!(room.current_phase(), Phase::P1);
    
    // Simulate restart: serialize and deserialize
    let serialized = serde_json::to_string(&room).unwrap();
    let restored_room: DeadbandRoom = serde_json::from_str(&serialized).unwrap();
    
    // State should be preserved
    assert_eq!(restored_room.current_phase(), Phase::P1);
    assert_eq!(restored_room.tile_count(), 2);
    
    // Should accept P2 tiles after restoration
    let result = restored_room.process_tile(Tile::new("agent_3", "P2 optimization"));
    assert!(result.is_ok(), "Restored room rejected valid P2 tile");
    assert_eq!(restored_room.current_phase(), Phase::P2);
}
```

### Test 3: Priority Regression Prevention
```rust
#[test]
fn test_priority_regression_rejected() {
    let mut room = DeadbandRoom::new();
    
    // Complete full protocol cycle
    room.process_tile(Tile::new("agent_1", "P0")).unwrap();
    room.process_tile(Tile::new("agent_2", "P1")).unwrap();
    room.process_tile(Tile::new("agent_3", "P2")).unwrap();
    
    assert_eq!(room.current_phase(), Phase::P2);
    
    // Attempt to submit P0 tile after reaching P2
    let result = room.process_tile(Tile::new("agent_4", "Attempt P0 regression"));
    
    // Should be rejected - cannot go backward in priorities
    assert!(result.is_err());
    assert_eq!(
        result.unwrap_err(),
        RoomError::InvalidPrioritySequence("Cannot regress from P2 to P0".to_string())
    );
    
    // Phase should remain at P2
    assert_eq!(room.current_phase(), Phase::P2);
}
```

### Test 4: Large Batch Processing Within Limits
```rust
#[tokio::test]
async fn test_large_batch_processing() {
    let room = DeadbandRoom::new();
    let start_memory = get_memory_usage();
    
    // Submit 1000 tiles (simulating training data import)
    let mut tasks = vec![];
    for i in 0..1000 {
        let room_clone = room.clone();
        let tile = Tile::new(
            format!("batch_agent_{}", i),
            &format!("Tile content {}", i).repeat(10) // 10x content size
        );
        tasks.push(tokio::spawn(async move {
            room_clone.submit_tile(tile).await
        }));
    }
    
    // Process with timeout to detect hangs
    let results = tokio::time::timeout(
        std::time::Duration::from_secs(5),
        futures::future::join_all(tasks)
    ).await.unwrap();
    
    // Check for resource exhaustion
    let end_memory = get_memory_usage();
    let memory_increase = end_memory - start_memory;
    
    // Memory increase should be reasonable (< 50MB for 1000 tiles)
    assert!(
        memory_increase < 50 * 1024 * 1024, // 50MB in bytes
        "Memory leak detected: increased by {} bytes",
        memory_increase
    );
    
    // All tasks should complete (success or rejection based on protocol)
    assert_eq!(results.len(), 1000);
}
```

```rust
#[test]
fn test_nonlinear_priority_sequence() {
    let mut room = DeadbandRoom::new();
    
    // Valid: P0 → P1
    room.process_tile(Tile::new("agent_1", "P0 mapping")).unwrap();
    assert_eq!(room.current_phase(), Phase::P0);
    
    room.process_tile(Tile::new("agent_2", "P1 channel")).unwrap();
    assert_eq!(room.current_phase(), Phase::P1);
    
    // Invalid: Try to jump back to P0 (P1 → P0)
    let result = room.process_tile(Tile::new("agent_3", "Attempt P0 after P1"));
    assert!(result.is_err());
    assert!(result.unwrap_err().to_string().contains("Cannot regress"));
    
    // Invalid: Try to skip to P2 without completing P1 (P1 → P2)
    // Actually this should be valid per protocol (P1 → P2 is allowed)
    // Let's test P0 → P2 skip instead
    let mut room2 = DeadbandRoom::new();
    room2.process_tile(Tile::new("agent_1", "P0")).unwrap();
    
    let result = room2.process_tile(Tile::new("agent_2", "Attempt P2 skip"));
    assert!(result.is_err());
    assert_eq!(
        result.unwrap_err(),
        RoomError::InvalidPrioritySequence("Cannot skip from P0 to P2".to_string())
    );
    
    // Edge case: Empty phase transition (same priority)
    let mut room3 = DeadbandRoom::new();
    room3.process_tile(Tile::new("agent_1", "P0")).unwrap();
    
    // Another P0 tile should be accepted (deepening P0 mapping)
    let result = room3.process_tile(Tile::new("agent_2", "More P0 mapping"));
    assert!(result.is_ok(), "Same-priority tiles should be accepted");
    assert_eq!(room3.current_phase(), Phase::P0);
}
```

```rust
// Helper function for memory testing
fn get_memory_usage() -> usize {
    #[cfg(target_os = "linux")]
    {
        use std::fs;
        let status = fs::read_to_string("/proc/self/status").unwrap();
        for line in status.lines() {
            if line.starts_with("VmRSS:") {
                let parts: Vec<&str> = line.split_whitespace().collect();
                return parts[1].parse::<usize>().unwrap() * 1024; // Convert kB to bytes
            }
        }
    }
    0 // Fallback for non-Linux
}
```
