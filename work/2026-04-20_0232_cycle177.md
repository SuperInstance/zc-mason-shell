### Test 3: Concurrent Tile Collisions
```python
def test_concurrent_tile_submission(self):
    """P1: Multiple agents submitting tiles simultaneously should not corrupt state."""
    room = DeadbandRoom()
    
    def submit_tile_async(agent_id, phase):
        tile = {
            "agent_id": agent_id,
            "phase": phase,
            "content": {"test": "data"},
            "timestamp": time.time()
        }
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    import threading
    results = []
    
    def worker(agent_id, phase):
        results.append(submit_tile_async(agent_id, phase))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=worker, args=(f"agent_{i}", "P0"))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All should be processed
    assert len(results) == 5
    
    # Check for duplicate agent state corruption
    agent_states = room.get_agent_states()
    assert len(agent_states) == 5  # All agents tracked separately
    
    # No data loss between agents
    for i in range(5):
        assert f"agent_{i}" in agent_states
```
