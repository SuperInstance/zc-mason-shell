## Implementation of Test 1: Concurrent Submission
```python
# test_deadband_room_edge.py
import threading
import time
import pytest
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_concurrent_tile_submission_at_capacity():
    """Edge-case: Multiple agents submitting tiles simultaneously at room capacity."""
    room = DeadbandRoom(capacity=2)
    room.current_phase = "P2"  # Assume room is active
    
    # Fill room to capacity
    room.submit_tile(agent_id="agent1", content="Tile 1", phase="P2")
    room.submit_tile(agent_id="agent2", content="Tile 2", phase="P2")
    assert room.tile_count == 2
    
    submission_results = []
    submission_lock = threading.Lock()
    
    def submit_tile(agent_id, content):
        try:
            result = room.submit_tile(agent_id=agent_id, content=content, phase="P2")
            with submission_lock:
                submission_results.append((agent_id, result))
        except Exception as e:
            with submission_lock:
                submission_results.append((agent_id, str(e)))
    
    # Start concurrent submissions
    threads = []
    for i in range(3):
        t = threading.Thread(
            target=submit_tile,
            args=(f"agent{i+3}", f"Concurrent tile {i+1}")
        )
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify results
    assert len(submission_results) == 3
    # Room at capacity: at least one should be rejected
    rejections = [r for _, r in submission_results if "capacity" in str(r).lower() or "full" in str(r).lower()]
    assert len(rejections) >= 1, "No rejections when room at capacity"
    # Room tile count should not exceed capacity
    assert room.tile_count == 2
```
