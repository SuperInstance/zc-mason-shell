def test_concurrent_tile_submission_race_condition():
    """P0: Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(config={"max_tiles": 1000})
    
    # Simulate 10 concurrent submissions
    tiles = [create_valid_tile(agent_id=f"agent_{i}") for i in range(10)]
    
    # Use threading to simulate concurrency
    results = []
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(e)
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify: All submissions processed, state consistent
    assert len(results) == 10
    assert room.get_tile_count() == 10
    assert room.get_state()["phase"] in ["P0", "P1", "P2"]  # Valid state
```

def test_state_persistence_after_partial_failure():
    """P1: Room should recover valid state after storage interruption."""
    room = DeadbandRoom(config={"persistence_path": "/tmp/test_deadband"})
    
    # Submit some valid tiles
    for i in range(5):
        room.submit_tile(create_valid_tile(agent_id=f"agent_{i}"))
    
    # Simulate partial write failure (e.g., disk full)
    original_save = room._save_state
    call_count = 0
    
    def failing_save(*args, **kwargs):
        nonlocal call_count
        call_count += 1
        if call_count == 3:  # Fail on third save attempt
            raise IOError("Simulated storage failure")
        return original_save(*args, **kwargs)
    
    room._save_state = failing_save
    
    # This submission should trigger failure
    try:
        room.submit_tile(create_valid_tile(agent_id="agent_fail"))
    except IOError:
        pass  # Expected
    
    # Restore normal save
    room._save_state = original_save
    
    # Verify: Room maintains previous valid state (4 tiles before failure)
    assert room.get_tile_count() == 4
    assert room.get_state()["phase"] == "P1"  # Should still be in valid phase
    assert "storage_error" in room.get_state().get("warnings", [])
```
