### Test 4: Protocol Reset After Max Iterations
```rust
#[tokio::test]
async fn test_deadband_reset_after_max_iterations() {
    let tile_store = MockTileStore::new();
    let config = RoomConfig {
        max_iterations: 3,
        ..Default::default()
    };
    
    let mut room = DeadbandRoom::new(tile_store, config);
    
    // Process 3 batches to reach max iterations
    for i in 0..3 {
        let tiles = vec![Tile::new(
            format!("tile_{}", i),
            "test_agent".to_string(),
            serde_json::json!({"phase": "P1", "content": format!("batch {}", i)})
        )];
        
        room.process_batch(tiles).await.unwrap();
    }
    
    // Next batch should reset protocol to P0
    let reset_tiles = vec![Tile::new(
        "reset_tile".to_string(),
        "test_agent".to_string(),
        serde_json::json!({"phase": "P2", "content": "should reset"})
    )];
    
    room.process_batch(reset_tiles).await.unwrap();
    
    assert_eq!(room.current_phase(), ProtocolPhase::P0);
    assert_eq!(room.iteration_count(), 0);
}
```

### Test 5: Concurrent Batch Processing
```rust
#[tokio::test]
async fn test_concurrent_batch_processing() {
    let tile_store = Arc::new(MockTileStore::new());
    let config = RoomConfig::default();
    
    let room = Arc::new(tokio::sync::Mutex::new(
        DeadbandRoom::new(tile_store.clone(), config)
    ));
    
    let mut handles = vec![];
    
    // Spawn 5 concurrent batch processors
    for i in 0..5 {
        let room_clone = room.clone();
        let handle = tokio::spawn(async move {
            let tiles = vec![Tile::new(
                format!("concurrent_tile_{}", i),
                "agent".to_string(),
                serde_json::json!({"phase": "P1", "sequence": i})
            )];
            
            let mut room = room_clone.lock().await;
            room.process_batch(tiles).await
        });
        handles.push(handle);
    }
    
    // Wait for all to complete
    let results = futures::future::join_all(handles).await;
    
    // Verify all succeeded
    for result in results {
        assert!(result.is_ok());
        assert!(result.unwrap().is_ok());
    }
    
    // Verify final state
    let final_room = room.lock().await;
    assert!(final_room.iteration_count() > 0);
}
```

    assert!(result.is_ok());
    assert_eq!(room.current_phase(), ProtocolPhase::P2);
}

### Test 7: Tile Store Failure Recovery
```rust
#[tokio::test]
async fn test_tile_store_failure_recovery() {
    let mut tile_store = MockTileStore::new();
    
    // Configure store to fail on third write
    let mut fail_counter = 0;
    tile_store.set_write_override(Box::new(move |_| {
        fail_counter += 1;
        if fail_counter == 3 {
            Err(anyhow::anyhow!("Simulated storage failure"))
        } else {
            Ok(())
        }
    }));
    
    let config = RoomConfig::default();
    let mut room = DeadbandRoom::new(tile_store, config);
    
    // Process batches that will trigger failure
    let mut all_succeeded = true;
    
    for i in 0..5 {
        let tiles = vec![Tile::new(
            format!("tile_{}", i),
            "agent".to_string(),
            serde_json::json!({"phase": "P1", "batch": i})
        )];
        
        match room.process_batch(tiles).await {
            Ok(_) => println!("Batch {} succeeded", i),
            Err(e) => {
                println!("Batch {} failed: {}", i, e);
                all_succeeded = false;
                // Room should continue processing despite storage error
            }
        }
    }
    
    // Verify room continued processing
    assert!(!all_succeeded, "Should have experienced at least one failure");
    assert!(room.iteration_count() > 0, "Should have processed some batches");
}
```

### Test 8: Protocol Phase Timeout
```rust
#[tokio::test]
async fn test_protocol_phase_timeout() {
    let tile_store = MockTileStore::new();
    let config = RoomConfig {
        phase_timeout_secs: 1, // 1 second timeout
        ..Default::default()
    };
    
    let mut room = DeadbandRoom::new(tile_store, config);
    
    // Start in P0
    assert_eq!(room.current_phase(), ProtocolPhase::P0);
    
    // Wait longer than timeout without processing
    tokio::time::sleep(tokio::time::Duration::from_secs(2)).await;
    
    // Try to process - should fail due to timeout
    let tiles = vec![Tile::new(
        "timeout_tile".to_string(),
        "agent".to_string(),
        serde_json::json!({"phase": "P1", "content": "too late"})
    )];
    
    let result = room.process_batch(tiles).await;
    
    // Should either fail or reset to P0
    match result {
        Err(e) => {
            assert!(e.to_string().contains("timeout") || e.to_string().contains("expired"));
        },
        Ok(_) => {
            // If it succeeds, should have reset to P0
            assert_eq!(room.current_phase(), ProtocolPhase::P0);
        }
    }
}
```
