### Test 3: Concurrent Submission – Race Condition Handling
```python
def test_concurrent_tile_submission(self):
    """Room should handle simultaneous submissions without corruption."""
    room = DeadbandRoom()
    room.config['concurrent_agents'] = 5
    
    def submit_tile(agent_id):
        tile = {
            'phase': 'P0',
            'content': {'negative_space': f'from_{agent_id}'},
            'agent_id': agent_id,
            'signature': f'sig_{agent_id}'
        }
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    import threading
    results = []
    threads = []
    for i in range(5):
        t = threading.Thread(target=lambda i=i: results.append(submit_tile(f'agent_{i}')))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All should be accepted or properly queued
    accepted = sum(1 for r in results if r.get('status') == 'accepted')
    assert accepted == 5
    assert len(room.state['tile_buffer']) == 5
```
