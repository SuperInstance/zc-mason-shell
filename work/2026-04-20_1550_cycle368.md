### Test 1: Concurrent Tile Submission Race Condition
```python
import asyncio
from unittest.mock import AsyncMock, patch
import pytest

@pytest.mark.asyncio
async def test_concurrent_tile_submission():
    """Test that room handles multiple simultaneous tile submissions without corruption."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom(room_id="edge-test-1")
    validator_mock = AsyncMock()
    validator_mock.validate_tile.return_value = {"valid": True, "score": 0.8}
    
    room.tile_validator = validator_mock
    
    # Simulate 5 agents submitting tiles at the same time
    tiles = [
        {"agent_id": f"agent-{i}", "phase": "P1", "content": f"test-{i}"}
        for i in range(5)
    ]
    
    tasks = [room.submit_tile(tile) for tile in tiles]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # All submissions should complete without exceptions
    assert not any(isinstance(r, Exception) for r in results)
    
    # Validator should be called exactly 5 times
    assert validator_mock.validate_tile.call_count == 5
    
    # Room state should reflect all submissions
    assert len(room.state.recent_tiles) == 5
    assert all(tile["validated"] for tile in room.state.recent_tiles)
```

### Test 2: Malformed Tile Payload Handling
```python
@pytest.mark.asyncio
async def test_malformed_tile_payloads():
    """Test room behavior with various invalid tile payloads."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from plato_torch.exceptions import InvalidTileError
    
    room = DeadbandRoom(room_id="edge-test-2")
    
    test_cases = [
        # (payload, expected_exception)
        (None, InvalidTileError),  # None payload
        ({}, InvalidTileError),  # Empty dict
        ({"agent_id": "test"}, InvalidTileError),  # Missing phase
        ({"phase": "P1"}, InvalidTileError),  # Missing agent_id
        ({"agent_id": 123, "phase": "P1"}, InvalidTileError),  # Wrong type
        ("not a dict", InvalidTileError),  # String instead of dict
        ({"agent_id": "test", "phase": "P99"}, ValueError),  # Invalid phase
    ]
    
    for payload, expected_exc in test_cases:
        with pytest.raises(expected_exc):
            await room.submit_tile(payload)
    
    # Validator should not be called for invalid payloads
    assert room.tile_validator.validate_tile.call_count == 0
```

### Test 3: Protocol Priority Violation Detection

```python
def test_protocol_priority_violation_detection():
    """Test that room detects and reports P2 attempts before P1 completion."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom(room_id="edge-test-3")
    
    # Simulate agent trying to submit P2 tile without P1 completion
    room.state.agent_progress = {"test-agent": {"current_phase": "P0"}}
    
    violation = room._check_protocol_violation(
        agent_id="test-agent",
        tile_phase="P2",
        tile_content="Attempting optimization before safe channels"
    )
    
    assert violation is True
    assert "P2 before P1" in room.state.violation_log[-1]["reason"]
    
    # Valid sequence should not trigger violation
    room.state.agent_progress["test-agent"]["current_phase"] = "P1"
    room.state.agent_progress["test-agent"]["p1_completed"] = True
    
    violation = room._check_protocol_violation(
        agent_id="test-agent",
        tile_phase="P2",
        tile_content="Valid optimization after P1 completion"
    )
    
    assert violation is False
```

### Test 4: Resource Exhaustion During State Persistence
```python
import tempfile
import os

def test_state_persistence_disk_full():
    """Test room behavior when disk is full during state save."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from plato_torch.exceptions import StatePersistenceError
    
    # Create a temporary directory and fill it
    with tempfile.TemporaryDirectory() as tmpdir:
        room = DeadbandRoom(room_id="edge-test-4", state_dir=tmpdir)
        
        # Fill the directory to simulate disk full
        large_file = os.path.join(tmpdir, "large.bin")
        with open(large_file, 'wb') as f:
            f.write(b'0' * (1024 * 1024 * 100))  # 100MB file
        
        # Try to save state - should raise appropriate error
        room.state.recent_tiles = [{"test": "tile"}]
        
        with pytest.raises(StatePersistenceError) as exc_info:
            room.save_state()
        
        assert "disk full" in str(exc_info.value).lower() or \
               "no space" in str(exc_info.value).lower()
        
        # State should remain in memory despite persistence failure
        assert len(room.state.recent_tiles) == 1
```

```python
@pytest.mark.asyncio
async def test_validator_service_unavailable():
    """Test room behavior when tile validator service is unavailable."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from plato_torch.exceptions import ValidatorUnavailableError
    
    room = DeadbandRoom(room_id="edge-test-5")
    
    # Mock validator to simulate service unavailable
    room.tile_validator.validate_tile.side_effect = ConnectionError("Service unavailable")
    
    tile = {
        "agent_id": "test-agent",
        "phase": "P1",
        "content": "Test content"
    }
    
    with pytest.raises(ValidatorUnavailableError):
        await room.submit_tile(tile)
    
    # Tile should be queued for retry
    assert len(room.state.pending_validation) == 1
    assert room.state.pending_validation[0]["tile"] == tile
    
    # Room should remain operational for other agents
    assert room.is_operational is True
```
