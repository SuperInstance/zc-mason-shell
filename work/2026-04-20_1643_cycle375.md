```python
def test_protocol_violation_p2_before_p1():
    """Agent attempts P2 optimization without completing P1 mapping."""
    room = DeadbandRoom(room_id="test_edge_1", max_tiles=100)
    agent_id = room.register_agent("test_agent")
    
    # Agent tries to submit P2 tile without P0/P1
    tile = {
        "phase": "P2",
        "content": "Optimizing channel",
        "timestamp": time.time()
    }
    
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.submit_tile(agent_id, tile)
    
    assert "P1 not completed" in str(exc_info.value)
    assert room.get_agent_state(agent_id)["current_phase"] == "P0"
```

### Test 2: Maximum Tile Capacity
```python
def test_max_tile_capacity():
    """Room rejects tiles when at maximum capacity."""
    max_tiles = 10
    room = DeadbandRoom(room_id="test_edge_2", max_tiles=max_tiles)
    agent_id = room.register_agent("test_agent")
    
    # Fill room to capacity
    for i in range(max_tiles):
        tile = {
            "phase": "P0",
            "content": f"Negative space mapping {i}",
            "timestamp": time.time()
        }
        room.submit_tile(agent_id, tile)
    
    # Attempt to submit one more tile
    overflow_tile = {
        "phase": "P1",
        "content": "Should be rejected",
        "timestamp": time.time()
    }
    
    with pytest.raises(RoomCapacityError) as exc_info:
        room.submit_tile(agent_id, overflow_tile)
    
    assert "maximum tile capacity" in str(exc_info.value)
    assert len(room.get_tiles()) == max_tiles
```

### Test 3: Concurrent Tile Submission
```python
def test_concurrent_tile_submission():
    """Multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(room_id="test_edge_3", max_tiles=100)
    agent_ids = [room.register_agent(f"agent_{i}") for i in range(3)]
    
    tiles_submitted = []
    
    def submit_tile_worker(agent_id, tile_data):
        try:
            room.submit_tile(agent_id, tile_data)
            tiles_submitted.append(True)
        except Exception:
            tiles_submitted.append(False)
    
    # Create concurrent submissions
    threads = []
    for i, agent_id in enumerate(agent_ids):
        tile = {
            "phase": "P0",
            "content": f"Concurrent tile {i}",
            "timestamp": time.time()
        }
        thread = threading.Thread(target=submit_tile_worker, args=(agent_id, tile))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # All submissions should succeed or fail gracefully
    assert all(tiles_submitted) or not any(tiles_submitted)
    assert room.get_tile_count() <= 3
```

### Test 4: Malformed Tile Data
```python
def test_malformed_tile_data():
    """Room handles invalid tile structures gracefully."""
    room = DeadbandRoom(room_id="test_edge_4", max_tiles=100)
    agent_id = room.register_agent("test_agent")
    
    test_cases = [
        {"missing_phase": "content"},  # No phase field
        {"phase": "P0"},  # No content
        {"phase": "P5", "content": "Invalid phase"},  # Invalid phase
        "not_a_dict",  # Wrong type
        None  # Null value
    ]
    
    for tile_data in test_cases:
        with pytest.raises((ValidationError, TypeError)) as exc_info:
            room.submit_tile(agent_id, tile_data)
        
        # Room should maintain state despite bad input
        assert room.get_agent_state(agent_id) is not None
```

### Test 5: Agent Reconnection State
```python
def test_agent_reconnection_state():
    """Agent disconnects and reconnects, room preserves phase progress."""
    room = DeadbandRoom(room_id="test_edge_5", max_tiles=100)
    agent_id = room.register_agent("test_agent")
    
    # Submit P0 tile
    p0_tile = {
        "phase": "P0",
        "content": "Negative space mapped",
        "timestamp": time.time()
    }
    room.submit_tile(agent_id, p0_tile)
    
    # Simulate disconnection (agent removed from active sessions)
    room._active_sessions.pop(agent_id, None)
    
    # Reconnect with same agent ID
    reconnected = room.register_agent("test_agent", existing_id=agent_id)
    assert reconnected == agent_id
    
    # Agent should be in P1 phase (next after P0 completion)
    state = room.get_agent_state(agent_id)
    assert state["current_phase"] == "P1"
    assert state["completed_phases"] == ["P0"]
```
