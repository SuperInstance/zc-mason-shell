import unittest
from unittest.mock import patch, MagicMock
from datetime import datetime, timedelta
import json

from plato_torch.rooms.deadband_room import DeadbandRoom


class TestDeadbandRoomEdgeCases(unittest.TestCase):
    
    def setUp(self):
        """Set up test fixture."""
        self.room = DeadbandRoom(room_id="test_edge_room")
        self.agent_id = "test_agent_edge"
        self.session_id = "test_session_edge"
        
    def test_p2_optimization_with_no_p1_channels(self):
        """Edge: Agent attempts P2 optimization when no P1 channels exist.
        
        Scenario: Agent has mapped P0 (negative space) but found zero
        safe channels. According to protocol, P2 should not be reachable.
        Expected: Reward penalty for protocol violation.
        """
        # Setup: Agent has only P0 knowledge, zero P1 channels
        self.room.state['agents'][self.agent_id] = {
            'p0_mapped': ['rock1', 'rock2'],
            'p1_channels': [],  # Empty - no safe channels
            'current_phase': 'P0',
            'violation_count': 0
        }
        
        # Action: Agent tries P2 optimization anyway
        action = {
            'type': 'P2_OPTIMIZE',
            'target': 'efficiency',
            'tile_data': {'test': 'data'}
        }
        
        result = self.room.process_action(
            agent_id=self.agent_id,
            session_id=self.session_id,
            action=action
        )
        
        # Assert: Should be violation (trying P2 without P1)
        self.assertLess(result['reward'], 0, 
                       "P2 attempt without P1 channels should yield negative reward")
        self.assertIn('protocol_violation', result['state']['last_error'])
        
    def test_consecutive_p0_violations_penalty_accumulation(self):
        """Edge: Multiple P0 violations in sequence.
        
        Scenario: Agent repeatedly hits known rocks (P0 violations).
        Expected: Penalty should increase with each violation.
        """
        violations = []
        
        # First violation
        self.room.state['agents'][self.agent_id] = {
            'p0_mapped': ['rock1'],
            'p1_channels': ['channel1'],
            'current_phase': 'P0',
            'violation_count': 0
        }
        
        action = {
            'type': 'P0_VIOLATION',  # Hitting known rock
            'rock_id': 'rock1',
            'tile_data': {}
        }
        
        result1 = self.room.process_action(
            agent_id=self.agent_id,
            session_id=self.session_id,
            action=action
        )
        violations.append(result1['reward'])
        
        # Second violation (same rock, same session)
        result2 = self.room.process_action(
            agent_id=self.agent_id,
            session_id=self.session_id,
            action=action
        )
        violations.append(result2['reward'])
        
        # Assert: Second penalty should be more severe
        self.assertLess(violations[1], violations[0],
                       "Consecutive violations should have increasing penalties")
        self.assertEqual(self.room.state['agents'][self.agent_id]['violation_count'], 2)
        
    def test_mixed_protocol_steps_single_action(self):
        """Edge: Agent submits action containing both P0 and P2 elements.
        
        Scenario: Single action JSON tries to map negative space AND optimize.
        Expected: Room should reject mixed protocol actions.
        """
        action = {
            'type': 'MIXED_PROTOCOL',
            'p0_map': ['rock1', 'rock2'],
            'p2_optimize': 'efficiency',
            'tile_data': {'mixed': 'action'}
        }
        
        result = self.room.process_action(
            agent_id=self.agent_id,
            session_id=self.session_id,
            action=action
        )
        
        # Assert: Should be rejected as invalid action type
        self.assertLess(result['reward'], 0)
        self.assertIn('invalid_action', result['state']['last_error'].lower())
        
    def test_maximum_tile_size_boundary(self):
        """Edge: Submit tile at exact maximum size limit.
        
        Scenario: Tile data size equals room's MAX_TILE_SIZE.
        Expected: Should be accepted (not exceed limit).
        """
        # Get max size from room config (assuming it's accessible)
        max_size = getattr(self.room, 'MAX_TILE_SIZE', 1024 * 1024)  # Default 1MB
        
        # Create tile data at exact boundary
        boundary_data = {'content': 'x' * (max_size - 100)}  # Approximate
        # Note: Actual test would need serialized JSON size calculation
        
        action = {
            'type': 'P1_EXPLORE',
            'channel_id': 'test_channel',
            'tile_data': boundary_data
        }
        
        # Mock the size check if needed
        with patch.object(self.room, '_validate_tile_size') as mock_validate:
            mock_validate.return_value = True  # Assume passes
            
            result = self.room.process_action(
                agent_id=self.agent_id,
                session_id=self.session_id,
                action=action
            )
            
            # Assert: Should process normally
            self.assertGreaterEqual(result['reward'], 0)
            mock_validate.assert_called_once()
            
    def test_session_expiration_mid_cycle(self):
        """Edge: Session expires during action processing.
        
        Scenario: Valid session at start, expires during tile validation.
        Expected: Action should be rejected with session expired error.
        """
        # Create room with short session timeout
        room = DeadbandRoom(
            room_id="test_expire_room",
            session_timeout_seconds=0.1  # Very short
        )
        
        # Setup initial state
        room.state['agents'][self.agent_id] = {
            'p0_mapped': [],
            'p1_channels': ['channel1'],
            'current_phase': 'P1',
            'violation_count': 0
        }
        
        # Create session
        session_id = room.create_session(self.agent_id)
        
        # Wait for expiration
        import time
        time.sleep(0.2)
        
        action = {
            'type': 'P1_EXPLORE',
            'channel_id': 'channel1',
            'tile_data': {'test': 'data'}
        }
        
        result = room.process_action(
            agent_id=self.agent_id,
            session_id=session_id,
            action=action
        )
        
        # Assert: Should reject expired session
        self.assertLess(result['reward'], 0)
        self.assertIn('expired', result['state']['last_error'].lower())


if __name__ == '__main__':
    unittest.main()
