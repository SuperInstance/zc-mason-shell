# plato-torch/tests/test_deadband_room_edge_cases.py
"""
Edge-case tests for DeadbandRoom protocol implementation.
Tests scenarios beyond normal operation to ensure robustness.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
import json
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import TrainingTile


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""
    
    @pytest.mark.asyncio
    async def test_concurrent_tile_submission(self):
        """Test multiple agents submitting tiles simultaneously.
        
        Scenario: 3 agents submit tiles at the same moment.
        Expected: All tiles processed without corruption or race conditions.
        """
        room = DeadbandRoom(room_id="edge-test-1")
        
        # Create 3 agents with different tile data
        agents = [
            {"agent_id": "agent-1", "tile_data": {"phase": "P0", "content": "Avoid rocks"}},
            {"agent_id": "agent-2", "tile_data": {"phase": "P1", "content": "Find channel"}},
            {"agent_id": "agent-3", "tile_data": {"phase": "P2", "content": "Optimize path"}},
        ]
        
        async def submit_tile(agent_data):
            tile = TrainingTile(
                agent_id=agent_data["agent_id"],
                content=json.dumps(agent_data["tile_data"]),
                timestamp=room._get_timestamp()
            )
            return await room.submit_tile(tile)
        
        # Submit all tiles concurrently
        tasks = [submit_tile(agent) for agent in agents]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Verify all submissions succeeded
        assert all(isinstance(r, bool) for r in results), "Some submissions failed"
        assert all(results), "Not all concurrent submissions succeeded"
        
        # Verify tiles are stored correctly
        stored_tiles = await room.get_recent_tiles(limit=10)
        assert len(stored_tiles) == 3, f"Expected 3 tiles, got {len(stored_tiles)}"
        
        # Verify no data corruption
        tile_contents = [json.loads(tile.content) for tile in stored_tiles]
        expected_contents = [agent["tile_data"] for agent in agents]
        assert all(ec in tile_contents for ec in expected_contents), "Tile content corrupted"
    
    @pytest.mark.asyncio
    async def test_protocol_violation_recovery(self):
        """Test room recovery after protocol violation (P0→P2 skip).
        
        Scenario: Agent attempts illegal P0→P2 transition, then corrects.
        Expected: Room rejects violation, allows correction, maintains state.
        """
        room = DeadbandRoom(room_id="edge-test-2")
        
        # Initial P0 tile (valid)
        tile_p0 = TrainingTile(
            agent_id="violator-agent",
            content=json.dumps({"phase": "P0", "lesson": "Don't skip phases"}),
            timestamp=room._get_timestamp()
        )
        assert await room.submit_tile(tile_p0), "P0 submission should succeed"
        
        # Attempt illegal P0→P2 skip
        tile_p2_illegal = TrainingTile(
            agent_id="violator-agent",
            content=json.dumps({"phase": "P2", "lesson": "Trying to optimize too soon"}),
            timestamp=room._get_timestamp()
        )
        
        # Room should reject this
        with pytest.raises(ValueError, match="Protocol violation"):
            await room.submit_tile(tile_p2_illegal)
        
        # Verify room state is still P0
        current_state = await room.get_current_state()
        assert current_state["current_phase"] == "P0", "Room should remain in P0 after violation"
        
        # Submit valid P1 tile (recovery)
        tile_p1 = TrainingTile(
            agent_id="violator-agent",
            content=json.dumps({"phase": "P1", "lesson": "Finding safe channel after error"}),
            timestamp=room._get_timestamp()
        )
        assert await room.submit_tile(tile_p1), "Recovery P1 submission should succeed"
        
        # Now P2 should be allowed
        tile_p2_valid = TrainingTile(
            agent_id="violator-agent",
            content=json.dumps({"phase": "P2", "lesson": "Now optimizing correctly"}),
            timestamp=room._get_timestamp()
        )
        assert await room.submit_tile(tile_p2_valid), "Valid P2 submission should succeed"
    
    @pytest.mark.asyncio

        """Test room handling of malformed tile data.
        
        Scenarios:
        1. Missing required fields
        2. Invalid JSON content
        3. Excessively large content
        4. Wrong data types
        """
        room = DeadbandRoom(room_id="edge-test-3")
        
        test_cases = [
            # (tile_data, expected_error, description)
            (
                {"agent_id": "test-agent"},  # Missing content
                KeyError,
                "Tile missing required 'content' field"
            ),
            (
                {"agent_id": "test-agent", "content": "{invalid json"},
                json.JSONDecodeError,
                "Invalid JSON content"
            ),
            (
                {"agent_id": "test-agent", "content": "x" * 1000000},  # 1MB content
                ValueError,
                "Content exceeds size limit"
            ),
            (
                {"agent_id": 123, "content": json.dumps({"phase": "P0"})},  # Wrong agent_id type
                TypeError,
                "agent_id must be string"
            ),
        ]
        
        for tile_data, expected_error, description in test_cases:
            with pytest.subTest(description=description):
                tile = TrainingTile(**tile_data) if "content" in tile_data else None
                
                if tile:
                    with pytest.raises(expected_error):
                        await room.submit_tile(tile)
                else:
                    with pytest.raises(expected_error):
                        TrainingTile(**tile_data)  # Should fail during construction
