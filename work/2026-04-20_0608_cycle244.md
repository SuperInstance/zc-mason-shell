### Test 1: Empty State Transition Matrix
**Scenario:** DeadbandRoom initialized with empty state transition matrix
**Edge Case:** Protocol phases with no defined state transitions
**Expected Behavior:** Room should handle gracefully with default transitions or clear error
```python
def test_empty_transition_matrix():
    """Test DeadbandRoom with empty state transition matrix."""
    room = DeadbandRoom(
        state_transitions={},  # Empty matrix
        initial_state="idle",
        max_iterations=100
    )
    # Should either use default transitions or raise informative error
    with pytest.raises(ValueError) as exc_info:
        room.run_protocol()
    assert "state transition" in str(exc_info.value).lower()
```

### Test 2: Circular State Dependencies
**Scenario:** State transitions form a cycle that could cause infinite loops
**Edge Case:** P0→P1→P2→P0 cycle within max_iterations
**Expected Behavior:** Room should detect cycles or respect max_iterations limit
```python
def test_circular_state_dependencies():
    """Test protocol with circular state dependencies."""
    circular_transitions = {
        "P0": {"next": "P1", "condition": lambda x: True},
        "P1": {"next": "P2", "condition": lambda x: True},
        "P2": {"next": "P0", "condition": lambda x: True}
    }
    room = DeadbandRoom(
        state_transitions=circular_transitions,
        initial_state="P0",
        max_iterations=5  # Small limit to test termination
    )
    result = room.run_protocol()
    # Should complete within max_iterations
    assert result["iterations"] <= 5
    assert result["termination_reason"] == "max_iterations_reached"
```

### Test 3: Concurrent Protocol Execution
**Scenario:** Multiple agents attempting to run protocol simultaneously
**Edge Case:** Race conditions in state management
**Expected Behavior:** Thread-safe state transitions or clear concurrency error
```python
def test_concurrent_protocol_execution():
    """Test multiple concurrent protocol executions."""
    room = DeadbandRoom(
        state_transitions=standard_transitions,
        initial_state="idle",
        max_iterations=10
    )
    
    def run_protocol_agent(agent_id):
        try:
            return room.run_protocol(context={"agent_id": agent_id})
        except Exception as e:
            return {"agent_id": agent_id, "error": str(e)}
    
    # Simulate concurrent execution
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = [executor.submit(run_protocol_agent, i) for i in range(3)]
        results = [f.result() for f in futures]
    
    # Should either all succeed with proper locking or fail with concurrency error
    errors = [r for r in results if "error" in r]
    if errors:
        # If errors occur, they should be about concurrency
        assert any("concurrent" in e["error"].lower() for e in errors)
```

### Test 4: Memory Exhaustion Scenario
**Scenario:** Protocol generates excessive state history
**Edge Case:** Memory consumption during long-running protocols
**Expected Behavior:** Memory management or early termination
```python
def test_memory_exhaustion_scenario():
    """Test protocol behavior under memory pressure."""
    # Create transitions that generate large state history
    memory_intensive_transitions = {
        "start": {
            "next": "generate",
            "condition": lambda ctx: True,
            "action": lambda ctx: ctx.update({"data": "x" * 1000000})  # 1MB per state
        },
        "generate": {
            "next": "start",
            "condition": lambda ctx: len(ctx.get("history", [])) < 50,
            "action": lambda ctx: ctx["history"].append(ctx.get("data", ""))
        }
    }
    
    room = DeadbandRoom(
        state_transitions=memory_intensive_transitions,
        initial_state="start",
        max_iterations=100
    )
    
    # Monitor memory usage
    import tracemalloc
    tracemalloc.start()
    
    try:
        result = room.run_protocol()
        current, peak = tracemalloc.get_traced_memory()
        tracemalloc.stop()
        
        # Protocol should complete or terminate before excessive memory use
        assert peak < 200 * 1024 * 1024  # Less than 200MB peak
    except MemoryError:
        # MemoryError is acceptable if properly handled
        assert room.current_state == "error"
```

### Test 5: Partial Protocol Completion Recovery
**Scenario:** Protocol interrupted mid-execution (simulated crash)
**Edge Case:** Room recovery from partial state
**Expected Behavior:** Clean recovery or consistent error state
```python
def test_partial_protocol_recovery():
    """Test recovery after protocol interruption."""
    room = DeadbandRoom(
        state_transitions=standard_transitions,
        initial_state="P0",
        max_iterations=10
    )
    
    # Simulate partial execution
    room.run_protocol()
    mid_state = room.current_state
    mid_context = room.context.copy()
    
    # Simulate interruption (e.g., crash, restart)
    room2 = DeadbandRoom(
        state_transitions=standard_transitions,
        initial_state=mid_state,
        initial_context=mid_context,
        max_iterations=10
    )
    
    # Continue protocol from interrupted state
    result = room2.run_protocol()
    
    # Should complete successfully from interruption point
    assert result["completed"] == True
    assert result["initial_state"] == mid_state
    # Context should preserve data from before interruption
    assert all(k in result["final_context"] for k in mid_context.keys())
```
