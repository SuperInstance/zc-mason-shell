def test_concurrent_tile_submission_race_condition():
    """Test that concurrent tile submissions from multiple agents don't corrupt room state."""
    room = DeadbandRoom(room_id="race-test")
    
    # Simulate 3 agents submitting tiles simultaneously
    tiles = [
        {"agent_id": "agent1", "content": "P0: don't crash", "phase": "P0"},
        {"agent_id": "agent2", "content": "P1: safe path", "phase": "P1"},
        {"agent_id": "agent3", "content": "P2: optimize", "phase": "P2"}
    ]
    
    # Submit tiles concurrently (simulated with threads)
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append((tile["agent_id"], result))
        except Exception as e:
            results.append((tile["agent_id"], str(e)))
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify: All tiles should be accepted or properly rejected
    assert len(results) == 3
    # State should remain consistent
    assert room.get_state()["tile_count"] <= 3
    # No duplicate agent submissions should corrupt state
    agent_ids = [r[0] for r in results]
    assert len(set(agent_ids)) == 3
```

def test_network_partition_during_p1_to_p2_transition():
    """Test room behavior when network partition occurs during protocol phase transition."""
    room = DeadbandRoom(room_id="partition-test")
    
    # Mock network failure during state persistence
    original_save = room._save_state_to_storage
    
    partition_triggered = False
    
    def failing_save(state):
        nonlocal partition_triggered
        if not partition_triggered and room.current_phase == "P1":
            partition_triggered = True
            raise ConnectionError("Network partition: storage unavailable")
        return original_save(state)
    
    room._save_state_to_storage = failing_save
    
    # Setup: Complete P0, start P1
    room.current_phase = "P1"
    room.negative_space = {"crash": True, "timeout": True}
    
    # Trigger P1 → P2 transition (should fail due to partition)
    with pytest.raises(ConnectionError):
        room.complete_phase("P1")
    
    # Verify: Room should remain in P1, not corrupted
    assert room.current_phase == "P1"
    assert partition_triggered is True
    
    # Restore network and retry
    room._save_state_to_storage = original_save
    room.complete_phase("P1")
    
    # Should now transition to P2
    assert room.current_phase == "P2"
```

def test_malformed_tile_data_injection():
    """Test room resilience against malformed or malicious tile data."""
    room = DeadbandRoom(room_id="injection-test")
    
    test_cases = [
        # (tile_data, expected_behavior)
        ({"agent_id": None, "content": None}, "reject"),
        ({"agent_id": "evil", "content": "<script>alert('xss')</script>"}, "sanitize"),
        ({"agent_id": "a" * 1000, "content": "normal"}, "reject"),
        ({"agent_id": "normal", "content": {"nested": "object"}}, "reject"),
        ({"agent_id": "normal", "content": "valid", "extra_field": "malicious"}, "ignore_extra"),
    ]
    
    for tile_data, expected in test_cases:
        try:
            result = room.submit_tile(tile_data)
            
            if expected == "reject":
                assert result.get("status") == "rejected"
                assert "error" in result
            elif expected == "sanitize":
                assert result.get("status") == "accepted"
                # Content should be sanitized
                content = result.get("content", "")
                assert "<script>" not in content
            elif expected == "ignore_extra":
                assert result.get("status") == "accepted"
                # Extra field should not be in stored tile
                stored = room.get_tile_by_agent("normal")
                assert "extra_field" not in stored
        except Exception as e:
            # Some malformed data should raise validation errors
            if expected != "reject":
                pytest.fail(f"Unexpected error for {tile_data}: {e}")
    
    # Final state should be clean
    assert room.get_state()["integrity_check"] is True
```

def test_resource_exhaustion_handling():
    """Test room behavior when system resources are exhausted."""
    room = DeadbandRoom(room_id="resource-test")
    
    # Mock memory allocation failure
    import sys
    
    original_tile_storage = room._tile_storage
    
    memory_exhausted = False
    
    class ExhaustedStorage:
        def add_tile(self, tile):
            nonlocal memory_exhausted
            if not memory_exhausted and len(self) > 100:  # Simulate memory limit
                memory_exhausted = True
                raise MemoryError("Out of memory")
            return original_tile_storage.add_tile(tile)
        
        def __len__(self):
            return len(original_tile_storage)
        
        def __getattr__(self, name):
            return getattr(original_tile_storage, name)
    
    room._tile_storage = ExhaustedStorage()
    
    # Submit tiles until exhaustion
    submitted = 0
    for i in range(150):  # More than limit
        try:
            room.submit_tile({
                "agent_id": f"agent{i}",
                "content": f"Test tile {i}",
                "phase": "P0"
            })
            submitted += 1
        except MemoryError:
            # Should trigger graceful degradation
            assert memory_exhausted is True
            # Room should enter safe mode
            assert room.safe_mode is True
            break
    
    # Verify: Some tiles accepted before exhaustion
    assert submitted > 0
    assert submitted <= 100
    
    # Room should still be operational in safe mode
    assert room.get_state()["operational"] is True
    
    # Cleanup should work
    room.cleanup_old_tiles(keep_last=50)
    assert len(room._tile_storage) <= 50
```
