### Test 2: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """P1: Room should handle simultaneous submissions from multiple agents."""
    room = DeadbandRoom(room_id="edge-test-2")
    agent_ids = [room.register_agent(f"agent-{i}") for i in range(3)]
    
    # Simulate concurrent submissions
    tiles = []
    for i, agent_id in enumerate(agent_ids):
        tile = {
            "cycle": i,
            "content": f"Test content from agent {i}",
            "priority": i % 3,  # Mix of P0, P1, P2
            "metadata": {"source": f"agent-{i}"}
        }
        tiles.append((agent_id, tile))
    
    # Submit all tiles
    results = []
    for agent_id, tile in tiles:
        try:
            result = room.submit_tile(agent_id, tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    self.assertEqual(len(results), 3)
    self.assertTrue(all(r.get("accepted", False) for r in results if isinstance(r, dict)))
```

### Test 3: Multi-Cycle State Persistence
```python
def test_multi_cycle_state_persistence(self):
    """P1: Room should maintain agent state across protocol cycles."""
    room = DeadbandRoom(room_id="edge-test-3")
    agent_id = room.register_agent("persistent-agent")
    
    # Run 5 complete protocol cycles
    for cycle in range(5):
        # Execute P0, P1, P2 for this agent
        room.execute_protocol_step(agent_id, "P0")
        room.execute_protocol_step(agent_id, "P1")
        room.execute_protocol_step(agent_id, "P2")
        
        # Submit a tile each cycle
        tile = {
            "cycle": cycle,
            "content": f"Cycle {cycle} completion",
            "priority": cycle % 3,
            "metadata": {"cycle": cycle}
        }
        room.submit_tile(agent_id, tile)
    
    # Verify agent's history is preserved
    agent_state = room.get_agent_state(agent_id)
    self.assertEqual(len(agent_state.get("submitted_tiles", [])), 5)
    self.assertEqual(agent_state.get("completed_cycles"), 5)
    
    # Verify tiles are in chronological order
    tiles = agent_state.get("submitted_tiles", [])
    for i, tile in enumerate(tiles):
        self.assertEqual(tile["metadata"]["cycle"], i)
```

def test_edge_agent_behaviors(self):
    """P0/P1: Room should handle silent, rapid, and malformed agent actions."""
    room = DeadbandRoom(room_id="edge-test-4")
    
    # Test 1: Silent agent (no actions)
    silent_id = room.register_agent("silent-agent")
    # Agent registers but does nothing - room should not crash
    self.assertIsNotNone(room.get_agent_state(silent_id))
    
    # Test 2: Rapid-fire actions
    rapid_id = room.register_agent("rapid-agent")
    for _ in range(100):  # Rapid protocol execution
        room.execute_protocol_step(rapid_id, "P0")
    
    # Room should throttle or handle gracefully
    state = room.get_agent_state(rapid_id)
    self.assertLessEqual(len(state.get("action_log", [])), 100)  # May be throttled
    
    # Test 3: Invalid protocol step order
    disorder_id = room.register_agent("disorder-agent")
    # Skip P0 (violates protocol)
    with self.assertRaises(ValueError):
        room.execute_protocol_step(disorder_id, "P1")
```

def test_resource_management(self):
    """P0: Room should handle resource constraints gracefully."""
    room = DeadbandRoom(room_id="edge-test-5")
    
    # Test many agent registrations
    agent_ids = []
    for i in range(1000):  # Large number of agents
        try:
            agent_id = room.register_agent(f"mass-agent-{i}")
            agent_ids.append(agent_id)
        except Exception as e:
            # Room may hit limits - should fail gracefully
            self.assertIn("limit", str(e).lower())
            break
    
    # If registration succeeded, test tile submission volume
    if len(agent_ids) > 0:
        successful_submissions = 0
        for i, agent_id in enumerate(agent_ids[:10]):  # Test subset
            try:
                tile = {
                    "cycle": i,
                    "content": "x" * 10000,  # Large content
                    "priority": 0,
                    "metadata": {}
                }
                room.submit_tile(agent_id, tile)
                successful_submissions += 1
            except Exception:
                # Room may reject due to size limits
                pass
        
        # Room should handle at least some submissions
        self.assertGreater(successful_submissions, 0)
    
    # Verify room can still function after stress test
    final_agent = room.register_agent("final-test-agent")
    self.assertIsNotNone(final_agent)
```
