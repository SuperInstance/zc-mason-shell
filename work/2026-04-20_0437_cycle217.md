```python
def test_concurrent_phase_transitions(self):
    """Test multiple agents attempting phase transitions simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # All agents start in P0
    for agent_id in agent_ids:
        room.register_agent(agent_id)
    
    # Simulate concurrent P0→P1 transitions
    results = []
    for agent_id in agent_ids:
        try:
            # Simulate concurrent access (thread-safe check)
            result = room.submit_tile(agent_id, {
                "phase": "P1",
                "content": f"Channel map from {agent_id}",
                "timestamp": time.time()
            })
            results.append((agent_id, result))
        except Exception as e:
            results.append((agent_id, str(e)))
    
    # Verify: All transitions should succeed or fail gracefully
    # No deadlocks, no corrupted state
    success_count = sum(1 for _, r in results if r.get("accepted") == True)
    assert success_count > 0, "At least one agent should succeed"
    assert room.state_integrity_check(), "Room state should remain consistent"
```

### Test 2: Malformed Tile Submission Resilience
```python
def test_malformed_tile_handling(self):
    """Test room's resilience against various malformed tile submissions."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    room.register_agent(agent_id)
    
    malformed_tiles = [
        None,  # Null tile
        {},  # Empty dict
        {"phase": "P5"},  # Invalid phase
        {"phase": "P1"},  # Missing content
        {"content": "test", "phase": 123},  # Wrong phase type
        {"phase": "P1", "content": "", "timestamp": "not-a-number"},  # Bad timestamp
        {"phase": "P1", "content": "x" * 1000000},  # Excessively large content
        {"phase": "P1", "content": "<script>alert()</script>"},  # Injection attempt
    ]
    
    rejection_count = 0
    for i, tile in enumerate(malformed_tiles):
        try:
            result = room.submit_tile(agent_id, tile)
            if not result.get("accepted", False):
                rejection_count += 1
        except (ValueError, TypeError, KeyError):
            rejection_count += 1
    
    # All malformed tiles should be rejected without crashing
    assert rejection_count == len(malformed_tiles), \
        f"Expected all {len(malformed_tiles)} malformed tiles rejected, got {rejection_count}"
    assert room.get_agent_phase(agent_id) == "P0", \
        "Agent should remain in P0 after malformed submissions"
```

```python
def test_memory_boundary_conditions(self):
    """Test room behavior near memory/resource limits."""
    room = DeadbandRoom()
    
    # Test 1: Maximum concurrent agents
    max_agents = 1000
    for i in range(max_agents):
        agent_id = f"agent_{i}"

            # Should fail gracefully, not crash
            assert "resource" in str(e).lower() or "limit" in str(e).lower()

    # Test 2: Tile storage limits

    assert room.get_agent_count() > 0, "Room should still have agents"
    assert room.state_integrity_check(), "State integrity maintained"
```

### Test 4: Network Partition Simulation
```python
def test_partial_network_failure(self):
    """Test room behavior during simulated network partitions."""
    room = DeadbandRoom()
    
    # Create agent groups in different "network segments"
    segment_a = [f"agent_a_{i}" for i in range(3)]
    segment_b = [f"agent_b_{i}" for i in range(3)]
    
    for agent_id in segment_a + segment_b:
        room.register_agent(agent_id)
    
    # Simulate partition: Segment A can submit, Segment B cannot
    partition_active = True
    
    # Mock network failure for segment B
    original_submit = room.submit_tile
    def partitioned_submit(agent_id, tile):
        if partition_active and agent_id in segment_b:
            raise ConnectionError(f"Network partition for {agent_id}")
        return original_submit(agent_id, tile)
    
    room.submit_tile = partitioned_submit
    
    # Segment A should work normally
    for agent_id in segment_a:
        result = room.submit_tile(agent_id, {
            "phase": "P1",
            "content": f"From {agent_id} during partition",
            "timestamp": time.time()
        })
        assert result.get("accepted") == True, f"Segment A agent {agent_id} should succeed"
    
    # Segment B should fail gracefully
    for agent_id in segment_b:
        try:
            room.submit_tile(agent_id, {
                "phase": "P1",
                "content": f"From {agent_id} during partition",
                "timestamp": time.time()
            })
            assert False, f"Segment B agent {agent_id} should raise ConnectionError"
        except ConnectionError:
            pass  # Expected
    
    # Restore network
    partition_active = False
    room.submit_tile = original_submit
    
    # Verify: Segment B agents should still be in original state
    for agent_id in segment_b:
        assert room.get_agent_phase(agent_id) == "P0", \
            f"Agent {agent_id} should remain in P0 after partition"
    
    # Room should recover fully
    for agent_id in segment_b:
        result = room.submit_tile(agent_id, {
            "phase": "P1",
            "content": f"Recovery from {agent_id}",
            "timestamp": time.time()
        })
        assert result.get("accepted") == True, \
            f"Agent {agent_id} should recover after partition"
```
