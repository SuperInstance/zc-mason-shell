### Test 1: Concurrent Tile Submission Race Condition
**Scenario:** Multiple agents attempt to submit tiles simultaneously while the room is in P1 state.
**Edge:** Tests thread safety and state consistency during concurrent operations.
```python
def test_concurrent_tile_submission_p1_state():
    """Test that concurrent tile submissions in P1 state maintain protocol integrity."""
    room = DeadbandRoom(room_id="race_test", preset="deadband")
    room.current_state = "P1"  # Force P1 state
    
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "content": f"Test from {agent_id}",
            "protocol_phase": "P1",
            "timestamp": time.time()
        }
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = [executor.submit(submit_tile, f"agent_{i}") for i in range(5)]
        results = [f.result() for f in futures]
    
    # Verify all submissions were processed
    assert all(r["status"] == "accepted" for r in results)
    # Verify state didn't regress to P0
    assert room.current_state == "P1"
    # Verify tile count increased correctly
    assert len(room.tiles) == 5
```

def test_state_oscillation_protection():
    """Test that rapid phase changes are prevented or throttled."""
    room = DeadbandRoom(room_id="oscillation_test", preset="deadband")
    
    # Simulate rapid submissions trying to force state changes
    states = ["P0", "P1", "P2", "P1", "P0"]
    results = []
    
    for i, target_state in enumerate(states):
        tile = {
            "agent_id": "oscillating_agent",
            "content": f"Attempt {i} to reach {target_state}",
            "protocol_phase": target_state,
            "timestamp": time.time() + i * 0.001  # Very close timestamps
        }
        results.append(room.submit_tile(tile))
    
    # Check that not all submissions succeeded (some should be throttled)
    accepted = [r for r in results if r["status"] == "accepted"]
    rejected = [r for r in results if r["status"] == "rejected"]
    
    assert len(accepted) > 0  # Some should work
    assert len(rejected) > 0  # Some should be rejected
    assert len(accepted) < len(states)  # Not all should work
    
    # Verify final state is valid (not oscillating)
    assert room.current_state in ["P0", "P1", "P2"]
```

### Test 5: Recovery from Corrupted Tile Storage

def test_corrupted_tile_recovery():
    """Test room behavior when encountering corrupted tile data."""
    room = DeadbandRoom(room_id="corruption_test", preset="deadband")
    
    # First, add some valid tiles
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"Valid tile {i}",
            "protocol_phase": "P0",
            "timestamp": time.time() + i
        }
        room.submit_tile(tile)
    
    # Simulate corruption by directly manipulating internal storage
    # (This tests the room's resilience to data corruption)
    room.tiles.append({"corrupted": True, "garbage": b"\xff\xfe\x00"})
    
    # Try to perform normal operations
    new_tile = {
        "agent_id": "recovery_agent",
        "content": "Test after corruption",
        "protocol_phase": "P0",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(new_tile)
    
    # Room should handle corruption gracefully
    assert result["status"] == "accepted" or "error" in result
    
    # Attempt to serialize/deserialize (common operation)
    try:
        serialized = room.serialize_state()
        # If serialization works, deserialize should too
        room.load_state(serialized)
        recovery_successful = True
    except Exception:
        recovery_successful = False
    
    # Room should either recover or fail cleanly
    assert recovery_successful or room.current_state == "P0"  # Fallback to P0
```
