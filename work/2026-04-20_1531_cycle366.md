### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents should be able to submit tiles concurrently without corruption."""
    room = DeadbandRoom(capacity=1000)
    tiles = [
        {"agent": "agent1", "phase": "P0", "content": "Avoid rock A"},
        {"agent": "agent2", "phase": "P0", "content": "Avoid rock B"},
        {"agent": "agent3", "phase": "P1", "content": "Channel X safe"}
    ]
    
    # Simulate concurrent submission
    import threading
    results = []
    
    def submit_tile(tile):
        results.append(room.add_tile(tile))
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # All tiles should be accepted
    assert all(results), "Concurrent tile submission failed"
    assert len(room.tiles) == 3, f"Expected 3 tiles, got {len(room.tiles)}"
    
    # Verify tile integrity
    tile_contents = {t["content"] for t in room.tiles}
    expected = {"Avoid rock A", "Avoid rock B", "Channel X safe"}
    assert tile_contents == expected, "Tile content corrupted during concurrent write"
```

### Test 2: Memory Capacity Enforcement
```python
def test_memory_capacity_enforcement(self):
    """Room should reject tiles when at capacity, preserving existing tiles."""
    room = DeadbandRoom(capacity=2)  # Small capacity for testing
    
    # Fill room to capacity
    tile1 = {"agent": "agent1", "phase": "P0", "content": "Tile 1"}
    tile2 = {"agent": "agent2", "phase": "P0", "content": "Tile 2"}
    
    assert room.add_tile(tile1) == True
    assert room.add_tile(tile2) == True
    assert len(room.tiles) == 2
    
    # Attempt to exceed capacity
    tile3 = {"agent": "agent3", "phase": "P1", "content": "Tile 3"}
    assert room.add_tile(tile3) == False, "Should reject tile when at capacity"
    assert len(room.tiles) == 2, "Tile count should remain at capacity"
    
    # Verify existing tiles unchanged
    contents = {t["content"] for t in room.tiles}
    assert contents == {"Tile 1", "Tile 2"}, "Existing tiles corrupted"
```

### Test 3: Invalid Phase Transition Enforcement
```python
def test_invalid_phase_transition(self):
    """Room should reject attempts to skip protocol phases."""
    room = DeadbandRoom()
    
    # Valid P0 tile
    p0_tile = {"agent": "agent1", "phase": "P0", "content": "Map negative space"}
    assert room.add_tile(p0_tile) == True
    
    # Attempt P2 before P1 (should fail)
    p2_tile = {"agent": "agent1", "phase": "P2", "content": "Optimize path"}
    assert room.add_tile(p2_tile) == False, "Should reject P2 before P1 completion"
    
    # Add P1 tile
    p1_tile = {"agent": "agent1", "phase": "P1", "content": "Find safe channel"}
    assert room.add_tile(p1_tile) == True
    
    # Now P2 should be accepted
    assert room.add_tile(p2_tile) == True, "Should accept P2 after P1"
    
    # Verify phase sequence in tiles
    phases = [t["phase"] for t in room.tiles]
    assert phases == ["P0", "P1", "P2"], f"Invalid phase sequence: {phases}"
```

### Test 4: Ghost Tile Handling
```python
def test_ghost_tile_handling(self):
    """Tiles from terminated agents should be marked but preserved."""
    room = DeadbandRoom()
    
    # Normal tile from active agent
    normal_tile = {
        "agent": "active_agent",
        "phase": "P0", 
        "content": "Normal operation",
        "timestamp": "2026-04-20T15:30:00Z"
    }
    
    # Ghost tile from terminated agent
    ghost_tile = {
        "agent": "terminated_agent",
        "phase": "P0",
        "content": "Lesson from dead agent",
        "timestamp": "2026-04-20T14:30:00Z",
        "status": "terminated"
    }
    
    assert room.add_tile(normal_tile) == True
    assert room.add_tile(ghost_tile) == True
    
    # Ghost tiles should be included in count
    assert len(room.tiles) == 2
    
    # Verify ghost tile metadata preserved
    ghost = next(t for t in room.tiles if t["agent"] == "terminated_agent")
    assert ghost["status"] == "terminated"
    assert "Lesson from dead agent" in ghost["content"]
```
