### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(max_agents=5)
    agent_ids = [f"agent_{i}" for i in range(3)]
    
    # Simulate concurrent submissions using threads
    import threading
    results = []
    
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        try:
            room.submit_tile(tile)
            results.append((agent_id, "success"))
        except Exception as e:
            results.append((agent_id, str(e)))
    
    threads = [threading.Thread(target=submit_tile, args=(aid,)) for aid in agent_ids]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    self.assertEqual(len(results), 3)
    self.assertEqual(len([r for r in results if r[1] == "success"]), 3)
    
    # Verify room state consistency
    self.assertEqual(room.get_tile_count(), 3)
    self.assertEqual(len(set(room.get_agent_ids())), 3)
```

### Test 2: Malformed Tile Payloads
```python
def test_malformed_tile_payloads(self):
    """Room should reject tiles with invalid structure or missing required fields."""
    room = DeadbandRoom()
    
    test_cases = [
        ("empty_dict", {}, "Missing required field: agent_id"),
        ("missing_phase", {"agent_id": "test_agent"}, "Missing required field: phase"),
        ("invalid_phase", {"agent_id": "test_agent", "phase": "P3"}, "Invalid phase: P3"),
        ("non_string_content", {"agent_id": "test_agent", "phase": "P1", "content": 123}, "Content must be string"),
        ("malformed_json", '{"incomplete": true', "Invalid JSON structure"),
    ]
    
    for case_name, tile, expected_error in test_cases:
        with self.subTest(case=case_name):
            with self.assertRaises(DeadbandViolation) as cm:
                if isinstance(tile, dict):
                    room.submit_tile(tile)
                else:
                    room._validate_tile_structure(tile)  # Direct validation test
            self.assertIn(expected_error, str(cm.exception))
```

### Test 3: Room Capacity Limits
```python
def test_room_capacity_limits(self):
    """Room should enforce max_agents limit and handle capacity gracefully."""
    room = DeadbandRoom(max_agents=2)
    
    # Fill room to capacity
    agent1_tile = {"agent_id": "agent1", "phase": "P0", "content": "Test"}
    agent2_tile = {"agent_id": "agent2", "phase": "P1", "content": "Test"}
    
    room.submit_tile(agent1_tile)
    room.submit_tile(agent2_tile)
    
    # Third agent should be rejected
    agent3_tile = {"agent_id": "agent3", "phase": "P2", "content": "Test"}
    with self.assertRaises(DeadbandViolation) as cm:
        room.submit_tile(agent3_tile)
    self.assertIn("Room at capacity", str(cm.exception))
    
    # After reset, new agents should be accepted
    room.reset()
    room.submit_tile(agent3_tile)
    self.assertEqual(room.get_agent_ids(), ["agent3"])
```

### Test 4: Network Partition Simulation
```python
def test_network_partition_recovery(self):
    """Room should handle agent disconnection and maintain protocol state."""
    room = DeadbandRoom()
    
    # Agent starts P0 phase
    tile1 = {"agent_id": "agent_a", "phase": "P0", "content": "Mapping negative space"}
    room.submit_tile(tile1)
    
    # Simulate network partition - agent disappears
    room.mark_agent_inactive("agent_a")
    
    # Another agent should be able to continue
    tile2 = {"agent_id": "agent_b", "phase": "P0", "content": "Alternative mapping"}
    room.submit_tile(tile2)
    
    # Original agent reconnects - should be treated as new session
    tile3 = {"agent_id": "agent_a", "phase": "P0", "content": "Reconnected mapping"}
    room.submit_tile(tile3)
    
    # Verify both agents in room state
    self.assertEqual(sorted(room.get_agent_ids()), ["agent_a", "agent_b"])
    self.assertEqual(room.get_agent_state("agent_a")["phase"], "P0")
```

### Test 5: Resource Exhaustion Protection
```python
def test_resource_exhaustion_protection(self):
    """Room should handle extremely large tiles or high frequency submissions."""
    room = DeadbandRoom()
    
    # Test 1: Very large content (10MB string)
    large_content = "x" * (10 * 1024 * 1024)  # 10MB
    large_tile = {
        "agent_id": "memory_test",
        "phase": "P1",
        "content": large_content
    }
    
    with self.assertRaises(DeadbandViolation) as cm:
        room.submit_tile(large_tile)
    self.assertIn("Tile size limit exceeded", str(cm.exception))
    
    # Test 2: High frequency submissions (rate limiting)
    import time
    rapid_agent = "rapid_fire"
    submissions = []
    
    for i in range(10):
        tile = {
            "agent_id": rapid_agent,
            "phase": "P1",
            "content": f"Submission {i}"
        }
        try:
            room.submit_tile(tile)
            submissions.append(True)
            time.sleep(0.01)  # Small delay
        except DeadbandViolation as e:
            if "Rate limit exceeded" in str(e):
                submissions.append(False)
                break
    
    # Should accept first few, then rate limit
    self.assertTrue(any(submissions))  # At least some succeeded
    self.assertFalse(all(submissions))  # But not all 10
```
