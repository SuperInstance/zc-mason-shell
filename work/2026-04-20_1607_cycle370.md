```python
def test_concurrent_tile_submissions(self):
    """Simulate multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(max_agents=10, tile_capacity=1000)
    
    # Create mock agents with different agent_ids
    tiles = []
    for i in range(10):
        tile = {
            'agent_id': f'agent_{i}',
            'phase': 'P1',
            'content': f'Test content from agent {i}',
            'timestamp': time.time() + i  # Staggered timestamps
        }
        tiles.append(tile)
    
    # Submit all tiles (simulating concurrent access)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    self.assertEqual(len(results), 10)
    # Check no data corruption occurred
    stored_tiles = room.get_recent_tiles(count=20)
    self.assertEqual(len(stored_tiles), 10)
```

```python
def test_partial_connectivity_scenario(self):
    """Test room behavior when agents have intermittent connectivity."""
    room = DeadbandRoom(max_agents=5, tile_capacity=50)
    
    # Simulate agent connecting, submitting, then disconnecting
    agent_id = "flaky_agent_1"
    
    # First connection successful
    tile1 = {'agent_id': agent_id, 'phase': 'P0', 'content': 'Initial map'}
    result1 = room.submit_tile(tile1)
    self.assertTrue(result1['success'])
    
    # Simulate disconnect (agent object removed from room's tracking)
    room._active_agents.pop(agent_id, None)
    
    # Reconnect with same agent_id (should be allowed)
    tile2 = {'agent_id': agent_id, 'phase': 'P1', 'content': 'Reconnected content'}
    result2 = room.submit_tile(tile2)
    self.assertTrue(result2['success'])
    
    # Verify both tiles stored under same agent_id
    agent_tiles = room.get_tiles_by_agent(agent_id)
    self.assertEqual(len(agent_tiles), 2)
```

### Test 4: Malformed Tile Data Handling
```python
def test_malformed_tile_validation(self):
    """Test room's resilience against invalid tile structures."""
    room = DeadbandRoom(max_agents=3, tile_capacity=100)
    
    test_cases = [
        # (tile_data, expected_error_type)
        (None, TypeError),  # None input
        ("not a dict", TypeError),  # String instead of dict
        ({}, ValueError),  # Empty dict
        ({'agent_id': 123}, TypeError),  # Wrong type for agent_id
        ({'agent_id': 'agent1'}, ValueError),  # Missing required fields
        ({'agent_id': 'agent1', 'phase': 'P5', 'content': 'test'}, ValueError),  # Invalid phase
        ({'agent_id': 'agent1', 'phase': 'P0', 'content': ''}, ValueError),  # Empty content
        ({'agent_id': 'agent1', 'phase': 'P0', 'content': 'x' * 10000}, ValueError),  # Too large
    ]
    
    for tile_data, expected_error in test_cases:
        with self.subTest(tile_data=tile_data):
            with self.assertRaises(expected_error):
                room.submit_tile(tile_data)
```

```python
def test_resource_limits_enforced(self):
    """Test room behavior at capacity limits."""
    # Test tile capacity limit
    room = DeadbandRoom(max_agents=2, tile_capacity=3)
    
    # Fill to capacity
    for i in range(3):
        tile = {
            'agent_id': f'agent_{i % 2}',
            'phase': 'P0',
            'content': f'Tile {i}',
            'timestamp': time.time()
        }
        room.submit_tile(tile)
    
    # Next submission should trigger cleanup or rejection
    tile_overflow = {
        'agent_id': 'agent_0',
        'phase': 'P1',
        'content': 'Should trigger capacity management',
        'timestamp': time.time()
    }
    
    result = room.submit_tile(tile_overflow)
    # Either succeeds with eviction or fails gracefully
    self.assertIn(result['status'], ['accepted', 'rejected'])
    
    if result['status'] == 'accepted':
        # Verify old tiles were evicted (FIFO or LRU)
        self.assertLessEqual(len(room.get_recent_tiles(count=10)), 3)
    
    # Test agent capacity limit
    room2 = DeadbandRoom(max_agents=1, tile_capacity=10)
    tile1 = {'agent_id': 'agent_a', 'phase': 'P0', 'content': 'First agent'}
    room2.submit_tile(tile1)
    
    # Different agent should be rejected or cause eviction
    tile2 = {'agent_id': 'agent_b', 'phase': 'P0', 'content': 'Second agent'}
    result2 = room2.submit_tile(tile2)
    self.assertFalse(result2.get('success', True))  # Should fail
```
