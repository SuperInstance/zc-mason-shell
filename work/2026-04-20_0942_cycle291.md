### Test 2: Concurrent Agent Submissions with Race Conditions
**Scenario:** Multiple agents submit tiles simultaneously; room must maintain consistency.
```python
def test_concurrent_tile_submissions(self):
    """P1: Room should handle concurrent submissions without data corruption."""
    room = DeadbandRoom()
    tiles = [
        {"tile_id": f"concurrent_{i}", "content": f"test_{i}", "phase": "P1"}
        for i in range(10)
    ]
    # Simulate concurrent submissions via threads
    import threading
    results = []
    def submit(tile):
        try:
            results.append(room.submit_tile(tile))
        except Exception as e:
            results.append(e)
    threads = [threading.Thread(target=submit, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    # Check all tiles processed without errors
    successful = [r for r in results if isinstance(r, dict) and r.get("status") == "accepted"]
    self.assertEqual(len(successful), 10)
    self.assertEqual(len(room.tiles), 10)
```

```python
def test_tile_eviction_on_capacity(self):
    """P1: Room should evict oldest tiles when capacity exceeded, preserving P0 tiles."""
    room = DeadbandRoom()
    room.max_tiles = 5  # Set low capacity for test
    # Submit 3 P0 tiles (critical) and 3 P1 tiles (less critical)
    for i in range(3):
        room.submit_tile({"tile_id": f"p0_{i}", "content": f"P0 data {i}", "phase": "P0"})
    for i in range(3):
        room.submit_tile({"tile_id": f"p1_{i}", "content": f"P1 data {i}", "phase": "P1"})
    # Room should have 5 tiles max, evict oldest P1 tile first
    self.assertEqual(len(room.tiles), 5)
    # Ensure P0 tiles are retained
    p0_tiles = [t for t in room.tiles if t.get("phase") == "P0"]
    self.assertEqual(len(p0_tiles), 3)
```
