### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race():
    """P0: Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(capacity=100)
    tiles = [
        {"agent_id": f"agent_{i}", "content": f"test_tile_{i}", "phase": "P1"}
        for i in range(10)
    ]
    
    # Simulate concurrent submission
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            room.submit_tile(tile)
            results.append(True)
        except Exception:
            results.append(False)
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all tiles accepted and room state consistent
    assert all(results), "Some concurrent submissions failed"
    assert len(room.tiles) == 10, f"Expected 10 tiles, got {len(room.tiles)}"
    assert room.training_phase == "P1", "Room should remain in P1 phase"
```

```python
def test_malformed_tile_rejection():
    """P0: Invalid tile structures should be rejected with clear error messages."""
    room = DeadbandRoom(capacity=50)
    
    malformed_tiles = [
        None,  # null tile
        {"agent_id": "test"},  # missing content and phase
        {"content": "test", "phase": "P2"},  # missing agent_id
        {"agent_id": 123, "content": "test", "phase": "P1"},  # wrong agent_id type
        {"agent_id": "test", "content": "", "phase": "P0"},  # empty content
        {"agent_id": "test", "content": "x" * 10001, "phase": "P1"},  # content too large
    ]
    
    for i, tile in enumerate(malformed_tiles):
        try:
            room.submit_tile(tile)
            assert False, f"Malformed tile {i} should have been rejected"
        except (ValueError, TypeError) as e:
            # Should provide specific error message
            assert "tile" in str(e).lower() or "invalid" in str(e).lower()
    
    assert len(room.tiles) == 0, "No malformed tiles should be accepted"
```
