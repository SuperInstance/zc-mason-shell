### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    import threading
    
    room = DeadbandRoom(room_id="concurrent_test")
    agent1 = MagicMock(agent_id="agent1")
    agent2 = MagicMock(agent_id="agent2")
    
    tile1 = {"priority": "P0", "content": "test1", "valid": True}
    tile2 = {"priority": "P0", "content": "test2", "valid": True}
    
    results = []
    errors = []
    
    def submit_tile(agent, tile, result_list):
        try:
            result = room.submit_tile(agent, tile)
            result_list.append(result)
        except Exception as e:
            errors.append(str(e))
    
    thread1 = threading.Thread(target=submit_tile, args=(agent1, tile1, results))
    thread2 = threading.Thread(target=submit_tile, args=(agent2, tile2, results))
    
    thread1.start()
    thread2.start()
    thread1.join()
    thread2.join()
    
    # Should handle concurrent submissions without crashing
    self.assertEqual(len(results), 2)
    self.assertEqual(len(errors), 0)
    # Both tiles should be accepted or properly queued
    self.assertTrue(all(r.get('accepted', False) for r in results))
```

### Test 2: Persistence Failure Recovery
```python
def test_persistence_failure_recovery(self):
    """Test room recovery when persistence layer fails."""
    room = DeadbandRoom(room_id="persistence_fail")
    
    # Mock a failing save operation
    original_save = room._save_state
    save_calls = []
    
    def failing_save():
        save_calls.append("attempted")
        raise IOError("Disk write failed")
    
    room._save_state = failing_save
    
    agent = MagicMock(agent_id="test_agent")
    tile = {"priority": "P1", "content": "test", "valid": True}
    
    # Submission should handle persistence failure gracefully
    with self.assertLogs(level='ERROR') as log:
        result = room.submit_tile(agent, tile)
        
        # Should log the error
        self.assertTrue(any("persistence" in msg.lower() for msg in log.output))
        # Should still process the tile
        self.assertIn('accepted', result)
        # Should attempt to save
        self.assertEqual(len(save_calls), 1)
    
    # Restore original method
    room._save_state = original_save
```

### Test 4: Memory Pressure Handling
```python
def test_memory_pressure_handling(self):
    """Test behavior under memory-constrained conditions."""
    room = DeadbandRoom(room_id="memory_test")
    
    # Simulate many tile submissions
    large_tiles = []
    for i in range(1000):
        large_tile = {
            "priority": "P1",
            "content": "x" * 10000,  # 10KB per tile
            "valid": True,
            "metadata": {"index": i}
        }
        large_tiles.append(large_tile)
    
    agent = MagicMock(agent_id="stress_agent")
    
    # Submit all tiles
    accepted_count = 0
    for tile in large_tiles:
        try:
            result = room.submit_tile(agent, tile)
            if result.get('accepted'):
                accepted_count += 1
        except MemoryError:
            # Room should handle memory pressure gracefully
            self.fail("Room crashed on MemoryError")
    
    # Should accept tiles within reasonable limits
    self.assertGreater(accepted_count, 0)
    
    # Check if room implemented any cleanup mechanism
    if hasattr(room, '_cleanup_old_tiles'):
        self.assertTrue(callable(room._cleanup_old_tiles))
```

### Test 5: Protocol State Race Condition
```python
def test_protocol_state_race_condition(self):
    """Test P1→P2 transition while P0 still processing."""
    room = DeadbandRoom(room_id="race_test")
    
    # Start in P0 state
    room.current_priority = "P0"
    room._processing_p0 = True  # Simulate ongoing P0 processing
    
    agent = MagicMock(agent_id="race_agent")
    
    # Attempt P1 tile while P0 processing
    tile_p1 = {"priority": "P1", "content": "P1 test", "valid": True}
    
    with self.assertRaises(ValueError) as context:
        room.submit_tile(agent, tile_p1)
    
    # Should reject with appropriate error
    self.assertIn("P0", str(context.exception))
    self.assertIn("processing", str(context.exception).lower())
    
    # Complete P0 processing
    room._processing_p0 = False
    room.current_priority = "P1"
    
    # Now P1 should be accepted
    result = room.submit_tile(agent, tile_p1)
    self.assertTrue(result.get('accepted'))
```
