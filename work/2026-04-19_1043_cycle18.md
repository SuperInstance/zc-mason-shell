### Test 1: Concurrent Tile Submission Race Condition

def test_concurrent_tile_submission_race():
    """Test that concurrent tile submissions from multiple agents don't corrupt room state."""
    room = DeadbandRoom()
    agent_ids = ["agent_a", "agent_b", "agent_c"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 18,
            "content": f"Test tile from {agent_id}",
            "protocol_phase": "P1"
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (in rapid succession)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all tiles were accepted
    assert all(r["status"] == "accepted" for r in results), "Concurrent submissions should all succeed"
    
    # Verify room state integrity
    assert len(room.tile_history) == len(agent_ids), "All tiles should be preserved"
    assert set(t["agent_id"] for t in room.tile_history) == set(agent_ids), "All agent IDs should be present"
```

def test_protocol_regression_handling():
    """Test that regression from P2 to P1 or P0 is handled gracefully (not prevented but logged)."""
    room = DeadbandRoom()
    
    # Normal progression
    room.current_phase = "P0"
    room.advance_phase()  # P0 → P1
    assert room.current_phase == "P1"

    assert room.current_phase == "P2"
    
    # Simulate regression (e.g., due to external event or error)
    room.current_phase = "P1"  # Manual regression
    
    # Submit tile during regression state
    tile = {
        "agent_id": "test_agent",
        "cycle": 18,
        "content": "Tile during regression",
        "protocol_phase": "P1"

    assert result["status"] == "accepted"
    
    # Regression should be logged

### Test 3: Resource Exhaustion (Memory)

def test_memory_exhaustion_handling():
    """Test behavior when tile history approaches memory limits."""
    room = DeadbandRoom()
    
    # Determine current limit (or set a test limit)
    test_limit = 1000
    room.max_tiles = test_limit
    
    # Fill to near capacity
    for i in range(test_limit - 10):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": i,
            "content": f"Tile {i}",
            "protocol_phase": "P1"
        }
        room.submit_tile(tile)
    
    # Verify still accepting tiles
    tile = {
        "agent_id": "final_agent",
        "cycle": test_limit,
        "content": "Near limit tile",
        "protocol_phase": "P1"
    }
    result = room.submit_tile(tile)
    assert result["status"] == "accepted"
    
    # Hit exact limit
    for i in range(9):  # Fill remaining 9 slots
        tile = {
            "agent_id": f"overflow_{i}",
            "cycle": test_limit + i + 1,
            "content": f"Overflow {i}",
            "protocol_phase": "P1"
        }
        room.submit_tile(tile)
    
    # Next submission should trigger cleanup or rejection
    tile = {
        "agent_id": "should_fail",
        "cycle": test_limit + 10,
        "content": "Beyond limit",
        "protocol_phase": "P1"
    }
    result = room.submit_tile(tile)
    
    # Either old tiles are purged (LRU) or new tile is rejected
    assert result["status"] in ["accepted", "rejected"]
    if result["status"] == "accepted":
        assert len(room.tile_history) <= test_limit
    else:
        assert "limit" in result.get("reason", "").lower()
```

### Test 4: Network Partition Simulation
```python
def test_network_partition_recovery():
    """Test that room can recover state after simulated network partition."""
    room = DeadbandRoom()
    
    # Normal operation
    tile1 = {
        "agent_id": "agent1",
        "cycle": 18,
        "content": "Pre-partition tile",
        "protocol_phase": "P1"
    }
    room.submit_tile(tile1)
    
    # Simulate partition: room loses connection to persistence layer
    original_save = room.save_state
    partition_active = True
    
    def partitioned_save():
        if partition_active:
            raise ConnectionError("Network partition: cannot save")
        return original_save()
    
    room.save_state = partitioned_save
    
    # Try to submit during partition
    tile2 = {
        "agent_id": "agent2",
        "cycle": 19,
        "content": "During partition tile",
        "protocol_phase": "P2"
    }
    
    try:
        result = room.submit_tile(tile2)
        # If submission succeeds without save, tile might be in buffer
        if result["status"] == "accepted":
            assert hasattr(room, 'unsaved_tiles_buffer')
    except ConnectionError:
        # Expected during partition
        pass
    
    # End partition
    partition_active = False
    room.save_state = original_save
    
    # Verify recovery
    tile3 = {
        "agent_id": "agent3",
        "cycle": 20,
        "content": "Post-partition tile",
        "protocol_phase": "P0"
    }
    result = room.submit_tile(tile3)
    assert result["status"] == "accepted"
    
    # State should be consistent
    assert room.tile_history[-1]["agent_id"] == "agent3"
```

def test_malformed_p0_mapping_rejection():
    """Test that invalid P0 mappings (contradictory or empty) are rejected."""
    room = DeadbandRoom()
    room.current_phase = "P0"
    
    # Test 1: Empty negative space
    tile1 = {
        "agent_id": "agent_empty",
        "cycle": 18,
        "content": {
            "p0_mapping": {},  # Empty dict
            "p1_channels": ["safe_path"]

    assert result1["status"] == "rejected"

        "agent_id": "agent_contradict",
        "cycle": 19,

            "p0_mapping": {"path_a": "unsafe", "path_b": "unsafe"},
            "p1_channels": ["path_a"]  # Contradiction: path_a in both P0 and P1
        },
        "protocol_phase": "P0"
    }
    result2 = room.submit_tile(tile2)
    assert result2["status"] == "rejected"
    assert "contradict" in result2.get("reason", "").lower() or "invalid" in result2.get("reason", "").lower()
