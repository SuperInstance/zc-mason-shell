def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents without corruption."""
    room = DeadbandRoom()
    agent_ids = ["agent_alpha", "agent_beta", "agent_gamma"]
    tiles = [
        {"phase": "P0", "content": "Avoid null pointers", "agent_id": aid}
        for aid in agent_ids
    ]
    
    # Simulate concurrent submission (threaded or async in real use)
    # For test, we submit sequentially but verify isolation
    for tile in tiles:
        room.submit_tile(tile)
    
    # Each agent's tile should be stored separately
    stored = room.get_recent_tiles(count=10)
    stored_agent_ids = {t.get("agent_id") for t in stored if t.get("agent_id")}
    assert stored_agent_ids == set(agent_ids), "Concurrent submissions corrupted tile isolation"
```

### Test 2: Extreme Numeric Metadata

def test_extreme_numeric_metadata(self):
    """Test tiles with extremely large, small, or zero numeric values in metadata."""
    room = DeadbandRoom()
    
    edge_tiles = [
        {"phase": "P1", "content": "Large cycle", "cycle": 10**9},
        {"phase": "P2", "content": "Negative priority", "priority": -100},
        {"phase": "P0", "content": "Zero confidence", "confidence": 0.0},
        {"phase": "P1", "content": "Float overflow", "score": float("inf")},
    ]
    
    for tile in edge_tiles:
        # Room should accept but may clamp or log warnings
        result = room.submit_tile(tile)
        assert result["status"] in ("accepted", "warning"), f"Tile rejected unexpectedly: {tile}"
        
    # Verify tiles stored without crashing
    assert len(room.get_recent_tiles()) >= len(edge_tiles), "Extreme metadata caused data loss"
```

### Test 3: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Test that room recovers gracefully when agent skips P0 (violates Deadband Protocol)."""
    room = DeadbandRoom()
    
    # Submit P2 tile without P0/P1 (violation)
    violation_tile = {"phase": "P2", "content": "Optimize prematurely", "agent_id": "eager_agent"}
    result = room.submit_tile(violation_tile)
    
    # Room should reject or flag violation
    assert result["status"] == "rejected", "Protocol violation not caught"
    assert "violation" in result.get("reason", "").lower(), "Missing violation warning"
    
    # Room should remain operational for next valid submission
    valid_tile = {"phase": "P0", "content": "Now doing P0 properly", "agent_id": "eager_agent"}
    valid_result = room.submit_tile(valid_tile)
    assert valid_result["status"] == "accepted", "Room failed to recover after violation"
```

### Test 4: Malformed Sentiment Arrays
```python
def test_malformed_sentiment_arrays(self):
    """Test submission with incorrectly shaped or typed sentiment/embedding arrays."""
    room = DeadbandRoom()
    
    malformed_tiles = [
        {"phase": "P1", "content": "Wrong dim sentiment", "sentiment": [0.5]},  # 1D vs expected 2D
        {"phase": "P0", "content": "String sentiment", "sentiment": "positive"},
        {"phase": "P2", "content": "Oversized embedding", "embedding": [0.1] * 10000},
        {"phase": "P1", "content": "None embedding", "embedding": None},
    ]
    
    for tile in malformed_tiles:
        result = room.submit_tile(tile)
        # Room should either reject or normalize/ignore malformed fields
        assert result["status"] in ("accepted", "rejected", "warning"), f"Unexpected status: {result}"
        if result["status"] == "accepted":
            # If accepted, ensure room handled malformation gracefully
            stored = room.get_recent_tiles(count=1)[0]
            assert "sentiment" not in stored or isinstance(stored.get("sentiment"), (list, type(None)))
```

### Test 5: State Serialization Edge Cases
```python
def test_state_serialization_edge_cases(self):
    """Test room state persistence with empty, large, and corrupted states."""
    room = DeadbandRoom()
    
    # 1. Serialize empty room
    empty_state = room.serialize_state()
    assert isinstance(empty_state, (str, bytes, dict)), "Empty state serialization failed"
    
    # 2. Add many tiles then serialize
    for i in range(100):
        room.submit_tile({"phase": "P0", "content": f"Tile {i}", "cycle": i})
    full_state = room.serialize_state()
    assert len(full_state) > len(empty_state), "Full state not larger than empty"
    
    # 3. Deserialize corrupted state (should handle gracefully)
    corrupted_state = "{invalid_json: true}"
    try:
        room.load_state(corrupted_state)
        # If doesn't raise, ensure room is in valid fallback state
        assert room.get_recent_tiles() == [], "Corrupted state should reset to empty"
    except Exception as e:
        # Expected to raise specific error type
        assert "json" in str(e).lower() or "parse" in str(e).lower()
    
    # 4. Round-trip serialization
    original_tiles = room.get_recent_tiles()
    saved_state = room.serialize_state()
    new_room = DeadbandRoom()
    new_room.load_state(saved_state)
    assert new_room.get_recent_tiles() == original_tiles, "Round-trip serialization failed"
```
