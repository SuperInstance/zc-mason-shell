```python
def test_deadband_width_boundaries(self):
    """Test DeadbandRoom initialization with extreme deadband_width values."""
    # Minimum valid width (should succeed)
    room = DeadbandRoom(name="test_min", deadband_width=0.001, max_tiles=100)
    self.assertEqual(room.deadband_width, 0.001)
    
    # Maximum reasonable width (should succeed)
    room = DeadbandRoom(name="test_max", deadband_width=1000.0, max_tiles=100)
    self.assertEqual(room.deadband_width, 1000.0)
    
    # Zero width (should fail - P0 violation)
    with self.assertRaises(ValueError):
        DeadbandRoom(name="test_zero", deadband_width=0.0, max_tiles=100)
    
    # Negative width (should fail - P0 violation)
    with self.assertRaises(ValueError):
        DeadbandRoom(name="test_neg", deadband_width=-1.0, max_tiles=100)
```

### Test 2: Concurrent Tile Addition During Training
```python
def test_concurrent_tile_addition_during_training(self):
    """Test adding tiles while training cycle is active."""
    room = DeadbandRoom(name="concurrent_test", deadband_width=0.1, max_tiles=50)
    
    # Add initial tiles
    for i in range(10):
        room.add_tile({"cycle": i, "data": f"tile_{i}"})
    
    # Start training in background thread
    import threading
    training_complete = threading.Event()
    
    def train_in_background():
        room.train_cycle()
        training_complete.set()
    
    trainer = threading.Thread(target=train_in_background)
    trainer.start()
    
    # Attempt to add tiles during training
    tiles_added_during_training = 0
    for i in range(10, 20):
        try:
            room.add_tile({"cycle": i, "data": f"concurrent_tile_{i}"})
            tiles_added_during_training += 1
        except RuntimeError:
            # Expected if room locks during training
            pass
    
    trainer.join()
    training_complete.wait(timeout=5.0)
    
    # Verify training completed and tiles were either added or properly rejected
    self.assertTrue(training_complete.is_set())
    self.assertGreaterEqual(tiles_added_during_training, 0)
```

### Test 3: Max Tiles Exhaustion and Cleanup

```python
def test_max_tiles_exhaustion_and_cleanup(self):
    """Test behavior when exceeding max_tiles limit."""
    room = DeadbandRoom(name="exhaustion_test", deadband_width=0.1, max_tiles=5)
    
    # Fill to capacity
    for i in range(5):
        room.add_tile({"cycle": i, "data": f"tile_{i}"})
    self.assertEqual(len(room.tiles), 5)
    
    # Add one more - should trigger cleanup of oldest tile
    room.add_tile({"cycle": 5, "data": "overflow_tile"})
    
    # Should still have max_tiles, not max_tiles + 1
    self.assertEqual(len(room.tiles), 5)
    
    # Oldest tile (cycle 0) should be removed
    tile_cycles = [t["cycle"] for t in room.tiles]
    self.assertNotIn(0, tile_cycles)
    self.assertIn(5, tile_cycles)
```

### Test 4: Malformed Tile Content Resilience
```python
def test_malformed_tile_content_resilience(self):
    """Test handling of various malformed tile structures."""
    room = DeadbandRoom(name="malformed_test", deadband_width=0.1, max_tiles=100)
    
    test_cases = [
        # (tile_data, should_succeed, description)
        ({"valid": "data", "cycle": 1}, True, "Valid tile"),
        ({"cycle": "not_a_number"}, False, "Non-numeric cycle"),
        ({"cycle": 1, "data": None}, True, "None data field"),
        ({"cycle": 1, "data": {"nested": {"deep": []}}}, True, "Deeply nested data"),
        ({"cycle": 1, "data": "x" * 1000000}, False, "Oversized data payload"),
        ({"cycle": -1}, False, "Negative cycle number"),
        ({}, False, "Empty tile"),
        ({"cycle": 1, "data": b"binary_data"}, False, "Binary data in string field"),
    ]
    
    for tile_data, should_succeed, description in test_cases:
        with self.subTest(description):
            if should_succeed:
                room.add_tile(tile_data)
                # Verify tile was added
                self.assertIn(tile_data, room.tiles)
            else:
                with self.assertRaises((ValueError, TypeError)):
                    room.add_tile(tile_data)
```

```python
def test_export_ensign_invalid_adapter_config(self):
    """Test export_ensign with various invalid adapter configurations."""
    room = DeadbandRoom(name="export_test", deadband_width=0.1, max_tiles=10)
    
    # Add some training data
    for i in range(3):
        room.add_tile({"cycle": i, "data": f"training_{i}"})
    
    room.train_cycle()  # Ensure model is trained
    
    invalid_configs = [
        {"adapter_type": "invalid_type"},  # Unknown adapter type
        {"rank": 0},  # Invalid rank (must be positive)
        {"rank": -1},  # Negative rank
        {"alpha": "not_a_number"},  # Non-numeric alpha
        {"target_modules": []},  # Empty target modules
        {"dropout": 2.0},  # Dropout > 1.0
        None,  # None config
        {},  # Empty config dict
    ]
    
    for config in invalid_configs:
        with self.subTest(config=str(config)):
            with self.assertRaises((ValueError, TypeError)):
                room.export_ensign("test_ensign", config)
    
    # Verify valid config still works
    valid_config = {"rank": 8, "alpha": 16, "dropout": 0.1}
    result = room.export_ensign("valid_ensign", valid_config)
    self.assertIsNotNone(result)
    self.assertIn("adapter_path", result)
```
