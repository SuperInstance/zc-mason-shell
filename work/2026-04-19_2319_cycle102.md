### Test 3: Concurrent Tile Submission Race
```python
def test_concurrent_tile_submission():
    """P1: Multiple agents submitting simultaneously should not corrupt state."""
    room = DeadbandRoom(max_agents=5)
    agents = [MockAgent(f"agent_{i}") for i in range(3)]
    
    # Register all agents
    for agent in agents:
        room.register_agent(agent)
    
    # Simulate concurrent submissions
    tiles = []
    for i, agent in enumerate(agents):
        tile = {
            "agent_id": agent.agent_id,
            "cycle": 1,
            "protocol_step": "P0",
            "content": {"negative_space": f"area_{i}"}
        }
        tiles.append(tile)
    
    # Submit concurrently (simulated with threads)
    import threading
    results = []
    
    def submit_tile(tile):
        results.append(room.submit_tile(tile))
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all tiles processed and state consistent
    assert len(results) == 3
    assert all(r["status"] == "accepted" for r in results)
    assert room.tile_count == 3
    assert len(room.get_agent_ids()) == 3
```

```python
def test_network_partition_recovery():
    """P1: Room should handle agent disconnection and reconnection."""
    room = DeadbandRoom(max_agents=3, heartbeat_timeout_sec=2)
    agent = MockAgent("flaky_agent", connection_stable=False)
    
    # Register and submit initial tile
    room.register_agent(agent)
    tile1 = create_valid_tile(1, agent_id="flaky_agent")
    result1 = room.submit_tile(tile1)
    assert result1["status"] == "accepted"
    
    # Simulate network partition (agent disappears)
    agent.disconnect()
    
    # Room should detect missing heartbeat
    import time
    time.sleep(3)  # Exceed heartbeat timeout
    
    # Attempt tile submission from disconnected agent
    tile2 = create_valid_tile(2, agent_id="flaky_agent")
    result2 = room.submit_tile(tile2)
    assert result2["status"] == "rejected"
    assert "disconnected" in result2["reason"].lower()
    
    # Agent reconnects
    agent.reconnect()
    room.register_agent(agent)  # Re-registration required
    
    # Should accept tiles again
    tile3 = create_valid_tile(3, agent_id="flaky_agent")
    result3 = room.submit_tile(tile3)
    assert result3["status"] == "accepted"
```
