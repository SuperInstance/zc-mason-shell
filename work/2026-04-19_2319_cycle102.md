def test_concurrent_tile_submission_race():
    """Test DeadbandRoom handling of simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(room_id="race_test")
    
    # Simulate 10 agents submitting tiles at the same cycle
    tiles = []
    for i in range(10):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": 100,
            "content": f"Test content from agent {i}",
            "protocol_phase": "P1",
            "validation_hash": f"hash_{i}"
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (in rapid succession)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all tiles were processed without corruption
    assert len(results) == 10
    assert all(r.get("status") == "accepted" for r in results)
    
    # Verify room state reflects all submissions
    assert len(room.current_tiles) == 10
    assert room.cycle_count == 100
```

        "protocol_phase": "P1",
        "validation_hash": "hash_p1"
    }
    
    result = room.submit_tile(p1_tile)

def test_malformed_tile_structure_recovery():
    """Test DeadbandRoom handling of structurally invalid tiles."""
    room = DeadbandRoom(room_id="malformed_test")
    
    test_cases = [
        # Missing required fields
        {"agent_id": "agent1", "cycle": 1},  # Missing content, protocol_phase
        {"content": "test", "cycle": 1},  # Missing agent_id
        {"agent_id": "agent2", "content": "test"},  # Missing cycle
        
        # Invalid data types
        {"agent_id": 123, "cycle": "one", "content": "test", "protocol_phase": "P0"},
        {"agent_id": "agent3", "cycle": 1, "content": 456, "protocol_phase": "P0"},
        
        # Invalid protocol phase
        {"agent_id": "agent4", "cycle": 1, "content": "test", "protocol_phase": "P5"},
        {"agent_id": "agent5", "cycle": 1, "content": "test", "protocol_phase": ""},
        
        # Empty content
        {"agent_id": "agent6", "cycle": 1, "content": "", "protocol_phase": "P0"},
        
        # Excessively large content
        {"agent_id": "agent7", "cycle": 1, "content": "x" * 10001, "protocol_phase": "P0"},
    ]
    
    rejection_count = 0
    for tile in test_cases:
        result = room.submit_tile(tile)
        if result.get("status") == "rejected":
            rejection_count += 1
    
    # All malformed tiles should be rejected
    assert rejection_count == len(test_cases)
    
    # Room state should remain clean
    assert len(room.current_tiles) == 0
    assert room.cycle_count == 0
```
