def test_deadband_room_boundary_action_list():
    """Test minimum (1) and maximum (protocol limit) action list sizes."""
    room = DeadbandRoom()
    
    # Minimum: single action
    result = room.process_cycle(["action1"], ["constraint1"])
    assert "P0" in result
    
    # Maximum: at protocol limit (1000 actions)
    many_actions = [f"action_{i}" for i in range(1000)]
    many_constraints = [f"constraint_{i}" for i in range(1000)]

    assert "P0" in result or "P1" in result  # Depends on constraint matching

### Test 2: P1 Returns Empty Safe Channels
```python
def test_deadband_room_empty_safe_channels():
    """Test when P1 identifies NO safe channels (all actions violate constraints)."""
    room = DeadbandRoom()
    
    # All actions violate all constraints
    actions = ["hit_rock", "crash_into_wall", "violate_p0"]
    constraints = ["avoid_rocks", "stay_in_lane", "follow_p0"]
    
    result = room.process_cycle(actions, constraints)
    # Should not proceed to P2, should report no safe channels
    assert "No safe channels identified" in result
    assert "P2" not in result
```

### Test 4: Multiple Protocol Cycles with State
```python
def test_deadband_room_multiple_cycles_state():
    """Test that room maintains no hidden state between cycles (idempotent)."""
    room = DeadbandRoom()
    
    # First cycle
    result1 = room.process_cycle(["action1", "action2"], ["constraint1"])
    
    # Second cycle with same inputs
    result2 = room.process_cycle(["action1", "action2"], ["constraint1"])
    
    # Results should be identical (room is stateless by design)
    assert result1 == result2
    
    # Third cycle with different inputs
    result3 = room.process_cycle(["action3"], ["constraint2"])
    assert result3 != result1  # Should differ
```

### Test 5: Mixed Valid/Invalid Input Recovery
```python
def test_deadband_room_error_recovery():
    """Test that room recovers gracefully after invalid input."""
    room = DeadbandRoom()
    
    # First: invalid input (wrong type)
    with pytest.raises(TypeError):
        room.process_cycle("not_a_list", ["constraint1"])
    
    # Second: valid input immediately after error
    result = room.process_cycle(["valid_action"], ["valid_constraint"])
    assert "P0" in result  # Should process normally
    
    # Third: another invalid input (None)
    with pytest.raises(TypeError):
        room.process_cycle(None, None)
    
    # Fourth: back to valid
    result = room.process_cycle(["another_action"], ["another_constraint"])
    assert "P1" in result or "P2" in result  # Should still work
```
