### Test 1: Concurrent Tile Submission
```python
import asyncio
from unittest.mock import AsyncMock, patch
import pytest

@pytest.mark.asyncio
async def test_concurrent_tile_submission_no_corruption():
    """P0: Concurrent submissions should not corrupt room state."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    config = {"room_id": "test_concurrent", "max_tiles": 100}
    room = DeadbandRoom(config)
    
    # Simulate 10 agents submitting tiles simultaneously
    tiles = [
        {"agent_id": f"agent_{i}", "cycle": i, "content": f"test_{i}", "timestamp": i}
        for i in range(10)
    ]
    
    async def submit_tile(tile):
        return await room.submit_tile(tile)
    
    # Submit all tiles concurrently
    tasks = [submit_tile(tile) for tile in tiles]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # All submissions should succeed
    assert all(isinstance(r, dict) and r.get("status") == "accepted" for r in results)
    
    # Room should have all 10 tiles
    state = room.get_state()
    assert len(state["tiles"]) == 10
    
    # Verify no duplicate or missing tiles
    tile_ids = [t["agent_id"] for t in state["tiles"]]
    assert set(tile_ids) == {f"agent_{i}" for i in range(10)}
```

```python
def test_malicious_protocol_bypass():
    """P0: Room should detect and reject tiles that bypass P0/P1/P2 ordering."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    config = {"room_id": "test_protocol", "max_tiles": 100}
    room = DeadbandRoom(config)
    
    # Tile that claims to be P2 optimization without P0/P1
    malicious_tile = {
        "agent_id": "malicious_agent",
        "cycle": 1,
        "content": {
            "protocol_step": "P2",
            "optimization": "skip_to_p2",  # Violates "Never skip to P2"
            "data": {"score": 100}
        },
        "timestamp": 1234567890
    }
    
    result = room.submit_tile(malicious_tile)
    
    # Should be rejected for protocol violation
    assert result["status"] == "rejected"
    assert "protocol" in result.get("reason", "").lower()
    assert "p0" in result.get("reason", "").lower() or "p1" in result.get("reason", "").lower()
```

### Test 4: Memory Limit Enforcement
```python
def test_memory_limit_enforcement():
    """P0: Room should enforce max_tiles limit and evict oldest tiles."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    config = {"room_id": "test_memory", "max_tiles": 3}
    room = DeadbandRoom(config)
    
    # Submit 5 tiles (exceeds limit of 3)
    for i in range(5):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": i,
            "content": f"content_{i}",
            "timestamp": i  # Earlier timestamp = older
        }
        room.submit_tile(tile)
    
    state = room.get_state()
    
    # Should only have 3 tiles (max_tiles)
    assert len(state["tiles"]) == 3
    
    # Should have evicted the 2 oldest tiles (timestamp 0 and 1)
    remaining_agents = {t["agent_id"] for t in state["tiles"]}
    assert "agent_0" not in remaining_agents
    assert "agent_1" not in remaining_agents
    assert {"agent_2", "agent_3", "agent_4"} == remaining_agents
```
