### Test 3: Concurrent Tile Submission Race Condition
```rust
#[test]
fn test_deadband_room_concurrent_submissions() {
    let room = Arc::new(Mutex::new(DeadbandRoom::new("test-room".to_string(), 100)));
    let mut handles = vec![];
    
    for i in 0..10 {
        let room_clone = Arc::clone(&room);
        handles.push(thread::spawn(move || {
            let tile = Tile {
                agent_id: format!("agent-{}", i),
                content: format!("Content from agent {}", i),
                timestamp: Utc::now(),
                priority: i % 3,
            };
            
            let mut room = room_clone.lock().unwrap();
            room.submit_tile(tile)
        }));
    }
    
    let mut successes = 0;
    let mut failures = 0;
    
    for handle in handles {
        match handle.join().unwrap() {
            Ok(_) => successes += 1,
            Err(_) => failures += 1,
        }
    }
    
    // Should handle concurrent access without panicking
    assert!(successes + failures == 10);
}
```

### Test 5: Protocol Violation Detection
```rust
#[test]
fn test_deadband_room_p2_before_p0_detection() {
    let mut room = DeadbandRoom::new("test-room".to_string(), 100);
    
    // Simulate a tile that tries to optimize (P2) without establishing safe channel (P1)
    let violating_tile = Tile {
        agent_id: "test-agent".to_string(),
        content: "P2 optimization: Increase throughput by 50%".to_string(),
        timestamp: Utc::now(),
        priority: 2, // P2 without preceding P0/P1
    };
    
    let result = room.submit_tile(violating_tile);
    
    // Room should detect protocol violation
    match result {
        Ok(_) => {
            // If accepted, verify room state doesn't apply P2 without P1
            let state = room.get_state();
            assert!(!state.applied_p2_optimizations);
        }
        Err(e) => {
            assert!(e.contains("protocol") || e.contains("P0") || e.contains("P1"));
        }
    }
}
```
