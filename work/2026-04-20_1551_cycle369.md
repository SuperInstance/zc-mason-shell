## Identified Edge Cases (P0 Negative Space)
From reading the room implementation, I identified these untested edge conditions:

1. **State normalization failure** – When state values exceed the `state_bounds`.
2. **Action clipping at boundaries** – When actions exceed `action_bounds`, they should be clipped, not error.
3. **P0 violation with zero safe channels** – When all actions lead to constraint violations.
4. **Consecutive P0 violations** – Protocol requires resetting to a safe state after N violations.
5. **Random seed consistency** – Reproducibility of the environment's random sampling.

# test_deadband_room_edge.py
import unittest
import numpy as np
import gym
from plato_torch.rooms.deadband_room import DeadbandRoom

class TestDeadbandRoomEdgeCases(unittest.TestCase):
    
    def setUp(self):
        self.config = {
            'state_dim': 2,
            'action_dim': 2,
            'state_bounds': [[-1.0, 1.0], [-1.0, 1.0]],
            'action_bounds': [[-1.0, 1.0], [-1.0, 1.0]],
            'constraint_threshold': 0.8,
            'max_p0_violations': 3
        }
        self.env = DeadbandRoom(self.config)
    
    def test_state_normalization_out_of_bounds(self):
        """Test that states outside bounds are properly normalized."""
        # Create a state vector outside configured bounds
        raw_state = np.array([2.0, -2.0])  # Outside [-1, 1] bounds
        
        # The room should normalize this to within bounds
        self.env.state = raw_state
        normalized_state = self.env._normalize_state(raw_state)
        
        # Check normalization brings values within bounds
        self.assertTrue(np.all(normalized_state >= -1.0))
        self.assertTrue(np.all(normalized_state <= 1.0))
        
        # Verify the normalization is consistent (not just clipping)
        # Values should be scaled proportionally
        self.assertAlmostEqual(normalized_state[0], 1.0)  # 2.0 -> 1.0
        self.assertAlmostEqual(normalized_state[1], -1.0) # -2.0 -> -1.0
    
    def test_action_clipping_at_boundaries(self):
        """Test that actions exceeding bounds are clipped not rejected."""
        # Actions outside the [-1, 1] bounds
        extreme_actions = [
            np.array([1.5, 0.5]),   # First dimension exceeds upper bound
            np.array([-1.5, -0.5]), # First dimension exceeds lower bound
            np.array([0.5, 1.5]),   # Second dimension exceeds upper bound
        ]
        
        for action in extreme_actions:
            # This should not raise an error (clipping should happen internally)
            observation, reward, done, info = self.env.step(action)
            
            # The actual executed action should be clipped
            executed_action = info.get('executed_action', action)
            self.assertTrue(np.all(executed_action >= -1.0))
            self.assertTrue(np.all(executed_action <= 1.0))
            
            # Verify clipping occurred
            if np.any(action < -1.0) or np.any(action > 1.0):
                self.assertNotEqual(action.tolist(), executed_action.tolist())
    
    def test_p0_violation_with_no_safe_channels(self):
        """Test behavior when all possible actions violate constraints."""
        # Set up a state where all actions would violate constraints
        # Place state very close to constraint boundary
        self.env.state = np.array([0.79, 0.79])  # Just below threshold 0.8
        
        # The constraint function considers distance from origin
        # At this state, any non-zero action might push it over threshold
        
        # Try multiple actions that should all be P0 violations
        test_actions = [
            np.array([0.1, 0.1]),   # Small push over threshold
            np.array([-0.1, -0.1]), # Negative push still increases magnitude
            np.array([0.05, 0.05]), # Even smaller push
        ]
        
        violation_count = 0
        for action in test_actions:
            _, _, _, info = self.env.step(action)
            if info.get('phase') == 'P0':
                violation_count += 1
        
        # All actions should result in P0 violations in this setup
        self.assertEqual(violation_count, len(test_actions))
        
        # After max violations, environment should reset
        self.env.p0_violation_count = self.config['max_p0_violations'] - 1
        _, _, done, info = self.env.step(np.array([0.1, 0.1]))
        self.assertTrue(done)  # Should reset after max violations reached
    
    def test_consecutive_p0_violations_reset(self):
        """Test that consecutive P0 violations trigger a reset."""
        # Start with a clean state
        self.env.reset()
        initial_state = self.env.state.copy()
        
        # Force P0 violations
        self.env.state = np.array([0.9, 0.9])  # Already in violation territory
        
        violation_states = []
        for i in range(self.config['max_p0_violations'] + 1):
            _, _, done, info = self.env.step(np.array([0.0, 0.0]))  # Any action
            violation_states.append(self.env.state.copy())
            
            if info.get('phase') == 'P0':
                self.env.p0_violation_count += 1
            
            if done:
                # Reset should have occurred
                self.assertEqual(self.env.p0_violation_count, 0)
                # State should be different from violation state
                self.assertNotEqual(self.env.state.tolist(), violation_states[-1].tolist())
                break
    
    def test_random_seed_reproducibility(self):
        """Test that the environment is reproducible with same seed."""
        seed = 42
        
        # First run
        env1 = DeadbandRoom(self.config)
        env1.seed(seed)
        env1.reset()
        
        actions1 = []
        states1 = []
        for _ in range(5):
            action = env1.action_space.sample()
            actions1.append(action.copy())
            obs, _, _, _ = env1.step(action)
            states1.append(obs.copy())
        
        # Second run with same seed
        env2 = DeadbandRoom(self.config)
        env2.seed(seed)
        env2.reset()
        
        actions2 = []
        states2 = []
        for _ in range(5):
            action = env2.action_space.sample()
            actions2.append(action.copy())
            obs, _, _, _ = env2.step(action)
            states2.append(obs.copy())
        
        # All actions and states should be identical
        for a1, a2 in zip(actions1, actions2):
            np.testing.assert_array_equal(a1, a2)
        
        for s1, s2 in zip(states1, states2):
            np.testing.assert_array_equal(s1, s2)

if __name__ == '__main__':
    unittest.main()
