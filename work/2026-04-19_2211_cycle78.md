Edge-case tests for DeadbandRoom preset in plato-torch.

import pytest
import torch
from plato_torch.presets.deadband_room import DeadbandRoom
from plato_torch.core.tile import Tile

class TestDeadbandRoomEdgeCases:
    """Test suite for DeadbandRoom edge cases."""

    def test_p0_to_p2_priority_jump_raises_error(self):
        """P0: Attempting to jump from P0 directly to P2 should raise a protocol violation.
        
        Deadband Protocol requires strict priority: P0 → P1 → P2.
        Skipping P1 is a P0 violation (negative space).
        """
        room = DeadbandRoom()
        tile = Tile(priority="P2", content={"action": "optimize"})
        
        with pytest.raises(ValueError, match="Protocol violation: Cannot jump from P0 to P2"):
            room.process_tile(tile, current_priority="P0")

    def test_deadband_threshold_boundary_values(self):
        """P1: Testing boundary values for deadband thresholds.
        
        Deadband thresholds define safe channels. Values at exact boundaries
        should be handled deterministically (e.g., inclusive vs exclusive).
        """
        room = DeadbandRoom(deadband_low=0.1, deadband_high=0.9)
        
        # Test exactly at low boundary
        tile_low = Tile(priority="P1", content={"value": 0.1})
        result_low = room.process_tile(tile_low, current_priority="P0")
        assert result_low["in_deadband"] == True  # or False, depending on implementation
        
        # Test exactly at high boundary
        tile_high = Tile(priority="P1", content={"value": 0.9})
        result_high = room.process_tile(tile_high, current_priority="P0")
        assert result_high["in_deadband"] == True  # or False

    def test_concurrent_tiles_cause_state_conflict(self):
        """P0: Simulate concurrent tile submissions causing state conflicts.
        
        Multiple agents submitting tiles simultaneously may lead to race conditions.
        Room should handle conflicts gracefully (e.g., locking, conflict resolution).
        """
        room = DeadbandRoom()
        tile1 = Tile(priority="P1", content={"agent": "A", "action": "claim_channel"})
        tile2 = Tile(priority="P1", content={"agent": "B", "action": "claim_channel"})
        
        # Simulate concurrent processing (simplified)
        import threading
        results = []
        
        def process(tile):
            try:
                results.append(room.process_tile(tile, current_priority="P0"))
            except Exception as e:
                results.append(e)
        
        t1 = threading.Thread(target=process, args=(tile1,))
        t2 = threading.Thread(target=process, args=(tile2,))
        t1.start(); t2.start()
        t1.join(); t2.join()
        
        # At least one should succeed, the other may fail or be queued
        assert len(results) == 2
        assert any(isinstance(r, dict) for r in results)  # At least one successful tile

    def test_malformed_tile_missing_priority_field(self):
        """P0: Tile missing required 'priority' field should be rejected.
        
        Malformed input is a negative space—room should not crash.
        """
        room = DeadbandRoom()
