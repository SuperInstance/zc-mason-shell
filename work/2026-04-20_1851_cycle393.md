class DeadbandRoom:
    def __init__(self, config: Dict[str, Any]):
        self.validator = TileValidator(config.get('validator_rules', {}))
        self.ensign_exporter = config.get('ensign_exporter')
        self.tiles = []
        self.agents_trained = 0

    def submit_tile(self, agent_id: str, tile_data: Union[Dict, str]) -> Dict[str, Any]:
        # If tile_data is string, try to parse as JSON
        if isinstance(tile_data, str):
            try:
                tile_data = json.loads(tile_data)
            except json.JSONDecodeError:
                return {"status": "error", "reason": "invalid_json"}

        # P0: Check negative space
        p0_result = self.validator.check_p0(tile_data)
        if not p0_result["allowed"]:
            return {"status": "rejected", "phase": "P0", "reason": p0_result["reason"]}

        # P1: Find safe channels
        p1_channels = self.validator.find_p1_channels(tile_data)
        if not p1_channels:
            return {"status": "rejected", "phase": "P1", "reason": "no_safe_channels"}

        # P2: Optimize within first safe channel
        try:
            p2_optimized = self.validator.optimize_p2(tile_data, p1_channels[0])
        except Exception as e:
            return {"status": "error", "phase": "P2", "reason": str(e)}

        # Accept tile
        self.tiles.append({
            "agent_id": agent_id,
            "tile": p2_optimized,
            "timestamp": time.time()
        })
        self.agents_trained += 1

        # Export to ensign if configured
        if self.ensign_exporter:
            self.ensign_exporter.export(p2_optimized)

        return {"status": "accepted", "channel": p1_channels[0], "optimized": p2_optimized}

import pytest

from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom

class TestDeadbandRoomEdgeCases:
    """Edge-case tests for DeadbandRoom beyond basic coverage."""

    def test_submit_malformed_json_string(self):
        """Edge case: agent submits tile as string with invalid JSON syntax."""
        config = {"validator_rules": {}}
        room = DeadbandRoom(config)
        
        # Mock validator to not be called (should fail earlier)
        room.validator = Mock()
        
        result = room.submit_tile("agent1", "{invalid json, missing quote}")
        
        assert result["status"] == "error"
        assert result["reason"] == "invalid_json"
        room.validator.check_p0.assert_not_called()

    def test_ensign_exporter_none(self):
        """Edge case: ensign_exporter is None in config, ensure no export attempted."""
        config = {"validator_rules": {}, "ensign_exporter": None}
        room = DeadbandRoom(config)
        
        # Mock validator to return successful P0/P1/P2
        room.validator.check_p0.return_value = {"allowed": True}
        room.validator.find_p1_channels.return_value = ["channel_a"]
        room.validator.optimize_p2.return_value = {"optimized": True}
        
        result = room.submit_tile("agent1", {"test": "tile"})
        
        assert result["status"] == "accepted"
        assert room.ensign_exporter is None
        # No attribute to assert not called, but ensure no exception

    def test_p1_channels_empty_list(self):
        """Edge case: validator returns empty list for P1 channels (no safe channels)."""
        config = {"validator_rules": {}}
        room = DeadbandRoom(config)
        
        room.validator.check_p0.return_value = {"allowed": True}
        room.validator.find_p1_channels.return_value = []  # Empty!
        
        result = room.submit_tile("agent1", {"test": "tile"})
        
        assert result["status"] == "rejected"
        assert result["phase"] == "P1"
        assert result["reason"] == "no_safe_channels"
        room.validator.optimize_p2.assert_not_called()

    def test_p2_optimization_raises_exception(self):
        """Edge case: P2 optimization raises exception despite valid P0/P1."""
        config = {"validator_rules": {}}
        room = DeadbandRoom(config)
        
        room.validator.check_p0.return_value = {"allowed": True}
        room.validator.find_p1_channels.return_value = ["channel_a"]
        room.validator.optimize_p2.side_effect = ValueError("Optimization failed unexpectedly")
        
        result = room.submit_tile("agent1", {"test": "tile"})
        
        assert result["status"] == "error"
        assert result["phase"] == "P2"
        assert "Optimization failed" in result["reason"]

    @patch("threading.Thread")
    def test_concurrent_submissions_thread_safety(self, mock_thread):
        """Edge case: multiple agents submit tiles concurrently.
        Note: This test uses threading to simulate concurrency.
        """
        config = {"validator_rules": {}}
        room = DeadbandRoom(config)
        
        # Setup mock validator to accept all tiles quickly
        room.validator.check_p0.return_value = {"allowed": True}
        room.validator.find_p1_channels.return_value = ["channel_a"]
        room.validator.optimize_p2.return_value = {"optimized": True}
        
        tiles_submitted = []
        errors = []
        
        def submit_tile_task(agent_id, tile_data):
            try:
                result = room.submit_tile(agent_id, tile_data)
                tiles_submitted.append(result)
            except Exception as e:
                errors.append(e)
        
        # Simulate 10 concurrent submissions
        threads = []
        for i in range(10):
            t = mock_thread(target=submit_tile_task, args=(f"agent{i}", {"tile": i}))
            threads.append(t)
        
        # Start and join all threads (mocked)
        for t in threads:
            t.start()
        for t in threads:
            t.join()
        
        # Assertions
        assert len(errors) == 0, f"Concurrency errors: {errors}"
        assert len(tiles_submitted) == 10
        # All tiles should be accepted
        assert all(t["status"] == "accepted" for t in tiles_submitted)
        # Tile count should match
        assert len(room.tiles) == 10
