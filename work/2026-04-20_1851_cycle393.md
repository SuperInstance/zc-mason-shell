### Test 1: Maximum Tile Capacity
```python
def test_add_tile_exceeds_max_capacity(self, deadband_room):
    """Adding a tile beyond max_tiles should raise an error or drop the oldest tile."""
    room = deadband_room(max_tiles=2)
    room.add_tile({"id": "t1", "phase": "P0", "content": "test1"})
    room.add_tile({"id": "t2", "phase": "P1", "content": "test2"})
    # Attempt to add third tile
    with pytest.raises(RuntimeError) as exc_info:
        room.add_tile({"id": "t3", "phase": "P2", "content": "test3"})
    assert "max tiles" in str(exc_info.value).lower()
```

### Test 3: Concurrent Modification During Training

def test_concurrent_tile_addition_during_training(self, deadband_room, mocker):
    """Adding a tile while training should be blocked or queued."""
    room = deadband_room()
    room.add_tile({"id": "t1", "phase": "P0", "content": "test"})
    # Mock training to simulate long-running process
    mocker.patch.object(room, '_train_step', side_effect=lambda: time.sleep(0.1))
    # Start training in a separate thread
    import threading
    train_thread = threading.Thread(target=room.train_ensign, kwargs={'iterations': 1})
    train_thread.start()
    time.sleep(0.05)  # Let training start
    # Attempt to add tile during training
    with pytest.raises(RuntimeError) as exc_info:
        room.add_tile({"id": "t2", "phase": "P1", "content": "concurrent"})
    assert "training" in str(exc_info.value).lower()
    train_thread.join()
```

def test_export_ensign_corrupted_path(self, deadband_room, mocker):
    """Export should fail gracefully if adapter path is inaccessible."""
    room = deadband_room()
    room.add_tile({"id": "t1", "phase": "P0", "content": "test"})
    room.train_ensign(iterations=1)
    # Mock os.makedirs to raise PermissionError
    mocker.patch('os.makedirs', side_effect=PermissionError("Read-only filesystem"))
    with pytest.raises(PermissionError) as exc_info:
        room.export_ensign(adapter_path="/readonly/path/adapter")
    assert "read-only" in str(exc_info.value).lower()
```
