```python
def test_empty_constraint_sets():
    """P0: Room should handle empty constraint dictionaries gracefully."""
    room = DeadbandRoom(room_id="empty-test")
    
    # Empty P0 constraints (no negative space defined)
    result = room.validate_phase(agent_id="test-agent", phase=0, constraints={})
    assert result["valid"] == False
    assert "P0 constraints cannot be empty" in result["error"]
    
    # Empty P1 constraints after valid P0
    room.set_constraints(phase=0, constraints={"rock": "avoid"})
    result = room.validate_phase(agent_id="test-agent", phase=1, constraints={})
    assert result["valid"] == False
    assert "P1 constraints cannot be empty" in result["error"]
```

### Test 2: Concurrent Agent Phase Collisions
```python
def test_concurrent_phase_collisions():
    """P0: Multiple agents attempting same phase transition simultaneously."""
    room = DeadbandRoom(room_id="concurrent-test")
    room.set_constraints(phase=0, constraints={"rock": "avoid"})
    room.set_constraints(phase=1, constraints={"channel": "follow"})
    
    # Simulate 10 agents trying to move from P0 to P1 at same time
    agents = [f"agent-{i}" for i in range(10)]
    results = []
    
    for agent_id in agents:
        # All complete P0 first
        room.submit_tile(agent_id=agent_id, phase=0, data={"avoided": "rock"})
    
    # Concurrent P1 attempts
    for agent_id in agents:
        result = room.validate_phase(
            agent_id=agent_id, 
            phase=1, 
            constraints={"channel": "follow"}
        )
        results.append(result["valid"])
    
    # Should all succeed (room handles concurrency)
    assert all(results), "Concurrent phase transitions should not block"
```

### Test 3: Resource Exhaustion (Memory/Tiles)

```python
def test_tile_storage_exhaustion():
    """P0: Room behavior when tile storage reaches configured limit."""
    room = DeadbandRoom(
        room_id="exhaustion-test",
        max_tiles=5  # Small limit for testing
    )
    
    # Fill room with tiles
    for i in range(5):
        room.submit_tile(
            agent_id=f"filler-{i}",
            phase=0,
            data={"test": f"data-{i}"}
        )
    
    # Next tile should trigger cleanup or rejection
    result = room.submit_tile(
        agent_id="overflow-agent",
        phase=0,
        data={"test": "overflow"}
    )
    
    # Either old tiles are evicted or new tile is rejected
    assert "evicted" in result or "rejected" in result
    assert room.get_tile_count() <= 5
```

### Test 4: Network Partition Simulation
```python
def test_network_partition_recovery():
    """P1: Room should maintain consistency during simulated network issues."""
    room = DeadbandRoom(room_id="partition-test")
    
    # Simulate partition: room loses connection to validation service
    room._validation_available = False
    
    # Agent attempts phase transition during partition
    result = room.validate_phase(
        agent_id="partitioned-agent",
        phase=0,
        constraints={"rock": "avoid"}
    )
    
    # Should enter degraded mode, not crash
    assert result["valid"] == False
    assert "degraded" in result["mode"]
    assert "validation unavailable" in result["error"]
    
    # Recovery when network restored
    room._validation_available = True
    room.recover_from_partition()
    
    # Should process queued validations
    assert room._pending_validations == 0
```

```python
def test_byzantine_agent_input():
    """P0: Room should sanitize and handle malicious agent input."""
    room = DeadbandRoom(room_id="byzantine-test")
    
    # Test various attack vectors
    attack_vectors = [
        # SQL injection attempt
        {"constraints": {"'; DROP TABLE tiles; --": "attack"}},
        # XSS attempt
        {"constraints": {"<script>alert(1)</script>": "attack"}},
        # Path traversal
        {"constraints": {"../../../etc/passwd": "attack"}},
        # Extremely large payload
        {"constraints": {"A" * 1000000: "attack"}},
        # Malformed JSON/nested structures
        {"constraints": {"nested": {"deeper": {"malicious": True}}}},
    ]
    
    for attack in attack_vectors:
        result = room.validate_phase(
            agent_id="byzantine-agent",
            phase=0,
            constraints=attack["constraints"]
        )
        
        # Should reject or sanitize, not crash
        assert result["valid"] == False
        assert "invalid" in result["error"] or "sanitized" in result
        assert room._crash_count == 0  # Room should not crash
```
