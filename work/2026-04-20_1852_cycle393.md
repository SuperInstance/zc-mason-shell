### Test 3: Concurrent Tile Processing Race Condition
```python
def test_concurrent_tile_processing(self):
    """Simulate rapid tile submission from multiple agents."""
    room = DeadbandRoom()
    tiles = [
        Tile(content=f"Tile {i}", priority=i % 3, metadata={"agent_id": f"agent_{i % 2}"})
        for i in range(100)
    ]
    
    # Process all tiles rapidly
    results = []
    for tile in tiles:
        result = room.process_tile(tile)
        results.append(result)
    
    # Verify no data corruption
    processed_counts = {}
    for result in results:
        phase = result.get("assigned_phase", "unknown")
        processed_counts[phase] = processed_counts.get(phase, 0) + 1
    
    # Should maintain phase distribution integrity
    assert processed_counts.get("P0", 0) > 0  # Some tiles should be P0
    assert len(set([r.get("tile_id") for r in results])) == 100  # All unique
```

```python
def test_agent_state_rollback(self):
    """Test that agents skipping from P0 directly to P2 get rolled back."""
    agent = MockAgent(state={"phase": "P0", "channels": []})
    
    # Agent attempts to violate protocol by submitting P2 tile while in P0
    violating_tile = Tile(content="Skip attempt", priority=2, metadata={"agent_id": agent.id})
    
    room = DeadbandRoom()
    initial_state = agent.state.copy()
    
    result = room.process_tile(violating_tile)
    
    # Agent should be rolled back to P0 with violation flag
    assert result["protocol_violation"] is True
    assert agent.state["phase"] == "P0"
    assert agent.state.get("violation_count", 0) > 0
    assert "skip_attempt" in agent.state.get("violations", [])
```

### Test 5: Memory Exhaustion from Ghost Tile Accumulation
```python
def test_ghost_tile_memory_management(self):
    """Test handling of excessive ghost tile accumulation."""
    room = DeadbandRoom()
    
    # Simulate many ghost tiles from terminated agents
    for i in range(1000):
        ghost_tile = Tile(
            content=f"Ghost {i}",
            priority=0,
            metadata={"agent_id": f"terminated_{i}", "ghost": True}
        )
        room.archive_tile(ghost_tile)
    
    # Force memory cleanup
    room.cleanup_ghost_tiles(threshold=500)
    
    # Should retain recent ghosts but prune older ones
    assert len(room.ghost_tiles) <= 500
    assert room.memory_usage < room.config["max_ghost_memory"]
    
    # Recent ghosts (last 100) should remain
    recent_ghosts = [t for t in room.ghost_tiles if "terminated_9" in t.metadata.get("agent_id", "")]
    assert len(recent_ghosts) > 0
```
