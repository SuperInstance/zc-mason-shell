```python
def test_deadband_room_concurrent_collision(self):
    """Test multiple agents attempting conflicting P1 safe channels."""
    room = DeadbandRoom()
    
    # Agent 1 claims channel A
    decision1 = {
        "action": "occupy_channel",
        "channel": "A",
        "score": 0.8,
        "confidence": 0.8,
        "metadata": {"resource": "cpu"}
    }
    result1 = room.process_decision("agent1", decision1, cycle=1)
    self.assertTrue(result1["approved"])
    self.assertEqual(result1["assigned_channel"], "A")
    
    # Agent 2 attempts same channel (should be rejected or reassigned)
    decision2 = {
        "action": "occupy_channel", 
        "channel": "A",
        "score": 0.9,
        "confidence": 0.9,
        "metadata": {"resource": "cpu"}
    }
    result2 = room.process_decision("agent2", decision2, cycle=1)
    
    # Verify collision resolution
    self.assertNotEqual(result2.get("assigned_channel"), "A")
    self.assertTrue(result2["approved"])  # Should find alternative channel
    self.assertIn("alternative_channel", result2)
```

```python
def test_deadband_room_resource_exhaustion(self):
    """Test behavior when all safe channels are occupied."""
    room = DeadbandRoom()
    
    # Exhaust all available channels
    channels = ["A", "B", "C", "D", "E"]  # Assuming 5 channels
    for i, channel in enumerate(channels):
        decision = {
            "action": "occupy_channel",
            "channel": channel,
            "score": 0.7 + (i * 0.05),
            "confidence": 0.7,
            "metadata": {"agent_id": f"agent{i}"}
        }
        result = room.process_decision(f"agent{i}", decision, cycle=1)
        self.assertTrue(result["approved"])
    
    # Next agent should trigger resource exhaustion handling
    decision = {
        "action": "occupy_channel",
        "channel": "F",  # Non-existent channel
        "score": 0.8,
        "confidence": 0.8
    }
    result = room.process_decision("agentX", decision, cycle=1)
    
    # Should either queue or provide fallback behavior
    self.assertFalse(result["approved"])
    self.assertIn("exhausted", result["reason"].lower() or result.get("status", "").lower())
    self.assertIn("queue_position", result or "retry_after" in result)
```
