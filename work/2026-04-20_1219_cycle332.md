## Identified Edge Cases (P0 Negative Space)
Based on reading the `deadband_room.py` implementation, I identified the following untested edge conditions:

1. **Concurrent tile submission** – Room's `submit_tile` isn't thread-safe; simultaneous calls could corrupt internal state.
2. **Malformed sentiment scores** – Sentiment field expects float 0.0-1.0; values outside range or wrong type aren't validated.
3. **Protocol phase skipping** – The `execute_protocol_step` should reject P2 calls before P1 completion, but only checks via `current_phase`.
4. **Massive tile payloads** – No size limits on tile content; could cause memory issues.
5. **Room reset during active training** – `reset_room` method doesn't check if training is in progress.

### Test 1: Concurrent Tile Submission Race Condition
```python
import threading
import pytest

def test_concurrent_tile_submission_race(deadband_room):
    """P0: Simultaneous tile submissions should not corrupt room state."""
    room = deadband_room
    errors = []
    
    def submit_tile_thread():
        try:
            tile = {"content": "test", "sentiment": 0.5, "phase": "P0"}
            room.submit_tile(tile)
        except Exception as e:
            errors.append(e)
    
    threads = [threading.Thread(target=submit_tile_thread) for _ in range(10)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Room should still be in valid state
    assert room.current_phase in ["P0", "P1", "P2"]
    assert len(errors) == 0, f"Threads raised errors: {errors}"
```

### Test 2: Invalid Sentiment Score Boundaries
```python
def test_sentiment_score_boundary_validation(deadband_room):
    """P0: Sentiment scores outside 0.0-1.0 should be rejected."""
    room = deadband_room
    
    # Test below lower bound
    with pytest.raises(ValueError, match="Sentiment must be between 0.0 and 1.0"):
        room.submit_tile({"content": "test", "sentiment": -0.1, "phase": "P0"})
    
    # Test above upper bound  
    with pytest.raises(ValueError, match="Sentiment must be between 0.0 and 1.0"):
        room.submit_tile({"content": "test", "sentiment": 1.1, "phase": "P0"})
    
    # Test non-float type
    with pytest.raises(TypeError, match="Sentiment must be a float"):
        room.submit_tile({"content": "test", "sentiment": "high", "phase": "P0"})
```

### Test 3: Protocol Phase Skipping Enforcement
```python
def test_protocol_phase_skipping_prevention(deadband_room):
    """P0: Attempting P2 before P1 completion should fail."""
    room = deadband_room
    
    # Start in P0
    assert room.current_phase == "P0"
    
    # Try to jump directly to P2 (should fail)
    with pytest.raises(RuntimeError, match="Cannot execute P2 before completing P1"):
        room.execute_protocol_step("P2")
    
    # Complete P0 normally
    room.execute_protocol_step("P0")
    assert room.current_phase == "P1"
    
    # Try P2 again (should still fail since P1 not complete)
    with pytest.raises(RuntimeError, match="Cannot execute P2 before completing P1"):
        room.execute_protocol_step("P2")
```

def test_oversized_tile_payload_rejection(deadband_room):
    """P0: Extremely large tile content should be rejected."""
    room = deadband_room
    
    # Generate 10MB string (exceeds reasonable tile size)
    oversized_content = "x" * (10 * 1024 * 1024)
    
    with pytest.raises(ValueError, match="Tile content exceeds maximum size"):

            "sentiment": 0.5,
            "phase": "P0"
        })
    
    # Verify room state unchanged
    assert len(room.submitted_tiles) == 0

### Test 5: Room Reset During Active Training
```python
def test_reset_during_active_training(deadband_room):
    """P0: Resetting room while training active should warn or block."""
    room = deadband_room
    
    # Start training
    room.start_training()
    assert room.training_active is True
    
    # Attempt reset (should raise warning or error)
    with pytest.warns(UserWarning, match="Resetting room during active training"):
        room.reset_room()
    
    # Training should be stopped after reset
    assert room.training_active is False
    assert room.current_phase == "P0"
    assert len(room.submitted_tiles) == 0
```
