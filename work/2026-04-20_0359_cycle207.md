### Test 1: Protocol Priority Violation
Tests that the room rejects attempts to skip P0 or P1 and jump directly to P2 optimization.
```python
def test_protocol_priority_violation_skipping_p0(self):
    """P0: Attempt to submit P2 tile without P0/P1 mapping should fail."""
    room = DeadbandRoom(room_id="edge_priority")
    # Try to submit P2 tile directly
    tile = {"phase": "P2", "content": "optimized path", "timestamp": time.time()}
    result = room.submit_tile(tile)
    assert not result["accepted"]
    assert "P0" in result.get("error", "").lower()
```

### Test 2: Empty and Malformed Tile Submission
Tests boundary cases for tile structure validation.

```python
def test_empty_and_malformed_tiles(self):
    """P0: Empty, None, and structurally invalid tiles should be rejected."""
    room = DeadbandRoom(room_id="edge_malformed")
    
    test_cases = [
        None,
        {},
        {"phase": "P0"},  # Missing required fields
        {"content": "test", "timestamp": "not-a-float"},  # Wrong type

    ]

        result = room.submit_tile(tile)
        assert not result["accepted"]
        assert "invalid" in result.get("error", "").lower()
```

### Test 3: Concurrent Tile Submission Race Condition
Simulates multiple agents submitting tiles simultaneously.
```python
def test_concurrent_tile_submission(self):
    """P1: Multiple concurrent submissions should not corrupt room state."""
    room = DeadbandRoom(room_id="edge_concurrent")
    tiles = [
        {"phase": "P0", "content": f"negative_{i}", "timestamp": time.time()}
        for i in range(10)
    ]
    
    # Simulate concurrent submission
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All should be accepted
    accepted = sum(1 for r in results if r["accepted"])
    assert accepted == 10
    # Room tile count should match
    assert room.get_tile_count() == 10
```

```python
def test_state_persistence_across_reset(self):

    tile_p0 = {"phase": "P0", "content": "negative space mapped", "timestamp": time.time()}
    room.submit_tile(tile_p0)
    
    # Simulate reset by creating new instance with same ID
    # (In real scenario, this would be persistence layer)
    room2 = DeadbandRoom(room_id="edge_persistence")
    # Room should know P0 is complete, allow P1
    tile_p1 = {"phase": "P1", "content": "safe channel found", "timestamp": time.time()}
    result = room2.submit_tile(tile_p1)
    assert result["accepted"]
```
