### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Test that room handles multiple simultaneous tile submissions correctly."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Negative space map from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    assert all(r.get("status") == "accepted" for r in results)
    
    # Verify tiles are stored separately
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == len(agent_ids)
```

### Test 2: Malformed Tile Payloads
```python
def test_malformed_tile_payloads(self):
    """Test room rejection of various malformed tile payloads."""
    room = DeadbandRoom()
    
    test_cases = [
        # Missing required fields
        {"agent_id": "test_agent"},  # missing phase, content
        {"phase": "P0", "content": "test"},  # missing agent_id
        {"agent_id": "test_agent", "phase": "P0"},  # missing content
        
        # Invalid data types
        {"agent_id": 123, "phase": "P0", "content": "test"},  # agent_id not string
        {"agent_id": "test_agent", "phase": 123, "content": "test"},  # phase not string
        {"agent_id": "test_agent", "phase": "P0", "content": 123},  # content not string
        
        # Invalid phase values
        {"agent_id": "test_agent", "phase": "P3", "content": "test"},  # P3 doesn't exist
        {"agent_id": "test_agent", "phase": "", "content": "test"},  # empty phase
        {"agent_id": "test_agent", "phase": "P2", "content": "test"},  # P2 without P1
        
        # Invalid JSON structure
        "not_a_dict",
        [],
        None
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.submit_tile(tile)
        assert result.get("status") == "rejected", f"Test case {i} should be rejected: {tile}"
        assert "error" in result, f"Test case {i} should include error message"
```

### Test 3: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Test that agents can recover after protocol violations."""
    room = DeadbandRoom()
    agent_id = "recovery_agent"
    
    # Attempt P2 without completing P1 (should fail)
    tile_p2 = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Trying to optimize without safe channel",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p2)
    assert result.get("status") == "rejected"
    assert "P1" in result.get("error", "").lower()  # Error should mention P1
    
    # Submit proper P0 tile
    tile_p0 = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Negative space identified",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p0)
    assert result.get("status") == "accepted"
    
    # Submit P1 tile
    tile_p1 = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Safe channel found",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p1)
    assert result.get("status") == "accepted"
    
    # Now P2 should work
    tile_p2_retry = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Optimizing within safe channel",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p2_retry)
    assert result.get("status") == "accepted"
    
    # Verify all tiles stored
    agent_tiles = room.get_agent_tiles(agent_id)
    assert len(agent_tiles) == 3
    phases = [t.get("phase") for t in agent_tiles]
    assert phases == ["P0", "P1", "P2"]
```

### Test 4: State Persistence Across Resets
```python
def test_state_persistence_across_resets(self):
    """Test room state consistency after multiple reset cycles."""
    room = DeadbandRoom()
    
    # First session
    agent1 = "agent_session_1"
    tile1 = {
        "agent_id": agent1,
        "phase": "P0",
        "content": "Session 1 P0",
        "timestamp": time.time()
    }
    room.submit_tile(tile1)
    
    # Verify state
    assert len(room.get_tiles()) == 1
    assert room.get_agent_state(agent1)["current_phase"] == "P0"
    
    # Reset room
    room.reset()
    
    # Verify clean state
    assert len(room.get_tiles()) == 0
    assert room.get_agent_state(agent1) == {}  # Should be empty or default
    
    # Second session with same agent
    tile2 = {
        "agent_id": agent1,
        "phase": "P0",
        "content": "Session 2 P0 - fresh start",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile2)
    assert result.get("status") == "accepted"
    
    # Third session with different agent after another reset
    room.reset()
    agent2 = "agent_session_3"
    tile3 = {
        "agent_id": agent2,
        "phase": "P0",
        "content": "Session 3 P0",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile3)
    assert result.get("status") == "accepted"
    
    # Verify no cross-session contamination
    agent1_state = room.get_agent_state(agent1)
    agent2_state = room.get_agent_state(agent2)
    assert agent1_state == {} or agent1_state.get("current_phase") is None
    assert agent2_state.get("current_phase") == "P0"
```
