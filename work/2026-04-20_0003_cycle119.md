### Test 1: Extreme Constraint Values
```python
def test_deadband_extreme_constraints():
    """Test room behavior with constraint values at numerical limits."""
    room = DeadbandRoom()
    
    # Test with negative constraints (should be rejected)
    with pytest.raises(ValueError):
        room.process({"constraints": {"min": -100, "max": -50}})
    
    # Test with zero-range constraints (edge case for P1)
    result = room.process({"constraints": {"min": 42, "max": 42}})
    assert result["phase"] == "P1"
    assert len(result["channels"]) == 1  # Single point channel
    
    # Test with extremely large values (should not crash)
    large_val = 1e18
    result = room.process({"constraints": {"min": large_val, "max": large_val * 2}})
    assert "phase" in result
```

### Test 2: Concurrent Agent Simulation

```python
def test_deadband_concurrent_agents():
    """Simulate multiple agents interacting with the same room state."""
    room = DeadbandRoom()
    
    # Agent 1 establishes P0
    agent1_result = room.process({
        "agent_id": "agent_1",
        "constraints": {"min": 0, "max": 100}
    })
    assert agent1_result["phase"] == "P0"
    
    # Agent 2 tries to establish different P0 (should see existing)
    agent2_result = room.process({
        "agent_id": "agent_2", 
        "constraints": {"min": 50, "max": 150}
    })
    assert agent2_result["phase"] == "P0"
    # Should reference existing negative space from agent_1
    assert "existing_negative_space" in agent2_result
    
    # Agent 1 moves to P1
    agent1_p1 = room.process({
        "agent_id": "agent_1",
        "phase": "P1",
        "negative_space": agent1_result["negative_space"]
    })
    assert agent1_p1["phase"] == "P1"
    
    # Verify room state tracks both agents
    assert len(room.agent_states) == 2
```

### Test 3: Partial Protocol Corruption

```python
def test_deadband_corrupted_protocol():
    """Test handling of agents that partially follow protocol."""
    room = DeadbandRoom()
    
    # Agent completes P0 correctly
    p0_result = room.process({
        "constraints": {"min": 0, "max": 100},
        "phase": "P0"
    })
    
    # Agent tries to skip to P2 with corrupted P1 data
    with pytest.raises(ProtocolViolationError):
        room.process({
            "phase": "P2",
            "negative_space": p0_result["negative_space"],
            "channels": ["fake_channel"]  # Not from valid P1
        })
    
    # Agent provides valid P1 but corrupted optimization function
    p1_result = room.process({
        "phase": "P1",
        "negative_space": p0_result["negative_space"]
    })
    
    with pytest.raises(ValueError):
        room.process({
            "phase": "P2",
            "negative_space": p0_result["negative_space"],
            "channels": p1_result["channels"],
            "optimizer": "invalid_function()"  # Corrupted optimizer
        })
```

### Test 4: Resource Exhaustion Protection
```python
def test_deadband_resource_limits():
    """Test room behavior when approaching system limits."""
    room = DeadbandRoom(max_agents=3, max_constraints_per_agent=1000)
    
    # Fill room with agents
    for i in range(3):
        room.process({
            "agent_id": f"agent_{i}",
            "constraints": {"min": i * 100, "max": i * 100 + 50}
        })
    
    # Fourth agent should be rejected
    with pytest.raises(ResourceLimitError):
        room.process({
            "agent_id": "agent_4",
            "constraints": {"min": 400, "max": 450}
        })
    
    # Test constraint limit
    many_constraints = [{"min": i, "max": i + 1} for i in range(1001)]
    with pytest.raises(ResourceLimitError):
        room.process({
            "agent_id": "agent_0",
            "constraints": many_constraints
        })
    
    # Verify room continues working for existing agents
    result = room.process({
        "agent_id": "agent_0",
        "phase": "P1",
        "negative_space": {"min": 0, "max": 50}
    })
    assert result["phase"] == "P1"
```

```python
def test_deadband_state_recovery():
    """Test room recovery from corrupted or invalid states."""
    room = DeadbandRoom()
    
    # Establish normal state
    room.process({
        "agent_id": "recovery_test",
        "constraints": {"min": 0, "max": 100}
    })
    
    # Corrupt internal state (simulating bug or external interference)
    room.agent_states["recovery_test"]["phase"] = "INVALID_PHASE"
    
    # Room should detect and handle corrupted state
    result = room.process({
        "agent_id": "recovery_test",
        "phase": "P1",
        "negative_space": {"min": 0, "max": 100}
    })
    
    # Either recovers gracefully or resets agent
    assert result["phase"] in ["P0", "P1"]  # Valid phase
    if result["phase"] == "P0":
        assert "state_reset" in result  # Indicates recovery action
    
    # Test recovery from serialization errors
    import pickle
    corrupted_data = pickle.dumps({"corrupted": True})[:50]  # Truncated
    
    room = DeadbandRoom.from_bytes(corrupted_data)
    # Should either raise error or create fresh instance
    assert room is not None
    assert len(room.agent_states) == 0  # Fresh start
```
