### Test 1: Protocol Violation – Skipping Directly to P2
```python
def test_deadband_room_skips_to_p2_violation():
    """P0: Ensure room rejects interactions that skip P0/P1 and go straight to P2 optimization."""
    room = DeadbandRoom()
    interaction = {
        "user_input": "Optimize this channel",
        "metadata": {"priority": 2, "skip_safety_check": True}
    }
    result = room.process_interaction(interaction)
    assert "error" in result
    assert "P0" in result["error"]  # Should flag protocol violation
    assert result["tile_valid"] is False
```

def test_deadband_room_duplicate_interaction_id():
    """P2: Test handling of duplicate interaction IDs (simulating concurrent requests)."""
    room = DeadbandRoom()
    interaction_id = "test_123"
    interaction1 = {
        "id": interaction_id,
        "user_input": "First",
        "metadata": {"priority": 2, "channel": "test_channel"}
    }
    interaction2 = {
        "id": interaction_id,  # Same ID
        "user_input": "Second",
        "metadata": {"priority": 2, "channel": "test_channel"}
    }

    result2 = room.process_interaction(interaction2)
    
    # Should either reject second or handle with versioning
    assert result1["tile_valid"] is True
    # Second might be rejected or marked as conflict
    assert "conflict" in result2.get("status", "") or result2["tile_valid"] is False
