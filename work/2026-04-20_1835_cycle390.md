### 1. Test Tile Capacity Boundary
```python
def test_add_tile_at_capacity(self):
    """Test adding a tile when room is at max_tiles capacity."""
    room = DeadbandRoom(max_tiles=2)
    tile1 = {"id": "t1", "content": "test1", "phase": "P0"}
    tile2 = {"id": "t2", "content": "test2", "phase": "P1"}
    tile3 = {"id": "t3", "content": "test3", "phase": "P2"}
    
    # Add up to capacity
    room.add_tile(tile1)
    room.add_tile(tile2)
    self.assertEqual(len(room.tiles), 2)
    
    # Adding third tile should raise error or be ignored
    with self.assertRaises(RoomCapacityError):
        room.add_tile(tile3)
```

### 2. Test Malformed Tile Recovery
```python
def test_add_malformed_tile_recovery(self):
    """Test room recovers after receiving malformed tile data."""
    room = DeadbandRoom()
    
    # Partial JSON structure (missing required fields)
    bad_tile = {"id": "bad"}  # Missing 'content' and 'phase'
    
    with self.assertRaises(InvalidTileError):
        room.add_tile(bad_tile)
    
    # Room should still accept valid tiles after error
    good_tile = {"id": "good", "content": "valid", "phase": "P0"}
    room.add_tile(good_tile)
    self.assertEqual(len(room.tiles), 1)
    self.assertEqual(room.tiles[0]["id"], "good")
```

### 3. Test Protocol Priority Violation Detection
```python
def test_detect_p2_before_p1(self):
    """Test room detects and rejects P2 tiles when P1 not satisfied."""
    room = DeadbandRoom()
    
    # Add P0 tile (valid)
    room.add_tile({"id": "p0", "content": "avoid", "phase": "P0"})
    
    # Try to add P2 tile without P1 - should be rejected
    p2_tile = {"id": "p2", "content": "optimize", "phase": "P2"}
    
    with self.assertRaises(ProtocolViolationError):
        room.add_tile(p2_tile)
    
    # Add P1 tile, then P2 should work
    room.add_tile({"id": "p1", "content": "channel", "phase": "P1"})
    room.add_tile(p2_tile)
    self.assertEqual(len(room.tiles), 3)
```

### 5. Test Export With Partial Training
```python
def test_export_after_partial_training(self):
    """Test exporting ensign after interrupted training."""
    room = DeadbandRoom()
    
    # Add some tiles but don't complete full training
    room.add_tile({"id": "t1", "content": "test", "phase": "P0"})
    
    # Start training but simulate interruption
    room.train_ensign(epochs=1, complete=False)
    
    # Export should handle partially trained state
    adapter = room.export_ensign()
    
    # Adapter should exist but may have limited weights
    self.assertIsNotNone(adapter)
    self.assertTrue(hasattr(adapter, "layers"))
    self.assertLess(len(adapter.layers), 3)  # Few layers due to partial training
```
