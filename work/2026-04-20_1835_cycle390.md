### Test 2: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Simulate multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    
    # Create multiple tile submission threads
    submission_count = {"accepted": 0}
    
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "content": f"Test from agent {agent_id}",
            "timestamp": time.time()
        }
        if room.submit_tile(tile):
            submission_count["accepted"] += 1
    
    # Simulate 10 concurrent agents
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All submissions should be accepted (room handles concurrency)
    self.assertEqual(submission_count["accepted"], 10, 
                    "All concurrent submissions should be accepted")
```

```python
def test_partial_connectivity_recovery(self):
    """Test room recovery after simulated network partition."""
    room = DeadbandRoom()
    
    # Simulate network partition
    room._simulate_network_partition = True
    
    # Attempt tile submission during partition
    tile = {"agent_id": "test_agent", "content": "During partition"}
    result = room.submit_tile(tile)
    self.assertFalse(result, "Should reject tiles during network partition")
    
    # Recover from partition
    room._simulate_network_partition = False
    room.recover_from_partition()
    
    # Verify recovery
    self.assertTrue(room.is_healthy(), "Room should be healthy after recovery")
    
    # Resubmit tile
    result = room.submit_tile(tile)
    self.assertTrue(result, "Should accept tiles after recovery")
```

```python
def test_malformed_protocol_data(self):
    """Test handling of intentionally malformed P0/P1/P2 data."""
    room = DeadbandRoom()
    
    # Test with circular reference (should not crash)
    malicious_data = {"p0": {"self_ref": None}}
    malicious_data["p0"]["self_ref"] = malicious_data
    
    try:
        room.process_protocol_data(malicious_data)
        # Should handle gracefully, not crash
        self.assertTrue(True, "Handled circular reference without crash")
    except RecursionError:
        self.fail("Room crashed on circular reference")
    
    # Test with extremely nested data
    nested = {"p0": {}}
    current = nested["p0"]
    for i in range(1000):
        current[f"level_{i}"] = {}
        current = current[f"level_{i}"]
    
    result = room.process_protocol_data(nested)
    self.assertIsNotNone(result, "Should process deeply nested data")
```

```python
def test_resource_exhaustion_handling(self):
    """Test room behavior under resource constraints."""
    room = DeadbandRoom()
    
    # Simulate file descriptor exhaustion
    original_open = builtins.open
    
    def exhausted_open(*args, **kwargs):
        raise OSError(24, "Too many open files")
    
    builtins.open = exhausted_open
    
    try:
        # Attempt to save room state (should fail gracefully)
        result = room.save_state("/tmp/test_state.json")
        self.assertFalse(result, "Should handle file descriptor exhaustion")
        
        # Room should remain operational for in-memory operations
        self.assertTrue(room.is_operational(), 
                       "Should remain operational despite resource issues")
    finally:
        builtins.open = original_open
    
    # Test memory pressure simulation
    large_data = "x" * (10 * 1024 * 1024)  # 10MB string
    
    try:
        room.cache_large_data(large_data)
        # Should handle large data without crash
        self.assertTrue(True, "Handled memory pressure without crash")
    except MemoryError:
        self.fail("Room crashed on memory pressure")
```
