def test_concurrent_agent_phase_conflict():
    """Test multiple agents attempting different phase operations."""
    room = DeadbandRoom(room_id="test_concurrent")
    agent1 = Mock(agent_id="agent1")
    agent2 = Mock(agent_id="agent2")
    
    room.register_agent(agent1, "agent1")
    room.register_agent(agent2, "agent2")
    
    # Agent1 sets phase to P0
    room.set_phase("P0")
    
    # Agent2 tries to submit P2 tile (should fail)
    p2_tile = {"type": "optimization", "content": "skip", "priority": "P2"}
    result = room.submit_tile(agent2, p2_tile)
    assert result["status"] == "rejected"
    assert "phase violation" in result["reason"].lower()
    
    # Agent1 can still submit P0 tile
    p0_tile = {"type": "constraint", "content": "avoid", "priority": "P0"}
    result = room.submit_tile(agent1, p0_tile)
    assert result["status"] == "accepted"
```

def test_room_reset_persistence():
    """Test that certain room state persists across soft resets."""
    room = DeadbandRoom(room_id="test_persistence")
    mock_agent = Mock()
    
    room.register_agent(mock_agent, "test_agent")
    
    # Submit several tiles
    for i in range(3):
        tile = {"type": "test", "content": f"tile_{i}", "priority": "P0"}
        room.submit_tile(mock_agent, tile)
    
    initial_tile_count = room.get_tile_count()
    initial_agents = room.get_registered_agents()
    
    # Soft reset (simulate room restart)
    room.reset(soft=True)
    
    # Tile count should reset, agent registry should persist
    assert room.get_tile_count() == 0
    assert room.get_registered_agents() == initial_agents
    assert "test_agent" in room.get_registered_agents()
```
