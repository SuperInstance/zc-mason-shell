def test_protocol_state_valid_transitions():
    """Test that room correctly tracks P0→P1→P2 progression."""
    room = DeadbandRoom(
        negative_space=["invalid_action"],
        safe_channels=["channel_a", "channel_b"],
        optimization_metric="throughput"
    )
    
    # Should start in P0
    assert room.current_phase == "P0"
    
    # Complete P0
    room.complete_phase("P0")
    assert room.current_phase == "P1"
    
    # Complete P1
    room.complete_phase("P1")
    assert room.current_phase == "P2"
    
    # Should not regress
    with pytest.raises(ProtocolViolationError):
        room.complete_phase("P1")

    )

    ]
    
    room.load_optimization_data(test_data)
    
    # With ties, should select first occurrence or use tiebreaker
    result = room.optimize_channel("only_channel")

    assert result["score"] == 100

### Test 5: Protocol Violation Recovery and Continuation
```python
def test_protocol_violation_recovery():
    """Test that room recovers gracefully after protocol violation."""
    room = DeadbandRoom(
        negative_space=["skip_step"],
        safe_channels=["recovery_path"],
        optimization_metric="resilience"
    )
    
    # Force a P2 before P1 violation
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.complete_phase("P2")
    
    assert "P1 not completed" in str(exc_info.value)
    
    # Room should remain in P0 state
    assert room.current_phase == "P0"
    
    # Should be able to continue normal operation
    room.complete_phase("P0")
    assert room.current_phase == "P1"
    
    # Subsequent operations should work
    room.complete_phase("P1")
    assert room.current_phase == "P2"
    
    # Optimization should proceed normally
    result = room.optimize_channel("recovery_path")
    assert "optimized" in result
```
