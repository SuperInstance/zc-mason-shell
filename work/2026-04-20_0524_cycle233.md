### Test 1: Protocol Priority Enforcement with Partial P0 Completion
**Scenario:** Agent attempts P1 before P0 is fully mapped (some negative space identified but not all).
```python
def test_partial_p0_before_p1():
    """Test that P1 is blocked if P0 mapping is incomplete."""
    room = DeadbandRoom()
    # Simulate partial P0 mapping (only some negative space)
    room.current_state = {
        "phase": "P0",
        "negative_space": ["rock1", "rock2"],
        "p0_complete": False  # Still mapping
    }
    
    with pytest.raises(ProtocolViolationError) as excinfo:
        room.execute_phase("P1", {"constraints": ["channel_a"]})
    
    assert "P0 not complete" in str(excinfo.value)
```

### Test 2: Empty Safe Channels After P1
**Scenario:** P1 finds no safe channels (all constraints lead to negative space).
```python
def test_no_safe_channels_after_p1():
    """Test handling when P1 yields zero safe channels."""
    room = DeadbandRoom()
    room.current_state = {
        "phase": "P0",
        "negative_space": ["rock1", "rock2", "rock3"],
        "p0_complete": True
    }
    
    result = room.execute_phase("P1", {
        "constraints": ["through_rock1", "through_rock2"]
    })
    
    assert result["phase"] == "P1"
    assert result["safe_channels"] == []
    assert result["p1_complete"] == True
    # Should not allow progression to P2
    assert room.can_proceed_to_p2() == False
```

    room.negative_space = ["rock1", "rock2"]
    
    malformed_tile = {
        "phase": "P0",

### Test 5: Concurrent Protocol Execution Simulation
**Scenario:** Simulate multiple agents attempting different phases simultaneously.
```python
def test_concurrent_phase_execution():
    """Test room handling of simulated concurrent phase requests."""
    room = DeadbandRoom()
    
    # Simulate agent A at P0
    room.current_state = {
        "phase": "P0",
        "negative_space": [],
        "p0_complete": False,
        "agent_id": "agent_a"
    }
    
    # Simulate agent B trying to jump ahead (should fail)
    with pytest.raises(ConcurrentProtocolError) as excinfo:
        room.execute_phase("P2", {
            "agent_id": "agent_b",
            "optimization_metric": "speed"
        })
    
    assert "different agent" in str(excinfo.value).lower()
    assert "P0" in str(excinfo.value)
    
    # Agent A completes P0
    room.execute_phase("P0", {
        "agent_id": "agent_a",
        "negative_space": ["rock1"]
    })
    
    # Now agent B can start at P1
    result = room.execute_phase("P1", {
        "agent_id": "agent_b",
        "constraints": ["avoid_rock1"]
    })
    
    assert result["phase"] == "P1"
    assert result["agent_id"] == "agent_b"
```
