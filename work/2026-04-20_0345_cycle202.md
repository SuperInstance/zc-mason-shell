```python
def test_skip_p0_to_p2_violation():
    """P0: Attempt to execute P2 step without completing P1 should fail."""
    room = DeadbandRoom(room_id="edge_skip_test")
    # Force set current phase to P0
    room.current_phase = "P0"
    
    # Attempt P2 step directly
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.execute_p2_step(optimization_data={"path": "shortcut"})
    
    assert "Cannot skip from P0 to P2" in str(exc_info.value)
    assert room.current_phase == "P0"  # State unchanged
```

### Test 2: State Persistence Across Sessions

```python
def test_state_persistence_corruption():
    """P0: Room state should not leak between different agent sessions."""
    room = DeadbandRoom(room_id="edge_persistence_test")
    
    # First agent session
    agent1_id = "agent_alpha"
    room.register_agent(agent1_id)
    room.execute_p0_step(negative_space={"rock": True})
    room.submit_tile(agent1_id, {"lesson": "avoid rock"})
    
    # Simulate session end (room should reset agent-specific state)
    room.clear_agent_session(agent1_id)
    
    # Second agent session
    agent2_id = "agent_beta"
    room.register_agent(agent2_id)
    
    # Agent2 should not see Agent1's P0 results
    assert room.get_agent_tiles(agent2_id) == []
    assert room.current_phase == "P0"  # Fresh start
    # Agent1's tiles still in archive but not in active session
    assert len(room.archive.get_tiles(agent1_id)) == 1
```

### Test 3: Large Batch Tile Submission

```python
def test_large_tile_batch_performance():
    """P1: Room should handle large tile batches without resource exhaustion."""
    room = DeadbandRoom(room_id="edge_batch_test")
    agent_id = "batch_agent"
    room.register_agent(agent_id)
    
    # Generate 10,000 tiles (simulating training data dump)
    large_batch = [
        {"cycle": i, "observation": f"test_{i}", "priority": "P1"}
        for i in range(10000)
    ]
    
    import time
    start_time = time.time()
    
    for tile in large_batch:
        room.submit_tile(agent_id, tile)
    
    end_time = time.time()
    processing_time = end_time - start_time
    
    # Performance constraint: < 0.1ms per tile average
    avg_time_per_tile = processing_time / 10000
    assert avg_time_per_tile < 0.0001, f"Slow tile processing: {avg_time_per_tile}s per tile"
    
    # Memory check (approximate)
    import sys
    room_size = sys.getsizeof(room) + sum(sys.getsizeof(t) for t in large_batch[:100])
    assert room_size < 50 * 1024 * 1024  # < 50MB for room + sample
```

### Test 4: Concurrent Agent Interactions
```python
def test_concurrent_agent_deadlock():
    """P0: Multiple agents should not deadlock on shared room resources."""
    room = DeadbandRoom(room_id="edge_concurrent_test")
    
    def agent_worker(agent_id, results):
        try:
            room.register_agent(agent_id)
            room.execute_p0_step(negative_space={"conflict": True})
            tile_id = room.submit_tile(agent_id, {"agent": agent_id})
            results[agent_id] = tile_id
        except Exception as e:
            results[agent_id] = str(e)
    
    # Simulate 5 agents interacting concurrently
    import threading
    results = {}
    threads = []
    
    for i in range(5):
        agent_id = f"concurrent_agent_{i}"
        thread = threading.Thread(target=agent_worker, args=(agent_id, results))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join(timeout=5.0)  # 5 second timeout to detect deadlocks
    
    # Check all agents completed
    assert len(results) == 5
    # At least some agents should have succeeded
    successful = sum(1 for v in results.values() if isinstance(v, str) and v.startswith("tile_"))
    assert successful > 0, "All agents failed in concurrent scenario"
    
    # No threads should be stuck
    alive_threads = [t for t in threads if t.is_alive()]
    assert len(alive_threads) == 0, f"{len(alive_threads)} threads deadlocked"
```

### Test 5: Malformed Doctrine Configuration
```python
def test_malformed_doctrine_validation():
    """P0: Room should reject invalid deadband protocol definitions."""
    # Test missing P0 definition
    with pytest.raises(DoctrineValidationError) as exc_info:
        room = DeadbandRoom(
            room_id="edge_doctrine_test",
            doctrine_config={
                "P1": {"channels": ["safe"]},
                "P2": {"optimize": True}
            }
        )
    assert "Missing P0 definition" in str(exc_info.value)
    
    # Test circular dependency
    with pytest.raises(DoctrineValidationError) as exc_info:
        room = DeadbandRoom(
            room_id="edge_circular_test",
            doctrine_config={
                "P0": {"requires": ["P2"]},
                "P1": {"requires": ["P0"]},
                "P2": {"requires": ["P1"]}  # Circular: P0→P2→P1→P0
            }
        )
    assert "Circular dependency" in str(exc_info.value)
    
    # Test invalid phase name
    with pytest.raises(DoctrineValidationError) as exc_info:
        room = DeadbandRoom(
            room_id="edge_phase_test",
            doctrine_config={
                "P0": {"avoid": ["rocks"]},
                "PX": {"invalid": True}  # Not a valid phase
            }
        )
    assert "Invalid phase name" in str(exc_info.value)
```
