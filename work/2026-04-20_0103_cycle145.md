### Test 1: Empty Action Space
**Scenario:** Agent attempts to operate when action space is empty (no possible moves)
```python
def test_empty_action_space(self):
    """Test DeadbandRoom behavior when action space contains no valid actions."""
    room = DeadbandRoom(action_space=[], state_dim=3)
    agent_state = np.array([1.0, 2.0, 3.0])
    
    # Should raise specific error when no actions available
    with self.assertRaises(DeadbandProtocolError) as context:
        room.step(agent_state, action_idx=0)
    
    self.assertIn("No valid actions", str(context.exception))
```

### Test 2: Boundary State Values
**Scenario:** Agent state contains extreme values (NaN, Inf, very large numbers)
```python
def test_boundary_state_values(self):
    """Test handling of NaN, Inf, and extreme numerical values in agent state."""
    room = DeadbandRoom(action_space=[0, 1, 2], state_dim=3)
    
    test_states = [
        np.array([np.nan, 1.0, 2.0]),
        np.array([np.inf, 1.0, 2.0]),
        np.array([-np.inf, 1.0, 2.0]),
        np.array([1e308, 1.0, 2.0]),  # Near float max
    ]
    
    for state in test_states:
        with self.subTest(state=state):
            # Should either sanitize or reject gracefully
            result = room.step(state, action_idx=0)
            self.assertNotIn(np.nan, result['next_state'])
            self.assertNotIn(np.inf, result['next_state'])
```

### Test 3: Concurrent Protocol Violations
**Scenario:** Agent violates multiple protocol levels simultaneously
```python
def test_concurrent_protocol_violations(self):
    """Test when agent violates P0 and attempts P2 optimization in same step."""
    room = DeadbandRoom(
        action_space=['move_left', 'move_right', 'optimize'],
        state_dim=2,
        p0_constraints={'move_left': lambda s: s[0] > 0}  # Can't move left if x > 0
    )
    
    # Start in state where move_left violates P0
    agent_state = np.array([1.0, 0.0])
    
    # Try to optimize (P2) while in P0 violation state
    with self.assertRaises(DeadbandProtocolError) as context:
        room.step(agent_state, action_idx=2)  # 'optimize' action
    
    error_msg = str(context.exception)
    # Should mention both violations
    self.assertIn("P0 constraint", error_msg)
    self.assertIn("cannot skip to P2", error_msg)
```

### Test 4: Dynamic Constraint Changes
**Scenario:** Constraints change mid-episode (simulating environment shift)
```python
def test_dynamic_constraint_changes(self):
    """Test room adaptation when P0 constraints change during episode."""
    room = DeadbandRoom(
        action_space=['north', 'south', 'east', 'west'],
        state_dim=2
    )
    
    # Initial state with no constraints
    agent_state = np.array([0.0, 0.0])
    result1 = room.step(agent_state, action_idx=0)  # 'north'
    self.assertTrue(result1['valid'])
    
    # Dynamically add constraint: can't go north from certain states
    room.update_p0_constraint(
        'north',
        lambda s: s[1] < 5.0  # Can't go north if y < 5.0
    )
    
    # Now same action should violate P0
    with self.assertRaises(DeadbandProtocolError):
        room.step(agent_state, action_idx=0)
```

### Test 5: Reward Shaping Edge Cases
**Scenario:** Agent finds degenerate but technically valid P1 channel
```python
def test_degenerate_p1_channel(self):
    """Test when P1 safe channel exists but provides minimal/no reward."""
    room = DeadbandRoom(
        action_space=list(range(10)),
        state_dim=3,
        reward_fn=lambda s, a, ns: 0.001 if a == 5 else -1.0  # Tiny reward for action 5
    )
    
    agent_state = np.array([0.0, 0.0, 0.0])
    
    # Test all actions to find the degenerate channel
    valid_actions = []
    for action_idx in range(10):
        try:
            result = room.step(agent_state, action_idx)
            if result['valid']:
                valid_actions.append(action_idx)
        except DeadbandProtocolError:
            continue
    
    # Should identify action 5 as the only P1 safe channel
    self.assertEqual(valid_actions, [5])
    
    # Verify reward is minimal but positive
    result = room.step(agent_state, action_idx=5)
    self.assertAlmostEqual(result['reward'], 0.001, places=5)
```
