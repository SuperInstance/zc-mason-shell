# File: plato-torch/tests/test_deadband_room_edge.py
import pytest
from unittest.mock import Mock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import Tile

class MockAgent:
    def __init__(self, agent_id="test_agent"):
        self.agent_id = agent_id
        self.state = {}

def test_protocol_violation_skip_p0():
    """P0: Attempt to skip P0 and call P1 directly should fail."""
    room = DeadbandRoom(preset_id="deadband_protocol")
    agent = MockAgent()
    
    # Attempt P1 without P0
    tile = room.step(agent, {"step": "P1", "action": "move_forward"})
    assert tile is not None
    assert tile.data.get("error") == "protocol_violation"
    assert "P0 required first" in tile.data.get("message", "")

def test_p2_invalid_channel():
    """P2: Providing invalid safe channel to P2 should return error."""
    room = DeadbandRoom(preset_id="deadband_protocol")
    agent = MockAgent()
    
    # Complete P0 and P1 to get valid state
    room.step(agent, {"step": "P0", "action": "scan"})
    room.step(agent, {"step": "P1", "action": "explore"})
    
    # P2 with invalid channel
    tile = room.step(agent, {"step": "P2", "channel": "invalid_channel", "action": "optimize"})
    assert tile.data.get("error") == "invalid_channel"
    assert "not in safe channels" in tile.data.get("message", "")

def test_empty_action_p0():
    """P0: Empty or null action should be handled gracefully."""
    room = DeadbandRoom(preset_id="deadband_protocol")
    agent = MockAgent()
    
    # Empty action dict
    tile = room.step(agent, {"step": "P0", "action": {}})
    assert tile.data.get("status") == "rejected"
    assert "action cannot be empty" in tile.data.get("reason", "")
    
    # None action
    tile = room.step(agent, {"step": "P0", "action": None})
    assert tile.data.get("status") == "rejected"

def test_state_persistence_across_steps():
    """State should persist for an agent across P0→P1→P2 sequence."""
    room = DeadbandRoom(preset_id="deadband_protocol")
    agent = MockAgent(agent_id="persistent_agent")
    
    # P0
    tile_p0 = room.step(agent, {"step": "P0", "action": "scan", "target": "obstacle"})
    p0_state = tile_p0.data.get("state", {})
    assert "negative_space" in p0_state
    
    # P1 should have access to P0 state
    tile_p1 = room.step(agent, {"step": "P1", "action": "explore"})
    p1_state = tile_p1.data.get("state", {})
    assert p1_state.get("previous_step") == "P0"
    assert "safe_channels" in p1_state
    
    # P2 should have accumulated state
    tile_p2 = room.step(agent, {"step": "P2", "channel": p1_state["safe_channels"][0], "action": "optimize"})
    p2_state = tile_p2.data.get("state", {})
    assert "negative_space" in p2_state  # From P0
    assert "safe_channels" in p2_state   # From P1
    assert "optimized_path" in p2_state  # From P2

def test_concurrent_agents_isolation():
    """Two agents interacting concurrently should have isolated states."""
    room = DeadbandRoom(preset_id="deadband_protocol")
    agent1 = MockAgent(agent_id="agent_1")
    agent2 = MockAgent(agent_id="agent_2")
    
    # Agent1 completes P0
    tile1 = room.step(agent1, {"step": "P0", "action": "scan"})
    state1 = room.get_agent_state(agent1.agent_id)
    assert state1 is not None
    assert "negative_space" in state1
    
    # Agent2 should not see Agent1's state
    state2 = room.get_agent_state(agent2.agent_id)
    assert state2 is None or "negative_space" not in state2
    
    # Agent2 does its own P0
    tile2 = room.step(agent2, {"step": "P0", "action": "scan"})
    assert tile2.data.get("state", {}).get("negative_space") != tile1.data.get("state", {}).get("negative_space")
