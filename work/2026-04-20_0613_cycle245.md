```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions processed
    assert all(r["status"] == "accepted" for r in results)
    assert len(room.training_tiles) == len(agent_ids)
    
    # Verify no tile corruption
    stored_agents = {t["agent_id"] for t in room.training_tiles}
    assert stored_agents == set(agent_ids)
```

### Test 2: Malformed Tile Payloads

```python
def test_malformed_tile_rejection(self):
    """Test rejection of tiles with invalid structure."""
    room = DeadbandRoom()
    
    test_cases = [
        {"agent_id": "test_agent"},  # Missing required fields
        {"phase": "P0", "content": "test"},  # Missing agent_id
        {"agent_id": 123, "phase": "P0", "content": "test"},  # Wrong type
        None,  # Null payload
        "not_a_dict",  # String instead of dict
        {"agent_id": "test", "phase": "P5", "content": "test"},  # Invalid phase
    ]
    
    for i, tile in enumerate(test_cases):
        try:
            result = room.submit_tile(tile)
            # Should not reach here for invalid tiles
            assert False, f"Test case {i} should have raised exception"
        except (ValueError, TypeError, KeyError) as e:
            # Expected rejection
            assert "tile" in str(e).lower() or "invalid" in str(e).lower()
            assert len(room.training_tiles) == 0  # No tiles accepted
```

### Test 3: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Test agent attempting P2 without completing P1, then recovering."""
    room = DeadbandRoom()
    agent_id = "recovery_agent"
    
    # Attempt P2 directly (violation)
    tile_p2 = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Trying to optimize without safe channel",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(tile_p2)
    assert result["status"] == "rejected"
    assert "P1" in result.get("message", "")  # Should mention missing P1
    
    # Correctly do P0
    tile_p0 = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Mapping negative space",
        "timestamp": time.time()
    }
    room.submit_tile(tile_p0)
    
    # Correctly do P1
    tile_p1 = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Finding safe channels",
        "timestamp": time.time()
    }
    room.submit_tile(tile_p1)
    
    # Now P2 should work
    tile_p2_correct = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Optimizing within channels",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_p2_correct)
    assert result["status"] == "accepted"
    
    # Verify protocol sequence preserved
    agent_tiles = [t for t in room.training_tiles if t["agent_id"] == agent_id]
    phases = [t["phase"] for t in agent_tiles]
    assert phases == ["P0", "P1", "P2"]
```

### Test 4: Resource Exhaustion Handling

```python
def test_resource_limits(self):
    """Test behavior when hitting memory/disk limits."""
    room = DeadbandRoom()
    
    # Configure small limit for testing
    room.max_tiles = 3
    room.max_tile_size = 100  # bytes
    
    # Fill to capacity with valid tiles
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P0",
            "content": "X" * 50,  # 50 bytes
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # Attempt to exceed tile count limit
    tile_extra = {
        "agent_id": "agent_overflow",
        "phase": "P0",
        "content": "Should be rejected",
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_extra)
    assert result["status"] == "rejected"
    assert "limit" in result.get("message", "").lower()
    
    # Attempt to exceed size limit
    room.max_tiles = 10  # Reset count limit
    tile_oversized = {
        "agent_id": "agent_big",
        "phase": "P0",
        "content": "X" * 150,  # Exceeds 100 byte limit
        "timestamp": time.time()
    }
    result = room.submit_tile(tile_oversized)
    assert result["status"] == "rejected"
    assert "size" in result.get("message", "").lower()
```

### Test 5: Network Partition Simulation
```python
def test_partial_connectivity(self):
    """Test room behavior during simulated network issues."""
    room = DeadbandRoom()
    
    # Simulate intermittent connectivity
    original_submit = room._submit_to_plato_server
    
    call_count = 0
    def flaky_submit(tile):
        nonlocal call_count
        call_count += 1
        
        # Fail every other attempt
        if call_count % 2 == 0:
            raise ConnectionError("Network partition simulated")
        
        # Succeed on odd attempts
        return original_submit(tile)
    
    room._submit_to_plato_server = flaky_submit
    
    # Submit series of tiles
    results = []
    for i in range(5):
        tile = {
            "agent_id": f"flaky_agent",
            "phase": "P0",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except ConnectionError:
            results.append({"status": "failed", "attempt": i})
    
    # Verify some succeeded, some failed
    successes = [r for r in results if r.get("status") == "accepted"]
    failures = [r for r in results if r.get("status") == "failed"]
    
    assert len(successes) > 0
    assert len(failures) > 0
    assert len(successes) + len(failures) == 5
    
    # Verify room state consistent despite failures
    assert len(room.training_tiles) == len(successes)
    
    # Restore original method
    room._submit_to_plato_server = original_submit
```
