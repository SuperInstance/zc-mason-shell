```python
# work/explore_codebase.py
import os
import re

def find_python_files():
    python_files = []
    for root, dirs, files in os.walk('.'):
        # Skip hidden directories and work directory
        dirs[:] = [d for d in dirs if not d.startswith('.') and d != 'work']
        for file in files:
            if file.endswith('.py'):
                python_files.append(os.path.join(root, file))
    return python_files

def search_for_deadband(files):
    deadband_files = []
    for file in files:
        try:
            with open(file, 'r') as f:
                content = f.read()
                if 'deadband' in content.lower() or 'Deadband' in content:
                    deadband_files.append(file)
        except:
            continue
    return deadband_files

if __name__ == "__main__":
    print("Exploring codebase...")
    python_files = find_python_files()
    print(f"Found {len(python_files)} Python files")
    
    deadband_files = search_for_deadband(python_files)
    print(f"\nFiles containing 'deadband': {len(deadband_files)}")
    for file in deadband_files:
        print(f"  - {file}")
```
