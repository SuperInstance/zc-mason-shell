From `test_deadband_room.py`, I observed the following test patterns:

2. **Test P0 negative space mapping** – Checks that P0 logic rejects invalid actions.
3. **Test P1 safe channel discovery** – Validates that safe actions are identified.

## Identified Edge-Case Gaps
Based on the Deadband Protocol doctrine and room implementation, the following edge cases are not covered by existing tests:
1. **Concurrent tile processing** – Multiple agents submitting tiles simultaneously.
2. **Malformed tile data** – Invalid JSON, missing fields, or incorrect types.
3. **P0/P1 boundary conditions** – Actions that are on the threshold of "negative space" vs "safe channel".
4. **Resource exhaustion** – Memory/CPU spikes during large tile batches.
5. **Network partition simulation** – Tile server unreachable after P1/P2 computation.

### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Edge-case: Multiple agents submit tiles simultaneously."""
    import threading
    results = []
    def submit_tile(agent_id):
        tile = {"agent": agent_id, "phase": "P1", "action": "explore"}
        result = self.room.submit_tile(tile)
        results.append((agent_id, result))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all tiles were processed without corruption
    self.assertEqual(len(results), 5)
    for agent_id, result in results:
        self.assertIn("accepted", result)
        self.assertTrue(result["accepted"] or "error" in result)
```

```python
def test_p0_p1_boundary_condition(self):
    """Edge-case: Action that sits exactly on threshold between negative space and safe channel."""
    # Mock an action with risk_score = 0.5 (exactly at threshold)
    boundary_action = {
        "id": "boundary_action",
        "risk_score": 0.5,  # Threshold defined in room config
        "reward_potential": 0.5
    }
    
    # Should be rejected by P0 if >= 0.5, accepted if < 0.5
    p0_result = self.room._apply_p0(boundary_action)
    if self.room.p0_threshold <= 0.5:
        self.assertFalse(p0_result["safe"], "Action at threshold should be rejected by P0")
    else:
        self.assertTrue(p0_result["safe"], "Action below threshold should pass P0")
    
    # If passes P0, verify P1 evaluation
    if p0_result["safe"]:
        p1_result = self.room._apply_p1([boundary_action])
        self.assertIn(boundary_action["id"], p1_result["safe_channels"])
```

### Test 4: Resource Exhaustion During Batch Processing
```python
def test_resource_exhaustion_batch_processing(self):
    """Edge-case: Processing large batch of tiles triggers memory/CPU limits."""
    # Generate 10,000 simple tiles
    large_batch = []
    for i in range(10000):
        large_batch.append({
            "agent": f"bot_{i}",
            "phase": "P1",
            "action": f"action_{i % 100}",
            "timestamp": i
        })
    
    # Process batch with memory monitoring
    import psutil
    import os
    process = psutil.Process(os.getpid())
    initial_memory = process.memory_info().rss
    
    results = []
    for tile in large_batch:
        try:
            result = self.room.submit_tile(tile)
            results.append(result)
        except MemoryError:
            self.fail("MemoryError raised during large batch processing")
    
    final_memory = process.memory_info().rss
    memory_increase = final_memory - initial_memory
    
    # Assert memory increase is within reasonable bounds (e.g., < 100MB)
    self.assertLess(memory_increase, 100 * 1024 * 1024, 
                   f"Memory increase {memory_increase/1024/1024:.2f}MB exceeds 100MB limit")
    
    self.assertEqual(len(results), 10000, "All tiles should be processed")
```

### Test 5: Network Partition After Computation
```python
def test_network_partition_after_computation(self):
    """Edge-case: Tile server becomes unreachable after P1/P2 computation completes."""
    # Mock tile server that fails after first successful call
    call_count = 0
    original_send = self.room.tile_server.send_tile
    
    def failing_send(tile):
        nonlocal call_count
        call_count += 1
        if call_count > 1:
            raise ConnectionError("Tile server unreachable")
        return original_send(tile)
    
    self.room.tile_server.send_tile = failing_send
    
    # First tile should succeed
    tile1 = {"agent": "agent1", "phase": "P2", "action": "optimize"}
    result1 = self.room.submit_tile(tile1)
    self.assertTrue(result1["accepted"])
    
    # Second tile should fail with graceful handling
    tile2 = {"agent": "agent2", "phase": "P2", "action": "optimize"}
    result2 = self.room.submit_tile(tile2)
    
    # Should not crash; should return error in result
    self.assertFalse(result2["accepted"])
    self.assertIn("error", result2)
    self.assertIn("ConnectionError", result2["error"])
    
    # Verify room state is still consistent
    self.assertTrue(self.room.is_operational)
```
