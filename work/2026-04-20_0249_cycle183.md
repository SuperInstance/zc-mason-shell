```python
def test_concurrent_tile_submission(self):

    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:

            "protocol_phase": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions succeeded
    assert all(r["status"] == "accepted" for r in results), "Concurrent submissions failed"
    
    # Verify tile count matches
    assert len(room.get_tiles()) == len(agent_ids), "Tile count mismatch after concurrent submissions"
    
    # Verify no duplicate or corrupted tiles
    tile_agents = [t["agent_id"] for t in room.get_tiles()]
    assert len(set(tile_agents)) == len(agent_ids), "Duplicate or missing agent tiles"
```

```python
def test_protocol_violation_recovery(self):
    """Agent attempting to skip P1 should be rejected and guided back to correct phase."""
    room = DeadbandRoom()
    agent_id = "violating_agent"
    
    # Agent tries to submit P2 tile without completing P1 (violation)
    violation_tile = {
        "agent_id": agent_id,
        "cycle": 183,
        "protocol_phase": "P2",  # Skipping P1!
        "content": "Trying to optimize without safe channel",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(violation_tile)
    
    # Should be rejected with guidance
    assert result["status"] == "rejected", "Protocol violation should be rejected"
    assert "P1" in result.get("message", ""), "Should guide agent to P1"
    assert "violation" in result.get("message", "").lower(), "Should indicate protocol violation"
    
    # Agent corrects and submits P1 tile
    correction_tile = {
        "agent_id": agent_id,
        "cycle": 184,
        "protocol_phase": "P1",
        "content": "Finding safe channels as instructed",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(correction_tile)
    assert result["status"] == "accepted", "Corrected submission should be accepted"
    
    # Verify room state reflects the correction
    agent_tiles = [t for t in room.get_tiles() if t["agent_id"] == agent_id]
    assert len(agent_tiles) == 1, "Should have one accepted tile after correction"
    assert agent_tiles[0]["protocol_phase"] == "P1", "Accepted tile should be P1 phase"
```

```python
def test_malformed_tile_data(self):
    """Room should handle corrupted, partial, or malicious tile structures."""
    room = DeadbandRoom()
    
    test_cases = [
        # (tile_data, expected_handling)
        (None, "reject"),  # Null tile
        ({}, "reject"),  # Empty dict
        ({"agent_id": "test"}, "reject"),  # Missing required fields
        ({"agent_id": "test", "cycle": "not_a_number"}, "reject"),  # Wrong type
        ({"agent_id": "test", "cycle": 183, "protocol_phase": "P99"}, "reject"),  # Invalid phase
        ({"agent_id": "x" * 10000, "cycle": 183, "protocol_phase": "P1"}, "reject"),  # Too large
        ({"agent_id": "<script>alert()</script>", "cycle": 183, "protocol_phase": "P1"}, "sanitize_or_reject"),  # XSS attempt
    ]
    
    for tile_data, expected in test_cases:
        try:
            result = room.submit_tile(tile_data)
            
            if expected == "reject":
                assert result["status"] == "rejected", \
                       f"Malformed tile {tile_data} should be rejected"
                assert "invalid" in result.get("message", "").lower() or \
                       "malformed" in result.get("message", "").lower(), \
                       f"Should indicate why tile {tile_data} was rejected"
            
            elif expected == "sanitize_or_reject":
                # Either rejection or acceptance with sanitization
                if result["status"] == "accepted":
                    # Verify sanitization occurred
                    stored_tile = next((t for t in room.get_tiles() 
                                      if t.get("agent_id") == tile_data["agent_id"]), None)
                    if stored_tile:
                        assert "<script>" not in stored_tile["agent_id"], \
                               "XSS content should be sanitized"
        except Exception as e:
            # Room might raise exception for extreme cases
            assert "malformed" in str(e).lower() or "invalid" in str(e).lower(), \
                   f"Exception for {tile_data} should indicate problem: {e}"
    
    # Verify room integrity maintained
    assert room.get_state()["status"] == "active", "Room should survive malformed data"
    valid_tiles = [t for t in room.get_tiles() 
                   if all(key in t for key in ["agent_id", "cycle", "protocol_phase"])]
    assert len(valid_tiles) == len(room.get_tiles()), "All stored tiles should be valid"
```
