### Test 2: Tile Flood Attack Mitigation

```python
def test_tile_flood_mitigation():
    """Test room's rate limiting against rapid tile submission."""
    room = DeadbandRoom(capacity=5)
    agent_id = "flood_agent"
    
    # Submit tiles at maximum rate
    submission_times = []
    for i in range(20):  # Attempt to flood
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Flood tile {i}",
            "timestamp": time.time()
        }
        try:
            room.submit_tile(agent_id, tile)
            submission_times.append(time.time())
        except RateLimitExceededError:
            break  # Expected after certain threshold
    
    # Verify rate limiting kicked in
    assert len(submission_times) < 20, "Rate limiting failed"
    
    # Check timing between submissions increased
    if len(submission_times) > 2:
        intervals = [submission_times[i+1] - submission_times[i] 
                    for i in range(len(submission_times)-1)]
        # Later intervals should be larger (exponential backoff)
        assert intervals[-1] > intervals[0] * 1.5
```

```python
def test_concurrent_state_integrity():
    """Test room handles concurrent tile submissions without state corruption."""
    import threading
    
    room = DeadbandRoom(capacity=100)
    agent_ids = [f"concurrent_agent_{i}" for i in range(10)]
    results = {}
    lock = threading.Lock()
    
    def submit_tiles_for_agent(agent_id):
        """Submit 5 tiles from a single agent."""
        agent_tiles = []
        for i in range(5):
            tile = {
                "agent_id": agent_id,
                "phase": "P0",
                "content": f"Tile {i} from {agent_id}",
                "timestamp": time.time() + random.random() * 0.01
            }
            try:
                success = room.submit_tile(agent_id, tile)
                if success:
                    agent_tiles.append(tile)
            except Exception as e:
                agent_tiles.append(f"error: {e}")
        
        with lock:
            results[agent_id] = agent_tiles
    
    # Launch concurrent submissions
    threads = []
    for agent_id in agent_ids:
        t = threading.Thread(target=submit_tiles_for_agent, args=(agent_id,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all agents' tiles were processed
    for agent_id in agent_ids:
        agent_tiles = results[agent_id]
        assert len(agent_tiles) > 0, f"No tiles processed for {agent_id}"
        
        # Check for state corruption
        if isinstance(agent_tiles[0], dict):
            room_state = room.get_agent_state(agent_id)
            assert room_state["tile_count"] == len([t for t in agent_tiles 
                                                   if isinstance(t, dict)])
    
    # Verify total room tile count matches sum of individual agents
    total_tiles = sum(len([t for t in tiles if isinstance(t, dict)]) 
                     for tiles in results.values())
    assert room.get_room_stats()["total_tiles"] == total_tiles
```

```python
def test_malformed_tile_handling():
    """Test room gracefully handles various malformed tile structures."""
    room = DeadbandRoom(capacity=10)
    agent_id = "malformed_test_agent"
    
    malformed_tiles = [
        # Missing required fields
        {"agent_id": agent_id},  # Missing phase, content
        {"phase": "P0", "content": "test"},  # Missing agent_id
        
        # Invalid data types
        {"agent_id": 123, "phase": "P0", "content": "test"},  # agent_id not string
        {"agent_id": agent_id, "phase": 999, "content": "test"},  # phase not string
        
        # Invalid phase value
        {"agent_id": agent_id, "phase": "P99", "content": "invalid phase"},
        
        # Extremely large content
        {"agent_id": agent_id, "phase": "P0", "content": "X" * 1000000},
        
        # Negative timestamp
        {"agent_id": agent_id, "phase": "P0", "content": "test", "timestamp": -1},
        
        # Future timestamp (far future)
        {"agent_id": agent_id, "phase": "P0", "content": "test", 
         "timestamp": time.time() + 86400 * 365},  # 1 year in future
    ]
    
    rejection_count = 0
    for i, tile in enumerate(malformed_tiles):
        try:
            success = room.submit_tile(
                tile.get("agent_id", agent_id), 
                tile
            )
            if not success:
                rejection_count += 1
        except (ValidationError, ValueError):
            rejection_count += 1
    
    # All malformed tiles should be rejected
    assert rejection_count == len(malformed_tiles), \
        f"Expected all {len(malformed_tiles)} malformed tiles rejected, " \
        f"but {len(malformed_tiles) - rejection_count} were accepted"
    
    # Agent should not be penalized for malformed submissions
    agent_state = room.get_agent_state(agent_id)
    assert agent_state.get("violations", 0) == 0, \
        "Agent penalized for malformed tiles (should only reject, not penalize)"
```
