### Test 3: Concurrent Tile Submissions
```python
def test_concurrent_tile_submissions(self):
    """Room should maintain consistency with concurrent submissions."""
    room = DeadbandRoom(room_id="edge-test-3")
    
    def submit_tile_async(tile_data):
        return room.submit_tile(tile_data)
    
    # Create multiple tiles from different agents
    tiles = []
    for i in range(5):
        tile = {
            "phase": "P0",
            "content": {"negative_space": [f"rock-{i}"]},
            "agent_id": f"agent-{i}"
        }
        tiles.append(tile)
    
    # Simulate concurrent submission
    results = []
    for tile in tiles:
        result = submit_tile_async(tile)
        results.append(result)
    
    # All should be accepted
    accepted_count = sum(1 for r in results if r["accepted"])
    self.assertEqual(accepted_count, 5)
    
    # Room state should reflect all submissions
    self.assertEqual(len(room.get_tiles()), 5)
```

### Test 4: Corrupted State Recovery
```python
def test_corrupted_state_recovery(self):
    """Room should recover from corrupted internal state."""
    room = DeadbandRoom(room_id="edge-test-4")
    
    # First, establish normal state
    valid_tile = {
        "phase": "P0",
        "content": {"negative_space": ["test-rock"]},
        "agent_id": "test-agent"
    }
    room.submit_tile(valid_tile)
    
    # Simulate state corruption (inject invalid data)
    room._tiles.append({"corrupted": True})  # Internal attribute access
    
    # Room should detect and handle corrupted state on next operation
    recovery_tile = {
        "phase": "P1",
        "content": {"channel": "recovery-channel"},
        "agent_id": "test-agent"
    }
    
    # Should either recover or raise appropriate error
    try:
        result = room.submit_tile(recovery_tile)
        # If recovery succeeds, validate state
        self.assertTrue(room.validate_state())
    except StateCorruptionError as e:
        # Expected error for unrecoverable corruption
        self.assertIn("corrupted", str(e).lower())
```

### Test 5: Resource Exhaustion Handling
```python
def test_resource_exhaustion_handling(self):
    """Room should handle resource limits gracefully."""
    room = DeadbandRoom(room_id="edge-test-5", max_tiles=3)
    
    # Fill room to capacity
    for i in range(3):
        tile = {
            "phase": "P0",
            "content": {"negative_space": [f"rock-{i}"]},
            "agent_id": f"agent-{i}"
        }
        result = room.submit_tile(tile)
        self.assertTrue(result["accepted"])
    
    # Attempt to exceed capacity
    excess_tile = {
        "phase": "P0",
        "content": {"negative_space": ["excess-rock"]},
        "agent_id": "excess-agent"
    }
    
    with self.assertRaises(ResourceExhaustedError) as context:
        room.submit_tile(excess_tile)
    
    self.assertIn("capacity", str(context.exception).lower())
    
    # Verify existing tiles remain intact
    self.assertEqual(len(room.get_tiles()), 3)
```
