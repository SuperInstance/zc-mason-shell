### Test 1: Empty Input Space Handling
```rust
#[test]
fn test_deadband_empty_input_space() {
    let mut room = DeadbandRoom::new();
    let negative_space = vec![]; // No constraints
    let available_channels = vec![]; // No safe channels
    
    // P0: Map negative space (empty)
    room.map_negative_space(&negative_space);
    assert_eq!(room.get_mapped_constraints().len(), 0);
    
    // P1: Find safe channels (empty)
    let safe_channels = room.find_safe_channels(&available_channels);
    assert!(safe_channels.is_empty(), "Should return empty when no channels available");
    
    // P2: Should not proceed without safe channels
    let result = room.optimize_within_channels(&safe_channels);
    assert!(result.is_err(), "Optimization should fail without safe channels");
}
```

### Test 2: Boundary Tolerance Conditions
```rust
#[test]
fn test_deadband_boundary_tolerance() {
    let mut room = DeadbandRoom::with_tolerance(0.001); // Tight tolerance
    let negative_space = vec![Constraint::new("rock", 0.0, 1.0)];
    
    // Channel exactly at tolerance boundary
    let boundary_channel = Channel::new(0.999, 1.001); // Touches negative space at tolerance limit
    
    room.map_negative_space(&negative_space);
    let safe_channels = room.find_safe_channels(&vec![boundary_channel]);
    
    // Should accept boundary channel if within tolerance
    if room.get_tolerance() >= 0.001 {
        assert_eq!(safe_channels.len(), 1, "Boundary channel should be accepted within tolerance");
    } else {
        assert!(safe_channels.is_empty(), "Boundary channel should be rejected outside tolerance");
    }
}
```

### Test 3: Concurrent Tile Submission Collision
```rust
#[test]
fn test_concurrent_tile_collision() {
    use std::sync::{Arc, Mutex};
    use std::thread;
    
    let room = Arc::new(Mutex::new(DeadbandRoom::new()));
    let mut handles = vec![];
    let successful_submissions = Arc::new(Mutex::new(0));
    
    // Simulate 5 agents submitting tiles simultaneously
    for i in 0..5 {
        let room_clone = Arc::clone(&room);
        let success_clone = Arc::clone(&successful_submissions);
        
        handles.push(thread::spawn(move || {
            let tile = Tile::new(format!("agent_{}_work", i), "test_data".to_string());
            let mut room_lock = room_clone.lock().unwrap();
            
            match room_lock.submit_tile(tile) {
                Ok(_) => {
                    let mut count = success_clone.lock().unwrap();
                    *count += 1;
                }
                Err(e) => {
                    // Expected: some submissions may fail due to collision
                    assert!(e.to_string().contains("concurrent") || e.to_string().contains("locked"));
                }
            }
        }));
    }
    
    for handle in handles {
        handle.join().unwrap();
    }
    
    let final_count = successful_submissions.lock().unwrap();
    assert!(*final_count > 0, "At least one submission should succeed");
    assert!(*final_count <= 5, "Cannot exceed total submission attempts");
}
```

### Test 4: Protocol State Recovery After Violation
```rust
#[test]
fn test_protocol_recovery_after_violation() {
    let mut room = DeadbandRoom::new();
    
    // Force a protocol violation (skip to P2)
    let channels = vec![Channel::new(0.5, 1.5)];
    let violation_result = room.optimize_within_channels(&channels);
    assert!(violation_result.is_err(), "Should reject optimization without P0/P1");
    
    // Check error type
    let err = violation_result.unwrap_err();
    assert!(err.to_string().contains("P0") || err.to_string().contains("protocol"));
    
    // Reset and follow proper protocol
    room.reset_protocol_state();
    
    let negative_space = vec![Constraint::new("rock", 0.0, 1.0)];
    room.map_negative_space(&negative_space);
    
    let safe_channels = room.find_safe_channels(&channels);
    assert_eq!(safe_channels.len(), 1, "Should find safe channel after reset");
    
    let optimization_result = room.optimize_within_channels(&safe_channels);
    assert!(optimization_result.is_ok(), "Should succeed after proper protocol sequence");
}
```

### Test 5: Resource Exhaustion During Optimization
```rust
#[test]
#[should_panic(expected = "resource")] // or specific error message
fn test_optimization_resource_exhaustion() {
    let mut room = DeadbandRoom::new();
    
    // Create pathological case that could cause infinite loop or memory exhaustion
    let mut infinite_channels = Vec::new();
    for i in 0..1000000 {
        infinite_channels.push(Channel::new(i as f64, (i + 1) as f64));
    }
    
    // Map minimal negative space
    room.map_negative_space(&vec![Constraint::new("test", 0.0, 0.1)]);
    
    // This should trigger resource limits
    let safe_channels = room.find_safe_channels(&infinite_channels);
    
    // If we get here, optimization should fail
    let _ = room.optimize_within_channels(&safe_channels).expect_err("Should fail due to resource limits");
    
    // Alternative: set resource limit explicitly
    room.set_resource_limit(1000); // Only process 1000 channels max
    let limited_result = room.optimize_within_channels(&safe_channels);
    assert!(limited_result.is_err());
}
```

## Validation Approach
Each test follows P0/P1/P2 doctrine:
1. First establish what shouldn't happen (P0)
2. Then identify valid paths (P1)
3. Only then attempt optimization (P2)

Tests avoid absolute assertions about system behavior, instead checking for expected patterns given specific inputs.
