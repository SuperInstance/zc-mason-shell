"""
Edge-case tests for DeadbandRoom preset.
Focuses on boundary conditions, error paths, and stress scenarios.
"""

import pytest
import asyncio
from unittest.mock import Mock, patch, AsyncMock
import json
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import TrainingTile


class TestDeadbandRoomEdgeCases:
    """Edge-case test suite for DeadbandRoom."""
    
    @pytest.mark.asyncio
    async def test_concurrent_agent_collision(self):
        """Test P0: Two agents attempting simultaneous P1 channel discovery.
        
        Edge case: Concurrent writes to shared channel map could cause
        race conditions or inconsistent state.
        """
        room = DeadbandRoom(room_id="edge-concurrent")
        
        # Simulate two agents discovering channels simultaneously
        agent1_tile = TrainingTile(
            agent_id="agent1",
            protocol_phase="P1",
            content={"discovered_channels": ["channel_a"]},
            cycle=100
        )
        
        agent2_tile = TrainingTile(
            agent_id="agent2", 
            protocol_phase="P1",
            content={"discovered_channels": ["channel_b"]},
            cycle=100
        )
        
        # Submit tiles concurrently
        tasks = [
            asyncio.create_task(room.process_tile(agent1_tile)),
            asyncio.create_task(room.process_tile(agent2_tile))
        ]
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Verify no exceptions and both tiles processed
        assert not any(isinstance(r, Exception) for r in results)
        
        # Check that room state reflects both discoveries
        room_state = room.get_state()
        assert "channel_a" in room_state.get("known_channels", [])
        assert "channel_b" in room_state.get("known_channels", [])
    
    @pytest.mark.asyncio
    async def test_malformed_tile_content(self):
        """Test P0: Agent submits tile with invalid content structure.
        
        Edge case: Content missing required fields or with wrong data types
        should be handled gracefully without crashing the room.
        """
        room = DeadbandRoom(room_id="edge-malformed")
        
        # Tile missing required 'protocol_phase' field
        bad_tile = TrainingTile(
            agent_id="malformed_agent",
            content={"random_data": [1, 2, 3]},  # No protocol phase
            cycle=101
        )
        
        # Remove protocol_phase to simulate malformed data
        bad_tile.protocol_phase = None
        
        result = await room.process_tile(bad_tile)
        
        # Room should handle gracefully - either reject or default to P0
        assert result is not None
        # Result should indicate error or fallback to P0
        assert "error" in result or result.get("protocol_phase") == "P0"
    
    @pytest.mark.asyncio
    async def test_protocol_boundary_condition(self):
        """Test P1->P2 transition with minimal safe channel width.
        
        Edge case: When safe channel is at minimum viable width (epsilon),
        optimization attempts should be constrained or rejected.
        """
        room = DeadbandRoom(room_id="edge-boundary")
        
        # First establish a minimal safe channel
        p1_tile = TrainingTile(
            agent_id="boundary_agent",
            protocol_phase="P1",
            content={
                "discovered_channels": ["minimal_channel"],
                "channel_metrics": {"width": 0.001}  # Very narrow
            },
            cycle=102
        )
        
        p1_result = await room.process_tile(p1_tile)
        assert p1_result.get("status") == "accepted"
        
        # Attempt P2 optimization on minimal channel
        p2_tile = TrainingTile(
            agent_id="boundary_agent",
            protocol_phase="P2",
            content={
                "channel": "minimal_channel",
                "optimization": {"factor": 10.0}  # Aggressive optimization
            },
            cycle=103
        )
        
        p2_result = await room.process_tile(p2_tile)
        
        # Room should either reject or heavily constrain the optimization
        assert "constraint" in p2_result or p2_result.get("status") == "rejected"
    
    @pytest.mark.asyncio
    async def test_memory_exhaustion_recovery(self):
        """Test P0: Room recovers after hitting memory limits.
        
        Edge case: Simulate memory pressure by flooding with tiles,
        then verify room can recover and continue processing.
        """
        room = DeadbandRoom(room_id="edge-memory")
        
        # Flood room with tiles to simulate memory pressure
        flood_tiles = []
        for i in range(1000):  # Large number to simulate pressure

                agent_id=f"flood_agent_{i}",
                protocol_phase="P0",

                cycle=200 + i

            flood_tiles.append(tile)
        
        # Process all tiles (in reality would be batched or rate-limited)

            if result and result.get("status") == "accepted":
                processed += 1
        
        # Simulate memory cleanup
        room.cleanup_old_tiles(threshold_cycles=5)
        
        # Verify room can still process new tiles after cleanup
        recovery_tile = TrainingTile(
            agent_id="recovery_agent",
            protocol_phase="P1",
            content={"discovered_channels": ["recovery_channel"]},
            cycle=300
        )
        
        recovery_result = await room.process_tile(recovery_tile)
        assert recovery_result.get("status") == "accepted"
    
    @pytest.mark.asyncio
    async def test_corrupted_state_persistence(self):

            channels = room_state.get("known_channels", [])
            assert all(isinstance(c, str) for c in channels)

            test_tile = TrainingTile(
                agent_id="fresh_agent",
                protocol_phase="P0",
                content={"negative_space": ["new_hazard"]},
                cycle=400
            )
            
            result = await room.process_tile(test_tile)
            assert result.get("status") == "accepted"
