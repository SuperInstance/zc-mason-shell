### Test 3: Concurrent Submissions from Same Agent
```python
def test_concurrent_submission_race(room, agent):
    """P1: Simulate rapid consecutive submissions from same agent."""
    import threading
    
    results = []
    
    def submit_tile_thread(cycle_num):
        tile = {
            "cycle": cycle_num,
            "phase": "P1",
            "content": f"Safe channel analysis {cycle_num}",
            "agent_id": agent.id
        }
        results.append(room.submit_tile(tile))
    
    # Submit 3 tiles with same cycle number (simulating race condition)
    threads = []
    for i in range(3):
        t = threading.Thread(target=submit_tile_thread, args=(300,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Only one should be accepted, others rejected as duplicate
    accepted = [r for r in results if r["status"] == "accepted"]
    rejected = [r for r in results if r["status"] == "rejected"]
    
    assert len(accepted) == 1
    assert len(rejected) == 2
    for r in rejected:
        assert "duplicate" in r["reason"].lower() or "concurrent" in r["reason"].lower()
```

```python
def test_agent_state_persistence(room, agent):
    """P2: Verify agent phase persists through room soft reset."""
    # Complete P0 and P1 phases
    tile_p0 = {
        "cycle": 500,
        "phase": "P0",
        "content": "Negative space: don't hardcode API keys",
        "agent_id": agent.id
    }
    room.submit_tile(tile_p0)
    
    tile_p1 = {
        "cycle": 501,
        "phase": "P1",
        "content": "Safe channel: use environment variables",
        "agent_id": agent.id
    }
    room.submit_tile(tile_p1)
    
    # Agent should now be in P2 phase
    assert room.get_agent_phase(agent.id) == "P2"
    
    # Simulate room soft reset (preserves agent states)
    room.soft_reset()
    
    # Agent should remain in P2 phase after reset
    assert room.get_agent_phase(agent.id) == "P2"
    
    # P2 tile should be accepted
    tile_p2 = {
        "cycle": 502,
        "phase": "P2",
        "content": "Optimize: cache env vars",
        "agent_id": agent.id
    }
    result = room.submit_tile(tile_p2)
    assert result["status"] == "accepted"
```
