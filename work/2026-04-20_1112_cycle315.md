"""
Edge-case tests for DeadbandRoom.
Focus on P0 scenarios (what NOT to do) and boundary conditions.
"""

import pytest
import asyncio
from unittest.mock import AsyncMock, MagicMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import Tile


class TestDeadbandRoomEdgeCases:
    """Edge-case tests focusing on negative space and boundary conditions."""
    
    @pytest.mark.asyncio
    async def test_concurrent_join_leave_during_tile_processing(self):
        """P0: Race condition when agents join/leave while tiles are processing."""
        room = DeadbandRoom()
        
        # Create multiple agents
        agents = []
        for i in range(5):
            agent = AsyncMock()
            agent.id = f"agent_{i}"
            agent.send_state = AsyncMock()
            agents.append(agent)
        
        # Start concurrent operations
        join_tasks = [room.join(agent) for agent in agents]
        await asyncio.gather(*join_tasks)
        
        # Submit tiles from all agents concurrently
        tile = Tile(priority=1, content={"test": "data"}, agent_id="agent_0")
        submit_tasks = [room.submit_tile(tile) for _ in range(10)]
        
        # Have some agents leave during processing
        leave_tasks = [room.leave(agents[i]) for i in [1, 3]]
        
        # Run all concurrently
        results = await asyncio.gather(
            *submit_tasks,
            *leave_tasks,
            return_exceptions=True
        )
        
        # Verify no crashes and state consistency
        assert room.agent_count >= 3  # At least 3 agents should remain
        assert all(not isinstance(r, Exception) for r in results)

    @pytest.mark.asyncio
    async def test_malformed_tile_structures(self):
        """P0: Submit tiles with invalid data structures."""
        room = DeadbandRoom()
        
        agent = AsyncMock()
        agent.id = "test_agent"
        agent.send_state = AsyncMock()
        await room.join(agent)
        
        test_cases = [
            # (tile_data, expected_behavior)
            (None, "reject"),  # None tile
            ("not a dict", "reject"),  # String instead of dict
            ({"priority": 1}, "reject"),  # Missing content
            ({"content": {}}, "reject"),  # Missing priority
            ({"priority": "not_a_number", "content": {}}, "reject"),  # Wrong type
            ({"priority": 1, "content": {}, "extra": "field"}, "accept"),  # Extra fields OK
        ]
        
        for tile_data, expected in test_cases:
            if expected == "reject":
                with pytest.raises((ValueError, TypeError)):
                    await room.submit_tile(tile_data)
            else:
                # Should accept valid tile
                tile = Tile(**tile_data) if isinstance(tile_data, dict) else tile_data
                result = await room.submit_tile(tile)
                assert result is not None
    
    @pytest.mark.asyncio
    async def test_tile_flood_protection(self):
        """P0: Submit tiles faster than processing rate."""
        room = DeadbandRoom()
        
        agent = AsyncMock()
        agent.id = "flood_agent"
        agent.send_state = AsyncMock()
        await room.join(agent)
        
        # Submit large burst of tiles
        tiles = []
        for i in range(100):  # Large number to test queue limits
            tile = Tile(
                priority=i % 3 + 1,  # Cycle through P1-P3
                content={"index": i, "data": "x" * 1000},  # Include large payload
                agent_id=agent.id
            )
            tiles.append(tile)
        
        # Submit all tiles rapidly
        submit_tasks = [room.submit_tile(tile) for tile in tiles]
        results = await asyncio.gather(*submit_tasks, return_exceptions=True)
        
        # Check for rate limiting or queue protection
        exceptions = [r for r in results if isinstance(r, Exception)]
        
        # Either all succeed (if no flood protection) or some fail (if protection exists)
        # Both are valid implementations
        if exceptions:
            # If there's flood protection, verify it's a sensible error
            for exc in exceptions:
                assert "queue" in str(exc).lower() or "rate" in str(exc).lower() or "limit" in str(exc).lower()
        else:
            # If no protection, verify all tiles processed
            assert len([r for r in results if r is not None]) == len(tiles)
    
    @pytest.mark.asyncio

            with pytest.raises(ValueError):

            {"deadband_min": 0, "deadband_max": 1},  # Minimal valid range

            {"tile_queue_size": 1},  # Smallest valid queue
            {"heartbeat_timeout_seconds": 0.001},  # Very short but positive timeout
        ]
        
        for config in valid_configs:
            room = DeadbandRoom(config=config)
            assert room is not None
            # Verify config was applied
            for key, value in config.items():
                assert getattr(room.config, key) == value
