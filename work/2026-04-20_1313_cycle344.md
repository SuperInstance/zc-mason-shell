```python
def test_protocol_state_corruption():
    """P0: Attempting P1 before P0 completion should fail."""
    room = DeadbandRoom(room_id="edge-test-1")
    agent_id = "test-agent-edge-1"
    
    # Start P0
    room.start_protocol_phase(agent_id, "P0")
    
    # Attempt to skip to P1 without completing P0
    with pytest.raises(ProtocolViolationError):
        room.start_protocol_phase(agent_id, "P1")
    
    # Complete P0 properly
    room.complete_protocol_phase(agent_id, "P0")
    
    # Now P1 should work
    room.start_protocol_phase(agent_id, "P1")
    assert room.get_agent_phase(agent_id) == "P1"
```

### Test 5: Network Partition Simulation
```python
def test_network_partition_simulation():
    """P0: Handle partial writes and concurrent access patterns."""
    room = DeadbandRoom(room_id="edge-test-5")
    
    import threading
    import time
    
    results = []
    errors = []
    
    def concurrent_agent_work(agent_num):
        try:
            agent_id = f"concurrent-agent-{agent_num}"
            room.register_agent(agent_id)
            
            # Simulate network delay
            time.sleep(0.01 * agent_num)
            
            room.start_protocol_phase(agent_id, "P0")
            time.sleep(0.005)
            
            tile_id = room.submit_tile(agent_id, {"agent": agent_num})
            results.append((agent_id, tile_id))
            
        except Exception as e:
            errors.append((agent_num, str(e)))
    
    # Launch multiple concurrent agents
    threads = []
    for i in range(10):
        t = threading.Thread(target=concurrent_agent_work, args=(i,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify no data corruption
    assert len(errors) == 0, f"Errors in concurrent access: {errors}"
    
    # All agents should have unique tile IDs
    tile_ids = [tid for _, tid in results]
    assert len(set(tile_ids)) == len(tile_ids), "Duplicate tile IDs detected"
    
    # Room state should be consistent
    assert room.get_active_agent_count() == 10
```
