### Test 1: Concurrent Agent Protocol Adherence
```python
def test_concurrent_agents_deadband():
    """Test DeadbandRoom with multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent P0 submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 134,
            "phase": 4,
            "protocol_step": "P0",
            "content": {"negative_space": ["skip_p1", "assume_safe"]}
        }
        tiles.append(tile)
    
    results = room.process_batch(tiles)
    
    # All agents should be rejected for trying to skip P1
    assert all(r["status"] == "rejected" for r in results)
    assert all("P0 violation" in r["reason"] for r in results)
```

### Test 3: State Persistence Across Cycles

def test_state_persistence():
    """Test that agent state persists correctly through protocol phases."""
    room = DeadbandRoom()
    agent_id = "persistent_agent"
    
    # Complete protocol sequence
    tiles = [
        {"agent_id": agent_id, "cycle": 1, "protocol_step": "P0", "content": {"negative_space": ["rock"]}},
        {"agent_id": agent_id, "cycle": 2, "protocol_step": "P1", "content": {"safe_channels": ["channel_a"]}},
        {"agent_id": agent_id, "cycle": 3, "protocol_step": "P2", "content": {"optimization": "path_a"}}
    ]
    
    for tile in tiles:
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # Verify state tracking
    agent_state = room.get_agent_state(agent_id)
    assert agent_state["current_step"] == "P2"
    assert len(agent_state["completed_steps"]) == 3
    assert "P0" in agent_state["completed_steps"]
```

def test_malformed_tile_content():
    """Test handling of tiles with invalid content structures."""
    room = DeadbandRoom()
    
    test_cases = [
        # Binary data in content
        {"agent_id": "test", "content": b"binary_data"},
        # Nested recursion
        {"agent_id": "test", "content": {"self_ref": None}},
        # Oversized content (>1MB)
        {"agent_id": "test", "content": {"data": "x" * 1024 * 1025}},
        # Missing required protocol_step
        {"agent_id": "test", "content": {"something": "else"}},
    ]
    
    for tile in test_cases:
        result = room.submit_tile(tile)
        assert result["status"] == "rejected"
        assert "invalid" in result["reason"].lower() or "malformed" in result["reason"].lower()
```

### Test 5: Recovery After Protocol Violation
```python
def test_recovery_after_violation():
    """Test agent can recover and continue after protocol violation."""
    room = DeadbandRoom()
    agent_id = "recovering_agent"
    
    # Violation: Skip from P0 to P2
    tile_p0 = {"agent_id": agent_id, "protocol_step": "P0", "content": {"negative_space": []}}
    tile_p2 = {"agent_id": agent_id, "protocol_step": "P2", "content": {"optimization": "test"}}
    
    room.submit_tile(tile_p0)  # Accepted
    result_p2 = room.submit_tile(tile_p2)  # Rejected
    
    assert result_p2["status"] == "rejected"
    
    # Agent should be able to submit P1 after rejection
    tile_p1 = {"agent_id": agent_id, "protocol_step": "P1", "content": {"safe_channels": ["recovery"]}}
    result_p1 = room.submit_tile(tile_p1)
    
    # Should be accepted despite previous violation
    assert result_p1["status"] == "accepted"
    
    # Then P2 should work
    tile_p2_correct = {"agent_id": agent_id, "protocol_step": "P2", "content": {"optimization": "recovered"}}
    result_p2_correct = room.submit_tile(tile_p2_correct)
    assert result_p2_correct["status"] == "accepted"
```
