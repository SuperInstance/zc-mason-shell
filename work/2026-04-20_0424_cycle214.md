### Test 1: Partial Ghost Tile Recovery
**Scenario:** Ghost tile contains valid P0 data but corrupted P1/P2 sections
**Expected:** Room should extract usable P0 warnings while logging corruption
**Rationale:** Real-world data persistence failures create partial tiles

```python
def test_partial_ghost_tile_recovery(deadband_room):
    """Test recovery of usable P0 data from corrupted ghost tile."""
    corrupted_ghost = {
        "agent_id": "ghost_agent_001",
        "cycle": 42,
        "p0_warnings": ["avoid_north_channel", "watch_for_eddies"],
        "p1_channels": None,  # Corrupted/missing
        "p2_optimizations": b"corrupted_binary_data",
        "timestamp": "2026-04-20T04:30:00Z"
    }
    
    result = deadband_room._process_ghost_tile(corrupted_ghost)
    
    # Should extract P0 warnings
    assert "avoid_north_channel" in deadband_room.p0_warnings
    assert "watch_for_eddies" in deadband_room.p0_warnings
    
    # Should log corruption
    assert deadband_room.logs[-1]["type"] == "ghost_corruption"
    assert result["status"] == "partial_recovery"
```

### Test 2: Channel State Rollback on P2 Failure
**Scenario:** Agent successfully completes P1, starts P2 optimization, then submits invalid P2 tile
**Expected:** Room should roll back to P1 state, not reset to P0
**Rationale:** Failed optimizations shouldn't discard valid channel discovery

```python
def test_p2_failure_rollback_to_p1(deadband_room, valid_agent):
    """Test state preservation when P2 optimization fails."""
    # Complete P1 successfully
    p1_tile = {
        "agent_id": valid_agent,
        "cycle": 1,
        "p1_channels": ["safe_channel_a", "safe_channel_b"],
        "phase": "p1"
    }
    deadband_room.submit_tile(p1_tile)
    
    # Store P1 state for comparison
    original_channels = deadband_room.active_channels.copy()
    
    # Submit invalid P2 tile (missing required fields)
    invalid_p2 = {
        "agent_id": valid_agent,
        "cycle": 2,
        "phase": "p2"
        # Missing p2_optimizations field
    }
    
    with pytest.raises(ValueError, match="Missing p2_optimizations"):
        deadband_room.submit_tile(invalid_p2)
    
    # Should remain in P1 with channels intact
    assert deadband_room.current_phase == "p1"
    assert deadband_room.active_channels == original_channels
    assert "p2_optimization_failed" in deadband_room.logs[-1]["message"]
```

def test_concurrent_channel_conflict(deadband_room):
    """Test conflict resolution when agents report conflicting channels."""
    agent_a_tile = {
        "agent_id": "agent_alpha",
        "cycle": 10,
        "p1_channels": ["channel_x", "channel_y"],
        "timestamp": "2026-04-20T04:31:00.000Z",

    agent_b_tile = {
        "agent_id": "agent_beta", 
        "cycle": 10,
        "p1_channels": ["channel_y", "channel_z"],  # Overlap on channel_y
        "timestamp": "2026-04-20T04:31:00.001Z",  # 1ms later
        "phase": "p1"
    }
    
    # Submit concurrently (simulated)
    result_a = deadband_room.submit_tile(agent_a_tile)

    assert "channel_y" in deadband_room.active_channels

    assert "channel_z" in deadband_room.active_channels

### Test 4: Phase Transition Boundary Conditions
**Scenario:** Agent submits tile with phase field at exact boundary values
**Expected:** Room should handle "p0", "p1", "p2" case-insensitively with whitespace trimming
**Rationale:** Real data often has formatting inconsistencies

```python
def test_phase_field_boundary_conditions(deadband_room, valid_agent):
    """Test phase field parsing with edge cases."""
    test_cases = [
        ("P0", "p0"),  # Upper case
        (" p1 ", "p1"),  # Whitespace
        ("P2", "p2"),  # Upper case P2
        ("p0", "p0"),  # Normal
        ("", ValueError),  # Empty string
        ("p3", ValueError),  # Invalid phase
        (None, ValueError),  # Null value
    ]
    
    for input_phase, expected in test_cases:
        tile = {
            "agent_id": valid_agent,
            "cycle": 1,
            "phase": input_phase,
            "p0_warnings": ["test_warning"] if expected == "p0" else None,
            "p1_channels": ["test_channel"] if expected == "p1" else None,
            "p2_optimizations": {"test": "data"} if expected == "p2" else None
        }
        
        if expected == ValueError:
            with pytest.raises(ValueError, match="Invalid phase"):
                deadband_room.submit_tile(tile)
        else:
            result = deadband_room.submit_tile(tile)
            assert result["phase"] == expected
            assert deadband_room.current_phase == expected
            deadband_room.reset()  # Clean state for next test
```
