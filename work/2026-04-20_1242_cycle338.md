def test_concurrent_tile_submission_stress(self):
    """Test DeadbandRoom handling 100 concurrent tile submissions."""
    import threading
    from queue import Queue
    
    room = DeadbandRoom(room_id="stress_test")
    results = Queue()
    errors = []
    
    def submit_tile(agent_id):
        try:
            tile = {
                "agent_id": agent_id,
                "cycle": 1,
                "phase": 4,
                "content": f"Test from agent {agent_id}",
                "timestamp": time.time()
            }
            result = room.submit_tile(tile)
            results.put((agent_id, result))
        except Exception as e:
            errors.append((agent_id, str(e)))
    
    # Create 100 agent threads
    threads = []
    for i in range(100):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    # Wait for all threads
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    successful = results.qsize()
    self.assertEqual(successful, 100, 
                    f"Expected 100 successful submissions, got {successful}")
    self.assertEqual(len(errors), 0,
                    f"Expected 0 errors, got {len(errors)}: {errors}")
    
    # Verify tile count matches
    self.assertEqual(len(room.tiles), 100,
                    f"Expected 100 tiles in room, got {len(room.tiles)}")
```

def test_malformed_tile_validation(self):
    """Test DeadbandRoom rejection of various malformed tiles."""
    room = DeadbandRoom(room_id="validation_test")
    
    malformed_tiles = [
        # Missing required fields
        {"agent_id": "test_agent"},  # missing cycle, phase, content
        {"cycle": 1, "phase": 4},  # missing agent_id, content
        {"content": "test"},  # missing agent_id, cycle, phase
        
        # Wrong data types
        {"agent_id": 123, "cycle": 1, "phase": 4, "content": "test"},  # agent_id not string
        {"agent_id": "test", "cycle": "one", "phase": 4, "content": "test"},  # cycle not int
        {"agent_id": "test", "cycle": 1, "phase": "four", "content": "test"},  # phase not int
        {"agent_id": "test", "cycle": 1, "phase": 4, "content": 123},  # content not string
        
        # Invalid values
        {"agent_id": "test", "cycle": -1, "phase": 4, "content": "test"},  # negative cycle
        {"agent_id": "test", "cycle": 1, "phase": 5, "content": "test"},  # phase > 4
        {"agent_id": "test", "cycle": 1, "phase": 0, "content": "test"},  # phase < 1
        
        # Extra fields (should be ignored or rejected based on implementation)
        {"agent_id": "test", "cycle": 1, "phase": 4, "content": "test", "extra": "field"},
        
        # Empty content
        {"agent_id": "test", "cycle": 1, "phase": 4, "content": ""},
        
        # Content too large (10MB+)
        {"agent_id": "test", "cycle": 1, "phase": 4, "content": "x" * (10 * 1024 * 1024 + 1)},
    ]
    
    for i, tile in enumerate(malformed_tiles):
        with self.subTest(tile_index=i, tile=tile):
            with self.assertRaises((ValueError, TypeError, ValidationError)):
                room.submit_tile(tile)
    
    # Verify no tiles were accepted
    self.assertEqual(len(room.tiles), 0,
                    f"Expected 0 tiles accepted, got {len(room.tiles)}")

def test_partial_failure_recovery(self):
    """Test DeadbandRoom behavior during simulated network partitions."""
    room = DeadbandRoom(room_id="partition_test")
    
    # Mock external dependencies to simulate failures
    original_validate = room._validate_tile
    original_persist = room._persist_tile
    
    failure_modes = [
        # Database connection failure
        lambda: ConnectionError("Database unreachable"),
        # File system full
        lambda: OSError("No space left on device"),
        # Memory allocation failure
        lambda: MemoryError("Cannot allocate memory"),
        # Network timeout
        lambda: TimeoutError("Operation timed out"),
    ]
    
    successful_tiles = 0
    
    for i, failure_mode in enumerate(failure_modes):
        # Set up failure for this iteration
        room._validate_tile = lambda tile: original_validate(tile)
        room._persist_tile = lambda tile: (_ for _ in ()).throw(failure_mode())
        
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": i + 1,
            "phase": 4,
            "content": f"Test during failure mode {i}",
            "timestamp": time.time()
        }
        
        try:
            result = room.submit_tile(tile)
            # Some implementations might queue for retry
            if result.get("status") == "queued":
                successful_tiles += 1
        except Exception:
            # Expected failure
            pass
    
    # Restore original methods
    room._validate_tile = original_validate
    room._persist_tile = original_persist
    
    # Submit a final tile to verify recovery
    recovery_tile = {
        "agent_id": "recovery_agent",
        "cycle": 99,
        "phase": 4,
        "content": "Recovery test",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(recovery_tile)
    self.assertIn(result.get("status"), ["accepted", "queued"],
                 f"Recovery tile failed with status: {result.get('status')}")
    
    # Verify at least recovery tile is present
    self.assertGreaterEqual(len(room.tiles), 1,
                           f"Expected at least 1 tile after recovery, got {len(room.tiles)}")
```
