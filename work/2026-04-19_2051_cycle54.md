## Test 5: Consecutive Identical Input Handling

```rust
// plato-torch/tests/deadband_room_tests.rs
#[test]
fn test_consecutive_identical_inputs() {
    use plato_torch::presets::deadband_room::DeadbandRoom;
    use plato_torch::room::Room;
    
    // Setup room with standard thresholds
    let mut room = DeadbandRoom::new(0.1, 0.3, 0.7).unwrap();
    
    // First input in P1 zone
    let tile1 = serde_json::json!({
        "value": 0.5,
        "metadata": {"source": "test"}
    });
    
    let result1 = room.process_input(&tile1).unwrap();
    assert_eq!(result1.state, "P1_ACTIVE");
    assert!(result1.processed);
    
    // Identical second input
    let tile2 = serde_json::json!({
        "value": 0.5,
        "metadata": {"source": "test"}
    });
    
    let result2 = room.process_input(&tile2).unwrap();
    
    // Should remain in P1_ACTIVE state
    assert_eq!(result2.state, "P1_ACTIVE");
    assert!(result2.processed);
    
    // Verify no state oscillation
    assert_eq!(result1.state, result2.state);
}
```
