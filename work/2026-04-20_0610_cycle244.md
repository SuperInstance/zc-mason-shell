```python
def test_concurrent_agents_respect_protocol_priority():
    """Multiple agents in same room must not skip P0/P1 phases."""
    room = DeadbandRoom(room_id="concurrent_test")
    
    # Simulate two agents joining simultaneously
    agent1_state = room.join_agent("agent_1")
    agent2_state = room.join_agent("agent_2")
    
    # Both should start at P0
    assert agent1_state["phase"] == "P0"
    assert agent2_state["phase"] == "P0"
    
    # Agent 1 completes P0, moves to P1
    room.submit_tile("agent_1", {"phase": "P0", "content": "negative_space_mapped"})
    state1 = room.get_agent_state("agent_1")
    assert state1["phase"] == "P1"
    
    # Agent 2 should still be at P0 (independent progression)
    state2 = room.get_agent_state("agent_2")
    assert state2["phase"] == "P0"
    
    # Verify no protocol violation occurred
    assert room.protocol_violations == 0
```

### Test 2: Network Failure During Tile Submission
```python
def test_tile_submission_with_simulated_network_failure():
    """Room should handle IO errors gracefully without state corruption."""
    room = DeadbandRoom(room_id="network_failure_test")
    room.join_agent("flaky_agent")
    
    # Simulate network failure on first submission attempt
    original_save = room._save_state
    call_count = 0
    
    def failing_save(*args, **kwargs):
        nonlocal call_count
        call_count += 1
        if call_count == 1:
            raise IOError("Simulated network failure")
        return original_save(*args, **kwargs)
    
    room._save_state = failing_save
    
    # First submission should fail but not crash
    try:
        room.submit_tile("flaky_agent", {"phase": "P0", "content": "test"})
    except IOError:
        pass  # Expected
    
    # Agent state should remain consistent (still at P0)
    state = room.get_agent_state("flaky_agent")
    assert state["phase"] == "P0"
    assert len(state["submitted_tiles"]) == 0
    
    # Second submission should succeed
    room._save_state = original_save
    result = room.submit_tile("flaky_agent", {"phase": "P0", "content": "recovered"})
    assert result["success"] is True
```

```python
def test_malformed_tile_payload_rejection():
    """Room must reject tiles missing required fields or with invalid structure."""
    room = DeadbandRoom(room_id="validation_test")
    room.join_agent("test_agent")
    
    test_cases = [
        # (payload, expected_error)
        ({}, "Missing 'phase' field"),
        ({"phase": "P0"}, "Missing 'content' field"),
        ({"phase": "P5", "content": "test"}, "Invalid phase: P5"),
        ({"phase": "P0", "content": ""}, "Empty content"),
        ({"phase": "P0", "content": "x" * 10001}, "Content too long"),
        ({"phase": "P1", "content": "test", "extra": b"binary_data"}, "Invalid data type"),
    ]
    
    for payload, expected_error in test_cases:
        result = room.submit_tile("test_agent", payload)
        assert result["success"] is False
        assert expected_error in result.get("error", "")
```

```python
def test_protocol_boundary_conditions():
    """Test edge cases: empty safe channels, max tile limits, phase transitions."""
    room = DeadbandRoom(room_id="boundary_test")
    room.join_agent("boundary_agent")
    
    # Test: Agent completes P0 but reports empty safe channels
    room.submit_tile("boundary_agent", {
        "phase": "P0",
        "content": "negative_space_mapped",
        "safe_channels": []  # Empty list
    })
    
    # Should still transition to P1 (empty channels is valid finding)
    state = room.get_agent_state("boundary_agent")
    assert state["phase"] == "P1"
    
    # Test: Submit maximum allowed tiles
    max_tiles = room.config.get("max_tiles_per_phase", 100)
    for i in range(max_tiles):
        room.submit_tile("boundary_agent", {
            "phase": "P1",
            "content": f"channel_analysis_{i}"
        })
    
    # Next submission should be rejected
    result = room.submit_tile("boundary_agent", {
        "phase": "P1",
        "content": "extra_tile"
    })
    assert result["success"] is False
    assert "tile limit" in result.get("error", "").lower()
```

```python
def test_recovery_from_corrupted_state():
    """Room should detect and recover from corrupted state persistence."""
    import json
    import tempfile
    import os
    
    # Create a temporary state file with corruption
    with tempfile.NamedTemporaryFile(mode='w', suffix='.json', delete=False) as f:
        # Write valid JSON then corrupt it
        valid_state = {
            "room_id": "recovery_test",
            "agents": {
                "recovery_agent": {
                    "phase": "P1",
                    "submitted_tiles": ["tile1", "tile2"]
                }
            }
        }
        json.dump(valid_state, f)
        f.write("\nCORRUPTED_LINE\n")  # Add corruption
        temp_path = f.name
    
    try:
        # Attempt to load corrupted state
        room = DeadbandRoom(room_id="recovery_test")
        room.state_file = temp_path
        
        # Should either: 1) Load valid portion, 2) Reset to clean state, or 3) Raise specific error
        try:
            room.load_state()
            # If loaded, verify critical data intact
            if "recovery_agent" in room.agents:
                assert room.agents["recovery_agent"]["phase"] in ["P0", "P1", "P2"]
        except (json.JSONDecodeError, KeyError) as e:
            # Expected for severe corruption - room should initialize fresh
            assert len(room.agents) == 0
            assert room.phase == "P0"
    finally:
        os.unlink(temp_path)
    
    # Verify room remains functional after corruption handling
    room.join_agent("new_agent")
    result = room.submit_tile("new_agent", {"phase": "P0", "content": "post_recovery"})
    assert result["success"] is True
```
