## 5 Additional Edge-Case Tests

### Test 1: Concurrent Tile Submission Race Condition
**Scenario:** Two agents submit tiles simultaneously for the same room cycle.
**Edge:** Potential race in `_validate_and_store_tile` leading to state corruption.
**Test:** Use `threading` or `asyncio` to simulate concurrent submissions, verify atomicity and final room state consistency.

### Test 2: Protocol Violation with Auto-Correction
**Scenario:** Agent submits a tile that violates Deadband Protocol (skips P1, goes P0 → P2).
**Edge:** Room's validation logic should reject or auto-correct to P1.
**Test:** Submit malformed protocol tile, assert room either rejects it or logs correction, ensures P1 channel is mapped.

### Test 3: Room State Rollback on Validation Failure
**Scenario:** A batch of tiles is submitted, one fails validation mid-batch.
**Edge:** Room should rollback partial state changes, maintain consistency.
**Test:** Submit multiple tiles where the 3rd is invalid, verify first two are not persisted, room state unchanged.

### Test 4: High Volume Tile Flood
**Scenario:** 10,000 tiles submitted in rapid succession.
**Edge:** Memory/performance degradation, queue overflow.
**Test:** Load test with mocked agents, measure memory usage and processing time, ensure no tile loss or deadlock.

### Test 5: Network Partition During Submission
**Scenario:** Agent submits tile, network fails before acknowledgment.
**Edge:** Tile may be processed but agent receives timeout, leading to duplicate submission.
**Test:** Mock network failure after tile storage but before ACK, simulate retry, ensure idempotency (no duplicate tiles).

import pytest
import asyncio
from unittest.mock import patch, AsyncMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.schemas import Tile, ProtocolPhase

class TestDeadbandRoomEdgeCases:
    """Edge-case tests for DeadbandRoom."""
    
    @pytest.mark.asyncio
    async def test_concurrent_tile_submission(self):
        """Test race condition handling for concurrent tile submissions."""
        room = DeadbandRoom(config={"room_id": "edge_concurrent"})
        tiles = [Tile(agent_id=f"agent_{i}", phase=ProtocolPhase.P0, content=f"test_{i}") for i in range(10)]
        
        async def submit(tile):
            return await room.submit_tile(tile)
        
        # Submit all tiles concurrently
        results = await asyncio.gather(*[submit(tile) for tile in tiles])
        assert all(r["status"] == "accepted" for r in results)
        assert len(room.get_tiles()) == 10  # All tiles stored
        
    @pytest.mark.asyncio
    async def test_protocol_violation_auto_correct(self):
        """Test room handles P0→P2 skip by auto-correcting to P1."""
        room = DeadbandRoom(config={"room_id": "edge_violation"})
        # Simulate agent history: last phase was P0
        room.agent_phases["test_agent"] = ProtocolPhase.P0
        violating_tile = Tile(agent_id="test_agent", phase=ProtocolPhase.P2, content="skip_p1")
        
        with pytest.warns(UserWarning, match="Protocol violation"):
            result = await room.submit_tile(violating_tile)
        
        # Room should either reject or correct
        assert result["status"] in ("rejected", "corrected")
        if result["status"] == "corrected":
            assert room.agent_phases["test_agent"] == ProtocolPhase.P1  # Now in P1
            
    @pytest.mark.asyncio
    async def test_batch_rollback_on_validation_failure(self):
        """Test state rollback when a tile in batch fails validation."""
        room = DeadbandRoom(config={"room_id": "edge_rollback"})
        valid_tile = Tile(agent_id="agent1", phase=ProtocolPhase.P0, content="ok")
        invalid_tile = Tile(agent_id="agent2", phase=None, content="")  # Invalid: missing phase
        
        # Submit valid tile
        await room.submit_tile(valid_tile)
        # Submit invalid tile, should raise
        with pytest.raises(ValueError):
            await room.submit_tile(invalid_tile)
        
        # Ensure valid tile was not persisted due to rollback
        assert len(room.get_tiles()) == 0
        
    @pytest.mark.asyncio
    async def test_high_volume_tile_flood(self):
        """Test performance under high load."""
        room = DeadbandRoom(config={"room_id": "edge_flood"})
        tiles = [Tile(agent_id=f"agent_{i}", phase=ProtocolPhase.P0, content="x"*1000) for i in range(10000)]
        
        import time
        start = time.time()
        for tile in tiles:
            await room.submit_tile(tile)
        elapsed = time.time() - start
        
        # Should process within reasonable time (e.g., < 30 seconds)
        assert elapsed < 30.0
        assert len(room.get_tiles()) == 10000
        
    @pytest.mark.asyncio
    async def test_network_partition_idempotency(self):
        """Test idempotent submission on network failure."""
        room = DeadbandRoom(config={"room_id": "edge_network"})
        tile = Tile(agent_id="agent1", phase=ProtocolPhase.P0, content="test")
        
        # Simulate storage success then network failure
        with patch.object(room, '_send_ack', side_effect=ConnectionError):
            with pytest.raises(ConnectionError):
                await room.submit_tile(tile)
        
        # Tile may have been stored despite error
        stored_count = len(room.get_tiles())
        
        # Retry submission (agent retries after timeout)
        result = await room.submit_tile(tile)
        # Should be idempotent: no duplicate if already stored
        assert len(room.get_tiles()) == max(1, stored_count)
