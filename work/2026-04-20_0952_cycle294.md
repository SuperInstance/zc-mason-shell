```python
def test_p0_to_p2_skip_violation():
    """P0: Agent attempts to skip P1 and go directly to P2.
    Expected: Room rejects with DeadbandViolation error."""
    room = DeadbandRoom(config={"max_agents": 1})
    agent_id = room.register_agent("test_agent")
    
    # Complete P0 mapping
    room.submit_p0(agent_id, {"negative_space": ["rock1", "rock2"]})
    
    # Attempt to submit P2 optimization without P1
    with pytest.raises(DeadbandViolation) as exc:
        room.submit_p2(agent_id, {"optimized_path": "shortcut"})
    
    assert "P1 channel finding required" in str(exc.value)
```

### Test 3: Concurrent Agent Contention
```python
def test_concurrent_p1_submissions():
    """Multiple agents submit P1 channels simultaneously.
    Expected: Thread-safe processing, no data corruption."""
    room = DeadbandRoom(config={"max_agents": 3})
    agents = [room.register_agent(f"agent_{i}") for i in range(3)]
    
    # All agents complete P0
    for aid in agents:
        room.submit_p0(aid, {"negative_space": ["shared_rock"]})
    
    # Concurrent P1 submissions
    import threading
    results = []
    
    def submit_p1(agent_id, channel):
        result = room.submit_p1(agent_id, {"channels": [channel]})
        results.append((agent_id, result))
    
    threads = []
    for i, aid in enumerate(agents):
        t = threading.Thread(target=submit_p1, args=(aid, f"channel_{i}"))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions processed
    assert len(results) == 3
    assert len(set(r[1]["status"] for r in results)) == 1  # All same status
    assert room.get_agent_count() == 3
```

```python
        "optimized_path": "normal",

    # Should be accepted but sanitized
    assert result["status"] == "accepted"
    assert "DROP TABLE" not in room.get_last_tile(agent_id)["content"]
    assert room.tile_count() > 0  # Table not actually dropped
```

### Test 5: Network Partition During Protocol
```python
def test_partial_failure_p1_to_p2_transition():
    """Network partition occurs between P1 and P2 submission.
    Expected: Room maintains consistency, allows retry."""
    room = DeadbandRoom(config={"retry_count": 3})
    agent_id = room.register_agent("unstable_agent")
    
    # Mock network failure
    original_submit = room._submit_tile
    call_count = 0
    
    def failing_submit(agent_id, data, phase):
        nonlocal call_count
        call_count += 1
        if call_count == 2:  # Fail on second attempt
            raise NetworkError("Partition detected")
        return original_submit(agent_id, data, phase)
    
    room._submit_tile = failing_submit
    
    # Complete P0
    room.submit_p0(agent_id, {"negative_space": ["rock"]})
    
    # P1 should succeed
    p1_result = room.submit_p1(agent_id, {"channels": ["redundant"]})
    assert p1_result["status"] == "accepted"
    
    # P2 will fail once, then succeed on retry
    p2_result = room.submit_p2(agent_id, {"optimized_path": "recovery"})
    assert p2_result["status"] == "accepted"
    assert call_count == 3  # Initial + fail + retry success
```
