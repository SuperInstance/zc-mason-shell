### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        state = {
            "phase": "P1",
            "channels": ["safe_channel_1"],
            "agent_id": agent_id
        }
        tile = {"state": state, "action": "channel_discovery"}
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (simulated)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions were processed
    assert len(results) == 5
    # Verify no data corruption occurred
    for result in results:
        assert "processed" in result or "rejected" in result
    # Verify room state consistency
    assert room.get_agent_count() == 5
```

### Test 5: Network Partition Simulation
```python
def test_network_partition_recovery(self):
    """Test agent disconnection and reconnection during protocol."""
    room = DeadbandRoom()
    agent_id = "network_agent"
    
    # Start P0 mapping
    state_p0 = {
        "phase": "P0",
        "negative_space": ["rock_1", "hazard_zone"],
        "agent_id": agent_id
    }
    
    tile_p0 = {"state": state_p0, "action": "p0_mapping"}
    result_p0 = room.submit_tile(tile_p0)
    assert result_p0.get("status") == "accepted"
    
    # Simulate network partition (agent disappears)
    room.simulate_network_partition(agent_id)
    
    # Agent tries to continue P1 after partition
    state_p1 = {
        "phase": "P1",
        "negative_space": ["rock_1", "hazard_zone"],
        "channels": ["safe_channel_1"],
        "agent_id": agent_id,
        "session_id": "expired_session"  # Old session due to partition
    }
    
    tile_p1 = {"state": state_p1, "action": "channel_discovery"}
    result_p1 = room.submit_tile(tile_p1)
    
    # Should handle expired session - either reject or create new session
    assert result_p1 is not None
    # Room should maintain consistency despite partition
    assert room.get_agent_state(agent_id) is not None or room.is_agent_expired(agent_id)
```
