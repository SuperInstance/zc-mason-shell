### Test 3: Concurrent Agent Interactions
```python
def test_concurrent_agent_interactions(room):
    """P0: Multiple agents submitting simultaneously should maintain protocol isolation."""
    from threading import Thread
    
    agent1 = MockAgent(id="agent_1")
    agent2 = MockAgent(id="agent_2")
    
    results = []
    
    def submit_tile(agent, priority):
        tile = {
            "priority": priority,
            "content": f"Test from {agent.id}",
            "agent_id": agent.id,
            "timestamp": "2026-04-20T12:00:00Z"
        }
        results.append(room.process_interaction(agent, tile))
    
    # Both agents attempt P0 simultaneously
    threads = [
        Thread(target=submit_tile, args=(agent1, "P0")),
        Thread(target=submit_tile, args=(agent2, "P0"))
    ]
    
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Both should succeed with isolated histories
    assert all(r["valid"] for r in results)
    assert len(room.get_agent_history(agent1.id)) == 1
    assert len(room.get_agent_history(agent2.id)) == 1
    assert room.get_agent_history(agent1.id) != room.get_agent_history(agent2.id)
```

def test_malformed_tile_recovery(room, agent):
    """P0: Room should recover gracefully after rejecting malformed tile."""
    # First valid tile
    valid_tile = {
        "priority": "P0",
        "content": "Valid P0 tile",
        "agent_id": agent.id,
        "timestamp": "2026-04-20T12:00:00Z"
    }
    room.process_interaction(agent, valid_tile)
    
    # Malformed tile (missing required field)
    malformed_tile = {
        "priority": "P1",
        # Missing "content" field
        "agent_id": agent.id,
        "timestamp": "2026-04-20T12:01:00Z"
    }
    
    result = room.process_interaction(agent, malformed_tile)
    assert not result["valid"]
    
    # Room should still accept next valid tile
    recovery_tile = {
        "priority": "P1",
        "content": "Recovery after malformed",
        "agent_id": agent.id,
        "timestamp": "2026-04-20T12:02:00Z"
    }
    
    recovery_result = room.process_interaction(agent, recovery_tile)
    assert recovery_result["valid"]
    assert len(room.get_agent_history(agent.id)) == 2  # Only valid tiles counted
```

### Test 5: Protocol Reset Mid-Session

def test_protocol_reset_mid_session(room, agent):
    """P0: Room reset during agent session should clear history but maintain agent state."""
    # Build some history
    for priority in ["P0", "P1", "P2"]:
        tile = {
            "priority": priority,
            "content": f"Test {priority}",
            "agent_id": agent.id,
            "timestamp": f"2026-04-20T12:0{priority[-1]}:00Z"
        }
        room.process_interaction(agent, tile)
    
    assert len(room.get_agent_history(agent.id)) == 3
    
    # Reset room
    room.reset()
    
    # History should be cleared
    assert room.get_agent_history(agent.id) == []
    assert room.get_tile_count() == 0
    
    # Agent should be able to start fresh protocol
    new_p0_tile = {
        "priority": "P0",
        "content": "Fresh start after reset",
        "agent_id": agent.id,
        "timestamp": "2026-04-20T12:10:00Z"
    }
    
    result = room.process_interaction(agent, new_p0_tile)
    assert result["valid"]
    assert len(room.get_agent_history(agent.id)) == 1
```
