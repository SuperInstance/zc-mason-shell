```python
def test_priority_violation_recovery():
    """Test that room recovers when P2 tiles arrive before P0 channels are mapped."""
    room = DeadbandRoom(room_id="test_recovery")
    
    # Submit P2 tile first (violation)
    p2_tile = Tile(content="optimize path", priority="P2", agent_id="test_agent")
    result = room.submit_tile(p2_tile)
    
    # Should be queued, not rejected
    assert result["status"] == "queued"
    assert "pending_p0" in result["message"]
    
    # Now submit P0 tile
    p0_tile = Tile(content="avoid rocks", priority="P0", agent_id="test_agent")
    result = room.submit_tile(p0_tile)
    assert result["status"] == "accepted"
    
    # P2 tile should now process
    processed = room.process_queued_tiles()
    assert len(processed) == 1
    assert processed[0].content == "optimize path"
```

### Test 2: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission():
    """Test thread-safe tile aggregation with simulated concurrent access."""
    room = DeadbandRoom(room_id="test_concurrent")
    
    def submit_tiles(agent_id, count):
        tiles = []
        for i in range(count):
            tile = Tile(
                content=f"test_{agent_id}_{i}",
                priority="P0",
                agent_id=agent_id
            )
            tiles.append(tile)
        return [room.submit_tile(t) for t in tiles]
    
    # Simulate 3 agents submitting simultaneously
    import threading
    results = []
    
    def worker(agent_id):
        results.extend(submit_tiles(agent_id, 10))
    
    threads = [
        threading.Thread(target=worker, args=(f"agent_{i}",))
        for i in range(3)
    ]
    
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # All tiles should be accepted without corruption
    accepted = [r for r in results if r["status"] == "accepted"]
    assert len(accepted) == 30
    
    # Verify no duplicate or missing tiles
    contents = set()
    for tile_batch in room.tile_batches:
        for tile in tile_batch:
            contents.add(tile.content)
    
    assert len(contents) == 30
```

```python
def test_partial_write_failure():
    """Test resilience to partial disk writes (simulated network partition)."""
    import tempfile
    import os
    
    room = DeadbandRoom(room_id="test_partial")
    
    # Monkey-patch the save method to simulate partial write
    original_save = room._save_to_disk
    
    def failing_save(data, path):
        # Write first half, then simulate failure
        with open(path, 'wb') as f:
            half_data = data[:len(data)//2]
            f.write(half_data)
        raise IOError("Simulated disk full / network partition")
    
    room._save_to_disk = failing_save
    
    # Submit tile - should handle save failure gracefully
    tile = Tile(content="test_partial", priority="P0", agent_id="test_agent")
    
    try:
        result = room.submit_tile(tile)
        # If submission succeeds despite save failure, tile should be in memory only
        if result["status"] == "accepted":
            assert any(t.content == "test_partial" 
                      for batch in room.tile_batches 
                      for t in batch)
    except IOError:
        # Save failed, but room should remain in consistent state
        assert room.tile_batches[-1][-1].content != "test_partial"  # Not partially added
    
    # Restore original save method
    room._save_to_disk = original_save
    
    # Verify room still works
    tile = Tile(content="recovery", priority="P0", agent_id="test_agent")
    result = room.submit_tile(tile)
    assert result["status"] == "accepted"
```
