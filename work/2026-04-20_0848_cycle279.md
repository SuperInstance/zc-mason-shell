```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom(room_id="concurrent_test")
    
    # Simulate 5 agents submitting at same time
    tiles = []
    for i in range(5):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": 100 + i,
            "content": f"Test content from agent {i}",
            "protocol_phase": "P1",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles (simulating concurrent access)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent submissions should all succeed"
    
    # Verify all tiles are stored
    stored_tiles = room.get_tiles()
    assert len(stored_tiles) == 5, \
        "All 5 tiles should be stored"
    
    # Verify no data corruption
    agent_ids = {t["agent_id"] for t in stored_tiles}
    assert agent_ids == {f"agent_{i}" for i in range(5)}, \
        "All agent IDs should be preserved"
```

```python
def test_malformed_tile_data(self):

    room = DeadbandRoom(room_id="malformed_test")
    
    test_cases = [
        # (tile_data, expected_error)

        ("not json", "Tile must be valid JSON object"),

        ({"agent_id": "test"}, "Missing required field: content"),
        ({"agent_id": "test", "content": "x" * 10000}, "Content too large"),
        ({"agent_id": "test", "content": "test", "protocol_phase": "P3"}, "Invalid protocol phase"),
        ({"agent_id": "test", "content": "test", "cycle": -1}, "Invalid cycle number"),
    ]
    
    for tile_data, expected_error in test_cases:
        try:
            result = room.submit_tile(tile_data)
            if result["status"] == "accepted":
                pytest.fail(f"Should have rejected: {tile_data}")
        except (ValueError, TypeError) as e:
            assert expected_error in str(e), \
                f"Wrong error for {tile_data}: got {e}, expected {expected_error}"
```

```python
def test_state_corruption_recovery(self):
    """Test room recovery from corrupted state file."""
    room = DeadbandRoom(room_id="corruption_test")
    
    # First, add some valid tiles
    for i in range(3):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": 200 + i,
            "content": f"Valid tile {i}",
            "protocol_phase": "P1"
        }
        room.submit_tile(tile)
    
    # Simulate file corruption by writing garbage
    state_file = room._get_state_file_path()
    with open(state_file, 'w') as f:
        f.write("CORRUPTED_DATA" * 100)
    
    # Try to load room - should handle corruption gracefully
    try:
        recovered_room = DeadbandRoom(room_id="corruption_test")
        # Should either initialize fresh or recover what it can
        tiles = recovered_room.get_tiles()
        # Acceptable outcomes:
        # 1. Empty tiles list (fresh start)
        # 2. Partial recovery of valid tiles
        # 3. Full recovery if corruption was detected and ignored
        assert isinstance(tiles, list), \
            "Should return list even after corruption"
    except Exception as e:
        # Room should not crash on corruption
        pytest.fail(f"Room crashed on corruption: {e}")
    
    # Verify room can still accept new tiles after corruption
    new_tile = {
        "agent_id": "recovery_agent",
        "cycle": 300,
        "content": "Post-corruption tile",
        "protocol_phase": "P2"
    }
    result = room.submit_tile(new_tile)
    assert result["status"] == "accepted", \
        "Should accept tiles after corruption recovery"
```

### Test 4: Protocol Violation and Correction
```python
def test_protocol_violation_correction(self):
    """Test agent violating then correcting protocol."""
    room = DeadbandRoom(room_id="protocol_violation_test")
    
    # Agent tries to skip to P2 without P1 (violation)
    violating_tile = {
        "agent_id": "eager_agent",
        "cycle": 400,
        "content": "Trying to optimize without safe channel",
        "protocol_phase": "P2",
        "metadata": {"attempted_skip": True}
    }
    
    # Submission should be rejected or marked as protocol violation
    result = room.submit_tile(violating_tile)
    
    if result["status"] == "rejected":
        assert "protocol violation" in result.get("reason", "").lower(), \
            "Rejection should mention protocol violation"
    elif result["status"] == "accepted":
        # If accepted, tile should be marked as protocol violation
        tile = room.get_tiles()[-1]
        assert tile.get("protocol_violation", False) == True, \
            "Accepted violation should be flagged"
    
    # Agent corrects itself - does P1 properly
    correcting_tile = {
        "agent_id": "eager_agent",
        "cycle": 401,
        "content": "Mapping negative space (P0)",
        "protocol_phase": "P0"
    }
    
    result = room.submit_tile(correcting_tile)
    assert result["status"] == "accepted", \
        "Corrected protocol should be accepted"
    
    # Now P1 should work
    p1_tile = {
        "agent_id": "eager_agent",
        "cycle": 402,
        "content": "Finding safe channels (P1)",
        "protocol_phase": "P1"
    }
    
    result = room.submit_tile(p1_tile)
    assert result["status"] == "accepted", \
        "P1 after P0 should work"
    
    # Now P2 should be allowed
    p2_tile = {
        "agent_id": "eager_agent",
        "cycle": 403,
        "content": "Optimizing within channel (P2)",
        "protocol_phase": "P2"
    }
    
    result = room.submit_tile(p2_tile)
    assert result["status"] == "accepted", \
        "P2 after P1 should work when protocol followed"
```
