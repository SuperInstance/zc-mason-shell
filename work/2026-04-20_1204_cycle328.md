### Test 1: Concurrent Tile Submission Race Condition
```python
import asyncio
from unittest.mock import AsyncMock
import pytest

@pytest.mark.asyncio
async def test_concurrent_tile_submission_race(deadband_room):
    """Test multiple agents submitting tiles simultaneously."""
    agent1 = AsyncMock(agent_id="agent_1", submit_tile=AsyncMock(return_value={"status": "queued"}))
    agent2 = AsyncMock(agent_id="agent_2", submit_tile=AsyncMock(return_value={"status": "queued"}))
    
    # Simulate concurrent submission
    tasks = [
        deadband_room.process_tile_submission(agent1, {"phase": "P1", "content": "test1"}),
        deadband_room.process_tile_submission(agent2, {"phase": "P2", "content": "test2"})
    ]
    
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # Verify both submissions were handled without deadlock
    assert len(results) == 2
    assert not any(isinstance(r, Exception) for r in results)
    
    # Check tile queue state
    assert deadband_room.tile_queue.qsize() == 2
```

### Test 3: Deeply Nested Malformed Tile Payload
```python
def test_deeply_nested_malformed_tile_validation(deadband_room):
    """Test validation of complex malformed payloads."""
    # Circular reference payload
    malicious_payload = {"data": {}}
    malicious_payload["data"]["self"] = malicious_payload
    
    # Extremely deep nesting
    deep_payload = {"level": 1}
    current = deep_payload
    for i in range(2, 1000):
        current[f"level_{i}"] = {"level": i}
        current = current[f"level_{i}"]
    
    # Binary data injection attempt
    binary_payload = {"content": b"\x00\x01\x02\x03\x04"}
    
    test_cases = [
        (malicious_payload, "circular_reference"),
        (deep_payload, "max_depth_exceeded"),
        (binary_payload, "invalid_encoding"),
        ({"phase": "P0", "content": None}, "null_content"),
        ({"phase": "P5", "content": "invalid_phase"}, "invalid_phase"),
    ]
    
    for payload, expected_error in test_cases:
        result = deadband_room.validate_tile_payload(payload)
        assert not result["valid"], f"Should reject {expected_error}"
        assert expected_error in result.get("error", "").lower()
```

### Test 4: Protocol Sequence Violation Recovery
```python
@pytest.mark.asyncio
async def test_protocol_sequence_violation_recovery(deadband_room):
    """Test agent skipping P0→P1→P2 sequence."""
    agent = AsyncMock(agent_id="sequence_violator")
    
    # Start at P0
    await deadband_room.start_protocol_session(agent)
    
    # Attempt to jump directly to P2 (violation)
    violation_result = await deadband_room.submit_protocol_action(
        agent.agent_id, 
        {"phase": "P2", "action": "optimize", "data": "skipped_p1"}
    )
    
    # Should be rejected
    assert violation_result["status"] == "rejected"
    assert "sequence" in violation_result["error"].lower()
    
    # Verify agent can recover by following correct sequence
    await deadband_room.advance_protocol_phase(agent.agent_id, "P1")
    await deadband_room.advance_protocol_phase(agent.agent_id, "P2")
    
    # Now P2 action should succeed
    valid_result = await deadband_room.submit_protocol_action(
        agent.agent_id,
        {"phase": "P2", "action": "optimize", "data": "valid_after_correction"}
    )
    assert valid_result["status"] == "accepted"
```

### Test 5: Resource Exhaustion During Training Cycle
```python
import psutil
import tempfile
import os

@pytest.mark.asyncio
async def test_resource_exhaustion_handling(deadband_room, monkeypatch):
    """Test behavior when system resources are constrained."""
    # Mock psutil to simulate high memory usage
    original_virtual_memory = psutil.virtual_memory
    mock_memory = type('obj', (object,), {
        'percent': 95.0,  # 95% memory used
        'available': 100 * 1024 * 1024  # 100MB available
    })
    
    monkeypatch.setattr(psutil, 'virtual_memory', lambda: mock_memory)
    
    # Mock disk space
    mock_disk = type('obj', (object,), {
        'free': 50 * 1024 * 1024,  # 50MB free
        'total': 100 * 1024 * 1024 * 1024  # 100GB total
    })
    monkeypatch.setattr(psutil, 'disk_usage', lambda path: mock_disk)
    
    # Attempt training cycle with constrained resources
    agent = AsyncMock(agent_id="resource_test_agent")
    
    with tempfile.TemporaryDirectory() as temp_dir:
        # Set training directory to temp location
        deadband_room.training_output_dir = temp_dir
        
        # Fill directory to simulate disk pressure
        large_file = os.path.join(temp_dir, "large.bin")
        with open(large_file, 'wb') as f:
            f.write(b'0' * (40 * 1024 * 1024))  # 40MB file
        
        result = await deadband_room.execute_training_cycle(agent)
        
        # Should either throttle or reject, not crash
        assert result["status"] in ["throttled", "deferred", "failed"]
        assert "resource" in result.get("message", "").lower()
    
    # Restore original
    monkeypatch.setattr(psutil, 'virtual_memory', original_virtual_memory)
```
