def test_concurrent_tile_submission(self):
    """Test that DeadbandRoom handles concurrent tile submissions correctly."""
    room = DeadbandRoom()
    num_agents = 10
    tiles = []
    
    # Create tiles from multiple agents simultaneously
    for i in range(num_agents):
        tile = {
            "agent_id": f"agent_{i}",
            "phase": "P0",
            "content": f"Negative space mapping {i}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles concurrently using threads
    results = []
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append((tile["agent_id"], result))
        except Exception as e:
            results.append((tile["agent_id"], str(e)))
    
    threads = []
    for tile in tiles:
        t = threading.Thread(target=submit_tile, args=(tile,))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    assert len(results) == num_agents
    # Check no data corruption occurred
    for agent_id, result in results:
        assert "error" not in str(result).lower() or "deadband" in str(result)
```

def test_protocol_violation_correction(self):
    """Test agent violating P0→P2 skip, then correcting to P1."""
    room = DeadbandRoom()
    agent_id = "violation_test_agent"
    
    # P0: Normal start
    p0_tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Mapping negative space: avoid direct optimization",
        "timestamp": time.time()
    }
    p0_result = room.submit_tile(p0_tile)
    assert p0_result.get("status") == "accepted"
    
    # VIOLATION: Try to skip to P2 (should fail or be corrected)
    p2_violation_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Trying to optimize without safe channel",
        "timestamp": time.time()
    }
    p2_result = room.submit_tile(p2_violation_tile)
    
    # Room should either reject or auto-correct to P1
    assert p2_result.get("status") in ["rejected", "corrected"]
    if p2_result.get("status") == "corrected":
        assert p2_result.get("corrected_phase") == "P1"
    
    # CORRECTION: Submit proper P1 tile
    p1_tile = {
        "agent_id": agent_id,
        "phase": "P1",
        "content": "Finding safe channels after violation",
        "timestamp": time.time()
    }
    p1_result = room.submit_tile(p1_tile)
    assert p1_result.get("status") == "accepted"
    
    # Now P2 should work
    p2_tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Optimizing within safe channel",
        "timestamp": time.time()
    }
    p2_result = room.submit_tile(p2_tile)
    assert p2_result.get("status") == "accepted"
```

### Test 4: Resource Exhaustion Handling
```python
def test_resource_exhaustion_handling(self):
    """Test room behavior under memory/CPU pressure."""
    room = DeadbandRoom()
    
    # Simulate memory pressure by submitting extremely large tiles
    large_content = "X" * (1024 * 1024 * 10)  # 10MB content
    
    large_tile = {
        "agent_id": "memory_test",
        "phase": "P0",
        "content": large_content,
        "timestamp": time.time()
    }
    
    try:
        result = room.submit_tile(large_tile)
        # Room should either process or reject gracefully
        assert result.get("status") in ["accepted", "rejected", "error"]
        if result.get("status") == "error":
            assert "memory" in result.get("message", "").lower() or \
                   "size" in result.get("message", "").lower()
    except MemoryError:
        print("✓ Room raised MemoryError for oversized tile (expected)")
    except Exception as e:
        # Room should handle gracefully, not crash
        assert "memory" in str(e).lower() or "resource" in str(e).lower()
    
    # Test CPU exhaustion with rapid submissions
    start_time = time.time()
    submission_count = 0
    max_submissions = 1000
    
    while time.time() - start_time < 2.0 and submission_count < max_submissions:
        tile = {
            "agent_id": f"stress_agent_{submission_count}",
            "phase": "P0",
            "content": f"Stress test content {submission_count}",
            "timestamp": time.time()
        }
        try:
            room.submit_tile(tile)
            submission_count += 1
        except Exception:
            # Room throttling is acceptable
            break
    
    print(f"✓ Room handled {submission_count} rapid submissions")
    assert submission_count > 0  # Should process at least some
```
