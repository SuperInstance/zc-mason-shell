### Test 1: Concurrent Tile Processing Stress Test
```python
def test_concurrent_tile_processing(self):
    """Test DeadbandRoom handling of simultaneous tiles from multiple agents."""
    room = DeadbandRoom()
    
    # Simulate 10 agents submitting tiles simultaneously
    agent_tiles = []
    for i in range(10):
        tile = {
            'agent_id': f'agent_{i}',
            'protocol_phase': 'P0',
            'negative_space': [f'constraint_{i}'],
            'timestamp': time.time()
        }
        agent_tiles.append(tile)
    
    # Process tiles concurrently (simulated)
    results = []
    for tile in agent_tiles:
        result = room.process_tile(tile)
        results.append(result)
    
    # Verify all tiles processed without state corruption
    assert len(results) == 10
    assert len(set([r['room_state_hash'] for r in results])) == 1  # Consistent state
    assert room.protocol_state['P0_map']['total_constraints'] == 10
```

### Test 2: Protocol State Persistence After Reset
```python
def test_state_persistence_across_reset(self):
    """Test that critical P0 knowledge persists through room resets."""
    room = DeadbandRoom()
    
    # Build substantial P0 map
    for i in range(50):
        tile = {
            'protocol_phase': 'P0',
            'negative_space': [f'rock_{i}', f'hazard_{i}'],
            'lesson_priority': 'high' if i % 5 == 0 else 'medium'
        }
        room.process_tile(tile)
    
    initial_p0_count = len(room.protocol_state['P0_map']['constraints'])
    initial_ghost_tiles = len(room.ghost_tiles)
    
    # Simulate room reset (maintenance, crash recovery)
    room.reset_room_state(preserve_ghosts=True)
    
    # Verify P0 constraints preserved as ghost tiles
    assert len(room.ghost_tiles) == initial_ghost_tiles
    assert 'rock_0' in room.ghost_tiles[0]['negative_space']
    assert room.protocol_state['P0_map']['constraints'] == []  # Fresh state
    
    # Verify new tiles can be processed normally
    new_tile = {'protocol_phase': 'P0', 'negative_space': ['new_hazard']}
    result = room.process_tile(new_tile)
    assert result['status'] == 'processed'
```

### Test 3: Resource Exhaustion Boundary Test

def test_memory_exhaustion_handling(self):
    """Test room behavior when system resources are constrained."""
    room = DeadbandRoom()
    
    # Simulate memory pressure by creating excessively large tiles
    oversized_tile = {
        'protocol_phase': 'P0',
        'negative_space': ['x' * 10**6] * 100,  # ~100MB of data
        'metadata': {'large_array': list(range(10**5))}
    }
    
    # Room should reject or truncate oversized tiles
    result = room.process_tile(oversized_tile)
    
    # Acceptable outcomes: rejection, truncation, or error with graceful fallback
    assert result['status'] in ['rejected', 'truncated', 'error']
    if result['status'] == 'error':
        assert 'memory' in result['message'].lower() or 'size' in result['message'].lower()
    
    # Verify room remains operational for normal tiles
    normal_tile = {'protocol_phase': 'P1', 'safe_channels': ['test_channel']}
    normal_result = room.process_tile(normal_tile)
    assert normal_result['status'] == 'processed'
```

def test_partial_network_isolation(self):
    """Test room behavior when agent connectivity is intermittent."""
    room = DeadbandRoom()
    
    # Simulate network partition: agent can send but not receive
    partition_tile = {
        'protocol_phase': 'P2',
        'optimization_path': ['a', 'b', 'c'],
        'network_status': 'send_only',
        'timestamp': time.time() - 30  # Stale tile
    }
    
    result = room.process_tile(partition_tile)
    
    # Room should process tile but flag network issues
    assert result['status'] == 'processed_with_warnings'
    assert 'network' in result.get('warnings', '').lower()
    
    # Subsequent tiles should trigger partition detection
    room.detect_network_partitions()
    assert room.network_status['partitioned_agents'] >= 1
    
    # Recovery simulation
    recovery_tile = {
        'protocol_phase': 'P2',
        'optimization_path': ['d', 'e', 'f'],
        'network_status': 'healthy',
        'timestamp': time.time()  # Fresh tile
    }
    recovery_result = room.process_tile(recovery_tile)
    assert recovery_result['status'] == 'processed'
```

### Test 5: Contradictory P0 Constraint Resolution
```python
def test_contradictory_p0_constraints(self):
    """Test room handling of conflicting negative space definitions."""
    room = DeadbandRoom()
    
    # Agent A defines a constraint
    tile_a = {
        'agent_id': 'agent_a',
        'protocol_phase': 'P0',
        'negative_space': ['avoid_region_x'],
        'confidence': 0.9,
        'evidence': ['sensor_data_1']
    }
    room.process_tile(tile_a)
    
    # Agent B defines the opposite constraint
    tile_b = {
        'agent_id': 'agent_b', 
        'protocol_phase': 'P0',
        'negative_space': ['seek_region_x'],  # Direct contradiction
        'confidence': 0.8,
        'evidence': ['sensor_data_2']
    }
    result_b = room.process_tile(tile_b)
    
    # Room should detect contradiction and apply resolution logic
    assert 'contradiction_detected' in result_b.get('flags', [])
    
    # Verify resolution strategy (highest confidence, most evidence, or merge)
    p0_map = room.protocol_state['P0_map']
    if 'avoid_region_x' in p0_map['constraints']:
        assert p0_map['constraints']['avoid_region_x']['confidence'] == 0.9
        assert 'contradictory' in p0_map['constraints']['avoid_region_x']['flags']
    elif 'seek_region_x' in p0_map['constraints']:
        assert p0_map['constraints']['seek_region_x']['confidence'] == 0.8
        assert 'contradictory' in p0_map['constraints']['seek_region_x']['flags']
    
    # Room should log contradiction for review
    assert len(room.contradiction_log) >= 1
```
