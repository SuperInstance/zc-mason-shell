### Test 1: Concurrent Agent Protocol Adherence
```python
def test_concurrent_agents_deadband_priority():
    """Test that multiple agents maintain P0>P1>P2 priority under concurrent load."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent P0 violations from different agents
    violations = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Negative space mapping from {agent_id}",
            "timestamp": time.time()
        }
        violations.append(room.submit_tile(tile))
    
    # Verify all P0 violations are recorded before any P1/P2 processing
    room_state = room.get_state()
    assert len(room_state["p0_violations"]) == len(agent_ids)
    assert room_state["current_phase"] == "P0"  # Should not advance while P0 exists
```

### Test 2: Protocol Violation Recovery
```python
def test_p2_before_p1_recovery():
    """Test room recovery when agent incorrectly submits P2 before P1."""
    room = DeadbandRoom()
    
    # Agent incorrectly tries to optimize (P2) without finding safe channels (P1)
    invalid_tile = {
        "agent_id": "eager_agent",
        "phase": "P2",
        "content": "Trying to optimize without safe channel",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(invalid_tile)
    assert result["status"] == "rejected"
    assert "P1 required before P2" in result["message"]
    
    # Verify room allows correction with proper P1 tile
    valid_tile = {
        "agent_id": "eager_agent",
        "phase": "P1",
        "content": "Finding safe channels after rejection",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(valid_tile)
    assert result["status"] == "accepted"
```

### Test 3: Resource Exhaustion Handling

```python
def test_tile_storage_limits():
    """Test behavior when tile storage approaches/exceeds limits."""
    room = DeadbandRoom(max_tiles=100)
    
    # Fill room to capacity
    for i in range(100):
        tile = {
            "agent_id": f"flood_agent",
            "phase": "P0",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        room.submit_tile(tile)
    
    # Attempt to exceed capacity
    excess_tile = {
        "agent_id": "late_agent",
        "phase": "P1",
        "content": "Should trigger cleanup",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(excess_tile)
    assert result["status"] == "accepted"
    
    # Verify oldest tiles were removed to make space
    room_state = room.get_state()
    assert len(room_state["tiles"]) <= 100
    assert room_state["tiles_evicted"] > 0
```

### Test 4: Malformed Tile Payload Resilience
```python
def test_malformed_tile_handling():
    """Test room resilience against various malformed tile structures."""
    room = DeadbandRoom()
    
    test_cases = [
        {"agent_id": "agent1"},  # Missing required fields
        {"phase": "P0", "content": "test"},  # Missing agent_id
        {"agent_id": None, "phase": "P0", "content": "test"},  # Null agent_id
        {"agent_id": "agent1", "phase": "P99", "content": "test"},  # Invalid phase
        {"agent_id": "agent1", "phase": "P0", "content": "", "timestamp": -1},  # Invalid timestamp
        {},  # Empty tile
    ]
    
    for malformed_tile in test_cases:
        result = room.submit_tile(malformed_tile)
        assert result["status"] == "rejected"
        assert "validation" in result["message"].lower() or "invalid" in result["message"].lower()
    
    # Verify room state remains stable
    room_state = room.get_state()
    assert room_state["error_count"] == len(test_cases)
    assert room_state["current_phase"] == "P0"  # Default state maintained
```

### Test 5: Network Partition Simulation
```python
def test_partial_state_synchronization():
    """Test room behavior during simulated network partitions."""
    room = DeadbandRoom()
    
    # Simulate partition: agent can read but not write
    room._simulate_partition = True
    
    read_result = room.get_state()
    assert read_result is not None  # Read operations succeed
    
    # Write operation should fail or queue
    tile = {
        "agent_id": "partitioned_agent",
        "phase": "P0",
        "content": "Tile during partition",
        "timestamp": time.time()
    }
    
    write_result = room.submit_tile(tile)
    assert write_result["status"] in ["queued", "failed"]
    
    # End partition simulation
    room._simulate_partition = False
    
    # Verify queued tiles are processed
    if write_result["status"] == "queued":
        room._process_queued_tiles()
        room_state = room.get_state()
        assert any(t["agent_id"] == "partitioned_agent" for t in room_state["tiles"])
```
