### 1. Test: Concurrent Tile Submission from Multiple Agents
**Scenario:** Simulate two agents submitting tiles simultaneously to the same room.
**Edge:** Race conditions in room state updates.
```python
def test_concurrent_tile_submission(room, sample_tile):
    """Multiple agents submitting tiles concurrently should not corrupt room state."""
    import threading
    
    results = []
    def submit_tile(agent_id):
        tile = sample_tile.copy()
        tile["agent_id"] = agent_id
        tile["content"] = f"Test from agent {agent_id}"
        results.append(room.submit_tile(tile))
    
    threads = [
        threading.Thread(target=submit_tile, args=("agent_1",)),
        threading.Thread(target=submit_tile, args=("agent_2",))
    ]
    
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Both submissions should succeed
    assert all(results), "Concurrent submissions failed"
    assert len(room.tiles) == 2, "Room should have both tiles"
    assert room.tiles[0]["agent_id"] != room.tiles[1]["agent_id"], "Tiles from different agents preserved"
```

### 2. Test: Extremely Large Tile Content
**Scenario:** Submit a tile with content at size limit boundary.
**Edge:** Memory/processing limits, truncation behavior.
```python
def test_large_tile_content(room):
    """Tile with maximum allowed content size should be accepted."""
    max_size = room.config.get("max_tile_size", 10_000)
    
    # Create content at exact limit
    large_content = "x" * max_size
    tile = {
        "agent_id": "test_agent",
        "cycle": 999,
        "content": large_content,
        "protocol_phase": "P1"
    }
    
    result = room.submit_tile(tile)
    assert result is True, "Max-size tile should be accepted"
    assert room.tiles[-1]["content"] == large_content, "Content should not be truncated"
    
    # One byte over should fail
    tile["content"] = "x" * (max_size + 1)
    result = room.submit_tile(tile)
    assert result is False, "Oversized tile should be rejected"
```

### 3. Test: Protocol Phase Skipping Detection
**Scenario:** Agent attempts to submit P2 tile without completing P0 and P1.
**Edge:** Protocol enforcement across multiple submissions.
```python
def test_protocol_phase_skipping_detection(room):
    """Room should detect and reject attempts to skip protocol phases."""
    # First tile must be P0
    tile_p0 = {
        "agent_id": "test_agent",
        "cycle": 1,
        "content": "P0: Avoid invalid paths",
        "protocol_phase": "P0"
    }
    assert room.submit_tile(tile_p0), "P0 tile should be accepted"
    
    # Attempt to submit P2 directly (skipping P1)
    tile_p2 = {
        "agent_id": "test_agent",
        "cycle": 2,
        "content": "P2: Optimize path",
        "protocol_phase": "P2"
    }
    result = room.submit_tile(tile_p2)
    assert result is False, "P2 tile should be rejected when P1 not completed"
    
    # Submit P1 properly
    tile_p1 = {
        "agent_id": "test_agent",
        "cycle": 3,
        "content": "P1: Find safe channels",
        "protocol_phase": "P1"
    }
    assert room.submit_tile(tile_p1), "P1 tile should be accepted after P0"
    
    # Now P2 should work
    tile_p2["cycle"] = 4
    assert room.submit_tile(tile_p2), "P2 tile should be accepted after P1"
```

```python
def test_partial_validation_failure(room, mocker):
    """Tile that passes format check but fails content validation should be rejected cleanly."""
    # Mock validate_tile to simulate content validation failure
    mocker.patch.object(room.validator, 'validate_tile', 
                       side_effect=[True, False])  # Format OK, content invalid

        "agent_id": "test_agent",
        "cycle": 100,
        "content": "Seems valid but has hidden issue",
        "protocol_phase": "P0"

    assert result is False, "Tile failing content validation should be rejected"
    assert len(room.tiles) == 0, "Invalid tile should not be stored"

```

### 5. Test: Room State Recovery After Crash
**Scenario:** Simulate room crash and recovery from persisted state.
**Edge:** State persistence and recovery mechanisms.
```python
def test_room_state_recovery(room, tmp_path):
    """Room should recover full state from persistence layer after restart."""
    # Submit several tiles
    for i in range(5):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": i,
            "content": f"Test tile {i}",
            "protocol_phase": "P0" if i == 0 else "P1" if i == 1 else "P2"
        }
        room.submit_tile(tile)
    
    original_state = {
        "tile_count": len(room.tiles),
        "last_cycle": room.last_cycle,
        "protocol_state": room.protocol_state.copy()
    }
    
    # Simulate crash and recovery
    persistence_file = tmp_path / "room_state.json"
    room._persist_state(persistence_file)
    
    # Create new room instance with persisted state
    recovered_room = DeadbandRoom(config=room.config)
    recovered_room._load_state(persistence_file)
    
    # Verify full recovery
    assert len(recovered_room.tiles) == original_state["tile_count"]
    assert recovered_room.last_cycle == original_state["last_cycle"]
    assert recovered_room.protocol_state == original_state["protocol_state"]
    
    # Verify tiles maintain integrity
    for i, tile in enumerate(recovered_room.tiles):
        assert tile["agent_id"] == f"agent_{i}"
        assert tile["content"] == f"Test tile {i}"
```
