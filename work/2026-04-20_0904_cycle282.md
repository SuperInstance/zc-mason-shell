### Test 1: Concurrent Tile Submission Race Condition

```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom(room_id="race_test")
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "priority": "P1",
            "content": f"Test from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles (simulating concurrent access)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    # Verify all submissions were processed
    self.assertEqual(len(results), 5)
    self.assertEqual(room.get_tile_count(), 5)
    
    # Verify no duplicate processing or lost tiles
    unique_agents = set(tile["agent_id"] for tile in tiles)
    processed_agents = set(room.get_agent_ids())
    self.assertEqual(unique_agents, processed_agents)
```

### Test 2: Network Partition During Protocol Phase
```python
def test_network_partition_recovery(self):
    """Room should handle partial write failures and maintain consistency."""
    room = DeadbandRoom(room_id="partition_test")
    
    # Simulate successful P0 tile
    p0_tile = {
        "agent_id": "agent_1",
        "priority": "P0",
        "content": "Negative space mapped",
        "timestamp": time.time()
    }
    room.submit_tile(p0_tile)
    
    # Simulate network partition during P1 submission
    original_save = room._save_state
    partition_triggered = False
    
    def failing_save():
        nonlocal partition_triggered
        if not partition_triggered:
            partition_triggered = True
            raise IOError("Network partition: write failed")
        return original_save()
    
    room._save_state = failing_save
    
    p1_tile = {
        "agent_id": "agent_1",
        "priority": "P1",
        "content": "Finding safe channels",
        "timestamp": time.time()
    }
    
    # Should raise error on first attempt
    with self.assertRaises(IOError):
        room.submit_tile(p1_tile)
    
    # Restore normal operation
    room._save_state = original_save
    
    # Resubmit should work
    result = room.submit_tile(p1_tile)
    self.assertTrue(result["success"])
    
    # Verify state consistency
    self.assertEqual(room.get_current_phase("agent_1"), "P1")
```

### Test 3: Malformed Tile Payload Injection
```python
def test_malformed_tile_payloads(self):
    """Room should reject tiles with schema violations or injection attempts."""
    room = DeadbandRoom(room_id="injection_test")
    
    test_cases = [
        # SQL injection attempt
        {
            "agent_id": "agent'; DROP TABLE tiles; --",
            "priority": "P0",
            "content": "Test",
            "timestamp": time.time()
        },
        # Missing required field
        {
            "priority": "P0",
            "content": "Test"
        },
        # Invalid priority value
        {
            "agent_id": "agent_1",
            "priority": "P99",
            "content": "Test",
            "timestamp": time.time()
        },
        # Extremely large content (potential DoS)
        {
            "agent_id": "agent_1",
            "priority": "P0",
            "content": "X" * 10_000_000,
            "timestamp": time.time()
        },
        # Nested malicious objects
        {
            "agent_id": "agent_1",
            "priority": "P0",
            "content": {"__class__": "Exploit"},
            "timestamp": time.time()
        }
    ]
    
    for i, tile in enumerate(test_cases):
        with self.subTest(f"malformed_tile_{i}"):
            with self.assertRaises((ValueError, TypeError, ValidationError)):
                room.submit_tile(tile)
    
    # Verify room remains clean
    self.assertEqual(room.get_tile_count(), 0)
```

### Test 4: Resource Exhaustion During Training
```python
def test_resource_exhaustion_handling(self):
    """Room should handle memory/disk limits gracefully during training cycles."""
    room = DeadbandRoom(room_id="resource_test")
    
    # Fill room with tiles up to configured limit
    max_tiles = room.config.get("max_tiles_per_room", 1000)
    
    for i in range(max_tiles + 10):  # Attempt to exceed limit
        tile = {
            "agent_id": f"agent_{i % 5}",
            "priority": "P1",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        
        if i < max_tiles:
            # Should succeed within limits
            result = room.submit_tile(tile)
            self.assertTrue(result["success"])
        else:
            # Should reject when at capacity
            with self.assertRaises(ResourceWarning):
                room.submit_tile(tile)
    
    # Verify room at capacity but functional
    self.assertEqual(room.get_tile_count(), max_tiles)
    
    # Training should still work on existing tiles
    training_result = room.run_training_cycle()
    self.assertIn("tiles_processed", training_result)
    self.assertLessEqual(training_result["tiles_processed"], max_tiles)
```
