```python
import threading
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_concurrent_tile_submission():
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom()
    tiles_submitted = []
    
    def submit_tile(agent_id):
        tile = {
            "agent_id": agent_id,
            "content": f"Test from agent {agent_id}",
            "phase": "P0",
            "timestamp": time.time()
        }
        result = room.submit_tile(tile)
        tiles_submitted.append((agent_id, result))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All submissions should be processed
    assert len(tiles_submitted) == 5
    # Room tile count should match
    assert room.get_tile_count() == 5
    # No duplicate agent IDs in accepted tiles
    accepted_agents = [agent_id for agent_id, result in tiles_submitted if result["status"] == "accepted"]
    assert len(set(accepted_agents)) == len(accepted_agents)
```

### Test 4: Protocol Violation Sequences

```python
def test_protocol_violation_sequences():
    """Room should enforce Deadband Protocol priority rules."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    
    # Attempt to submit P2 without P0 or P1
    p2_tile = {
        "agent_id": agent_id,
        "content": "Jumping straight to optimization",
        "phase": "P2",
        "timestamp": time.time()
    }
    p2_result = room.submit_tile(p2_tile)
    assert p2_result["status"] == "rejected", "Should reject P2 without prior phases"
    assert "P0" in p2_result.get("error", ""), "Error should mention missing P0"
    
    # Submit valid P0
    p0_tile = {
        "agent_id": agent_id,
        "content": "Negative space mapped",
        "phase": "P0",
        "timestamp": time.time()
    }
    p0_result = room.submit_tile(p0_tile)
    assert p0_result["status"] == "accepted", "Should accept valid P0"
    
    # Attempt to submit another P0 (should be allowed but might trigger warning)
    p0_again = {
        "agent_id": agent_id,
        "content": "Additional negative space",
        "phase": "P0",
        "timestamp": time.time() + 1
    }
    p0_again_result = room.submit_tile(p0_again)
    # Either accept or reject with "move to P1" suggestion
    assert p0_again_result["status"] in ["accepted", "rejected"]
```

```python
def test_room_capacity_limits():
    """Room should handle hitting maximum tiles per cycle."""
    room = DeadbandRoom()
    max_tiles = room.MAX_TILES_PER_CYCLE  # Assuming this attribute exists
    
    # Fill room to capacity
    for i in range(max_tiles):
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"Tile {i}",
            "phase": "P0",
            "timestamp": time.time() + i
        }
        result = room.submit_tile(tile)
        assert result["status"] == "accepted", f"Should accept tile {i}"
    
    # Attempt one more tile beyond capacity
    overflow_tile = {
        "agent_id": "overflow_agent",
        "content": "This should be rejected",
        "phase": "P1",
        "timestamp": time.time()
    }
    overflow_result = room.submit_tile(overflow_tile)
    assert overflow_result["status"] == "rejected", "Should reject beyond capacity"
    assert "capacity" in overflow_result.get("error", "").lower() or "full" in overflow_result.get("error", "").lower()
    
    # Reset and verify new tiles can be submitted
    room.reset()
    assert room.get_tile_count() == 0
    new_tile = {
        "agent_id": "new_agent",
        "content": "After reset",
        "phase": "P0",
        "timestamp": time.time()
    }
    new_result = room.submit_tile(new_tile)
    assert new_result["status"] == "accepted", "Should accept after reset"
```
