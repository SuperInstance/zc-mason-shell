### Test 5: Concurrent Deadband Application (P2 Optimization Edge Case)
```python
def test_concurrent_deadband_application(self):
    """Test that DeadbandRoom can handle rapid, concurrent apply_deadband calls."""
    import threading
    
    room = DeadbandRoom(deadband=0.3)
    results = []
    lock = threading.Lock()
    
    def apply_deadband_thread(input_val, thread_id):
        """Thread function to apply deadband concurrently."""
        input_tensor = torch.tensor([input_val])
        output = room.apply_deadband(input_tensor)
        with lock:
            results.append((thread_id, input_val, output.item()))
    
    # Create multiple threads with different inputs
    threads = []
    test_inputs = [-0.8, -0.2, 0.0, 0.2, 0.8]
    
    for i, val in enumerate(test_inputs):
        thread = threading.Thread(target=apply_deadband_thread, args=(val, i))
        threads.append(thread)
        thread.start()
    
    # Wait for all threads to complete
    for thread in threads:
        thread.join()
    
    # Verify all threads completed and results are correct
    assert len(results) == len(test_inputs)
    
    # Sort by thread_id for consistent verification
    results.sort(key=lambda x: x[0])
    
    # Check each result follows deadband logic
    for (thread_id, input_val, output_val) in results:
        if abs(input_val) <= 0.3:  # Within deadband
            assert abs(output_val) < 1e-6  # Should be approximately zero
        else:  # Outside deadband
            assert abs(output_val - input_val) < 1e-6  # Should be unchanged
```
