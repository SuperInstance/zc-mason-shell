### 1. Test: Concurrent Tile Submission from Multiple Agents
**Scenario:** Simulate race condition when multiple agents submit tiles simultaneously
**Purpose:** Verify thread safety and tile queue integrity
```python
def test_concurrent_tile_submission(deadband_room):
    """Test that concurrent submissions don't corrupt tile storage."""
    import threading
    
    tiles_submitted = []
    def submit_tile(agent_id):
        tile = {"agent_id": agent_id, "content": f"test from {agent_id}", "priority": "P1"}
        result = deadband_room.submit_tile(tile)
        tiles_submitted.append((agent_id, result))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all tiles were accepted
    assert len(tiles_submitted) == 5
    for agent_id, result in tiles_submitted:
        assert result["status"] == "accepted", f"Tile from {agent_id} rejected"
    
    # Verify tile count matches
    assert deadband_room.get_tile_count() >= 5
```

### 2. Test: Priority Rollback After P2 Violation
**Scenario:** Agent attempts P2 optimization, fails validation, must roll back to P1
**Purpose:** Test protocol enforcement and state recovery
```python
def test_priority_rollback_after_p2_violation(deadband_room):
    """Test that agent reverts to P1 after invalid P2 attempt."""
    agent_id = "test_agent_rollback"
    
    # Start with P0 (negative space mapping)
    deadband_room.set_agent_priority(agent_id, "P0")
    assert deadband_room.get_agent_priority(agent_id) == "P0"
    
    # Skip P1 and attempt P2 (should fail)
    with pytest.raises(ValueError, match="Cannot skip P1"):
        deadband_room.set_agent_priority(agent_id, "P2")
    
    # Agent should remain at P0
    assert deadband_room.get_agent_priority(agent_id) == "P0"
    
    # Proper progression: P0 → P1 → P2
    deadband_room.set_agent_priority(agent_id, "P1")
    assert deadband_room.get_agent_priority(agent_id) == "P1"
    
    deadband_room.set_agent_priority(agent_id, "P2")
    assert deadband_room.get_agent_priority(agent_id) == "P2"
```

### 4. Test: Agent State Persistence Across Room Resets
**Scenario:** Room reset should preserve completed tiles but reset agent states
**Purpose:** Verify data persistence boundaries
```python
def test_agent_state_persistence_after_reset(deadband_room):
    """Test what persists and what resets after room reset."""
    agent_id = "test_agent_persistence"
    
    # Set up initial state
    deadband_room.set_agent_priority(agent_id, "P1")
    deadband_room.submit_tile({
        "agent_id": agent_id,
        "content": "persistent tile",
        "priority": "P1"
    })
    
    initial_tile_count = deadband_room.get_tile_count()
    initial_priority = deadband_room.get_agent_priority(agent_id)
    
    # Reset room
    deadband_room.reset()
    
    # Tiles should persist (training data)
    assert deadband_room.get_tile_count() == initial_tile_count
    
    # Agent state should reset to default
    assert deadband_room.get_agent_priority(agent_id) != initial_priority
    assert deadband_room.get_agent_priority(agent_id) == "P0"  # Default
    
    # New tile submission should work
    result = deadband_room.submit_tile({
        "agent_id": agent_id,
        "content": "post-reset tile",
        "priority": "P0"
    })
    assert result["status"] == "accepted"
    assert deadband_room.get_tile_count() == initial_tile_count + 1
```

```python
def test_maximum_tile_size_enforcement(deadband_room):
    """Test that oversized tiles are rejected."""
    # Get current max size from room config
    max_size = deadband_room.config.get("max_tile_size", 1024 * 1024)  # Default 1MB
    
    # Create tile at limit
    content_at_limit = "x" * (max_size - 100)  # Leave room for JSON structure
    tile_at_limit = {
        "agent_id": "test_agent",
        "content": content_at_limit,
        "priority": "P1"
    }
    result1 = deadband_room.submit_tile(tile_at_limit)
    assert result1["status"] == "accepted"
    
    # Create tile exceeding limit
    content_exceeding = "x" * (max_size + 100)
    tile_exceeding = {
        "agent_id": "test_agent",
        "content": content_exceeding,
        "priority": "P1"
    }
    result2 = deadband_room.submit_tile(tile_exceeding)
    assert result2["status"] == "rejected"
    assert "size" in result2.get("error", "").lower() or "limit" in result2.get("error", "").lower()
    
    # Test with binary data
    binary_data = b"\x00" * (max_size + 50)
    tile_binary = {
        "agent_id": "test_agent",
        "content": binary_data,
        "priority": "P1"
    }
    result3 = deadband_room.submit_tile(tile_binary)
    assert result3["status"] == "rejected"
```
