### Test 1: Concurrent Tile Processing
```python
def test_concurrent_tile_processing(self):
    """Test that DeadbandRoom handles simultaneous tile submissions correctly."""
    room = DeadbandRoom(config=self.valid_config)
    
    # Create 10 identical valid tiles
    tiles = [self.create_valid_tile() for _ in range(10)]
    
    # Process all tiles concurrently (simulated with threads)
    import threading
    results = []
    
    def process_tile(tile):
        result = room.process_tile(tile)
        results.append(result)
    
    threads = [threading.Thread(target=process_tile, args=(tile,)) for tile in tiles]
    for thread in threads:
        thread.start()
    for thread in threads:
        thread.join()
    
    # All tiles should be processed without crashing
    assert len(results) == 10
    # State should be consistent (no race conditions corrupted room state)
    assert room.state.current_phase in ['P0', 'P1', 'P2']
```

### Test 5: Protocol Phase Regression Detection
```python
def test_phase_regression_detection(self):
    """Test that room detects and handles unexpected phase regression."""
    room = DeadbandRoom(config=self.valid_config)
    
    # Force room into P2 (optimization phase)
    room.state.current_phase = 'P2'
    room.state.safe_channels = ['channel_a', 'channel_b']
    
    # Create a tile that would normally be P0 violation
    # but submitted when room thinks it's in P2
    regression_tile = self.create_tile_with_p0_violation()
    
    result = room.process_tile(regression_tile)
    
    # Room should either:
    # 1. Detect regression and reset to P0
    # 2. Reject tile with regression warning
    # 3. Log the anomaly
    
    assert result['status'] == 'rejected'
    # Check for regression indicator in response
    assert any(keyword in result.get('reason', '').lower() 
               for keyword in ['regression', 'phase', 'p0', 'violation'])
    
    # Verify room state updated appropriately
    if 'regression' in result.get('reason', '').lower():
        assert room.state.current_phase == 'P0'
```
