```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles in same cycle should be handled safely."""
    room = DeadbandRoom(max_tiles_per_cycle=100)
    agent_ids = [f"agent_{i}" for i in range(10)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "cycle": 1,
            "content": f"Test from {agent_id}",
            "protocol_phase": "P1"
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    assert len(room.tile_buffer) == 10
```

### Test 2: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Agent skipping P1 to P2 should be rejected, then succeed after correction."""
    room = DeadbandRoom()
    agent_id = "test_agent"
    
    # Attempt P2 without completing P1 (violation)
    tile_p2 = {
        "agent_id": agent_id,
        "cycle": 1,
        "content": "Optimizing without safe channel",
        "protocol_phase": "P2"
    }
    result_p2 = room.submit_tile(tile_p2)
    assert result_p2["status"] == "rejected"
    assert "P1 not completed" in result_p2.get("reason", "")
    
    # Correctly submit P1
    tile_p1 = {
        "agent_id": agent_id,
        "cycle": 1,
        "content": "Finding safe channels",
        "protocol_phase": "P1"
    }
    result_p1 = room.submit_tile(tile_p1)
    assert result_p1["status"] == "accepted"
    
    # Now P2 should succeed
    tile_p2["cycle"] = 2  # Advance cycle
    result_p2_corrected = room.submit_tile(tile_p2)
    assert result_p2_corrected["status"] == "accepted"
```

### Test 3: Resource Exhaustion Handling
```python
def test_tile_storage_limits(self):
    """Room should handle hitting tile storage limits gracefully."""
    room = DeadbandRoom(max_total_tiles=5)
    
    # Fill room to capacity
    for i in range(5):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": 1,
            "content": f"Tile {i}",
            "protocol_phase": "P0"
        }
        result = room.submit_tile(tile)
        assert result["status"] == "accepted"
    
    # Attempt to exceed capacity
    overflow_tile = {
        "agent_id": "overflow_agent",
        "cycle": 1,
        "content": "This should be rejected",
        "protocol_phase": "P1"
    }
    result = room.submit_tile(overflow_tile)
    assert result["status"] == "rejected"
    assert "capacity" in result.get("reason", "").lower()
    
    # Verify existing tiles remain intact
    assert len(room.tile_buffer) == 5
```

```python
def test_partial_connectivity_recovery(self):
    """Room should handle agents with intermittent connectivity."""
    room = DeadbandRoom()
    agent_id = "flaky_agent"
    
    # Agent submits P0 successfully
    tile_p0 = {
        "agent_id": agent_id,
        "cycle": 1,
        "content": "Mapping negative space",
        "protocol_phase": "P0"
    }
    result_p0 = room.submit_tile(tile_p0)
    assert result_p0["status"] == "accepted"
    
    # Simulate network partition - agent disappears for 3 cycles
    # Room should maintain agent's last known state
    
    # Agent reconnects and attempts to continue from last phase
    tile_p1 = {
        "agent_id": agent_id,
        "cycle": 5,  # Jump ahead due to partition
        "content": "Resuming after partition",
        "protocol_phase": "P1"
    }
    result_p1 = room.submit_tile(tile_p1)
    
    # Should be accepted if protocol allows resume
    # Implementation-specific: check if room tracks agent state
    if hasattr(room, 'agent_states'):
        assert result_p1["status"] == "accepted"
    else:
        # Without state tracking, might be rejected
        assert result_p1["status"] in ["accepted", "rejected"]
```

### Test 5: Malformed Tile Payloads
```python
def test_malformed_tile_handling(self):
    """Room should reject various malformed tile payloads safely."""
    room = DeadbandRoom()
    
    test_cases = [
        # (tile_payload, expected_result, description)
        (
            {"agent_id": "", "cycle": 1, "content": "test", "protocol_phase": "P0"},
            "rejected",
            "empty agent_id"
        ),
        (
            {"agent_id": "test", "cycle": -1, "content": "test", "protocol_phase": "P0"},
            "rejected",
            "negative cycle"
        ),
        (
            {"agent_id": "test", "cycle": 1, "content": "", "protocol_phase": "P0"},
            "rejected",
            "empty content"
        ),
        (
            {"agent_id": "test", "cycle": 1, "content": "x" * 10000, "protocol_phase": "P0"},
            "rejected",
            "oversized content"
        ),
        (
            {"agent_id": "test", "cycle": 1, "content": "test", "protocol_phase": "INVALID"},
            "rejected",
            "invalid protocol phase"
        ),
        (
            {"agent_id": "test", "cycle": 1, "content": "test", "protocol_phase": "P0", "extra": "<script>alert(1)</script>"},
            "rejected",
            "injection attempt"
        ),
    ]
    
    for tile, expected_status, description in test_cases:
        result = room.submit_tile(tile)
        assert result["status"] == expected_status, f"Failed: {description}"
        if expected_status == "rejected":
            assert "reason" in result, f"No reason provided for: {description}"
```
