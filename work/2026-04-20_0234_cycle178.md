### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Negative space map from {agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (in rapid succession)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All tiles should be accepted or properly queued
    accepted = sum(1 for r in results if r.get("status") == "accepted")
    self.assertGreater(accepted, 0, "At least some tiles should be accepted")
    
    # Room state should be consistent
    self.assertEqual(len(room.tile_buffer), accepted)
    self.assertEqual(room.current_phase, "P0")
```

### Test 2: Network Partition with Partial State
```python
def test_partial_state_recovery(self):
    """Test room recovery when agents see inconsistent state due to partitions."""
    room = DeadbandRoom()
    
    # Create partition scenario: some tiles accepted, some lost
    valid_tile = {
        "agent_id": "agent_1",
        "phase": "P0",
        "content": "Valid P0 content",
        "timestamp": time.time()
    }
    
    # Simulate partition: room accepts tile but agent doesn't see confirmation
    room.submit_tile(valid_tile)
    
    # Agent re-submits same tile (idempotency check)
    result = room.submit_tile(valid_tile)
    
    # Room should handle duplicate gracefully
    self.assertIn(result.get("status"), ["accepted", "duplicate"])
    
    # Agent requests state sync
    sync_state = room.get_state(for_agent="agent_1")
    
    # State should include the tile and current phase
    self.assertIn("tiles", sync_state)
    self.assertIn("current_phase", sync_state)
    self.assertEqual(sync_state["current_phase"], "P0")
```

### Test 3: Malformed Tile Payload Deep Validation
```python
def test_deep_malformed_payload(self):
    """Test handling of deeply nested malformed tile structures."""
    room = DeadbandRoom()
    
    # Test various malformed payloads
    malformed_tiles = [
        # Circular reference attempt
        {"agent_id": "agent_1", "content": {"self_ref": None}},
        
        # Extremely large payload
        {"agent_id": "agent_2", "content": "X" * 1000000},
        
        # Invalid UTF-8 sequences
        {"agent_id": "agent_3", "content": b"\xff\xfe".decode("latin-1")},
        
        # Missing required fields but extra nonsense
        {"extra_field": "nonsense", "nested": {"deep": {"deeper": {}}}},
        
        # Protocol phase injection attempt
        {"agent_id": "agent_4", "phase": "P2", "content": "Skipping to optimization"}
    ]
    
    for i, tile in enumerate(malformed_tiles):
        try:
            result = room.submit_tile(tile)
            # Should either reject or sanitize
            self.assertNotEqual(result.get("status"), "accepted", 
                              f"Tile {i} should not be accepted as-is")
        except Exception as e:
            # Exceptions are acceptable for truly invalid input
            self.assertIsInstance(e, (ValueError, TypeError))
```

### Test 4: Protocol Violation with Self-Correction
```python
def test_protocol_violation_recovery(self):
    """Test agent skipping phases then self-correcting."""
    room = DeadbandRoom()
    
    # Agent attempts to skip P0 (violation)
    violation_tile = {
        "agent_id": "bad_agent",
        "phase": "P2",
        "content": "Trying to optimize without mapping",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(violation_tile)
    self.assertEqual(result.get("status"), "rejected")
    self.assertIn("protocol", result.get("reason", "").lower())
    
    # Agent receives feedback and corrects
    feedback = room.get_feedback(agent_id="bad_agent")
    self.assertIn("P0", feedback)
    
    # Agent submits proper P0 tile
    correction_tile = {
        "agent_id": "bad_agent",
        "phase": "P0",
        "content": "Mapping negative space as instructed",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(correction_tile)
    self.assertEqual(result.get("status"), "accepted")
    
    # Verify room tracks violation history
    self.assertIn("bad_agent", room.violation_log)
```

### Test 5: Resource Pressure During Training Cycle
```python
def test_resource_exhaustion_resilience(self):
    """Test room behavior under memory/CPU pressure."""
    room = DeadbandRoom()
    
    # Simulate resource pressure by submitting many tiles rapidly
    tile_count = 1000
    accepted_count = 0
    
    for i in range(tile_count):
        tile = {
            "agent_id": f"pressure_agent_{i}",
            "phase": "P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
            "content": f"Tile under pressure {i}",
            "timestamp": time.time()
        }
        
        try:
            result = room.submit_tile(tile)
            if result.get("status") == "accepted":
                accepted_count += 1
                
            # Simulate GC/memory pressure every 100 tiles
            if i % 100 == 0:
                import gc
                gc.collect()
                
        except MemoryError:
            # Room should handle gracefully, not crash
            self.fail("Room crashed under memory pressure")
        except Exception as e:
            # Other exceptions should be logged but not break room
            self.assertIsInstance(e, (ValueError, RuntimeError))
    
    # Room should still be functional
    self.assertTrue(room.is_operational())
    
    # Some tiles should be processed (not necessarily all)
    self.assertGreater(accepted_count, 0)
    self.assertLessEqual(accepted_count, tile_count)
    
    # Room state should be consistent
    self.assertEqual(len(room.tile_buffer), accepted_count)
```
