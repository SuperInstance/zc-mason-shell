### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race():
    """Test that room handles multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    room.reset()
    
    # Simulate 5 agents submitting tiles at same "time"
    tiles = []
    for i in range(5):
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"Test tile {i}",
            "phase": "P1",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles (simulating concurrent access)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should succeed
    assert all(r["status"] == "accepted" for r in results)
    
    # Room should have all 5 tiles
    assert len(room.get_tiles()) == 5
    
    # No tile data corruption
    stored_tiles = room.get_tiles()
    for i, tile in enumerate(tiles):
        assert stored_tiles[i]["agent_id"] == tile["agent_id"]
        assert stored_tiles[i]["content"] == tile["content"]
```

```python
def test_malformed_state_recovery():

    try:
        tiles = room.get_tiles()
        # If we get here, room should have recovered or defaulted

        # Room should raise specific error, not crash

    # Reset should always work even after corruption
    room.reset()
    assert room.get_tiles() == []
    assert room.get_current_phase() == "P0"
```

### Test 3: Network Partition Simulation
```python
def test_network_partition_handling():
    """Test room behavior during simulated network issues."""
    room = DeadbandRoom()
    room.reset()
    
    # Simulate network partition mid-operation
    original_send = room._send_to_plato_server
    
    partition_happened = False
    
    def partitioned_send(data):
        nonlocal partition_happened
        if not partition_happened:
            partition_happened = True
            raise ConnectionError("Network partition simulated")
        return original_send(data)
    
    room._send_to_plato_server = partitioned_send
    
    # First submission should fail due to partition
    tile = {
        "agent_id": "test_agent",
        "content": "Tile during partition",
        "phase": "P1"
    }
    
    try:
        result = room.submit_tile(tile)
        # If submission succeeds, tile should be queued locally
        if result["status"] == "accepted":
            assert room._pending_tiles is not None
            assert len(room._pending_tiles) > 0
    except ConnectionError:
        # Expected during partition
        pass
    
    # Room should remain functional for local operations
    assert room.get_current_phase() == "P1"
    
    # Clear partition
    room._send_to_plato_server = original_send
    
    # Room should recover and sync pending tiles
    if hasattr(room, '_pending_tiles') and room._pending_tiles:
        room._sync_pending_tiles()
        assert len(room._pending_tiles) == 0
```

### Test 4: Memory Pressure and Large Tiles

```python
def test_memory_pressure_large_tiles():
    """Test room handling of memory pressure from large tile submissions."""
    room = DeadbandRoom()
    room.reset()
    
    # Submit increasingly large tiles
    sizes = [1000, 10000, 100000, 1000000]  # bytes
    
    for i, size in enumerate(sizes):
        # Generate tile with large content
        large_content = "x" * size
        tile = {
            "agent_id": f"agent_{i}",
            "content": large_content,
            "phase": "P2",
            "metadata": {"size": size}
        }
        
        try:
            result = room.submit_tile(tile)
            
            # Room should either accept or reject with clear reason
            if result["status"] == "rejected":
                assert "size" in result.get("reason", "").lower() or \
                       "memory" in result.get("reason", "").lower()
            else:
                # If accepted, verify content integrity
                stored = room.get_tiles()[-1]
                assert len(stored["content"]) == size
                
        except MemoryError:
            # Room should handle memory errors gracefully
            # Clear some state if possible
            room._cleanup_old_tiles()
            continue
    
    # Room should still be operational after memory pressure
    assert room.get_current_phase() == "P2"
    
    # Submit normal tile after memory pressure
    normal_tile = {
        "agent_id": "recovery_agent",
        "content": "Normal tile",
        "phase": "P0"
    }
    result = room.submit_tile(normal_tile)
    assert result["status"] == "accepted"
```

def test_protocol_violation_detection():
    """Test room detection and handling of Deadband Protocol violations."""
    room = DeadbandRoom()
    room.reset()
    
    # Violation 1: Skip from P0 directly to P2
    tile_p2_skip = {
        "agent_id": "violating_agent",
        "content": "Trying to optimize without safe channel",
        "phase": "P2",  # Should be P1 first
        "violation": "skip_p1"
    }
    
    result = room.submit_tile(tile_p2_skip)
    
    # Room should detect violation
    if result["status"] == "rejected":
        assert "protocol" in result.get("reason", "").lower() or \
               "sequence" in result.get("reason", "").lower()
    elif result["status"] == "accepted":
        # If accepted, room should have corrected phase
        assert room.get_current_phase() == "P1"  # Auto-corrected to P1
        # And tile should be marked with correction
        tiles = room.get_tiles()
        last_tile = tiles[-1]
        assert last_tile.get("corrected_phase") == "P1"
    
    # Violation 2: Go backwards in protocol (P2 -> P1)
    room.set_phase("P2")  # Manually advance to P2
    
    tile_backwards = {
        "agent_id": "backwards_agent",
        "content": "Trying to go back to P1",
        "phase": "P1"  # Should not go backwards
    }
    
    result = room.submit_tile(tile_backwards)
    
    # Room should reject or correct backwards movement
    if result["status"] == "rejected":
        assert "backwards" in result.get("reason", "").lower() or \
               "regress" in result.get("reason", "").lower()

        # If accepted, phase should remain P2
        assert room.get_current_phase() == "P2"
    
    # Valid sequence: P0 -> P1 -> P2
    room.reset()
    
    phases = ["P0", "P1", "P2"]
    for phase in phases:
        tile = {
            "agent_id": "compliant_agent",
            "content": f"Phase {phase} work",
            "phase": phase
        }
        result = room.submit_tile(tile)
