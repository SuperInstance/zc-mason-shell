### 1. Test Protocol Violation: Skipping Directly to P2
**Scenario:** An agent attempts to execute P2 optimization without completing P0 and P1 mapping.
**Expected:** Room should reject the request and return a P0 violation error.
**Rationale:** Enforces strict priority doctrine from fleet context.

```python
def test_protocol_violation_skip_to_p2():
    """DeadbandRoom should reject P2 execution without prior P0/P1 completion."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from plato_torch.exceptions import ProtocolViolationError
    
    room = DeadbandRoom(config={"max_agents": 5})
    agent_id = room.register_agent(agent_type="zeroclaw")
    
    # Agent tries to execute P2 without P0/P1
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.execute_step(agent_id=agent_id, step_type="P2", payload={"optimize": True})
    
    assert "P0 not completed" in str(exc_info.value)
    assert room.get_agent_state(agent_id)["current_phase"] == "P0"
```

### 3. Test Concurrent Agent Phase Transitions
**Scenario:** Multiple agents attempt phase transitions simultaneously with shared resource constraints.
**Expected:** Room should handle concurrency without state corruption or deadlock.
**Rationale:** Stress tests room's thread safety for fleet-scale operations.

```python
def test_concurrent_phase_transitions():
    """DeadbandRoom should handle concurrent agent phase transitions safely."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    import threading
    
    room = DeadbandRoom(config={"max_agents": 10})
    agent_ids = [room.register_agent(agent_type=f"zeroclaw_{i}") for i in range(5)]
    
    results = []
    errors = []
    
    def execute_agent_flow(agent_id):
        try:
            # P0
            p0_result = room.execute_step(
                agent_id=agent_id,
                step_type="P0",
                payload={"negative_space": ["test_rock"]}
            )
            # P1
            p1_result = room.execute_step(
                agent_id=agent_id,
                step_type="P1",
                payload={"safe_channels": ["channel_a"]}
            )
            results.append((agent_id, p0_result, p1_result))
        except Exception as e:
            errors.append((agent_id, str(e)))
    
    threads = []
    for agent_id in agent_ids:
        thread = threading.Thread(target=execute_agent_flow, args=(agent_id,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all agents completed without errors
    assert len(errors) == 0
    assert len(results) == 5
    
    # Verify each agent reached P1 phase
    for agent_id, p0_result, p1_result in results:
        state = room.get_agent_state(agent_id)
        assert state["current_phase"] == "P1"
        assert "safe_channels" in p1_result
```

### 4. Test Malformed Tile Payload
**Scenario:** Agent submits a tile with malformed JSON structure or missing required fields.
**Expected:** Room should reject the tile with descriptive error, not crash.
**Rationale:** Ensures robustness against malformed inputs from various agent implementations.

```python
def test_malformed_tile_payload():
    """DeadbandRoom should handle malformed tile payloads gracefully."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from plato_torch.exceptions import InvalidTileError
    
    room = DeadbandRoom(config={"max_agents": 3})
    agent_id = room.register_agent(agent_type="zeroclaw")
    
    # Complete P0 and P1 first
    room.execute_step(agent_id=agent_id, step_type="P0", payload={"negative_space": []})
    room.execute_step(agent_id=agent_id, step_type="P1", payload={"safe_channels": ["test"]})
    
    # Attempt P2 with malformed payload
    malformed_payloads = [
        {"optimize": True},  # Missing 'path' field
        {"path": []},  # Missing 'optimize' field
        None,  # Null payload
        "not_a_dict",  # Wrong type
        {"optimize": "yes", "path": {}}  # Wrong type for optimize
    ]
    
    for payload in malformed_payloads:
        with pytest.raises(InvalidTileError) as exc_info:
            room.execute_step(agent_id=agent_id, step_type="P2", payload=payload)
        
        assert "Invalid tile payload" in str(exc_info.value)
        assert room.get_agent_state(agent_id)["current_phase"] == "P1"  # State unchanged
```

### 5. Test Room Capacity Exhaustion
**Scenario:** Room reaches maximum agent capacity, then attempts to register additional agents.
**Expected:** Room should reject new registrations with capacity error, not crash existing agents.
**Rationale:** Validates resource boundary enforcement for fleet resource management.

```python
def test_room_capacity_exhaustion():
    """DeadbandRoom should handle maximum agent capacity correctly."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    from plato_torch.exceptions import RoomCapacityError
    
    room = DeadbandRoom(config={"max_agents": 2})
    
    # Register up to capacity
    agent1 = room.register_agent(agent_type="zeroclaw_1")
    agent2 = room.register_agent(agent_type="zeroclaw_2")
    
    # Verify both agents are active
    assert room.get_agent_state(agent1)["status"] == "active"
    assert room.get_agent_state(agent2)["status"] == "active"
    
    # Attempt to exceed capacity
    with pytest.raises(RoomCapacityError) as exc_info:
        room.register_agent(agent_type="zeroclaw_3")
    
    assert "maximum capacity" in str(exc_info.value).lower()
    
    # Verify existing agents unaffected
    assert room.get_agent_state(agent1)["status"] == "active"
    assert room.get_agent_state(agent2)["status"] == "active"
    
    # Verify agent count unchanged
    assert room.get_agent_count() == 2
```
