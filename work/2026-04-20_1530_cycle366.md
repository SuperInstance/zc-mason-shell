### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions correctly."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    tiles = [self._create_valid_tile(agent_id=aid) for aid in agent_ids]
    
    # Simulate concurrent submission
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(e)
    
    # Verify all submissions processed
    assert len(results) == 5
    # Room should maintain consistency (no crashes, state corruption)
    assert room.training_queue_length() <= 5
    # Priority ordering should be preserved despite concurrency
    priorities = [t.get('priority', 'P0') for t in tiles]
    if 'P0' in priorities:
        assert any(r.get('processed_priority') == 'P0' for r in results if isinstance(r, dict))
```

def test_partial_tile_recovery(self):
    """Test room recovery when tile delivery is interrupted."""
    room = DeadbandRoom()
    
    # Create tile with multiple chunks (simulating streaming)
    tile_chunks = [
        {'header': 'P0', 'agent_id': 'test_agent', 'timestamp': '2026-04-20T15:30:00Z'},

        {'footer': 'END_TILE', 'checksum': 'abc123'}
    ]
    
    # Submit partial tile (missing footer)
    partial_tile = tile_chunks[0:2]
    with pytest.raises(TileValidationError) as exc_info:
        room.submit_tile(partial_tile)
    
    assert "incomplete" in str(exc_info.value).lower()
    
    # Verify room didn't accept partial tile
    assert room.training_queue_length() == 0
    
    # Submit complete tile after partition heals
    complete_tile = self._reconstruct_tile(tile_chunks)
    result = room.submit_tile(complete_tile)
    assert result['status'] == 'accepted'

### Test 3: Invalid Priority Sequence Detection
```python
def test_priority_sequence_violation(self):
    """Test that P2-before-P1 sequences are rejected."""
    room = DeadbandRoom()
    
    # Agent tries to submit P2 without completing P1
    p2_tile = {
        'priority': 'P2',
        'agent_id': 'eager_agent',
        'content': 'Optimizing path without safe channel...',
        'timestamp': '2026-04-20T15:35:00Z'
    }
    
    # Should be rejected
    with pytest.raises(ProtocolViolationError) as exc_info:
        room.submit_tile(p2_tile)
    
    assert "P1 not completed" in str(exc_info.value)
    
    # Submit proper P0 then P1 sequence
    p0_tile = {**p2_tile, 'priority': 'P0', 'content': 'Negative space mapped...'}
    p1_tile = {**p2_tile, 'priority': 'P1', 'content': 'Safe channel identified...'}
    
    room.submit_tile(p0_tile)
    room.submit_tile(p1_tile)
    
    # Now P2 should be accepted
    result = room.submit_tile(p2_tile)
    assert result['status'] == 'accepted'
```

### Test 4: Resource Exhaustion During Training
```python
def test_memory_pressure_handling(self):
    """Test room behavior when system resources are constrained."""
    room = DeadbandRoom()
    
    # Simulate memory pressure by creating many large tiles
    large_tiles = []
    for i in range(1000):
        large_tile = self._create_valid_tile(
            agent_id=f"agent_{i}",
            content="x" * 10000,  # 10KB per tile
            metadata={'size': 10000}
        )
        large_tiles.append(large_tile)
    
    accepted_count = 0
    rejected_count = 0
    
    for tile in large_tiles:
        try:
            result = room.submit_tile(tile)
            if result.get('status') == 'accepted':
                accepted_count += 1
            else:
                rejected_count += 1
        except ResourceExhaustionError:
            rejected_count += 1
            # Room should continue operating
            assert room.is_operational()
    
    # Verify graceful degradation
    assert accepted_count > 0  # Some tiles processed
    assert rejected_count > 0  # Some tiles rejected
    assert room.is_operational()  # Room still functional
    
    # After pressure eases, new tiles should be accepted
    normal_tile = self._create_valid_tile(agent_id="recovery_agent")
    result = room.submit_tile(normal_tile)
    assert result['status'] == 'accepted'
```

### Test 5: Ghost Tile Integration with Live Agents
```python
def test_ghost_tile_influence(self):
    """Test that dead agent tiles affect live room training."""
    room = DeadbandRoom()
    
    # Submit tiles from a now-dead agent
    ghost_agent_id = "ghost_agent_001"
    ghost_tiles = [
        self._create_valid_tile(
            agent_id=ghost_agent_id,
            priority='P0',
            content='CRITICAL: Avoid this failure mode...',
            timestamp='2026-04-19T12:00:00Z'
        ),
        self._create_valid_tile(
            agent_id=ghost_agent_id,
            priority='P0',
            content='Another negative space mapping...',
            timestamp='2026-04-19T12:05:00Z'
        )
    ]
    
    for tile in ghost_tiles:
        room.submit_tile(tile)
    
    # Mark agent as dead (simulate)
    room.mark_agent_inactive(ghost_agent_id)
    
    # Live agent submits similar P0 tile
    live_tile = self._create_valid_tile(
        agent_id="live_agent_001",
        priority='P0',
        content='Mapping negative space...',
        timestamp='2026-04-20T15:40:00Z'
    )
    
    result = room.submit_tile(live_tile)
    
    # Room should recognize ghost tile similarity
    if 'ghost_correlation' in result:
        # May adjust training weight or flag for review
        assert result['ghost_correlation'] > 0.5
        assert result['status'] == 'accepted_with_ghost_note'
    else:
        # Or process normally
        assert result['status'] == 'accepted'
    
    # Verify ghost tiles still in training set
    training_data = room.get_training_batch()
    ghost_content = [t for t in training_data if t['agent_id'] == ghost_agent_id]
    assert len(ghost_content) == 2
```
