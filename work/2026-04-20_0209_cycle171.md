## Identified Edge Cases (P0 Negative Space)
Based on reading the implementation, these scenarios represent potential failure modes or untested behavior:

1. **Empty action space** - What happens when no actions are possible?
2. **Conflicting constraints** - When P1 channels contradict each other
3. **Zero-tolerance deadband** - Constraints that eliminate all options
4. **State mutation during protocol** - External state changes between P0/P1/P2
5. **Non-deterministic channel selection** - Multiple valid P1 channels with equal priority

```python
def test_deadband_empty_action_space(self):
    """P0: When no actions are possible, protocol should return empty result."""
    room = DeadbandRoom()
    state = {"position": "blocked"}
    actions = []  # Empty action space
    constraints = ["avoid_walls"]
    
    result = room.apply_deadband_protocol(state, actions, constraints)
    
    # P0 should map negative space (empty), P1 should find no channels
    assert result["p0_mapped"] == []
    assert result["p1_channels"] == []
    assert result["p2_optimized"] is None
    assert result["final_action"] is None
```

### Test 2: Conflicting P1 Constraints
```python
def test_deadband_conflicting_constraints(self):
    """P1: When constraints conflict, protocol should handle gracefully."""
    room = DeadbandRoom()
    state = {"fuel": 50, "distance": 100}
    actions = [
        {"id": "fast", "fuel_cost": 60, "time": 10},
        {"id": "slow", "fuel_cost": 40, "time": 30},
        {"id": "medium", "fuel_cost": 50, "time": 20}
    ]
    # Conflicting constraints: need fuel < 45 AND time < 15
    constraints = ["fuel_cost < 45", "time < 15"]
    
    result = room.apply_deadband_protocol(state, actions, constraints)
    
    # P0 should filter based on individual constraints
    # P1 should find no channels that satisfy ALL constraints
    assert len(result["p1_channels"]) == 0
    assert result["final_action"] is None
```

### Test 3: Zero-Tolerance Deadband Edge
```python
def test_deadband_zero_tolerance(self):
    """P0: Constraints with zero tolerance (exact values)."""
    room = DeadbandRoom()
    state = {"temperature": 22.5}
    actions = [
        {"id": "cool", "temp_change": -2.5},
        {"id": "heat", "temp_change": 2.5},
        {"id": "maintain", "temp_change": 0}
    ]
    # Must maintain EXACTLY 22.5°C (zero tolerance)
    constraints = ["abs(temperature + temp_change - 22.5) < 0.01"]
    
    result = room.apply_deadband_protocol(state, actions, constraints)
    
    # Only "maintain" action should pass P0
    assert len(result["p0_mapped"]) == 1
    assert result["p0_mapped"][0]["id"] == "maintain"
    assert result["final_action"]["id"] == "maintain"
```

### Test 4: State Mutation During Protocol
```python
def test_deadband_state_mutation(self):
    """Protocol should handle external state changes between phases."""
    room = DeadbandRoom()
    
    # Mutable state that can change externally
    mutable_state = {"counter": 0}
    actions = [
        {"id": "inc", "change": 1},
        {"id": "dec", "change": -1}
    ]
    constraints = ["counter + change >= 0"]
    
    # Simulate external mutation after P0 but before P1
    original_apply = room.apply_deadband_protocol
    
    def mutated_apply(state, actions, constraints):
        # External process modifies state
        state["counter"] = -1  # Invalid state
        return original_apply(state, actions, constraints)
    
    room.apply_deadband_protocol = mutated_apply
    
    result = room.apply_deadband_protocol(mutable_state, actions, constraints)
    
    # With counter = -1, no actions satisfy constraint
    assert len(result["p1_channels"]) == 0
    assert result["final_action"] is None
```

### Test 5: Non-Deterministic Channel Selection
```python
def test_deadband_tie_breaking(self):
    """P1: Multiple channels with equal priority should have deterministic selection."""
    room = DeadbandRoom()
    state = {"resources": 100}
    actions = [
        {"id": "A", "cost": 10, "value": 5},
        {"id": "B", "cost": 10, "value": 5},  # Identical to A
        {"id": "C", "cost": 20, "value": 8}
    ]
    constraints = ["cost <= 10"]
    
    result = room.apply_deadband_protocol(state, actions, constraints)
    
    # Both A and B should be in P1 channels
    assert len(result["p1_channels"]) == 2
    
    # P2 should select deterministically (e.g., by id or insertion order)
    # This tests that the protocol doesn't randomly select between equal options
    final_id = result["final_action"]["id"]
    assert final_id in ["A", "B"]
    
    # Verify deterministic behavior by running multiple times
    results = []
    for _ in range(10):
        r = room.apply_deadband_protocol(state, actions, constraints)
        results.append(r["final_action"]["id"])
    
    # All runs should produce the same result
    assert len(set(results)) == 1, "Non-deterministic selection detected"
```
