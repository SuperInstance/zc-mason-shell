```python
def test_protocol_sequence_with_gaps():
    """Test that protocols with missing intermediate priorities are rejected."""
    # Missing P1
    with pytest.raises(ValueError, match="Protocol must contain contiguous priorities"):
        DeadbandRoom(name="test", channels=["test"], protocol=["P0", "P2"])
    
    # Missing P0 (should still fail as P0 is required)
    with pytest.raises(ValueError, match="Protocol must start with P0"):
        DeadbandRoom(name="test", channels=["test"], protocol=["P1", "P2"])
    
    # Valid: P0 only (P1 and P2 optional per implementation)
    room = DeadbandRoom(name="test", channels=["test"], protocol=["P0"])
    assert room.protocol == ["P0"]
```

### Test 4: State Persistence After Reset
```python
def test_state_persistence_after_reset():
    """Test that room reset clears state but preserves configuration."""
    room = DeadbandRoom(name="test", channels=["ch1", "ch2"], protocol=["P0", "P1", "P2"])
    
    # Submit some tiles
    room.submit_tile({"content": "Tile 1", "priority": "P0", "channel": "ch1"})
    room.submit_tile({"content": "Tile 2", "priority": "P1", "channel": "ch2"})
    
    # Verify state exists
    assert room.get_tile_count() == 2
    assert room.get_current_priority() == "P1"  # Assuming state tracking
    
    # Reset room
    room.reset()
    
    # Configuration should remain
    assert room.name == "test"
    assert room.channels == ["ch1", "ch2"]
    assert room.protocol == ["P0", "P1", "P2"]
    
    # State should be cleared
    assert room.get_tile_count() == 0
    assert room.get_current_priority() == "P0"  # Back to start
```

```python
def test_channel_switching_mid_protocol():
    """Test that agents can switch channels between priorities."""
    room = DeadbandRoom(
        name="test", 
        channels=["analysis", "synthesis", "validation"], 
        protocol=["P0", "P1", "P2"]
    )
    
    # P0 in one channel
    tile1 = {"content": "P0 analysis", "priority": "P0", "channel": "analysis"}
    assert room.submit_tile(tile1) is True
    
    # P1 in different channel (should be allowed)
    tile2 = {"content": "P1 synthesis", "priority": "P1", "channel": "synthesis"}
    assert room.submit_tile(tile2) is True
    
    # P2 in third channel
    tile3 = {"content": "P2 validation", "priority": "P2", "channel": "validation"}
    assert room.submit_tile(tile3) is True
    
    # Invalid: Try to go back to P1 after P2 (protocol violation)
    tile4 = {"content": "Another P1", "priority": "P1", "channel": "analysis"}
    with pytest.raises(ValueError, match="Cannot regress to priority P1 after completing P2"):
        room.submit_tile(tile4)
```
