### Test 4: Protocol Phase Order Enforcement Under Stress

```python
def test_concurrent_phase_violation_detection():
    """Test that protocol violations are detected even with rapid phase calls."""
    room = DeadbandRoom(threshold=0.05, channels=["a", "b", "c"])
    
    violation_detected = False
    
    # Rapid, illegal phase sequence
    try:
        room.execute_phase("P0", constraints=["test"])
        room.execute_phase("P2", constraints=[])  # Illegal: skipping P1
    except ProtocolViolationError:
        violation_detected = True
    
    assert violation_detected, "Should detect P2 without P1"
    
    # Verify room state is consistent after violation
    assert room.current_phase == "P0"  # Should not advance to P2
    assert room.valid_channels == []   # P1 never completed
```

### Test 5: Historical Data Integrity Across Multiple Cycles
```python
def test_historical_data_integrity():
    """Test that optimization history maintains integrity across full protocol cycles."""
    room = DeadbandRoom(threshold=0.1, channels=["x", "y", "z"])
    
    histories = []
    
    # Run 5 complete protocol cycles
    for cycle in range(5):
        room.execute_phase("P0", constraints=[f"cycle_{cycle}"])
        room.execute_phase("P1", constraints=[f"cycle_{cycle}"])
        room.execute_phase("P2", constraints=[])
        
        # Record history at each cycle
        histories.append({
            'cycle': cycle,
            'phase': room.current_phase,
            'channels': room.valid_channels.copy(),
            'history': room.optimization_history.copy()
        })
    
    # Verify each cycle's history is preserved and distinct
    for i in range(1, 5):
        prev_history = histories[i-1]['history']
        curr_history = histories[i]['history']
        
        # Current should contain all previous entries plus new ones
        assert len(curr_history) > len(prev_history)
        
        # All previous entries should be unchanged
        for entry in prev_history:
            assert entry in curr_history
        
        # No duplicate cycle identifiers in history
        cycle_ids = [h.get('cycle_id', '') for h in curr_history if 'cycle_id' in h]
        assert len(cycle_ids) == len(set(cycle_ids))
```
