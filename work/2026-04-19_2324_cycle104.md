### Test 1: Concurrent Channel Contention
```python
def test_concurrent_channel_contention(self):
    """Test multiple agents attempting to claim the same P1 channel simultaneously."""
    room = DeadbandRoom()
    agent_ids = ["agent_a", "agent_b", "agent_c"]
    
    # All agents map same negative space (P0)
    for agent_id in agent_ids:
        room.process_tile(agent_id, {"phase": "P0", "violations": ["rock1"]})
    
    # All attempt same channel (should handle contention)
    results = []
    for agent_id in agent_ids:
        result = room.process_tile(agent_id, {"phase": "P1", "channel_candidate": "safe_path_1"})
        results.append(result.get("channel_assigned"))
    
    # Verify: Only one agent gets the channel, others get alternatives
    assigned_channels = [c for c in results if c is not None]
    assert len(set(assigned_channels)) == len(agent_ids)  # All get unique channels
    assert None not in results  # No agent gets blocked entirely
```

### Test 2: Partial State Recovery After Disconnect
```python
def test_partial_state_recovery(self):
    """Test room maintains protocol state when agent disconnects mid-phase."""
    room = DeadbandRoom()
    agent_id = "flaky_agent"
    
    # Agent completes P0, starts P1
    room.process_tile(agent_id, {"phase": "P0", "violations": ["rock1", "rock2"]})
    room.process_tile(agent_id, {"phase": "P1", "channel_candidate": "safe_a"})
    
    # Simulate agent disconnect (no tile for 5 cycles)
    for _ in range(6):
        room.cycle()
    
    # Agent reconnects with same ID
    result = room.process_tile(agent_id, {"phase": "P1", "channel_candidate": "safe_b"})
    
    # Should either: 1) Resume from P1, or 2) Restart from P0 with clear signal
    assert "phase" in result
    assert result["phase"] in ["P0", "P1", "P2"]
    if result["phase"] == "P0":
        assert "reason" in result and "timeout" in result["reason"].lower()
```

### Test 3: Memory Constraint During P2 Optimization
```python
def test_memory_constrained_optimization(self):
    """Test P2 optimization with artificially limited memory."""
    room = DeadbandRoom()
    agent_id = "memory_limited_agent"
    
    # Set up memory constraint
    room.memory_limit = 100  # bytes
    
    # Complete P0 and P1
    room.process_tile(agent_id, {"phase": "P0", "violations": ["x" * 50]})  # Large violation data
    room.process_tile(agent_id, {"phase": "P1", "channel_candidate": "narrow_path"})
    
    # Attempt P2 with data that would exceed limit
    result = room.process_tile(agent_id, {
        "phase": "P2",
        "optimization_data": "y" * 200  # Exceeds 100 byte limit
    })
    
    # Should gracefully handle constraint
    assert "error" in result or "constraint" in result.get("status", "")
    assert room.get_agent_phase(agent_id) == "P1"  # Should not advance to P2
```

```python
def test_syntactically_valid_protocol_violation(self):
    """Test tiles that are syntactically valid but violate protocol semantics."""
    room = DeadbandRoom()

    })

    assert "violation" in result or "invalid_transition" in result.get("status", "")

```

### Test 5: Malformed Tile Ingestion
```python
def test_malformed_tile_recovery(self):
    """Test room handling of structurally invalid tiles."""
    room = DeadbandRoom()
    agent_id = "buggy_agent"
    
    malformed_tiles = [
        None,  # Null tile
        {"wrong_key": "value"},  # Missing 'phase'
        {"phase": "P3"},  # Invalid phase value
        {"phase": "P0", "violations": "not_a_list"},  # Wrong type
        b"binary_data",  # Non-dict type
    ]
    
    for tile in malformed_tiles:
        try:
            result = room.process_tile(agent_id, tile)
            # If no exception, should return error structure
            assert "error" in result
            assert "malformed" in result["error"].lower() or "invalid" in result["error"].lower()
        except Exception as e:
            # Exception is acceptable if caught and logged
            assert hasattr(room, "log_error")
    
    # Agent should still be able to proceed after errors
    valid_result = room.process_tile(agent_id, {"phase": "P0", "violations": []})
    assert valid_result.get("phase") == "P0" or valid_result.get("next_phase") == "P1"
```
