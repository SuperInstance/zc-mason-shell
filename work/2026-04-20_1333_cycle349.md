### Test 1: Concurrent Tile Submission

import threading
from plato_torch.rooms.deadband_room import DeadbandRoom

def test_concurrent_tile_submission(room, agent_id):
    """P0: Concurrent submissions should not corrupt room state."""
    tiles = [
        {"cycle": 100, "phase": 1, "content": "Tile A", "protocol_step": "P1"},
        {"cycle": 100, "phase": 1, "content": "Tile B", "protocol_step": "P1"},
    ]
    results = []
    
    def submit(tile):
        try:
            results.append(room.submit_tile(agent_id, tile))
        except Exception as e:
            results.append(e)
    
    threads = [threading.Thread(target=submit, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # At least one submission should succeed
    assert any(r is True for r in results if isinstance(r, bool))
    # Room state should remain consistent
    assert room.get_agent_state(agent_id)["last_cycle"] == 100
```

def test_protocol_step_skipping(room, agent_id):
    """P0: Attempting P2 without P1 should be rejected."""
    # Submit P0 tile
    p0_tile = {"cycle": 200, "phase": 0, "content": "Mapping rocks", "protocol_step": "P0"}
    assert room.submit_tile(agent_id, p0_tile) == True
    
    # Attempt to submit P2 directly (skipping P1)
    p2_tile = {"cycle": 201, "phase": 2, "content": "Optimizing", "protocol_step": "P2"}
    # Room should reject this based on agent's state
    result = room.submit_tile(agent_id, p2_tile)
    assert result == False or isinstance(result, Exception)
    
    # Verify agent state hasn't advanced to phase 2
    state = room.get_agent_state(agent_id)
    assert state["current_phase"] < 2
```
