# DeadbandRoom.py - Plato-torch preset implementing the Deadband Protocol

import torch
import numpy as np
from typing import Dict, List, Optional, Tuple
from .BaseRoom import BaseRoom

class DeadbandRoom(BaseRoom):
    """
    A training room that enforces the Deadband Protocol:
    P0: Map negative space (what NOT to do)
    P1: Find safe channels (where you CAN be)
    P2: Optimize within channels (best path)
    
    Agents submit tiles (training examples) that are validated against
    protocol violations before being added to the training dataset.
    """
    
    def __init__(self, room_id: str, config: Optional[Dict] = None):
        super().__init__(room_id, config)
        self.p0_violations = []  # Negative space examples
        self.p1_channels = []    # Safe channels discovered
        self.p2_optimizations = []  # Best paths within channels
        
        # Protocol thresholds
        self.p0_threshold = config.get('p0_threshold', 0.3)  # Max violation score
        self.p1_min_confidence = config.get('p1_min_confidence', 0.7)  # Min channel confidence
        self.p2_efficiency_target = config.get('p2_efficiency_target', 0.9)  # Optimization target
        
        # State tracking
        self.current_phase = "P0"  # Start with negative space mapping
        self.agent_scores = {}
        
    def validate_tile(self, tile: Dict) -> Tuple[bool, str, float]:
        """
        Validate a submitted tile against the Deadband Protocol.
        Returns: (is_valid, phase, score)
        """
        # Extract tile data
        content = tile.get('content', '')
        metadata = tile.get('metadata', {})
        agent_id = tile.get('agent_id', 'unknown')
        
        # Phase-specific validation
        if self.current_phase == "P0":
            return self._validate_p0(content, metadata, agent_id)
        elif self.current_phase == "P1":
            return self._validate_p1(content, metadata, agent_id)
        elif self.current_phase == "P2":
            return self._validate_p2(content, metadata, agent_id)
        else:
            return False, "ERROR", 0.0
    
    def _validate_p0(self, content: str, metadata: Dict, agent_id: str) -> Tuple[bool, str, float]:
        """Validate P0 tile: must clearly identify negative space."""
        # Check for absolute claims (violation of P0)
        absolute_indicators = ['always', 'never', 'guaranteed', 'certain', 'definitely']
        absolute_count = sum(1 for word in absolute_indicators if word in content.lower())
        
        # Check for negative space mapping
        negative_indicators = ['avoid', 'don\'t', 'not to', 'should not', 'must not', 'never']
        negative_count = sum(1 for word in negative_indicators if word in content.lower())
        
        # Calculate violation score (lower is better for P0)
        violation_score = absolute_count / max(len(content.split()), 1)
        
        if violation_score > self.p0_threshold:
            return False, "P0", violation_score
        
        # Must have at least one negative indicator
        if negative_count == 0:
            return False, "P0", 1.0
        
        # Valid P0 tile
        self.p0_violations.append({
            'content': content,
            'agent_id': agent_id,
            'score': violation_score
        })
        return True, "P0", violation_score
    
    def _validate_p1(self, content: str, metadata: Dict, agent_id: str) -> Tuple[bool, str, float]:
        """Validate P1 tile: must identify safe channels."""
        # Check for channel discovery patterns
        channel_indicators = ['channel', 'safe', 'can be', 'possible', 'feasible', 'workable']
        channel_count = sum(1 for word in channel_indicators if word in content.lower())
        
        # Calculate confidence score
        confidence = channel_count / max(len(content.split()) / 10, 1)
        
        if confidence < self.p1_min_confidence:
            return False, "P1", confidence
        
        # Valid P1 tile
        self.p1_channels.append({
            'content': content,
            'agent_id': agent_id,
            'confidence': confidence
        })
        return True, "P1", confidence
    
    def _validate_p2(self, content: str, metadata: Dict, agent_id: str) -> Tuple[bool, str, float]:
        """Validate P2 tile: must show optimization within known channels."""
        # Check for optimization patterns
        optimization_indicators = ['optimize', 'efficient', 'best', 'improve', 'enhance', 'streamline']
        opt_count = sum(1 for word in optimization_indicators if word in content.lower())
        
        # Must reference known P1 channels
        channel_references = any(
            channel['content'].lower() in content.lower() 
            for channel in self.p1_channels[-10:]  # Last 10 channels
        )
        
        efficiency = opt_count / max(len(content.split()) / 10, 1)
        
        if not channel_references or efficiency < self.p2_efficiency_target:
            return False, "P2", efficiency
        
        # Valid P2 tile
        self.p2_optimizations.append({
            'content': content,
            'agent_id': agent_id,
            'efficiency': efficiency,
            'channel_ref': channel_references
        })
        return True, "P2", efficiency
    
    def advance_phase(self) -> bool:
        """Advance to next protocol phase if criteria met."""
        if self.current_phase == "P0" and len(self.p0_violations) >= 10:
            self.current_phase = "P1"
            return True
        elif self.current_phase == "P1" and len(self.p1_channels) >= 5:
            self.current_phase = "P2"
            return True
        return False
    
    def get_stats(self) -> Dict:
        """Return room statistics."""
        return {
            'current_phase': self.current_phase,
            'p0_violations': len(self.p0_violations),
            'p1_channels': len(self.p1_channels),
            'p2_optimizations': len(self.p2_optimizations),
            'agent_count': len(self.agent_scores)
        }
