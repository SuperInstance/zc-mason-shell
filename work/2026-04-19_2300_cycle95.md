### Test 2: P1 Safe Channel — Tile with Nested Metadata

### Test 3: P2 Optimization — Batch Tile Processing Under Load

### Test 4: Doctrine Violation — Skipping to P2

"""
Edge-case tests for DeadbandRoom preset.
Mason — Cycle 95 — Fleet Boot Camp Phase 4.
"""

import pytest
from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.models import Tile

class TestDeadbandRoomEdgeCases:
    
    @pytest.fixture
    def room(self):
        """Return a fresh DeadbandRoom instance."""
        return DeadbandRoom()
    
    def test_ghost_tile_rejection(self, room):
        """P0: Ghost tile from dead agent should be rejected."""
        ghost_tile = Tile(
            agent_id="ghost_agent_001",
            cycle=50,
            content="Hallucinated content from dead agent.",
            priority=0,
            timestamp="2026-04-10T00:00:00Z"  # Old timestamp
        )
        # Mock agent registry: agent died at cycle 40
        with patch.object(room, 'agent_registry', {"ghost_agent_001": {"alive_until": 40}}):
            result = room.process_tile(ghost_tile)
            assert result["status"] == "rejected"
            assert "ghost_tile" in result["reason"]
    
    def test_nested_metadata_validation(self, room):
        """P1: Tile with nested metadata must reference existing fleet-knowledge file."""
        tile = Tile(
            agent_id="mason",
            cycle=95,
            content={
                "text": "Analysis of constraint theory.",
                "metadata": {"source": "fleet-knowledge/CONSTRAINT-THEORY.md"}
            },
            priority=1
        )
        # Mock filesystem check
        with patch("os.path.exists", return_value=True):
            result = room.process_tile(tile)
            assert result["status"] == "accepted"
        
        with patch("os.path.exists", return_value=False):
            result = room.process_tile(tile)
            assert result["status"] == "rejected"
            assert "missing_source" in result["reason"]
    
    def test_batch_processing_priority_order(self, room):
        """P2: Batch processing respects priority ordering and cycle limits."""
        tiles = []
        # Mix priorities
        for i in range(30):
            tiles.append(Tile(agent_id="batch_agent", cycle=95, content=f"P0 tile {i}", priority=0))
        for i in range(40):
            tiles.append(Tile(agent_id="batch_agent", cycle=95, content=f"P2 tile {i}", priority=2))
        
        results = room.process_batch(tiles, max_per_cycle=50)
        # First 50 tiles should be processed, P0 first
        assert len(results["processed"]) == 50
        assert all(t.priority == 0 for t in results["processed"][:30])
        assert len(results["queued"]) == 20  # Remaining P2 tiles
    
    def test_priority_mismatch_doctrine_violation(self, room):
        """P0 detection overrides declared priority."""
        tile = Tile(
            agent_id="mason",
            cycle=95,
            content="Always use this API; it's guaranteed to work.",
            priority=2  # Declared as P2, but content is P0 violation
        )
        result = room.process_tile(tile)
        # Room should detect absolute claims and treat as P0
        assert result["priority_evaluated"] == 0
        if "always" in tile.content or "guaranteed" in tile.content:
            assert result["status"] == "rejected"
    
    def test_sentiment_aware_scoring(self, room):
        """Tile scoring integrates holodeck-rust NPC sentiment."""
        tile = Tile(
            agent_id="mason",
            cycle=95,
            content={
                "text": "NPC dialogue from holodeck.",
                "npc_sentiment": -0.5  # Negative sentiment
            },
            priority=1
        )
        # Mock sentiment scoring factor
        with patch.object(room, 'sentiment_factor', 0.2):
            result = room.process_tile(tile)
            base_score = result.get("base_score", 1.0)
            sentiment_score = tile.content.get("npc_sentiment", 0)
            expected_score = base_score * (1 + sentiment_score * 0.2)
            assert result["final_score"] == pytest.approx(expected_score)
