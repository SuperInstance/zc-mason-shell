### Test 1: P2 Attempt Without P1 Channel (Protocol Violation)
```rust
#[test]
fn test_p2_attempt_without_p1_channel() {
    let config = RoomConfig {
        max_agents: 10,
        tile_validation: true,
        ..Default::default()
    };
    let mut room = DeadbandRoom::new(config);
    let agent_id = room.add_agent("test_agent".to_string());
    
    // Agent tries to submit P2 tile without establishing P1 channel
    let p2_tile = Tile {
        phase: Phase::P2,
        content: "Optimizing path".to_string(),
        agent_id: agent_id.clone(),
        timestamp: Utc::now(),
    };
    
    let result = room.submit_tile(agent_id.clone(), p2_tile);
    assert!(result.is_err());
    assert_eq!(result.unwrap_err(), RoomError::ProtocolViolation);
    
    // Agent should be removed for skipping P1
    assert!(!room.has_agent(&agent_id));
}
```

### Test 2: Maximum Agent Capacity Boundary
```rust
#[test]
fn test_maximum_agent_capacity() {
    let config = RoomConfig {
        max_agents: 3,  // Small limit for testing
        tile_validation: true,
        ..Default::default()
    };
    let mut room = DeadbandRoom::new(config);
    
    // Add agents up to capacity
    let agent1 = room.add_agent("agent1".to_string());
    let agent2 = room.add_agent("agent2".to_string());
    let agent3 = room.add_agent("agent3".to_string());
    
    assert_eq!(room.agent_count(), 3);
    
    // Attempt to add fourth agent should fail
    let result = room.add_agent("agent4".to_string());
    assert!(result.is_err());
    assert_eq!(result.unwrap_err(), RoomError::CapacityExceeded);
    
    // Remove one agent and capacity should work again
    room.remove_agent(&agent1);
    let agent4 = room.add_agent("agent4".to_string());
    assert!(agent4.is_ok());
}
```

### Test 3: Concurrent Tile Submission Race Condition
```rust
#[test]
fn test_concurrent_tile_submission() {
    let config = RoomConfig {
        max_agents: 5,
        tile_validation: true,
        ..Default::default()
    };
    let mut room = DeadbandRoom::new(config);
    
    // Setup two agents with P1 channels
    let agent1 = room.add_agent("agent1".to_string());
    let agent2 = room.add_agent("agent2".to_string());
    
    let p1_tile1 = Tile {
        phase: Phase::P1,
        content: "Channel A".to_string(),
        agent_id: agent1.clone(),
        timestamp: Utc::now(),
    };
    
    let p1_tile2 = Tile {
        phase: Phase::P1,
        content: "Channel B".to_string(),
        agent_id: agent2.clone(),
        timestamp: Utc::now(),
    };
    
    // Simulate concurrent submission (sequential in test, but testing state consistency)
    let result1 = room.submit_tile(agent1.clone(), p1_tile1);
    let result2 = room.submit_tile(agent2.clone(), p1_tile2);
    
    assert!(result1.is_ok());
    assert!(result2.is_ok());
    
    // Both agents should have active P1 channels
    assert!(room.get_agent_phase(&agent1) == Some(Phase::P1));
    assert!(room.get_agent_phase(&agent2) == Some(Phase::P1));
    
    // Room should track both channels
    assert_eq!(room.active_channels(), 2);
}
```

### Test 5: Room State Persistence Simulation

```rust
#[test]
fn test_state_persistence_simulation() {
    let config = RoomConfig {
        max_agents: 10,
        tile_validation: true,
        persist_state: true,  // Assuming this flag exists
        ..Default::default()
    };
    
    // Create room and add some state
    let mut room1 = DeadbandRoom::new(config.clone());
    let agent1 = room1.add_agent("agent1".to_string());
    
    let p0_tile = Tile {
        phase: Phase::P0,
        content: "Negative space mapped".to_string(),
        agent_id: agent1.clone(),
        timestamp: Utc::now(),
    };
    
    room1.submit_tile(agent1.clone(), p0_tile).unwrap();
    
    // Simulate persistence by serializing/deserializing
    let serialized = serde_json::to_string(&room1).unwrap();
    let mut room2: DeadbandRoom = serde_json::from_str(&serialized).unwrap();
    
    // State should be preserved
    assert_eq!(room2.agent_count(), 1);
    assert!(room2.has_agent(&agent1));
    assert_eq!(room2.get_agent_phase(&agent1), Some(Phase::P0));
    assert_eq!(room2.total_tiles(), 1);
}
```
