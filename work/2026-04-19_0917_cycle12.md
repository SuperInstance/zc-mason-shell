### Test 4: Concurrent Tile Submission Race Condition
```python
def test_concurrent_submission_handling(self):
    """Test handling of rapid concurrent tile submissions."""
    room = DeadbandRoom()
    
    def submit_concurrent(agent_id):
        tile = {
            "agent_id": agent_id,
            "cycle": 1,
            "priority": 0,
            "content": f"Content from {agent_id}",
            "phase": "P0"
        }
        return room.submit_tile(tile)
    
    # Simulate concurrent submissions
    import threading
    results = []
    
    def worker(agent_id):
        try:
            result = submit_concurrent(agent_id)
            results.append((agent_id, "success", result))
        except Exception as e:
            results.append((agent_id, "error", str(e)))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=worker, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify all submissions were processed
    success_count = sum(1 for r in results if r[1] == "success")
    self.assertEqual(success_count, 5, "All concurrent submissions should succeed")
    
    # Verify no duplicate cycle numbers were assigned
    cycle_numbers = [r[2].get("cycle") for r in results if r[1] == "success"]
    self.assertEqual(len(set(cycle_numbers)), 5, "Each tile should have unique cycle tracking")
```

```python
def test_malformed_content_handling(self):
    """Test submission with potentially malicious or malformed content."""
    room = DeadbandRoom()
    
    test_cases = [
        {
            "name": "SQL injection attempt",
            "content": "Test'; DROP TABLE tiles; --",
            "should_fail": False  # Should be sanitized, not rejected
        },
        {
            "name": "Excessively large content",
            "content": "X" * 10001,  # 10KB+ content
            "should_fail": True
        },
        {
            "name": "Binary data in content",
            "content": b"binary\x00data".decode('latin-1'),
            "should_fail": False  # Should be converted to string
        },
        {
            "name": "Empty string content",
            "content": "",
            "should_fail": True
        }
    ]
    
    for test_case in test_cases:
        tile = {
            "agent_id": "test_agent",
            "cycle": 1,
            "priority": 0,
            "content": test_case["content"],
            "phase": "P0"
        }
        
        if test_case["should_fail"]:
            with self.assertRaises(ValueError):
                room.submit_tile(tile)
        else:
            # Should process without error
            result = room.submit_tile(tile)
            self.assertIsNotNone(result)
            # Verify content was properly stored/truncated
            stored_content = result.get("content", "")
            self.assertTrue(len(stored_content) > 0)
```
