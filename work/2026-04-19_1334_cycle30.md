## 5 Additional Edge-Case Tests

### Test 1: Concurrent Agent Tile Submission
**Scenario:** Multiple agents attempt to submit tiles simultaneously during the same training cycle.
**Edge:** Race conditions in tile queue management.
**Assertion:** All valid tiles are processed; no data corruption occurs.

```python
def test_concurrent_tile_submission(self):
    """Test DeadbandRoom handles concurrent tile submissions from multiple agents."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3", "agent_4"]
    tiles = [
        {"agent_id": aid, "content": f"test from {aid}", "phase": "P1"}
        for aid in agent_ids
    ]
    
    # Simulate concurrent submission
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All tiles should be accepted
    assert all(r["status"] == "accepted" for r in results), \
        "Concurrent tile submission failed: some tiles rejected unexpectedly"
    
    # Verify all tiles appear in room state
    room_state = room.get_state()
    submitted_tiles = room_state.get("tiles", [])
    assert len(submitted_tiles) == len(tiles), \
        f"Expected {len(tiles)} tiles in room state, found {len(submitted_tiles)}"
```

### Test 3: Malformed Tile Payload Injection
**Scenario:** Agent submits tile with invalid structure or malicious payload.
**Edge:** Input validation and sanitization.
**Assertion:** Room rejects malformed tiles without crashing; error logged.

```python
def test_malformed_tile_payload(self):
    """Test DeadbandRoom handles malformed tile payloads gracefully."""
    room = DeadbandRoom()
    
    malformed_tiles = [
        None,  # null payload
        {"agent_id": "test"},  # missing required fields
        {"agent_id": "x" * 1000, "content": "", "phase": "P0"},  # oversized ID
        {"agent_id": "test", "content": 12345, "phase": "P1"},  # wrong content type
        {"agent_id": "test", "content": "valid", "phase": "P5"},  # invalid phase
        {"agent_id": "test", "content": "<script>alert()</script>", "phase": "P1"},  # XSS attempt
    ]
    
    for i, tile in enumerate(malformed_tiles):
        try:
            result = room.submit_tile(tile)
            assert result["status"] == "rejected", \
                f"Malformed tile {i} should be rejected: {tile}"
        except Exception as e:
            # Room should not crash on invalid input
            pytest.fail(f"Room crashed on malformed tile {i}: {e}")
```

### Test 4: Resource Exhaustion – Memory Limit
**Scenario:** Room processes extremely large tiles or excessive tile volume.
**Edge:** Memory management and quota enforcement.
**Assertion:** Room enforces memory limits; rejects oversized tiles; maintains operation.

```python
def test_memory_limit_enforcement(self):
    """Test DeadbandRoom enforces memory limits on tile storage."""
    room = DeadbandRoom()
    room.max_tile_size = 1024  # 1KB limit
    room.max_total_tiles = 5   # Small limit for testing
    
    # Test oversized tile
    large_content = "x" * 2048  # 2KB
    tile = {
        "agent_id": "test",
        "content": large_content,
        "phase": "P1"
    }
    
    result = room.submit_tile(tile)
    assert result["status"] == "rejected", \
        "Oversized tile should be rejected"
    assert "size" in result.get("reason", "").lower(), \
        f"Expected size limit reason, got: {result.get('reason')}"
    
    # Test tile count limit
    for i in range(room.max_total_tiles + 2):
        tile = {
            "agent_id": f"agent_{i}",
            "content": f"tile {i}",
            "phase": "P1"
        }
        result = room.submit_tile(tile)
        
        if i >= room.max_total_tiles:
            assert result["status"] == "rejected", \
                f"Tile {i} should be rejected due to count limit"
            assert "limit" in result.get("reason", "").lower() or \
                   "full" in result.get("reason", "").lower(), \
                f"Expected limit reason, got: {result.get('reason')}"
        else:
            assert result["status"] == "accepted", \
                f"Tile {i} should be accepted (under limit)"
```

### Test 5: Protocol Regression After Recovery
**Scenario:** Room recovers from failure state; protocol enforcement must persist.
**Edge:** State persistence and protocol integrity.
**Assertion:** Recovered room maintains P0>P1>P2 priority; previous violations remembered.

```python
def test_protocol_regression_after_recovery(self):
    """Test DeadbandRoom maintains protocol enforcement after recovery."""
    room = DeadbandRoom()
    
    # First, establish some state
    valid_tiles = [
        {"agent_id": "a1", "content": "P0 safety", "phase": "P0"},
        {"agent_id": "a2", "content": "P1 channel", "phase": "P1"},
    ]
    
    for tile in valid_tiles:
        room.submit_tile(tile)
    
    # Simulate room recovery (re-initialization with existing state)
    room_state = room.get_state()
    recovered_room = DeadbandRoom()
    recovered_room.load_state(room_state)
    
    # Attempt protocol violation on recovered room
    violation_tile = {
        "agent_id": "violator",
        "content": "Skip to P2 without P1",
        "phase": "P2"
    }
    
    # Room should reject P2 without prior P1 completion
    result = recovered_room.submit_tile(violation_tile)
    assert result["status"] == "rejected", \
        "Protocol violation should be rejected after recovery"
    assert "p1" in result.get("reason", "").lower() or \
           "priority" in result.get("reason", "").lower(), \
        f"Expected protocol priority reason, got: {result.get('reason')}"
    
    # Verify previous valid tiles still present
    recovered_state = recovered_room.get_state()
    assert len(recovered_state.get("tiles", [])) == len(valid_tiles), \
        "Recovered room lost previous valid tiles"
```
