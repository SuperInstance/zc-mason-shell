def test_concurrent_tile_addition_race_condition():
    """Test that multiple threads adding tiles concurrently doesn't corrupt state."""
    import threading
    from plato_torch.rooms.deadband_room import DeadbandRoom
    
    room = DeadbandRoom()
    tiles_to_add = 1000
    tiles_added = []
    
    def add_tiles(thread_id):
        for i in range(tiles_to_add // 10):
            tile = {"thread": thread_id, "index": i, "content": f"tile_{thread_id}_{i}"}
            room.add_tile(tile)
            tiles_added.append((thread_id, i))
    
    threads = []
    for i in range(10):
        thread = threading.Thread(target=add_tiles, args=(i,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all tiles were added
    state = room.get_state()
    assert len(state["tiles"]) == tiles_to_add
    
    # Verify no duplicate tile IDs (if IDs are generated)
    if "tile_ids" in state:
        assert len(state["tile_ids"]) == len(set(state["tile_ids"]))
```

def test_state_persistence_and_recovery():
    """Test that room state can be properly serialized and restored."""
    from plato_torch.rooms.deadband_room import DeadbandRoom
    import pickle
    import copy
    
    # Create room and add some tiles
    room1 = DeadbandRoom()
    tiles = [
        {"id": f"tile_{i}", "content": f"content_{i}", "priority": i % 3}
        for i in range(10)
    ]
    for tile in tiles:
        room1.add_tile(tile)
    
    # Get initial state
    state1 = room1.get_state()
    
    # Test 1: Deep copy state and restore to new room
    state_copy = copy.deepcopy(state1)
    
    room2 = DeadbandRoom()
    # Note: This assumes set_state() method exists or similar restoration mechanism
    # If not available in current API, this tests the need for such functionality
    if hasattr(room2, 'set_state'):
        room2.set_state(state_copy)
        state2 = room2.get_state()
        
        # Verify states are equivalent
        assert state1["tiles"] == state2["tiles"]
        assert state1["preset"]["name"] == state2["preset"]["name"]
    
    # Test 2: Pickle/unpickle the entire room object
    try:
        pickled = pickle.dumps(room1)
        room3 = pickle.loads(pickled)
        
        state3 = room3.get_state()
        # Verify unpickled room has same state
        assert len(state3["tiles"]) == len(state1["tiles"])
        
        # Verify unpickled room is still functional
        room3.add_tile({"id": "after_pickle", "test": True})
        state3_updated = room3.get_state()
        assert len(state3_updated["tiles"]) == len(state1["tiles"]) + 1
    except (pickle.PickleError, AttributeError):
        # Pickling might not be supported - this is acceptable
        pass
```
