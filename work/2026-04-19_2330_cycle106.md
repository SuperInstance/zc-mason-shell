## 5 Additional Edge-Case Tests for DeadbandRoom

### Test 1: Protocol Violation - Skipping Directly to P2
**Scenario**: Agent attempts to optimize (P2) without first establishing safe channels (P1)
**Expected**: Room should reject the action and remain in current state
**Rationale**: Tests strict priority enforcement of Deadband Protocol

```python
def test_protocol_violation_skip_to_p2():
    """Test that agent cannot skip from P0 directly to P2."""
    room = DeadbandRoom()
    # Start in P0 state
    room.state = "P0"
    room.current_channel = None
    
    # Attempt P2 optimization without P1
    result = room.process_action("optimize", {"target": 0.5})
    
    assert result["success"] == False
    assert "must complete P1 first" in result["message"].lower()
    assert room.state == "P0"  # State unchanged
```

### Test 3: Concurrent Action Interference
**Scenario**: Multiple rapid actions from same agent ID
**Expected**: Room should handle race conditions or queue actions properly
**Rationale**: Tests thread safety and action sequencing

```python
def test_concurrent_action_sequence():
    """Test that rapid sequential actions maintain protocol integrity."""
    room = DeadbandRoom()
    agent_id = "test_agent_1"
    
    # Start protocol
    room.process_action("start_p0", {"agent_id": agent_id})
    
    # Simulate rapid, potentially conflicting actions
    actions = [
        ("complete_p0", {}),
        ("explore_channel", {"channel": "safe_1"}),
        ("optimize", {"target": 0.7}),
        ("complete_p0", {})  # Duplicate P0 completion
    ]
    
    results = []
    for action, params in actions:
        params["agent_id"] = agent_id
        results.append(room.process_action(action, params))
    
    # Verify protocol was followed correctly despite rapid actions
    assert results[0]["success"] == True  # First P0 completion
    assert results[1]["success"] == True  # P1 exploration
    assert results[2]["success"] == False  # P2 should fail (no P1 completion)
    assert results[3]["success"] == False  # Duplicate P0 should fail
```

### Test 4: Memory Persistence Across Resets
**Scenario**: Agent completes P0-P2 cycle, resets, then starts new cycle
**Expected**: Negative space knowledge (P0) should persist or be archived
**Rationale**: Tests "ghost tiles" concept - lessons should survive resets

```python
def test_p0_knowledge_persistence():
    """Test that P0 negative space knowledge persists after reset."""
    room = DeadbandRoom()
    
    # Complete a full cycle
    room.process_action("start_p0", {})
    room.negative_space = ["rock_a", "rock_b"]
    room.process_action("complete_p0", {})
    
    # Store P0 knowledge
    p0_knowledge = room.negative_space.copy()
    
    # Complete cycle and reset
    room.process_action("complete_p2", {})
    room.process_action("reset", {})
    
    # Start new cycle - P0 knowledge should influence new analysis
    room.process_action("start_p0", {})
    
    assert hasattr(room, 'previous_negative_space')
    assert room.previous_negative_space == p0_knowledge
    # New P0 analysis should reference previous knowledge
    assert "historical" in room.process_action("get_p0_guidance", {})["message"]
```

### Test 5: Malformed Channel Data
**Scenario**: P1 returns channel with invalid structure or extreme values
**Expected**: Room should sanitize or reject malformed channel data
**Rationale**: Tests robustness against corrupted state from upstream systems

```python
def test_malformed_channel_data():
    """Test handling of invalid channel data from P1 exploration."""
    room = DeadbandRoom()
    room.state = "P1"
    
    # Test various malformed channel data
    malformed_channels = [
        {"name": "valid", "safety_score": 1.5},  # Score > 1.0
        {"name": "", "safety_score": 0.5},  # Empty name
        {"safety_score": 0.7},  # Missing name
        {"name": "channel_x", "safety_score": -0.1},  # Negative score
        None,  # Null channel
        [],  # Array instead of dict
    ]
    
    for channel in malformed_channels:
        result = room.process_action("add_channel", {"channel": channel})
        
        if channel is None or not isinstance(channel, dict):
            assert result["success"] == False
        elif not channel.get("name") or not 0 <= channel.get("safety_score", -1) <= 1:
            assert result["success"] == False
            assert "invalid" in result["message"].lower()
```
