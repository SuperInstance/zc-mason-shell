```python
def test_protocol_with_mutating_state(self):
    """Test protocol behavior when state is modified during execution."""
    room = DeadbandRoom()
    
    state = {
        "position": [0, 0, 0],
        "velocity": [1, 0, 0],
        "constraints": {"max_speed": 5},
        "negative_space": [[1, 1, 1]]
    }
    
    # Simulate concurrent modification by callback
    modified_constraints = []
    
    def constraint_modifier(phase_result):
        if "P1" in phase_result:
            # Modify constraints after P0 completes
            state["constraints"]["max_speed"] = 2
        modified_constraints.append(state["constraints"]["max_speed"])
    
    # Run protocol with monitoring
    result = room.run_protocol(state)
    
    # Protocol should either:
    # 1. Use original constraints consistently, OR
    # 2. Handle constraint changes gracefully
    assert "P2" in result
    
    # Verify behavior is deterministic
    if len(modified_constraints) > 1:
        # If constraints changed, path should respect final constraints
        final_speed = state["constraints"]["max_speed"]
        path = result["P2"]["optimized_path"]
        for point in path:
            speed = sum(v**2 for v in point.get("velocity", [0,0,0]))**0.5
            assert speed <= final_speed + 1e-9
```

```python
def test_partial_negative_space_mapping(self):
    """Test P0 when negative space only partially blocks channels."""

    state = {
        "position": [0, 0, 0],
        "velocity": [0, 0, 0],
        "constraints": {"max_speed": 10},
        "negative_space": [
            [[0, 0, 0], [5, 0, 0]],  # Block +X direction
            [[0, 0, 0], [0, 5, 0]],  # Block +Y direction
            # Leave -X, -Y, and Z directions open
        ]
    }
    
    result = room.run_protocol(state)
    
    # P0 should identify the blocked directions
    assert "P0" in result
    blocked_dirs = result["P0"]["negative_space_mapped"]
    assert len(blocked_dirs) >= 2
    
    # P1 should find at least one safe channel (e.g., -X or -Y direction)
    assert "P1" in result
    channels = result["P1"]["safe_channels"]
    assert len(channels) > 0
    
    # Verify channels avoid blocked directions
    for channel in channels:
        direction = channel.get("direction", [0, 0, 0])
        # Should not be primarily in +X or +Y direction
        assert not (direction[0] > 0.5 and direction[1] < 0.5)
        assert not (direction[1] > 0.5 and direction[0] < 0.5)
```

### Test 4: Protocol Interruption and Resume
```python
def test_protocol_interruption_resume(self):
    """Test protocol can be interrupted and resumed with partial results."""
    room = DeadbandRoom()
    
    state = {
        "position": [0, 0, 0],
        "velocity": [2, 0, 0],
        "constraints": {"max_speed": 5, "min_speed": 1},
        "negative_space": [[3, 0, 0], [4, 0, 0]]
    }
    
    # Run P0 only
    phase_results = {}
    try:
        # Simulate interruption after P0
        p0_result = room._run_p0(state)
        phase_results["P0"] = p0_result
        
        # Simulate system interruption
        raise KeyboardInterrupt("Simulated interruption")
        
    except KeyboardInterrupt:
        # Resume with P1 using P0 results
        p1_input = {**state, **phase_results["P0"]}
        p1_result = room._run_p1(p1_input)
        phase_results["P1"] = p1_result
        
        # Continue with P2
        p2_input = {**state, **phase_results["P0"], **phase_results["P1"]}
        p2_result = room._run_p2(p2_input)
        phase_results["P2"] = p2_result
    
    # Verify all phases completed successfully
    assert "negative_space_mapped" in phase_results["P0"]
    assert "safe_channels" in phase_results["P1"]
    assert "optimized_path" in phase_results["P2"]
    
    # Verify consistency across phases
    channels = phase_results["P1"]["safe_channels"]
    path = phase_results["P2"]["optimized_path"]
    
    # Path should stay within safe channels
    for point in path:
        in_channel = False
        for channel in channels:
            if self._point_in_channel(point, channel):
                in_channel = True
                break
        assert in_channel, f"Point {point} outside safe channels"
```
