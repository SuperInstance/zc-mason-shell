### Test 1: Minimal Grid Size (1x1)

```python
def test_deadband_room_minimal_grid(self):
    """Test DeadbandRoom with 1x1 grid - edge case for space constraints."""
    room = DeadbandRoom(grid_size=(1, 1), obstacles=[])
    
    # P0: Negative space should be empty (no obstacles)
    negative_space = room.map_negative_space()
    assert negative_space == set(), "1x1 grid with no obstacles should have empty negative space"
    
    # P1: Safe channels should be the single cell
    safe_channels = room.find_safe_channels(negative_space)
    assert safe_channels == {(0, 0)}, "1x1 grid should have single safe channel at (0,0)"
    
    # P2: Optimization should return the only possible path
    optimal_path = room.optimize_within_channels(safe_channels)
    assert optimal_path == [(0, 0)], "Optimal path in 1x1 grid should be single cell"
```

### Test 2: Fully Blocked Grid (No Safe Channels)
```python
def test_deadband_room_fully_blocked(self):
    """Test DeadbandRoom where obstacles cover all cells - P1 should return empty."""
    grid_size = (3, 3)
    obstacles = [(x, y) for x in range(3) for y in range(3)]  # All cells blocked
    
    room = DeadbandRoom(grid_size=grid_size, obstacles=obstacles)
    
    # P0: Negative space should be all cells
    negative_space = room.map_negative_space()
    assert len(negative_space) == 9, "All 9 cells should be in negative space"
    
    # P1: Safe channels should be empty set
    safe_channels = room.find_safe_channels(negative_space)
    assert safe_channels == set(), "Fully blocked grid should have no safe channels"
    
    # P2: Should raise appropriate error when no channels exist
    with pytest.raises(ValueError, match="No safe channels available for optimization"):
        room.optimize_within_channels(safe_channels)
```

### Test 5: Protocol Phase Tie (P0 and P1 Both Valid)
```python  
def test_deadband_room_protocol_tie(self):
    """Test scenario where P0 and P1 recommendations conflict."""
    # Create grid where obstacle detection (P0) and channel finding (P1) could conflict
    grid_size = (4, 4)
    obstacles = [(1, 1), (2, 2)]
    
    room = DeadbandRoom(grid_size=grid_size, obstacles=obstacles)
    
    # P0: Map negative space
    negative_space = room.map_negative_space()
    
    # P1: Find safe channels  
    safe_channels = room.find_safe_channels(negative_space)
    
    # Create a mock agent trying to skip from P0 to P2
    # This tests protocol enforcement
    try:
        # Attempt optimization without P1 (should fail)
        room.optimize_within_channels(negative_space)  # Passing negative_space instead of safe_channels
        assert False, "Should have raised protocol violation"
    except ProtocolViolationError as e:
        assert "P1 required before P2" in str(e), "Should enforce protocol phase order"
    
    # Valid path: P0 → P1 → P2
    optimal_path = room.optimize_within_channels(safe_channels)
    assert len(optimal_path) > 0, "Valid protocol execution should produce a path"
    
    # Verify path doesn't go through obstacles
    for point in optimal_path:
        assert point not in obstacles, "Optimal path should avoid obstacles"
```
