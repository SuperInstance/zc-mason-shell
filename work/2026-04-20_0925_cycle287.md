```python
def test_concurrent_tile_submission(self):
    """Test that room handles simultaneous tile submissions correctly."""
    room = DeadbandRoom()
    tiles = [
        self._create_valid_tile(priority="P0", content="Negative space A"),
        self._create_valid_tile(priority="P0", content="Negative space B"),
        self._create_valid_tile(priority="P1", content="Safe channel X")
    ]
    
    # Simulate concurrent submissions
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(str(e))
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify: All submissions processed, no data corruption
    self.assertEqual(len(results), 3)
    self.assertEqual(room.tile_count, 3)
    # Check that tile order is preserved or handled consistently
    self.assertTrue(all(r.get("status") == "accepted" for r in results if isinstance(r, dict)))
```

```python
def test_p0_to_p2_skip_violation(self):
    """Test that skipping from P0 directly to P2 is detected and rejected."""
    room = DeadbandRoom()
    
    # Submit valid P0 tile
    p0_tile = self._create_valid_tile(priority="P0", content="Avoid rocks")
    room.submit_tile(p0_tile)
    
    # Attempt to submit P2 tile without P1 (should fail)
    p2_tile = self._create_valid_tile(priority="P2", content="Optimize path")
    result = room.submit_tile(p2_tile)
    
    # Verify rejection with appropriate error
    self.assertEqual(result["status"], "rejected")
    self.assertIn("protocol violation", result.get("message", "").lower())
    self.assertIn("P1", result.get("message", ""))
    
    # Submit valid P1 tile, then P2 should work
    p1_tile = self._create_valid_tile(priority="P1", content="Find safe channel")
    room.submit_tile(p1_tile)
    result = room.submit_tile(p2_tile)
    self.assertEqual(result["status"], "accepted")
```

```python
def test_training_memory_limit(self):
    """Test room behavior when system hits memory limits during training."""
    room = DeadbandRoom()
    
    # Fill room with tiles to trigger training
    for i in range(1000):  # Large number to potentially exhaust resources
        tile = self._create_valid_tile(
            priority="P0" if i % 3 == 0 else "P1" if i % 3 == 1 else "P2",
            content=f"Test tile {i}"
        )
        room.submit_tile(tile)
    
    # Mock memory exhaustion during training
    import psutil
    original_virtual_memory = psutil.virtual_memory
    
    def mock_virtual_memory():
        class Memory:
            percent = 99.8  # Simulate 99.8% memory usage
            available = 1024 * 1024  # 1MB available
        return Memory()
    
    psutil.virtual_memory = mock_virtual_memory
    
    try:
        # Attempt to start training (should handle gracefully)
        training_result = room.start_training()
        
        # Verify: Room should enter error state or pause, not crash
        self.assertIn(room.state, ["error", "paused", "idle"])
        if training_result:
            self.assertIn("memory", training_result.get("message", "").lower())
    finally:
        psutil.virtual_memory = original_virtual_memory
```
