### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission():
    """Test that concurrent tile submissions don't corrupt room state."""
    room = DeadbandRoom()
    tiles = [
        {"phase": "P0", "content": "test1", "agent_id": "agent1"},
        {"phase": "P0", "content": "test2", "agent_id": "agent2"},
        {"phase": "P1", "content": "test3", "agent_id": "agent3"},
    ]
    
    # Simulate concurrent submissions
    import threading
    results = []
    
    def submit_tile(tile):
        try:
            room.submit_tile(tile)
            results.append(("success", tile["agent_id"]))
        except Exception as e:
            results.append(("error", str(e)))
    
    threads = [threading.Thread(target=submit_tile, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all tiles were processed
    assert len(room.tiles) == 3
    assert len(results) == 3
    assert all(r[0] == "success" for r in results)
```

def test_server_unreachable_backoff():
    """Test exponential backoff when tile server is unreachable."""
    room = DeadbandRoom()
    room.tile_server_url = "http://unreachable-server:9999"
    
    tile = {"phase": "P0", "content": "test", "agent_id": "test_agent"}
    
    # Mock requests to simulate failures
    import requests
    from unittest.mock import patch
    
    with patch('requests.post') as mock_post:
        mock_post.side_effect = requests.exceptions.ConnectionError("Server unreachable")
        
        # Should raise after max retries
        with pytest.raises(RuntimeError, match="Failed to submit tile after"):
            room.submit_tile(tile)
        
        # Verify backoff pattern was attempted
        assert mock_post.call_count == room.max_retries
        call_delays = [call[1].get('timeout', 0) for call in mock_post.call_args_list]
        assert all(delay > 0 for delay in call_delays)
```

### Test 4: State Persistence Across Soft Reset

def test_state_persistence_soft_reset():
    """Test that P0/P1 knowledge persists across room soft resets."""
    room = DeadbandRoom()
    
    # Submit P0 and P1 tiles
    p0_tile = {"phase": "P0", "content": "Avoid null pointers", "agent_id": "agent1"}
    p1_tile = {"phase": "P1", "content": "Use type hints", "agent_id": "agent2"}
    
    room.submit_tile(p0_tile)
    room.submit_tile(p1_tile)
    
    # Verify initial state
    assert len(room.negative_space) > 0
    assert len(room.safe_channels) > 0
    
    # Perform soft reset (clear current phase but keep knowledge)
    room.soft_reset()
    
    # P0 and P1 knowledge should persist
    assert len(room.negative_space) > 0, "P0 knowledge lost after soft reset"
    assert len(room.safe_channels) > 0, "P1 knowledge lost after soft reset"
    
    # Current phase should reset to P0
    assert room.current_phase == "P0"
    
    # Tiles should be archived
    assert len(room.archived_tiles) == 2
```

### Test 5: Protocol Phase Regression Detection
```python
def test_phase_regression_detection():
    """Test detection and prevention of protocol phase regression."""
    room = DeadbandRoom()
    
    # Normal progression P0 -> P1 -> P2
    room.submit_tile({"phase": "P0", "content": "P0 tile", "agent_id": "a1"})
    assert room.current_phase == "P0"
    
    room.submit_tile({"phase": "P1", "content": "P1 tile", "agent_id": "a2"})
    assert room.current_phase == "P1"
    
    room.submit_tile({"phase": "P2", "content": "P2 tile", "agent_id": "a3"})
    assert room.current_phase == "P2"
    
    # Attempt regression P2 -> P1 (should be rejected)
    with pytest.raises(ValueError, match="Protocol regression detected"):
        room.submit_tile({"phase": "P1", "content": "Attempt regression", "agent_id": "a4"})
    
    # Attempt regression P2 -> P0 (should also be rejected)
    with pytest.raises(ValueError, match="Protocol regression detected"):
        room.submit_tile({"phase": "P0", "content": "Attempt regression", "agent_id": "a5"})
    
    # Valid P2 continuation should work
    room.submit_tile({"phase": "P2", "content": "Another P2 tile", "agent_id": "a6"})
    assert room.current_phase == "P2"
    
    # Verify tile counts
    assert len(room.tiles) == 4  # 3 successful + 1 after regression attempts
```
