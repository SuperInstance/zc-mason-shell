def test_concurrent_tile_submission_race_condition():
    """Test DeadbandRoom handles simultaneous tile submissions from multiple agents."""
    room = DeadbandRoom(config={"max_concurrent_agents": 10})
    
    # Simulate 5 agents submitting tiles at the same "cycle"
    agent_ids = [f"agent_{i}" for i in range(5)]
    tiles = [
        Tile(agent_id=aid, content=f"P0 test from {aid}", protocol_phase="P0")
        for aid in agent_ids
    ]
    
    # Submit all tiles concurrently (simulated)
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # All submissions should be processed without corruption
    assert all(r["status"] == "accepted" for r in results), "Concurrent submissions failed"
    assert len(room.get_tile_history()) == 5, "Tile count mismatch after concurrent submission"
    
    # Verify tile ordering or timestamp consistency
    timestamps = [r["timestamp"] for r in results]
    assert len(set(timestamps)) == 5, "Timestamp collisions detected"

def test_protocol_phase_transition_validation():
    """Test DeadbandRoom enforces P0→P1→P2 strict progression."""
    room = DeadbandRoom()
    agent_id = "test_agent_phase"
    
    # Valid progression: P0 → P1 → P2
    p0_tile = Tile(agent_id=agent_id, content="Mapping negative space", protocol_phase="P0")
    p1_tile = Tile(agent_id=agent_id, content="Finding safe channels", protocol_phase="P1")
    p2_tile = Tile(agent_id=agent_id, content="Optimizing within channels", protocol_phase="P2")
    
    assert room.submit_tile(p0_tile)["status"] == "accepted"
    assert room.submit_tile(p1_tile)["status"] == "accepted"
    assert room.submit_tile(p2_tile)["status"] == "accepted"
    
    # Invalid: Attempt to submit P2 without completing P1
    room2 = DeadbandRoom()
    p2_only = Tile(agent_id=agent_id, content="Skipping to optimization", protocol_phase="P2")
    result = room2.submit_tile(p2_only)
    
    assert result["status"] == "rejected"
    assert "P1 not completed" in result.get("reason", ""), "Should reject P2 without P1"
    
    # Invalid: Regression from P2 back to P0
    room3 = DeadbandRoom()
    room3.submit_tile(p0_tile)
    room3.submit_tile(p1_tile)
    room3.submit_tile(p2_tile)
    
    # Now try to submit another P0 tile (regression)
    p0_again = Tile(agent_id=agent_id, content="Trying to go back", protocol_phase="P0")
    result = room3.submit_tile(p0_again)
    
    assert result["status"] == "rejected"
    assert "protocol regression" in result.get("reason", "").lower()

def test_resource_exhaustion_handling():
    """Test DeadbandRoom gracefully handles memory/disk limits."""
    # Configure room with artificially low limits
    config = {
        "max_tiles_per_agent": 100,
        "max_total_tiles": 1000,
        "max_tile_size_bytes": 1024  # 1KB per tile
    }
    room = DeadbandRoom(config=config)
    
    # Fill room to capacity with valid tiles
    for i in range(1000):
        tile = Tile(
            agent_id=f"agent_{i % 10}",
            content=f"Test tile {i}" * 10,  # Ensure content isn't too small
            protocol_phase="P0"
        )
        result = room.submit_tile(tile)
        
        if i >= 999:  # Last tile should hit limit
            if result["status"] == "rejected":
                assert "capacity" in result.get("reason", "").lower()
                break
    
    # Verify room maintains operational state after hitting limits
    assert room.get_room_state()["status"] == "operational"
    
    # Test oversized tile rejection
    oversized_content = "X" * 2048  # 2KB > 1KB limit
    oversized_tile = Tile(
        agent_id="oversize_agent",
        content=oversized_content,
        protocol_phase="P0"
    )
    result = room.submit_tile(oversized_tile)
    assert result["status"] == "rejected"
    assert "size" in result.get("reason", "").lower()
```

def test_network_partition_partial_data_recovery():
    """Test DeadbandRoom handles incomplete tile data and recovers state."""
    room = DeadbandRoom()
    
    # Simulate partial tile data (missing required fields)
    partial_tiles = [
        {"agent_id": "agent1"},  # Missing content and protocol_phase
        {"content": "Some content"},  # Missing agent_id
        {"protocol_phase": "P0"},  # Missing agent_id and content
        None,  # Null tile
        "",  # Empty string
        {"agent_id": "agent2", "content": "Valid", "protocol_phase": "INVALID"}  # Invalid phase
    ]
    
    rejection_count = 0
    for partial in partial_tiles:
        try:
            # Attempt to create Tile object (should fail validation)
            if partial is None:
                tile = None
            else:
                tile = Tile(**partial) if isinstance(partial, dict) else Tile(content=partial)
            
            result = room.submit_tile(tile)
            if result["status"] == "rejected":
                rejection_count += 1
        except (TypeError, ValueError):
            # Tile constructor validation failed
            rejection_count += 1
    
    # All partial/invalid tiles should be rejected
    assert rejection_count == len(partial_tiles)
    
    # Room should maintain clean state despite garbage inputs
    assert room.get_room_state()["error_count"] == 0  # Errors logged but not fatal
    assert room.get_tile_history() == []  # No valid tiles accepted
```
