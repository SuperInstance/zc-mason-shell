def test_ghost_tile_inherits_p0_knowledge(self):
    """Test that ghost tiles from terminated agents preserve P0 violations."""
    room = DeadbandRoom()
    
    # Create a ghost tile from a terminated agent
    ghost_tile = {
        "agent_id": "terminated_agent_001",
        "timestamp": "2026-04-20T10:00:00Z",
        "phase": "P0",
        "content": "Attempted to optimize before mapping channels (P2 before P1)",
        "violation": True,
        "ghost": True
    }
    
    # Submit ghost tile
    result = room.process_tile(ghost_tile)
    
    # Verify ghost tile is stored in P0 knowledge base
    assert result["status"] == "accepted"
    assert "ghost" in room.p0_knowledge_base
    assert ghost_tile["content"] in room.p0_knowledge_base["ghost"]
    
    # Verify new agents receive ghost warnings
    new_agent_tile = {
        "agent_id": "new_agent_001",
        "phase": "P2",
        "content": "Optimizing path without channel mapping"
    }
    validation = room.validate_tile(new_agent_tile)
    assert "ghost_warning" in validation
    assert "P2 before P1" in validation["ghost_warning"]

def test_mirror_play_screen_synchronization(self):
    """Test that two PLATO systems can iterate screen-to-screen correctly."""
    room_a = DeadbandRoom(preset="plato_system_a")
    room_b = DeadbandRoom(preset="plato_system_b")
    
    # Initial screen state from system A
    screen_a = {
        "system": "A",
        "iteration": 1,
        "tiles": ["P0: Avoid rocks", "P1: Find channels"],
        "checksum": "abc123"
    }
    
    # Mirror to system B
    mirror_result = room_a.mirror_to(screen_a, room_b)
    assert mirror_result["status"] == "mirrored"
    assert room_b.current_screen["system"] == "A"
    assert room_b.current_screen["iteration"] == 1
    
    # System B processes and responds
    screen_b = room_b.process_mirror(screen_a)
    assert screen_b["system"] == "B"
    assert screen_b["iteration"] == 2  # Incremented
    assert "response_tiles" in screen_b

    corrupted_screen = screen_a.copy()
    corrupted_screen["checksum"] = "invalid"
    with pytest.raises(ValueError, match="Checksum mismatch"):
        room_b.process_mirror(corrupted_screen)

### Test 4: Concurrent Tile Submission Race Conditions
```python
def test_concurrent_tile_submission_thread_safety(self):
    """Test that multiple agents can submit tiles concurrently without corruption."""
    import threading
    import queue
    
    room = DeadbandRoom()
    results = queue.Queue()
    errors = []
    
    def submit_tiles(agent_id, count):
        for i in range(count):
            tile = {
                "agent_id": agent_id,
                "phase": "P1",
                "content": f"Channel finding attempt {i}",
                "timestamp": f"2026-04-20T10:{i:02d}:00Z"
            }
            try:
                result = room.process_tile(tile)
                results.put((agent_id, i, result))
            except Exception as e:
                errors.append((agent_id, i, str(e)))
    
    # Create 10 agents submitting 5 tiles each concurrently
    threads = []
    for agent_num in range(10):
        t = threading.Thread(
            target=submit_tiles,
            args=(f"agent_{agent_num}", 5)
        )
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # Verify no errors occurred
    assert len(errors) == 0, f"Errors during concurrent submission: {errors}"
    
    # Verify all tiles were processed
    processed_tiles = []
    while not results.empty():
        agent_id, i, result = results.get()
        processed_tiles.append((agent_id, i))
        assert result["status"] == "accepted"
    
    # Should have 50 tiles total (10 agents × 5 tiles)
    assert len(processed_tiles) == 50
    
    # Verify agent states are consistent
    for agent_num in range(10):
        agent_state = room.get_agent_state(f"agent_{agent_num}")
        assert agent_state["tile_count"] == 5
        assert agent_state["last_phase"] == "P1"
```
