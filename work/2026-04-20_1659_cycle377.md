## Identified Edge Cases (P0 - Negative Space)
From reading the room implementation, these scenarios are not explicitly tested:
1. **Constraint nesting**: P1/P2 constraints containing nested dicts/lists.
2. **Tile sentiment overflow**: Tile `sentiment` field exceeding expected bounds (-1.0 to 1.0).
3. **Concurrent training cycles**: Calling `train_cycle()` multiple times without new tiles.
4. **Malformed metadata**: Tile `metadata` field with unserializable types.
5. **State persistence**: Room state after ensign export (should it reset?).

def test_repeated_training_no_tiles():
    """Calling train_cycle() multiple times without new tiles should handle gracefully."""
    config = {
        "room_id": "test_repeat",
        "p0_constraint": {"disallowed": []},
        "p1_constraint": {"allowed": ["test"]},
        "p2_constraint": {"optimize": "test"}
    }
    room = DeadbandRoom(config)
    
    # Initial training with tiles
    tiles = [
        {"content": "test1", "sentiment": 0.5, "metadata": {}},
        {"content": "test2", "sentiment": -0.2, "metadata": {}},
    ]
    for tile in tiles:
        room.submit_tile(tile)
    
    first_loss = room.train_cycle()
    assert isinstance(first_loss, float)
    
    # Second training without new tiles
    # Implementation may raise RuntimeError or return same loss
    try:
        second_loss = room.train_cycle()
        assert isinstance(second_loss, float)
        # Loss might be same or NaN
    except RuntimeError as e:
        assert "insufficient" in str(e).lower() or "no new" in str(e).lower()

### Test 5: State After Ensign Export
```python
def test_state_after_ensign_export():
    """Room state should be reset or preserved after exporting ensign adapter?"""
    config = {
        "room_id": "test_export",
        "p0_constraint": {"disallowed": []},
        "p1_constraint": {"allowed": ["test"]},
        "p2_constraint": {"optimize": "test"}
    }
    room = DeadbandRoom(config)
    
    # Add tiles and train
    for i in range(3):
        room.submit_tile({"content": f"test{i}", "sentiment": 0.1*i, "metadata": {}})
    
    room.train_cycle()
    initial_state = room.training_cycles
    
    # Export ensign
    adapter = room.export_ensign()
    assert adapter is not None
    assert hasattr(adapter, "apply_to")  # Basic adapter interface
    
    # Check state
    # Implementation may reset tile_buffer or keep it
    # Both are valid, but should be consistent
    if hasattr(room, 'tile_buffer'):
        # Buffer might be cleared or kept
        assert isinstance(room.tile_buffer, list)
    
    # Training cycles should be preserved or incremented
    assert room.training_cycles >= initial_state
```
