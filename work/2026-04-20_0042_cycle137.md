### Test 2: Concurrent Agent Tile Submission
```python
def test_concurrent_agent_submissions():
    """Test room handles multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agents = [f"agent_{i}" for i in range(3)]
    
    # Simulate concurrent submissions
    tiles = []
    for i, agent_id in enumerate(agents):
        tile = {
            "agent_id": agent_id,
            "content": f"P0 negative space mapping {i}",
            "phase": "P0",
            "timestamp": datetime.now().isoformat()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        results.append(room.submit_tile(tile))
    
    # All should be accepted
    assert all(r["accepted"] for r in results)
    
    # All agents should be at P1 after P0 completion
    for agent_id in agents:
        assert room.get_agent_phase(agent_id) == "P1"
```

```python
def test_agent_identity_validation():
    """Test room validates agent identity and prevents spoofing."""
    room = DeadbandRoom()
    legitimate_agent = "legit_agent"
    spoofing_agent = "spoof_agent"
    
    # Legitimate agent submits P0 tile
    legit_tile = {
        "agent_id": legitimate_agent,
        "content": "Legitimate P0 work",
        "phase": "P0",
        "timestamp": datetime.now().isoformat()
    }
    
    result = room.submit_tile(legit_tile)
    assert result["accepted"]
    
    # Spoofing agent tries to submit tile as legitimate agent
    spoof_tile = {
        "agent_id": legitimate_agent,  # Using legitimate agent's ID
        "content": "Malicious P1 content",
        "phase": "P1",
        "timestamp": datetime.now().isoformat()
    }
    
    # This should fail if room tracks submission source
    result = room.submit_tile(spoof_tile)
    
    # Either rejected or accepted but tracked as different source
    if not result["accepted"]:
        assert "identity" in result["message"].lower() or "validation" in result["message"].lower()
    else:
        # If accepted, check that room tracks the actual source
        tiles = room.get_agent_tiles(legitimate_agent)
        assert len(tiles) == 2
        # Last tile should have metadata indicating different source
        assert "submission_source" in tiles[-1] or "ip_address" in tiles[-1]
```
