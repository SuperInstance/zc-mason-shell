```python
def test_concurrent_tile_submission(self):
    """Test multiple agents submitting tiles simultaneously."""
    room = DeadbandRoom()
    agent_ids = [f"agent_{i}" for i in range(5)]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "content": f"Test from {agent_id}",
            "phase": "P1",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles "concurrently" (simulated)
    results = []
    for tile in tiles:
        try:
            result = room.submit_tile(tile)
            results.append(result)
        except Exception as e:
            results.append(f"Error: {e}")
    
    # Verify all submissions were processed
    assert len(results) == len(agent_ids)
    # Verify no data corruption occurred
    stored_tiles = room.get_recent_tiles(count=5)
    assert len(stored_tiles) == 5
```

### Test 3: Protocol Violation Sequence Detection
```python
def test_protocol_violation_sequences(self):
    """Test detection of P0→P2 skipping violations."""
    room = DeadbandRoom()
    agent_id = "protocol_test_agent"
    
    # Attempt P2 without completing P1
    p2_tile = {
        "agent_id": agent_id,
        "content": "Attempting P2 optimization",
        "phase": "P2",
        "timestamp": time.time()
    }
    
    # Should reject or flag violation
    try:
        result = room.submit_tile(p2_tile)
        # If accepted, check if violation was logged
        if hasattr(room, 'violation_log'):
            violations = room.violation_log.get(agent_id, [])
            assert any("P2 without P1" in str(v) for v in violations)
    except ProtocolViolationError:
        # Expected rejection
        pass
    
    # Now submit proper P0→P1→P2 sequence
    p0_tile = {
        "agent_id": agent_id,
        "content": "P0: Negative space mapping",
        "phase": "P0",
        "timestamp": time.time()
    }
    room.submit_tile(p0_tile)
    
    p1_tile = {
        "agent_id": agent_id,
        "content": "P1: Safe channel identification",
        "phase": "P1",
        "timestamp": time.time()
    }
    room.submit_tile(p1_tile)
    
    # Now P2 should be accepted
    p2_tile["timestamp"] = time.time()
    result = room.submit_tile(p2_tile)
    assert result["status"] == "accepted"
```

### Test 4: Resource Exhaustion Handling
```python
def test_resource_exhaustion_handling(self):
    """Test behavior when storage limits are reached."""
    room = DeadbandRoom()
    
    # Simulate filling up tile storage
    max_tiles = getattr(room, 'max_tiles', 1000)  # Get actual limit or use default
    
    # Submit tiles until limit (or near limit)
    for i in range(max_tiles + 10):
        tile = {
            "agent_id": f"flood_agent_{i}",
            "content": f"Test tile {i}",
            "phase": "P0",
            "timestamp": time.time()
        }
        
        if i >= max_tiles:
            # Beyond limit - should handle gracefully
            try:
                result = room.submit_tile(tile)
                # If accepted, check cleanup occurred
                if result["status"] == "accepted":
                    current_count = len(room.get_recent_tiles(count=max_tiles*2))
                    assert current_count <= max_tiles * 1.1  # Allow 10% buffer
            except ResourceExhaustedError:
                # Expected when limits are strictly enforced
                pass
        else:
            # Within limits - should accept
            result = room.submit_tile(tile)
            assert result["status"] == "accepted"
    
    # Verify room still functional after exhaustion test
    functional_tile = {
        "agent_id": "recovery_agent",
        "content": "Post-exhaustion test",
        "phase": "P1",
        "timestamp": time.time()
    }
    result = room.submit_tile(functional_tile)
    assert result["status"] in ["accepted", "queued"]
```

### Test 5: Network Partition Simulation
```python
def test_network_partition_resilience(self):
    """Test room behavior during simulated network issues."""
    room = DeadbandRoom()
    
    # Simulate intermittent connectivity
    connection_states = [True, False, True, False, True]  # Flapping connection
    
    submission_results = []
    for i, connected in enumerate(connection_states):
        tile = {
            "agent_id": f"partition_agent_{i}",
            "content": f"Tile during connection state: {connected}",
            "phase": "P0",
            "timestamp": time.time()
        }
        
        if not connected:
            # Simulate network failure
            with pytest.raises(ConnectionError) as exc_info:
                # Mock network failure in submission
                room._simulate_network_failure = True
                room.submit_tile(tile)
            assert "network" in str(exc_info.value).lower()
            room._simulate_network_failure = False
        else:
            # Normal submission
            result = room.submit_tile(tile)
            submission_results.append(result)
    
    # Verify accepted tiles persisted
    accepted_count = sum(1 for r in submission_results if r.get("status") == "accepted")
    stored_tiles = room.get_recent_tiles(count=10)
    
    # Should have at least the accepted tiles stored
    assert len(stored_tiles) >= accepted_count
    
    # Verify no data corruption from partition events
    for tile in stored_tiles:
        assert "agent_id" in tile
        assert "content" in tile
        assert tile["phase"] in ["P0", "P1", "P2"]
```
