### Test 1: Concurrent Tile Submission Race Condition
```rust
#[tokio::test]
async fn test_concurrent_tile_submission_race() {
    use std::sync::Arc;
    use tokio::sync::Barrier;
    
    let room = DeadbandRoom::new();
    let room = Arc::new(tokio::sync::Mutex::new(room));
    let barrier = Arc::new(Barrier::new(5));
    
    let mut handles = vec![];
    for i in 0..5 {
        let room_clone = Arc::clone(&room);
        let barrier_clone = Arc::clone(&barrier);
        
        handles.push(tokio::spawn(async move {
            barrier_clone.wait().await;
            
            let tile = Tile {
                agent_id: format!("agent_{}", i),
                content: format!("Test tile from agent {}", i),
                priority: 1,
                timestamp: chrono::Utc::now(),
            };
            
            let mut room = room_clone.lock().await;
            match room.submit_tile(tile).await {
                Ok(_) => println!("Agent {} submitted successfully", i),
                Err(e) => println!("Agent {} failed: {}", i, e),
            }
        }));
    }
    
    for handle in handles {
        handle.await.unwrap();
    }
    
    let room = room.lock().await;
    assert_eq!(room.get_tile_count(), 5, "All concurrent submissions should be processed");
}
```

### Test 2: Network Partition with Partial Write
```rust
#[test]
fn test_partial_write_recovery() {
    use std::io::{self, Write};
    use tempfile::NamedTempFile;
    
    // Simulate partial file write (like network failure mid-write)
    let mut temp_file = NamedTempFile::new().unwrap();
    let valid_data = b"{\"agent_id\":\"test\",\"content\":\"complete tile\"}";
    let partial_data = &valid_data[0..valid_data.len() / 2]; // Cut in half
    
    temp_file.write_all(partial_data).unwrap();
    temp_file.flush().unwrap();
    
    // Try to load room from corrupted state
    let path = temp_file.path();
    let result = DeadbandRoom::load_from_file(path);
    
    // Should either fail cleanly or recover gracefully
    match result {
        Ok(_) => panic!("Should not successfully load corrupted file"),
        Err(e) => {
            // Verify error is descriptive
            assert!(
                e.to_string().contains("corrupt") || 
                e.to_string().contains("parse") ||
                e.to_string().contains("incomplete"),
                "Error should indicate corruption: {}",
                e
            );
        }
    }
}
```

### Test 4: Extreme Load with Batch Submission
```rust
#[tokio::test]
async fn test_extreme_batch_load() {
    let mut room = DeadbandRoom::new();
    let batch_size = 10_000;
    let mut successful = 0;
    let mut failed = 0;
    
    // Submit large batch
    for i in 0..batch_size {
        let tile = Tile {
            agent_id: format!("load_agent_{}", i % 100), // Reuse agent IDs
            content: format!("Load test tile {}", i),
            priority: i % 3, // Distribute across P0, P1, P2
            timestamp: chrono::Utc::now(),
        };
        
        match room.submit_tile(tile).await {
            Ok(_) => successful += 1,
            Err(e) => {
                failed += 1;
                // Log first few failures for debugging
                if failed <= 5 {
                    println!("Failure {}: {}", failed, e);
                }
            }
        }
    }
    
    println!("Batch results: {} successful, {} failed", successful, failed);
    
    // System should handle load without crashing
    assert!(successful > 0, "Should process at least some tiles");
    
    // If failures occur, they should be for identifiable reasons
    if failed > 0 {
        let tile_count = room.get_tile_count();
        assert!(
            tile_count == successful,
            "Tile count should match successful submissions"
        );
        
        // Check memory usage isn't pathological
        let estimated_memory = tile_count * 1024; // Rough estimate
        assert!(
            estimated_memory < 512 * 1024 * 1024, // 512MB reasonable limit
            "Memory usage should be reasonable for {} tiles",
            tile_count
        );
    }
}
```

### Test 5: State Corruption and Rollback Recovery
```rust
#[test]
fn test_state_rollback_mechanism() {
    use std::fs;
    use std::path::PathBuf;
    
    // Create test directory
    let test_dir = tempfile::tempdir().unwrap();
    let state_file = test_dir.path().join("room_state.json");
    let backup_file = test_dir.path().join("room_state.json.bak");
    
    // Create initial valid state
    let mut room = DeadbandRoom::new();
    let tile = Tile {
        agent_id: "agent1".to_string(),
        content: "Valid tile before corruption".to_string(),
        priority: 1,
        timestamp: chrono::Utc::now(),
    };
    
    room.submit_tile(tile).unwrap();
    room.save_to_file(&state_file).unwrap();
    
    // Create backup
    fs::copy(&state_file, &backup_file).unwrap();
    
    // Corrupt the primary state file
    fs::write(&state_file, b"{{{ corrupted json }}}").unwrap();
    
    // Attempt to load - should detect corruption
    let load_result = DeadbandRoom::load_from_file(&state_file);
    assert!(load_result.is_err(), "Should fail to load corrupted file");
    
    // Test rollback mechanism (if implemented)
    if backup_file.exists() {
        let rollback_result = DeadbandRoom::load_from_file(&backup_file);
        assert!(
            rollback_result.is_ok(),
            "Should successfully load from backup"
        );
        
        let recovered_room = rollback_result.unwrap();
        assert_eq!(
            recovered_room.get_tile_count(),
            1,
            "Recovered room should have pre-corruption state"
        );
    }
    
    // Cleanup
    drop(test_dir);
}
```
