These are positive-path tests. The Deadband Protocol's strict priority (P0→P1→P2, no skipping) and the room's role as a training tile accumulator suggest several edge cases that should be validated.

## Edge Cases Identified (P0 Negative Space)
1. **Invalid Protocol Skipping**: Attempting P2 before P1, or P1 before P0.
2. **Malformed Tile Submission**: Tiles missing required fields, or with incorrect sentiment/confidence scoring.
3. **Resource Exhaustion**: Room memory limits, tile storage capacity.
4. **Concurrent Access**: Simulated multi-agent tile submission race conditions.
5. **State Corruption Recovery**: Room interrupted mid-training (simulated crash), then restored.

## Implementation: 5 Edge-Case Tests

**File:** `plato-torch/tests/test_deadband_room_edge.py`

```python
"""
Edge-case tests for DeadbandRoom.
Focuses on protocol violations, error conditions, and resilience.
"""

import pytest
import asyncio
from unittest.mock import patch, AsyncMock
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.core.tile import TrainingTile

class TestDeadbandRoomEdgeCases:
    """Test suite for DeadbandRoom edge cases."""

    def test_protocol_skip_raises_error(self):
        """P0: Attempting to execute P2 before P1 should raise ProtocolError."""
        room = DeadbandRoom(room_id="edge_skip", max_tiles=100)
        # Force state to P0 complete, but skip P1
        room.current_phase = "P0"
        room.phase_complete = {"P0": True, "P1": False, "P2": False}
        
        with pytest.raises(ProtocolError, match="Cannot execute P2 before P1"):
            room.execute_phase("P2")

    def test_malformed_tile_rejection(self):
        """P0: Tile missing required 'content' field should be rejected."""
        room = DeadbandRoom(room_id="edge_malformed", max_tiles=100)
        malformed_tile = {
            "agent_id": "test_agent",  # Missing 'content'
            "sentiment": 0.5,
            "confidence": 0.8
        }
        
        result = room.submit_tile(malformed_tile)
        assert result["accepted"] is False
        assert "missing required field" in result["reason"].lower()

    @patch('plato_torch.storage.TileStorage.save')
    def test_tile_capacity_enforced(self, mock_save):
        """P0: Submission beyond max_tiles should be rejected."""
        max_tiles = 5
        room = DeadbandRoom(room_id="edge_capacity", max_tiles=max_tiles)
        
        # Fill room to capacity
        for i in range(max_tiles):
            tile = TrainingTile(
                content=f"Tile {i}",
                agent_id="test_agent",
                sentiment=0.5,
                confidence=0.7
            )
            room.submit_tile(tile.to_dict())
        
        # Attempt one more tile
        extra_tile = TrainingTile(
            content="Excess tile",
            agent_id="test_agent",
            sentiment=0.5,
            confidence=0.7
        )
        result = room.submit_tile(extra_tile.to_dict())
        
        assert result["accepted"] is False
        assert "capacity" in result["reason"].lower()
        assert mock_save.call_count == max_tiles  # Only saves up to limit

    async def test_concurrent_tile_submission(self):
        """P1: Multiple agents submitting tiles concurrently should not corrupt state."""
        room = DeadbandRoom(room_id="edge_concurrent", max_tiles=100)
        num_agents = 10
        tiles_per_agent = 5
        
        async def submit_tiles(agent_id):
            for i in range(tiles_per_agent):
                tile = TrainingTile(
                    content=f"Tile from {agent_id}-{i}",
                    agent_id=agent_id,
                    sentiment=0.5,
                    confidence=0.7
                )
                await asyncio.sleep(0.01)  # Simulate network delay
                room.submit_tile(tile.to_dict())
        
        # Simulate concurrent submissions
        tasks = [submit_tiles(f"agent_{i}") for i in range(num_agents)]
        await asyncio.gather(*tasks)
        
        # Verify all tiles were recorded
        expected_total = num_agents * tiles_per_agent
        assert len(room.tiles) == expected_total
        
        # Verify no duplicate tile IDs (check for corruption)
        tile_ids = [tile["tile_id"] for tile in room.tiles]
        assert len(tile_ids) == len(set(tile_ids)), "Duplicate tile IDs indicate race condition"

    def test_state_recovery_after_interruption(self):
        """P1: Room should recover valid state after simulated interruption."""
        room = DeadbandRoom(room_id="edge_recovery", max_tiles=100)
        
        # Simulate partial training progress
        room.current_phase = "P1"
        room.phase_complete = {"P0": True, "P1": False, "P2": False}
        room.tiles.append(TrainingTile(
            content="Partial work",
            agent_id="test_agent",
            sentiment=0.5,
            confidence=0.7
        ).to_dict())
        
        # Simulate interruption: corrupt internal state
        room._training_lock = None  # This would normally cause failure
        room._phase_history = []    # Clear history
        
        # Attempt recovery via re-initialization
        recovered_room = DeadbandRoom.from_state(room.get_state())
        
        # Verify essential state restored
        assert recovered_room.room_id == room.room_id
        assert recovered_room.current_phase == "P1"
        assert len(recovered_room.tiles) == 1
        assert recovered_room.tiles[0]["content"] == "Partial work"
        # P2 optimization should remain unavailable until P1 completes
        assert recovered_room.phase_complete["P2"] is False
```
