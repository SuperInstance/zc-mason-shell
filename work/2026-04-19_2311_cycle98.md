```python
def test_max_tile_capacity(self):
    """Test behavior when room reaches maximum tile capacity."""
    room = DeadbandRoom(name="test_room", max_tiles=3)
    
    # Fill room to capacity
    for i in range(3):
        room.submit_tile(agent_id=f"agent_{i}", content=f"Tile {i}")
    
    # Attempt to exceed capacity
    result = room.submit_tile(agent_id="overflow_agent", content="Should be rejected")
    self.assertFalse(result["accepted"])
    self.assertIn("capacity", result["reason"].lower())
    self.assertEqual(len(room.tiles), 3)
```

### Test 4: Concurrent Tile Submission Race Condition
```python
def test_concurrent_submission_race(self):
    """Test handling of near-simultaneous tile submissions."""
    import threading
    
    room = DeadbandRoom(name="test_room", max_tiles=100)
    results = []
    
    def submit_tile_thread(agent_id, content):
        results.append(room.submit_tile(agent_id, content))
    
    threads = []
    for i in range(10):
        t = threading.Thread(target=submit_tile_thread, 
                           args=(f"agent_{i}", f"Concurrent tile {i}"))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All submissions should be processed
    self.assertEqual(len(results), 10)
    
    # Check for duplicate agent submissions (should be prevented)
    agent_ids = [r.get("agent_id") for r in results if r.get("accepted")]
    self.assertEqual(len(agent_ids), len(set(agent_ids)))  # No duplicates
```

```python
def test_malformed_tile_structure(self):
    """Test validation of tiles with missing required fields."""
    room = DeadbandRoom(name="test_room", max_tiles=100)
    
    test_cases = [
        {"content": "Missing phase"},  # Missing phase field
        {"phase": "P1"},  # Missing content
        {},  # Empty tile
        {"phase": "P0", "content": "Valid", "extra_field": "unexpected"},  # Extra field
    ]
    
    for i, tile in enumerate(test_cases):
        result = room.validate_tile(tile)
        self.assertFalse(result["valid"], f"Test case {i} should be invalid")
        self.assertTrue(len(result["errors"]) > 0)
```
