### Test 1: Concurrent Tile Submission Race Condition
**Scenario:** Multiple agents attempt to submit tiles simultaneously to the same room.
**Edge:** Thread safety of the tile queue and state validation.
```python
def test_concurrent_tile_submission(self):
    """DeadbandRoom should handle concurrent tile submissions without corruption."""
    room = DeadbandRoom(room_id="concurrent_test")
    tiles = [
        {"agent": "agent1", "phase": "P0", "content": "Avoid rock A"},
        {"agent": "agent2", "phase": "P1", "content": "Channel B clear"},
        {"agent": "agent3", "phase": "P2", "content": "Optimize path C"},
    ]
    
    # Simulate concurrent submissions
    import threading
    results = []
    def submit(tile):
        results.append(room.submit_tile(tile))
    
    threads = [threading.Thread(target=submit, args=(tile,)) for tile in tiles]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # Verify all tiles were accepted and queue order is maintained
    assert all(results), "All concurrent submissions should succeed"
    assert len(room.tile_queue) == 3, "Queue should contain all submitted tiles"
    # Additional check: no duplicate or missing tiles
    queued_contents = [tile["content"] for tile in room.tile_queue]
    expected_contents = ["Avoid rock A", "Channel B clear", "Optimize path C"]
    assert sorted(queued_contents) == sorted(expected_contents), "All tiles should be preserved"
```

def test_malformed_tile_recovery(self):
    """DeadbandRoom should reject malformed tiles and maintain operational state."""

    assert room.tile_queue[0]["content"] == "Valid", "Valid tile should be preserved"
