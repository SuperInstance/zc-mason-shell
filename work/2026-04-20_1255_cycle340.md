```python
import pytest
from unittest.mock import Mock, patch
from plato_torch.rooms.deadband_room import DeadbandRoom
from plato_torch.exceptions import RoomCapacityError, ValidatorUnavailableError

def test_concurrent_tile_submission(deadband_room, sample_tile):
    """Edge case: Multiple agents submitting tiles simultaneously."""
    from concurrent.futures import ThreadPoolExecutor
    
    # Create multiple tile copies with different agent IDs
    tiles = []
    for i in range(3):
        tile_copy = sample_tile.copy()
        tile_copy['agent_id'] = f'agent_{i}'
        tile_copy['tile_id'] = f'tile_{i}'
        tiles.append(tile_copy)
    
    # Submit all tiles concurrently
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = [executor.submit(deadband_room.submit_tile, tile) 
                  for tile in tiles]
        results = [f.result() for f in futures]
    
    # All submissions should succeed
    assert all(r['status'] == 'accepted' for r in results)
    assert len(deadband_room.submitted_tiles) == 3

def test_validator_service_unavailable(deadband_room, sample_tile):
    """Edge case: Tile validator service is unavailable."""
    # Mock the validator to raise connection error
    with patch.object(deadband_room.validator, 'validate', 
                     side_effect=ValidatorUnavailableError("Service unreachable")):
        result = deadband_room.submit_tile(sample_tile)
        
        assert result['status'] == 'rejected'
        assert 'validator_unavailable' in result['reason']
        assert deadband_room.metrics['validator_errors'] == 1

def test_malformed_protocol_step_data():
    """Edge case: Protocol steps with contradictory constraints."""
    # Create room with P0 that contradicts P1
    config = {
        'max_agents': 5,
        'protocol_steps': [
            {'priority': 0, 'constraints': {'action': 'avoid_rocks'}},
            {'priority': 1, 'constraints': {'action': 'approach_rocks'}}  # Contradicts P0
        ]
    }
    
    room = DeadbandRoom(config)
    
    # Room should still initialize but log warning
    assert room is not None
    assert len(room.protocol_steps) == 2
    # Implementation detail: check if warning was logged
    # (would require capturing logs in actual test)

def test_room_capacity_exceeded():
    """Edge case: More agents than room capacity."""
    config = {'max_agents': 2}
    room = DeadbandRoom(config)
    
    # Register 2 agents (at capacity)
    room.register_agent('agent_1')
    room.register_agent('agent_2')
    
    # Third registration should fail
    with pytest.raises(RoomCapacityError) as exc_info:
        room.register_agent('agent_3')
    
    assert "Capacity exceeded" in str(exc_info.value)
    assert len(room.active_agents) == 2

def test_state_persistence_after_full_protocol(deadband_room):
    """Edge case: Room state after completing all protocol steps."""
    # Run through all protocol steps
    for step in deadband_room.protocol_steps:
        deadband_room.execute_step(step['priority'])
    
    # Room should be in "completed" state
    assert deadband_room.state == 'completed'
    
    # Reset room for next cycle
    deadband_room.reset()
    
    # Verify clean state but preserved configuration
    assert deadband_room.state == 'ready'
    assert len(deadband_room.submitted_tiles) == 0
    assert len(deadband_room.active_agents) == 0
    assert deadband_room.protocol_steps is not None  # Config persists
```
