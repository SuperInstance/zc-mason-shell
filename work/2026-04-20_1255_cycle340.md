### Test 1: Concurrent Tile Submission
```python
def test_concurrent_tile_submission(self):
    """Multiple agents submitting tiles simultaneously should not corrupt room state."""
    room = DeadbandRoom()
    agent_ids = ["agent_1", "agent_2", "agent_3"]
    
    # Simulate concurrent submissions
    tiles = []
    for agent_id in agent_ids:
        tile = {
            "agent_id": agent_id,
            "priority": "P1",
            "action": f"test_action_{agent_id}",
            "timestamp": time.time()
        }
        tiles.append(tile)
    
    # Submit all tiles
    results = []
    for tile in tiles:
        result = room.submit_tile(tile)
        results.append(result)
    
    # Verify all submissions succeeded
    assert all(r["status"] == "accepted" for r in results)
    
    # Verify room state contains all tiles
    room_state = room.get_state()
    submitted_agent_ids = {t["agent_id"] for t in room_state["recent_tiles"]}
    assert submitted_agent_ids == set(agent_ids)
```

### Test 2: Malformed Tile Data Handling
```python
def test_malformed_tile_data(self):
    """Room should reject tiles with invalid structure and provide clear error."""
    room = DeadbandRoom()
    
    test_cases = [
        # (tile_data, expected_error_keyword)
        ("not json at all", "JSON"),
        ({"priority": "P1"}, "missing"),  # Missing agent_id
        ({"agent_id": "test", "priority": "P5"}, "invalid priority"),  # Invalid priority
        ({"agent_id": "test", "priority": "P1", "extra": "field" * 10000}, "too large"),  # Oversized
    ]
    
    for tile_data, error_keyword in test_cases:
        try:
            result = room.submit_tile(tile_data)
            # If we get here, the tile was accepted unexpectedly
            assert False, f"Tile should have been rejected: {tile_data}"
        except (ValueError, TypeError, RuntimeError) as e:
            error_msg = str(e).lower()
            assert error_keyword in error_msg, \
                f"Error message '{error_msg}' should contain '{error_keyword}'"
```

### Test 3: Protocol Violation Recovery
```python
def test_protocol_violation_recovery(self):
    """Agent skipping P0 should be corrected, not blocked from future submissions."""
    room = DeadbandRoom()
    agent_id = "protocol_violator"
    
    # First tile: Skip P0, go directly to P2 (violation)
    violation_tile = {
        "agent_id": agent_id,
        "priority": "P2",
        "action": "optimize_before_mapping",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(violation_tile)
    assert result["status"] == "rejected"
    assert "P0" in result.get("message", "").lower()
    
    # Second tile: Correct P0 submission
    correction_tile = {
        "agent_id": agent_id,
        "priority": "P0",
        "action": "map_negative_space",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(correction_tile)
    assert result["status"] == "accepted"
    
    # Verify agent can continue with P1 after correction
    p1_tile = {
        "agent_id": agent_id,
        "priority": "P1",
        "action": "find_safe_channels",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(p1_tile)
    assert result["status"] == "accepted"
```

### Test 4: Resource Exhaustion Handling
```python
def test_resource_exhaustion_handling(self):
    """Room should handle memory/disk limits gracefully during long sessions."""
    room = DeadbandRoom()
    
    # Simulate many tile submissions
    max_tiles_before_cleanup = 1000  # From room config
    accepted_count = 0
    
    for i in range(max_tiles_before_cleanup * 2):
        tile = {
            "agent_id": f"stress_agent_{i % 10}",
            "priority": "P1",
            "action": f"action_{i}",
            "timestamp": time.time(),
            "data": "x" * 1024  # 1KB per tile
        }
        
        try:
            result = room.submit_tile(tile)
            if result["status"] == "accepted":
                accepted_count += 1
        except MemoryError:
            # Room should trigger cleanup before crashing
            room_state = room.get_state()
            assert len(room_state["recent_tiles"]) < max_tiles_before_cleanup
            break
    
    # Verify room is still functional after cleanup
    test_tile = {
        "agent_id": "test_after_cleanup",
        "priority": "P0",
        "action": "verify_functional",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(test_tile)
    assert result["status"] == "accepted"
```

### Test 5: Network Partition Simulation
```python
def test_network_partition_handling(self):
    """Room should handle partial writes and timeouts during network issues."""
    room = DeadbandRoom()
    
    # Mock a network partition during tile submission
    original_save_method = room._save_tile_to_storage
    
    def failing_save(tile):
        # Simulate network partition: first write succeeds, second fails
        if not hasattr(failing_save, "call_count"):
            failing_save.call_count = 0
        failing_save.call_count += 1
        
        if failing_save.call_count == 2:
            raise ConnectionError("Network partition simulated")
        return original_save_method(tile)
    
    room._save_tile_to_storage = failing_save
    
    # First tile should succeed
    tile1 = {
        "agent_id": "network_test",
        "priority": "P0",
        "action": "first_tile",
        "timestamp": time.time()
    }
    
    result1 = room.submit_tile(tile1)
    assert result1["status"] == "accepted"
    
    # Second tile should fail with network error
    tile2 = {
        "agent_id": "network_test",
        "priority": "P1",
        "action": "second_tile",
        "timestamp": time.time()
    }
    
    try:
        result2 = room.submit_tile(tile2)
        assert False, "Should have raised ConnectionError"
    except ConnectionError as e:
        assert "partition" in str(e).lower()
    
    # Room state should still be consistent (first tile persisted)
    room_state = room.get_state()
    assert len(room_state["recent_tiles"]) == 1
    assert room_state["recent_tiles"][0]["action"] == "first_tile"
    
    # Restore original method and verify recovery
    room._save_tile_to_storage = original_save_method
    result3 = room.submit_tile(tile2)
    assert result3["status"] == "accepted"
```
