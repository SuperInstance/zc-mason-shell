### Test 2: Concurrent State Transitions
```python
def test_concurrent_p1_p2_transitions():
    """Edge case: Multiple agents attempting phase transitions simultaneously."""
    room = DeadbandRoom(config=valid_config)
    agent_ids = ["agent_a", "agent_b", "agent_c"]
    
    # All agents join
    for aid in agent_ids:
        room.join_agent(aid)
        room.submit_tile(aid, {"phase": "P0", "content": f"P0 from {aid}"})
    
    # Simulate concurrent P1 submissions (thread-safe check)
    import threading
    
    results = []
    def submit_p1(aid):
        try:
            room.submit_tile(aid, {"phase": "P1", "content": f"channel from {aid}"})
            results.append((aid, "success"))
        except Exception as e:
            results.append((aid, str(e)))
    
    threads = [threading.Thread(target=submit_p1, args=(aid,)) for aid in agent_ids]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    # All should succeed or fail gracefully (no data corruption)
    assert len(results) == 3
    assert all("corrupt" not in r[1] for r in results)
```

### Test 4: Malformed Tile Submission
```python
def test_malformed_tile_submission():
    """Edge case: Agent submits non-JSON, wrong schema, or oversized tile."""
    room = DeadbandRoom(config=valid_config)
    agent_id = "agent_malformed"
    room.join_agent(agent_id)
    
    test_cases = [
        ("not json at all", ValueError),  # Invalid JSON
        ({"wrong_key": "data"}, KeyError),  # Missing required fields
        ({"phase": "P0", "content": "x" * 10000}, ValueError),  # Too large
        ({"phase": "P5", "content": "invalid phase"}, ValueError),  # Invalid phase
    ]
    
    for tile_data, expected_error in test_cases:
        try:
            room.submit_tile(agent_id, tile_data)
            assert False, f"Should have raised {expected_error} for {tile_data}"
        except expected_error:
            pass  # Expected
        except Exception as e:
            # Should not crash with unrelated errors
            assert False, f"Unexpected error {type(e)} for {tile_data}"
    
    # Agent state should remain intact despite bad submissions
    assert agent_id in room._agents
```
