### 1. Test Concurrent Tile Submission

def test_concurrent_tile_submission(deadband_room):
    """Multiple agents submitting tiles simultaneously should not corrupt state."""
    import threading
    
    results = []
    def submit_agent_tile(agent_id):
        tile = {"agent_id": agent_id, "phase": "P1", "content": f"test from {agent_id}"}
        try:
            result = deadband_room.submit_tile(tile)
            results.append((agent_id, result))
        except Exception as e:
            results.append((agent_id, e))
    
    threads = []
    for i in range(5):
        t = threading.Thread(target=submit_agent_tile, args=(f"agent_{i}",))
        threads.append(t)
        t.start()
    
    for t in threads:
        t.join()
    
    # All submissions should complete without exception
    success_count = sum(1 for _, r in results if r.get("status") == "accepted")
    assert success_count == 5, f"Expected 5 successes, got {success_count}"
    assert deadband_room.tile_queue.qsize() == 5
```

def test_malformed_tile_rejection(deadband_room):
    """Tiles missing required fields or with invalid types should be rejected with specific error."""
    # Missing 'agent_id'
    tile1 = {"phase": "P0", "content": "test"}

### 3. Test Protocol Violation Sequence

def test_protocol_sequence_violation(deadband_room):
    """Agent attempting P2 action before P1 completion should be blocked."""
    agent_id = "impatient_agent"
    
    # Start with P0
    deadband_room.set_agent_phase(agent_id, "P0")
    assert deadband_room.get_agent_phase(agent_id) == "P0"
    
    # Attempt direct jump to P2 (should fail)
    with pytest.raises(RuntimeError, match="Cannot proceed to P2: P1 not completed"):
        deadband_room.set_agent_phase(agent_id, "P2")
    
    # Complete P0 -> P1 transition properly
    deadband_room.set_agent_phase(agent_id, "P1")
    assert deadband_room.get_agent_phase(agent_id) == "P1"
    
    # Now P2 should succeed
    deadband_room.set_agent_phase(agent_id, "P2")
    assert deadband_room.get_agent_phase(agent_id) == "P2"
```

def test_room_capacity_limits(deadband_room):
    """Room should enforce maximum concurrent agents and queue limits."""
    # Add maximum agents
    for i in range(deadband_room.MAX_AGENTS):
        deadband_room.register_agent(f"agent_{i}")
    
    # Next registration should fail
    with pytest.raises(RuntimeError, match="Room at capacity"):
        deadband_room.register_agent("extra_agent")
    
    # Test tile queue limit
    deadband_room.MAX_QUEUE_SIZE = 10
    for i in range(10):
        tile = {"agent_id": "agent_0", "phase": "P1", "content": f"tile_{i}"}
        deadband_room.submit_tile(tile)
    
    # 11th submission should be rejected
    tile = {"agent_id": "agent_0", "phase": "P1", "content": "overflow"}
    with pytest.raises(RuntimeError, match="Queue full"):
        deadband_room.submit_tile(tile)
```

def test_validator_network_failure(deadband_room, mocker):
    """Tile submission should handle validator timeout or network errors gracefully."""
    # Mock validator to simulate network timeout
    mock_validator = mocker.MagicMock()
    mock_validator.validate_tile.side_effect = TimeoutError("Validator unreachable")
    deadband_room.tile_validator = mock_validator
    
    tile = {"agent_id": "agent_1", "phase": "P1", "content": "test tile"}
    
    # Should not crash; should return specific error state
    result = deadband_room.submit_tile(tile)
    assert result["status"] == "failed"
    assert "validator_timeout" in result["error"]
    
    # Tile should remain in retry queue
    assert deadband_room.retry_queue.qsize() == 1
```
