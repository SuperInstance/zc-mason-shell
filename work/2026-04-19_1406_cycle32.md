## 5 Additional Edge-Case Tests

### Test 1: Protocol Priority Violation - Skipping P0
**Scenario:** Agent attempts to submit P2 tile without completing P0 and P1 phases.
**Expected:** Room rejects tile, maintains P0 phase, logs violation.
**Rationale:** Ensures strict adherence to Deadband Protocol priority.

```python
def test_protocol_priority_violation_skip_p0():
    """Test that room rejects P2 tile when P0 not completed."""
    room = DeadbandRoom()
    agent_id = "test_agent_1"
    
    # Agent tries to submit P2 tile immediately (skipping P0 and P1)
    tile = {
        "agent_id": agent_id,
        "phase": "P2",
        "content": "Optimizing without mapping negative space",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(agent_id, tile)
    assert not result["accepted"]
    assert "P0" in result.get("message", "")  # Should reference P0 requirement
    assert room.get_agent_phase(agent_id) == "P0"  # Should remain in P0
```

### Test 2: Concurrent Agent State Isolation
**Scenario:** Multiple agents interact with room simultaneously, each with independent phase progression.
**Expected:** Agent states remain isolated; one agent's progress doesn't affect another's.
**Rationale:** Ensures room handles concurrent sessions correctly.

```python
def test_concurrent_agent_state_isolation():
    """Test that multiple agents have independent phase states."""
    room = DeadbandRoom()
    agent_a = "agent_alpha"
    agent_b = "agent_beta"
    
    # Agent A completes P0
    tile_p0_a = {
        "agent_id": agent_a,
        "phase": "P0",
        "content": "Mapping negative space for agent A",
        "timestamp": time.time()
    }
    room.submit_tile(agent_a, tile_p0_a)
    
    # Agent B should still be in P0
    assert room.get_agent_phase(agent_a) == "P1"
    assert room.get_agent_phase(agent_b) == "P0"
    
    # Agent B submits P0 tile
    tile_p0_b = {
        "agent_id": agent_b,
        "phase": "P0",
        "content": "Mapping negative space for agent B",
        "timestamp": time.time()
    }
    room.submit_tile(agent_b, tile_p0_b)
    
    # Both should now be in P1, independently
    assert room.get_agent_phase(agent_a) == "P1"
    assert room.get_agent_phase(agent_b) == "P1"
```

### Test 3: Tile Queue Exhaustion and Recovery
**Scenario:** Room's tile queue reaches capacity; new submissions should be handled gracefully.
**Expected:** Room either queues tiles with backpressure or rejects with clear error; recovers when space available.
**Rationale:** Tests resource limits and recovery mechanisms.

```python
def test_tile_queue_exhaustion_recovery():
    """Test room behavior when tile queue reaches capacity."""
    room = DeadbandRoom(max_queue_size=3)  # Small queue for testing
    agent_id = "test_agent_3"
    
    # Fill the queue
    for i in range(3):
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": f"Tile {i}",
            "timestamp": time.time()
        }
        result = room.submit_tile(agent_id, tile)
        assert result["accepted"]
    
    # Fourth tile should be rejected or queued with backpressure
    tile_overflow = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Overflow tile",
        "timestamp": time.time()
    }
    result = room.submit_tile(agent_id, tile_overflow)
    
    # Either rejection or successful with queue management
    if not result["accepted"]:
        assert "queue" in result.get("message", "").lower() or "full" in result.get("message", "").lower()
    else:
        # If accepted, verify queue management
        assert room.get_queue_size() <= 3  # Should not exceed max
    
    # Process some tiles and verify recovery
    room.process_tiles(2)
    assert room.get_queue_size() <= 1
```

### Test 4: Malformed State Recovery After Corruption
**Scenario:** Room's internal state becomes corrupted (simulated); room should detect and recover.
**Expected:** Room either resets to safe state or rejects further tiles until repaired.
**Rationale:** Tests robustness against state corruption.

```python
def test_malformed_state_recovery():
    """Test room recovery from corrupted internal state."""
    room = DeadbandRoom()
    agent_id = "test_agent_4"
    
    # Simulate state corruption (direct manipulation for test)
    room.agent_phases[agent_id] = "INVALID_PHASE"
    
    # Attempt to submit tile
    tile = {
        "agent_id": agent_id,
        "phase": "P0",
        "content": "Tile after corruption",
        "timestamp": time.time()
    }
    
    result = room.submit_tile(agent_id, tile)
    
    # Room should either:
    # 1. Reject tile and indicate state error
    # 2. Reset agent to P0 and accept tile
    if not result["accepted"]:
        assert "state" in result.get("message", "").lower() or "invalid" in result.get("message", "").lower()
    else:
        # If accepted, state should be repaired
        assert room.get_agent_phase(agent_id) in ["P0", "P1", "P2"]
```

### Test 5: Boundary Condition - Empty Tile Content
**Scenario:** Agent submits tile with empty or whitespace-only content.
**Expected:** Room should reject empty content tiles with appropriate error message.
**Rationale:** Ensures content validation handles edge cases.

```python
def test_empty_tile_content_rejection():
    """Test that tiles with empty content are rejected."""
    room = DeadbandRoom()
    agent_id = "test_agent_5"
    
    test_cases = [
        {"content": "", "description": "empty string"},
        {"content": "   ", "description": "whitespace only"},
        {"content": "\n\t\r", "description": "control characters only"},
    ]
    
    for test_case in test_cases:
        tile = {
            "agent_id": agent_id,
            "phase": "P0",
            "content": test_case["content"],
            "timestamp": time.time()
        }
        
        result = room.submit_tile(agent_id, tile)
        assert not result["accepted"], f"Should reject {test_case['description']}"
        assert "content" in result.get("message", "").lower() or "empty" in result.get("message", "").lower()
```
