### Test 1: Protocol Priority Violation (Skip P1 to P2)
**Scenario:** Agent attempts P2 optimization without completing P1 channel identification.
**Expected:** Room rejects tile with "P1_NOT_COMPLETE" error.
**Rationale:** Tests strict priority enforcement from Fleet Doctrine.

```python
def test_protocol_priority_violation_skip_p1_to_p2():
    """DeadbandRoom must reject P2 optimization attempts before P1 completion."""
    room = DeadbandRoom(room_id="edge_priority")
    agent_state = {"position": [0.5, 0.5], "velocity": [0.1, 0.1]}
    
    # Agent tries to submit P2 tile without P1
    tile = {
        "agent_id": "test_agent",
        "cycle": 100,
        "phase": "P2",
        "content": {"optimization": "gradient_descent", "step_size": 0.01},
        "protocol_check": False  # Deliberately missing P1
    }
    
    result = room.submit_tile(tile, agent_state)
    assert not result["accepted"]
    assert "P1_NOT_COMPLETE" in result["error"]
    assert room.tile_count == 0
```

def test_p1_channel_collision_detection():

        "agent_id": "agent_alpha",
        "cycle": 200,
        "phase": "P1",
        "content": {
            "safe_channels": [

        }

    # Second agent identifies overlapping channel B
    tile2 = {
        "agent_id": "agent_beta",
        "cycle": 201,
        "phase": "P1",
        "content": {
            "safe_channels": [
                {"id": "channel_B", "bounds": [[5, 5], [15, 15]], "confidence": 0.8}

    assert result2["accepted"]
    assert hasattr(room, "channel_conflicts")
    assert len(room.channel_conflicts) > 0
    assert room.get_channel_overlap("channel_A", "channel_B") is not None

### Test 5: Tile Flood Protection
**Scenario:** Agent submits tiles faster than room processing rate.
**Expected:** Room implements rate limiting or queue management.
**Rationale:** Tests system stability under high load.

```python
def test_tile_flood_protection():
    """Room should handle rapid tile submissions without crashing."""
    room = DeadbandRoom(room_id="edge_flood", processing_delay=0.1)
    agent_state = {"position": [0, 0]}
    
    submission_results = []
    
    # Submit 20 tiles in rapid succession
    for i in range(20):
        tile = {
            "agent_id": "flood_agent",
            "cycle": 400 + i,
            "phase": "P0",
            "content": {"negative_space": [{"constraint": f"test_{i}", "value": i}]}
        }
        
        try:
            result = room.submit_tile(tile, agent_state)
            submission_results.append(result)
        except Exception as e:
            submission_results.append({"error": str(e), "cycle": 400 + i})
    
    # Check system stability
    assert room.is_healthy()  # Room should still be responsive
    assert len(submission_results) == 20
    
    # Some late submissions might be queued or rejected, but no crashes
    crash_count = sum(1 for r in submission_results if "RuntimeError" in str(r.get("error", "")))
    assert crash_count == 0
    
    # Verify queue management exists
    assert hasattr(room, "queue_length") or hasattr(room, "processing_queue")
```
