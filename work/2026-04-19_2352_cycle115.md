### Test 1: Concurrent P1 Channel Collision
```python
def test_concurrent_p1_collision(self):
    """Two agents attempt to claim the same P1 channel simultaneously."""
    room = DeadbandRoom()
    state = {"position": [0, 0], "constraints": {"rock": [1, 1]}}
    
    # Agent A claims channel [2, 2]
    action_a = {"type": "p1", "channel": [2, 2]}
    result_a = room.step(state, action_a)
    
    # Agent B attempts same channel before state update
    action_b = {"type": "p1", "channel": [2, 2]}
    result_b = room.step(state, action_b)
    
    # First agent should succeed, second should receive collision error
    self.assertTrue(result_a["success"])
    self.assertFalse(result_b["success"])
    self.assertIn("channel_occupied", result_b["error"])
```

### Test 2: P2 Before P1 Protocol Violation

def test_p2_before_p1_violation(self):
    """Agent attempts P2 optimization without established P1 channel."""
    room = DeadbandRoom()
    state = {"position": [0, 0], "constraints": {}, "p1_channel": None}
    
    action = {"type": "p2", "optimization": "gradient_descent"}
    result = room.step(state, action)
    
    # Should fail with protocol violation
    self.assertFalse(result["success"])
    self.assertIn("p1_required", result["error"])
    self.assertEqual(result["reward"], -10)  # Protocol violation penalty
```

### Test 3: Complete P0 Constraint Lock

def test_complete_p0_lock(self):
    """P0 constraints eliminate all possible actions."""
    room = DeadbandRoom()
    # Surround agent with constraints in all directions
    state = {
        "position": [5, 5],
        "constraints": {
            "rock": [[4, 5], [6, 5], [5, 4], [5, 6], [4, 4], [4, 6], [6, 4], [6, 6]]
        }
    }
    
    action = {"type": "p1", "channel": [5, 7]}  # Attempt any move
    result = room.step(state, action)
    
    # Should fail with no valid channels
    self.assertFalse(result["success"])
    self.assertIn("no_valid_channels", result["error"])
    self.assertEqual(result["reward"], -5)  # Stuck penalty
```

### Test 4: Reward Boundary Conditions
```python
def test_reward_boundary_conditions(self):
    """Test min/max reward values and accumulation."""
    room = DeadbandRoom()
    state = {"position": [0, 0], "constraints": {}, "total_reward": 0}
    
    # Test reward ceiling (should cap at 1000)
    for _ in range(150):
        action = {"type": "p2", "optimization": "efficient"}
        state = room.step(state, action)["next_state"]
    
    self.assertLessEqual(state["total_reward"], 1000)
    
    # Test reward floor (should cap at -100)
    state["total_reward"] = -50
    for _ in range(10):
        action = {"type": "p0", "constraint": "invalid"}  # Deliberate error
        state = room.step(state, action)["next_state"]
    
    self.assertGreaterEqual(state["total_reward"], -100)
```

### Test 5: Persistent Channel State Validation
```python
def test_persistent_channel_state(self):
    """P1 channel should persist across multiple P2 optimizations."""
    room = DeadbandRoom()
    state = {"position": [0, 0], "constraints": {}, "p1_channel": None}
    
    # Establish P1 channel
    p1_action = {"type": "p1", "channel": [3, 3]}
    result = room.step(state, p1_action)
    self.assertIsNotNone(result["next_state"]["p1_channel"])
    
    # Perform 5 P2 optimizations
    channel_id = result["next_state"]["p1_channel"]
    for i in range(5):
        p2_action = {"type": "p2", "optimization": f"path_{i}"}
        result = room.step(result["next_state"], p2_action)
        # Channel should remain unchanged
        self.assertEqual(result["next_state"]["p1_channel"], channel_id)
    
    # Attempt to change channel without explicit release
    new_p1 = {"type": "p1", "channel": [4, 4]}
    result = room.step(result["next_state"], new_p1)
    self.assertFalse(result["success"])
    self.assertIn("channel_locked", result["error"])
```
