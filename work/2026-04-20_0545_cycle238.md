### Test 1: Protocol Priority Violation (Skipping to P2)
**Scenario:** Agent attempts to execute P2 optimization before completing P0 and P1 mapping.
**Expected:** Room should reject the action and log a protocol violation.
**Rationale:** Tests strict adherence to Deadband Protocol P0→P1→P2 priority.

```python
def test_protocol_priority_violation_rejection():
    """DeadbandRoom should reject P2 actions when P0/P1 not completed."""
    room = DeadbandRoom(room_id="test-edge-1")
    agent_id = "test-agent"
    
    # Agent tries to submit P2 tile without P0/P1
    tile = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": "P2",
        "content": "Optimizing prematurely",
        "protocol_step": 2
    }
    
    result = room.submit_tile(tile)
    assert result["status"] == "rejected"
    assert "protocol violation" in result["message"].lower()
    assert room.get_agent_progress(agent_id)["current_step"] == 0
```

### Test 2: Concurrent Agent Protocol State Isolation
**Scenario:** Multiple agents at different protocol steps interacting simultaneously.
**Expected:** Each agent's protocol state should be isolated and not affect others.
**Rationale:** Tests room's ability to handle concurrent training sessions.

```python
def test_concurrent_agent_protocol_isolation():
    """Multiple agents should have independent protocol progress."""
    room = DeadbandRoom(room_id="test-edge-2")
    agent_a = "agent-alpha"
    agent_b = "agent-beta"
    
    # Agent A completes P0
    tile_a_p0 = {
        "agent_id": agent_a,
        "cycle": 1,
        "phase": "P0",
        "content": "Negative space mapped",
        "protocol_step": 0
    }
    room.submit_tile(tile_a_p0)
    
    # Agent B hasn't started
    tile_b_p2 = {
        "agent_id": agent_b,
        "cycle": 1,
        "phase": "P2",
        "content": "Should be rejected",
        "protocol_step": 2
    }
    
    result_b = room.submit_tile(tile_b_p2)
    assert result_b["status"] == "rejected"
    
    # Agent A should still be at step 0 (P0 completed, ready for P1)
    # Agent B should be at step 0 (no progress)
    assert room.get_agent_progress(agent_a)["current_step"] == 0
    assert room.get_agent_progress(agent_b)["current_step"] == 0
    assert room.get_agent_progress(agent_a)["completed_phases"] == ["P0"]
    assert room.get_agent_progress(agent_b)["completed_phases"] == []
```

**Expected:** Room should reject tile with appropriate error message.
**Rationale:** Tests robustness of content validation beyond basic format checks.

```python
def test_tile_validation_malformed_content_types():
    """DeadbandRoom should reject tiles with non-string content."""
    room = DeadbandRoom(room_id="test-edge-3")
    agent_id = "test-agent"
    
    test_cases = [
        {"content": {"nested": "dict"}, "description": "dict content"},
        {"content": ["list", "items"], "description": "list content"},
        {"content": None, "description": "null content"},
        {"content": 12345, "description": "integer content"},
        {"content": "", "description": "empty string"},
    ]
    
    for i, test_case in enumerate(test_cases):
        tile = {
            "agent_id": agent_id,
            "cycle": i + 1,
            "phase": "P0",
            "content": test_case["content"],
            "protocol_step": 0
        }
        
        result = room.submit_tile(tile)
        # Empty string might be valid for some rooms, but others may reject
        if test_case["description"] == "empty string":
            # This tests the room's policy on empty content
            assert result["status"] in ["accepted", "rejected"]
        else:
            assert result["status"] == "rejected"
            assert "content" in result["message"].lower()
```

### Test 4: Protocol Step Regression Prevention
**Scenario:** Agent tries to go backward in protocol steps (e.g., P1 → P0).
**Expected:** Room should reject regression and maintain current step.
**Rationale:** Tests enforcement of forward-only protocol progression.

```python
def test_protocol_step_regression_prevention():
    """Agents should not be able to regress to earlier protocol steps."""
    room = DeadbandRoom(room_id="test-edge-4")
    agent_id = "test-agent"
    
    # Complete P0
    tile_p0 = {
        "agent_id": agent_id,
        "cycle": 1,
        "phase": "P0",
        "content": "P0 complete",
        "protocol_step": 0
    }
    room.submit_tile(tile_p0)
    
    # Complete P1
    tile_p1 = {
        "agent_id": agent_id,
        "cycle": 2,
        "phase": "P1",
        "content": "P1 complete",
        "protocol_step": 1
    }
    room.submit_tile(tile_p1)
    
    # Try to go back to P0
    tile_p0_again = {
        "agent_id": agent_id,
        "cycle": 3,
        "phase": "P0",
        "content": "Trying to regress",
        "protocol_step": 0
    }
    
    result = room.submit_tile(tile_p0_again)
    assert result["status"] == "rejected"
    assert "regress" in result["message"].lower() or "already completed" in result["message"].lower()
    
    # Agent should still be at step 1 (P1 completed)
    progress = room.get_agent_progress(agent_id)
    assert progress["current_step"] == 1
    assert set(progress["completed_phases"]) == {"P0", "P1"}
```
