### Test 4: Concurrent Tile Submission Simulation
```python
def test_rapid_sequential_submissions(deadband_room):
    """Test rapid sequential submissions from same agent."""
    tiles = []
    for i in range(5):
        tile = {
            "cycle": 313 + i,
            "agent": "mason",
            "content": {"iteration": i, "data": "x" * 1000},
            "protocol_phase": "P1"
        }
        tiles.append(tile)
    
    results = []
    for tile in tiles:
        results.append(deadband_room.submit_tile(tile))
    
    # All should be accepted if cycles are sequential
    accepted = [r for r in results if r["status"] == "accepted"]
    assert len(accepted) == 5
    assert deadband_room.get_tile_count() == 5
```
