### Test 3: Concurrent Agent Isolation
```rust
#[test]
fn test_concurrent_agents_protocol_isolation() {
    let mut room = DeadbandRoom::new();
    let agent1 = "agent_alpha".to_string();
    let agent2 = "agent_beta".to_string();
    
    // Agent 1 completes P0
    room.submit_tile(TrainingTile {
        agent_id: agent1.clone(),
        phase: Some(Phase::P0MapNegative),
        content: "P0 from alpha".to_string(),
        timestamp: Utc::now(),
    }).unwrap();
    
    // Agent 2 attempts P1 without P0
    let result = room.submit_tile(TrainingTile {
        agent_id: agent2.clone(),
        phase: Some(Phase::P1FindChannels),
        content: "P1 from beta".to_string(),
        timestamp: Utc::now(),
    });
    
    // Should fail - each agent has independent protocol state
    assert!(result.is_err());
    
    // Agent 1 should still be able to proceed to P1
    assert!(room.can_transition_to_p1(&agent1));
    assert!(!room.can_transition_to_p1(&agent2));
}
```

### Test 4: Invalid State Recovery
```rust
#[test]
fn test_invalid_state_reset_and_recovery() {
    let mut room = DeadbandRoom::new();
    let agent_id = "recovery_agent".to_string();
    
    // Corrupt agent state (simulating bug or crash)
    room.force_invalid_state(&agent_id);
    
    // Attempt normal operation - should fail
    let tile = TrainingTile {
        agent_id: agent_id.clone(),
        phase: Some(Phase::P0MapNegative),
        content: "Normal tile".to_string(),
        timestamp: Utc::now(),
    };
    
    let initial_result = room.submit_tile(tile);
    assert!(initial_result.is_err());
    
    // Reset agent state
    room.reset_agent(&agent_id);
    
    // Should now accept valid tiles
    let recovery_tile = TrainingTile {
        agent_id: agent_id.clone(),
        phase: Some(Phase::P0MapNegative),
        content: "Recovery tile".to_string(),
        timestamp: Utc::now(),
    };
    
    let recovery_result = room.submit_tile(recovery_tile);
    assert!(recovery_result.is_ok());
}
```
