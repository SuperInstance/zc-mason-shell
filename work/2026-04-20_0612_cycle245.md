### Test 2: Concurrent Tile Submission Handling
**Scenario:** Multiple agents submit tiles simultaneously
**Expected:** Room processes tiles sequentially, maintains state consistency
**Purpose:** Test thread safety in tile processing

```python
def test_concurrent_tile_submission(self):
    """Room should handle concurrent submissions without corruption."""
    room = DeadbandRoom()
    room.start_training()
    
    tiles = []
    for i in range(5):
        tile = {
            'phase': 'P0',
            'content': f'Negative space mapping {i}',
            'agent_id': f'agent_{i}',
            'timestamp': time.time()
        }
        tiles.append(tile)
    
    # Simulate concurrent submission
    import threading
    results = []
    
    def submit_tile(tile):
        results.append(room.submit_tile(tile))
    
    threads = []
    for tile in tiles:
        thread = threading.Thread(target=submit_tile, args=(tile,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all tiles processed
    assert len(results) == 5
    assert all(r['accepted'] for r in results)
    assert len(room.training_tiles) == 5
```

### Test 4: Malformed Tile Structure Recovery
**Scenario:** Tile missing required fields or with invalid types
**Expected:** Room validates structure, rejects invalid tiles
**Purpose:** Test input validation robustness

```python
def test_malformed_tile_validation(self):
    """Room should validate tile structure and reject malformed tiles."""
    room = DeadbandRoom()
    room.start_training()
    
    test_cases = [
        {'phase': 'P0'},  # Missing content and agent_id
        {'content': 'test', 'agent_id': 'agent'},  # Missing phase
        {'phase': 'P5', 'content': 'test', 'agent_id': 'agent'},  # Invalid phase
        {'phase': 'P0', 'content': 123, 'agent_id': 'agent'},  # Wrong content type
        {'phase': 'P0', 'content': 'test', 'agent_id': 456},  # Wrong agent_id type
    ]
    
    for tile in test_cases:
        result = room.submit_tile(tile)
        assert not result['accepted']
        assert 'invalid' in result['message'].lower() or 'missing' in result['message'].lower()
    
    # Room should still accept valid tiles after rejections
    valid_tile = {
        'phase': 'P0',
        'content': 'Valid negative space mapping',
        'agent_id': 'test_agent',
        'timestamp': time.time()
    }
    
    result = room.submit_tile(valid_tile)
    assert result['accepted']
    assert len(room.training_tiles) == 1
```

### Test 5: Training Interruption Recovery
**Scenario:** Room stopped mid-training, then restarted
**Expected:** Room recovers state, preserves completed tiles
**Purpose:** Test state persistence and recovery mechanisms

```python
def test_training_interruption_recovery(self):
    """Room should recover state after interruption."""
    room = DeadbandRoom()
    room.start_training()
    
    # Submit some tiles
    for i in range(3):
        tile = {
            'phase': ['P0', 'P1', 'P2'][i],
            'content': f'Test content {i}',
            'agent_id': 'test_agent',
            'timestamp': time.time()
        }
        room.submit_tile(tile)
    
    # Simulate interruption (room instance lost)
    tile_count_before = len(room.training_tiles)
    phase_progress = room.current_phase
    
    # Create new room instance (simulating recovery)
    recovered_room = DeadbandRoom()
    
    # Room should initialize with previous state if persistence implemented
    # This test validates the expected interface for recovery
    if hasattr(recovered_room, 'recover_state'):
        recovered_room.recover_state()
        assert len(recovered_room.training_tiles) == tile_count_before
        assert recovered_room.current_phase == phase_progress
    else:
        # If no recovery mechanism, at least shouldn't crash
        recovered_room.start_training()
        assert recovered_room.training_state == 'active'
```
