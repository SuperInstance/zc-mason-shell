### Test 1: Concurrent Tile Submission Race Condition
```python
def test_concurrent_tile_submission_race():
    """Test that room handles concurrent tile submissions without corruption."""
    room = DeadbandRoom()
    
    # Simulate 10 agents submitting tiles simultaneously
    tiles = []
    for i in range(10):
        tile = {
            "agent_id": f"agent_{i}",
            "cycle": 36,
            "phase": 4,
            "content": f"Test tile {i}",
            "protocol_step": "P1" if i % 2 == 0 else "P2"
        }
        tiles.append(tile)
    
    # Submit all tiles concurrently (simulated with threads)
    results = []
    import threading
    
    def submit_tile(tile):
        try:
            result = room.submit_tile(tile)
            results.append((tile["agent_id"], result))
        except Exception as e:
            results.append((tile["agent_id"], str(e)))
    
    threads = []
    for tile in tiles:
        thread = threading.Thread(target=submit_tile, args=(tile,))
        threads.append(thread)
        thread.start()
    
    for thread in threads:
        thread.join()
    
    # Verify all tiles were processed without data corruption
    assert len(results) == 10
    assert all(r[1] is not None for r in results)
    
    # Check room state integrity
    assert room.get_tile_count() == 10
    assert len(set(r[0] for r in results)) == 10  # All agents accounted for
```

```python
def test_protocol_interruption_resume():

    room = DeadbandRoom()

    room.set_protocol_phase("P0")

    # Resume protocol
    room.resume_protocol()

    # Complete P0 and move to P1
```

```python
def test_network_partition_recovery():
    """Test room recovery after simulated network partition."""
    room = DeadbandRoom()
    
    # Normal operation
    room.set_protocol_phase("P1")
    tile1 = {"agent_id": "agent1", "content": "Normal tile", "protocol_step": "P1"}
    result1 = room.submit_tile(tile1)
    assert result1["status"] == "accepted"
    
    # Simulate network partition
    room._simulate_network_partition(duration=5)  # 5 cycle partition
    
    # During partition, submissions should fail or queue
    tile2 = {"agent_id": "agent2", "content": "Partition tile", "protocol_step": "P1"}
    
    try:
        result2 = room.submit_tile(tile2)
        # If submission succeeded, it was queued
        assert result2["status"] == "queued"
        assert "queued_at" in result2
    except ConnectionError:
        # Or it raised an exception
        pass
    
    # End partition
    room._end_network_partition()
    
    # Verify recovery
    assert room.is_network_healthy() == True
    
    # Queued tiles should process
    if "result2" in locals() and result2["status"] == "queued":
        room.process_queued_tiles()
        assert room.get_tile_count() == 2
```

```python
def test_malformed_protocol_transitions():
    """Test handling of invalid protocol step transitions."""
    room = DeadbandRoom()
    
    # Test invalid transition: P0 -> P2 (skipping P1)
    room.set_protocol_phase("P0")
    
    # Attempt to submit P2 tile during P0 phase
    invalid_tile = {
        "agent_id": "impatient_agent",
        "content": "Trying to skip P1",
        "protocol_step": "P2"  # Invalid: should be P0
    }
    
    result = room.submit_tile(invalid_tile)
    assert result["status"] == "rejected"
    assert "protocol_violation" in result.get("reason", "").lower()
    
    # Test invalid transition: P1 -> P0 (going backward)
    room.complete_phase("P0")
    room.set_protocol_phase("P1")
    
    backward_tile = {
        "agent_id": "confused_agent",
        "content": "Trying to go back to P0",
        "protocol_step": "P0"  # Invalid: cannot go backward
    }
    
    result2 = room.submit_tile(backward_tile)
    assert result2["status"] == "rejected"
    assert "cannot regress" in result2.get("reason", "").lower()
    
    # Test valid transition: P1 -> P2
    valid_tile = {
        "agent_id": "good_agent",
        "content": "Valid P1 work",
        "protocol_step": "P1"
    }
    
    result3 = room.submit_tile(valid_tile)
    assert result3["status"] == "accepted"
    
    # Complete P1 and move to P2
    room.complete_phase("P1")
    assert room.get_protocol_phase() == "P2"
```
